017package org.apache.commons.math3.ode;
018
019import org.apache.commons.math3.exception.DimensionMismatchException;
020import org.apache.commons.math3.exception.MaxCountExceededException;
021
022/** Interface to compute exactly Jacobian matrix for some parameter
023 *  when computing {@link JacobianMatrices partial derivatives equations}.
024 *
025 * @since 3.0
026 */
027public interface ParameterJacobianProvider extends Parameterizable {
028
029    /** Compute the Jacobian matrix of ODE with respect to one parameter.
030     * <p>If the parameter does not belong to the collection returned by
031     * {@link #getParametersNames()}, the Jacobian will be set to 0,
032     * but no errors will be triggered.</p>
033     * @param t current value of the independent <I>time</I> variable
034     * @param y array containing the current value of the main state vector
035     * @param yDot array containing the current value of the time derivative
036     * of the main state vector
037     * @param paramName name of the parameter to consider
038     * @param dFdP placeholder array where to put the Jacobian matrix of the
039     * ODE with respect to the parameter
040     * @exception MaxCountExceededException if the number of functions evaluations is exceeded
041     * @exception DimensionMismatchException if arrays dimensions do not match equations settings
042     * @exception UnknownParameterException if the parameter is not supported
043     */
044    void computeParameterJacobian(double t, double[] y, double[] yDot,
045                                  String paramName, double[] dFdP)
046        throws DimensionMismatchException, MaxCountExceededException, UnknownParameterException;
047
048}
