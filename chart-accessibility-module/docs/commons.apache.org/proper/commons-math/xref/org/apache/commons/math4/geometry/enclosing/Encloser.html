17  package org.apache.commons.math4.geometry.enclosing;
18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.geometry.Space;
21  
22  /** Interface for algorithms computing enclosing balls.
23   * @param <S> Space type.
24   * @param <P> Point type.
25   * @see EnclosingBall
26   * @since 3.3
27   */
28  public interface Encloser<S extends Space, P extends Point<S>> {
29  
30      /** Find a ball enclosing a list of points.
31       * @param points points to enclose
32       * @return enclosing ball
33       */
34      EnclosingBall<S, P> enclose(Iterable<P> points);
35  
36  }
