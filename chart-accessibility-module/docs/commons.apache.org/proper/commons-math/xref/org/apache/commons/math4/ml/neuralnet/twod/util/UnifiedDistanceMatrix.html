18  package org.apache.commons.math4.ml.neuralnet.twod.util;
19  
20  import java.util.Collection;
21  import org.apache.commons.math4.ml.neuralnet.Neuron;
22  import org.apache.commons.math4.ml.neuralnet.Network;
23  import org.apache.commons.math4.ml.neuralnet.twod.NeuronSquareMesh2D;
24  import org.apache.commons.math4.ml.distance.DistanceMeasure;
25  
26  /**
27   * <a href="http://en.wikipedia.org/wiki/U-Matrix">U-Matrix</a>
28   * visualization of high-dimensional data projection.
29   * @since 3.6
30   */
31  public class UnifiedDistanceMatrix implements MapVisualization {
32      /** Whether to show distance between each pair of neighbouring units. */
33      private final boolean individualDistances;
34      /** Distance. */
35      private final DistanceMeasure distance;
36  
37      /**
38       * Simple constructor.
39       *
40       * @param individualDistances If {@code true}, the 8 individual
41       * inter-units distances will be {@link #computeImage(NeuronSquareMesh2D)
42       * computed}.  They will be stored in additional pixels around each of
43       * the original units of the 2D-map.  The additional pixels that lie
44       * along a "diagonal" are shared by <em>two</em> pairs of units: their
45       * value will be set to the average distance between the units belonging
46       * to each of the pairs.  The value zero will be stored in the pixel
47       * corresponding to the location of a unit of the 2D-map.
48       * <br>
49       * If {@code false}, only the average distance between a unit and all its
50       * neighbours will be computed (and stored in the pixel corresponding to
51       * that unit of the 2D-map).  In that case, the number of neighbours taken
52       * into account depends on the network's
53       * {@link org.apache.commons.math4.ml.neuralnet.SquareNeighbourhood
54       * neighbourhood type}.
55       * @param distance Distance.
56       */
57      public UnifiedDistanceMatrix(boolean individualDistances,
58                                   DistanceMeasure distance) {
59          this.individualDistances = individualDistances;
60          this.distance = distance;
61      }
62  
63      /** {@inheritDoc} */
64      @Override
65      public double[][] computeImage(NeuronSquareMesh2D map) {
66          if (individualDistances) {
67              return individualDistances(map);
68          } else {
69              return averageDistances(map);
70          }
71      }
72  
73      /**
74       * Computes the distances between a unit of the map and its
75       * neighbours.
76       * The image will contain more pixels than the number of neurons
77       * in the given {@code map} because each neuron has 8 neighbours.
78       * The value zero will be stored in the pixels corresponding to
79       * the location of a map unit.
80       *
81       * @param map Map.
82       * @return an image representing the individual distances.
83       */
84      private double[][] individualDistances(NeuronSquareMesh2D map) {
85          final int numRows = map.getNumberOfRows();
86          final int numCols = map.getNumberOfColumns();
87  
88          final double[][] uMatrix = new double[numRows * 2 + 1][numCols * 2 + 1];
89  
90          // 1.
91          // Fill right and bottom slots of each unit's location with the
92          // distance between the current unit and each of the two neighbours,
93          // respectively.
94          for (int i = 0; i < numRows; i++) {
95              // Current unit's row index in result image.
96              final int iR = 2 * i + 1;
97  
98              for (int j = 0; j < numCols; j++) {
99                  // Current unit's column index in result image.
100                 final int jR = 2 * j + 1;
101 
102                 final double[] current = map.getNeuron(i, j).getFeatures();
103                 Neuron neighbour;
104 
105                 // Right neighbour.
106                 neighbour = map.getNeuron(i, j,
107                                           NeuronSquareMesh2D.HorizontalDirection.RIGHT,
108                                           NeuronSquareMesh2D.VerticalDirection.CENTER);
109                 if (neighbour != null) {
110                     uMatrix[iR][jR + 1] = distance.compute(current,
111                                                            neighbour.getFeatures());
112                 }
113 
114                 // Bottom-center neighbour.
115                 neighbour = map.getNeuron(i, j,
116                                           NeuronSquareMesh2D.HorizontalDirection.CENTER,
117                                           NeuronSquareMesh2D.VerticalDirection.DOWN);
118                 if (neighbour != null) {
119                     uMatrix[iR + 1][jR] = distance.compute(current,
120                                                            neighbour.getFeatures());
121                 }
122             }
123         }
124 
125         // 2.
126         // Fill the bottom-rigth slot of each unit's location with the average
127         // of the distances between
128         //  * the current unit and its bottom-right neighbour, and
129         //  * the bottom-center neighbour and the right neighbour.
130         for (int i = 0; i < numRows; i++) {
131             // Current unit's row index in result image.
132             final int iR = 2 * i + 1;
133 
134             for (int j = 0; j < numCols; j++) {
135                 // Current unit's column index in result image.
136                 final int jR = 2 * j + 1;
137 
138                 final Neuron current = map.getNeuron(i, j);
139                 final Neuron right = map.getNeuron(i, j,
140                                                    NeuronSquareMesh2D.HorizontalDirection.RIGHT,
141                                                    NeuronSquareMesh2D.VerticalDirection.CENTER);
142                 final Neuron bottom = map.getNeuron(i, j,
143                                                     NeuronSquareMesh2D.HorizontalDirection.CENTER,
144                                                     NeuronSquareMesh2D.VerticalDirection.DOWN);
145                 final Neuron bottomRight = map.getNeuron(i, j,
146                                                          NeuronSquareMesh2D.HorizontalDirection.RIGHT,
147                                                          NeuronSquareMesh2D.VerticalDirection.DOWN);
148 
149                 final double current2BottomRight = bottomRight == null ?
150                     0 :
151                     distance.compute(current.getFeatures(),
152                                      bottomRight.getFeatures());
153                 final double right2Bottom = (right == null ||
154                                              bottom == null) ?
155                     0 :
156                     distance.compute(right.getFeatures(),
157                                      bottom.getFeatures());
158 
159                 // Bottom-right slot.
160                 uMatrix[iR + 1][jR + 1] = 0.5 * (current2BottomRight + right2Bottom);
161             }
162         }
163 
164         // 3. Copy last row into first row.
165         final int lastRow = uMatrix.length - 1;
166         uMatrix[0] = uMatrix[lastRow];
167 
168         // 4.
169         // Copy last column into first column.
170         final int lastCol = uMatrix[0].length - 1;
171         for (int r = 0; r < lastRow; r++) {
172             uMatrix[r][0] = uMatrix[r][lastCol];
173         }
174 
175         return uMatrix;
176     }
177 
178     /**
179      * Computes the distances between a unit of the map and its neighbours.
180      *
181      * @param map Map.
182      * @return an image representing the average distances.
183      */
184     private double[][] averageDistances(NeuronSquareMesh2D map) {
185         final int numRows = map.getNumberOfRows();
186         final int numCols = map.getNumberOfColumns();
187         final double[][] uMatrix = new double[numRows][numCols];
188 
189         final Network net = map.getNetwork();
190 
191         for (int i = 0; i < numRows; i++) {
192             for (int j = 0; j < numCols; j++) {
193                 final Neuron neuron = map.getNeuron(i, j);
194                 final Collection<Neuron> neighbours = net.getNeighbours(neuron);
195                 final double[] features = neuron.getFeatures();
196 
197                 double d = 0;
198                 int count = 0;
199                 for (Neuron n : neighbours) {
200                     ++count;
201                     d += distance.compute(features, n.getFeatures());
202                 }
203 
204                 uMatrix[i][j] = d / count;
205             }
206         }
207 
208         return uMatrix;
209     }
