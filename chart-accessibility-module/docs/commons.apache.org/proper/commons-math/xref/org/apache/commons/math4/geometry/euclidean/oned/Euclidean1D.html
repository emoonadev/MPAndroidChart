18  package org.apache.commons.math4.geometry.euclidean.oned;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  import org.apache.commons.math4.geometry.Space;
25  
26  /**
27   * This class implements a one-dimensional space.
28   * @since 3.0
29   */
30  public class Euclidean1D implements Serializable, Space {
31  
32      /** Serializable version identifier. */
33      private static final long serialVersionUID = -1178039568877797126L;
34  
35      /** Private constructor for the singleton.
36       */
37      private Euclidean1D() {
38      }
39  
40      /** Get the unique instance.
41       * @return the unique instance
42       */
43      public static Euclidean1D getInstance() {
44          return LazyHolder.INSTANCE;
45      }
46  
47      /** {@inheritDoc} */
48      @Override
49      public int getDimension() {
50          return 1;
51      }
52  
53      /** {@inheritDoc}
54       * <p>
55       * As the 1-dimension Euclidean space does not have proper sub-spaces,
56       * this method always throws a {@link NoSubSpaceException}
57       * </p>
58       * @return nothing
59       * @throws NoSubSpaceException in all cases
60       */
61      @Override
62      public Space getSubSpace() throws NoSubSpaceException {
63          throw new NoSubSpaceException();
64      }
65  
66      // CHECKSTYLE: stop HideUtilityClassConstructor
67      /** Holder for the instance.
68       * <p>We use here the Initialization On Demand Holder Idiom.</p>
69       */
70      private static class LazyHolder {
71          /** Cached field instance. */
72          private static final Euclidean1D INSTANCE = new Euclidean1D();
73      }
74      // CHECKSTYLE: resume HideUtilityClassConstructor
75  
76      /** Handle deserialization of the singleton.
77       * @return the singleton instance
78       */
79      private Object readResolve() {
80          // return the singleton instance
81          return LazyHolder.INSTANCE;
82      }
83  
84      /** Specialized exception for inexistent sub-space.
85       * <p>
86       * This exception is thrown when attempting to get the sub-space of a one-dimensional space
87       * </p>
88       */
89      public static class NoSubSpaceException extends MathUnsupportedOperationException {
90  
91          /** Serializable UID. */
92          private static final long serialVersionUID = 20140225L;
93  
94          /** Simple constructor.
95           */
96          public NoSubSpaceException() {
97              super(LocalizedFormats.NOT_SUPPORTED_IN_DIMENSION_N, 1);
98          }
99  
100     }
101 
102 }
