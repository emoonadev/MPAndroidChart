17  package org.apache.commons.math4.distribution;
18  
19  import java.util.ArrayList;
20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Map.Entry;
24  
25  import org.apache.commons.math4.exception.DimensionMismatchException;
26  import org.apache.commons.math4.exception.MathArithmeticException;
27  import org.apache.commons.math4.exception.NotANumberException;
28  import org.apache.commons.math4.exception.NotFiniteNumberException;
29  import org.apache.commons.math4.exception.NotPositiveException;
30  import org.apache.commons.rng.UniformRandomProvider;
31  import org.apache.commons.math4.util.Pair;
32  
33  /**
34   * <p>Implementation of an integer-valued {@link EnumeratedDistribution}.</p>
35   *
36   * <p>Values with zero-probability are allowed but they do not extend the
37   * support.<br/>
38   * Duplicate values are allowed. Probabilities of duplicate values are combined
39   * when computing cumulative probabilities and statistics.</p>
40   *
41   * @since 3.2
42   */
43  public class EnumeratedIntegerDistribution extends AbstractIntegerDistribution {
44      /** Serializable UID. */
45      private static final long serialVersionUID = 20130308L;
46      /**
47       * {@link EnumeratedDistribution} instance (using the {@link Integer} wrapper)
48       * used to generate the pmf.
49       */
50      protected final EnumeratedDistribution<Integer> innerDistribution;
51  
52      /**
53       * Create a discrete distribution.
54       *
55       * @param singletons array of random variable values.
56       * @param probabilities array of probabilities.
57       * @throws DimensionMismatchException if
58       * {@code singletons.length != probabilities.length}
59       * @throws NotPositiveException if any of the probabilities are negative.
60       * @throws NotFiniteNumberException if any of the probabilities are infinite.
61       * @throws NotANumberException if any of the probabilities are NaN.
62       * @throws MathArithmeticException all of the probabilities are 0.
63       */
64      public EnumeratedIntegerDistribution(final int[] singletons,
65                                           final double[] probabilities)
66          throws DimensionMismatchException,
67                 NotPositiveException,
68                 MathArithmeticException,
69                 NotFiniteNumberException,
70                 NotANumberException {
71          innerDistribution = new EnumeratedDistribution<Integer>(createDistribution(singletons,
72                                                                                     probabilities));
73      }
74  
75      /**
76       * Create a discrete integer-valued distribution from the input data.
77       * Values are assigned mass based on their frequency.
78       *
79       * @param data input dataset
80       */
81      public EnumeratedIntegerDistribution(final int[] data) {
82          final Map<Integer, Integer> dataMap = new HashMap<Integer, Integer>();
83          for (int value : data) {
84              Integer count = dataMap.get(value);
85              if (count == null) {
86                  count = 0;
87              }
88              dataMap.put(value, ++count);
89          }
90          final int massPoints = dataMap.size();
91          final double denom = data.length;
92          final int[] values = new int[massPoints];
93          final double[] probabilities = new double[massPoints];
94          int index = 0;
95          for (Entry<Integer, Integer> entry : dataMap.entrySet()) {
96              values[index] = entry.getKey();
97              probabilities[index] = entry.getValue().intValue() / denom;
98              index++;
99          }
100         innerDistribution = new EnumeratedDistribution<Integer>(createDistribution(values, probabilities));
101     }
102 
103     /**
104      * Create the list of Pairs representing the distribution from singletons and probabilities.
105      *
106      * @param singletons values
107      * @param probabilities probabilities
108      * @return list of value/probability pairs
109      */
110     private static List<Pair<Integer, Double>>  createDistribution(int[] singletons, double[] probabilities) {
111         if (singletons.length != probabilities.length) {
112             throw new DimensionMismatchException(probabilities.length, singletons.length);
113         }
114 
115         final List<Pair<Integer, Double>> samples = new ArrayList<Pair<Integer, Double>>(singletons.length);
116 
117         for (int i = 0; i < singletons.length; i++) {
118             samples.add(new Pair<Integer, Double>(singletons[i], probabilities[i]));
119         }
120         return samples;
121     }
122 
123     /**
124      * {@inheritDoc}
125      */
126     @Override
127     public double probability(final int x) {
128         return innerDistribution.probability(x);
129     }
130 
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     public double cumulativeProbability(final int x) {
136         double probability = 0;
137 
138         for (final Pair<Integer, Double> sample : innerDistribution.getPmf()) {
139             if (sample.getKey() <= x) {
140                 probability += sample.getValue();
141             }
142         }
143 
144         return probability;
145     }
146 
147     /**
148      * {@inheritDoc}
149      *
150      * @return {@code sum(singletons[i] * probabilities[i])}
151      */
152     @Override
153     public double getNumericalMean() {
154         double mean = 0;
155 
156         for (final Pair<Integer, Double> sample : innerDistribution.getPmf()) {
157             mean += sample.getValue() * sample.getKey();
158         }
159 
160         return mean;
161     }
162 
163     /**
164      * {@inheritDoc}
165      *
166      * @return {@code sum((singletons[i] - mean) ^ 2 * probabilities[i])}
167      */
168     @Override
169     public double getNumericalVariance() {
170         double mean = 0;
171         double meanOfSquares = 0;
172 
173         for (final Pair<Integer, Double> sample : innerDistribution.getPmf()) {
174             mean += sample.getValue() * sample.getKey();
175             meanOfSquares += sample.getValue() * sample.getKey() * sample.getKey();
176         }
177 
178         return meanOfSquares - mean * mean;
179     }
180 
181     /**
182      * {@inheritDoc}
183      *
184      * Returns the lowest value with non-zero probability.
185      *
186      * @return the lowest value with non-zero probability.
187      */
188     @Override
189     public int getSupportLowerBound() {
190         int min = Integer.MAX_VALUE;
191         for (final Pair<Integer, Double> sample : innerDistribution.getPmf()) {
192             if (sample.getKey() < min && sample.getValue() > 0) {
193                 min = sample.getKey();
194             }
195         }
196 
197         return min;
198     }
199 
200     /**
201      * {@inheritDoc}
202      *
203      * Returns the highest value with non-zero probability.
204      *
205      * @return the highest value with non-zero probability.
206      */
207     @Override
208     public int getSupportUpperBound() {
209         int max = Integer.MIN_VALUE;
210         for (final Pair<Integer, Double> sample : innerDistribution.getPmf()) {
211             if (sample.getKey() > max && sample.getValue() > 0) {
212                 max = sample.getKey();
213             }
214         }
215 
216         return max;
217     }
218 
219     /**
220      * {@inheritDoc}
221      *
222      * The support of this distribution is connected.
223      *
224      * @return {@code true}
225      */
226     @Override
227     public boolean isSupportConnected() {
228         return true;
229     }
230 
231     /** {@inheritDoc} */
232     @Override
233     public IntegerDistribution.Sampler createSampler(final UniformRandomProvider rng) {
234         return new IntegerDistribution.Sampler() {
235             /** Delegate. */
236             private final EnumeratedDistribution<Integer>.Sampler inner =
237                 innerDistribution.createSampler(rng);
238 
239             /** {@inheritDoc} */
240             @Override
241             public int sample() {
242                 return inner.sample();
243             }
244         };
245     }
246 }
