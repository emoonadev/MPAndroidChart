18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.BivariateFunction;
21  import org.apache.commons.math4.util.FastMath;
22  
23  /**
24   * Minimum function.
25   *
26   * @since 3.0
27   */
28  public class Min implements BivariateFunction {
29      /** {@inheritDoc} */
30      @Override
31      public double value(double x, double y) {
32          return FastMath.min(x, y);
33      }
34  }
