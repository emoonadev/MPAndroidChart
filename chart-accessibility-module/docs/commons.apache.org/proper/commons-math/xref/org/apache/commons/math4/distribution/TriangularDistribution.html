18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.exception.NumberIsTooLargeException;
21  import org.apache.commons.math4.exception.NumberIsTooSmallException;
22  import org.apache.commons.math4.exception.OutOfRangeException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  import org.apache.commons.math4.util.FastMath;
25  
26  /**
27   * Implementation of the triangular real distribution.
28   *
29   * @see <a href="http://en.wikipedia.org/wiki/Triangular_distribution">
30   * Triangular distribution (Wikipedia)</a>
31   *
32   * @since 3.0
33   */
34  public class TriangularDistribution extends AbstractRealDistribution {
35      /** Serializable version identifier. */
36      private static final long serialVersionUID = 20160311L;
37      /** Lower limit of this distribution (inclusive). */
38      private final double a;
39      /** Upper limit of this distribution (inclusive). */
40      private final double b;
41      /** Mode of this distribution. */
42      private final double c;
43      /** Inverse cumulative probability accuracy. */
44      private final double solverAbsoluteAccuracy;
45  
46      /**
47       * Creates a distribution.
48       *
49       * @param a Lower limit of this distribution (inclusive).
50       * @param b Upper limit of this distribution (inclusive).
51       * @param c Mode of this distribution.
52       * @throws NumberIsTooLargeException if {@code a >= b} or if {@code c > b}.
53       * @throws NumberIsTooSmallException if {@code c < a}.
54       * @since 3.1
55       */
56      public TriangularDistribution(double a,
57                                    double c,
58                                    double b)
59          throws NumberIsTooLargeException,
60                 NumberIsTooSmallException {
61          if (a >= b) {
62              throw new NumberIsTooLargeException(
63                              LocalizedFormats.LOWER_BOUND_NOT_BELOW_UPPER_BOUND,
64                              a, b, false);
65          }
66          if (c < a) {
67              throw new NumberIsTooSmallException(
68                      LocalizedFormats.NUMBER_TOO_SMALL, c, a, true);
69          }
70          if (c > b) {
71              throw new NumberIsTooLargeException(
72                      LocalizedFormats.NUMBER_TOO_LARGE, c, b, true);
73          }
74  
75          this.a = a;
76          this.c = c;
77          this.b = b;
78          solverAbsoluteAccuracy = FastMath.max(FastMath.ulp(a), FastMath.ulp(b));
79      }
80  
81      /**
82       * Returns the mode {@code c} of this distribution.
83       *
84       * @return the mode {@code c} of this distribution
85       */
86      public double getMode() {
87          return c;
88      }
89  
90      /**
91       * {@inheritDoc}
92       *
93       * <p>
94       * For this distribution, the returned value is not really meaningful,
95       * since exact formulas are implemented for the computation of the
96       * {@link #inverseCumulativeProbability(double)} (no solver is invoked).
97       * </p>
98       * <p>
99       * For lower limit {@code a} and upper limit {@code b}, the current
100      * implementation returns {@code max(ulp(a), ulp(b)}.
101      * </p>
102      */
103     @Override
104     protected double getSolverAbsoluteAccuracy() {
105         return solverAbsoluteAccuracy;
106     }
107 
108     /**
109      * {@inheritDoc}
110      *
111      * For lower limit {@code a}, upper limit {@code b} and mode {@code c}, the
112      * PDF is given by
113      * <ul>
114      * <li>{@code 2 * (x - a) / [(b - a) * (c - a)]} if {@code a <= x < c},</li>
115      * <li>{@code 2 / (b - a)} if {@code x = c},</li>
116      * <li>{@code 2 * (b - x) / [(b - a) * (b - c)]} if {@code c < x <= b},</li>
117      * <li>{@code 0} otherwise.
118      * </ul>
119      */
120     @Override
121     public double density(double x) {
122         if (x < a) {
123             return 0;
124         }
125         if (a <= x && x < c) {
126             double divident = 2 * (x - a);
127             double divisor = (b - a) * (c - a);
128             return divident / divisor;
129         }
130         if (x == c) {
131             return 2 / (b - a);
132         }
133         if (c < x && x <= b) {
134             double divident = 2 * (b - x);
135             double divisor = (b - a) * (b - c);
136             return divident / divisor;
137         }
138         return 0;
139     }
140 
141     /**
142      * {@inheritDoc}
143      *
144      * For lower limit {@code a}, upper limit {@code b} and mode {@code c}, the
145      * CDF is given by
146      * <ul>
147      * <li>{@code 0} if {@code x < a},</li>
148      * <li>{@code (x - a)^2 / [(b - a) * (c - a)]} if {@code a <= x < c},</li>
149      * <li>{@code (c - a) / (b - a)} if {@code x = c},</li>
150      * <li>{@code 1 - (b - x)^2 / [(b - a) * (b - c)]} if {@code c < x <= b},</li>
151      * <li>{@code 1} if {@code x > b}.</li>
152      * </ul>
153      */
154     @Override
155     public double cumulativeProbability(double x)  {
156         if (x < a) {
157             return 0;
158         }
159         if (a <= x && x < c) {
160             double divident = (x - a) * (x - a);
161             double divisor = (b - a) * (c - a);
162             return divident / divisor;
163         }
164         if (x == c) {
165             return (c - a) / (b - a);
166         }
167         if (c < x && x <= b) {
168             double divident = (b - x) * (b - x);
169             double divisor = (b - a) * (b - c);
170             return 1 - (divident / divisor);
171         }
172         return 1;
173     }
174 
175     /**
176      * {@inheritDoc}
177      *
178      * For lower limit {@code a}, upper limit {@code b}, and mode {@code c},
179      * the mean is {@code (a + b + c) / 3}.
180      */
181     @Override
182     public double getNumericalMean() {
183         return (a + b + c) / 3;
184     }
185 
186     /**
187      * {@inheritDoc}
188      *
189      * For lower limit {@code a}, upper limit {@code b}, and mode {@code c},
190      * the variance is {@code (a^2 + b^2 + c^2 - a * b - a * c - b * c) / 18}.
191      */
192     @Override
193     public double getNumericalVariance() {
194         return (a * a + b * b + c * c - a * b - a * c - b * c) / 18;
195     }
196 
197     /**
198      * {@inheritDoc}
199      *
200      * The lower bound of the support is equal to the lower limit parameter
201      * {@code a} of the distribution.
202      *
203      * @return lower bound of the support
204      */
205     @Override
206     public double getSupportLowerBound() {
207         return a;
208     }
209 
210     /**
211      * {@inheritDoc}
212      *
213      * The upper bound of the support is equal to the upper limit parameter
214      * {@code b} of the distribution.
215      *
216      * @return upper bound of the support
217      */
218     @Override
219     public double getSupportUpperBound() {
220         return b;
221     }
222 
223     /**
224      * {@inheritDoc}
225      *
226      * The support of this distribution is connected.
227      *
228      * @return {@code true}
229      */
230     @Override
231     public boolean isSupportConnected() {
232         return true;
233     }
234 
235     /** {@inheritDoc} */
236     @Override
237     public double inverseCumulativeProbability(double p)
238         throws OutOfRangeException {
239         if (p < 0 || p > 1) {
240             throw new OutOfRangeException(p, 0, 1);
241         }
242         if (p == 0) {
243             return a;
244         }
245         if (p == 1) {
246             return b;
247         }
248         if (p < (c - a) / (b - a)) {
249             return a + FastMath.sqrt(p * (b - a) * (c - a));
250         }
251         return b - FastMath.sqrt((1 - p) * (b - a) * (b - c));
252     }
253 }
