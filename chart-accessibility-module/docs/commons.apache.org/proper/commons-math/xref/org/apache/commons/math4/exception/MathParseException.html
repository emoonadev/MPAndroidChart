17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.LocalizedFormats;
20  
21  /**
22   * Class to signal parse failures.
23   *
24   * @since 2.2
25   */
26  public class MathParseException extends MathIllegalStateException {
27      /** Serializable version Id. */
28      private static final long serialVersionUID = -6024911025449780478L;
29  
30      /**
31       * @param wrong Bad string representation of the object.
32       * @param position Index, in the {@code wrong} string, that caused the
33       * parsing to fail.
34       * @param type Class of the object supposedly represented by the
35       * {@code wrong} string.
36       */
37      public MathParseException(String wrong, int position, Class<?> type) {
38          getContext().addMessage(LocalizedFormats.CANNOT_PARSE_AS_TYPE,
39                                  wrong, Integer.valueOf(position), type.getName());
40      }
41  
42      /**
43       * @param wrong Bad string representation of the object.
44       * @param position Index, in the {@code wrong} string, that caused the
45       * parsing to fail.
46       */
47      public MathParseException(String wrong, int position) {
48          getContext().addMessage(LocalizedFormats.CANNOT_PARSE,
49                                  wrong, Integer.valueOf(position));
50      }
51  }
