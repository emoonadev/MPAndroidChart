17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.exception.DimensionMismatchException;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when a square matrix is expected.
24   *
25   * @since 3.0
26   */
27  public class NonSquareMatrixException extends DimensionMismatchException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -660069396594485772L;
30  
31      /**
32       * Construct an exception from the mismatched dimensions.
33       *
34       * @param wrong Row dimension.
35       * @param expected Column dimension.
36       */
37      public NonSquareMatrixException(int wrong,
38                                      int expected) {
39          super(LocalizedFormats.NON_SQUARE_MATRIX, wrong, expected);
40      }
41  }
