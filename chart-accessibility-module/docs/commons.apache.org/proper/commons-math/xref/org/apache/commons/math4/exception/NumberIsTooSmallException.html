17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when a number is too small.
24   *
25   * @since 2.2
26   */
27  public class NumberIsTooSmallException extends MathIllegalNumberException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -6100997100383932834L;
30      /**
31       * Higher bound.
32       */
33      private final Number min;
34      /**
35       * Whether the maximum is included in the allowed range.
36       */
37      private final boolean boundIsAllowed;
38  
39      /**
40       * Construct the exception.
41       *
42       * @param wrong Value that is smaller than the minimum.
43       * @param min Minimum.
44       * @param boundIsAllowed Whether {@code min} is included in the allowed range.
45       */
46      public NumberIsTooSmallException(Number wrong,
47                                       Number min,
48                                       boolean boundIsAllowed) {
49          this(boundIsAllowed ?
50               LocalizedFormats.NUMBER_TOO_SMALL :
51               LocalizedFormats.NUMBER_TOO_SMALL_BOUND_EXCLUDED,
52               wrong, min, boundIsAllowed);
53      }
54  
55      /**
56       * Construct the exception with a specific context.
57       *
58       * @param specific Specific context pattern.
59       * @param wrong Value that is smaller than the minimum.
60       * @param min Minimum.
61       * @param boundIsAllowed Whether {@code min} is included in the allowed range.
62       */
63      public NumberIsTooSmallException(Localizable specific,
64                                       Number wrong,
65                                       Number min,
66                                       boolean boundIsAllowed) {
67          super(specific, wrong, min);
68  
69          this.min = min;
70          this.boundIsAllowed = boundIsAllowed;
71      }
72  
73      /**
74       * @return {@code true} if the minimum is included in the allowed range.
75       */
76      public boolean getBoundIsAllowed() {
77          return boundIsAllowed;
78      }
79  
80      /**
81       * @return the minimum.
82       */
83      public Number getMin() {
84          return min;
85      }
86  }
