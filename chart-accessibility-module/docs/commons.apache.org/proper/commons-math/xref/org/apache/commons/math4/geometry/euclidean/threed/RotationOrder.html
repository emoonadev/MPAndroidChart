18  package org.apache.commons.math4.geometry.euclidean.threed;
19  
20  /**
21   * This class is a utility representing a rotation order specification
22   * for Cardan or Euler angles specification.
23   *
24   * This class cannot be instanciated by the user. He can only use one
25   * of the twelve predefined supported orders as an argument to either
26   * the {@link Rotation#Rotation(RotationOrder,double,double,double)}
27   * constructor or the {@link Rotation#getAngles} method.
28   *
29   * @since 1.2
30   */
31  public final class RotationOrder {
32  
33      /** Set of Cardan angles.
34       * this ordered set of rotations is around X, then around Y, then
35       * around Z
36       */
37      public static final RotationOrder XYZ =
38        new RotationOrder("XYZ", Vector3D.PLUS_I, Vector3D.PLUS_J, Vector3D.PLUS_K);
39  
40      /** Set of Cardan angles.
41       * this ordered set of rotations is around X, then around Z, then
42       * around Y
43       */
44      public static final RotationOrder XZY =
45        new RotationOrder("XZY", Vector3D.PLUS_I, Vector3D.PLUS_K, Vector3D.PLUS_J);
46  
47      /** Set of Cardan angles.
48       * this ordered set of rotations is around Y, then around X, then
49       * around Z
50       */
51      public static final RotationOrder YXZ =
52        new RotationOrder("YXZ", Vector3D.PLUS_J, Vector3D.PLUS_I, Vector3D.PLUS_K);
53  
54      /** Set of Cardan angles.
55       * this ordered set of rotations is around Y, then around Z, then
56       * around X
57       */
58      public static final RotationOrder YZX =
59        new RotationOrder("YZX", Vector3D.PLUS_J, Vector3D.PLUS_K, Vector3D.PLUS_I);
60  
61      /** Set of Cardan angles.
62       * this ordered set of rotations is around Z, then around X, then
63       * around Y
64       */
65      public static final RotationOrder ZXY =
66        new RotationOrder("ZXY", Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J);
67  
68      /** Set of Cardan angles.
69       * this ordered set of rotations is around Z, then around Y, then
70       * around X
71       */
72      public static final RotationOrder ZYX =
73        new RotationOrder("ZYX", Vector3D.PLUS_K, Vector3D.PLUS_J, Vector3D.PLUS_I);
74  
75      /** Set of Euler angles.
76       * this ordered set of rotations is around X, then around Y, then
77       * around X
78       */
79      public static final RotationOrder XYX =
80        new RotationOrder("XYX", Vector3D.PLUS_I, Vector3D.PLUS_J, Vector3D.PLUS_I);
81  
82      /** Set of Euler angles.
83       * this ordered set of rotations is around X, then around Z, then
84       * around X
85       */
86      public static final RotationOrder XZX =
87        new RotationOrder("XZX", Vector3D.PLUS_I, Vector3D.PLUS_K, Vector3D.PLUS_I);
88  
89      /** Set of Euler angles.
90       * this ordered set of rotations is around Y, then around X, then
91       * around Y
92       */
93      public static final RotationOrder YXY =
94        new RotationOrder("YXY", Vector3D.PLUS_J, Vector3D.PLUS_I, Vector3D.PLUS_J);
95  
96      /** Set of Euler angles.
97       * this ordered set of rotations is around Y, then around Z, then
98       * around Y
99       */
100     public static final RotationOrder YZY =
101       new RotationOrder("YZY", Vector3D.PLUS_J, Vector3D.PLUS_K, Vector3D.PLUS_J);
102 
103     /** Set of Euler angles.
104      * this ordered set of rotations is around Z, then around X, then
105      * around Z
106      */
107     public static final RotationOrder ZXZ =
108       new RotationOrder("ZXZ", Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_K);
109 
110     /** Set of Euler angles.
111      * this ordered set of rotations is around Z, then around Y, then
112      * around Z
113      */
114     public static final RotationOrder ZYZ =
115       new RotationOrder("ZYZ", Vector3D.PLUS_K, Vector3D.PLUS_J, Vector3D.PLUS_K);
116 
117     /** Name of the rotations order. */
118     private final String name;
119 
120     /** Axis of the first rotation. */
121     private final Vector3D a1;
122 
123     /** Axis of the second rotation. */
124     private final Vector3D a2;
125 
126     /** Axis of the third rotation. */
127     private final Vector3D a3;
128 
129     /** Private constructor.
130      * This is a utility class that cannot be instantiated by the user,
131      * so its only constructor is private.
132      * @param name name of the rotation order
133      * @param a1 axis of the first rotation
134      * @param a2 axis of the second rotation
135      * @param a3 axis of the third rotation
136      */
137     private RotationOrder(final String name,
138                           final Vector3D a1, final Vector3D a2, final Vector3D a3) {
139         this.name = name;
140         this.a1   = a1;
141         this.a2   = a2;
142         this.a3   = a3;
143     }
144 
145     /** Get a string representation of the instance.
146      * @return a string representation of the instance (in fact, its name)
147      */
148     @Override
149     public String toString() {
150         return name;
151     }
152 
153     /** Get the axis of the first rotation.
154      * @return axis of the first rotation
155      */
156     public Vector3D getA1() {
157         return a1;
158     }
159 
160     /** Get the axis of the second rotation.
161      * @return axis of the second rotation
162      */
163     public Vector3D getA2() {
164         return a2;
165     }
166 
167     /** Get the axis of the second rotation.
168      * @return axis of the second rotation
169      */
170     public Vector3D getA3() {
171         return a3;
172     }
173 
174 }
