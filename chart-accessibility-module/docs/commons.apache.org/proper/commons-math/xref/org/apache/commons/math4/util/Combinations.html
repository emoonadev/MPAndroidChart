25  import org.apache.commons.math4.exception.DimensionMismatchException;

29  /**
30   * Utility to create <a href="http://en.wikipedia.org/wiki/Combination">
31   * combinations</a> {@code (n, k)} of {@code k} elements in a set of
32   * {@code n} elements.
33   *
34   * @since 3.3
35   */
36  public class Combinations implements Iterable<int[]> {
37      /** Size of the set from which combinations are drawn. */
38      private final int n;
39      /** Number of elements in each combination. */
40      private final int k;
41      /** Iteration order. */
42      private final IterationOrder iterationOrder;
43  
44      /**
45       * Describes the type of iteration performed by the
46       * {@link #iterator() iterator}.
47       */
48      private enum IterationOrder {
49          /** Lexicographic order. */
50          LEXICOGRAPHIC
51      }
52  
53     /**
54       * Creates an instance whose range is the k-element subsets of
55       * {0, ..., n - 1} represented as {@code int[]} arrays.
56       * <p>
57       * The iteration order is lexicographic: the arrays returned by the
58       * {@link #iterator() iterator} are sorted in descending order and
59       * they are visited in lexicographic order with significance from
60       * right to left.
61       * For example, {@code new Combinations(4, 2).iterator()} returns
62       * an iterator that will generate the following sequence of arrays
63       * on successive calls to
64       * {@code next()}:<br/>
65       * {@code [0, 1], [0, 2], [1, 2], [0, 3], [1, 3], [2, 3]}
66       * </p>
67       * If {@code k == 0} an iterator containing an empty array is returned;
68       * if {@code k == n} an iterator containing [0, ..., n - 1] is returned.
69       *
70       * @param n Size of the set from which subsets are selected.
71       * @param k Size of the subsets to be enumerated.
72       * @throws org.apache.commons.math4.exception.NotPositiveException if {@code n < 0}.
73       * @throws org.apache.commons.math4.exception.NumberIsTooLargeException if {@code k > n}.
74       */
75      public Combinations(int n,
76                          int k) {
77          this(n, k, IterationOrder.LEXICOGRAPHIC);
78      }
79  
80      /**
81       * Creates an instance whose range is the k-element subsets of
82       * {0, ..., n - 1} represented as {@code int[]} arrays.
83       * <p>
84       * If the {@code iterationOrder} argument is set to
85       * {@link IterationOrder#LEXICOGRAPHIC}, the arrays returned by the
86       * {@link #iterator() iterator} are sorted in descending order and
87       * they are visited in lexicographic order with significance from
88       * right to left.
89       * For example, {@code new Combinations(4, 2).iterator()} returns
90       * an iterator that will generate the following sequence of arrays
91       * on successive calls to
92       * {@code next()}:<br/>
93       * {@code [0, 1], [0, 2], [1, 2], [0, 3], [1, 3], [2, 3]}
94       * </p>
95       * If {@code k == 0} an iterator containing an empty array is returned;
96       * if {@code k == n} an iterator containing [0, ..., n - 1] is returned.
97       *
98       * @param n Size of the set from which subsets are selected.
99       * @param k Size of the subsets to be enumerated.
100      * @param iterationOrder Specifies the {@link #iterator() iteration order}.
101      * @throws org.apache.commons.math4.exception.NotPositiveException if {@code n < 0}.
102      * @throws org.apache.commons.math4.exception.NumberIsTooLargeException if {@code k > n}.
103      */
104     private Combinations(int n,
105                          int k,
106                          IterationOrder iterationOrder) {
107         CombinatoricsUtils.checkBinomial(n, k);
108         this.n = n;
109         this.k = k;
110         this.iterationOrder = iterationOrder;
111     }
112 
113     /**
114      * Gets the size of the set from which combinations are drawn.
115      *
116      * @return the size of the universe.
117      */
118     public int getN() {
119         return n;
120     }
121 
122     /**
123      * Gets the number of elements in each combination.
124      *
125      * @return the size of the subsets to be enumerated.
126      */
127     public int getK() {
128         return k;
129     }
130 
131     /** {@inheritDoc} */
132     @Override
133     public Iterator<int[]> iterator() {
134         if (k == 0 ||
135             k == n) {
136             return new SingletonIterator(MathArrays.natural(k));
137         }
138 
139         switch (iterationOrder) {
140         case LEXICOGRAPHIC:
141             return new LexicographicIterator(n, k);
142         default:
143             throw new MathInternalError(); // Should never happen.
144         }
145     }
146 
147     /**
148      * Defines a lexicographic ordering of combinations.
149      * The returned comparator allows to compare any two combinations
150      * that can be produced by this instance's {@link #iterator() iterator}.
151      * Its {@code compare(int[],int[])} method will throw exceptions if
152      * passed combinations that are inconsistent with this instance:
153      * <ul>
154      *  <li>{@code DimensionMismatchException} if the array lengths are not
155      *      equal to {@code k},</li>
156      *  <li>{@code OutOfRangeException} if an element of the array is not
157      *      within the interval [0, {@code n}).</li>
158      * </ul>
159      * @return a lexicographic comparator.
160      */
161     public Comparator<int[]> comparator() {
162         return new LexicographicComparator(n, k);
163     }
164 
165     /**
166      * Lexicographic combinations iterator.
167      * <p>
168      * Implementation follows Algorithm T in <i>The Art of Computer Programming</i>
169      * Internet Draft (PRE-FASCICLE 3A), "A Draft of Section 7.2.1.3 Generating All
170      * Combinations</a>, D. Knuth, 2004.</p>
171      * <p>
172      * The degenerate cases {@code k == 0} and {@code k == n} are NOT handled by this
173      * implementation.  If constructor arguments satisfy {@code k == 0}
174      * or {@code k >= n}, no exception is generated, but the iterator is empty.
175      * </p>
176      *
177      */
178     private static class LexicographicIterator implements Iterator<int[]> {
179         /** Size of subsets returned by the iterator */
180         private final int k;
181 
182         /**
183          * c[1], ..., c[k] stores the next combination; c[k + 1], c[k + 2] are
184          * sentinels.
185          * <p>
186          * Note that c[0] is "wasted" but this makes it a little easier to
187          * follow the code.
188          * </p>
189          */
190         private final int[] c;
191 
192         /** Return value for {@link #hasNext()} */
193         private boolean more = true;
194 
195         /** Marker: smallest index such that c[j + 1] > j */
196         private int j;
197 
198         /**
199          * Construct a CombinationIterator to enumerate k-sets from n.
200          * <p>
201          * NOTE: If {@code k === 0} or {@code k >= n}, the Iterator will be empty
202          * (that is, {@link #hasNext()} will return {@code false} immediately.
203          * </p>
204          *
205          * @param n size of the set from which subsets are enumerated
206          * @param k size of the subsets to enumerate
207          */
208         LexicographicIterator(int n, int k) {
209             this.k = k;
210             c = new int[k + 3];
211             if (k == 0 || k >= n) {
212                 more = false;
213                 return;
214             }
215             // Initialize c to start with lexicographically first k-set
216             for (int i = 1; i <= k; i++) {
217                 c[i] = i - 1;
218             }
219             // Initialize sentinels
220             c[k + 1] = n;
221             c[k + 2] = 0;
222             j = k; // Set up invariant: j is smallest index such that c[j + 1] > j
223         }
224 
225         /**
226          * {@inheritDoc}
227          */
228         @Override
229         public boolean hasNext() {
230             return more;
231         }
232 
233         /**
234          * {@inheritDoc}
235          */
236         @Override
237         public int[] next() {
238             if (!more) {
239                 throw new NoSuchElementException();
240             }
241             // Copy return value (prepared by last activation)
242             final int[] ret = new int[k];
243             System.arraycopy(c, 1, ret, 0, k);
244 
245             // Prepare next iteration
246             // T2 and T6 loop
247             int x = 0;
248             if (j > 0) {
249                 x = j;
250                 c[j] = x;
251                 j--;
252                 return ret;
253             }
254             // T3
255             if (c[1] + 1 < c[2]) {
256                 c[1]++;
257                 return ret;
258             } else {
259                 j = 2;
260             }
261             // T4
262             boolean stepDone = false;
263             while (!stepDone) {
264                 c[j - 1] = j - 2;
265                 x = c[j] + 1;
266                 if (x == c[j + 1]) {
267                     j++;
268                 } else {
269                     stepDone = true;
270                 }
271             }
272             // T5
273             if (j > k) {
274                 more = false;
275                 return ret;
276             }
277             // T6
278             c[j] = x;
279             j--;
280             return ret;
281         }
282 
283         /**
284          * Not supported.
285          */
286         @Override
287         public void remove() {
288             throw new UnsupportedOperationException();
289         }
290     }

292     /**
293      * Iterator with just one element to handle degenerate cases (full array,
294      * empty array) for combination iterator.
295      */
296     private static class SingletonIterator implements Iterator<int[]> {
297         /** Singleton array */
298         private final int[] singleton;
299         /** True on initialization, false after first call to next */
300         private boolean more = true;
301         /**
302          * Create a singleton iterator providing the given array.
303          * @param singleton array returned by the iterator
304          */
305         SingletonIterator(final int[] singleton) {
306             this.singleton = singleton;
307         }
308         /** @return True until next is called the first time, then false */
309         @Override
310         public boolean hasNext() {
311             return more;
312         }
313         /** @return the singleton in first activation; throws NSEE thereafter */
314         @Override
315         public int[] next() {
316             if (more) {
317                 more = false;
318                 return singleton;
319             } else {
320                 throw new NoSuchElementException();
321             }
322         }
323         /** Not supported */
324         @Override
325         public void remove() {
326             throw new UnsupportedOperationException();
327         }
328     }

330     /**
331      * Defines the lexicographic ordering of combinations, using
332      * the {@link #lexNorm(int[])} method.
333      */
334     private static class LexicographicComparator
335         implements Comparator<int[]>, Serializable {
336         /** Serializable version identifier. */
337         private static final long serialVersionUID = 20130906L;
338         /** Size of the set from which combinations are drawn. */
339         private final int n;
340         /** Number of elements in each combination. */
341         private final int k;
342 
343         /**
344          * @param n Size of the set from which subsets are selected.
345          * @param k Size of the subsets to be enumerated.
346          */
347         LexicographicComparator(int n, int k) {
348             this.n = n;
349             this.k = k;
350         }
351 
352         /**
353          * {@inheritDoc}
354          *
355          * @throws DimensionMismatchException if the array lengths are not
356          * equal to {@code k}.
357          * @throws OutOfRangeException if an element of the array is not
358          * within the interval [0, {@code n}).
359          */
360         @Override
361         public int compare(int[] c1,
362                            int[] c2) {
363             if (c1.length != k) {
364                 throw new DimensionMismatchException(c1.length, k);
365             }
366             if (c2.length != k) {
367                 throw new DimensionMismatchException(c2.length, k);
368             }
369 
370             // Method "lexNorm" works with ordered arrays.
371             final int[] c1s = MathArrays.copyOf(c1);
372             Arrays.sort(c1s);
373             final int[] c2s = MathArrays.copyOf(c2);
374             Arrays.sort(c2s);
375 
376             final long v1 = lexNorm(c1s);
377             final long v2 = lexNorm(c2s);
378 
379             if (v1 < v2) {
380                 return -1;
381             } else if (v1 > v2) {
382                 return 1;
383             } else {
384                 return 0;
385             }
386         }
387 
388         /**
389          * Computes the value (in base 10) represented by the digit
390          * (interpreted in base {@code n}) in the input array in reverse
391          * order.
392          * For example if {@code c} is {@code {3, 2, 1}}, and {@code n}
393          * is 3, the method will return 18.
394          *
395          * @param c Input array.
396          * @return the lexicographic norm.
397          * @throws OutOfRangeException if an element of the array is not
398          * within the interval [0, {@code n}).
399          */
400         private long lexNorm(int[] c) {
401             long ret = 0;
402             for (int i = 0; i < c.length; i++) {
403                 final int digit = c[i];
404                 if (digit < 0 ||
405                     digit >= n) {
406                     throw new OutOfRangeException(digit, 0, n - 1);
407                 }
408 
409                 ret += c[i] * ArithmeticUtils.pow(n, i);
410             }
411             return ret;
412         }
413     }
