17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.math4.util.FastMath;
23  
24  /**
25   * This class implements the Laplace distribution.
26   *
27   * @see <a href="http://en.wikipedia.org/wiki/Laplace_distribution">Laplace distribution (Wikipedia)</a>
28   *
29   * @since 3.4
30   */
31  public class LaplaceDistribution extends AbstractRealDistribution {
32  
33      /** Serializable version identifier. */
34      private static final long serialVersionUID = 20160311L;
35  
36      /** The location parameter. */
37      private final double mu;
38      /** The scale parameter. */
39      private final double beta;
40  
41      /**
42       * Creates a distribution.
43       *
44       * @param mu location parameter
45       * @param beta scale parameter (must be positive)
46       * @throws NotStrictlyPositiveException if {@code beta <= 0}
47       */
48      public LaplaceDistribution(double mu, double beta) {
49          if (beta <= 0.0) {
50              throw new NotStrictlyPositiveException(LocalizedFormats.NOT_POSITIVE_SCALE, beta);
51          }
52  
53          this.mu = mu;
54          this.beta = beta;
55      }
56  
57      /**
58       * Access the location parameter, {@code mu}.
59       *
60       * @return the location parameter.
61       */
62      public double getLocation() {
63          return mu;
64      }
65  
66      /**
67       * Access the scale parameter, {@code beta}.
68       *
69       * @return the scale parameter.
70       */
71      public double getScale() {
72          return beta;
73      }
74  
75      /** {@inheritDoc} */
76      @Override
77      public double density(double x) {
78          return FastMath.exp(-FastMath.abs(x - mu) / beta) / (2.0 * beta);
79      }
80  
81      /** {@inheritDoc} */
82      @Override
83      public double cumulativeProbability(double x) {
84          if (x <= mu) {
85              return FastMath.exp((x - mu) / beta) / 2.0;
86          } else {
87              return 1.0 - FastMath.exp((mu - x) / beta) / 2.0;
88          }
89      }
90  
91      /** {@inheritDoc} */
92      @Override
93      public double inverseCumulativeProbability(double p) throws OutOfRangeException {
94          if (p < 0.0 || p > 1.0) {
95              throw new OutOfRangeException(p, 0.0, 1.0);
96          } else if (p == 0) {
97              return Double.NEGATIVE_INFINITY;
98          } else if (p == 1) {
99              return Double.POSITIVE_INFINITY;
100         }
101         double x = (p > 0.5) ? -Math.log(2.0 - 2.0 * p) : Math.log(2.0 * p);
102         return mu + beta * x;
103     }
104 
105     /** {@inheritDoc} */
106     @Override
107     public double getNumericalMean() {
108         return mu;
109     }
110 
111     /** {@inheritDoc} */
112     @Override
113     public double getNumericalVariance() {
114         return 2.0 * beta * beta;
115     }
116 
117     /** {@inheritDoc} */
118     @Override
119     public double getSupportLowerBound() {
120         return Double.NEGATIVE_INFINITY;
121     }
122 
123     /** {@inheritDoc} */
124     @Override
125     public double getSupportUpperBound() {
126         return Double.POSITIVE_INFINITY;
127     }
128 
129     /** {@inheritDoc} */
130     @Override
131     public boolean isSupportConnected() {
132         return true;
133     }
134 
135 }
