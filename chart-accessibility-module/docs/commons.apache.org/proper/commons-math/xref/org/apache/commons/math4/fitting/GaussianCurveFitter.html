37  /**
38   * Fits points to a {@link
39   * org.apache.commons.math4.analysis.function.Gaussian.Parametric Gaussian}
40   * function.
41   * <br/>
42   * The {@link #withStartPoint(double[]) initial guess values} must be passed
43   * in the following order:
44   * <ul>
45   *  <li>Normalization</li>
46   *  <li>Mean</li>
47   *  <li>Sigma</li>
48   * </ul>
49   * The optimal values will be returned in the same order.
50   *
51   * <p>
52   * Usage example:
53   * <pre>
54   *   WeightedObservedPoints obs = new WeightedObservedPoints();
55   *   obs.add(4.0254623,  531026.0);
56   *   obs.add(4.03128248, 984167.0);
57   *   obs.add(4.03839603, 1887233.0);
58   *   obs.add(4.04421621, 2687152.0);
59   *   obs.add(4.05132976, 3461228.0);
60   *   obs.add(4.05326982, 3580526.0);
61   *   obs.add(4.05779662, 3439750.0);
62   *   obs.add(4.0636168,  2877648.0);
63   *   obs.add(4.06943698, 2175960.0);
64   *   obs.add(4.07525716, 1447024.0);
65   *   obs.add(4.08237071, 717104.0);
66   *   obs.add(4.08366408, 620014.0);
67   *   double[] parameters = GaussianCurveFitter.create().fit(obs.toList());
68   * </pre>
69   *
70   * @since 3.3
71   */
72  public class GaussianCurveFitter extends AbstractCurveFitter {
73      /** Parametric function to be fitted. */
74      private static final Gaussian.Parametric FUNCTION = new Gaussian.Parametric() {
75              /** {@inheritDoc} */
76              @Override
77              public double value(double x, double ... p) {
78                  double v = Double.POSITIVE_INFINITY;
79                  try {
80                      v = super.value(x, p);
81                  } catch (NotStrictlyPositiveException e) { // NOPMD
82                      // Do nothing.
83                  }
84                  return v;
85              }
86  
87              /** {@inheritDoc} */
88              @Override
89              public double[] gradient(double x, double ... p) {
90                  double[] v = { Double.POSITIVE_INFINITY,
91                                 Double.POSITIVE_INFINITY,
92                                 Double.POSITIVE_INFINITY };
93                  try {
94                      v = super.gradient(x, p);
95                  } catch (NotStrictlyPositiveException e) { // NOPMD
96                      // Do nothing.
97                  }
98                  return v;
99              }
100         };
101     /** Initial guess. */
102     private final double[] initialGuess;
103     /** Maximum number of iterations of the optimization algorithm. */
104     private final int maxIter;
105 
106     /**
107      * Contructor used by the factory methods.
108      *
109      * @param initialGuess Initial guess. If set to {@code null}, the initial guess
110      * will be estimated using the {@link ParameterGuesser}.
111      * @param maxIter Maximum number of iterations of the optimization algorithm.
112      */
113     private GaussianCurveFitter(double[] initialGuess,
114                                 int maxIter) {
115         this.initialGuess = initialGuess;
116         this.maxIter = maxIter;
117     }

119     /**
120      * Creates a default curve fitter.
121      * The initial guess for the parameters will be {@link ParameterGuesser}
122      * computed automatically, and the maximum number of iterations of the
123      * optimization algorithm is set to {@link Integer#MAX_VALUE}.
124      *
125      * @return a curve fitter.
126      *
127      * @see #withStartPoint(double[])
128      * @see #withMaxIterations(int)
129      */
130     public static GaussianCurveFitter create() {
131         return new GaussianCurveFitter(null, Integer.MAX_VALUE);
132     }

134     /**
135      * Configure the start point (initial guess).
136      * @param newStart new start point (initial guess)
137      * @return a new instance.
138      */
139     public GaussianCurveFitter withStartPoint(double[] newStart) {
140         return new GaussianCurveFitter(newStart.clone(),
141                                        maxIter);
142     }

144     /**
145      * Configure the maximum number of iterations.
146      * @param newMaxIter maximum number of iterations
147      * @return a new instance.
148      */
149     public GaussianCurveFitter withMaxIterations(int newMaxIter) {
150         return new GaussianCurveFitter(initialGuess,
151                                        newMaxIter);
152     }

154     /** {@inheritDoc} */
155     @Override
156     protected LeastSquaresProblem getProblem(Collection<WeightedObservedPoint> observations) {
157 
158         // Prepare least-squares problem.
159         final int len = observations.size();
160         final double[] target  = new double[len];
161         final double[] weights = new double[len];
162 
163         int i = 0;
164         for (WeightedObservedPoint obs : observations) {
165             target[i]  = obs.getY();
166             weights[i] = obs.getWeight();
167             ++i;
168         }
169 
170         final AbstractCurveFitter.TheoreticalValuesFunction model =
171                 new AbstractCurveFitter.TheoreticalValuesFunction(FUNCTION, observations);
172 
173         final double[] startPoint = initialGuess != null ?
174             initialGuess :
175             // Compute estimation.
176             new ParameterGuesser(observations).guess();
177 
178         // Return a new least squares problem set up to fit a Gaussian curve to the
179         // observed points.
180         return new LeastSquaresBuilder().
181                 maxEvaluations(Integer.MAX_VALUE).
182                 maxIterations(maxIter).
183                 start(startPoint).
184                 target(target).
185                 weight(new DiagonalMatrix(weights)).
186                 model(model.getModelFunction(), model.getModelFunctionJacobian()).
187                 build();
188 
189     }

191     /**
192      * Guesses the parameters {@code norm}, {@code mean}, and {@code sigma}
193      * of a {@link org.apache.commons.math4.analysis.function.Gaussian.Parametric}
194      * based on the specified observed points.
195      */
196     public static class ParameterGuesser {
197         /** Normalization factor. */
198         private final double norm;
199         /** Mean. */
200         private final double mean;
201         /** Standard deviation. */
202         private final double sigma;
203 
204         /**
205          * Constructs instance with the specified observed points.
206          *
207          * @param observations Observed points from which to guess the
208          * parameters of the Gaussian.
209          * @throws NullArgumentException if {@code observations} is
210          * {@code null}.
211          * @throws NumberIsTooSmallException if there are less than 3
212          * observations.
213          */
214         public ParameterGuesser(Collection<WeightedObservedPoint> observations) {
215             if (observations == null) {
216                 throw new NullArgumentException(LocalizedFormats.INPUT_ARRAY);
217             }
218             if (observations.size() < 3) {
219                 throw new NumberIsTooSmallException(observations.size(), 3, true);
220             }
221 
222             final List<WeightedObservedPoint> sorted = sortObservations(observations);
223             final double[] params = basicGuess(sorted.toArray(new WeightedObservedPoint[0]));
224 
225             norm = params[0];
226             mean = params[1];
227             sigma = params[2];
228         }
229 
230         /**
231          * Gets an estimation of the parameters.
232          *
233          * @return the guessed parameters, in the following order:
234          * <ul>
235          *  <li>Normalization factor</li>
236          *  <li>Mean</li>
237          *  <li>Standard deviation</li>
238          * </ul>
239          */
240         public double[] guess() {
241             return new double[] { norm, mean, sigma };
242         }
243 
244         /**
245          * Sort the observations.
246          *
247          * @param unsorted Input observations.
248          * @return the input observations, sorted.
249          */
250         private List<WeightedObservedPoint> sortObservations(Collection<WeightedObservedPoint> unsorted) {
251             final List<WeightedObservedPoint> observations = new ArrayList<WeightedObservedPoint>(unsorted);
252 
253             final Comparator<WeightedObservedPoint> cmp = new Comparator<WeightedObservedPoint>() {
254                 /** {@inheritDoc} */
255                 @Override
256                 public int compare(WeightedObservedPoint p1,
257                                    WeightedObservedPoint p2) {
258                     if (p1 == null && p2 == null) {
259                         return 0;
260                     }
261                     if (p1 == null) {
262                         return -1;
263                     }
264                     if (p2 == null) {
265                         return 1;
266                     }
267                     int comp = Double.compare(p1.getX(), p2.getX());
268                     if (comp != 0) {
269                         return comp;
270                     }
271                     comp = Double.compare(p1.getY(), p2.getY());
272                     if (comp != 0) {
273                         return comp;
274                     }
275                     comp = Double.compare(p1.getWeight(), p2.getWeight());
276                     if (comp != 0) {
277                         return comp;
278                     }
279                     return 0;
280                 }
281             };
282 
283             Collections.sort(observations, cmp);
284             return observations;
285         }
286 
287         /**
288          * Guesses the parameters based on the specified observed points.
289          *
290          * @param points Observed points, sorted.
291          * @return the guessed parameters (normalization factor, mean and
292          * sigma).
293          */
294         private double[] basicGuess(WeightedObservedPoint[] points) {
295             final int maxYIdx = findMaxY(points);
296             final double n = points[maxYIdx].getY();
297             final double m = points[maxYIdx].getX();
298 
299             double fwhmApprox;
300             try {
301                 final double halfY = n + ((m - n) / 2);
302                 final double fwhmX1 = interpolateXAtY(points, maxYIdx, -1, halfY);
303                 final double fwhmX2 = interpolateXAtY(points, maxYIdx, 1, halfY);
304                 fwhmApprox = fwhmX2 - fwhmX1;
305             } catch (OutOfRangeException e) {
306                 // TODO: Exceptions should not be used for flow control.
307                 fwhmApprox = points[points.length - 1].getX() - points[0].getX();
308             }
309             final double s = fwhmApprox / (2 * FastMath.sqrt(2 * FastMath.log(2)));
310 
311             return new double[] { n, m, s };
312         }
313 
314         /**
315          * Finds index of point in specified points with the largest Y.
316          *
317          * @param points Points to search.
318          * @return the index in specified points array.
319          */
320         private int findMaxY(WeightedObservedPoint[] points) {
321             int maxYIdx = 0;
322             for (int i = 1; i < points.length; i++) {
323                 if (points[i].getY() > points[maxYIdx].getY()) {
324                     maxYIdx = i;
325                 }
326             }
327             return maxYIdx;
328         }
329 
330         /**
331          * Interpolates using the specified points to determine X at the
332          * specified Y.
333          *
334          * @param points Points to use for interpolation.
335          * @param startIdx Index within points from which to start the search for
336          * interpolation bounds points.
337          * @param idxStep Index step for searching interpolation bounds points.
338          * @param y Y value for which X should be determined.
339          * @return the value of X for the specified Y.
340          * @throws ZeroException if {@code idxStep} is 0.
341          * @throws OutOfRangeException if specified {@code y} is not within the
342          * range of the specified {@code points}.
343          */
344         private double interpolateXAtY(WeightedObservedPoint[] points,
345                                        int startIdx,
346                                        int idxStep,
347                                        double y)
348             throws OutOfRangeException {
349             if (idxStep == 0) {
350                 throw new ZeroException();
351             }
352             final WeightedObservedPoint[] twoPoints
353                 = getInterpolationPointsForY(points, startIdx, idxStep, y);
354             final WeightedObservedPoint p1 = twoPoints[0];
355             final WeightedObservedPoint p2 = twoPoints[1];
356             if (p1.getY() == y) {
357                 return p1.getX();
358             }
359             if (p2.getY() == y) {
360                 return p2.getX();
361             }
362             return p1.getX() + (((y - p1.getY()) * (p2.getX() - p1.getX())) /
363                                 (p2.getY() - p1.getY()));
364         }
365 
366         /**
367          * Gets the two bounding interpolation points from the specified points
368          * suitable for determining X at the specified Y.
369          *
370          * @param points Points to use for interpolation.
371          * @param startIdx Index within points from which to start search for
372          * interpolation bounds points.
373          * @param idxStep Index step for search for interpolation bounds points.
374          * @param y Y value for which X should be determined.
375          * @return the array containing two points suitable for determining X at
376          * the specified Y.
377          * @throws ZeroException if {@code idxStep} is 0.
378          * @throws OutOfRangeException if specified {@code y} is not within the
379          * range of the specified {@code points}.
380          */
381         private WeightedObservedPoint[] getInterpolationPointsForY(WeightedObservedPoint[] points,
382                                                                    int startIdx,
383                                                                    int idxStep,
384                                                                    double y)
385             throws OutOfRangeException {
386             if (idxStep == 0) {
387                 throw new ZeroException();
388             }
389             for (int i = startIdx;
390                  idxStep < 0 ? i + idxStep >= 0 : i + idxStep < points.length;
391                  i += idxStep) {
392                 final WeightedObservedPoint p1 = points[i];
393                 final WeightedObservedPoint p2 = points[i + idxStep];
394                 if (isBetween(y, p1.getY(), p2.getY())) {
395                     if (idxStep < 0) {
396                         return new WeightedObservedPoint[] { p2, p1 };
397                     } else {
398                         return new WeightedObservedPoint[] { p1, p2 };
399                     }
400                 }
401             }
402 
403             // Boundaries are replaced by dummy values because the raised
404             // exception is caught and the message never displayed.
405             // TODO: Exceptions should not be used for flow control.
406             throw new OutOfRangeException(y,
407                                           Double.NEGATIVE_INFINITY,
408                                           Double.POSITIVE_INFINITY);
409         }
410 
411         /**
412          * Determines whether a value is between two other values.
413          *
414          * @param value Value to test whether it is between {@code boundary1}
415          * and {@code boundary2}.
416          * @param boundary1 One end of the range.
417          * @param boundary2 Other end of the range.
418          * @return {@code true} if {@code value} is between {@code boundary1} and
419          * {@code boundary2} (inclusive), {@code false} otherwise.
420          */
421         private boolean isBetween(double value,
422                                   double boundary1,
423                                   double boundary2) {
424             return (value >= boundary1 && value <= boundary2) ||
425                 (value >= boundary2 && value <= boundary1);
426         }
427     }
