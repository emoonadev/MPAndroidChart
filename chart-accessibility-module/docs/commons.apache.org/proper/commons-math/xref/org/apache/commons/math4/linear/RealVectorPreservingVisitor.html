17  package org.apache.commons.math4.linear;
18  
19  /**
20   * This interface defines a visitor for the entries of a vector. Visitors
21   * implementing this interface do not alter the entries of the vector being
22   * visited.
23   *
24   * @since 3.1
25   */
26  public interface RealVectorPreservingVisitor {
27      /**
28       * Start visiting a vector. This method is called once, before any entry
29       * of the vector is visited.
30       *
31       * @param dimension the size of the vector
32       * @param start the index of the first entry to be visited
33       * @param end the index of the last entry to be visited (inclusive)
34       */
35      void start(int dimension, int start, int end);
36  
37      /**
38       * Visit one entry of the vector.
39       *
40       * @param index the index of the entry being visited
41       * @param value the value of the entry being visited
42       */
43      void visit(int index, double value);
44  
45      /**
46       * End visiting a vector. This method is called once, after all entries of
47       * the vector have been visited.
48       *
49       * @return the value returned by
50       * {@link RealVector#walkInDefaultOrder(RealVectorPreservingVisitor)},
51       * {@link RealVector#walkInDefaultOrder(RealVectorPreservingVisitor, int, int)},
52       * {@link RealVector#walkInOptimizedOrder(RealVectorPreservingVisitor)}
53       * or
54       * {@link RealVector#walkInOptimizedOrder(RealVectorPreservingVisitor, int, int)}
55       */
56      double end();
57  }
