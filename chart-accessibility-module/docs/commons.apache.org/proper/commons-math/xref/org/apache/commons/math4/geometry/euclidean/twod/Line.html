35  /** This class represents an oriented line in the 2D plane.
36  
37   * <p>An oriented line can be defined either by prolongating a line
38   * segment between two points past these points, or by one point and
39   * an angular direction (in trigonometric orientation).</p>
40  
41   * <p>Since it is oriented the two half planes at its two sides are
42   * unambiguously identified as a left half plane and a right half
43   * plane. This can be used to identify the interior and the exterior
44   * in a simple way by local properties only when part of a line is
45   * used to define part of a polygon boundary.</p>
46  
47   * <p>A line can also be used to completely define a reference frame
48   * in the plane. It is sufficient to select one specific point in the
49   * line (the orthogonal projection of the original reference frame on
50   * the line) and to use the unit vector in the line direction and the
51   * orthogonal vector oriented from left half plane to right half
52   * plane. We define two coordinates by the process, the
53   * <em>abscissa</em> along the line, and the <em>offset</em> across
54   * the line. All points of the plane are uniquely identified by these
55   * two coordinates. The line is the set of points at zero offset, the
56   * left half plane is the set of points with negative offsets and the
57   * right half plane is the set of points with positive offsets.</p>
58  
59   * @since 3.0
60   */
61  public class Line implements Hyperplane<Euclidean2D>, Embedding<Euclidean2D, Euclidean1D> {
62  
63      /** Angle with respect to the abscissa axis. */
64      private double angle;
65  
66      /** Cosine of the line angle. */
67      private double cos;
68  
69      /** Sine of the line angle. */
70      private double sin;
71  
72      /** Offset of the frame origin. */
73      private double originOffset;
74  
75      /** Tolerance below which points are considered identical. */
76      private final double tolerance;
77  
78      /** Reverse line. */
79      private Line reverse;

81      /** Build a line from two points.
82       * <p>The line is oriented from p1 to p2</p>
83       * @param p1 first point
84       * @param p2 second point
85       * @param tolerance tolerance below which points are considered identical
86       * @since 3.3
87       */
88      public Line(final Vector2D p1, final Vector2D p2, final double tolerance) {
89          reset(p1, p2);
90          this.tolerance = tolerance;
91      }
92  
93      /** Build a line from a point and an angle.
94       * @param p point belonging to the line
95       * @param angle angle of the line with respect to abscissa axis
96       * @param tolerance tolerance below which points are considered identical
97       * @since 3.3
98       */
99      public Line(final Vector2D p, final double angle, final double tolerance) {
100         reset(p, angle);
101         this.tolerance = tolerance;
102     }

104     /** Build a line from its internal characteristics.
105      * @param angle angle of the line with respect to abscissa axis
106      * @param cos cosine of the angle
107      * @param sin sine of the angle
108      * @param originOffset offset of the origin
109      * @param tolerance tolerance below which points are considered identical
110      * @since 3.3
111      */
112     private Line(final double angle, final double cos, final double sin,
113                  final double originOffset, final double tolerance) {
114         this.angle        = angle;
115         this.cos          = cos;
116         this.sin          = sin;
117         this.originOffset = originOffset;
118         this.tolerance    = tolerance;
119         this.reverse      = null;
120     }
121 
122     /** Copy constructor.
123      * <p>The created instance is completely independent from the
124      * original instance, it is a deep copy.</p>
125      * @param line line to copy
126      */
127     public Line(final Line line) {
128         angle        = MathUtils.normalizeAngle(line.angle, FastMath.PI);
129         cos          = line.cos;
130         sin          = line.sin;
131         originOffset = line.originOffset;
132         tolerance    = line.tolerance;
133         reverse      = null;
134     }

136     /** {@inheritDoc} */
137     @Override
138     public Line copySelf() {
139         return new Line(this);
140     }

142     /** Reset the instance as if built from two points.
143      * <p>The line is oriented from p1 to p2</p>
144      * @param p1 first point
145      * @param p2 second point
146      */
147     public void reset(final Vector2D p1, final Vector2D p2) {
148         unlinkReverse();
149         final double dx = p2.getX() - p1.getX();
150         final double dy = p2.getY() - p1.getY();
151         final double d = FastMath.hypot(dx, dy);
152         if (d == 0.0) {
153             angle        = 0.0;
154             cos          = 1.0;
155             sin          = 0.0;
156             originOffset = p1.getY();
157         } else {
158             angle        = FastMath.PI + FastMath.atan2(-dy, -dx);
159             cos          = dx / d;
160             sin          = dy / d;
161             originOffset = MathArrays.linearCombination(p2.getX(), p1.getY(), -p1.getX(), p2.getY()) / d;
162         }
163     }
164 
165     /** Reset the instance as if built from a line and an angle.
166      * @param p point belonging to the line
167      * @param alpha angle of the line with respect to abscissa axis
168      */
169     public void reset(final Vector2D p, final double alpha) {
170         unlinkReverse();
171         this.angle   = MathUtils.normalizeAngle(alpha, FastMath.PI);
172         cos          = FastMath.cos(this.angle);
173         sin          = FastMath.sin(this.angle);
174         originOffset = MathArrays.linearCombination(cos, p.getY(), -sin, p.getX());
175     }
176 
177     /** Revert the instance.
178      */
179     public void revertSelf() {
180         unlinkReverse();
181         if (angle < FastMath.PI) {
182             angle += FastMath.PI;
183         } else {
184             angle -= FastMath.PI;
185         }
186         cos          = -cos;
187         sin          = -sin;
188         originOffset = -originOffset;
189     }
190 
191     /** Unset the link between an instance and its reverse.
192      */
193     private void unlinkReverse() {
194         if (reverse != null) {
195             reverse.reverse = null;
196         }
197         reverse = null;
198     }
199 
200     /** Get the reverse of the instance.
201      * <p>Get a line with reversed orientation with respect to the
202      * instance.</p>
203      * <p>
204      * As long as neither the instance nor its reverse are modified
205      * (i.e. as long as none of the {@link #reset(Vector2D, Vector2D)},
206      * {@link #reset(Vector2D, double)}, {@link #revertSelf()},
207      * {@link #setAngle(double)} or {@link #setOriginOffset(double)}
208      * methods are called), then the line and its reverse remain linked
209      * together so that {@code line.getReverse().getReverse() == line}.
210      * When one of the line is modified, the link is deleted as both
211      * instance becomes independent.
212      * </p>
213      * @return a new line, with orientation opposite to the instance orientation
214      */
215     public Line getReverse() {
216         if (reverse == null) {
217             reverse = new Line((angle < FastMath.PI) ? (angle + FastMath.PI) : (angle - FastMath.PI),
218                                -cos, -sin, -originOffset, tolerance);
219             reverse.reverse = this;
220         }
221         return reverse;
222     }

225      * @param vector n-dimension point of the space
226      * @return (n-1)-dimension point of the sub-space corresponding to
227      * the specified space point

231     }

233     /** Transform a sub-space point into a space point.
234      * @param vector (n-1)-dimension point of the sub-space
235      * @return n-dimension point of the space corresponding to the
236      * specified sub-space point
237      */
238     public Vector2D toSpace(Vector<Euclidean1D> vector) {
239         return toSpace((Point<Euclidean1D>) vector);
240     }

242     /** {@inheritDoc} */
243     @Override
244     public Vector1D toSubSpace(final Point<Euclidean2D> point) {
245         Vector2D p2 = (Vector2D) point;
246         return new Vector1D(MathArrays.linearCombination(cos, p2.getX(), sin, p2.getY()));
247     }
248 
249     /** {@inheritDoc} */
250     @Override
251     public Vector2D toSpace(final Point<Euclidean1D> point) {
252         final double abscissa = ((Vector1D) point).getX();
253         return new Vector2D(MathArrays.linearCombination(abscissa, cos, -originOffset, sin),
254                             MathArrays.linearCombination(abscissa, sin,  originOffset, cos));
255     }
256 
257     /** Get the intersection point of the instance and another line.
258      * @param other other line
259      * @return intersection point of the instance and the other line
260      * or null if there are no intersection points
261      */
262     public Vector2D intersection(final Line other) {
263         final double d = MathArrays.linearCombination(sin, other.cos, -other.sin, cos);
264         if (FastMath.abs(d) < tolerance) {
265             return null;
266         }
267         return new Vector2D(MathArrays.linearCombination(cos, other.originOffset, -other.cos, originOffset) / d,
268                             MathArrays.linearCombination(sin, other.originOffset, -other.sin, originOffset) / d);
269     }

271     /** {@inheritDoc}
272      * @since 3.3
273      */
274     @Override
275     public Point<Euclidean2D> project(Point<Euclidean2D> point) {
276         return toSpace(toSubSpace(point));
277     }

302     /** Get the offset (oriented distance) of a parallel line.
303      * <p>This method should be called only for parallel lines otherwise
304      * the result is not meaningful.</p>
305      * <p>The offset is 0 if both lines are the same, it is
306      * positive if the line is on the right side of the instance and
307      * negative if it is on the left side, according to its natural
308      * orientation.</p>
309      * @param line line to check
310      * @return offset of the line
311      */
312     public double getOffset(final Line line) {
313         return originOffset +
314                (MathArrays.linearCombination(cos, line.cos, sin, line.sin) > 0 ? -line.originOffset : line.originOffset);
315     }
316 
317     /** Get the offset (oriented distance) of a vector.
318      * @param vector vector to check
319      * @return offset of the vector
320      */
321     public double getOffset(Vector<Euclidean2D> vector) {
322         return getOffset((Point<Euclidean2D>) vector);
323     }

325     /** {@inheritDoc} */
326     @Override
327     public double getOffset(final Point<Euclidean2D> point) {
328         Vector2D p2 = (Vector2D) point;
329         return MathArrays.linearCombination(sin, p2.getX(), -cos, p2.getY(), 1.0, originOffset);
330     }

332     /** {@inheritDoc} */
333     @Override
334     public boolean sameOrientationAs(final Hyperplane<Euclidean2D> other) {
335         final Line otherL = (Line) other;
336         return MathArrays.linearCombination(sin, otherL.sin, cos, otherL.cos) >= 0.0;
337     }

339     /** Get one point from the plane.
340      * @param abscissa desired abscissa for the point
341      * @param offset desired offset for the point
342      * @return one point in the plane, with given abscissa and offset
343      * relative to the line
344      */
345     public Vector2D getPointAt(final Vector1D abscissa, final double offset) {
346         final double x       = abscissa.getX();
347         final double dOffset = offset - originOffset;
348         return new Vector2D(MathArrays.linearCombination(x, cos,  dOffset, sin),
349                             MathArrays.linearCombination(x, sin, -dOffset, cos));
350     }
351 
352     /** Check if the line contains a point.
353      * @param p point to check
354      * @return true if p belongs to the line
355      */
356     public boolean contains(final Vector2D p) {
357         return FastMath.abs(getOffset(p)) < tolerance;
358     }

360     /** Compute the distance between the instance and a point.
361      * <p>This is a shortcut for invoking FastMath.abs(getOffset(p)),
362      * and provides consistency with what is in the
363      * org.apache.commons.math4.geometry.euclidean.threed.Line class.</p>
364      *
365      * @param p to check
366      * @return distance between the instance and the point
367      * @since 3.1
368      */
369     public double distance(final Vector2D p) {
370         return FastMath.abs(getOffset(p));
371     }
372 
373     /** Check the instance is parallel to another line.
374      * @param line other line to check
375      * @return true if the instance is parallel to the other line
376      * (they can have either the same or opposite orientations)
377      */
378     public boolean isParallelTo(final Line line) {
379         return FastMath.abs(MathArrays.linearCombination(sin, line.cos, -cos, line.sin)) < tolerance;
380     }

382     /** Translate the line to force it passing by a point.
383      * @param p point by which the line should pass
384      */
385     public void translateToPoint(final Vector2D p) {
386         originOffset = MathArrays.linearCombination(cos, p.getY(), -sin, p.getX());
387     }

421     /** Get a {@link org.apache.commons.math4.geometry.partitioning.Transform
422      * Transform} embedding an affine transform.
423      * @param cXX transform factor between input abscissa and output abscissa
424      * @param cYX transform factor between input abscissa and output ordinate
425      * @param cXY transform factor between input ordinate and output abscissa
426      * @param cYY transform factor between input ordinate and output ordinate
427      * @param cX1 transform addendum for output abscissa
428      * @param cY1 transform addendum for output ordinate
429      * @return a new transform that can be applied to either {@link
430      * Vector2D Vector2D}, {@link Line Line} or {@link
431      * org.apache.commons.math4.geometry.partitioning.SubHyperplane
432      * SubHyperplane} instances
433      * @exception MathIllegalArgumentException if the transform is non invertible
434      * @since 4.0
435      */
436     public static Transform<Euclidean2D, Euclidean1D> getTransform(final double cXX,
437                                                                    final double cYX,
438                                                                    final double cXY,
439                                                                    final double cYY,
440                                                                    final double cX1,
441                                                                    final double cY1)
442         throws MathIllegalArgumentException {
443         return new LineTransform(cXX, cYX, cXY, cYY, cX1, cY1);
444     }

446     /** Class embedding an affine transform.
447      * <p>This class is used in order to apply an affine transform to a
448      * line. Using a specific object allow to perform some computations
449      * on the transform only once even if the same transform is to be
450      * applied to a large number of lines (for example to a large
451      * polygon)./<p>
452      */
453     private static class LineTransform implements Transform<Euclidean2D, Euclidean1D> {
454 
455         /** Transform factor between input abscissa and output abscissa. */
456         private final double cXX;
457 
458         /** Transform factor between input abscissa and output ordinate. */
459         private final double cYX;
460 
461         /** Transform factor between input ordinate and output abscissa. */
462         private final double cXY;
463 
464         /** Transform factor between input ordinate and output ordinate. */
465         private final double cYY;
466 
467         /** Transform addendum for output abscissa. */
468         private final double cX1;
469 
470         /** Transform addendum for output ordinate. */
471         private final double cY1;
472 
473         /** cXY * cY1 - cYY * cX1. */
474         private final double c1Y;
475 
476         /** cXX * cY1 - cYX * cX1. */
477         private final double c1X;
478 
479         /** cXX * cYY - cYX * cXY. */
480         private final double c11;
481 
482         /** Build an affine line transform from a n {@code AffineTransform}.
483          * @param cXX transform factor between input abscissa and output abscissa
484          * @param cYX transform factor between input abscissa and output ordinate
485          * @param cXY transform factor between input ordinate and output abscissa
486          * @param cYY transform factor between input ordinate and output ordinate
487          * @param cX1 transform addendum for output abscissa
488          * @param cY1 transform addendum for output ordinate
489          * @exception MathIllegalArgumentException if the transform is non invertible
490          * @since 4.0
491          */
492         LineTransform(final double cXX, final double cYX, final double cXY,
493                       final double cYY, final double cX1, final double cY1)
494             throws MathIllegalArgumentException {
495 
496             this.cXX = cXX;
497             this.cYX = cYX;
498             this.cXY = cXY;
499             this.cYY = cYY;
500             this.cX1 = cX1;
501             this.cY1 = cY1;
502 
503             c1Y = MathArrays.linearCombination(cXY, cY1, -cYY, cX1);
504             c1X = MathArrays.linearCombination(cXX, cY1, -cYX, cX1);
505             c11 = MathArrays.linearCombination(cXX, cYY, -cYX, cXY);
506 
507             if (FastMath.abs(c11) < 1.0e-20) {
508                 throw new MathIllegalArgumentException(LocalizedFormats.NON_INVERTIBLE_TRANSFORM);
509             }
510 
511         }
512 
513         /** {@inheritDoc} */
514         @Override
515         public Vector2D apply(final Point<Euclidean2D> point) {
516             final Vector2D p2D = (Vector2D) point;
517             final double  x   = p2D.getX();
518             final double  y   = p2D.getY();
519             return new Vector2D(MathArrays.linearCombination(cXX, x, cXY, y, cX1, 1),
520                                 MathArrays.linearCombination(cYX, x, cYY, y, cY1, 1));
521         }
522 
523         /** {@inheritDoc} */
524         @Override
525         public Line apply(final Hyperplane<Euclidean2D> hyperplane) {
526             final Line   line    = (Line) hyperplane;
527             final double rOffset = MathArrays.linearCombination(c1X, line.cos, c1Y, line.sin, c11, line.originOffset);
528             final double rCos    = MathArrays.linearCombination(cXX, line.cos, cXY, line.sin);
529             final double rSin    = MathArrays.linearCombination(cYX, line.cos, cYY, line.sin);
530             final double inv     = 1.0 / FastMath.sqrt(rSin * rSin + rCos * rCos);
531             return new Line(FastMath.PI + FastMath.atan2(-rSin, -rCos),
532                             inv * rCos, inv * rSin,
533                             inv * rOffset, line.tolerance);
534         }
535 
536         /** {@inheritDoc} */
537         @Override
538         public SubHyperplane<Euclidean1D> apply(final SubHyperplane<Euclidean1D> sub,
539                                                 final Hyperplane<Euclidean2D> original,
540                                                 final Hyperplane<Euclidean2D> transformed) {
541             final OrientedPoint op     = (OrientedPoint) sub.getHyperplane();
542             final Line originalLine    = (Line) original;
543             final Line transformedLine = (Line) transformed;
544             final Vector1D newLoc =
545                 transformedLine.toSubSpace(apply(originalLine.toSpace(op.getLocation())));
546             return new OrientedPoint(newLoc, op.isDirect(), originalLine.tolerance).wholeHyperplane();
547         }
548 
549     }
