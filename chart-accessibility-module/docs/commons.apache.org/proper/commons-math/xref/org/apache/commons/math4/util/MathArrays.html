324     /**
325      * Check that an array is monotonically increasing or decreasing.
326      *
327      * @param <T> the type of the elements in the specified array
328      * @param val Values.
329      * @param dir Ordering direction.
330      * @param strict Whether the order should be strict.
331      * @return {@code true} if sorted, {@code false} otherwise.
332      */
333     public static  <T extends Comparable<? super T>> boolean isMonotonic(T[] val,
334                                       OrderDirection dir,
335                                       boolean strict) {
336         T previous = val[0];
337         final int max = val.length;
338         for (int i = 1; i < max; i++) {
339             final int comp;
340             switch (dir) {
341             case INCREASING:
342                 comp = previous.compareTo(val[i]);
343                 if (strict) {
344                     if (comp >= 0) {
345                         return false;
346                     }
347                 } else {
348                     if (comp > 0) {
349                         return false;
350                     }
351                 }
352                 break;
353             case DECREASING:
354                 comp = val[i].compareTo(previous);
355                 if (strict) {
356                     if (comp >= 0) {
357                         return false;
358                     }
359                 } else {
360                     if (comp > 0) {
361                        return false;
362                     }
363                 }
364                 break;
365             default:
366                 // Should never happen.
367                 throw new MathInternalError();
368             }
369 
370             previous = val[i];
371         }
372         return true;
373     }

462     /**
463      * Check that the given array is sorted.
464      *
465      * @param val Values.
466      * @param dir Ordering direction.
467      * @param strict Whether the order should be strict.
468      * @param abort Whether to throw an exception if the check fails.
469      * @return {@code true} if the array is sorted.
470      * @throws NonMonotonicSequenceException if the array is not sorted
471      * and {@code abort} is {@code true}.
472      */
473     public static boolean checkOrder(double[] val, OrderDirection dir,
474                                      boolean strict, boolean abort)
475         throws NonMonotonicSequenceException {
476         double previous = val[0];
477         final int max = val.length;
478 
479         int index;
480         ITEM:
481         for (index = 1; index < max; index++) {
482             switch (dir) {
483             case INCREASING:
484                 if (strict) {
485                     if (val[index] <= previous) {
486                         break ITEM;
487                     }
488                 } else {
489                     if (val[index] < previous) {
490                         break ITEM;
491                     }
492                 }
493                 break;
494             case DECREASING:
495                 if (strict) {
496                     if (val[index] >= previous) {
497                         break ITEM;
498                     }
499                 } else {
500                     if (val[index] > previous) {
501                         break ITEM;
502                     }
503                 }
504                 break;
505             default:
506                 // Should never happen.
507                 throw new MathInternalError();
508             }
509 
510             previous = val[index];
511         }
512 
513         if (index == max) {
514             // Loop completed.
515             return true;
516         }
517 
518         // Loop early exit means wrong ordering.
519         if (abort) {
520             throw new NonMonotonicSequenceException(val[index], previous, index, dir, strict);
521         } else {
522             return false;
523         }
524     }

708         for (int i = 0; i < v.length; i++) {
709             double xabs = FastMath.abs(v[i]);
710             if (xabs < rdwarf || xabs > agiant) {
711                 if (xabs > rdwarf) {
712                     if (xabs > x1max) {
713                         double r = x1max / xabs;
714                         s1= 1 + s1 * r * r;
715                         x1max = xabs;
716                     } else {
717                         double r = xabs / x1max;
718                         s1 += r * r;
719                     }
720                 } else {
721                     if (xabs > x3max) {
722                         double r = x3max / xabs;
723                         s3= 1 + s3 * r * r;
724                         x3max = xabs;
725                     } else {
726                         if (xabs != 0) {
727                             double r = xabs / x3max;
728                             s3 += r * r;
729                         }
730                     }
731                 }
732             } else {
733                 s2 += xabs * xabs;
734             }
735         }
736         double norm;
737         if (s1 != 0) {
738             norm = x1max * Math.sqrt(s1 + (s2 / x1max) / x1max);
739         } else {
740             if (s2 == 0) {
741                 norm = x3max * Math.sqrt(s3);
742             } else {
743                 if (s2 >= x3max) {
744                     norm = Math.sqrt(s2 * (1 + (x3max / s2) * (x3max * s3)));
745                 } else {
746                     norm = Math.sqrt(x3max * ((s2 / x3max) + (x3max * s3)));
747                 }
748             }
749         }
750         return norm;
751     }

807      * allows the order of the sort to be provided in the {@code dir}

812      * @param dir Order direction.

815      * @throws DimensionMismatchException if any {@code y} is not the same
816      * size as {@code x}.
817      * @throws NullArgumentException if {@code x} or any {@code y} is null
818      * @since 3.0
819      */
820     public static void sortInPlace(double[] x,
821                                    final OrderDirection dir,
822                                    double[] ... yList)
823         throws NullArgumentException,
824                DimensionMismatchException {
825 
826         // Consistency checks.
827         if (x == null) {
828             throw new NullArgumentException();
829         }
830 
831         final int yListLen = yList.length;
832         final int len = x.length;
833 
834         for (int j = 0; j < yListLen; j++) {
835             final double[] y = yList[j];
836             if (y == null) {
837                 throw new NullArgumentException();
838             }
839             if (y.length != len) {
840                 throw new DimensionMismatchException(y.length, len);
841             }
842         }
843 
844         // Associate each abscissa "x[i]" with its index "i".
845         final List<PairDoubleInteger> list
846             = new ArrayList<PairDoubleInteger>(len);
847         for (int i = 0; i < len; i++) {
848             list.add(new PairDoubleInteger(x[i], i));
849         }
850 
851         // Create comparators for increasing and decreasing orders.
852         final Comparator<PairDoubleInteger> comp
853             = dir == MathArrays.OrderDirection.INCREASING ?
854             new Comparator<PairDoubleInteger>() {
855             /** {@inheritDoc} */
856             @Override
857             public int compare(PairDoubleInteger o1,
858                                PairDoubleInteger o2) {
859                 return Double.compare(o1.getKey(), o2.getKey());
860             }
861         } : new Comparator<PairDoubleInteger>() {
862             /** {@inheritDoc} */
863             @Override
864             public int compare(PairDoubleInteger o1,
865                                PairDoubleInteger o2) {
866                 return Double.compare(o2.getKey(), o1.getKey());
867             }
868         };
869 
870         // Sort.
871         Collections.sort(list, comp);
872 
873         // Modify the original array so that its elements are in
874         // the prescribed order.
875         // Retrieve indices of original locations.
876         final int[] indices = new int[len];
877         for (int i = 0; i < len; i++) {
878             final PairDoubleInteger e = list.get(i);
879             x[i] = e.getKey();
880             indices[i] = e.getValue();
881         }
882 
883         // In each of the associated arrays, move the
884         // elements to their new location.
885         for (int j = 0; j < yListLen; j++) {
886             // Input array will be modified in place.
887             final double[] yInPlace = yList[j];
888             final double[] yOrig = yInPlace.clone();
889 
890             for (int i = 0; i < len; i++) {
891                 yInPlace[i] = yOrig[indices[i]];
892             }
893         }
894     }

961     /**
962      * Compute a linear combination accurately.
963      * This method computes the sum of the products
964      * <code>a<sub>i</sub> b<sub>i</sub></code> to high accuracy.
965      * It does so by using specific multiplication and addition algorithms to
966      * preserve accuracy and reduce cancellation effects.
967      * <br/>
968      * It is based on the 2005 paper
969      * <a href="http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.2.1547">
970      * Accurate Sum and Dot Product</a> by Takeshi Ogita, Siegfried M. Rump,
971      * and Shin'ichi Oishi published in SIAM J. Sci. Comput.
972      *
973      * @param a Factors.
974      * @param b Factors.
975      * @return <code>&Sigma;<sub>i</sub> a<sub>i</sub> b<sub>i</sub></code>.
976      * @throws DimensionMismatchException if arrays dimensions don't match
977      */
978     public static double linearCombination(final double[] a, final double[] b)
979         throws DimensionMismatchException {
980         checkEqualLength(a, b);
981         final int len = a.length;
982 
983         if (len == 1) {
984             // Revert to scalar multiplication.
985             return a[0] * b[0];
986         }
987 
988         final double[] prodHigh = new double[len];
989         double prodLowSum = 0;
990 
991         for (int i = 0; i < len; i++) {
992             final double ai    = a[i];
993             final double aHigh = Double.longBitsToDouble(Double.doubleToRawLongBits(ai) & ((-1L) << 27));
994             final double aLow  = ai - aHigh;
995 
996             final double bi    = b[i];
997             final double bHigh = Double.longBitsToDouble(Double.doubleToRawLongBits(bi) & ((-1L) << 27));
998             final double bLow  = bi - bHigh;
999             prodHigh[i] = ai * bi;
1000             final double prodLow = aLow * bLow - (((prodHigh[i] -
1001                                                     aHigh * bHigh) -
1002                                                    aLow * bHigh) -
1003                                                   aHigh * bLow);
1004             prodLowSum += prodLow;
1005         }
1006 
1007 
1008         final double prodHighCur = prodHigh[0];
1009         double prodHighNext = prodHigh[1];
1010         double sHighPrev = prodHighCur + prodHighNext;
1011         double sPrime = sHighPrev - prodHighNext;
1012         double sLowSum = (prodHighNext - (sHighPrev - sPrime)) + (prodHighCur - sPrime);
1013 
1014         final int lenMinusOne = len - 1;
1015         for (int i = 1; i < lenMinusOne; i++) {
1016             prodHighNext = prodHigh[i + 1];
1017             final double sHighCur = sHighPrev + prodHighNext;
1018             sPrime = sHighCur - prodHighNext;
1019             sLowSum += (prodHighNext - (sHighCur - sPrime)) + (sHighPrev - sPrime);
1020             sHighPrev = sHighCur;
1021         }
1022 
1023         double result = sHighPrev + (prodLowSum + sLowSum);
1024 
1025         if (Double.isNaN(result)) {
1026             // either we have split infinite numbers or some coefficients were NaNs,
1027             // just rely on the naive implementation and let IEEE754 handle this
1028             result = 0;
1029             for (int i = 0; i < len; ++i) {
1030                 result += a[i] * b[i];
1031             }
1032         }
1033 
1034         return result;
1035     }

1248         final double prod1High  = a1 * b1;
1249         final double prod1Low   = a1Low * b1Low - (((prod1High - a1High * b1High) - a1Low * b1High) - a1High * b1Low);
1250 
1251         // split a2 and b2 as one 26 bits number and one 27 bits number
1252         final double a2High     = Double.longBitsToDouble(Double.doubleToRawLongBits(a2) & ((-1L) << 27));
1253         final double a2Low      = a2 - a2High;
1254         final double b2High     = Double.longBitsToDouble(Double.doubleToRawLongBits(b2) & ((-1L) << 27));
1255         final double b2Low      = b2 - b2High;
1256 
1257         // accurate multiplication a2 * b2
1258         final double prod2High  = a2 * b2;
1259         final double prod2Low   = a2Low * b2Low - (((prod2High - a2High * b2High) - a2Low * b2High) - a2High * b2Low);
1260 
1261         // split a3 and b3 as one 26 bits number and one 27 bits number
1262         final double a3High     = Double.longBitsToDouble(Double.doubleToRawLongBits(a3) & ((-1L) << 27));
1263         final double a3Low      = a3 - a3High;
1264         final double b3High     = Double.longBitsToDouble(Double.doubleToRawLongBits(b3) & ((-1L) << 27));
1265         final double b3Low      = b3 - b3High;
1266 
1267         // accurate multiplication a3 * b3
1268         final double prod3High  = a3 * b3;
1269         final double prod3Low   = a3Low * b3Low - (((prod3High - a3High * b3High) - a3Low * b3High) - a3High * b3Low);
1270 
1271         // split a4 and b4 as one 26 bits number and one 27 bits number
1272         final double a4High     = Double.longBitsToDouble(Double.doubleToRawLongBits(a4) & ((-1L) << 27));
1273         final double a4Low      = a4 - a4High;
1274         final double b4High     = Double.longBitsToDouble(Double.doubleToRawLongBits(b4) & ((-1L) << 27));
1275         final double b4Low      = b4 - b4High;
1276 
1277         // accurate multiplication a4 * b4
1278         final double prod4High  = a4 * b4;
1279         final double prod4Low   = a4Low * b4Low - (((prod4High - a4High * b4High) - a4Low * b4High) - a4High * b4Low);
1280 
1281         // accurate addition a1 * b1 + a2 * b2
1282         final double s12High    = prod1High + prod2High;
1283         final double s12Prime   = s12High - prod2High;
1284         final double s12Low     = (prod2High - (s12High - s12Prime)) + (prod1High - s12Prime);
1285 
1286         // accurate addition a1 * b1 + a2 * b2 + a3 * b3
1287         final double s123High   = s12High + prod3High;
1288         final double s123Prime  = s123High - prod3High;
1289         final double s123Low    = (prod3High - (s123High - s123Prime)) + (s12High - s123Prime);
1290 
1291         // accurate addition a1 * b1 + a2 * b2 + a3 * b3 + a4 * b4
1292         final double s1234High  = s123High + prod4High;
1293         final double s1234Prime = s1234High - prod4High;
1294         final double s1234Low   = (prod4High - (s1234High - s1234Prime)) + (s123High - s1234Prime);
1295 
1296         // final rounding, s1234 may have suffered many cancellations, we try
1297         // to recover some bits from the extra words we have saved up to now
1298         double result = s1234High + (prod1Low + prod2Low + prod3Low + prod4Low + s12Low + s123Low + s1234Low);
1299 
1300         if (Double.isNaN(result)) {
1301             // either we have split infinite numbers or some coefficients were NaNs,
1302             // just rely on the naive implementation and let IEEE754 handle this
1303             result = a1 * b1 + a2 * b2 + a3 * b3 + a4 * b4;
1304         }
1305 
1306         return result;
1307     }

1514      * Calculates the <a href="http://en.wikipedia.org/wiki/Convolution">

1516      * <p>
1517      * The solution is obtained via straightforward computation of the
1518      * convolution sum (and not via FFT). Whenever the computation needs
1519      * an element that would be located at an index outside the input arrays,
1520      * the value is assumed to be zero.
1521      *
1522      * @param x First sequence.
1523      * Typically, this sequence will represent an input signal to a system.
1524      * @param h Second sequence.
1525      * Typically, this sequence will represent the impulse response of the system.
1526      * @return the convolution of {@code x} and {@code h}.
1527      * This array's length will be {@code x.length + h.length - 1}.
1528      * @throws NullArgumentException if either {@code x} or {@code h} is {@code null}.
1529      * @throws NoDataException if either {@code x} or {@code h} is empty.
1530      *
1531      * @since 3.3
1532      */
1533     public static double[] convolve(double[] x, double[] h)
1534         throws NullArgumentException,
1535                NoDataException {
1536         MathUtils.checkNotNull(x);
1537         MathUtils.checkNotNull(h);
1538 
1539         final int xLen = x.length;
1540         final int hLen = h.length;
1541 
1542         if (xLen == 0 || hLen == 0) {
1543             throw new NoDataException();
1544         }
1545 
1546         // initialize the output array
1547         final int totalLength = xLen + hLen - 1;
1548         final double[] y = new double[totalLength];
1549 
1550         // straightforward implementation of the convolution sum
1551         for (int n = 0; n < totalLength; n++) {
1552             double yn = 0;
1553             int k = FastMath.max(0, n + 1 - xLen);
1554             int j = n - k;
1555             while (k < hLen && j >= 0) {
1556                 yn += x[j--] * h[k++];
1557             }
1558             y[n] = yn;
1559         }
1560 
1561         return y;
1562     }

1618                 } else {
1619                     target = rng.nextInt(i - start + 1) + start;
1620                 }
1621                 final int temp = list[target];
1622                 list[target] = list[i];
1623                 list[i] = temp;
1624             }
1625         }
1626             break;
1627         case HEAD: {
1628             for (int i = 0; i <= start; i++) {
1629                 final int target;
1630                 if (i == start) {
1631                     target = start;
1632                 } else {
1633                     target = rng.nextInt(start - i + 1) + i;
1634                 }
1635                 final int temp = list[target];
1636                 list[target] = list[i];
1637                 list[i] = temp;
1638             }
1639         }

1832      * </ul></p>
1833      *
1834      * @param values the input array.
1835      * @param weights the weights array.
1836      * @param begin index of the first array element to include.
1837      * @param length the number of elements to include.
1838      * @param allowEmpty if {@code true} than allow zero length arrays to pass.
1839      * @return {@code true} if the parameters are valid.
1840      * @throws NullArgumentException if either of the arrays are null
1841      * @throws MathIllegalArgumentException if the array indices are not valid,
1842      * the weights array contains NaN, infinite or negative elements, or there
1843      * are no positive weights.
1844      * @since 3.3
1845      */
1846     public static boolean verifyValues(final double[] values, final double[] weights,
1847             final int begin, final int length, final boolean allowEmpty) throws MathIllegalArgumentException {
1848 
1849         if (weights == null || values == null) {
1850             throw new NullArgumentException(LocalizedFormats.INPUT_ARRAY);
1851         }
1852 
1853         checkEqualLength(weights, values);
1854 
1855         boolean containsPositiveWeight = false;
1856         for (int i = begin; i < begin + length; i++) {
1857             final double weight = weights[i];
1858             if (Double.isNaN(weight)) {
1859                 throw new MathIllegalArgumentException(LocalizedFormats.NAN_ELEMENT_AT_INDEX, Integer.valueOf(i));
1860             }
1861             if (Double.isInfinite(weight)) {
1862                 throw new MathIllegalArgumentException(LocalizedFormats.INFINITE_ARRAY_ELEMENT, Double.valueOf(weight), Integer.valueOf(i));
1863             }
1864             if (weight < 0) {
1865                 throw new MathIllegalArgumentException(LocalizedFormats.NEGATIVE_ELEMENT_AT_INDEX, Integer.valueOf(i), Double.valueOf(weight));
1866             }
1867             if (!containsPositiveWeight && weight > 0.0) {
1868                 containsPositiveWeight = true;
1869             }
1870         }
1871 
1872         if (!containsPositiveWeight) {
1873             throw new MathIllegalArgumentException(LocalizedFormats.WEIGHT_AT_LEAST_ONE_NON_ZERO);
1874         }
1875 
1876         return verifyValues(values, begin, length, allowEmpty);
1877     }

1879     /**
1880      * Concatenates a sequence of arrays. The return array consists of the
1881      * entries of the input arrays concatenated in the order they appear in
1882      * the argument list.  Null arrays cause NullPointerExceptions; zero
1883      * length arrays are allowed (contributing nothing to the output array).
1884      *
1885      * @param x list of double[] arrays to concatenate
1886      * @return a new array consisting of the entries of the argument arrays
1887      * @throws NullPointerException if any of the arrays are null
1888      * @since 3.6
1889      */
1890     public static double[] concatenate(double[] ...x) {
1891         int combinedLength = 0;
1892         for (double[] a : x) {
1893             combinedLength += a.length;
1894         }
1895         int offset = 0;
1896         int curLength = 0;
1897         final double[] combined = new double[combinedLength];
1898         for (int i = 0; i < x.length; i++) {
1899             curLength = x[i].length;
1900             System.arraycopy(x[i], 0, combined, offset, curLength);
1901             offset += curLength;
1902         }
1903         return combined;
1904     }

1906     /**
1907      * Returns an array consisting of the unique values in {@code data}.
1908      * The return array is sorted in descending order.  Empty arrays
1909      * are allowed, but null arrays result in NullPointerException.
1910      * Infinities are allowed.  NaN values are allowed with maximum
1911      * sort order - i.e., if there are NaN values in {@code data},
1912      * {@code Double.NaN} will be the first element of the output array,
1913      * even if the array also contains {@code Double.POSITIVE_INFINITY}.
1914      *
1915      * @param data array to scan
1916      * @return descending list of values included in the input array
1917      * @throws NullPointerException if data is null
1918      * @since 3.6
1919      */
1920     public static double[] unique(double[] data) {
1921         TreeSet<Double> values = new TreeSet<Double>();
1922         for (int i = 0; i < data.length; i++) {
1923             values.add(data[i]);
1924         }
1925         final int count = values.size();
1926         final double[] out = new double[count];
1927         Iterator<Double> iterator = values.descendingIterator();
1928         int i = 0;
1929         while (iterator.hasNext()) {
1930             out[i++] = iterator.next();
1931         }
1932         return out;
1933     }
