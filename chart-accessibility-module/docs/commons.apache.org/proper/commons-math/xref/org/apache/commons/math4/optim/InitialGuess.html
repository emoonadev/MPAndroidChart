18  package org.apache.commons.math4.optim;
19  
20  /**
21   * Starting point (first guess) of the optimization procedure.
22   * <br/>
23   * Immutable class.
24   *
25   * @since 3.1
26   */
27  public class InitialGuess implements OptimizationData {
28      /** Initial guess. */
29      private final double[] init;
30  
31      /**
32       * @param startPoint Initial guess.
33       */
34      public InitialGuess(double[] startPoint) {
35          init = startPoint.clone();
36      }
37  
38      /**
39       * Gets the initial guess.
40       *
41       * @return the initial guess.
42       */
43      public double[] getInitialGuess() {
44          return init.clone();
45      }
46  }
