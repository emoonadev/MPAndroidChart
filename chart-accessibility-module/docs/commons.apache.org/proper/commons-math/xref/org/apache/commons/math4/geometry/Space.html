17  package org.apache.commons.math4.geometry;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
22  
23  /** This interface represents a generic space, with affine and vectorial counterparts.
24   * @see Vector
25   * @since 3.0
26   */
27  public interface Space extends Serializable {
28  
29      /** Get the dimension of the space.
30       * @return dimension of the space
31       */
32      int getDimension();
33  
34      /** Get the n-1 dimension subspace of this space.
35       * @return n-1 dimension sub-space of this space
36       * @see #getDimension()
37       * @exception MathUnsupportedOperationException for dimension-1 spaces
38       * which do not have sub-spaces
39       */
40      Space getSubSpace() throws MathUnsupportedOperationException;
41  
42  }
