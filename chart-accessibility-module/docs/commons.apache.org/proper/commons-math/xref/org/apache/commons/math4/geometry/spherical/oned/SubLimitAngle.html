17  package org.apache.commons.math4.geometry.spherical.oned;
18  
19  import org.apache.commons.math4.geometry.partitioning.AbstractSubHyperplane;
20  import org.apache.commons.math4.geometry.partitioning.Hyperplane;
21  import org.apache.commons.math4.geometry.partitioning.Region;
22  
23  /** This class represents sub-hyperplane for {@link LimitAngle}.
24   * <p>Instances of this class are guaranteed to be immutable.</p>
25   * @since 3.3
26   */
27  public class SubLimitAngle extends AbstractSubHyperplane<Sphere1D, Sphere1D> {
28  
29      /** Simple constructor.
30       * @param hyperplane underlying hyperplane
31       * @param remainingRegion remaining region of the hyperplane
32       */
33      public SubLimitAngle(final Hyperplane<Sphere1D> hyperplane,
34                           final Region<Sphere1D> remainingRegion) {
35          super(hyperplane, remainingRegion);
36      }
37  
38      /** {@inheritDoc} */
39      @Override
40      public double getSize() {
41          return 0;
42      }
43  
44      /** {@inheritDoc} */
45      @Override
46      public boolean isEmpty() {
47          return false;
48      }
49  
50      /** {@inheritDoc} */
51      @Override
52      protected AbstractSubHyperplane<Sphere1D, Sphere1D> buildNew(final Hyperplane<Sphere1D> hyperplane,
53                                                                   final Region<Sphere1D> remainingRegion) {
54          return new SubLimitAngle(hyperplane, remainingRegion);
55      }
56  
57      /** {@inheritDoc} */
58      @Override
59      public SplitSubHyperplane<Sphere1D> split(final Hyperplane<Sphere1D> hyperplane) {
60          final double global = hyperplane.getOffset(((LimitAngle) getHyperplane()).getLocation());
61          return (global < -1.0e-10) ?
62                                      new SplitSubHyperplane<Sphere1D>(null, this) :
63                                      new SplitSubHyperplane<Sphere1D>(this, null);
64      }
65  
66  }
