42  /**
43   * Maintains a dataset of values of a single variable and computes descriptive
44   * statistics based on stored data.
45   * <p>
46   * The {@link #getWindowSize() windowSize}
47   * property sets a limit on the number of values that can be stored in the
48   * dataset. The default value, INFINITE_WINDOW, puts no limit on the size of
49   * the dataset. This value should be used with caution, as the backing store
50   * will grow without bound in this case.  For very large datasets,
51   * {@link SummaryStatistics}, which does not store the dataset, should be used
52   * instead of this class. If <code>windowSize</code> is not INFINITE_WINDOW and
53   * more values are added than can be stored in the dataset, new values are
54   * added in a "rolling" manner, with new values replacing the "oldest" values
55   * in the dataset.
56   * <p>
57   * Note: this class is not threadsafe.  Use
58   * {@link SynchronizedDescriptiveStatistics} if concurrent access from multiple
59   * threads is required.
60   */
61  public class DescriptiveStatistics implements StatisticalSummary, Serializable {

154     /**
155      * Adds the value to the dataset. If the dataset is at the maximum size
156      * (i.e., the number of stored elements equals the currently configured
157      * windowSize), the first (oldest) element in the dataset is discarded
158      * to make room for the new value.
159      *
160      * @param v the value to be added
161      */
162     public void addValue(double v) {
163         if (windowSize != INFINITE_WINDOW) {
164             if (getN() == windowSize) {
165                 eDA.addElementRolling(v);
166             } else if (getN() < windowSize) {
167                 eDA.addElement(v);
168             }
169         } else {
170             eDA.addElement(v);
171         }
172     }

174     /**
175      * Removes the most recent value from the dataset.
176      *
177      * @throws MathIllegalStateException if there are no elements stored
178      */
179     public void removeMostRecentValue() throws MathIllegalStateException {
180         try {
181             eDA.discardMostRecentElements(1);
182         } catch (MathIllegalArgumentException ex) {
183             throw new MathIllegalStateException(LocalizedFormats.NO_DATA);
184         }

248     /**
249      * Returns the standard deviation of the available values.
250      * @return The standard deviation, Double.NaN if no values have been added
251      * or 0.0 for a single value set.
252      */
253     @Override
254     public double getStandardDeviation() {
255         double stdDev = Double.NaN;
256         if (getN() > 0) {
257             if (getN() > 1) {
258                 stdDev = FastMath.sqrt(getVariance());
259             } else {
260                 stdDev = 0.0;
261             }
262         }
263         return stdDev;
264     }

268      * <a href="http://mathworld.wolfram.com/Root-Mean-Square.html">

271      * have been added.

361     /**
362      * WindowSize controls the number of values that contribute to the
363      * reported statistics.  For example, if windowSize is set to 3 and the
364      * values {1,2,3,4,5} have been added <strong> in that order</strong> then
365      * the <i>available values</i> are {3,4,5} and all reported statistics will
366      * be based on these values. If {@code windowSize} is decreased as a result
367      * of this call and there are more than the new value of elements in the
368      * current dataset, values from the front of the array are discarded to
369      * reduce the dataset to {@code windowSize} elements.
370      *
371      * @param windowSize sets the size of the window.
372      * @throws MathIllegalArgumentException if window size is less than 1 but
373      * not equal to {@link #INFINITE_WINDOW}
374      */
375     public void setWindowSize(int windowSize) throws MathIllegalArgumentException {
376         if (windowSize < 1 && windowSize != INFINITE_WINDOW) {
377             throw new MathIllegalArgumentException(
378                     LocalizedFormats.NOT_POSITIVE_WINDOW_SIZE, windowSize);
379         }
380 
381         this.windowSize = windowSize;
382 
383         // We need to check to see if we need to discard elements
384         // from the front of the array.  If the windowSize is less than
385         // the current number of elements.
386         if (windowSize != INFINITE_WINDOW && windowSize < eDA.getNumElements()) {
387             eDA.discardFrontElements(eDA.getNumElements() - windowSize);
388         }
389     }

427     /**
428      * Returns an estimate for the pth percentile of the stored values.
429      * <p>
430      * The implementation provided here follows the first estimation procedure presented
431      * <a href="http://www.itl.nist.gov/div898/handbook/prc/section2/prc252.htm">here.</a>
432      * </p><p>
433      * <strong>Preconditions</strong>:<ul>
434      * <li><code>0 &lt; p &le; 100</code> (otherwise an
435      * <code>MathIllegalArgumentException</code> is thrown)</li>
436      * <li>at least one value must be stored (returns <code>Double.NaN
437      *     </code> otherwise)</li>
438      * </ul></p>
439      *
440      * @param p the requested percentile (scaled from 0 - 100)
441      * @return An estimate for the pth percentile of the stored data
442      * @throws MathIllegalStateException if percentile implementation has been
443      *  overridden and the supplied implementation does not support setQuantile
444      * @throws MathIllegalArgumentException if p is not a valid quantile
445      */
446     public double getPercentile(double p) throws MathIllegalStateException, MathIllegalArgumentException {
447         if (percentileImpl instanceof Percentile) {
448             ((Percentile) percentileImpl).setQuantile(p);
449         } else {
450             try {
451                 percentileImpl.getClass().getMethod(SET_QUANTILE_METHOD_NAME,
452                         new Class[] {Double.TYPE}).invoke(percentileImpl,
453                                 new Object[] {Double.valueOf(p)});
454             } catch (NoSuchMethodException e1) { // Setter guard should prevent
455                 throw new MathIllegalStateException(
456                       LocalizedFormats.PERCENTILE_IMPLEMENTATION_UNSUPPORTED_METHOD,
457                       percentileImpl.getClass().getName(), SET_QUANTILE_METHOD_NAME);
458             } catch (IllegalAccessException e2) {
459                 throw new MathIllegalStateException(
460                       LocalizedFormats.PERCENTILE_IMPLEMENTATION_CANNOT_ACCESS_METHOD,
461                       SET_QUANTILE_METHOD_NAME, percentileImpl.getClass().getName());
462             } catch (InvocationTargetException e3) {
463                 throw new IllegalStateException(e3.getCause());
464             }
465         }
466         return apply(percentileImpl);
467     }

469     /**
470      * Generates a text report displaying univariate statistics from values
471      * that have been added.  Each statistic is displayed on a separate
472      * line.
473      *
474      * @return String with line feeds displaying statistics
475      */
476     @Override
477     public String toString() {
478         StringBuilder outBuffer = new StringBuilder();
479         String endl = "\n";
480         outBuffer.append("DescriptiveStatistics:").append(endl);
481         outBuffer.append("n: ").append(getN()).append(endl);
482         outBuffer.append("min: ").append(getMin()).append(endl);
483         outBuffer.append("max: ").append(getMax()).append(endl);
484         outBuffer.append("mean: ").append(getMean()).append(endl);
485         outBuffer.append("std dev: ").append(getStandardDeviation())
486             .append(endl);
487         try {
488             // No catch for MIAE because actual parameter is valid below
489             outBuffer.append("median: ").append(getPercentile(50)).append(endl);
490         } catch (MathIllegalStateException ex) {
491             outBuffer.append("median: unavailable").append(endl);
492         }
493         outBuffer.append("skewness: ").append(getSkewness()).append(endl);
494         outBuffer.append("kurtosis: ").append(getKurtosis()).append(endl);
495         return outBuffer.toString();
496     }

755     /**
756      * Copies source to dest.
757      * <p>Neither source nor dest can be null.</p>
758      *
759      * @param source DescriptiveStatistics to copy
760      * @param dest DescriptiveStatistics to copy to
761      * @throws NullArgumentException if either source or dest is null
762      */
763     public static void copy(DescriptiveStatistics source, DescriptiveStatistics dest)
764         throws NullArgumentException {
765         MathUtils.checkNotNull(source);
766         MathUtils.checkNotNull(dest);
767         // Copy data and window size
768         dest.eDA = source.eDA.copy();
769         dest.windowSize = source.windowSize;
770 
771         // Copy implementations
772         dest.maxImpl = source.maxImpl.copy();
773         dest.meanImpl = source.meanImpl.copy();
774         dest.minImpl = source.minImpl.copy();
775         dest.sumImpl = source.sumImpl.copy();
776         dest.varianceImpl = source.varianceImpl.copy();
777         dest.sumsqImpl = source.sumsqImpl.copy();
778         dest.geometricMeanImpl = source.geometricMeanImpl.copy();
779         dest.kurtosisImpl = source.kurtosisImpl;
780         dest.skewnessImpl = source.skewnessImpl;
781         dest.percentileImpl = source.percentileImpl;
782     }
