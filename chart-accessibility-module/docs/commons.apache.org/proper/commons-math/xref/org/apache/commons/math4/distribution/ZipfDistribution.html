20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.rng.UniformRandomProvider;
23  import org.apache.commons.math4.util.FastMath;

25  /**
26   * Implementation of the Zipf distribution.
27   * <p>
28   * <strong>Parameters:</strong>
29   * For a random variable {@code X} whose values are distributed according to this
30   * distribution, the probability mass function is given by
31   * <pre>
32   *   P(X = k) = H(N,s) * 1 / k^s    for {@code k = 1,2,...,N}.
33   * </pre>
34   * {@code H(N,s)} is the normalizing constant
35   * which corresponds to the generalized harmonic number of order N of s.
36   * <p>
37   * <ul>
38   * <li>{@code N} is the number of elements</li>
39   * <li>{@code s} is the exponent</li>
40   * </ul>
41   * @see <a href="https://en.wikipedia.org/wiki/Zipf's_law">Zipf's law (Wikipedia)</a>
42   * @see <a href="https://en.wikipedia.org/wiki/Harmonic_number#Generalized_harmonic_numbers">Generalized harmonic numbers</a>
43   */
44  public class ZipfDistribution extends AbstractIntegerDistribution {
45      /** Serializable version identifier. */
46      private static final long serialVersionUID = 20150501L;
47      /** Number of elements. */
48      private final int numberOfElements;
49      /** Exponent parameter of the distribution. */
50      private final double exponent;
51      /** Cached values of the nth generalized harmonic. */
52      private final double nthHarmonic;
53      /** Cached numerical mean */
54      private double numericalMean = Double.NaN;
55      /** Whether or not the numerical mean has been calculated */
56      private boolean numericalMeanIsCalculated = false;
57      /** Cached numerical variance */
58      private double numericalVariance = Double.NaN;
59      /** Whether or not the numerical variance has been calculated */
60      private boolean numericalVarianceIsCalculated = false;
61  
62      /**
63       * Creates a distribution.
64       *
65       * @param numberOfElements Number of elements.
66       * @param exponent Exponent.
67       * @exception NotStrictlyPositiveException if {@code numberOfElements <= 0}
68       * or {@code exponent <= 0}.
69       */
70      public ZipfDistribution(int numberOfElements,
71                              double exponent)
72          throws NotStrictlyPositiveException {
73          if (numberOfElements <= 0) {
74              throw new NotStrictlyPositiveException(LocalizedFormats.DIMENSION,
75                                                     numberOfElements);
76          }
77          if (exponent <= 0) {
78              throw new NotStrictlyPositiveException(LocalizedFormats.EXPONENT,
79                                                     exponent);
80          }
81  
82          this.numberOfElements = numberOfElements;
83          this.exponent = exponent;
84          this.nthHarmonic = generalizedHarmonic(numberOfElements, exponent);
85      }
86  
87      /**
88       * Get the number of elements (e.g. corpus size) for the distribution.
89       *
90       * @return the number of elements
91       */
92      public int getNumberOfElements() {
93          return numberOfElements;
94      }
95  
96      /**
97       * Get the exponent characterizing the distribution.
98       *
99       * @return the exponent
100      */
101     public double getExponent() {
102         return exponent;
103     }
104 
105     /** {@inheritDoc} */
106     @Override
107     public double probability(final int x) {
108         if (x <= 0 || x > numberOfElements) {
109             return 0.0;
110         }
111 
112         return (1.0 / FastMath.pow(x, exponent)) / nthHarmonic;
113     }
114 
115     /** {@inheritDoc} */
116     @Override
117     public double logProbability(int x) {
118         if (x <= 0 || x > numberOfElements) {
119             return Double.NEGATIVE_INFINITY;
120         }
121 
122         return -FastMath.log(x) * exponent - FastMath.log(nthHarmonic);
123     }
124 
125     /** {@inheritDoc} */
126     @Override
127     public double cumulativeProbability(final int x) {
128         if (x <= 0) {
129             return 0.0;
130         } else if (x >= numberOfElements) {
131             return 1.0;
132         }
133 
134         return generalizedHarmonic(x, exponent) / nthHarmonic;
135     }
136 
137     /**
138      * {@inheritDoc}
139      *
140      * For number of elements {@code N} and exponent {@code s}, the mean is
141      * {@code Hs1 / Hs}, where
142      * <ul>
143      *  <li>{@code Hs1 = generalizedHarmonic(N, s - 1)},</li>
144      *  <li>{@code Hs = generalizedHarmonic(N, s)}.</li>
145      * </ul>
146      */
147     @Override
148     public double getNumericalMean() {
149         if (!numericalMeanIsCalculated) {
150             numericalMean = calculateNumericalMean();
151             numericalMeanIsCalculated = true;
152         }
153         return numericalMean;
154     }
155 
156     /**
157      * Used by {@link #getNumericalMean()}.
158      *
159      * @return the mean of this distribution
160      */
161     protected double calculateNumericalMean() {
162         final int N = getNumberOfElements();
163         final double s = getExponent();
164 
165         final double Hs1 = generalizedHarmonic(N, s - 1);
166         final double Hs = nthHarmonic;
167 
168         return Hs1 / Hs;
169     }
170 
171     /**
172      * {@inheritDoc}
173      *
174      * For number of elements {@code N} and exponent {@code s}, the mean is
175      * {@code (Hs2 / Hs) - (Hs1^2 / Hs^2)}, where
176      * <ul>
177      *  <li>{@code Hs2 = generalizedHarmonic(N, s - 2)},</li>
178      *  <li>{@code Hs1 = generalizedHarmonic(N, s - 1)},</li>
179      *  <li>{@code Hs = generalizedHarmonic(N, s)}.</li>
180      * </ul>
181      */
182     @Override
183     public double getNumericalVariance() {
184         if (!numericalVarianceIsCalculated) {
185             numericalVariance = calculateNumericalVariance();
186             numericalVarianceIsCalculated = true;
187         }
188         return numericalVariance;
189     }
190 
191     /**
192      * Used by {@link #getNumericalVariance()}.
193      *
194      * @return the variance of this distribution
195      */
196     protected double calculateNumericalVariance() {
197         final int N = getNumberOfElements();
198         final double s = getExponent();
199 
200         final double Hs2 = generalizedHarmonic(N, s - 2);
201         final double Hs1 = generalizedHarmonic(N, s - 1);
202         final double Hs = nthHarmonic;
203 
204         return (Hs2 / Hs) - ((Hs1 * Hs1) / (Hs * Hs));
205     }
206 
207     /**
208      * Calculates the Nth generalized harmonic number. See
209      * <a href="http://mathworld.wolfram.com/HarmonicSeries.html">Harmonic
210      * Series</a>.
211      *
212      * @param n Term in the series to calculate (must be larger than 1)
213      * @param m Exponent (special case {@code m = 1} is the harmonic series).
214      * @return the n<sup>th</sup> generalized harmonic number.
215      */
216     private double generalizedHarmonic(final int n, final double m) {
217         double value = 0;
218         for (int k = n; k > 0; --k) {
219             value += 1.0 / FastMath.pow(k, m);
220         }
221         return value;
222     }

229      * @return lower bound of the support (always 1)

241      * @return upper bound of the support

260     /**{@inheritDoc} */
261     @Override
262     public IntegerDistribution.Sampler createSampler(final UniformRandomProvider rng) {
263         return new IntegerDistribution.Sampler() {
264             /** Helper. */
265             private final ZipfRejectionInversionSampler sampler =
266                 new ZipfRejectionInversionSampler(numberOfElements, exponent);
267 
268             /** {@inheritDoc} */
269             @Override
270             public int sample() {
271                 return sampler.sample(rng);
272             }
273         };
274     }

276     /**
277      * Utility class implementing a rejection inversion sampling method for a discrete,
278      * bounded Zipf distribution that is based on the method described in
279      * <p>
280      * Wolfgang HÃ¶rmann and Gerhard Derflinger
281      * "Rejection-inversion to generate variates from monotone discrete distributions."
282      * ACM Transactions on Modeling and Computer Simulation (TOMACS) 6.3 (1996): 169-184.
283      * <p>
284      * The paper describes an algorithm for exponents larger than 1 (Algorithm ZRI).
285      * The original method uses {@code H(x) := (v + x)^(1 - q) / (1 - q)}
286      * as the integral of the hat function. This function is undefined for
287      * q = 1, which is the reason for the limitation of the exponent.
288      * If instead the integral function
289      * {@code H(x) := ((v + x)^(1 - q) - 1) / (1 - q)} is used,
290      * for which a meaningful limit exists for q = 1,
291      * the method works for all positive exponents.
292      * <p>
293      * The following implementation uses v := 0 and generates integral numbers
294      * in the range [1, numberOfElements]. This is different to the original method
295      * where v is defined to be positive and numbers are taken from [0, i_max].
296      * This explains why the implementation looks slightly different.
297      *
298      * @since 3.6
299      */
300     static class ZipfRejectionInversionSampler {
301         /** Exponent parameter of the distribution. */
302         private final double exponent;
303         /** Number of elements. */
304         private final int numberOfElements;
305         /** Constant equal to {@code hIntegral(1.5) - 1}. */
306         private final double hIntegralX1;
307         /** Constant equal to {@code hIntegral(numberOfElements + 0.5)}. */
308         private final double hIntegralNumberOfElements;
309         /** Constant equal to {@code 2 - hIntegralInverse(hIntegral(2.5) - h(2)}. */
310         private final double s;
311 
312         /**
313          * @param numberOfElements number of elements
314          * @param exponent exponent parameter of the distribution
315          */
316         ZipfRejectionInversionSampler(final int numberOfElements, final double exponent) {
317             this.exponent = exponent;
318             this.numberOfElements = numberOfElements;
319             this.hIntegralX1 = hIntegral(1.5) - 1d;
320             this.hIntegralNumberOfElements = hIntegral(numberOfElements + 0.5);
321             this.s = 2d - hIntegralInverse(hIntegral(2.5) - h(2));
322         }
323 
324         /**
325          * Generates one integer in the range [1, numberOfElements].
326          *
327          * @param random random generator to use
328          * @return generated integral number in the range [1, numberOfElements]
329          */
330         int sample(final UniformRandomProvider random) {
331             while(true) {
332                 final double u = hIntegralNumberOfElements + random.nextDouble() * (hIntegralX1 - hIntegralNumberOfElements);
333                 // u is uniformly distributed in (hIntegralX1, hIntegralNumberOfElements]
334 
335                 double x = hIntegralInverse(u);
336                 int k = (int)(x + 0.5);
337 
338                 // Limit k to the range [1, numberOfElements]
339                 // (k could be outside due to numerical inaccuracies)
340                 if (k < 1) {
341                     k = 1;
342                 }
343                 else if (k > numberOfElements) {
344                     k = numberOfElements;
345                 }
346 
347                 // Here, the distribution of k is given by:
348                 //
349                 //   P(k = 1) = C * (hIntegral(1.5) - hIntegralX1) = C
350                 //   P(k = m) = C * (hIntegral(m + 1/2) - hIntegral(m - 1/2)) for m >= 2
351                 //
352                 //   where C := 1 / (hIntegralNumberOfElements - hIntegralX1)
353 
354                 if (k - x <= s || u >= hIntegral(k + 0.5) - h(k)) {
355 
356                     // Case k = 1:
357                     //
358                     //   The right inequality is always true, because replacing k by 1 gives
359                     //   u >= hIntegral(1.5) - h(1) = hIntegralX1 and u is taken from
360                     //   (hIntegralX1, hIntegralNumberOfElements].
361                     //
362                     //   Therefore, the acceptance rate for k = 1 is P(accepted | k = 1) = 1
363                     //   and the probability that 1 is returned as random value is
364                     //   P(k = 1 and accepted) = P(accepted | k = 1) * P(k = 1) = C = C / 1^exponent
365                     //
366                     // Case k >= 2:
367                     //
368                     //   The left inequality (k - x <= s) is just a short cut
369                     //   to avoid the more expensive evaluation of the right inequality
370                     //   (u >= hIntegral(k + 0.5) - h(k)) in many cases.
371                     //
372                     //   If the left inequality is true, the right inequality is also true:
373                     //     Theorem 2 in the paper is valid for all positive exponents, because
374                     //     the requirements h'(x) = -exponent/x^(exponent + 1) < 0 and
375                     //     (-1/hInverse'(x))'' = (1+1/exponent) * x^(1/exponent-1) >= 0
376                     //     are both fulfilled.
377                     //     Therefore, f(x) := x - hIntegralInverse(hIntegral(x + 0.5) - h(x))
378                     //     is a non-decreasing function. If k - x <= s holds,
379                     //     k - x <= s + f(k) - f(2) is obviously also true which is equivalent to
380                     //     -x <= -hIntegralInverse(hIntegral(k + 0.5) - h(k)),
381                     //     -hIntegralInverse(u) <= -hIntegralInverse(hIntegral(k + 0.5) - h(k)),
382                     //     and finally u >= hIntegral(k + 0.5) - h(k).
383                     //
384                     //   Hence, the right inequality determines the acceptance rate:
385                     //   P(accepted | k = m) = h(m) / (hIntegrated(m+1/2) - hIntegrated(m-1/2))
386                     //   The probability that m is returned is given by
387                     //   P(k = m and accepted) = P(accepted | k = m) * P(k = m) = C * h(m) = C / m^exponent.
388                     //
389                     // In both cases the probabilities are proportional to the probability mass function
390                     // of the Zipf distribution.
391 
392                     return k;
393                 }
394             }
395         }
396 
397         /**
398          * {@code H(x) :=}
399          * <ul>
400          * <li>{@code (x^(1-exponent) - 1)/(1 - exponent)}, if {@code exponent != 1}</li>
401          * <li>{@code log(x)}, if {@code exponent == 1}</li>
402          * </ul>
403          * H(x) is an integral function of h(x),
404          * the derivative of H(x) is h(x).
405          *
406          * @param x free parameter
407          * @return {@code H(x)}
408          */
409         private double hIntegral(final double x) {
410             final double logX = FastMath.log(x);
411             return helper2((1d-exponent)*logX)*logX;
412         }
413 
414         /**
415          * {@code h(x) := 1/x^exponent}
416          *
417          * @param x free parameter
418          * @return h(x)
419          */
420         private double h(final double x) {
421             return FastMath.exp(-exponent * FastMath.log(x));
422         }
423 
424         /**
425          * The inverse function of H(x).
426          *
427          * @param x free parameter
428          * @return y for which {@code H(y) = x}
429          */
430         private double hIntegralInverse(final double x) {
431             double t = x*(1d-exponent);
432             if (t < -1d) {
433                 // Limit value to the range [-1, +inf).
434                 // t could be smaller than -1 in some rare cases due to numerical errors.
435                 t = -1;
436             }
437             return FastMath.exp(helper1(t)*x);
438         }
439 
440         /**
441          * Helper function that calculates {@code log(1+x)/x}.
442          * <p>
443          * A Taylor series expansion is used, if x is close to 0.
444          *
445          * @param x a value larger than or equal to -1
446          * @return {@code log(1+x)/x}
447          */
448         static double helper1(final double x) {
449             if (FastMath.abs(x)>1e-8) {
450                 return FastMath.log1p(x)/x;
451             }
452             else {
453                 return 1.-x*((1./2.)-x*((1./3.)-x*(1./4.)));
454             }
455         }
456 
457         /**
458          * Helper function to calculate {@code (exp(x)-1)/x}.
459          * <p>
460          * A Taylor series expansion is used, if x is close to 0.
461          *
462          * @param x free parameter
463          * @return {@code (exp(x)-1)/x} if x is non-zero, or 1 if x=0
464          */
465         static double helper2(final double x) {
466             if (FastMath.abs(x)>1e-8) {
467                 return FastMath.expm1(x)/x;
468             }
469             else {
470                 return 1.+x*(1./2.)*(1.+x*(1./3.)*(1.+x*(1./4.)));
471             }
472         }
473     }
