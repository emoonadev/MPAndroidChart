33  /** Extractor for {@link PolygonsSet polyhedrons sets} outlines.
34   * <p>This class extracts the 2D outlines from {{@link PolygonsSet
35   * polyhedrons sets} in a specified projection plane.</p>
36   * @since 3.0
37   */
38  public class OutlineExtractor {
39  
40      /** Abscissa axis of the projection plane. */
41      private final Vector3D u;
42  
43      /** Ordinate axis of the projection plane. */
44      private final Vector3D v;
45  
46      /** Normal of the projection plane (viewing direction). */
47      private final Vector3D w;
48  
49      /** Build an extractor for a specific projection plane.
50       * @param u abscissa axis of the projection point
51       * @param v ordinate axis of the projection point
52       */
53      public OutlineExtractor(final Vector3D u, final Vector3D v) {
54          this.u = u;
55          this.v = v;
56          w = Vector3D.crossProduct(u, v);
57      }
58  
59      /** Extract the outline of a polyhedrons set.
60       * @param polyhedronsSet polyhedrons set whose outline must be extracted
61       * @return an outline, as an array of loops.
62       */
63      public Vector2D[][] getOutline(final PolyhedronsSet polyhedronsSet) {
64  
65          // project all boundary facets into one polygons set
66          final BoundaryProjector projector = new BoundaryProjector(polyhedronsSet.getTolerance());
67          polyhedronsSet.getTree(true).visit(projector);
68          final PolygonsSet projected = projector.getProjected();
69  
70          // Remove the spurious intermediate vertices from the outline
71          final Vector2D[][] outline = projected.getVertices();
72          for (int i = 0; i < outline.length; ++i) {
73              final Vector2D[] rawLoop = outline[i];
74              int end = rawLoop.length;
75              int j = 0;
76              while (j < end) {
77                  if (pointIsBetween(rawLoop, end, j)) {
78                      // the point should be removed
79                      for (int k = j; k < (end - 1); ++k) {
80                          rawLoop[k] = rawLoop[k + 1];
81                      }
82                      --end;
83                  } else {
84                      // the point remains in the loop
85                      ++j;
86                  }
87              }
88              if (end != rawLoop.length) {
89                  // resize the array
90                  outline[i] = new Vector2D[end];
91                  System.arraycopy(rawLoop, 0, outline[i], 0, end);
92              }
93          }
94  
95          return outline;
96  
97      }
98  
99      /** Check if a point is geometrically between its neighbor in an array.
100      * <p>The neighbors are computed considering the array is a loop
101      * (i.e. point at index (n-1) is before point at index 0)</p>
102      * @param loop points array
103      * @param n number of points to consider in the array
104      * @param i index of the point to check (must be between 0 and n-1)
105      * @return true if the point is exactly between its neighbors
106      */
107     private boolean pointIsBetween(final Vector2D[] loop, final int n, final int i) {
108         final Vector2D previous = loop[(i + n - 1) % n];
109         final Vector2D current  = loop[i];
110         final Vector2D next     = loop[(i + 1) % n];
111         final double dx1       = current.getX() - previous.getX();
112         final double dy1       = current.getY() - previous.getY();
113         final double dx2       = next.getX()    - current.getX();
114         final double dy2       = next.getY()    - current.getY();
115         final double cross     = dx1 * dy2 - dx2 * dy1;
116         final double dot       = dx1 * dx2 + dy1 * dy2;
117         final double d1d2      = FastMath.sqrt((dx1 * dx1 + dy1 * dy1) * (dx2 * dx2 + dy2 * dy2));
118         return (FastMath.abs(cross) <= (1.0e-6 * d1d2)) && (dot >= 0.0);
119     }

121     /** Visitor projecting the boundary facets on a plane. */
122     private class BoundaryProjector implements BSPTreeVisitor<Euclidean3D> {
123 
124         /** Projection of the polyhedrons set on the plane. */
125         private PolygonsSet projected;
126 
127         /** Tolerance below which points are considered identical. */
128         private final double tolerance;
129 
130         /** Simple constructor.
131          * @param tolerance tolerance below which points are considered identical
132          */
133         BoundaryProjector(final double tolerance) {
134             this.projected = new PolygonsSet(new BSPTree<Euclidean2D>(Boolean.FALSE), tolerance);
135             this.tolerance = tolerance;
136         }
137 
138         /** {@inheritDoc} */
139         @Override
140         public Order visitOrder(final BSPTree<Euclidean3D> node) {
141             return Order.MINUS_SUB_PLUS;
142         }
143 
144         /** {@inheritDoc} */
145         @Override
146         public void visitInternalNode(final BSPTree<Euclidean3D> node) {
147             @SuppressWarnings("unchecked")
148             final BoundaryAttribute<Euclidean3D> attribute =
149                 (BoundaryAttribute<Euclidean3D>) node.getAttribute();
150             if (attribute.getPlusOutside() != null) {
151                 addContribution(attribute.getPlusOutside(), false);
152             }
153             if (attribute.getPlusInside() != null) {
154                 addContribution(attribute.getPlusInside(), true);
155             }
156         }
157 
158         /** {@inheritDoc} */
159         @Override
160         public void visitLeafNode(final BSPTree<Euclidean3D> node) {
161         }
162 
163         /** Add he contribution of a boundary facet.
164          * @param facet boundary facet
165          * @param reversed if true, the facet has the inside on its plus side
166          */
167         private void addContribution(final SubHyperplane<Euclidean3D> facet, final boolean reversed) {
168 
169             // extract the vertices of the facet
170             @SuppressWarnings("unchecked")
171             final AbstractSubHyperplane<Euclidean3D, Euclidean2D> absFacet =
172                 (AbstractSubHyperplane<Euclidean3D, Euclidean2D>) facet;
173             final Plane plane    = (Plane) facet.getHyperplane();
174 
175             final double scal = plane.getNormal().dotProduct(w);
176             if (FastMath.abs(scal) > 1.0e-3) {
177                 Vector2D[][] vertices =
178                     ((PolygonsSet) absFacet.getRemainingRegion()).getVertices();
179 
180                 if ((scal < 0) ^ reversed) {
181                     // the facet is seen from the inside,
182                     // we need to invert its boundary orientation
183                     final Vector2D[][] newVertices = new Vector2D[vertices.length][];
184                     for (int i = 0; i < vertices.length; ++i) {
185                         final Vector2D[] loop = vertices[i];
186                         final Vector2D[] newLoop = new Vector2D[loop.length];
187                         if (loop[0] == null) {
188                             newLoop[0] = null;
189                             for (int j = 1; j < loop.length; ++j) {
190                                 newLoop[j] = loop[loop.length - j];
191                             }
192                         } else {
193                             for (int j = 0; j < loop.length; ++j) {
194                                 newLoop[j] = loop[loop.length - (j + 1)];
195                             }
196                         }
197                         newVertices[i] = newLoop;
198                     }
199 
200                     // use the reverted vertices
201                     vertices = newVertices;
202 
203                 }
204 
205                 // compute the projection of the facet in the outline plane
206                 final ArrayList<SubHyperplane<Euclidean2D>> edges = new ArrayList<SubHyperplane<Euclidean2D>>();
207                 for (Vector2D[] loop : vertices) {
208                     final boolean closed = loop[0] != null;
209                     int previous         = closed ? (loop.length - 1) : 1;
210                     Vector3D previous3D  = plane.toSpace((Point<Euclidean2D>) loop[previous]);
211                     int current          = (previous + 1) % loop.length;
212                     Vector2D pPoint       = new Vector2D(previous3D.dotProduct(u),
213                                                          previous3D.dotProduct(v));
214                     while (current < loop.length) {
215 
216                         final Vector3D current3D = plane.toSpace((Point<Euclidean2D>) loop[current]);
217                         final Vector2D  cPoint    = new Vector2D(current3D.dotProduct(u),
218                                                                  current3D.dotProduct(v));
219                         final org.apache.commons.math4.geometry.euclidean.twod.Line line =
220                             new org.apache.commons.math4.geometry.euclidean.twod.Line(pPoint, cPoint, tolerance);
221                         SubHyperplane<Euclidean2D> edge = line.wholeHyperplane();
222 
223                         if (closed || (previous != 1)) {
224                             // the previous point is a real vertex
225                             // it defines one bounding point of the edge
226                             final double angle = line.getAngle() + 0.5 * FastMath.PI;
227                             final org.apache.commons.math4.geometry.euclidean.twod.Line l =
228                                 new org.apache.commons.math4.geometry.euclidean.twod.Line(pPoint, angle, tolerance);
229                             edge = edge.split(l).getPlus();
230                         }
231 
232                         if (closed || (current != (loop.length - 1))) {
233                             // the current point is a real vertex
234                             // it defines one bounding point of the edge
235                             final double angle = line.getAngle() + 0.5 * FastMath.PI;
236                             final org.apache.commons.math4.geometry.euclidean.twod.Line l =
237                                 new org.apache.commons.math4.geometry.euclidean.twod.Line(cPoint, angle, tolerance);
238                             edge = edge.split(l).getMinus();
239                         }
240 
241                         edges.add(edge);
242 
243                         previous   = current++;
244                         previous3D = current3D;
245                         pPoint     = cPoint;
246 
247                     }
248                 }
249                 final PolygonsSet projectedFacet = new PolygonsSet(edges, tolerance);
250 
251                 // add the contribution of the facet to the global outline
252                 projected = (PolygonsSet) new RegionFactory<Euclidean2D>().union(projected, projectedFacet);
253 
254             }
255         }
256 
257         /** Get the projection of the polyhedrons set on the plane.
258          * @return projection of the polyhedrons set on the plane
259          */
260         public PolygonsSet getProjected() {
261             return projected;
262         }
263 
264     }
