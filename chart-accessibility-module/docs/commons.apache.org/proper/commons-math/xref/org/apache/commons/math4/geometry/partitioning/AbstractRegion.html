31  /** Abstract class for all regions, independently of geometry type or dimension.
32  
33   * @param <S> Type of the space.
34   * @param <T> Type of the sub-space.

36   * @since 3.0
37   */
38  public abstract class AbstractRegion<S extends Space, T extends Space> implements Region<S> {

60      /** Build a region from an inside/outside BSP tree.
61       * <p>The leaf nodes of the BSP tree <em>must</em> have a
62       * {@code Boolean} attribute representing the inside status of
63       * the corresponding cell (true for inside cells, false for outside
64       * cells). In order to avoid building too many small objects, it is
65       * recommended to use the predefined constants
66       * {@code Boolean.TRUE} and {@code Boolean.FALSE}. The
67       * tree also <em>must</em> have either null internal nodes or
68       * internal nodes representing the boundary as specified in the
69       * {@link #getTree getTree} method).</p>
70       * @param tree inside/outside BSP tree representing the region
71       * @param tolerance tolerance below which points are considered identical.
72       */
73      protected AbstractRegion(final BSPTree<S> tree, final double tolerance) {
74          this.tree      = tree;
75          this.tolerance = tolerance;
76      }
77  
78      /** Build a Region from a Boundary REPresentation (B-rep).
79       * <p>The boundary is provided as a collection of {@link
80       * SubHyperplane sub-hyperplanes}. Each sub-hyperplane has the
81       * interior part of the region on its minus side and the exterior on
82       * its plus side.</p>
83       * <p>The boundary elements can be in any order, and can form
84       * several non-connected sets (like for example polygons with holes
85       * or a set of disjoints polyhedrons considered as a whole). In
86       * fact, the elements do not even need to be connected together
87       * (their topological connections are not used here). However, if the
88       * boundary does not really separate an inside open from an outside
89       * open (open having here its topological meaning), then subsequent
90       * calls to the {@link #checkPoint(Point) checkPoint} method will not be
91       * meaningful anymore.</p>
92       * <p>If the boundary is empty, the region will represent the whole
93       * space.</p>
94       * @param boundary collection of boundary elements, as a
95       * collection of {@link SubHyperplane SubHyperplane} objects
96       * @param tolerance tolerance below which points are considered identical.
97       */
98      protected AbstractRegion(final Collection<SubHyperplane<S>> boundary, final double tolerance) {
99  
100         this.tolerance = tolerance;
101 
102         if (boundary.size() == 0) {
103 
104             // the tree represents the whole space
105             tree = new BSPTree<S>(Boolean.TRUE);
106 
107         } else {
108 
109             // sort the boundary elements in decreasing size order
110             // (we don't want equal size elements to be removed, so
111             // we use a trick to fool the TreeSet)
112             final TreeSet<SubHyperplane<S>> ordered = new TreeSet<SubHyperplane<S>>(new Comparator<SubHyperplane<S>>() {
113                 /** {@inheritDoc} */
114                 @Override
115                 public int compare(final SubHyperplane<S> o1, final SubHyperplane<S> o2) {
116                     final double size1 = o1.getSize();
117                     final double size2 = o2.getSize();
118                     return (size2 < size1) ? -1 : ((o1 == o2) ? 0 : +1);
119                 }
120             });
121             ordered.addAll(boundary);
122 
123             // build the tree top-down
124             tree = new BSPTree<S>();
125             insertCuts(tree, ordered);
126 
127             // set up the inside/outside flags
128             tree.visit(new BSPTreeVisitor<S>() {
129 
130                 /** {@inheritDoc} */
131                 @Override
132                 public Order visitOrder(final BSPTree<S> node) {
133                     return Order.PLUS_SUB_MINUS;
134                 }
135 
136                 /** {@inheritDoc} */
137                 @Override
138                 public void visitInternalNode(final BSPTree<S> node) {
139                 }
140 
141                 /** {@inheritDoc} */
142                 @Override
143                 public void visitLeafNode(final BSPTree<S> node) {
144                     if (node.getParent() == null || node == node.getParent().getMinus()) {
145                         node.setAttribute(Boolean.TRUE);
146                     } else {
147                         node.setAttribute(Boolean.FALSE);
148                     }
149                 }
150             });
151 
152         }
153 
154     }

156     /** Build a convex region from an array of bounding hyperplanes.
157      * @param hyperplanes array of bounding hyperplanes (if null, an
158      * empty region will be built)
159      * @param tolerance tolerance below which points are considered identical.
160      */
161     public AbstractRegion(final Hyperplane<S>[] hyperplanes, final double tolerance) {
162         this.tolerance = tolerance;
163         if ((hyperplanes == null) || (hyperplanes.length == 0)) {
164             tree = new BSPTree<S>(Boolean.FALSE);
165         } else {
166 
167             // use the first hyperplane to build the right class
168             tree = hyperplanes[0].wholeSpace().getTree(false);
169 
170             // chop off parts of the space
171             BSPTree<S> node = tree;
172             node.setAttribute(Boolean.TRUE);
173             for (final Hyperplane<S> hyperplane : hyperplanes) {
174                 if (node.insertCut(hyperplane)) {
175                     node.setAttribute(null);
176                     node.getPlus().setAttribute(Boolean.FALSE);
177                     node = node.getMinus();
178                     node.setAttribute(Boolean.TRUE);
179                 }
180             }
181 
182         }
183 
184     }

197     /** Recursively build a tree by inserting cut sub-hyperplanes.
198      * @param node current tree node (it is a leaf node at the beginning
199      * of the call)
200      * @param boundary collection of edges belonging to the cell defined
201      * by the node
202      */
203     private void insertCuts(final BSPTree<S> node, final Collection<SubHyperplane<S>> boundary) {
204 
205         final Iterator<SubHyperplane<S>> iterator = boundary.iterator();
206 
207         // build the current level
208         Hyperplane<S> inserted = null;
209         while ((inserted == null) && iterator.hasNext()) {
210             inserted = iterator.next().getHyperplane();
211             if (!node.insertCut(inserted.copySelf())) {
212                 inserted = null;
213             }
214         }
215 
216         if (!iterator.hasNext()) {
217             return;
218         }
219 
220         // distribute the remaining edges in the two sub-trees
221         final ArrayList<SubHyperplane<S>> plusList  = new ArrayList<SubHyperplane<S>>();
222         final ArrayList<SubHyperplane<S>> minusList = new ArrayList<SubHyperplane<S>>();
223         while (iterator.hasNext()) {
224             final SubHyperplane<S> other = iterator.next();
225             final SubHyperplane.SplitSubHyperplane<S> split = other.split(inserted);
226             switch (split.getSide()) {
227             case PLUS:
228                 plusList.add(other);
229                 break;
230             case MINUS:
231                 minusList.add(other);
232                 break;
233             case BOTH:
234                 plusList.add(split.getPlus());
235                 minusList.add(split.getMinus());
236                 break;
237             default:
238                 // ignore the sub-hyperplanes belonging to the cut hyperplane
239             }
240         }
241 
242         // recurse through lower levels
243         insertCuts(node.getPlus(),  plusList);
244         insertCuts(node.getMinus(), minusList);
245 
246     }

260     /** {@inheritDoc} */
261     @Override
262     public boolean isEmpty(final BSPTree<S> node) {
263 
264         // we use a recursive function rather than the BSPTreeVisitor
265         // interface because we can stop visiting the tree as soon as we
266         // have found an inside cell
267 
268         if (node.getCut() == null) {
269             // if we find an inside node, the region is not empty
270             return !((Boolean) node.getAttribute());
271         }
272 
273         // check both sides of the sub-tree
274         return isEmpty(node.getMinus()) && isEmpty(node.getPlus());
275 
276     }

284     /** {@inheritDoc} */
285     @Override
286     public boolean isFull(final BSPTree<S> node) {
287 
288         // we use a recursive function rather than the BSPTreeVisitor
289         // interface because we can stop visiting the tree as soon as we
290         // have found an outside cell
291 
292         if (node.getCut() == null) {
293             // if we find an outside node, the region does not cover full space
294             return (Boolean) node.getAttribute();
295         }
296 
297         // check both sides of the sub-tree
298         return isFull(node.getMinus()) && isFull(node.getPlus());
299 
300     }

334     /** Check a point with respect to the region starting at a given node.
335      * @param node root node of the region
336      * @param point point to check
337      * @return a code representing the point status: either {@link
338      * Region.Location#INSIDE INSIDE}, {@link Region.Location#OUTSIDE
339      * OUTSIDE} or {@link Region.Location#BOUNDARY BOUNDARY}

341     protected Location checkPoint(final BSPTree<S> node, final Vector<S> point) {
342         return checkPoint(node, (Point<S>) point);
343     }
344 
345     /** Check a point with respect to the region starting at a given node.
346      * @param node root node of the region
347      * @param point point to check
348      * @return a code representing the point status: either {@link
349      * Region.Location#INSIDE INSIDE}, {@link Region.Location#OUTSIDE
350      * OUTSIDE} or {@link Region.Location#BOUNDARY BOUNDARY}
351      */
352     protected Location checkPoint(final BSPTree<S> node, final Point<S> point) {
353         final BSPTree<S> cell = node.getCell(point, tolerance);
354         if (cell.getCut() == null) {
355             // the point is in the interior of a cell, just check the attribute
356             return ((Boolean) cell.getAttribute()) ? Location.INSIDE : Location.OUTSIDE;
357         }
358 
359         // the point is on a cut-sub-hyperplane, is it on a boundary ?
360         final Location minusCode = checkPoint(cell.getMinus(), point);
361         final Location plusCode  = checkPoint(cell.getPlus(),  point);
362         return (minusCode == plusCode) ? minusCode : Location.BOUNDARY;
363 
364     }

434     /** Recursively compute the parts of a sub-hyperplane that are
435      * contained in the region.
436      * @param node current BSP tree node
437      * @param sub sub-hyperplane traversing the region
438      * @return filtered sub-hyperplane
439      */
440     private SubHyperplane<S> recurseIntersection(final BSPTree<S> node, final SubHyperplane<S> sub) {
441 
442         if (node.getCut() == null) {
443             return (Boolean) node.getAttribute() ? sub.copySelf() : null;
444         }
445 
446         final Hyperplane<S> hyperplane = node.getCut().getHyperplane();
447         final SubHyperplane.SplitSubHyperplane<S> split = sub.split(hyperplane);
448         if (split.getPlus() != null) {
449             if (split.getMinus() != null) {
450                 // both sides
451                 final SubHyperplane<S> plus  = recurseIntersection(node.getPlus(),  split.getPlus());
452                 final SubHyperplane<S> minus = recurseIntersection(node.getMinus(), split.getMinus());
453                 if (plus == null) {
454                     return minus;
455                 } else if (minus == null) {
456                     return plus;
457                 } else {
458                     return plus.reunite(minus);
459                 }
460             } else {
461                 // only on plus side
462                 return recurseIntersection(node.getPlus(), sub);
463             }
464         } else if (split.getMinus() != null) {
465             // only on minus side
466             return recurseIntersection(node.getMinus(), sub);
467         } else {
468             // on hyperplane
469             return recurseIntersection(node.getPlus(),
470                                        recurseIntersection(node.getMinus(), sub));
471         }
472 
473     }

475     /** Transform a region.
476      * <p>Applying a transform to a region consist in applying the
477      * transform to all the hyperplanes of the underlying BSP tree and
478      * of the boundary (and also to the sub-hyperplanes embedded in
479      * these hyperplanes) and to the barycenter. The instance is not
480      * modified, a new instance is built.</p>
481      * @param transform transform to apply
482      * @return a new region, resulting from the application of the
483      * transform to the instance
484      */
485     public AbstractRegion<S, T> applyTransform(final Transform<S, T> transform) {
486 
487         // transform the tree, except for boundary attribute splitters
488         final Map<BSPTree<S>, BSPTree<S>> map = new HashMap<BSPTree<S>, BSPTree<S>>();
489         final BSPTree<S> transformedTree = recurseTransform(getTree(false), transform, map);
490 
491         // set up the boundary attributes splitters
492         for (final Map.Entry<BSPTree<S>, BSPTree<S>> entry : map.entrySet()) {
493             if (entry.getKey().getCut() != null) {
494                 @SuppressWarnings("unchecked")
495                 BoundaryAttribute<S> original = (BoundaryAttribute<S>) entry.getKey().getAttribute();
496                 if (original != null) {
497                     @SuppressWarnings("unchecked")
498                     BoundaryAttribute<S> transformed = (BoundaryAttribute<S>) entry.getValue().getAttribute();
499                     for (final BSPTree<S> splitter : original.getSplitters()) {
500                         transformed.getSplitters().add(map.get(splitter));
501                     }
502                 }
503             }
504         }
505 
506         return buildNew(transformedTree);
507 
508     }

510     /** Recursively transform an inside/outside BSP-tree.
511      * @param node current BSP tree node
512      * @param transform transform to apply
513      * @param map transformed nodes map
514      * @return a new tree
515      */
516     @SuppressWarnings("unchecked")
517     private BSPTree<S> recurseTransform(final BSPTree<S> node, final Transform<S, T> transform,
518                                         final Map<BSPTree<S>, BSPTree<S>> map) {
519 
520         final BSPTree<S> transformedNode;
521         if (node.getCut() == null) {
522             transformedNode = new BSPTree<S>(node.getAttribute());
523         } else {
524 
525             final SubHyperplane<S>  sub = node.getCut();
526             final SubHyperplane<S> tSub = ((AbstractSubHyperplane<S, T>) sub).applyTransform(transform);
527             BoundaryAttribute<S> attribute = (BoundaryAttribute<S>) node.getAttribute();
528             if (attribute != null) {
529                 final SubHyperplane<S> tPO = (attribute.getPlusOutside() == null) ?
530                     null : ((AbstractSubHyperplane<S, T>) attribute.getPlusOutside()).applyTransform(transform);
531                 final SubHyperplane<S> tPI = (attribute.getPlusInside()  == null) ?
532                     null  : ((AbstractSubHyperplane<S, T>) attribute.getPlusInside()).applyTransform(transform);
533                 // we start with an empty list of splitters, it will be filled in out of recursion
534                 attribute = new BoundaryAttribute<S>(tPO, tPI, new NodesSet<S>());
535             }
536 
537             transformedNode = new BSPTree<S>(tSub,
538                                              recurseTransform(node.getPlus(),  transform, map),
539                                              recurseTransform(node.getMinus(), transform, map),
540                                              attribute);
541         }
542 
543         map.put(node, transformedNode);
544         return transformedNode;
545 
546     }
