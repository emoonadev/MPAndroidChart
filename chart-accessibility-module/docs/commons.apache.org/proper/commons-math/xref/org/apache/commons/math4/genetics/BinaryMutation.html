17  package org.apache.commons.math4.genetics;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import org.apache.commons.math4.exception.MathIllegalArgumentException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  
25  /**
26   * Mutation for {@link BinaryChromosome}s. Randomly changes one gene.
27   *
28   * @since 2.0
29   */
30  public class BinaryMutation implements MutationPolicy {
31  
32      /**
33       * Mutate the given chromosome. Randomly changes one gene.
34       *
35       * @param original the original chromosome.
36       * @return the mutated chromosome.
37       * @throws MathIllegalArgumentException if <code>original</code> is not an instance of {@link BinaryChromosome}.
38       */
39      @Override
40      public Chromosome mutate(Chromosome original) throws MathIllegalArgumentException {
41          if (!(original instanceof BinaryChromosome)) {
42              throw new MathIllegalArgumentException(LocalizedFormats.INVALID_BINARY_CHROMOSOME);
43          }
44  
45          BinaryChromosome origChrom = (BinaryChromosome) original;
46          List<Integer> newRepr = new ArrayList<Integer>(origChrom.getRepresentation());
47  
48          // randomly select a gene
49          int geneIndex = GeneticAlgorithm.getRandomGenerator().nextInt(origChrom.getLength());
50          // and change it
51          newRepr.set(geneIndex, origChrom.getRepresentation().get(geneIndex) == 0 ? 1 : 0);
52  
53          Chromosome newChrom = origChrom.newFixedLengthChromosome(newRepr);
54          return newChrom;
55      }
56  
57  }
