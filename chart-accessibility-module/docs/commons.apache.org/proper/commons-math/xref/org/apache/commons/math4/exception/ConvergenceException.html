17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Error thrown when a numerical computation can not be performed because the
24   * numerical result failed to converge to a finite value.
25   *
26   * @since 2.2
27   */
28  public class ConvergenceException extends MathIllegalStateException {
29      /** Serializable version Id. */
30      private static final long serialVersionUID = 4330003017885151975L;
31  
32      /**
33       * Construct the exception.
34       */
35      public ConvergenceException() {
36          this(LocalizedFormats.CONVERGENCE_FAILED);
37      }
38  
39      /**
40       * Construct the exception with a specific context and arguments.
41       *
42       * @param pattern Message pattern providing the specific context of
43       * the error.
44       * @param args Arguments.
45       */
46      public ConvergenceException(Localizable pattern, Object ... args) {
47          getContext().addMessage(pattern, args);
48      }
49  }
