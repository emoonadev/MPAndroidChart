17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception triggered when something that shouldn't happen does happen.
24   *
25   * @since 2.2
26   */
27  public class MathInternalError extends MathIllegalStateException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -6276776513966934846L;
30      /** URL for reporting problems. */
31      private static final String REPORT_URL = "https://issues.apache.org/jira/browse/MATH";
32  
33      /**
34       * Simple constructor.
35       */
36      public MathInternalError() {
37          getContext().addMessage(LocalizedFormats.INTERNAL_ERROR, REPORT_URL);
38      }
39  
40      /**
41       * Simple constructor.
42       * @param cause root cause
43       */
44      public MathInternalError(final Throwable cause) {
45          super(cause, LocalizedFormats.INTERNAL_ERROR, REPORT_URL);
46      }
47  
48      /**
49       * Constructor accepting a localized message.
50       *
51       * @param pattern Message pattern explaining the cause of the error.
52       * @param args Arguments.
53       */
54      public MathInternalError(Localizable pattern, Object ... args) {
55          super(pattern, args);
56      }
57  }
