28  /**
29   * This class implements the 8(5,3) Dormand-Prince integrator for Ordinary
30   * Differential Equations.
31   *
32   * <p>This integrator is an embedded Runge-Kutta integrator
33   * of order 8(5,3) used in local extrapolation mode (i.e. the solution
34   * is computed using the high order formula) with stepsize control
35   * (and automatic step initialization) and continuous output. This
36   * method uses 12 functions evaluations per step for integration and 4
37   * evaluations for interpolation. However, since the first
38   * interpolation evaluation is the same as the first integration
39   * evaluation of the next step, we have included it in the integrator
40   * rather than in the interpolator and specified the method was an
41   * <i>fsal</i>. Hence, despite we have 13 stages here, the cost is
42   * really 12 evaluations per step even if no interpolation is done,
43   * and the overcost of interpolation is only 3 evaluations.</p>
44   *
45   * <p>This method is based on an 8(6) method by Dormand and Prince
46   * (i.e. order 8 for the integration and order 6 for error estimation)
47   * modified by Hairer and Wanner to use a 5th order error estimator
48   * with 3rd order correction. This modification was introduced because
49   * the original method failed in some cases (wrong steps can be
50   * accepted when step size is too large, for example in the
51   * Brusselator problem) and also had <i>severe difficulties when
52   * applied to problems with discontinuities</i>. This modification is
53   * explained in the second edition of the first volume (Nonstiff
54   * Problems) of the reference book by Hairer, Norsett and Wanner:
55   * <i>Solving Ordinary Differential Equations</i> (Springer-Verlag,
56   * ISBN 3-540-56670-8).</p>
57   *
58   * @param <T> the type of the field elements
59   * @since 3.6
60   */
61

414     /** {@inheritDoc} */
415     @Override
416     protected T estimateError(final T[][] yDotK, final T[] y0, final T[] y1, final T h) {
417         T error1 = h.getField().getZero();
418         T error2 = h.getField().getZero();
419 
420         for (int j = 0; j < mainSetDimension; ++j) {
421             final T errSum1 =      yDotK[ 0][j].multiply(e1_01).
422                                add(yDotK[ 5][j].multiply(e1_06)).
423                                add(yDotK[ 6][j].multiply(e1_07)).
424                                add(yDotK[ 7][j].multiply(e1_08)).
425                                add(yDotK[ 8][j].multiply(e1_09)).
426                                add(yDotK[ 9][j].multiply(e1_10)).
427                                add(yDotK[10][j].multiply(e1_11)).
428                                add(yDotK[11][j].multiply(e1_12));
429             final T errSum2 =      yDotK[ 0][j].multiply(e2_01).
430                                add(yDotK[ 5][j].multiply(e2_06)).
431                                add(yDotK[ 6][j].multiply(e2_07)).
432                                add(yDotK[ 7][j].multiply(e2_08)).
433                                add(yDotK[ 8][j].multiply(e2_09)).
434                                add(yDotK[ 9][j].multiply(e2_10)).
435                                add(yDotK[10][j].multiply(e2_11)).
436                                add(yDotK[11][j].multiply(e2_12));
437 
438             final T yScale = MathUtils.max(y0[j].abs(), y1[j].abs());
439             final T tol = vecAbsoluteTolerance == null ?
440                           yScale.multiply(scalRelativeTolerance).add(scalAbsoluteTolerance) :
441                           yScale.multiply(vecRelativeTolerance[j]).add(vecAbsoluteTolerance[j]);
442             final T ratio1  = errSum1.divide(tol);
443             error1        = error1.add(ratio1.multiply(ratio1));
444             final T ratio2  = errSum2.divide(tol);
445             error2        = error2.add(ratio2.multiply(ratio2));
446         }
447 
448         T den = error1.add(error2.multiply(0.01));
449         if (den.getReal() <= 0.0) {
450             den = h.getField().getOne();
451         }
452 
453         return h.abs().multiply(error1).divide(den.multiply(mainSetDimension).sqrt());
454 
455     }
