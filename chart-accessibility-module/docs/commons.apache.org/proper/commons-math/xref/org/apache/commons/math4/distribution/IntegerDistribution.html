17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NumberIsTooLargeException;
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.rng.UniformRandomProvider;
22  
23  /**
24   * Interface for distributions on the integers.
25   */
26  public interface IntegerDistribution {
27  
28      /**
29       * For a random variable {@code X} whose values are distributed according to
30       * this distribution, this method returns {@code log(P(X = x))}, where
31       * {@code log} is the natural logarithm. In other words, this method
32       * represents the logarithm of the probability mass function (PMF) for the
33       * distribution. Note that due to the floating point precision and
34       * under/overflow issues, this method will for some distributions be more
35       * precise and faster than computing the logarithm of
36       * {@link #probability(int)}.
37       *
38       * @param x the point at which the PMF is evaluated
39       * @return the logarithm of the value of the probability mass function at {@code x}
40       * @since 4.0
41       */
42      double logProbability(int x);
43  
44      /**
45       * For a random variable {@code X} whose values are distributed according
46       * to this distribution, this method returns {@code P(X = x)}. In other
47       * words, this method represents the probability mass function (PMF)
48       * for the distribution.
49       *
50       * @param x the point at which the PMF is evaluated
51       * @return the value of the probability mass function at {@code x}
52       */
53      double probability(int x);
54  
55      /**
56       * For a random variable {@code X} whose values are distributed according
57       * to this distribution, this method returns {@code P(x0 < X <= x1)}.
58       *
59       * @param x0 the exclusive lower bound
60       * @param x1 the inclusive upper bound
61       * @return the probability that a random variable with this distribution
62       * will take a value between {@code x0} and {@code x1},
63       * excluding the lower and including the upper endpoint
64       * @throws NumberIsTooLargeException if {@code x0 > x1}
65       *
66       * @since 4.0, was previously named cumulativeProbability
67       */
68      double probability(int x0, int x1) throws NumberIsTooLargeException;
69  
70      /**
71       * For a random variable {@code X} whose values are distributed according
72       * to this distribution, this method returns {@code P(X <= x)}.  In other
73       * words, this method represents the (cumulative) distribution function
74       * (CDF) for this distribution.
75       *
76       * @param x the point at which the CDF is evaluated
77       * @return the probability that a random variable with this
78       * distribution takes a value less than or equal to {@code x}
79       */
80      double cumulativeProbability(int x);
81  
82      /**
83       * Computes the quantile function of this distribution.
84       * For a random variable {@code X} distributed according to this distribution,
85       * the returned value is
86       * <ul>
87       * <li><code>inf{x in Z | P(X<=x) >= p}</code> for {@code 0 < p <= 1},</li>
88       * <li><code>inf{x in Z | P(X<=x) > 0}</code> for {@code p = 0}.</li>
89       * </ul>
90       * If the result exceeds the range of the data type {@code int},
91       * then {@code Integer.MIN_VALUE} or {@code Integer.MAX_VALUE} is returned.
92       *
93       * @param p the cumulative probability
94       * @return the smallest {@code p}-quantile of this distribution
95       * (largest 0-quantile for {@code p = 0})
96       * @throws OutOfRangeException if {@code p < 0} or {@code p > 1}
97       */
98      int inverseCumulativeProbability(double p) throws OutOfRangeException;
99  
100     /**
101      * Use this method to get the numerical value of the mean of this
102      * distribution.
103      *
104      * @return the mean or {@code Double.NaN} if it is not defined
105      */
106     double getNumericalMean();
107 
108     /**
109      * Use this method to get the numerical value of the variance of this
110      * distribution.
111      *
112      * @return the variance (possibly {@code Double.POSITIVE_INFINITY} or
113      * {@code Double.NaN} if it is not defined)
114      */
115     double getNumericalVariance();
116 
117     /**
118      * Access the lower bound of the support. This method must return the same
119      * value as {@code inverseCumulativeProbability(0)}. In other words, this
120      * method must return
121      * <p><code>inf {x in Z | P(X <= x) > 0}</code>.</p>
122      *
123      * @return lower bound of the support ({@code Integer.MIN_VALUE}
124      * for negative infinity)
125      */
126     int getSupportLowerBound();
127 
128     /**
129      * Access the upper bound of the support. This method must return the same
130      * value as {@code inverseCumulativeProbability(1)}. In other words, this
131      * method must return
132      * <p><code>inf {x in R | P(X <= x) = 1}</code>.</p>
133      *
134      * @return upper bound of the support ({@code Integer.MAX_VALUE}
135      * for positive infinity)
136      */
137     int getSupportUpperBound();
138 
139     /**
140      * Use this method to get information about whether the support is
141      * connected, i.e. whether all integers between the lower and upper bound of
142      * the support are included in the support.
143      *
144      * @return whether the support is connected or not
145      */
146     boolean isSupportConnected();
147 
148     /**
149      * Creates a sampler.
150      *
151      * @param rng Generator of uniformly distributed numbers.
152      * @return a sampler that produces random numbers according this
153      * distribution.
154      */
155     Sampler createSampler(UniformRandomProvider rng);
156 
157     /**
158      * Sampling functionality.
159      */
160     interface Sampler {
161         /**
162          * Generates a random value sampled from this distribution.
163          *
164          * @return a random value.
165          */
166         int sample();
167     }
168 }
