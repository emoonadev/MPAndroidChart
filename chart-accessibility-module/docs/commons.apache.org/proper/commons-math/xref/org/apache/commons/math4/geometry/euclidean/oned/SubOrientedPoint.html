17  package org.apache.commons.math4.geometry.euclidean.oned;
18  
19  import org.apache.commons.math4.geometry.partitioning.AbstractSubHyperplane;
20  import org.apache.commons.math4.geometry.partitioning.Hyperplane;
21  import org.apache.commons.math4.geometry.partitioning.Region;
22  
23  /** This class represents sub-hyperplane for {@link OrientedPoint}.
24   * <p>An hyperplane in 1D is a simple point, its orientation being a
25   * boolean.</p>
26   * <p>Instances of this class are guaranteed to be immutable.</p>
27   * @since 3.0
28   */
29  public class SubOrientedPoint extends AbstractSubHyperplane<Euclidean1D, Euclidean1D> {
30  
31      /** Simple constructor.
32       * @param hyperplane underlying hyperplane
33       * @param remainingRegion remaining region of the hyperplane
34       */
35      public SubOrientedPoint(final Hyperplane<Euclidean1D> hyperplane,
36                              final Region<Euclidean1D> remainingRegion) {
37          super(hyperplane, remainingRegion);
38      }
39  
40      /** {@inheritDoc} */
41      @Override
42      public double getSize() {
43          return 0;
44      }
45  
46      /** {@inheritDoc} */
47      @Override
48      public boolean isEmpty() {
49          return false;
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      protected AbstractSubHyperplane<Euclidean1D, Euclidean1D> buildNew(final Hyperplane<Euclidean1D> hyperplane,
55                                                                         final Region<Euclidean1D> remainingRegion) {
56          return new SubOrientedPoint(hyperplane, remainingRegion);
57      }
58  
59      /** {@inheritDoc} */
60      @Override
61      public SplitSubHyperplane<Euclidean1D> split(final Hyperplane<Euclidean1D> hyperplane) {
62          final double global = hyperplane.getOffset(((OrientedPoint) getHyperplane()).getLocation());
63          if (global < -1.0e-10) {
64              return new SplitSubHyperplane<Euclidean1D>(null, this);
65          } else if (global > 1.0e-10) {
66              return new SplitSubHyperplane<Euclidean1D>(this, null);
67          } else {
68              return new SplitSubHyperplane<Euclidean1D>(null, null);
69          }
70      }
71  
72  }
