17  package org.apache.commons.math4.geometry.partitioning;
18  
19  import org.apache.commons.math4.geometry.Space;
20  
21  /** Utility class checking if inside nodes can be found
22   * on the plus and minus sides of an hyperplane.
23   * @param <S> Type of the space.
24   * @since 3.4
25   */
26  class InsideFinder<S extends Space> {
27  
28      /** Region on which to operate. */
29      private final Region<S> region;
30  
31      /** Indicator of inside leaf nodes found on the plus side. */
32      private boolean plusFound;
33  
34      /** Indicator of inside leaf nodes found on the plus side. */
35      private boolean minusFound;
36  
37      /** Simple constructor.
38       * @param region region on which to operate
39       */
40      InsideFinder(final Region<S> region) {
41          this.region = region;
42          plusFound  = false;
43          minusFound = false;
44      }
45  
46      /** Search recursively for inside leaf nodes on each side of the given hyperplane.
47  
48       * <p>The algorithm used here is directly derived from the one
49       * described in section III (<i>Binary Partitioning of a BSP
50       * Tree</i>) of the Bruce Naylor, John Amanatides and William
51       * Thibault paper <a
52       * href="http://www.cs.yorku.ca/~amana/research/bsptSetOp.pdf">Merging
53       * BSP Trees Yields Polyhedral Set Operations</a> Proc. Siggraph
54       * '90, Computer Graphics 24(4), August 1990, pp 115-124, published
55       * by the Association for Computing Machinery (ACM)..</p>
56  
57       * @param node current BSP tree node
58       * @param sub sub-hyperplane
59       */
60      public void recurseSides(final BSPTree<S> node, final SubHyperplane<S> sub) {
61  
62          if (node.getCut() == null) {
63              if ((Boolean) node.getAttribute()) {
64                  // this is an inside cell expanding across the hyperplane
65                  plusFound  = true;
66                  minusFound = true;
67              }
68              return;
69          }
70  
71          final Hyperplane<S> hyperplane = node.getCut().getHyperplane();
72          final SubHyperplane.SplitSubHyperplane<S> split = sub.split(hyperplane);
73          switch (split.getSide()) {
74          case PLUS :
75              // the sub-hyperplane is entirely in the plus sub-tree
76              if (node.getCut().split(sub.getHyperplane()).getSide() == Side.PLUS) {
77                  if (!region.isEmpty(node.getMinus())) {
78                      plusFound  = true;
79                  }
80              } else {
81                  if (!region.isEmpty(node.getMinus())) {
82                      minusFound = true;
83                  }
84              }
85              if (!(plusFound && minusFound)) {
86                  recurseSides(node.getPlus(), sub);
87              }
88              break;
89          case MINUS :
90              // the sub-hyperplane is entirely in the minus sub-tree
91              if (node.getCut().split(sub.getHyperplane()).getSide() == Side.PLUS) {
92                  if (!region.isEmpty(node.getPlus())) {
93                      plusFound  = true;
94                  }
95              } else {
96                  if (!region.isEmpty(node.getPlus())) {
97                      minusFound = true;
98                  }
99              }
100             if (!(plusFound && minusFound)) {
101                 recurseSides(node.getMinus(), sub);
102             }
103             break;
104         case BOTH :
105             // the sub-hyperplane extends in both sub-trees
106 
107             // explore first the plus sub-tree
108             recurseSides(node.getPlus(), split.getPlus());
109 
110             // if needed, explore the minus sub-tree
111             if (!(plusFound && minusFound)) {
112                 recurseSides(node.getMinus(), split.getMinus());
113             }
114             break;
115         default :
116             // the sub-hyperplane and the cut sub-hyperplane share the same hyperplane
117             if (node.getCut().getHyperplane().sameOrientationAs(sub.getHyperplane())) {
118                 if ((node.getPlus().getCut() != null) || ((Boolean) node.getPlus().getAttribute())) {
119                     plusFound  = true;
120                 }
121                 if ((node.getMinus().getCut() != null) || ((Boolean) node.getMinus().getAttribute())) {
122                     minusFound = true;
123                 }
124             } else {
125                 if ((node.getPlus().getCut() != null) || ((Boolean) node.getPlus().getAttribute())) {
126                     minusFound = true;
127                 }
128                 if ((node.getMinus().getCut() != null) || ((Boolean) node.getMinus().getAttribute())) {
129                     plusFound  = true;
130                 }
131             }
132         }
133 
134     }
135 
136     /** Check if inside leaf nodes have been found on the plus side.
137      * @return true if inside leaf nodes have been found on the plus side
138      */
139     public boolean plusFound() {
140         return plusFound;
141     }
142 
143     /** Check if inside leaf nodes have been found on the minus side.
144      * @return true if inside leaf nodes have been found on the minus side
145      */
146     public boolean minusFound() {
147         return minusFound;
148     }
149 
150 }
