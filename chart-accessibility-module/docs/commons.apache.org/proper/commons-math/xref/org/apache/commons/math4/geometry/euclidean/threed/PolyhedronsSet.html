57      /** Build a polyhedrons set from a BSP tree.
58       * <p>The leaf nodes of the BSP tree <em>must</em> have a
59       * {@code Boolean} attribute representing the inside status of
60       * the corresponding cell (true for inside cells, false for outside
61       * cells). In order to avoid building too many small objects, it is
62       * recommended to use the predefined constants
63       * {@code Boolean.TRUE} and {@code Boolean.FALSE}</p>
64       * <p>
65       * This constructor is aimed at expert use, as building the tree may
66       * be a difficult task. It is not intended for general use and for
67       * performances reasons does not check thoroughly its input, as this would
68       * require walking the full tree each time. Failing to provide a tree with
69       * the proper attributes, <em>will</em> therefore generate problems like
70       * {@link NullPointerException} or {@link ClassCastException} only later on.
71       * This limitation is known and explains why this constructor is for expert
72       * use only. The caller does have the responsibility to provided correct arguments.
73       * </p>
74       * @param tree inside/outside BSP tree representing the region
75       * @param tolerance tolerance below which points are considered identical
76       * @since 3.3
77       */
78      public PolyhedronsSet(final BSPTree<Euclidean3D> tree, final double tolerance) {
79          super(tree, tolerance);
80      }

82      /** Build a polyhedrons set from a Boundary REPresentation (B-rep) specified by sub-hyperplanes.
83       * <p>The boundary is provided as a collection of {@link
84       * SubHyperplane sub-hyperplanes}. Each sub-hyperplane has the
85       * interior part of the region on its minus side and the exterior on
86       * its plus side.</p>
87       * <p>The boundary elements can be in any order, and can form
88       * several non-connected sets (like for example polyhedrons with holes
89       * or a set of disjoint polyhedrons considered as a whole). In
90       * fact, the elements do not even need to be connected together
91       * (their topological connections are not used here). However, if the
92       * boundary does not really separate an inside open from an outside
93       * open (open having here its topological meaning), then subsequent
94       * calls to the {@link Region#checkPoint(Point) checkPoint} method will
95       * not be meaningful anymore.</p>
96       * <p>If the boundary is empty, the region will represent the whole
97       * space.</p>
98       * @param boundary collection of boundary elements, as a
99       * collection of {@link SubHyperplane SubHyperplane} objects
100      * @param tolerance tolerance below which points are considered identical
101      * @since 3.3
102      */
103     public PolyhedronsSet(final Collection<SubHyperplane<Euclidean3D>> boundary,
104                           final double tolerance) {
105         super(boundary, tolerance);
106     }

108     /** Build a polyhedrons set from a Boundary REPresentation (B-rep) specified by connected vertices.
109      * <p>
110      * The boundary is provided as a list of vertices and a list of facets.
111      * Each facet is specified as an integer array containing the arrays vertices
112      * indices in the vertices list. Each facet normal is oriented by right hand
113      * rule to the facet vertices list.
114      * </p>
115      * <p>
116      * Some basic sanity checks are performed but not everything is thoroughly
117      * assessed, so it remains under caller responsibility to ensure the vertices
118      * and facets are consistent and properly define a polyhedrons set.
119      * </p>
120      * @param vertices list of polyhedrons set vertices
121      * @param facets list of facets, as vertices indices in the vertices list
122      * @param tolerance tolerance below which points are considered identical
123      * @exception MathIllegalArgumentException if some basic sanity checks fail
124      * @since 3.5
125      */
126     public PolyhedronsSet(final List<Vector3D> vertices, final List<int[]> facets,
127                           final double tolerance) {
128         super(buildBoundary(vertices, facets, tolerance), tolerance);
129     }

148     /** Build a parallellepipedic box boundary.
149      * @param xMin low bound along the x direction
150      * @param xMax high bound along the x direction
151      * @param yMin low bound along the y direction
152      * @param yMax high bound along the y direction
153      * @param zMin low bound along the z direction
154      * @param zMax high bound along the z direction
155      * @param tolerance tolerance below which points are considered identical
156      * @return boundary tree
157      * @since 3.3
158      */
159     private static BSPTree<Euclidean3D> buildBoundary(final double xMin, final double xMax,
160                                                       final double yMin, final double yMax,
161                                                       final double zMin, final double zMax,
162                                                       final double tolerance) {
163         if ((xMin >= xMax - tolerance) || (yMin >= yMax - tolerance) || (zMin >= zMax - tolerance)) {
164             // too thin box, build an empty polygons set
165             return new BSPTree<Euclidean3D>(Boolean.FALSE);
166         }
167         final Plane pxMin = new Plane(new Vector3D(xMin, 0,    0),   Vector3D.MINUS_I, tolerance);
168         final Plane pxMax = new Plane(new Vector3D(xMax, 0,    0),   Vector3D.PLUS_I,  tolerance);
169         final Plane pyMin = new Plane(new Vector3D(0,    yMin, 0),   Vector3D.MINUS_J, tolerance);
170         final Plane pyMax = new Plane(new Vector3D(0,    yMax, 0),   Vector3D.PLUS_J,  tolerance);
171         final Plane pzMin = new Plane(new Vector3D(0,    0,   zMin), Vector3D.MINUS_K, tolerance);
172         final Plane pzMax = new Plane(new Vector3D(0,    0,   zMax), Vector3D.PLUS_K,  tolerance);
173         final Region<Euclidean3D> boundary =
174         new RegionFactory<Euclidean3D>().buildConvex(pxMin, pxMax, pyMin, pyMax, pzMin, pzMax);
175         return boundary.getTree(false);
176     }

178     /** Build boundary from vertices and facets.
179      * @param vertices list of polyhedrons set vertices
180      * @param facets list of facets, as vertices indices in the vertices list
181      * @param tolerance tolerance below which points are considered identical
182      * @return boundary as a list of sub-hyperplanes
183      * @exception MathIllegalArgumentException if some basic sanity checks fail
184      * @since 3.5
185      */
186     private static List<SubHyperplane<Euclidean3D>> buildBoundary(final List<Vector3D> vertices,
187                                                                   final List<int[]> facets,
188                                                                   final double tolerance) {
189 
190         // check vertices distances
191         for (int i = 0; i < vertices.size() - 1; ++i) {
192             final Vector3D vi = vertices.get(i);
193             for (int j = i + 1; j < vertices.size(); ++j) {
194                 if (Vector3D.distance(vi, vertices.get(j)) <= tolerance) {
195                     throw new MathIllegalArgumentException(LocalizedFormats.CLOSE_VERTICES,
196                                                            vi.getX(), vi.getY(), vi.getZ());
197                 }
198             }
199         }
200 
201         // find how vertices are referenced by facets
202         final int[][] references = findReferences(vertices, facets);
203 
204         // find how vertices are linked together by edges along the facets they belong to
205         final int[][] successors = successors(vertices, facets, references);
206 
207         // check edges orientations
208         for (int vA = 0; vA < vertices.size(); ++vA) {
209             for (final int vB : successors[vA]) {
210 
211                 if (vB >= 0) {
212                     // when facets are properly oriented, if vB is the successor of vA on facet f1,
213                     // then there must be an adjacent facet f2 where vA is the successor of vB
214                     boolean found = false;
215                     for (final int v : successors[vB]) {
216                         found = found || (v == vA);
217                     }
218                     if (!found) {
219                         final Vector3D start = vertices.get(vA);
220                         final Vector3D end   = vertices.get(vB);
221                         throw new MathIllegalArgumentException(LocalizedFormats.EDGE_CONNECTED_TO_ONE_FACET,
222                                                                start.getX(), start.getY(), start.getZ(),
223                                                                end.getX(),   end.getY(),   end.getZ());
224                     }
225                 }
226             }
227         }
228 
229         final List<SubHyperplane<Euclidean3D>> boundary = new ArrayList<SubHyperplane<Euclidean3D>>();
230 
231         for (final int[] facet : facets) {
232 
233             // define facet plane from the first 3 points
234             Plane plane = new Plane(vertices.get(facet[0]), vertices.get(facet[1]), vertices.get(facet[2]),
235                                     tolerance);
236 
237             // check all points are in the plane
238             final Vector2D[] two2Points = new Vector2D[facet.length];
239             for (int i = 0 ; i < facet.length; ++i) {
240                 final Vector3D v = vertices.get(facet[i]);
241                 if (!plane.contains(v)) {
242                     throw new MathIllegalArgumentException(LocalizedFormats.OUT_OF_PLANE,
243                                                            v.getX(), v.getY(), v.getZ());
244                 }
245                 two2Points[i] = plane.toSubSpace(v);
246             }
247 
248             // create the polygonal facet
249             boundary.add(new SubPlane(plane, new PolygonsSet(tolerance, two2Points)));
250 
251         }
252 
253         return boundary;
254 
255     }

257     /** Find the facets that reference each edges.
258      * @param vertices list of polyhedrons set vertices
259      * @param facets list of facets, as vertices indices in the vertices list
260      * @return references array such that r[v][k] = f for some k if facet f contains vertex v
261      * @exception MathIllegalArgumentException if some facets have fewer than 3 vertices
262      * @since 3.5
263      */
264     private static int[][] findReferences(final List<Vector3D> vertices, final List<int[]> facets) {
265 
266         // find the maximum number of facets a vertex belongs to
267         final int[] nbFacets = new int[vertices.size()];
268         int maxFacets  = 0;
269         for (final int[] facet : facets) {
270             if (facet.length < 3) {
271                 throw new NumberIsTooSmallException(LocalizedFormats.WRONG_NUMBER_OF_POINTS,
272                                                     3, facet.length, true);
273             }
274             for (final int index : facet) {
275                 maxFacets = FastMath.max(maxFacets, ++nbFacets[index]);
276             }
277         }
278 
279         // set up the references array
280         final int[][] references = new int[vertices.size()][maxFacets];
281         for (int[] r : references) {
282             Arrays.fill(r, -1);
283         }
284         for (int f = 0; f < facets.size(); ++f) {
285             for (final int v : facets.get(f)) {
286                 // vertex v is referenced by facet f
287                 int k = 0;
288                 while (k < maxFacets && references[v][k] >= 0) {
289                     ++k;
290                 }
291                 references[v][k] = f;
292             }
293         }
294 
295         return references;
296 
297     }

299     /** Find the successors of all vertices among all facets they belong to.
300      * @param vertices list of polyhedrons set vertices
301      * @param facets list of facets, as vertices indices in the vertices list
302      * @param references facets references array
303      * @return indices of vertices that follow vertex v in some facet (the array
304      * may contain extra entries at the end, set to negative indices)
305      * @exception MathIllegalArgumentException if the same vertex appears more than
306      * once in the successors list (which means one facet orientation is wrong)
307      * @since 3.5
308      */
309     private static int[][] successors(final List<Vector3D> vertices, final List<int[]> facets,
310                                       final int[][] references) {
311 
312         // create an array large enough
313         final int[][] successors = new int[vertices.size()][references[0].length];
314         for (final int[] s : successors) {
315             Arrays.fill(s, -1);
316         }
317 
318         for (int v = 0; v < vertices.size(); ++v) {
319             for (int k = 0; k < successors[v].length && references[v][k] >= 0; ++k) {
320 
321                 // look for vertex v
322                 final int[] facet = facets.get(references[v][k]);
323                 int i = 0;
324                 while (i < facet.length && facet[i] != v) {
325                     ++i;
326                 }
327 
328                 // we have found vertex v, we deduce its successor on current facet
329                 successors[v][k] = facet[(i + 1) % facet.length];
330                 for (int l = 0; l < k; ++l) {
331                     if (successors[v][l] == successors[v][k]) {
332                         final Vector3D start = vertices.get(v);
333                         final Vector3D end   = vertices.get(successors[v][k]);
334                         throw new MathIllegalArgumentException(LocalizedFormats.FACET_ORIENTATION_MISMATCH,
335                                                                start.getX(), start.getY(), start.getZ(),
336                                                                end.getX(),   end.getY(),   end.getZ());
337                     }
338                 }
339 
340             }
341         }
342 
343         return successors;
344 
345     }

353     /** {@inheritDoc} */
354     @Override
355     protected void computeGeometricalProperties() {
356 
357         // compute the contribution of all boundary facets
358         getTree(true).visit(new FacetsContributionVisitor());
359 
360         if (getSize() < 0) {
361             // the polyhedrons set as a finite outside
362             // surrounded by an infinite inside
363             setSize(Double.POSITIVE_INFINITY);
364             setBarycenter((Point<Euclidean3D>) Vector3D.NaN);
365         } else {
366             // the polyhedrons set is finite, apply the remaining scaling factors
367             setSize(getSize() / 3.0);
368             setBarycenter((Point<Euclidean3D>) new Vector3D(1.0 / (4 * getSize()), (Vector3D) getBarycenter()));
369         }
370 
371     }

373     /** Visitor computing geometrical properties. */
374     private class FacetsContributionVisitor implements BSPTreeVisitor<Euclidean3D> {
375 
376         /** Simple constructor. */
377         FacetsContributionVisitor() {
378             setSize(0);
379             setBarycenter((Point<Euclidean3D>) new Vector3D(0, 0, 0));
380         }
381 
382         /** {@inheritDoc} */
383         @Override
384         public Order visitOrder(final BSPTree<Euclidean3D> node) {
385             return Order.MINUS_SUB_PLUS;
386         }
387 
388         /** {@inheritDoc} */
389         @Override
390         public void visitInternalNode(final BSPTree<Euclidean3D> node) {
391             @SuppressWarnings("unchecked")
392             final BoundaryAttribute<Euclidean3D> attribute =
393                 (BoundaryAttribute<Euclidean3D>) node.getAttribute();
394             if (attribute.getPlusOutside() != null) {
395                 addContribution(attribute.getPlusOutside(), false);
396             }
397             if (attribute.getPlusInside() != null) {
398                 addContribution(attribute.getPlusInside(), true);
399             }
400         }
401 
402         /** {@inheritDoc} */
403         @Override
404         public void visitLeafNode(final BSPTree<Euclidean3D> node) {
405         }

407         /** Add he contribution of a boundary facet.
408          * @param facet boundary facet
409          * @param reversed if true, the facet has the inside on its plus side
410          */
411         private void addContribution(final SubHyperplane<Euclidean3D> facet, final boolean reversed) {
412 
413             final Region<Euclidean2D> polygon = ((SubPlane) facet).getRemainingRegion();
414             final double area    = polygon.getSize();
415 
416             if (Double.isInfinite(area)) {
417                 setSize(Double.POSITIVE_INFINITY);
418                 setBarycenter((Point<Euclidean3D>) Vector3D.NaN);
419             } else {
420 
421                 final Plane    plane  = (Plane) facet.getHyperplane();
422                 final Vector3D facetB = plane.toSpace(polygon.getBarycenter());
423                 double   scaled = area * facetB.dotProduct(plane.getNormal());
424                 if (reversed) {
425                     scaled = -scaled;
426                 }
427 
428                 setSize(getSize() + scaled);
429                 setBarycenter((Point<Euclidean3D>) new Vector3D(1.0, (Vector3D) getBarycenter(), scaled, facetB));
430 
431             }
432 
433         }
434 
435     }

437     /** Get the first sub-hyperplane crossed by a semi-infinite line.
438      * @param point start point of the part of the line considered
439      * @param line line to consider (contains point)
440      * @return the first sub-hyperplane crossed by the line after the
441      * given point, or null if the line does not intersect any
442      * sub-hyperplane
443      */
444     public SubHyperplane<Euclidean3D> firstIntersection(final Vector3D point, final Line line) {
445         return recurseFirstIntersection(getTree(true), point, line);
446     }
447 
448     /** Get the first sub-hyperplane crossed by a semi-infinite line.
449      * @param node current node
450      * @param point start point of the part of the line considered
451      * @param line line to consider (contains point)
452      * @return the first sub-hyperplane crossed by the line after the
453      * given point, or null if the line does not intersect any
454      * sub-hyperplane
455      */
456     private SubHyperplane<Euclidean3D> recurseFirstIntersection(final BSPTree<Euclidean3D> node,
457                                                                 final Vector3D point,
458                                                                 final Line line) {
459 
460         final SubHyperplane<Euclidean3D> cut = node.getCut();
461         if (cut == null) {
462             return null;
463         }
464         final BSPTree<Euclidean3D> minus = node.getMinus();
465         final BSPTree<Euclidean3D> plus  = node.getPlus();
466         final Plane                plane = (Plane) cut.getHyperplane();
467 
468         // establish search order
469         final double offset = plane.getOffset((Point<Euclidean3D>) point);
470         final boolean in    = FastMath.abs(offset) < getTolerance();
471         final BSPTree<Euclidean3D> near;
472         final BSPTree<Euclidean3D> far;
473         if (offset < 0) {
474             near = minus;
475             far  = plus;
476         } else {
477             near = plus;
478             far  = minus;
479         }
480 
481         if (in) {
482             // search in the cut hyperplane
483             final SubHyperplane<Euclidean3D> facet = boundaryFacet(point, node);
484             if (facet != null) {
485                 return facet;
486             }
487         }
488 
489         // search in the near branch
490         final SubHyperplane<Euclidean3D> crossed = recurseFirstIntersection(near, point, line);
491         if (crossed != null) {
492             return crossed;
493         }
494 
495         if (!in) {
496             // search in the cut hyperplane
497             final Vector3D hit3D = plane.intersection(line);
498             if (hit3D != null && line.getAbscissa(hit3D) > line.getAbscissa(point)) {
499                 final SubHyperplane<Euclidean3D> facet = boundaryFacet(hit3D, node);
500                 if (facet != null) {
501                     return facet;
502                 }
503             }
504         }
505 
506         // search in the far branch
507         return recurseFirstIntersection(far, point, line);
508 
509     }

511     /** Check if a point belongs to the boundary part of a node.
512      * @param point point to check
513      * @param node node containing the boundary facet to check
514      * @return the boundary facet this points belongs to (or null if it
515      * does not belong to any boundary facet)
516      */
517     private SubHyperplane<Euclidean3D> boundaryFacet(final Vector3D point,
518                                                      final BSPTree<Euclidean3D> node) {
519         final Vector2D point2D = ((Plane) node.getCut().getHyperplane()).toSubSpace((Point<Euclidean3D>) point);
520         @SuppressWarnings("unchecked")
521         final BoundaryAttribute<Euclidean3D> attribute =
522             (BoundaryAttribute<Euclidean3D>) node.getAttribute();
523         if ((attribute.getPlusOutside() != null) &&
524             (((SubPlane) attribute.getPlusOutside()).getRemainingRegion().checkPoint(point2D) == Location.INSIDE)) {
525             return attribute.getPlusOutside();
526         }
527         if ((attribute.getPlusInside() != null) &&
528             (((SubPlane) attribute.getPlusInside()).getRemainingRegion().checkPoint(point2D) == Location.INSIDE)) {
529             return attribute.getPlusInside();
530         }
531         return null;
532     }

544     /** 3D rotation as a Transform. */
545     private static class RotationTransform implements Transform<Euclidean3D, Euclidean2D> {
546 
547         /** Center point of the rotation. */
548         private final Vector3D   center;
549 
550         /** Vectorial rotation. */
551         private final Rotation   rotation;
552 
553         /** Cached original hyperplane. */
554         private Plane cachedOriginal;
555 
556         /** Cached 2D transform valid inside the cached original hyperplane. */
557         private Transform<Euclidean2D, Euclidean1D>  cachedTransform;
558 
559         /** Build a rotation transform.
560          * @param center center point of the rotation
561          * @param rotation vectorial rotation
562          */
563         RotationTransform(final Vector3D center, final Rotation rotation) {
564             this.center   = center;
565             this.rotation = rotation;
566         }
567 
568         /** {@inheritDoc} */
569         @Override
570         public Vector3D apply(final Point<Euclidean3D> point) {
571             final Vector3D delta = ((Vector3D) point).subtract(center);
572             return new Vector3D(1.0, center, 1.0, rotation.applyTo(delta));
573         }
574 
575         /** {@inheritDoc} */
576         @Override
577         public Plane apply(final Hyperplane<Euclidean3D> hyperplane) {
578             return ((Plane) hyperplane).rotate(center, rotation);
579         }
580 
581         /** {@inheritDoc} */
582         @Override
583         public SubHyperplane<Euclidean2D> apply(final SubHyperplane<Euclidean2D> sub,
584                                                 final Hyperplane<Euclidean3D> original,
585                                                 final Hyperplane<Euclidean3D> transformed) {
586             if (original != cachedOriginal) {
587                 // we have changed hyperplane, reset the in-hyperplane transform
588 
589                 final Plane    oPlane = (Plane) original;
590                 final Plane    tPlane = (Plane) transformed;
591                 final Vector3D p00    = oPlane.getOrigin();
592                 final Vector3D p10    = oPlane.toSpace((Point<Euclidean2D>) new Vector2D(1.0, 0.0));
593                 final Vector3D p01    = oPlane.toSpace((Point<Euclidean2D>) new Vector2D(0.0, 1.0));
594                 final Vector2D tP00   = tPlane.toSubSpace((Point<Euclidean3D>) apply(p00));
595                 final Vector2D tP10   = tPlane.toSubSpace((Point<Euclidean3D>) apply(p10));
596                 final Vector2D tP01   = tPlane.toSubSpace((Point<Euclidean3D>) apply(p01));
597 
598                 cachedOriginal  = (Plane) original;
599                 cachedTransform =
600                         org.apache.commons.math4.geometry.euclidean.twod.Line.getTransform(tP10.getX() - tP00.getX(),
601                                                                                            tP10.getY() - tP00.getY(),
602                                                                                            tP01.getX() - tP00.getX(),
603                                                                                            tP01.getY() - tP00.getY(),
604                                                                                            tP00.getX(),
605                                                                                            tP00.getY());
606 
607             }
608             return ((SubLine) sub).applyTransform(cachedTransform);
609         }
610 
611     }
612 
613     /** Translate the region by the specified amount.
614      * <p>The instance is not modified, a new instance is created.</p>
615      * @param translation translation to apply
616      * @return a new instance representing the translated region
617      */
618     public PolyhedronsSet translate(final Vector3D translation) {
619         return (PolyhedronsSet) applyTransform(new TranslationTransform(translation));
620     }
621 
622     /** 3D translation as a transform. */
623     private static class TranslationTransform implements Transform<Euclidean3D, Euclidean2D> {
624 
625         /** Translation vector. */
626         private final Vector3D   translation;
627 
628         /** Cached original hyperplane. */
629         private Plane cachedOriginal;
630 
631         /** Cached 2D transform valid inside the cached original hyperplane. */
632         private Transform<Euclidean2D, Euclidean1D>  cachedTransform;
633 
634         /** Build a translation transform.
635          * @param translation translation vector
636          */
637         TranslationTransform(final Vector3D translation) {
638             this.translation = translation;
639         }
640 
641         /** {@inheritDoc} */
642         @Override
643         public Vector3D apply(final Point<Euclidean3D> point) {
644             return new Vector3D(1.0, (Vector3D) point, 1.0, translation);
645         }
646 
647         /** {@inheritDoc} */
648         @Override
649         public Plane apply(final Hyperplane<Euclidean3D> hyperplane) {
650             return ((Plane) hyperplane).translate(translation);
651         }
652 
653         /** {@inheritDoc} */
654         @Override
655         public SubHyperplane<Euclidean2D> apply(final SubHyperplane<Euclidean2D> sub,
656                                                 final Hyperplane<Euclidean3D> original,
657                                                 final Hyperplane<Euclidean3D> transformed) {
658             if (original != cachedOriginal) {
659                 // we have changed hyperplane, reset the in-hyperplane transform
660 
661                 final Plane   oPlane = (Plane) original;
662                 final Plane   tPlane = (Plane) transformed;
663                 final Vector2D shift  = tPlane.toSubSpace((Point<Euclidean3D>) apply(oPlane.getOrigin()));
664 
665                 cachedOriginal  = (Plane) original;
666                 cachedTransform =
667                         org.apache.commons.math4.geometry.euclidean.twod.Line.getTransform(1, 0, 0, 1,
668                                                                                            shift.getX(),
669                                                                                            shift.getY());
670 
671             }
672 
673             return ((SubLine) sub).applyTransform(cachedTransform);
674 
675         }
676 
677     }
