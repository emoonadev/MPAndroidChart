17  package org.apache.commons.math4.geometry.euclidean.twod;
18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.util.FastMath;
21  
22  /** Simple container for a two-points segment.
23   * @since 3.0
24   */
25  public class Segment {
26  
27      /** Start point of the segment. */
28      private final Vector2D start;
29  
30      /** End point of the segment. */
31      private final Vector2D end;
32  
33      /** Line containing the segment. */
34      private final Line     line;
35  
36      /** Build a segment.
37       * @param start start point of the segment
38       * @param end end point of the segment
39       * @param line line containing the segment
40       */
41      public Segment(final Vector2D start, final Vector2D end, final Line line) {
42          this.start  = start;
43          this.end    = end;
44          this.line   = line;
45      }
46  
47      /** Get the start point of the segment.
48       * @return start point of the segment
49       */
50      public Vector2D getStart() {
51          return start;
52      }
53  
54      /** Get the end point of the segment.
55       * @return end point of the segment
56       */
57      public Vector2D getEnd() {
58          return end;
59      }
60  
61      /** Get the line containing the segment.
62       * @return line containing the segment
63       */
64      public Line getLine() {
65          return line;
66      }
67  
68      /** Calculates the shortest distance from a point to this line segment.
69       * <p>
70       * If the perpendicular extension from the point to the line does not
71       * cross in the bounds of the line segment, the shortest distance to
72       * the two end points will be returned.
73       * </p>
74       *
75       * Algorithm adapted from:
76       * <a href="http://www.codeguru.com/forum/printthread.php?s=cc8cf0596231f9a7dba4da6e77c29db3&t=194400&pp=15&page=1">
77       * Thread @ Codeguru</a>
78       *
79       * @param p to check
80       * @return distance between the instance and the point
81       * @since 3.1
82       */
83      public double distance(final Vector2D p) {
84          final double deltaX = end.getX() - start.getX();
85          final double deltaY = end.getY() - start.getY();
86  
87          final double r = ((p.getX() - start.getX()) * deltaX + (p.getY() - start.getY()) * deltaY) /
88                           (deltaX * deltaX + deltaY * deltaY);
89  
90          // r == 0 => P = startPt
91          // r == 1 => P = endPt
92          // r < 0 => P is on the backward extension of the segment
93          // r > 1 => P is on the forward extension of the segment
94          // 0 < r < 1 => P is on the segment
95  
96          // if point isn't on the line segment, just return the shortest distance to the end points
97          if (r < 0 || r > 1) {
98              final double dist1 = getStart().distance((Point<Euclidean2D>) p);
99              final double dist2 = getEnd().distance((Point<Euclidean2D>) p);
100 
101             return FastMath.min(dist1, dist2);
102         }
103         else {
104             // find point on line and see if it is in the line segment
105             final double px = start.getX() + r * deltaX;
106             final double py = start.getY() + r * deltaY;
107 
108             final Vector2D interPt = new Vector2D(px, py);
109             return interPt.distance((Point<Euclidean2D>) p);
110         }
111     }
112 }
