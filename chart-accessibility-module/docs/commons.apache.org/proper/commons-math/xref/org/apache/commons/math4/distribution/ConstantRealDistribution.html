18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.rng.UniformRandomProvider;
22  
23  /**
24   * Implementation of the constant real distribution.
25   *
26   * @since 3.4
27   */
28  public class ConstantRealDistribution extends AbstractRealDistribution {
29  
30      /** Serialization ID */
31      private static final long serialVersionUID = -4157745166772046273L;
32  
33      /** Constant value of the distribution */
34      private final double value;
35  
36      /**
37       * Create a constant real distribution with the given value.
38       *
39       * @param value the constant value of this distribution
40       */
41      public ConstantRealDistribution(double value) {
42          this.value = value;
43      }
44  
45      /** {@inheritDoc} */
46      @Override
47      public double density(double x) {
48          return x == value ? 1 : 0;
49      }
50  
51      /** {@inheritDoc} */
52      @Override
53      public double cumulativeProbability(double x)  {
54          return x < value ? 0 : 1;
55      }
56  
57      /** {@inheritDoc} */
58      @Override
59      public double inverseCumulativeProbability(final double p)
60              throws OutOfRangeException {
61          if (p < 0.0 || p > 1.0) {
62              throw new OutOfRangeException(p, 0, 1);
63          }
64          return value;
65      }
66  
67      /**
68       * {@inheritDoc}
69       */
70      @Override
71      public double getNumericalMean() {
72          return value;
73      }
74  
75      /**
76       * {@inheritDoc}
77       */
78      @Override
79      public double getNumericalVariance() {
80          return 0;
81      }
82  
83      /**
84       * {@inheritDoc}
85       */
86      @Override
87      public double getSupportLowerBound() {
88          return value;
89      }
90  
91      /**
92       * {@inheritDoc}
93       */
94      @Override
95      public double getSupportUpperBound() {
96          return value;
97      }
98  
99      /**
100      * {@inheritDoc}
101      */
102     @Override
103     public boolean isSupportConnected() {
104         return true;
105     }
106 
107     /**
108      * {@inheritDoc}
109      *
110      * @param rng Not used: distribution contains a single value.
111      * @return the value of the distribution.
112      */
113     @Override
114     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
115         return new RealDistribution.Sampler() {
116             /** {@inheritDoc} */
117             @Override
118             public double sample() {
119                 return value;
120             }
121         };
122     }
123 }
