17  package org.apache.commons.math4.geometry.euclidean.threed;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import org.apache.commons.math4.exception.MathIllegalArgumentException;
23  import org.apache.commons.math4.geometry.Point;
24  import org.apache.commons.math4.geometry.euclidean.oned.Euclidean1D;
25  import org.apache.commons.math4.geometry.euclidean.oned.Interval;
26  import org.apache.commons.math4.geometry.euclidean.oned.IntervalsSet;
27  import org.apache.commons.math4.geometry.euclidean.oned.Vector1D;
28  import org.apache.commons.math4.geometry.partitioning.Region.Location;
29  
30  /** This class represents a subset of a {@link Line}.
31   * @since 3.0
32   */
33  public class SubLine {
34  
35      /** Underlying line. */
36      private final Line line;
37  
38      /** Remaining region of the hyperplane. */
39      private final IntervalsSet remainingRegion;
40  
41      /** Simple constructor.
42       * @param line underlying line
43       * @param remainingRegion remaining region of the line
44       */
45      public SubLine(final Line line, final IntervalsSet remainingRegion) {
46          this.line            = line;
47          this.remainingRegion = remainingRegion;
48      }
49  
50      /** Create a sub-line from two endpoints.
51       * @param start start point
52       * @param end end point
53       * @param tolerance tolerance below which points are considered identical
54       * @exception MathIllegalArgumentException if the points are equal
55       * @since 3.3
56       */
57      public SubLine(final Vector3D start, final Vector3D end, final double tolerance)
58          throws MathIllegalArgumentException {
59          this(new Line(start, end, tolerance), buildIntervalSet(start, end, tolerance));
60      }
61  
62      /** Create a sub-line from a segment.
63       * @param segment single segment forming the sub-line
64       * @exception MathIllegalArgumentException if the segment endpoints are equal
65       */
66      public SubLine(final Segment segment) throws MathIllegalArgumentException {
67          this(segment.getLine(),
68               buildIntervalSet(segment.getStart(), segment.getEnd(), segment.getLine().getTolerance()));
69      }
70  
71      /** Get the endpoints of the sub-line.
72       * <p>
73       * A subline may be any arbitrary number of disjoints segments, so the endpoints
74       * are provided as a list of endpoint pairs. Each element of the list represents
75       * one segment, and each segment contains a start point at index 0 and an end point
76       * at index 1. If the sub-line is unbounded in the negative infinity direction,
77       * the start point of the first segment will have infinite coordinates. If the
78       * sub-line is unbounded in the positive infinity direction, the end point of the
79       * last segment will have infinite coordinates. So a sub-line covering the whole
80       * line will contain just one row and both elements of this row will have infinite
81       * coordinates. If the sub-line is empty, the returned list will contain 0 segments.
82       * </p>
83       * @return list of segments endpoints
84       */
85      public List<Segment> getSegments() {
86  
87          final List<Interval> list = remainingRegion.asList();
88          final List<Segment> segments = new ArrayList<Segment>(list.size());
89  
90          for (final Interval interval : list) {
91              final Vector3D start = line.toSpace((Point<Euclidean1D>) new Vector1D(interval.getInf()));
92              final Vector3D end   = line.toSpace((Point<Euclidean1D>) new Vector1D(interval.getSup()));
93              segments.add(new Segment(start, end, line));
94          }
95  
96          return segments;
97  
98      }
99  
100     /** Get the intersection of the instance and another sub-line.
101      * <p>
102      * This method is related to the {@link Line#intersection(Line)
103      * intersection} method in the {@link Line Line} class, but in addition
104      * to compute the point along infinite lines, it also checks the point
105      * lies on both sub-line ranges.
106      * </p>
107      * @param subLine other sub-line which may intersect instance
108      * @param includeEndPoints if true, endpoints are considered to belong to
109      * instance (i.e. they are closed sets) and may be returned, otherwise endpoints
110      * are considered to not belong to instance (i.e. they are open sets) and intersection
111      * occurring on endpoints lead to null being returned
112      * @return the intersection point if there is one, null if the sub-lines don't intersect
113      */
114     public Vector3D intersection(final SubLine subLine, final boolean includeEndPoints) {
115 
116         // compute the intersection on infinite line
117         Vector3D v1D = line.intersection(subLine.line);
118         if (v1D == null) {
119             return null;
120         }
121 
122         // check location of point with respect to first sub-line
123         Location loc1 = remainingRegion.checkPoint((Point<Euclidean1D>) line.toSubSpace((Point<Euclidean3D>) v1D));
124 
125         // check location of point with respect to second sub-line
126         Location loc2 = subLine.remainingRegion.checkPoint((Point<Euclidean1D>) subLine.line.toSubSpace((Point<Euclidean3D>) v1D));
127 
128         if (includeEndPoints) {
129             return ((loc1 != Location.OUTSIDE) && (loc2 != Location.OUTSIDE)) ? v1D : null;
130         } else {
131             return ((loc1 == Location.INSIDE) && (loc2 == Location.INSIDE)) ? v1D : null;
132         }
133 
134     }
135 
136     /** Build an interval set from two points.
137      * @param start start point
138      * @param end end point
139      * @return an interval set
140      * @param tolerance tolerance below which points are considered identical
141      * @exception MathIllegalArgumentException if the points are equal
142      */
143     private static IntervalsSet buildIntervalSet(final Vector3D start, final Vector3D end, final double tolerance)
144         throws MathIllegalArgumentException {
145         final Line line = new Line(start, end, tolerance);
146         return new IntervalsSet(line.toSubSpace((Point<Euclidean3D>) start).getX(),
147                                 line.toSubSpace((Point<Euclidean3D>) end).getX(),
148                                 tolerance);
149     }
150 
151 }
