91      /**
92       * Create a Chebyshev polynomial of the first kind.
93       * <p><a href="https://en.wikipedia.org/wiki/Chebyshev_polynomials">Chebyshev
94       * polynomials of the first kind</a> are orthogonal polynomials.
95       * They can be defined by the following recurrence relations:</p><p>
96       * \(
97       *    T_0(x) = 1 \\
98       *    T_1(x) = x \\
99       *    T_{k+1}(x) = 2x T_k(x) - T_{k-1}(x)
100      * \)
101      * </p>
102      * @param degree degree of the polynomial
103      * @return Chebyshev polynomial of specified degree
104      */
105     public static PolynomialFunction createChebyshevPolynomial(final int degree) {
106         return buildPolynomial(degree, CHEBYSHEV_COEFFICIENTS,
107                 new RecurrenceCoefficientsGenerator() {
108             /** Fixed recurrence coefficients. */
109             private final BigFraction[] coeffs = { BigFraction.ZERO, BigFraction.TWO, BigFraction.ONE };
110             /** {@inheritDoc} */
111             @Override
112             public BigFraction[] generate(int k) {
113                 return coeffs;
114             }
115         });
116     }

118     /**
119      * Create a Hermite polynomial.
120      * <p><a href="http://mathworld.wolfram.com/HermitePolynomial.html">Hermite
121      * polynomials</a> are orthogonal polynomials.
122      * They can be defined by the following recurrence relations:</p><p>
123      * \(
124      *  H_0(x) = 1 \\
125      *  H_1(x) = 2x \\
126      *  H_{k+1}(x) = 2x H_k(X) - 2k H_{k-1}(x)
127      * \)
128      * </p>
129 
130      * @param degree degree of the polynomial
131      * @return Hermite polynomial of specified degree
132      */
133     public static PolynomialFunction createHermitePolynomial(final int degree) {
134         return buildPolynomial(degree, HERMITE_COEFFICIENTS,
135                 new RecurrenceCoefficientsGenerator() {
136             /** {@inheritDoc} */
137             @Override
138             public BigFraction[] generate(int k) {
139                 return new BigFraction[] {
140                         BigFraction.ZERO,
141                         BigFraction.TWO,
142                         new BigFraction(2 * k)};
143             }
144         });
145     }

147     /**
148      * Create a Laguerre polynomial.
149      * <p><a href="http://mathworld.wolfram.com/LaguerrePolynomial.html">Laguerre
150      * polynomials</a> are orthogonal polynomials.
151      * They can be defined by the following recurrence relations:</p><p>
152      * \(
153      *   L_0(x) = 1 \\
154      *   L_1(x) = 1 - x \\
155      *   (k+1) L_{k+1}(x) = (2k + 1 - x) L_k(x) - k L_{k-1}(x)
156      * \)
157      * </p>
158      * @param degree degree of the polynomial
159      * @return Laguerre polynomial of specified degree
160      */
161     public static PolynomialFunction createLaguerrePolynomial(final int degree) {
162         return buildPolynomial(degree, LAGUERRE_COEFFICIENTS,
163                 new RecurrenceCoefficientsGenerator() {
164             /** {@inheritDoc} */
165             @Override
166             public BigFraction[] generate(int k) {
167                 final int kP1 = k + 1;
168                 return new BigFraction[] {
169                         new BigFraction(2 * k + 1, kP1),
170                         new BigFraction(-1, kP1),
171                         new BigFraction(k, kP1)};
172             }
173         });
174     }

176     /**
177      * Create a Legendre polynomial.
178      * <p><a href="http://mathworld.wolfram.com/LegendrePolynomial.html">Legendre
179      * polynomials</a> are orthogonal polynomials.
180      * They can be defined by the following recurrence relations:</p><p>
181      * \(
182      *   P_0(x) = 1 \\
183      *   P_1(x) = x \\
184      *   (k+1) P_{k+1}(x) = (2k+1) x P_k(x) - k P_{k-1}(x)
185      * \)
186      * </p>
187      * @param degree degree of the polynomial
188      * @return Legendre polynomial of specified degree
189      */
190     public static PolynomialFunction createLegendrePolynomial(final int degree) {
191         return buildPolynomial(degree, LEGENDRE_COEFFICIENTS,
192                                new RecurrenceCoefficientsGenerator() {
193             /** {@inheritDoc} */
194             @Override
195             public BigFraction[] generate(int k) {
196                 final int kP1 = k + 1;
197                 return new BigFraction[] {
198                         BigFraction.ZERO,
199                         new BigFraction(k + kP1, kP1),
200                         new BigFraction(k, kP1)};
201             }
202         });
203     }

205     /**
206      * Create a Jacobi polynomial.
207      * <p><a href="http://mathworld.wolfram.com/JacobiPolynomial.html">Jacobi
208      * polynomials</a> are orthogonal polynomials.
209      * They can be defined by the following recurrence relations:</p><p>
210      * \(
211      *    P_0^{vw}(x) = 1 \\
212      *    P_{-1}^{vw}(x) = 0 \\
213      *    2k(k + v + w)(2k + v + w - 2) P_k^{vw}(x) = \\
214      *    (2k + v + w - 1)[(2k + v + w)(2k + v + w - 2) x + v^2 - w^2] P_{k-1}^{vw}(x) \\
215      *  - 2(k + v - 1)(k + w - 1)(2k + v + w) P_{k-2}^{vw}(x)
216      * \)
217      * </p>
218      * @param degree degree of the polynomial
219      * @param v first exponent
220      * @param w second exponent
221      * @return Jacobi polynomial of specified degree
222      */
223     public static PolynomialFunction createJacobiPolynomial(final int degree, final int v, final int w) {
224 
225         // select the appropriate list
226         final JacobiKey key = new JacobiKey(v, w);
227 
228         if (!JACOBI_COEFFICIENTS.containsKey(key)) {
229 
230             // allocate a new list for v, w
231             final List<BigFraction> list = new ArrayList<BigFraction>();
232             JACOBI_COEFFICIENTS.put(key, list);
233 
234             // Pv,w,0(x) = 1;
235             list.add(BigFraction.ONE);
236 
237             // P1(x) = (v - w) / 2 + (2 + v + w) * X / 2
238             list.add(new BigFraction(v - w, 2));
239             list.add(new BigFraction(2 + v + w, 2));
240 
241         }
242 
243         return buildPolynomial(degree, JACOBI_COEFFICIENTS.get(key),
244                                new RecurrenceCoefficientsGenerator() {
245             /** {@inheritDoc} */
246             @Override
247             public BigFraction[] generate(int k) {
248                 k++;
249                 final int kvw      = k + v + w;
250                 final int twoKvw   = kvw + k;
251                 final int twoKvwM1 = twoKvw - 1;
252                 final int twoKvwM2 = twoKvw - 2;
253                 final int den      = 2 * k *  kvw * twoKvwM2;
254 
255                 return new BigFraction[] {
256                     new BigFraction(twoKvwM1 * (v * v - w * w), den),
257                     new BigFraction(twoKvwM1 * twoKvw * twoKvwM2, den),
258                     new BigFraction(2 * (k + v - 1) * (k + w - 1) * twoKvw, den)
259                 };
260             }
261         });
262 
263     }

265     /** Inner class for Jacobi polynomials keys. */
266     private static class JacobiKey {
267 
268         /** First exponent. */
269         private final int v;
270 
271         /** Second exponent. */
272         private final int w;
273 
274         /** Simple constructor.
275          * @param v first exponent
276          * @param w second exponent
277          */
278         JacobiKey(final int v, final int w) {
279             this.v = v;
280             this.w = w;
281         }
282 
283         /** Get hash code.
284          * @return hash code
285          */
286         @Override
287         public int hashCode() {
288             return (v << 16) ^ w;
289         }
290 
291         /** Check if the instance represent the same key as another instance.
292          * @param key other key
293          * @return true if the instance and the other key refer to the same polynomial
294          */
295         @Override
296         public boolean equals(final Object key) {
297 
298             if ((key == null) || !(key instanceof JacobiKey)) {
299                 return false;
300             }
301 
302             final JacobiKey otherK = (JacobiKey) key;
303             return (v == otherK.v) && (w == otherK.w);
304 
305         }
306     }

308     /**
309      * Compute the coefficients of the polynomial \(P_s(x)\)
310      * whose values at point {@code x} will be the same as the those from the
311      * original polynomial \(P(x)\) when computed at {@code x + shift}.
312      * <p>
313      * More precisely, let \(\Delta = \) {@code shift} and let
314      * \(P_s(x) = P(x + \Delta)\).  The returned array
315      * consists of the coefficients of \(P_s\).  So if \(a_0, ..., a_{n-1}\)
316      * are the coefficients of \(P\), then the returned array
317      * \(b_0, ..., b_{n-1}\) satisfies the identity
318      * \(\sum_{i=0}^{n-1} b_i x^i = \sum_{i=0}^{n-1} a_i (x + \Delta)^i\) for all \(x\).
319      *
320      * @param coefficients Coefficients of the original polynomial.
321      * @param shift Shift value.
322      * @return the coefficients \(b_i\) of the shifted
323      * polynomial.
324      */
325     public static double[] shift(final double[] coefficients,
326                                  final double shift) {
327         final int dp1 = coefficients.length;
328         final double[] newCoefficients = new double[dp1];
329 
330         // Pascal triangle.
331         final int[][] coeff = new int[dp1][dp1];
332         for (int i = 0; i < dp1; i++){
333             for(int j = 0; j <= i; j++){
334                 coeff[i][j] = (int) CombinatoricsUtils.binomialCoefficient(i, j);
335             }
336         }
337 
338         // First polynomial coefficient.
339         for (int i = 0; i < dp1; i++){
340             newCoefficients[0] += coefficients[i] * FastMath.pow(shift, i);
341         }
342 
343         // Superior order.
344         final int d = dp1 - 1;
345         for (int i = 0; i < d; i++) {
346             for (int j = i; j < d; j++){
347                 newCoefficients[i + 1] += coeff[j + 1][j - i] *
348                     coefficients[j + 1] * FastMath.pow(shift, j - i);
349             }
350         }
351 
352         return newCoefficients;
353     }

356     /** Get the coefficients array for a given degree.
357      * @param degree degree of the polynomial
358      * @param coefficients list where the computed coefficients are stored
359      * @param generator recurrence coefficients generator
360      * @return coefficients array
361      */
362     private static PolynomialFunction buildPolynomial(final int degree,
363                                                       final List<BigFraction> coefficients,
364                                                       final RecurrenceCoefficientsGenerator generator) {
365 
366         synchronized (coefficients) {
367             final int maxDegree = (int) FastMath.floor(FastMath.sqrt(2 * coefficients.size())) - 1;
368             if (degree > maxDegree) {
369                 computeUpToDegree(degree, maxDegree, generator, coefficients);
370             }
371         }
372 
373         // coefficient  for polynomial 0 is  l [0]
374         // coefficients for polynomial 1 are l [1] ... l [2] (degrees 0 ... 1)
375         // coefficients for polynomial 2 are l [3] ... l [5] (degrees 0 ... 2)
376         // coefficients for polynomial 3 are l [6] ... l [9] (degrees 0 ... 3)
377         // coefficients for polynomial 4 are l[10] ... l[14] (degrees 0 ... 4)
378         // coefficients for polynomial 5 are l[15] ... l[20] (degrees 0 ... 5)
379         // coefficients for polynomial 6 are l[21] ... l[27] (degrees 0 ... 6)
380         // ...
381         final int start = degree * (degree + 1) / 2;
382 
383         final double[] a = new double[degree + 1];
384         for (int i = 0; i <= degree; ++i) {
385             a[i] = coefficients.get(start + i).doubleValue();
386         }
387 
388         // build the polynomial
389         return new PolynomialFunction(a);
390 
391     }

393     /** Compute polynomial coefficients up to a given degree.
394      * @param degree maximal degree
395      * @param maxDegree current maximal degree
396      * @param generator recurrence coefficients generator
397      * @param coefficients list where the computed coefficients should be appended
398      */
399     private static void computeUpToDegree(final int degree, final int maxDegree,
400                                           final RecurrenceCoefficientsGenerator generator,
401                                           final List<BigFraction> coefficients) {
402 
403         int startK = (maxDegree - 1) * maxDegree / 2;
404         for (int k = maxDegree; k < degree; ++k) {
405 
406             // start indices of two previous polynomials Pk(X) and Pk-1(X)
407             int startKm1 = startK;
408             startK += k;
409 
410             // Pk+1(X) = (a[0] + a[1] X) Pk(X) - a[2] Pk-1(X)
411             BigFraction[] ai = generator.generate(k);
412 
413             BigFraction ck     = coefficients.get(startK);
414             BigFraction ckm1   = coefficients.get(startKm1);
415 
416             // degree 0 coefficient
417             coefficients.add(ck.multiply(ai[0]).subtract(ckm1.multiply(ai[2])));
418 
419             // degree 1 to degree k-1 coefficients
420             for (int i = 1; i < k; ++i) {
421                 final BigFraction ckPrev = ck;
422                 ck     = coefficients.get(startK + i);
423                 ckm1   = coefficients.get(startKm1 + i);
424                 coefficients.add(ck.multiply(ai[0]).add(ckPrev.multiply(ai[1])).subtract(ckm1.multiply(ai[2])));
425             }
426 
427             // degree k coefficient
428             final BigFraction ckPrev = ck;
429             ck = coefficients.get(startK + k);
430             coefficients.add(ck.multiply(ai[0]).add(ckPrev.multiply(ai[1])));
431 
432             // degree k+1 coefficient
433             coefficients.add(ck.multiply(ai[1]));
434 
435         }
436 
437     }

439     /** Interface for recurrence coefficients generation. */
440     private interface RecurrenceCoefficientsGenerator {
441         /**
442          * Generate recurrence coefficients.
443          * @param k highest degree of the polynomials used in the recurrence
444          * @return an array of three coefficients such that
445          * \( P_{k+1}(x) = (a[0] + a[1] x) P_k(x) - a[2] P_{k-1}(x) \)
446          */
447         BigFraction[] generate(int k);
448     }
