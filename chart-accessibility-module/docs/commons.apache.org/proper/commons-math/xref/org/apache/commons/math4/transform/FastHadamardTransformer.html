17  package org.apache.commons.math4.transform;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.analysis.FunctionUtils;
22  import org.apache.commons.math4.analysis.UnivariateFunction;
23  import org.apache.commons.math4.exception.MathIllegalArgumentException;
24  import org.apache.commons.math4.exception.util.LocalizedFormats;
25  import org.apache.commons.math4.util.ArithmeticUtils;

27  /**
28   * Implements the <a href="http://www.archive.chipcenter.com/dsp/DSP000517F1.html">Fast Hadamard Transform</a> (FHT).
29   * Transformation of an input vector x to the output vector y.
30   * <p>
31   * In addition to transformation of real vectors, the Hadamard transform can
32   * transform integer vectors into integer vectors. However, this integer transform
33   * cannot be inverted directly. Due to a scaling factor it may lead to rational results.
34   * As an example, the inverse transform of integer vector (0, 1, 0, 1) is rational
35   * vector (1/2, -1/2, 0, 0).
36   *
37   * @since 2.0
38   */
39  public class FastHadamardTransformer implements RealTransformer, Serializable {
40  
41      /** Serializable version identifier. */
42      static final long serialVersionUID = 20120211L;
43  
44      /**
45       * {@inheritDoc}
46       *
47       * @throws MathIllegalArgumentException if the length of the data array is
48       * not a power of two
49       */
50      @Override
51      public double[] transform(final double[] f, final TransformType type) {
52          if (type == TransformType.FORWARD) {
53              return fht(f);
54          }
55          return TransformUtils.scaleArray(fht(f), 1.0 / f.length);
56      }
57  
58      /**
59       * {@inheritDoc}
60       *
61       * @throws org.apache.commons.math4.exception.NonMonotonicSequenceException
62       *   if the lower bound is greater than, or equal to the upper bound
63       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
64       *   if the number of sample points is negative
65       * @throws MathIllegalArgumentException if the number of sample points is not a power of two
66       */
67      @Override
68      public double[] transform(final UnivariateFunction f,
69          final double min, final double max, final int n,
70          final TransformType type) {
71  
72          return transform(FunctionUtils.sample(f, min, max, n), type);
73      }
74  
75      /**
76       * Returns the forward transform of the specified integer data set.The
77       * integer transform cannot be inverted directly, due to a scaling factor
78       * which may lead to double results.
79       *
80       * @param f the integer data array to be transformed (signal)
81       * @return the integer transformed array (spectrum)
82       * @throws MathIllegalArgumentException if the length of the data array is not a power of two
83       */
84      public int[] transform(final int[] f) {
85          return fht(f);
86      }
87  
88      /**
89       * The FHT (Fast Hadamard Transformation) which uses only subtraction and
90       * addition. Requires {@code N * log2(N) = n * 2^n} additions.
91       *
92       * <h3>Short Table of manual calculation for N=8</h3>
93       * <ol>
94       * <li><b>x</b> is the input vector to be transformed,</li>
95       * <li><b>y</b> is the output vector (Fast Hadamard transform of <b>x</b>),</li>
96       * <li>a and b are helper rows.</li>
97       * </ol>
98       * <table align="center" border="1" cellpadding="3">
99       * <tbody align="center">
100      * <tr>
101      *     <th>x</th>
102      *     <th>a</th>
103      *     <th>b</th>
104      *     <th>y</th>
105      * </tr>
106      * <tr>
107      *     <th>x<sub>0</sub></th>
108      *     <td>a<sub>0</sub> = x<sub>0</sub> + x<sub>1</sub></td>
109      *     <td>b<sub>0</sub> = a<sub>0</sub> + a<sub>1</sub></td>
110      *     <td>y<sub>0</sub> = b<sub>0</sub >+ b<sub>1</sub></td>
111      * </tr>
112      * <tr>
113      *     <th>x<sub>1</sub></th>
114      *     <td>a<sub>1</sub> = x<sub>2</sub> + x<sub>3</sub></td>
115      *     <td>b<sub>0</sub> = a<sub>2</sub> + a<sub>3</sub></td>
116      *     <td>y<sub>0</sub> = b<sub>2</sub> + b<sub>3</sub></td>
117      * </tr>
118      * <tr>
119      *     <th>x<sub>2</sub></th>
120      *     <td>a<sub>2</sub> = x<sub>4</sub> + x<sub>5</sub></td>
121      *     <td>b<sub>0</sub> = a<sub>4</sub> + a<sub>5</sub></td>
122      *     <td>y<sub>0</sub> = b<sub>4</sub> + b<sub>5</sub></td>
123      * </tr>
124      * <tr>
125      *     <th>x<sub>3</sub></th>
126      *     <td>a<sub>3</sub> = x<sub>6</sub> + x<sub>7</sub></td>
127      *     <td>b<sub>0</sub> = a<sub>6</sub> + a<sub>7</sub></td>
128      *     <td>y<sub>0</sub> = b<sub>6</sub> + b<sub>7</sub></td>
129      * </tr>
130      * <tr>
131      *     <th>x<sub>4</sub></th>
132      *     <td>a<sub>0</sub> = x<sub>0</sub> - x<sub>1</sub></td>
133      *     <td>b<sub>0</sub> = a<sub>0</sub> - a<sub>1</sub></td>
134      *     <td>y<sub>0</sub> = b<sub>0</sub> - b<sub>1</sub></td>
135      * </tr>
136      * <tr>
137      *     <th>x<sub>5</sub></th>
138      *     <td>a<sub>1</sub> = x<sub>2</sub> - x<sub>3</sub></td>
139      *     <td>b<sub>0</sub> = a<sub>2</sub> - a<sub>3</sub></td>
140      *     <td>y<sub>0</sub> = b<sub>2</sub> - b<sub>3</sub></td>
141      * </tr>
142      * <tr>
143      *     <th>x<sub>6</sub></th>
144      *     <td>a<sub>2</sub> = x<sub>4</sub> - x<sub>5</sub></td>
145      *     <td>b<sub>0</sub> = a<sub>4</sub> - a<sub>5</sub></td>
146      *     <td>y<sub>0</sub> = b<sub>4</sub> - b<sub>5</sub></td>
147      * </tr>
148      * <tr>
149      *     <th>x<sub>7</sub></th>
150      *     <td>a<sub>3</sub> = x<sub>6</sub> - x<sub>7</sub></td>
151      *     <td>b<sub>0</sub> = a<sub>6</sub> - a<sub>7</sub></td>
152      *     <td>y<sub>0</sub> = b<sub>6</sub> - b<sub>7</sub></td>
153      * </tr>
154      * </tbody>
155      * </table>
156      *
157      * <h3>How it works</h3>
158      * <ol>
159      * <li>Construct a matrix with {@code N} rows and {@code n + 1} columns,
160      * {@code hadm[n+1][N]}.<br/>
161      * <em>(If I use [x][y] it always means [row-offset][column-offset] of a
162      * Matrix with n rows and m columns. Its entries go from M[0][0]
163      * to M[n][N])</em></li>
164      * <li>Place the input vector {@code x[N]} in the first column of the
165      * matrix {@code hadm}.</li>
166      * <li>The entries of the submatrix {@code D_top} are calculated as follows
167      *     <ul>
168      *         <li>{@code D_top} goes from entry {@code [0][1]} to
169      *         {@code [N / 2 - 1][n + 1]},</li>
170      *         <li>the columns of {@code D_top} are the pairwise mutually
171      *         exclusive sums of the previous column.</li>
172      *     </ul>
173      * </li>
174      * <li>The entries of the submatrix {@code D_bottom} are calculated as
175      * follows
176      *     <ul>
177      *         <li>{@code D_bottom} goes from entry {@code [N / 2][1]} to
178      *         {@code [N][n + 1]},</li>
179      *         <li>the columns of {@code D_bottom} are the pairwise differences
180      *         of the previous column.</li>
181      *     </ul>
182      * </li>
183      * <li>The consputation of {@code D_top} and {@code D_bottom} are best
184      * understood with the above example (for {@code N = 8}).
185      * <li>The output vector {@code y} is now in the last column of
186      * {@code hadm}.</li>
187      * <li><em>Algorithm from <a href="http://www.archive.chipcenter.com/dsp/DSP000517F1.html">chipcenter</a>.</em></li>
188      * </ol>
189      * <h3>Visually</h3>
190      * <table border="1" align="center" cellpadding="3">
191      * <tbody align="center">
192      * <tr>
193      *     <td></td><th>0</th><th>1</th><th>2</th><th>3</th>
194      *     <th>&hellip;</th>
195      *     <th>n + 1</th>
196      * </tr>
197      * <tr>
198      *     <th>0</th>
199      *     <td>x<sub>0</sub></td>
200      *     <td colspan="5" rowspan="5" align="center" valign="middle">
201      *         &uarr;<br/>
202      *         &larr; D<sub>top</sub> &rarr;<br/>
203      *         &darr;
204      *     </td>
205      * </tr>
206      * <tr><th>1</th><td>x<sub>1</sub></td></tr>
207      * <tr><th>2</th><td>x<sub>2</sub></td></tr>
208      * <tr><th>&hellip;</th><td>&hellip;</td></tr>
209      * <tr><th>N / 2 - 1</th><td>x<sub>N/2-1</sub></td></tr>
210      * <tr>
211      *     <th>N / 2</th>
212      *     <td>x<sub>N/2</sub></td>
213      *     <td colspan="5" rowspan="5" align="center" valign="middle">
214      *         &uarr;<br/>
215      *         &larr; D<sub>bottom</sub> &rarr;<br/>
216      *         &darr;
217      *     </td>
218      * </tr>
219      * <tr><th>N / 2 + 1</th><td>x<sub>N/2+1</sub></td></tr>
220      * <tr><th>N / 2 + 2</th><td>x<sub>N/2+2</sub></td></tr>
221      * <tr><th>&hellip;</th><td>&hellip;</td></tr>
222      * <tr><th>N</th><td>x<sub>N</sub></td></tr>
223      * </tbody>
224      * </table>
225      *
226      * @param x the real data array to be transformed
227      * @return the real transformed array, {@code y}
228      * @throws MathIllegalArgumentException if the length of the data array is not a power of two
229      */
230     protected double[] fht(double[] x) throws MathIllegalArgumentException {
231 
232         final int n     = x.length;
233         final int halfN = n / 2;
234 
235         if (!ArithmeticUtils.isPowerOfTwo(n)) {
236             throw new MathIllegalArgumentException(
237                     LocalizedFormats.NOT_POWER_OF_TWO,
238                     Integer.valueOf(n));
239         }
240 
241         /*
242          * Instead of creating a matrix with p+1 columns and n rows, we use two
243          * one dimension arrays which we are used in an alternating way.
244          */
245         double[] yPrevious = new double[n];
246         double[] yCurrent  = x.clone();
247 
248         // iterate from left to right (column)
249         for (int j = 1; j < n; j <<= 1) {
250 
251             // switch columns
252             final double[] yTmp = yCurrent;
253             yCurrent  = yPrevious;
254             yPrevious = yTmp;
255 
256             // iterate from top to bottom (row)
257             for (int i = 0; i < halfN; ++i) {
258                 // Dtop: the top part works with addition
259                 final int twoI = 2 * i;
260                 yCurrent[i] = yPrevious[twoI] + yPrevious[twoI + 1];
261             }
262             for (int i = halfN; i < n; ++i) {
263                 // Dbottom: the bottom part works with subtraction
264                 final int twoI = 2 * i;
265                 yCurrent[i] = yPrevious[twoI - n] - yPrevious[twoI - n + 1];
266             }
267         }
268 
269         return yCurrent;
270 
271     }
272 
273     /**
274      * Returns the forward transform of the specified integer data set. The FHT
275      * (Fast Hadamard Transform) uses only subtraction and addition.
276      *
277      * @param x the integer data array to be transformed
278      * @return the integer transformed array, {@code y}
279      * @throws MathIllegalArgumentException if the length of the data array is not a power of two
280      */
281     protected int[] fht(int[] x) throws MathIllegalArgumentException {
282 
283         final int n     = x.length;
284         final int halfN = n / 2;
285 
286         if (!ArithmeticUtils.isPowerOfTwo(n)) {
287             throw new MathIllegalArgumentException(
288                     LocalizedFormats.NOT_POWER_OF_TWO,
289                     Integer.valueOf(n));
290         }
291 
292         /*
293          * Instead of creating a matrix with p+1 columns and n rows, we use two
294          * one dimension arrays which we are used in an alternating way.
295          */
296         int[] yPrevious = new int[n];
297         int[] yCurrent  = x.clone();
298 
299         // iterate from left to right (column)
300         for (int j = 1; j < n; j <<= 1) {
301 
302             // switch columns
303             final int[] yTmp = yCurrent;
304             yCurrent  = yPrevious;
305             yPrevious = yTmp;
306 
307             // iterate from top to bottom (row)
308             for (int i = 0; i < halfN; ++i) {
309                 // Dtop: the top part works with addition
310                 final int twoI = 2 * i;
311                 yCurrent[i] = yPrevious[twoI] + yPrevious[twoI + 1];
312             }
313             for (int i = halfN; i < n; ++i) {
314                 // Dbottom: the bottom part works with subtraction
315                 final int twoI = 2 * i;
316                 yCurrent[i] = yPrevious[twoI - n] - yPrevious[twoI - n + 1];
317             }
318         }
319 
320         // return the last computed output vector y
321         return yCurrent;
322 
323     }
