669     /**
670      * Returns the sum of the (signed) differences between corresponding elements of the
671      * input arrays -- i.e., sum(sample1[i] - sample2[i]).
672      *
673      * @param sample1  the first array
674      * @param sample2  the second array
675      * @return sum of paired differences
676      * @throws DimensionMismatchException if the arrays do not have the same (positive) length.
677      * @throws NoDataException if the sample arrays are empty.
678      */
679     public static double sumDifference(final double[] sample1, final double[] sample2)
680         throws DimensionMismatchException, NoDataException {
681 
682         int n = sample1.length;
683         if (n != sample2.length) {
684             throw new DimensionMismatchException(n, sample2.length);
685         }
686         if (n <= 0) {
687             throw new NoDataException(LocalizedFormats.INSUFFICIENT_DIMENSION);
688         }
689         double result = 0;
690         for (int i = 0; i < n; i++) {
691             result += sample1[i] - sample2[i];
692         }
693         return result;
694     }

698      * input arrays -- i.e., sum(sample1[i] - sample2[i]) / sample1.length.
699      *
700      * @param sample1  the first array
701      * @param sample2  the second array

703      * @throws DimensionMismatchException if the arrays do not have the same (positive) length.
704      * @throws NoDataException if the sample arrays are empty.
705      */
706     public static double meanDifference(final double[] sample1, final double[] sample2)
707         throws DimensionMismatchException, NoDataException {

711     /**
712      * Returns the variance of the (signed) differences between corresponding elements of the
713      * input arrays -- i.e., var(sample1[i] - sample2[i]).
714      *
715      * @param sample1  the first array
716      * @param sample2  the second array
717      * @param meanDifference   the mean difference between corresponding entries
718      * @return variance of paired differences
719      * @throws DimensionMismatchException if the arrays do not have the same length.
720      * @throws NumberIsTooSmallException if the arrays length is less than 2.
721      * @see #meanDifference(double[],double[])
722      */
723     public static double varianceDifference(final double[] sample1, final double[] sample2, double meanDifference)
724         throws DimensionMismatchException, NumberIsTooSmallException {
725 
726         double sum1 = 0d;
727         double sum2 = 0d;
728         double diff = 0d;
729         int n = sample1.length;
730         if (n != sample2.length) {
731             throw new DimensionMismatchException(n, sample2.length);
732         }
733         if (n < 2) {
734             throw new NumberIsTooSmallException(n, 2, true);
735         }
736         for (int i = 0; i < n; i++) {
737             diff = sample1[i] - sample2[i];
738             sum1 += (diff - meanDifference) *(diff - meanDifference);
739             sum2 += diff - meanDifference;
740         }
741         return (sum1 - (sum2 * sum2 / n)) / (n - 1);
742     }

744     /**
745      * Normalize (standardize) the sample, so it is has a mean of 0 and a standard deviation of 1.
746      *
747      * @param sample Sample to normalize.
748      * @return normalized (standardized) sample.
749      * @since 2.2
750      */
751     public static double[] normalize(final double[] sample) {
752         DescriptiveStatistics stats = new DescriptiveStatistics();
753 
754         // Add the data from the series to stats
755         for (int i = 0; i < sample.length; i++) {
756             stats.addValue(sample[i]);
757         }
758 
759         // Compute mean and standard deviation
760         double mean = stats.getMean();
761         double standardDeviation = stats.getStandardDeviation();
762 
763         // initialize the standardizedSample, which has the same length as the sample
764         double[] standardizedSample = new double[sample.length];
765 
766         for (int i = 0; i < sample.length; i++) {
767             // z = (x- mean)/standardDeviation
768             standardizedSample[i] = (sample[i] - mean) / standardDeviation;
769         }
770         return standardizedSample;
771     }

801     /**
802      * Returns the sample mode(s).
803      * <p>
804      * The mode is the most frequently occurring value in the sample.
805      * If there is a unique value with maximum frequency, this value is returned
806      * as the only element of the output array. Otherwise, the returned array
807      * contains the maximum frequency elements in increasing order.
808      * <p>
809      * For example, if {@code sample} is {0, 12, 5, 6, 0, 13, 5, 17},
810      * the returned array will have length two, with 0 in the first element and
811      * 5 in the second.
812      * <p>
813      * NaN values are ignored when computing the mode - i.e., NaNs will never
814      * appear in the output array.  If the sample includes only NaNs or has
815      * length 0, an empty array is returned.
816      *
817      * @param sample input data
818      * @param begin index (0-based) of the first array element to include
819      * @param length the number of elements to include
820      * @return array of array of the most frequently occurring element(s) sorted in ascending order.
821      * @throws MathIllegalArgumentException if the indices are invalid or the array is null
822      * @since 3.3
823      */
824     public static double[] mode(double[] sample, final int begin, final int length) {
825         if (sample == null) {
826             throw new NullArgumentException(LocalizedFormats.INPUT_ARRAY);
827         }
828 
829         if (begin < 0) {
830             throw new NotPositiveException(LocalizedFormats.START_POSITION, Integer.valueOf(begin));
831         }
832 
833         if (length < 0) {
834             throw new NotPositiveException(LocalizedFormats.LENGTH, Integer.valueOf(length));
835         }
836 
837         return getMode(sample, begin, length);
838     }

840     /**
841      * Private helper method.
842      * Assumes parameters have been validated.
843      * @param values input data
844      * @param begin index (0-based) of the first array element to include
845      * @param length the number of elements to include
846      * @return array of array of the most frequently occurring element(s) sorted in ascending order.
847      */
848     private static double[] getMode(double[] values, final int begin, final int length) {
849         // Add the values to the frequency table
850         Frequency freq = new Frequency();
851         for (int i = begin; i < begin + length; i++) {
852             final double value = values[i];
853             if (!Double.isNaN(value)) {
854                 freq.addValue(Double.valueOf(value));
855             }
856         }
857         List<Comparable<?>> list = freq.getMode();
858         // Convert the list to an array of primitive double
859         double[] modes = new double[list.size()];
860         int i = 0;
861         for(Comparable<?> c : list) {
862             modes[i++] = ((Double) c).doubleValue();
863         }
864         return modes;
865     }
