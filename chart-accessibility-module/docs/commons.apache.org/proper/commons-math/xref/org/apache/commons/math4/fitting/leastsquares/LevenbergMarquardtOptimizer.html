32  /**
33   * This class solves a least-squares problem using the Levenberg-Marquardt
34   * algorithm.
35   *
36   * <p>This implementation <em>should</em> work even for over-determined systems
37   * (i.e. systems having more point than equations). Over-determined systems
38   * are solved by ignoring the point which have the smallest impact according

43   * href="http://www.netlib.org/minpack/lmder.f">lmder</a> routine with minor

45   * inherited convergence checker and the Q.R. decomposition which has been

295     /** {@inheritDoc} */
296     @Override
297     public Optimum optimize(final LeastSquaresProblem problem) {
298         // Pull in relevant data from the problem as locals.
299         final int nR = problem.getObservationSize(); // Number of observed data.
300         final int nC = problem.getParameterSize(); // Number of parameters.
301         // Counters.
302         final Incrementor iterationCounter = problem.getIterationCounter();
303         final Incrementor evaluationCounter = problem.getEvaluationCounter();
304         // Convergence criterion.
305         final ConvergenceChecker<Evaluation> checker = problem.getConvergenceChecker();
306 
307         // arrays shared with the other private methods
308         final int solvedCols  = FastMath.min(nR, nC);
309         /* Parameters evolution direction associated with lmPar. */
310         double[] lmDir = new double[nC];
311         /* Levenberg-Marquardt parameter. */
312         double lmPar = 0;
313 
314         // local point
315         double   delta   = 0;
316         double   xNorm   = 0;
317         double[] diag    = new double[nC];
318         double[] oldX    = new double[nC];
319         double[] oldRes  = new double[nR];
320         double[] qtf     = new double[nR];
321         double[] work1   = new double[nC];
322         double[] work2   = new double[nC];
323         double[] work3   = new double[nC];
324 
325 
326         // Evaluate the function at the starting point and calculate its norm.
327         evaluationCounter.incrementCount();
328         //value will be reassigned in the loop
329         Evaluation current = problem.evaluate(problem.getStart());
330         double[] currentResiduals = current.getResiduals().toArray();
331         double currentCost = current.getCost();
332         double[] currentPoint = current.getPoint().toArray();
333 
334         // Outer loop.
335         boolean firstIteration = true;
336         while (true) {
337             iterationCounter.incrementCount();
338 
339             final Evaluation previous = current;
340 
341             // QR decomposition of the jacobian matrix
342             final InternalData internalData
343                     = qrDecomposition(current.getJacobian(), solvedCols);
344             final double[][] weightedJacobian = internalData.weightedJacobian;
345             final int[] permutation = internalData.permutation;
346             final double[] diagR = internalData.diagR;
347             final double[] jacNorm = internalData.jacNorm;
348 
349             //residuals already have weights applied
350             double[] weightedResidual = currentResiduals;
351             for (int i = 0; i < nR; i++) {
352                 qtf[i] = weightedResidual[i];
353             }
354 
355             // compute Qt.res
356             qTy(qtf, internalData);
357 
358             // now we don't need Q anymore,
359             // so let jacobian contain the R matrix with its diagonal elements
360             for (int k = 0; k < solvedCols; ++k) {
361                 int pk = permutation[k];
362                 weightedJacobian[k][pk] = diagR[pk];
363             }
364 
365             if (firstIteration) {
366                 // scale the point according to the norms of the columns
367                 // of the initial jacobian
368                 xNorm = 0;
369                 for (int k = 0; k < nC; ++k) {
370                     double dk = jacNorm[k];
371                     if (dk == 0) {
372                         dk = 1.0;
373                     }
374                     double xk = dk * currentPoint[k];
375                     xNorm  += xk * xk;
376                     diag[k] = dk;
377                 }
378                 xNorm = FastMath.sqrt(xNorm);
379 
380                 // initialize the step bound delta
381                 delta = (xNorm == 0) ? initialStepBoundFactor : (initialStepBoundFactor * xNorm);
382             }
383 
384             // check orthogonality between function vector and jacobian columns
385             double maxCosine = 0;
386             if (currentCost != 0) {
387                 for (int j = 0; j < solvedCols; ++j) {
388                     int    pj = permutation[j];
389                     double s  = jacNorm[pj];
390                     if (s != 0) {
391                         double sum = 0;
392                         for (int i = 0; i <= j; ++i) {
393                             sum += weightedJacobian[i][pj] * qtf[i];
394                         }
395                         maxCosine = FastMath.max(maxCosine, FastMath.abs(sum) / (s * currentCost));
396                     }
397                 }
398             }
399             if (maxCosine <= orthoTolerance) {
400                 // Convergence has been reached.
401                 return new OptimumImpl(
402                         current,
403                         evaluationCounter.getCount(),
404                         iterationCounter.getCount());
405             }
406 
407             // rescale if necessary
408             for (int j = 0; j < nC; ++j) {
409                 diag[j] = FastMath.max(diag[j], jacNorm[j]);
410             }
411 
412             // Inner loop.
413             for (double ratio = 0; ratio < 1.0e-4;) {
414 
415                 // save the state
416                 for (int j = 0; j < solvedCols; ++j) {
417                     int pj = permutation[j];
418                     oldX[pj] = currentPoint[pj];
419                 }
420                 final double previousCost = currentCost;
421                 double[] tmpVec = weightedResidual;
422                 weightedResidual = oldRes;
423                 oldRes    = tmpVec;
424 
425                 // determine the Levenberg-Marquardt parameter
426                 lmPar = determineLMParameter(qtf, delta, diag,
427                                      internalData, solvedCols,
428                                      work1, work2, work3, lmDir, lmPar);
429 
430                 // compute the new point and the norm of the evolution direction
431                 double lmNorm = 0;
432                 for (int j = 0; j < solvedCols; ++j) {
433                     int pj = permutation[j];
434                     lmDir[pj] = -lmDir[pj];
435                     currentPoint[pj] = oldX[pj] + lmDir[pj];
436                     double s = diag[pj] * lmDir[pj];
437                     lmNorm  += s * s;
438                 }
439                 lmNorm = FastMath.sqrt(lmNorm);
440                 // on the first iteration, adjust the initial step bound.
441                 if (firstIteration) {
442                     delta = FastMath.min(delta, lmNorm);
443                 }
444 
445                 // Evaluate the function at x + p and calculate its norm.
446                 evaluationCounter.incrementCount();
447                 current = problem.evaluate(new ArrayRealVector(currentPoint));
448                 currentResiduals = current.getResiduals().toArray();
449                 currentCost = current.getCost();
450                 currentPoint = current.getPoint().toArray();
451 
452                 // compute the scaled actual reduction
453                 double actRed = -1.0;
454                 if (0.1 * currentCost < previousCost) {
455                     double r = currentCost / previousCost;
456                     actRed = 1.0 - r * r;
457                 }
458 
459                 // compute the scaled predicted reduction
460                 // and the scaled directional derivative
461                 for (int j = 0; j < solvedCols; ++j) {
462                     int pj = permutation[j];
463                     double dirJ = lmDir[pj];
464                     work1[j] = 0;
465                     for (int i = 0; i <= j; ++i) {
466                         work1[i] += weightedJacobian[i][pj] * dirJ;
467                     }
468                 }
469                 double coeff1 = 0;
470                 for (int j = 0; j < solvedCols; ++j) {
471                     coeff1 += work1[j] * work1[j];
472                 }
473                 double pc2 = previousCost * previousCost;
474                 coeff1 /= pc2;
475                 double coeff2 = lmPar * lmNorm * lmNorm / pc2;
476                 double preRed = coeff1 + 2 * coeff2;
477                 double dirDer = -(coeff1 + coeff2);
478 
479                 // ratio of the actual to the predicted reduction
480                 ratio = (preRed == 0) ? 0 : (actRed / preRed);
481 
482                 // update the step bound
483                 if (ratio <= 0.25) {
484                     double tmp =
485                         (actRed < 0) ? (0.5 * dirDer / (dirDer + 0.5 * actRed)) : 0.5;
486                         if ((0.1 * currentCost >= previousCost) || (tmp < 0.1)) {
487                             tmp = 0.1;
488                         }
489                         delta = tmp * FastMath.min(delta, 10.0 * lmNorm);
490                         lmPar /= tmp;
491                 } else if ((lmPar == 0) || (ratio >= 0.75)) {
492                     delta = 2 * lmNorm;
493                     lmPar *= 0.5;
494                 }
495 
496                 // test for successful iteration.
497                 if (ratio >= 1.0e-4) {
498                     // successful iteration, update the norm
499                     firstIteration = false;
500                     xNorm = 0;
501                     for (int k = 0; k < nC; ++k) {
502                         double xK = diag[k] * currentPoint[k];
503                         xNorm += xK * xK;
504                     }
505                     xNorm = FastMath.sqrt(xNorm);
506 
507                     // tests for convergence.
508                     if (checker != null && checker.converged(iterationCounter.getCount(), previous, current)) {
509                         return new OptimumImpl(current, evaluationCounter.getCount(), iterationCounter.getCount());
510                     }
511                 } else {
512                     // failed iteration, reset the previous values
513                     currentCost = previousCost;
514                     for (int j = 0; j < solvedCols; ++j) {
515                         int pj = permutation[j];
516                         currentPoint[pj] = oldX[pj];
517                     }
518                     tmpVec    = weightedResidual;
519                     weightedResidual = oldRes;
520                     oldRes    = tmpVec;
521                     // Reset "current" to previous values.
522                     current = previous;
523                 }
524 
525                 // Default convergence criteria.
526                 if ((FastMath.abs(actRed) <= costRelativeTolerance &&
527                      preRed <= costRelativeTolerance &&
528                      ratio <= 2.0) ||
529                     delta <= parRelativeTolerance * xNorm) {
530                     return new OptimumImpl(current, evaluationCounter.getCount(), iterationCounter.getCount());
531                 }
532 
533                 // tests for termination and stringent tolerances
534                 if (FastMath.abs(actRed) <= TWO_EPS &&
535                     preRed <= TWO_EPS &&
536                     ratio <= 2.0) {
537                     throw new ConvergenceException(LocalizedFormats.TOO_SMALL_COST_RELATIVE_TOLERANCE,
538                                                    costRelativeTolerance);
539                 } else if (delta <= TWO_EPS * xNorm) {
540                     throw new ConvergenceException(LocalizedFormats.TOO_SMALL_PARAMETERS_RELATIVE_TOLERANCE,
541                                                    parRelativeTolerance);
542                 } else if (maxCosine <= TWO_EPS) {
543                     throw new ConvergenceException(LocalizedFormats.TOO_SMALL_ORTHOGONALITY_TOLERANCE,
544                                                    orthoTolerance);
545                 }
546             }
547         }
548     }

550     /**
551      * Holds internal data.
552      * This structure was created so that all optimizer fields can be "final".
553      * Code should be further refactored in order to not pass around arguments
554      * that will modified in-place (cf. "work" arrays).

556     private static class InternalData {
557         /** Weighted Jacobian. */

559         /** Columns permutation array. */

561         /** Rank of the Jacobian matrix. */
562         private final int rank;
563         /** Diagonal elements of the R matrix in the QR decomposition. */
564         private final double[] diagR;
565         /** Norms of the columns of the jacobian matrix. */
566         private final double[] jacNorm;
567         /** Coefficients of the Householder transforms vectors. */
568         private final double[] beta;

570         /**
571          * @param weightedJacobian Weighted Jacobian.

573          * @param rank Rank of the Jacobian matrix.
574          * @param diagR Diagonal elements of the R matrix in the QR decomposition.
575          * @param jacNorm Norms of the columns of the jacobian matrix.
576          * @param beta Coefficients of the Householder transforms vectors.
577          */
578         InternalData(double[][] weightedJacobian,
579                      int[] permutation,
580                      int rank,
581                      double[] diagR,
582                      double[] jacNorm,
583                      double[] beta) {
584             this.weightedJacobian = weightedJacobian;
585             this.permutation = permutation;
586             this.rank = rank;
587             this.diagR = diagR;
588             this.jacNorm = jacNorm;
589             this.beta = beta;
590         }

593     /**
594      * Determines the Levenberg-Marquardt parameter.
595      *
596      * <p>This implementation is a translation in Java of the MINPACK
597      * <a href="http://www.netlib.org/minpack/lmpar.f">lmpar</a>
598      * routine.</p>
599      * <p>This method sets the lmPar and lmDir attributes.</p>
600      * <p>The authors of the original fortran function are:</p>
601      * <ul>
602      *   <li>Argonne National Laboratory. MINPACK project. March 1980</li>
603      *   <li>Burton  S. Garbow</li>
604      *   <li>Kenneth E. Hillstrom</li>
605      *   <li>Jorge   J. More</li>
606      * </ul>
607      * <p>Luc Maisonobe did the Java translation.</p>
608      *
609      * @param qy Array containing qTy.
610      * @param delta Upper bound on the euclidean norm of diagR * lmDir.
611      * @param diag Diagonal matrix.
612      * @param internalData Data (modified in-place in this method).
613      * @param solvedCols Number of solved point.
614      * @param work1 work array
615      * @param work2 work array
616      * @param work3 work array
617      * @param lmDir the "returned" LM direction will be stored in this array.
618      * @param lmPar the value of the LM parameter from the previous iteration.
619      * @return the new LM parameter
620      */
621     private double determineLMParameter(double[] qy, double delta, double[] diag,
622                                       InternalData internalData, int solvedCols,
623                                       double[] work1, double[] work2, double[] work3,
624                                       double[] lmDir, double lmPar) {
625         final double[][] weightedJacobian = internalData.weightedJacobian;
626         final int[] permutation = internalData.permutation;
627         final int rank = internalData.rank;
628         final double[] diagR = internalData.diagR;
629 
630         final int nC = weightedJacobian[0].length;
631 
632         // compute and store in x the gauss-newton direction, if the
633         // jacobian is rank-deficient, obtain a least squares solution
634         for (int j = 0; j < rank; ++j) {
635             lmDir[permutation[j]] = qy[j];
636         }
637         for (int j = rank; j < nC; ++j) {
638             lmDir[permutation[j]] = 0;
639         }
640         for (int k = rank - 1; k >= 0; --k) {
641             int pk = permutation[k];
642             double ypk = lmDir[pk] / diagR[pk];
643             for (int i = 0; i < k; ++i) {
644                 lmDir[permutation[i]] -= ypk * weightedJacobian[i][pk];
645             }
646             lmDir[pk] = ypk;
647         }
648 
649         // evaluate the function at the origin, and test
650         // for acceptance of the Gauss-Newton direction
651         double dxNorm = 0;
652         for (int j = 0; j < solvedCols; ++j) {
653             int pj = permutation[j];
654             double s = diag[pj] * lmDir[pj];
655             work1[pj] = s;
656             dxNorm += s * s;
657         }
658         dxNorm = FastMath.sqrt(dxNorm);
659         double fp = dxNorm - delta;
660         if (fp <= 0.1 * delta) {
661             lmPar = 0;
662             return lmPar;
663         }
664 
665         // if the jacobian is not rank deficient, the Newton step provides
666         // a lower bound, parl, for the zero of the function,
667         // otherwise set this bound to zero
668         double sum2;
669         double parl = 0;
670         if (rank == solvedCols) {
671             for (int j = 0; j < solvedCols; ++j) {
672                 int pj = permutation[j];
673                 work1[pj] *= diag[pj] / dxNorm;
674             }
675             sum2 = 0;
676             for (int j = 0; j < solvedCols; ++j) {
677                 int pj = permutation[j];
678                 double sum = 0;
679                 for (int i = 0; i < j; ++i) {
680                     sum += weightedJacobian[i][pj] * work1[permutation[i]];
681                 }
682                 double s = (work1[pj] - sum) / diagR[pj];
683                 work1[pj] = s;
684                 sum2 += s * s;
685             }
686             parl = fp / (delta * sum2);
687         }
688 
689         // calculate an upper bound, paru, for the zero of the function
690         sum2 = 0;
691         for (int j = 0; j < solvedCols; ++j) {
692             int pj = permutation[j];
693             double sum = 0;
694             for (int i = 0; i <= j; ++i) {
695                 sum += weightedJacobian[i][pj] * qy[i];
696             }
697             sum /= diag[pj];
698             sum2 += sum * sum;
699         }
700         double gNorm = FastMath.sqrt(sum2);
701         double paru = gNorm / delta;
702         if (paru == 0) {
703             paru = Precision.SAFE_MIN / FastMath.min(delta, 0.1);
704         }
705 
706         // if the input par lies outside of the interval (parl,paru),
707         // set par to the closer endpoint
708         lmPar = FastMath.min(paru, FastMath.max(lmPar, parl));
709         if (lmPar == 0) {
710             lmPar = gNorm / dxNorm;
711         }
712 
713         for (int countdown = 10; countdown >= 0; --countdown) {
714 
715             // evaluate the function at the current value of lmPar
716             if (lmPar == 0) {
717                 lmPar = FastMath.max(Precision.SAFE_MIN, 0.001 * paru);
718             }
719             double sPar = FastMath.sqrt(lmPar);
720             for (int j = 0; j < solvedCols; ++j) {
721                 int pj = permutation[j];
722                 work1[pj] = sPar * diag[pj];
723             }
724             determineLMDirection(qy, work1, work2, internalData, solvedCols, work3, lmDir);
725 
726             dxNorm = 0;
727             for (int j = 0; j < solvedCols; ++j) {
728                 int pj = permutation[j];
729                 double s = diag[pj] * lmDir[pj];
730                 work3[pj] = s;
731                 dxNorm += s * s;
732             }
733             dxNorm = FastMath.sqrt(dxNorm);
734             double previousFP = fp;
735             fp = dxNorm - delta;
736 
737             // if the function is small enough, accept the current value
738             // of lmPar, also test for the exceptional cases where parl is zero
739             if (FastMath.abs(fp) <= 0.1 * delta ||
740                 (parl == 0 &&
741                  fp <= previousFP &&
742                  previousFP < 0)) {
743                 return lmPar;
744             }
745 
746             // compute the Newton correction
747             for (int j = 0; j < solvedCols; ++j) {
748                 int pj = permutation[j];
749                 work1[pj] = work3[pj] * diag[pj] / dxNorm;
750             }
751             for (int j = 0; j < solvedCols; ++j) {
752                 int pj = permutation[j];
753                 work1[pj] /= work2[j];
754                 double tmp = work1[pj];
755                 for (int i = j + 1; i < solvedCols; ++i) {
756                     work1[permutation[i]] -= weightedJacobian[i][pj] * tmp;
757                 }
758             }
759             sum2 = 0;
760             for (int j = 0; j < solvedCols; ++j) {
761                 double s = work1[permutation[j]];
762                 sum2 += s * s;
763             }
764             double correction = fp / (delta * sum2);
765 
766             // depending on the sign of the function, update parl or paru.
767             if (fp > 0) {
768                 parl = FastMath.max(parl, lmPar);
769             } else if (fp < 0) {
770                 paru = FastMath.min(paru, lmPar);
771             }
772 
773             // compute an improved estimate for lmPar
774             lmPar = FastMath.max(parl, lmPar + correction);
775         }
776 
777         return lmPar;
778     }

780     /**
781      * Solve a*x = b and d*x = 0 in the least squares sense.
782      * <p>This implementation is a translation in Java of the MINPACK
783      * <a href="http://www.netlib.org/minpack/qrsolv.f">qrsolv</a>
784      * routine.</p>
785      * <p>This method sets the lmDir and lmDiag attributes.</p>
786      * <p>The authors of the original fortran function are:</p>
787      * <ul>
788      *   <li>Argonne National Laboratory. MINPACK project. March 1980</li>
789      *   <li>Burton  S. Garbow</li>
790      *   <li>Kenneth E. Hillstrom</li>
791      *   <li>Jorge   J. More</li>
792      * </ul>
793      * <p>Luc Maisonobe did the Java translation.</p>
794      *
795      * @param qy array containing qTy
796      * @param diag diagonal matrix
797      * @param lmDiag diagonal elements associated with lmDir
798      * @param internalData Data (modified in-place in this method).
799      * @param solvedCols Number of sloved point.
800      * @param work work array
801      * @param lmDir the "returned" LM direction is stored in this array
802      */
803     private void determineLMDirection(double[] qy, double[] diag,
804                                       double[] lmDiag,
805                                       InternalData internalData,
806                                       int solvedCols,
807                                       double[] work,
808                                       double[] lmDir) {
809         final int[] permutation = internalData.permutation;
810         final double[][] weightedJacobian = internalData.weightedJacobian;
811         final double[] diagR = internalData.diagR;
812 
813         // copy R and Qty to preserve input and initialize s
814         //  in particular, save the diagonal elements of R in lmDir
815         for (int j = 0; j < solvedCols; ++j) {
816             int pj = permutation[j];
817             for (int i = j + 1; i < solvedCols; ++i) {
818                 weightedJacobian[i][pj] = weightedJacobian[j][permutation[i]];
819             }
820             lmDir[j] = diagR[pj];
821             work[j]  = qy[j];
822         }
823 
824         // eliminate the diagonal matrix d using a Givens rotation
825         for (int j = 0; j < solvedCols; ++j) {
826 
827             // prepare the row of d to be eliminated, locating the
828             // diagonal element using p from the Q.R. factorization
829             int pj = permutation[j];
830             double dpj = diag[pj];
831             if (dpj != 0) {
832                 Arrays.fill(lmDiag, j + 1, lmDiag.length, 0);
833             }
834             lmDiag[j] = dpj;
835 
836             //  the transformations to eliminate the row of d
837             // modify only a single element of Qty
838             // beyond the first n, which is initially zero.
839             double qtbpj = 0;
840             for (int k = j; k < solvedCols; ++k) {
841                 int pk = permutation[k];
842 
843                 // determine a Givens rotation which eliminates the
844                 // appropriate element in the current row of d
845                 if (lmDiag[k] != 0) {
846 
847                     final double sin;
848                     final double cos;
849                     double rkk = weightedJacobian[k][pk];
850                     if (FastMath.abs(rkk) < FastMath.abs(lmDiag[k])) {
851                         final double cotan = rkk / lmDiag[k];
852                         sin   = 1.0 / FastMath.sqrt(1.0 + cotan * cotan);
853                         cos   = sin * cotan;
854                     } else {
855                         final double tan = lmDiag[k] / rkk;
856                         cos = 1.0 / FastMath.sqrt(1.0 + tan * tan);
857                         sin = cos * tan;
858                     }
859 
860                     // compute the modified diagonal element of R and
861                     // the modified element of (Qty,0)
862                     weightedJacobian[k][pk] = cos * rkk + sin * lmDiag[k];
863                     final double temp = cos * work[k] + sin * qtbpj;
864                     qtbpj = -sin * work[k] + cos * qtbpj;
865                     work[k] = temp;
866 
867                     // accumulate the tranformation in the row of s
868                     for (int i = k + 1; i < solvedCols; ++i) {
869                         double rik = weightedJacobian[i][pk];
870                         final double temp2 = cos * rik + sin * lmDiag[i];
871                         lmDiag[i] = -sin * rik + cos * lmDiag[i];
872                         weightedJacobian[i][pk] = temp2;
873                     }
874                 }
875             }
876 
877             // store the diagonal element of s and restore
878             // the corresponding diagonal element of R
879             lmDiag[j] = weightedJacobian[j][permutation[j]];
880             weightedJacobian[j][permutation[j]] = lmDir[j];
881         }
882 
883         // solve the triangular system for z, if the system is
884         // singular, then obtain a least squares solution
885         int nSing = solvedCols;
886         for (int j = 0; j < solvedCols; ++j) {
887             if ((lmDiag[j] == 0) && (nSing == solvedCols)) {
888                 nSing = j;
889             }
890             if (nSing < solvedCols) {
891                 work[j] = 0;
892             }
893         }
894         if (nSing > 0) {
895             for (int j = nSing - 1; j >= 0; --j) {
896                 int pj = permutation[j];
897                 double sum = 0;
898                 for (int i = j + 1; i < nSing; ++i) {
899                     sum += weightedJacobian[i][pj] * work[i];
900                 }
901                 work[j] = (work[j] - sum) / lmDiag[j];
902             }
903         }
904 
905         // permute the components of z back to components of lmDir
906         for (int j = 0; j < lmDir.length; ++j) {
907             lmDir[permutation[j]] = work[j];
908         }
909     }

911     /**
912      * Decompose a matrix A as A.P = Q.R using Householder transforms.
913      * <p>As suggested in the P. Lascaux and R. Theodor book
914      * <i>Analyse num&eacute;rique matricielle appliqu&eacute;e &agrave;
915      * l'art de l'ing&eacute;nieur</i> (Masson, 1986), instead of representing
916      * the Householder transforms with u<sub>k</sub> unit vectors such that:
917      * <pre>
918      * H<sub>k</sub> = I - 2u<sub>k</sub>.u<sub>k</sub><sup>t</sup>
919      * </pre>
920      * we use <sub>k</sub> non-unit vectors such that:
921      * <pre>
922      * H<sub>k</sub> = I - beta<sub>k</sub>v<sub>k</sub>.v<sub>k</sub><sup>t</sup>
923      * </pre>
924      * where v<sub>k</sub> = a<sub>k</sub> - alpha<sub>k</sub> e<sub>k</sub>.
925      * The beta<sub>k</sub> coefficients are provided upon exit as recomputing
926      * them from the v<sub>k</sub> vectors would be costly.</p>
927      * <p>This decomposition handles rank deficient cases since the tranformations
928      * are performed in non-increasing columns norms order thanks to columns
929      * pivoting. The diagonal elements of the R matrix are therefore also in
930      * non-increasing absolute values order.</p>
931      *
932      * @param jacobian Weighted Jacobian matrix at the current point.
933      * @param solvedCols Number of solved point.
934      * @return data used in other methods of this class.
935      * @throws ConvergenceException if the decomposition cannot be performed.
936      */
937     private InternalData qrDecomposition(RealMatrix jacobian,
938                                          int solvedCols) throws ConvergenceException {
939         // Code in this class assumes that the weighted Jacobian is -(W^(1/2) J),
940         // hence the multiplication by -1.
941         final double[][] weightedJacobian = jacobian.scalarMultiply(-1).getData();
942 
943         final int nR = weightedJacobian.length;
944         final int nC = weightedJacobian[0].length;
945 
946         final int[] permutation = new int[nC];
947         final double[] diagR = new double[nC];
948         final double[] jacNorm = new double[nC];
949         final double[] beta = new double[nC];
950 
951         // initializations
952         for (int k = 0; k < nC; ++k) {
953             permutation[k] = k;
954             double norm2 = 0;
955             for (int i = 0; i < nR; ++i) {
956                 double akk = weightedJacobian[i][k];
957                 norm2 += akk * akk;
958             }
959             jacNorm[k] = FastMath.sqrt(norm2);
960         }
961 
962         // transform the matrix column after column
963         for (int k = 0; k < nC; ++k) {
964 
965             // select the column with the greatest norm on active components
966             int nextColumn = -1;
967             double ak2 = Double.NEGATIVE_INFINITY;
968             for (int i = k; i < nC; ++i) {
969                 double norm2 = 0;
970                 for (int j = k; j < nR; ++j) {
971                     double aki = weightedJacobian[j][permutation[i]];
972                     norm2 += aki * aki;
973                 }
974                 if (Double.isInfinite(norm2) || Double.isNaN(norm2)) {
975                     throw new ConvergenceException(LocalizedFormats.UNABLE_TO_PERFORM_QR_DECOMPOSITION_ON_JACOBIAN,
976                                                    nR, nC);
977                 }
978                 if (norm2 > ak2) {
979                     nextColumn = i;
980                     ak2        = norm2;
981                 }
982             }
983             if (ak2 <= qrRankingThreshold) {
984                 return new InternalData(weightedJacobian, permutation, k, diagR, jacNorm, beta);
985             }
986             int pk = permutation[nextColumn];
987             permutation[nextColumn] = permutation[k];
988             permutation[k] = pk;
989 
990             // choose alpha such that Hk.u = alpha ek
991             double akk = weightedJacobian[k][pk];
992             double alpha = (akk > 0) ? -FastMath.sqrt(ak2) : FastMath.sqrt(ak2);
993             double betak = 1.0 / (ak2 - akk * alpha);
994             beta[pk] = betak;
995 
996             // transform the current column
997             diagR[pk] = alpha;
998             weightedJacobian[k][pk] -= alpha;
999 
1000             // transform the remaining columns
1001             for (int dk = nC - 1 - k; dk > 0; --dk) {
1002                 double gamma = 0;
1003                 for (int j = k; j < nR; ++j) {
1004                     gamma += weightedJacobian[j][pk] * weightedJacobian[j][permutation[k + dk]];
1005                 }
1006                 gamma *= betak;
1007                 for (int j = k; j < nR; ++j) {
1008                     weightedJacobian[j][permutation[k + dk]] -= gamma * weightedJacobian[j][pk];
1009                 }
1010             }
1011         }
1012 
1013         return new InternalData(weightedJacobian, permutation, solvedCols, diagR, jacNorm, beta);
1014     }

1016     /**
1017      * Compute the product Qt.y for some Q.R. decomposition.
1018      *
1019      * @param y vector to multiply (will be overwritten with the result)
1020      * @param internalData Data.
1021      */
1022     private void qTy(double[] y,
1023                      InternalData internalData) {
1024         final double[][] weightedJacobian = internalData.weightedJacobian;
1025         final int[] permutation = internalData.permutation;
1026         final double[] beta = internalData.beta;
1027 
1028         final int nR = weightedJacobian.length;
1029         final int nC = weightedJacobian[0].length;
1030 
1031         for (int k = 0; k < nC; ++k) {
1032             int pk = permutation[k];
1033             double gamma = 0;
1034             for (int i = k; i < nR; ++i) {
1035                 gamma += weightedJacobian[i][pk] * y[i];
1036             }
1037             gamma *= beta[pk];
1038             for (int i = k; i < nR; ++i) {
1039                 y[i] -= gamma * weightedJacobian[i][pk];
1040             }
1041         }
1042     }
