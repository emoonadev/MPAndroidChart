17  package org.apache.commons.math4.optim.linear;
18  
19  /**
20   * Types of relationships between two cells in a Solver {@link LinearConstraint}.
21   *
22   * @since 2.0
23   */
24  public enum Relationship {
25      /** Equality relationship. */
26      EQ("="),
27      /** Lesser than or equal relationship. */
28      LEQ("<="),
29      /** Greater than or equal relationship. */
30      GEQ(">=");
31  
32      /** Display string for the relationship. */
33      private final String stringValue;
34  
35      /**
36       * Simple constructor.
37       *
38       * @param stringValue Display string for the relationship.
39       */
40      Relationship(String stringValue) {
41          this.stringValue = stringValue;
42      }
43  
44      /** {@inheritDoc} */
45      @Override
46      public String toString() {
47          return stringValue;
48      }
49  
50      /**
51       * Gets the relationship obtained when multiplying all coefficients by -1.
52       *
53       * @return the opposite relationship.
54       */
55      public Relationship oppositeRelationship() {
56          switch (this) {
57          case LEQ :
58              return GEQ;
59          case GEQ :
60              return LEQ;
61          default :
62              return EQ;
63          }
64      }
65  }
