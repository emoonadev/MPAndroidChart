17  package org.apache.commons.math4.stat.descriptive;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.exception.NullArgumentException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.math4.util.MathArrays;
23  import org.apache.commons.math4.util.MathUtils;
24  import org.apache.commons.math4.util.Precision;
25  
26  /**
27   * Abstract base class for implementations of the
28   * {@link StorelessUnivariateStatistic} interface.
29   * <p>
30   * Provides default {@code evaluate(double[],...)} and {@code incrementAll(double[])}
31   * implementations.
32   * <p>
33   * <strong>Note that these implementations are not synchronized.</strong>
34   */
35  public abstract class AbstractStorelessUnivariateStatistic
36      implements StorelessUnivariateStatistic {
37  
38      /**
39       * This default implementation creates a copy of this {@link StorelessUnivariateStatistic}
40       * instance, calls {@link #clear} on it, then calls {@link #incrementAll} with the specified
41       * portion of the input array, and then uses {@link #getResult} to compute the return value.
42       * <p>
43       * Note that this implementation does not change the internal state of the statistic.
44       * <p>
45       * Implementations may override this method with a more efficient and possibly more
46       * accurate implementation that works directly with the input array.
47       * <p>
48       * If the array is null, a MathIllegalArgumentException is thrown.
49       *
50       * @param values input array
51       * @return the value of the statistic applied to the input array
52       * @throws MathIllegalArgumentException if values is null
53       * @see org.apache.commons.math4.stat.descriptive.UnivariateStatistic#evaluate(double[])
54       */
55      @Override
56      public double evaluate(final double[] values) throws MathIllegalArgumentException {
57          if (values == null) {
58              throw new NullArgumentException(LocalizedFormats.INPUT_ARRAY);
59          }
60          return evaluate(values, 0, values.length);
61      }
62  
63      /**
64       * This default implementation creates a copy of this {@link StorelessUnivariateStatistic}
65       * instance, calls {@link #clear} on it, then calls {@link #incrementAll} with the specified
66       * portion of the input array, and then uses {@link #getResult} to compute the return value.
67       * <p>
68       * Note that this implementation does not change the internal state of the statistic.
69       * <p>
70       * Implementations may override this method with a more efficient and possibly more
71       * accurate implementation that works directly with the input array.
72       * <p>
73       * If the array is null or the index parameters are not valid, an
74       * MathIllegalArgumentException is thrown.
75       *
76       * @param values the input array
77       * @param begin the index of the first element to include
78       * @param length the number of elements to include
79       * @return the value of the statistic applied to the included array entries
80       * @throws MathIllegalArgumentException if the array is null or the indices are not valid
81       * @see org.apache.commons.math4.stat.descriptive.UnivariateStatistic#evaluate(double[], int, int)
82       */
83      @Override
84      public double evaluate(final double[] values, final int begin, final int length)
85          throws MathIllegalArgumentException {
86  
87          if (MathArrays.verifyValues(values, begin, length)) {
88              final StorelessUnivariateStatistic stat = copy();
89              stat.clear();
90              stat.incrementAll(values, begin, length);
91              return stat.getResult();
92          }
93          return Double.NaN;
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      @Override
100     public abstract StorelessUnivariateStatistic copy();
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public abstract void clear();
107 
108     /**
109      * {@inheritDoc}
110      */
111     @Override
112     public abstract double getResult();
113 
114     /**
115      * {@inheritDoc}
116      */
117     @Override
118     public abstract void increment(final double d);
119 
120     /**
121      * This default implementation just calls {@link #increment} in a loop over
122      * the input array.
123      * <p>
124      * Throws IllegalArgumentException if the input values array is null.
125      *
126      * @param values values to add
127      * @throws MathIllegalArgumentException if values is null
128      * @see StorelessUnivariateStatistic#incrementAll(double[])
129      */
130     @Override
131     public void incrementAll(double[] values) throws MathIllegalArgumentException {
132         if (values == null) {
133             throw new NullArgumentException(LocalizedFormats.INPUT_ARRAY);
134         }
135         incrementAll(values, 0, values.length);
136     }
137 
138     /**
139      * This default implementation just calls {@link #increment} in a loop over
140      * the specified portion of the input array.
141      * <p>
142      * Throws IllegalArgumentException if the input values array is null.
143      *
144      * @param values  array holding values to add
145      * @param begin   index of the first array element to add
146      * @param length  number of array elements to add
147      * @throws MathIllegalArgumentException if values is null
148      * @see StorelessUnivariateStatistic#incrementAll(double[], int, int)
149      */
150     @Override
151     public void incrementAll(double[] values, int begin, int length) throws MathIllegalArgumentException {
152         if (MathArrays.verifyValues(values, begin, length)) {
153             int k = begin + length;
154             for (int i = begin; i < k; i++) {
155                 increment(values[i]);
156             }
157         }
158     }
159 
160     /**
161      * Returns true iff <code>object</code> is the same type of
162      * {@link StorelessUnivariateStatistic} (the object's class equals this
163      * instance) returning the same values as this for <code>getResult()</code>
164      * and <code>getN()</code>.
165      *
166      * @param object object to test equality against.
167      * @return true if object returns the same value as this
168      */
169     @Override
170     public boolean equals(Object object) {
171         if (object == this ) {
172             return true;
173         }
174         if (object == null || object.getClass() != this.getClass()) {
175             return false;
176         }
177         StorelessUnivariateStatistic stat = (StorelessUnivariateStatistic) object;
178         return Precision.equalsIncludingNaN(stat.getResult(), this.getResult()) &&
179                Precision.equalsIncludingNaN(stat.getN(), this.getN());
180     }
181 
182     /**
183      * Returns hash code based on getResult() and getN().
184      *
185      * @return hash code
186      */
187     @Override
188     public int hashCode() {
189         return 31 * (31 + MathUtils.hash(getResult())) + MathUtils.hash(getN());
190     }
191 
192 }
