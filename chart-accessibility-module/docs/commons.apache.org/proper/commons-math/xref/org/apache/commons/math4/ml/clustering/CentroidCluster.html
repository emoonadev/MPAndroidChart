17  package org.apache.commons.math4.ml.clustering;
18  
19  /**
20   * A Cluster used by centroid-based clustering algorithms.
21   * <p>
22   * Defines additionally a cluster center which may not necessarily be a member
23   * of the original data set.
24   *
25   * @param <T> the type of points that can be clustered
26   * @since 3.2
27   */
28  public class CentroidCluster<T extends Clusterable> extends Cluster<T> {
29  
30      /** Serializable version identifier. */
31      private static final long serialVersionUID = -3075288519071812288L;
32  
33      /** Center of the cluster. */
34      private final Clusterable center;
35  
36      /**
37       * Build a cluster centered at a specified point.
38       * @param center the point which is to be the center of this cluster
39       */
40      public CentroidCluster(final Clusterable center) {
41          super();
42          this.center = center;
43      }
44  
45      /**
46       * Get the point chosen to be the center of this cluster.
47       * @return chosen cluster center
48       */
49      public Clusterable getCenter() {
50          return center;
51      }
52  
53  }
