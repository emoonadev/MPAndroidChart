17  package org.apache.commons.math4.distribution;
18  
19  /**
20   * Implementation of the chi-squared distribution.
21   *
22   * @see <a href="http://en.wikipedia.org/wiki/Chi-squared_distribution">Chi-squared distribution (Wikipedia)</a>
23   * @see <a href="http://mathworld.wolfram.com/Chi-SquaredDistribution.html">Chi-squared Distribution (MathWorld)</a>
24   */
25  public class ChiSquaredDistribution extends AbstractRealDistribution {
26      /**
27       * Default inverse cumulative probability accuracy
28       * @since 2.1
29       */
30      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
31      /** Serializable version identifier */
32      private static final long serialVersionUID = 20160311L;
33      /** Internal Gamma distribution. */
34      private final GammaDistribution gamma;
35      /** Inverse cumulative probability accuracy */
36      private final double solverAbsoluteAccuracy;
37  
38      /**
39       * Creates distribution with the given degrees of freedom.
40       *
41       * @param degreesOfFreedom Degrees of freedom.
42       */
43      public ChiSquaredDistribution(double degreesOfFreedom) {
44          this(degreesOfFreedom, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
45      }
46  
47      /**
48       * Creates a distribution with the given degrees of freedom and
49       * inverse cumulative probability accuracy.
50       *
51       * @param degreesOfFreedom Degrees of freedom.
52       * @param inverseCumAccuracy the maximum absolute error in inverse
53       * cumulative probability estimates (defaults to
54       * {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
55       *
56       * @since 2.1
57       */
58      public ChiSquaredDistribution(double degreesOfFreedom,
59                                    double inverseCumAccuracy) {
60          gamma = new GammaDistribution(degreesOfFreedom / 2, 2);
61          solverAbsoluteAccuracy = inverseCumAccuracy;
62      }
63  
64      /**
65       * Access the number of degrees of freedom.
66       *
67       * @return the degrees of freedom.
68       */
69      public double getDegreesOfFreedom() {
70          return gamma.getShape() * 2.0;
71      }
72  
73      /** {@inheritDoc} */
74      @Override
75      public double density(double x) {
76          return gamma.density(x);
77      }
78  
79      /** {@inheritDoc} **/
80      @Override
81      public double logDensity(double x) {
82          return gamma.logDensity(x);
83      }
84  
85      /** {@inheritDoc} */
86      @Override
87      public double cumulativeProbability(double x)  {
88          return gamma.cumulativeProbability(x);
89      }
90  
91      /** {@inheritDoc} */
92      @Override
93      protected double getSolverAbsoluteAccuracy() {
94          return solverAbsoluteAccuracy;
95      }
96  
97      /**
98       * {@inheritDoc}
99       *
100      * For {@code k} degrees of freedom, the mean is {@code k}.
101      */
102     @Override
103     public double getNumericalMean() {
104         return getDegreesOfFreedom();
105     }
106 
107     /**
108      * {@inheritDoc}
109      *
110      * @return {@code 2 * k}, where {@code k} is the number of degrees of freedom.
111      */
112     @Override
113     public double getNumericalVariance() {
114         return 2 * getDegreesOfFreedom();
115     }
116 
117     /**
118      * {@inheritDoc}
119      *
120      * The lower bound of the support is always 0 no matter the
121      * degrees of freedom.
122      *
123      * @return zero.
124      */
125     @Override
126     public double getSupportLowerBound() {
127         return 0;
128     }
129 
130     /**
131      * {@inheritDoc}
132      *
133      * The upper bound of the support is always positive infinity no matter the
134      * degrees of freedom.
135      *
136      * @return {@code Double.POSITIVE_INFINITY}.
137      */
138     @Override
139     public double getSupportUpperBound() {
140         return Double.POSITIVE_INFINITY;
141     }
142 
143     /**
144      * {@inheritDoc}
145      *
146      * The support of this distribution is connected.
147      *
148      * @return {@code true}
149      */
150     @Override
151     public boolean isSupportConnected() {
152         return true;
153     }
154 }
