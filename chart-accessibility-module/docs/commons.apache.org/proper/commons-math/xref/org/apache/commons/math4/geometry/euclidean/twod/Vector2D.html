32   * <p>Instances of this class are guaranteed to be immutable.</p>

62      /** Simple constructor.
63       * Build a vector from its coordinates
64       * @param x abscissa
65       * @param y ordinate
66       * @see #getX()
67       * @see #getY()
68       */
69      public Vector2D(double x, double y) {
70          this.x = x;
71          this.y = y;
72      }

74      /** Simple constructor.
75       * Build a vector from its coordinates
76       * @param v coordinates array
77       * @exception DimensionMismatchException if array does not have 2 elements
78       * @see #toArray()
79       */
80      public Vector2D(double[] v) throws DimensionMismatchException {
81          if (v.length != 2) {
82              throw new DimensionMismatchException(v.length, 2);
83          }
84          this.x = v[0];
85          this.y = v[1];
86      }

88      /** Multiplicative constructor
89       * Build a vector from another one and a scale factor.
90       * The vector built will be a * u
91       * @param a scale factor
92       * @param u base (unscaled) vector
93       */
94      public Vector2D(double a, Vector2D u) {
95          this.x = a * u.x;
96          this.y = a * u.y;
97      }

99      /** Linear constructor
100      * Build a vector from two other ones and corresponding scale factors.
101      * The vector built will be a1 * u1 + a2 * u2
102      * @param a1 first scale factor
103      * @param u1 first base (unscaled) vector
104      * @param a2 second scale factor
105      * @param u2 second base (unscaled) vector
106      */
107     public Vector2D(double a1, Vector2D u1, double a2, Vector2D u2) {
108         this.x = a1 * u1.x + a2 * u2.x;
109         this.y = a1 * u1.y + a2 * u2.y;
110     }
111 
112     /** Linear constructor
113      * Build a vector from three other ones and corresponding scale factors.
114      * The vector built will be a1 * u1 + a2 * u2 + a3 * u3
115      * @param a1 first scale factor
116      * @param u1 first base (unscaled) vector
117      * @param a2 second scale factor
118      * @param u2 second base (unscaled) vector
119      * @param a3 third scale factor
120      * @param u3 third base (unscaled) vector
121      */
122     public Vector2D(double a1, Vector2D u1, double a2, Vector2D u2,
123                    double a3, Vector2D u3) {
124         this.x = a1 * u1.x + a2 * u2.x + a3 * u3.x;
125         this.y = a1 * u1.y + a2 * u2.y + a3 * u3.y;
126     }
127 
128     /** Linear constructor
129      * Build a vector from four other ones and corresponding scale factors.
130      * The vector built will be a1 * u1 + a2 * u2 + a3 * u3 + a4 * u4
131      * @param a1 first scale factor
132      * @param u1 first base (unscaled) vector
133      * @param a2 second scale factor
134      * @param u2 second base (unscaled) vector
135      * @param a3 third scale factor
136      * @param u3 third base (unscaled) vector
137      * @param a4 fourth scale factor
138      * @param u4 fourth base (unscaled) vector
139      */
140     public Vector2D(double a1, Vector2D u1, double a2, Vector2D u2,
141                    double a3, Vector2D u3, double a4, Vector2D u4) {
142         this.x = a1 * u1.x + a2 * u2.x + a3 * u3.x + a4 * u4.x;
143         this.y = a1 * u1.y + a2 * u2.y + a3 * u3.y + a4 * u4.y;
144     }

234     /** {@inheritDoc} */
235     @Override
236     public Vector2D normalize() throws MathArithmeticException {
237         double s = getNorm();
238         if (s == 0) {
239             throw new MathArithmeticException(LocalizedFormats.CANNOT_NORMALIZE_A_ZERO_NORM_VECTOR);
240         }
241         return scalarMultiply(1 / s);
242     }

244     /** Compute the angular separation between two vectors.
245      * <p>This method computes the angular separation between two
246      * vectors using the dot product for well separated vectors and the
247      * cross product for almost aligned vectors. This allows to have a
248      * good accuracy in all cases, even for vectors very close to each
249      * other.</p>
250      * @param v1 first vector
251      * @param v2 second vector
252      * @return angular separation between v1 and v2
253      * @exception MathArithmeticException if either vector has a null norm
254      */
255     public static double angle(Vector2D v1, Vector2D v2) throws MathArithmeticException {
256 
257         double normProduct = v1.getNorm() * v2.getNorm();
258         if (normProduct == 0) {
259             throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
260         }
261 
262         double dot = v1.dotProduct(v2);
263         double threshold = normProduct * 0.9999;
264         if ((dot < -threshold) || (dot > threshold)) {
265             // the vectors are almost aligned, compute using the sine
266             final double n = FastMath.abs(MathArrays.linearCombination(v1.x, v2.y, -v1.y, v2.x));
267             if (dot >= 0) {
268                 return FastMath.asin(n / normProduct);
269             }
270             return FastMath.PI - FastMath.asin(n / normProduct);
271         }
272 
273         // the vectors are sufficiently separated to use the cosine
274         return FastMath.acos(dot / normProduct);
275 
276     }

302     /** {@inheritDoc} */
303     @Override
304     public double distance1(Vector<Euclidean2D> p) {
305         Vector2D p3 = (Vector2D) p;
306         final double dx = FastMath.abs(p3.x - x);
307         final double dy = FastMath.abs(p3.y - y);
308         return dx + dy;
309     }

311     /** {@inheritDoc} */
312     @Override
313     public double distance(Point<Euclidean2D> p) {
314         Vector2D p3 = (Vector2D) p;
315         final double dx = p3.x - x;
316         final double dy = p3.y - y;
317         return FastMath.sqrt(dx * dx + dy * dy);
318     }

321     @Override
322     public double distanceInf(Vector<Euclidean2D> p) {
323         Vector2D p3 = (Vector2D) p;
324         final double dx = FastMath.abs(p3.x - x);
325         final double dy = FastMath.abs(p3.y - y);
326         return FastMath.max(dx, dy);
327     }

345     /**
346      * Compute the cross-product of the instance and the given points.
347      * <p>
348      * The cross product can be used to determine the location of a point
349      * with regard to the line formed by (p1, p2) and is calculated as:
350      * \[
351      *    P = (x_2 - x_1)(y_3 - y_1) - (y_2 - y_1)(x_3 - x_1)
352      * \]
353      * with \(p3 = (x_3, y_3)\) being this instance.
354      * <p>
355      * If the result is 0, the points are collinear, i.e. lie on a single straight line L;
356      * if it is positive, this point lies to the left, otherwise to the right of the line
357      * formed by (p1, p2).
358      *
359      * @param p1 first point of the line
360      * @param p2 second point of the line
361      * @return the cross-product
362      *
363      * @see <a href="http://en.wikipedia.org/wiki/Cross_product">Cross product (Wikipedia)</a>
364      */
365     public double crossProduct(final Vector2D p1, final Vector2D p2) {
366         final double x1 = p2.getX() - p1.getX();
367         final double y1 = getY() - p1.getY();
368         final double x2 = getX() - p1.getX();
369         final double y2 = p2.getY() - p1.getY();
370         return MathArrays.linearCombination(x1, y1, -x2, y2);
371     }

409     /**
410      * Test for the equality of two 2D vectors.
411      * <p>
412      * If all coordinates of two 2D vectors are exactly the same, and none are
413      * <code>Double.NaN</code>, the two 2D vectors are considered to be equal.
414      * </p>
415      * <p>
416      * <code>NaN</code> coordinates are considered to affect globally the vector
417      * and be equals to each other - i.e, if either (or all) coordinates of the
418      * 2D vector are equal to <code>Double.NaN</code>, the 2D vector is equal to
419      * {@link #NaN}.
420      * </p>
421      *
422      * @param other Object to test for equality to this
423      * @return true if two 2D vector objects are equal, false if
424      *         object is null, not an instance of Vector2D, or
425      *         not equal to this Vector2D instance
426      *
427      */
428     @Override
429     public boolean equals(Object other) {
430 
431         if (this == other) {
432             return true;
433         }
434 
435         if (other instanceof Vector2D) {
436             final Vector2D rhs = (Vector2D)other;
437             if (rhs.isNaN()) {
438                 return this.isNaN();
439             }
440 
441             return (x == rhs.x) && (y == rhs.y);
442         }
443         return false;
444     }

446     /**
447      * Get a hashCode for the 2D vector.
448      * <p>
449      * All NaN values have the same hash code.</p>
450      *
451      * @return a hash code value for this object
452      */
453     @Override
454     public int hashCode() {
455         if (isNaN()) {
456             return 542;
457         }
458         return 122 * (76 * MathUtils.hash(x) +  MathUtils.hash(y));
459     }
