56       * Wraps an instance of the JDK's {@link Random} class.
57       * The actual generation of random numbers will be delegated to that
58       * instance.
59       * <p>
60       * If cryptographically secure data is required, one can use this
61       * factory method, with an instance of the {@link java.security.SecureRandom}
62       * class as the argument.
63       * Note that data generation will be much slower in this case.
64       * </p>
65       *
66       * @param rng Underlying generator. Reference is copied so the RNG
67       * is shared with the caller.
68       * @return a {@link DataGenerator data generator}.
69       */
70      public static DataGenerator createDataGenerator(final Random rng) {
71          return createDataGenerator(asUniformRandomProvider(rng));
72      }

74      /**
75       * Wraps a {@link Random} instance.
76       *
77       * @param rng JDK {@link Random} instance to which the random number
78       * generation is delegated. Reference is copied so the RNG is shared
79       * with the caller.
80       * @return a {@link UniformRandomProvider} instance.
81       */
82      public static UniformRandomProvider asUniformRandomProvider(final Random rng) {
83          return new UniformRandomProvider() {
84              /** {@inheritDoc} */
85              @Override
86              public void nextBytes(byte[] bytes) {
87                  rng.nextBytes(bytes);
88              }
89  
90              /** {@inheritDoc} */
91              @Override
92              public void nextBytes(byte[] bytes,
93                                    int start,
94                                    int len) {
95                  final byte[] reduced = new byte[len];
96                  rng.nextBytes(reduced);
97                  System.arraycopy(reduced, 0, bytes, start, len);
98              }
99  
100             /** {@inheritDoc} */
101             @Override
102             public int nextInt() {
103                 return rng.nextInt();
104             }
105 
106             /** {@inheritDoc} */
107             @Override
108             public int nextInt(int n) {
109                 if (n <= 0) {
110                     throw new NotStrictlyPositiveException(n);
111                 }
112                 return rng.nextInt(n);
113             }
114 
115             /** {@inheritDoc} */
116             @Override
117             public long nextLong() {
118                 return rng.nextLong();
119             }
120 
121             /** {@inheritDoc} */
122             @Override
123             public long nextLong(long n) {
124                 // Code copied from "o.a.c.m.rng.internal.BaseProvider".
125 
126                 if (n > 0) {
127                     long bits;
128                     long val;
129                     do {
130                         bits = rng.nextLong() >>> 1;
131                         val  = bits % n;
132                     } while (bits - val + (n - 1) < 0);
133                     return val;
134                 }
135 
136                 throw new NotStrictlyPositiveException(n);
137             }
138 
139             /** {@inheritDoc} */
140             @Override
141             public boolean nextBoolean() {
142                 return rng.nextBoolean();
143             }
144 
145             /** {@inheritDoc} */
146             @Override
147             public float nextFloat() {
148                 return rng.nextFloat();
149             }
150 
151             /** {@inheritDoc} */
152             @Override
153             public double nextDouble() {
154                 return rng.nextDouble();
155             }
156         };
157     }

173         /**
174          * Generates a random string of hex characters of length {@code len}.
175          *
176          * <strong>Algorithm Description:</strong> how hexadecimal strings are
177          * generated depends on the value of the {@code useSha1} argument.
178          *
179          * <ul>
180          *  <li>If {@code useSha1 == false}, a 2-step process is used:
181          *   <ol>
182          *    <li>
183          *     {@code len / 2 + 1} binary bytes are generated using the underlying
184          *     generator.
185          *    </li>
186          *    <li>
187          *     Each binary byte is translated into 2 hex digits.
188          *    </li>
189          *   </ol>
190          *  </li>
191          *  <li>
192          *   If {@code useSha1 == true}, hex strings are generated in 40-byte
193          *   segments using a 3-step process:
194          *   <ol>
195          *    <li>
196          *     20 random bytes are generated using the underlying generator.
197          *    </li>
198          *    <li>
199          *     SHA-1 hash is applied to yield a 20-byte binary digest.
200          *    </li>
201          *    <li>
202          *     Each byte of the binary digest is converted to 2 hex digits.
203          *    </li>
204          *   </ol>
205          *  </li>
206          * </ul>
207          *
208          * @param len Length of the generated string.
209          * @param useSha1 Whether to use a digest.
210          * If {@code true} (resp. {@code false}), the 3-step (resp. 2-step)
211          * process will be used.
212          * @return the random string.
213          * @throws NotStrictlyPositiveException if {@code len <= 0}.
214          */
215         public String nextHexString(int len,
216                                     boolean useSha1) {
217             if (len <= 0) {
218                 throw new NotStrictlyPositiveException(LocalizedFormats.LENGTH, len);
219             }
220 
221             // Initialize output buffer.
222             final StringBuilder outBuffer = new StringBuilder();
223 
224             if (!useSha1) {
225                 // Generate int(len/2)+1 random bytes.
226                 final byte[] randomBytes = new byte[(len / 2) + 1];
227                 rng.nextBytes(randomBytes);
228 
229                 // Convert each byte to 2 hex digits.
230                 for (int i = 0; i < randomBytes.length; i++) {
231                     final Integer c = Integer.valueOf(randomBytes[i]);
232 
233                     // Add 128 to byte value to make interval 0-255 before
234                     // conversion to hex.
235                     // This guarantees <= 2 hex digits from "toHexString".
236                     // "toHexString" would otherwise add 2^32 to negative arguments.
237                     String hex = Integer.toHexString(c.intValue() + 128);
238 
239                     // Make sure we add 2 hex digits for each byte.
240                     if (hex.length() == 1) {
241                         hex = "0" + hex;
242                     }
243                     outBuffer.append(hex);
244                 }
245             } else {
246                 MessageDigest alg = null;
247                 try {
248                     alg = MessageDigest.getInstance("SHA-1");
249                 } catch (NoSuchAlgorithmException ex) {
250                     // Should never happen.
251                     throw new MathInternalError(ex);
252                 }
253                 alg.reset();
254 
255                 // Compute number of iterations required (40 bytes each).
256                 final int numIter = (len / 40) + 1;
257 
258                 for (int iter = 1; iter < numIter + 1; iter++) {
259                     final byte[] randomBytes = new byte[40];
260                     rng.nextBytes(randomBytes);
261                     alg.update(randomBytes);
262 
263                     // Create 20-byte binary hash.
264                     final byte[] hash = alg.digest();
265 
266                     // Loop over the hash, converting each byte to 2 hex digits
267                     for (int i = 0; i < hash.length; i++) {
268                         final Integer c = Integer.valueOf(hash[i]);
269 
270                         // Add 128 to byte value to make interval 0-255.
271                         // This guarantees <= 2 hex digits from "toHexString".
272                         // "toHexString" would otherwise add 2^32 to negative arguments.
273                         String hex = Integer.toHexString(c.intValue() + 128);
274 
275                         // Keep strings uniform length: guarantees 40 bytes.
276                         if (hex.length() == 1) {
277                             hex = "0" + hex;
278                         }
279                         outBuffer.append(hex);
280                     }
281                 }
282             }
283 
284             return outBuffer.toString().substring(0, len);
285         }

287         /**
288          * Generates a uniformly distributed random long integer between {@code lower}
289          * and {@code upper} (endpoints included).
290          *
291          * @param lower Lower bound for generated long integer.
292          * @param upper Upper bound for generated long integer.
293          * @return a random long integer greater than or equal to {@code lower}
294          * and less than or equal to {@code upper}
295          * @throws NumberIsTooLargeException if {@code lower >= upper}
296          */
297         public long nextLong(final long lower,
298                              final long upper) {
299             if (lower >= upper) {
300                 throw new NumberIsTooLargeException(LocalizedFormats.LOWER_BOUND_NOT_BELOW_UPPER_BOUND,
301                                                     lower, upper, false);
302             }
303             final long max = (upper - lower) + 1;
304             if (max <= 0) {
305                 // Range is too wide to fit in a positive long (larger than 2^63);
306                 // as it covers more than half the long range, we use directly a
307                 // simple rejection method.
308                 while (true) {
309                     final long r = rng.nextLong();
310                     if (r >= lower && r <= upper) {
311                         return r;
312                     }
313                 }
314             } else if (max < Integer.MAX_VALUE){
315                 // We can shift the range and generate directly a positive int.
316                 return lower + rng.nextInt((int) max);
317             } else {
318                 // We can shift the range and generate directly a positive long.
319                 return lower + rng.nextLong(max);
320             }
321         }

323         /**
324          * Generates a uniformly distributed random value from the open interval
325          * {@code (lower, upper)} (i.e., endpoints excluded).
326          * <p>
327          * <strong>Definition</strong>:
328          * <a href="http://www.itl.nist.gov/div898/handbook/eda/section3/eda3662.htm">
329          * Uniform Distribution</a> {@code lower} and {@code upper - lower} are the
330          * <a href = "http://www.itl.nist.gov/div898/handbook/eda/section3/eda364.htm">
331          * location and scale parameters</a>, respectively.</p>
332          * <p>
333          * <strong>Algorithm Description</strong>: scales the output of
334          * Random.nextDouble(), but rejects 0 values (i.e., will generate another
335          * random double if Random.nextDouble() returns 0). This is necessary to
336          * provide a symmetric output interval (both endpoints excluded).
337          * </p>
338          *
339          * @param lower Lower bound of the support (excluded).
340          * @param upper Upper bound of the support (excluded).
341          * @return a uniformly distributed random value between lower and upper
342          * (both excluded).
343          * @throws NumberIsTooLargeException if {@code lower >= upper}.
344          * @throws NotFiniteNumberException if one of the bounds is infinite.
345          * @throws NotANumberException if one of the bounds is NaN.
346          */
347         public double nextUniform(double lower, double upper) {
348             return nextUniform(lower, upper, false);
349         }

351         /**
352          * Generates a uniformly distributed random value from the interval
353          * {@code (lower, upper)} or the interval {@code [lower, upper)}. The lower
354          * bound is thus optionally included, while the upper bound is always
355          * excluded.
356          * <p>
357          * <strong>Definition</strong>:
358          * <a href="http://www.itl.nist.gov/div898/handbook/eda/section3/eda3662.htm">
359          * Uniform Distribution</a> {@code lower} and {@code upper - lower} are the
360          * <a href = "http://www.itl.nist.gov/div898/handbook/eda/section3/eda364.htm">
361          * location and scale parameters</a>, respectively.</p>
362          * <p>
363          * <strong>Algorithm Description</strong>: if the lower bound is excluded,
364          * scales the output of "nextDouble()", but rejects 0 values (i.e. it
365          * will generate another random double if "nextDouble()" returns 0).
366          * This is necessary to provide a symmetric output interval (both
367          * endpoints excluded).
368          * </p>
369          *
370          * @param lower Lower bound of the support.
371          * @param upper Exclusive upper bound of the support.
372          * @param lowerInclusive {@code true} if the lower bound is inclusive.
373          * @return a uniformly distributed random value in the {@code (lower, upper)}
374          * interval, if {@code lowerInclusive} is {@code false}, or in the
375          * {@code [lower, upper)} interval, if {@code lowerInclusive} is
376          * {@code true}.
377          * @throws NumberIsTooLargeException if {@code lower >= upper}.
378          * @throws NotFiniteNumberException if one of the bounds is infinite.
379          * @throws NotANumberException if one of the bounds is NaN.
380          */
381         public double nextUniform(double lower,
382                                   double upper,
383                                   boolean lowerInclusive) {
384             if (lower >= upper) {
385                 throw new NumberIsTooLargeException(LocalizedFormats.LOWER_BOUND_NOT_BELOW_UPPER_BOUND,
386                                                     lower, upper, false);
387             }
388             if (Double.isInfinite(lower)) {
389                 throw new NotFiniteNumberException(LocalizedFormats.INFINITE_BOUND, lower);
390             }
391             if (Double.isInfinite(upper)) {
392                 throw new NotFiniteNumberException(LocalizedFormats.INFINITE_BOUND, upper);
393             }
394             if (Double.isNaN(lower) || Double.isNaN(upper)) {
395                 throw new NotANumberException();
396             }
397 
398             // Ensure nextDouble() isn't 0.0
399             double u = rng.nextDouble();
400             while (!lowerInclusive && u <= 0.0) {
401                 u = rng.nextDouble();
402             }
403 
404             return u * upper + (1.0 - u) * lower;
405         }

407         /**
408          * Generates an integer array of length {@code k} whose entries are selected
409          * randomly, without repetition, from the integers {@code 0, ..., n - 1}
410          * (inclusive).
411          * <p>
412          * Generated arrays represent permutations of {@code n} taken {@code k} at a
413          * time.
414          * </p>
415          * <p>
416          * This method calls {@link MathArrays#shuffle(int[],UniformRandomProvider)
417          * MathArrays.shuffle} in order to create a random shuffle of the set
418          * of natural numbers {@code { 0, 1, ..., n - 1 }}.
419          * </p>
420          *
421          * @param n Domain of the permutation.
422          * @param k Size of the permutation.
423          * @return a random {@code k}-permutation of {@code n}, as an array of
424          * integers.
425          * @throws NumberIsTooLargeException if {@code k > n}.
426          * @throws NotStrictlyPositiveException if {@code k <= 0}.
427          */
428         public int[] nextPermutation(int n,
429                                      int k)
430             throws NumberIsTooLargeException, NotStrictlyPositiveException {
431             if (k > n) {
432                 throw new NumberIsTooLargeException(LocalizedFormats.PERMUTATION_EXCEEDS_N,
433                                                     k, n, true);
434             }
435             if (k <= 0) {
436                 throw new NotStrictlyPositiveException(LocalizedFormats.PERMUTATION_SIZE,
437                                                        k);
438             }
439 
440             final int[] index = MathArrays.natural(n);
441             MathArrays.shuffle(index, rng);
442 
443             // Return a new array containing the first "k" entries of "index".
444             return MathArrays.copyOf(index, k);
445         }

447         /**
448          * Returns a list of {@code k} objects selected randomly from the
449          * given {@code collection}.
450          *
451          * <p>
452          * Sampling is without replacement; but if {@code collection} contains
453          * identical objects, the sample may include repeats.  If all elements
454          * are distinct, the resulting object array represents a
455          * <a href="http://rkb.home.cern.ch/rkb/AN16pp/node250.html#SECTION0002500000000000000000">
456          * Simple Random Sample</a> of size {@code k} from the elements of
457          * the {@code collection}.
458          * </p>
459          * <p>
460          * This method calls {@link #nextPermutation(int,int) nextPermutation(c.size(), k)}
461          * in order to sample the collection.
462          * </p>
463          *
464          * @param <T> Type of objects held in the {@code collection}.
465          * @param collection Collection to be sampled.
466          * @param k Size of the sample.
467          * @return a random sample of {@code k} elements from the {@code collection}.
468          * @throws NumberIsTooLargeException if {@code k > collection.size()}.
469          * @throws NotStrictlyPositiveException if {@code k <= 0}.
470          */
471         public <T> List<T> nextSample(Collection<T> collection,
472                                       int k) {
473             final int len = collection.size();
474             if (k > len) {
475                 throw new NumberIsTooLargeException(LocalizedFormats.SAMPLE_SIZE_EXCEEDS_COLLECTION_SIZE,
476                                                     k, len, true);
477             }
478             if (k <= 0) {
479                 throw new NotStrictlyPositiveException(LocalizedFormats.NUMBER_OF_SAMPLES, k);
480             }
481 
482             final ArrayList<T> objects = new ArrayList<T>(collection);
483             final int[] index = nextPermutation(len, k);
484             final List<T> result = new ArrayList<T>(k);
485             for (int i = 0; i < k; i++) {
486                 result.add(objects.get(index[i]));
487             }
488 
489             return result;
490         }
