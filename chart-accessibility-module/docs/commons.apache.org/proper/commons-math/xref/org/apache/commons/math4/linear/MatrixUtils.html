67      /**
68       * Returns a {@link RealMatrix} with specified dimensions.
69       * <p>The type of matrix returned depends on the dimension. Below
70       * 2<sup>12</sup> elements (i.e. 4096 elements or 64&times;64 for a
71       * square matrix) which can be stored in a 32kB array, a {@link
72       * Array2DRowRealMatrix} instance is built. Above this threshold a {@link
73       * BlockRealMatrix} instance is built.</p>
74       * <p>The matrix elements are all set to 0.0.</p>
75       * @param rows number of rows of the matrix
76       * @param columns number of columns of the matrix
77       * @return  RealMatrix with specified dimensions
78       * @see #createRealMatrix(double[][])
79       */
80      public static RealMatrix createRealMatrix(final int rows, final int columns) {
81          return (rows * columns <= 4096) ?
82                  new Array2DRowRealMatrix(rows, columns) : new BlockRealMatrix(rows, columns);
83      }

183     /**
184      * Returns <code>dimension x dimension</code> identity matrix.
185      *
186      * @param <T> the type of the field elements
187      * @param field field to which the elements belong
188      * @param dimension dimension of identity matrix to generate
189      * @return identity matrix
190      * @throws IllegalArgumentException if dimension is not positive
191      * @since 2.0
192      */
193     public static <T extends FieldElement<T>> FieldMatrix<T>
194         createFieldIdentityMatrix(final Field<T> field, final int dimension) {
195         final T zero = field.getZero();
196         final T one  = field.getOne();
197         final T[][] d = MathArrays.buildArray(field, dimension, dimension);
198         for (int row = 0; row < dimension; row++) {
199             final T[] dRow = d[row];
200             Arrays.fill(dRow, zero);
201             dRow[row] = one;
202         }
203         return new Array2DRowFieldMatrix<T>(field, d, false);
204     }

392     /**
393      * Checks whether a matrix is symmetric, within a given relative tolerance.
394      *
395      * @param matrix Matrix to check.
396      * @param relativeTolerance Tolerance of the symmetry check.
397      * @param raiseException If {@code true}, an exception will be raised if
398      * the matrix is not symmetric.
399      * @return {@code true} if {@code matrix} is symmetric.
400      * @throws NonSquareMatrixException if the matrix is not square.
401      * @throws NonSymmetricMatrixException if the matrix is not symmetric.
402      */
403     private static boolean isSymmetricInternal(RealMatrix matrix,
404                                                double relativeTolerance,
405                                                boolean raiseException) {
406         final int rows = matrix.getRowDimension();
407         if (rows != matrix.getColumnDimension()) {
408             if (raiseException) {
409                 throw new NonSquareMatrixException(rows, matrix.getColumnDimension());
410             } else {
411                 return false;
412             }
413         }
414         for (int i = 0; i < rows; i++) {
415             for (int j = i + 1; j < rows; j++) {
416                 final double mij = matrix.getEntry(i, j);
417                 final double mji = matrix.getEntry(j, i);
418                 if (FastMath.abs(mij - mji) >
419                     FastMath.max(FastMath.abs(mij), FastMath.abs(mji)) * relativeTolerance) {
420                     if (raiseException) {
421                         throw new NonSymmetricMatrixException(i, j, relativeTolerance);
422                     } else {
423                         return false;
424                     }
425                 }
426             }
427         }
428         return true;
429     }

505     /**
506      * Check if submatrix ranges indices are valid.
507      * Rows and columns are indicated counting from 0 to {@code n - 1}.
508      *
509      * @param m Matrix.
510      * @param startRow Initial row index.
511      * @param endRow Final row index.
512      * @param startColumn Initial column index.
513      * @param endColumn Final column index.
514      * @throws OutOfRangeException if the indices are invalid.
515      * @throws NumberIsTooSmallException if {@code endRow < startRow} or
516      * {@code endColumn < startColumn}.
517      */
518     public static void checkSubMatrixIndex(final AnyMatrix m,
519                                            final int startRow, final int endRow,
520                                            final int startColumn, final int endColumn)
521         throws NumberIsTooSmallException, OutOfRangeException {
522         checkRowIndex(m, startRow);
523         checkRowIndex(m, endRow);
524         if (endRow < startRow) {
525             throw new NumberIsTooSmallException(LocalizedFormats.INITIAL_ROW_AFTER_FINAL_ROW,
526                                                 endRow, startRow, false);
527         }
528 
529         checkColumnIndex(m, startColumn);
530         checkColumnIndex(m, endColumn);
531         if (endColumn < startColumn) {
532             throw new NumberIsTooSmallException(LocalizedFormats.INITIAL_COLUMN_AFTER_FINAL_COLUMN,
533                                                 endColumn, startColumn, false);
534         }
535

539     /**
540      * Check if submatrix ranges indices are valid.
541      * Rows and columns are indicated counting from 0 to n-1.
542      *
543      * @param m Matrix.
544      * @param selectedRows Array of row indices.
545      * @param selectedColumns Array of column indices.
546      * @throws NullArgumentException if {@code selectedRows} or
547      * {@code selectedColumns} are {@code null}.
548      * @throws NoDataException if the row or column selections are empty (zero
549      * length).
550      * @throws OutOfRangeException if row or column selections are not valid.
551      */
552     public static void checkSubMatrixIndex(final AnyMatrix m,
553                                            final int[] selectedRows,
554                                            final int[] selectedColumns)
555         throws NoDataException, NullArgumentException, OutOfRangeException {
556         if (selectedRows == null) {
557             throw new NullArgumentException();
558         }
559         if (selectedColumns == null) {
560             throw new NullArgumentException();
561         }
562         if (selectedRows.length == 0) {
563             throw new NoDataException(LocalizedFormats.EMPTY_SELECTED_ROW_INDEX_ARRAY);
564         }
565         if (selectedColumns.length == 0) {
566             throw new NoDataException(LocalizedFormats.EMPTY_SELECTED_COLUMN_INDEX_ARRAY);
567         }
568 
569         for (final int row : selectedRows) {
570             checkRowIndex(m, row);
571         }
572         for (final int column : selectedColumns) {
573             checkColumnIndex(m, column);
574         }
575     }

639     /** Converter for {@link FieldMatrix}/{@link Fraction}. */
640     private static class FractionMatrixConverter extends DefaultFieldMatrixPreservingVisitor<Fraction> {
641         /** Converted array. */
642         private double[][] data;
643         /** Simple constructor. */
644         FractionMatrixConverter() {
645             super(Fraction.ZERO);
646         }
647 
648         /** {@inheritDoc} */
649         @Override
650         public void start(int rows, int columns,
651                           int startRow, int endRow, int startColumn, int endColumn) {
652             data = new double[rows][columns];
653         }
654 
655         /** {@inheritDoc} */
656         @Override
657         public void visit(int row, int column, Fraction value) {
658             data[row][column] = value.doubleValue();

925     /**Solve  a  system of composed of a Lower Triangular Matrix
926      * {@link RealMatrix}.
927      * <p>
928      * This method is called to solve systems of equations which are
929      * of the lower triangular form. The matrix {@link RealMatrix}
930      * is assumed, though not checked, to be in lower triangular form.
931      * The vector {@link RealVector} is overwritten with the solution.
932      * The matrix is checked that it is square and its dimensions match
933      * the length of the vector.
934      * </p>
935      * @param rm RealMatrix which is lower triangular
936      * @param b  RealVector this is overwritten
937      * @throws DimensionMismatchException if the matrix and vector are not
938      * conformable
939      * @throws NonSquareMatrixException if the matrix {@code rm} is not square
940      * @throws MathArithmeticException if the absolute value of one of the diagonal
941      * coefficient of {@code rm} is lower than {@link Precision#SAFE_MIN}
942      */
943     public static void solveLowerTriangularSystem(RealMatrix rm, RealVector b)
944         throws DimensionMismatchException, MathArithmeticException,
945         NonSquareMatrixException {
946         if ((rm == null) || (b == null) || ( rm.getRowDimension() != b.getDimension())) {
947             throw new DimensionMismatchException(
948                     (rm == null) ? 0 : rm.getRowDimension(),
949                     (b == null) ? 0 : b.getDimension());
950         }
951         if( rm.getColumnDimension() != rm.getRowDimension() ){
952             throw new NonSquareMatrixException(rm.getRowDimension(),
953                                                rm.getColumnDimension());
954         }
955         int rows = rm.getRowDimension();
956         for( int i = 0 ; i < rows ; i++ ){
957             double diag = rm.getEntry(i, i);
958             if( FastMath.abs(diag) < Precision.SAFE_MIN ){
959                 throw new MathArithmeticException(LocalizedFormats.ZERO_DENOMINATOR);
960             }
961             double bi = b.getEntry(i)/diag;
962             b.setEntry(i,  bi );
963             for( int j = i+1; j< rows; j++ ){
964                 b.setEntry(j, b.getEntry(j)-bi*rm.getEntry(j,i)  );
965             }
966         }
967     }

972      * This method is called to solve systems of equations which are

974      * is assumed, though not checked, to be in upper triangular form.
975      * The vector {@link RealVector} is overwritten with the solution.
976      * The matrix is checked that it is square and its dimensions match

984      * square
985      * @throws MathArithmeticException if the absolute value of one of the diagonal
986      * coefficient of {@code rm} is lower than {@link Precision#SAFE_MIN}
987      */
988     public static void solveUpperTriangularSystem(RealMatrix rm, RealVector b)
989         throws DimensionMismatchException, MathArithmeticException,
990         NonSquareMatrixException {
991         if ((rm == null) || (b == null) || ( rm.getRowDimension() != b.getDimension())) {
992             throw new DimensionMismatchException(
993                     (rm == null) ? 0 : rm.getRowDimension(),

1014     /**
1015      * Computes the inverse of the given matrix by splitting it into
1016      * 4 sub-matrices.
1017      *
1018      * @param m Matrix whose inverse must be computed.
1019      * @param splitIndex Index that determines the "split" line and
1020      * column.
1021      * The element corresponding to this index will part of the
1022      * upper-left sub-matrix.
1023      * @return the inverse of {@code m}.
1024      * @throws NonSquareMatrixException if {@code m} is not square.
1025      */
1026     public static RealMatrix blockInverse(RealMatrix m,
1027                                           int splitIndex) {
1028         final int n = m.getRowDimension();
1029         if (m.getColumnDimension() != n) {
1030             throw new NonSquareMatrixException(m.getRowDimension(),
1031                                                m.getColumnDimension());
1032         }
1033 
1034         final int splitIndex1 = splitIndex + 1;
1035 
1036         final RealMatrix a = m.getSubMatrix(0, splitIndex, 0, splitIndex);
1037         final RealMatrix b = m.getSubMatrix(0, splitIndex, splitIndex1, n - 1);
1038         final RealMatrix c = m.getSubMatrix(splitIndex1, n - 1, 0, splitIndex);
1039         final RealMatrix d = m.getSubMatrix(splitIndex1, n - 1, splitIndex1, n - 1);
1040 
1041         final SingularValueDecomposition aDec = new SingularValueDecomposition(a);
1042         final DecompositionSolver aSolver = aDec.getSolver();
1043         if (!aSolver.isNonSingular()) {
1044             throw new SingularMatrixException();
1045         }
1046         final RealMatrix aInv = aSolver.getInverse();
1047 
1048         final SingularValueDecomposition dDec = new SingularValueDecomposition(d);
1049         final DecompositionSolver dSolver = dDec.getSolver();
1050         if (!dSolver.isNonSingular()) {
1051             throw new SingularMatrixException();
1052         }
1053         final RealMatrix dInv = dSolver.getInverse();
1054 
1055         final RealMatrix tmp1 = a.subtract(b.multiply(dInv).multiply(c));
1056         final SingularValueDecomposition tmp1Dec = new SingularValueDecomposition(tmp1);
1057         final DecompositionSolver tmp1Solver = tmp1Dec.getSolver();
1058         if (!tmp1Solver.isNonSingular()) {
1059             throw new SingularMatrixException();
1060         }
1061         final RealMatrix result00 = tmp1Solver.getInverse();
1062 
1063         final RealMatrix tmp2 = d.subtract(c.multiply(aInv).multiply(b));
1064         final SingularValueDecomposition tmp2Dec = new SingularValueDecomposition(tmp2);
1065         final DecompositionSolver tmp2Solver = tmp2Dec.getSolver();
1066         if (!tmp2Solver.isNonSingular()) {
1067             throw new SingularMatrixException();
1068         }
1069         final RealMatrix result11 = tmp2Solver.getInverse();
1070 
1071         final RealMatrix result01 = aInv.multiply(b).multiply(result11).scalarMultiply(-1);
1072         final RealMatrix result10 = dInv.multiply(c).multiply(result00).scalarMultiply(-1);
1073 
1074         final RealMatrix result = new Array2DRowRealMatrix(n, n);
1075         result.setSubMatrix(result00.getData(), 0, 0);
1076         result.setSubMatrix(result01.getData(), 0, splitIndex1);
1077         result.setSubMatrix(result10.getData(), splitIndex1, 0);
1078         result.setSubMatrix(result11.getData(), splitIndex1, splitIndex1);
1079 
1080         return result;
1081     }

1106      * <p>
1107      * By default, the inverse of the matrix is computed using the QR-decomposition,
1108      * unless a more efficient method can be determined for the input matrix.
1109      *
1110      * @param matrix Matrix whose inverse shall be computed
1111      * @param threshold Singularity threshold
1112      * @return the inverse of {@code m}
1113      * @throws NullArgumentException if {@code matrix} is {@code null}
1114      * @throws SingularMatrixException if matrix is singular
1115      * @throws NonSquareMatrixException if matrix is not square
1116      * @since 3.3
1117      */

1119             throws NullArgumentException, SingularMatrixException, NonSquareMatrixException {
1120 
1121         MathUtils.checkNotNull(matrix);
1122 
1123         if (!matrix.isSquare()) {
1124             throw new NonSquareMatrixException(matrix.getRowDimension(),
1125                                                matrix.getColumnDimension());
1126         }
1127

1129             return ((DiagonalMatrix) matrix).inverse(threshold);
