18  package org.apache.commons.math4.fraction;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.Field;
23  import org.apache.commons.math4.FieldElement;
24  
25  /**
26   * Representation of the fractional numbers field.
27   * <p>
28   * This class is a singleton.
29   * </p>
30   * @see Fraction
31   * @since 2.0
32   */
33  public class FractionField implements Field<Fraction>, Serializable  {
34  
35      /** Serializable version identifier */
36      private static final long serialVersionUID = -1257768487499119313L;
37  
38      /** Private constructor for the singleton.
39       */
40      private FractionField() {
41      }
42  
43      /** Get the unique instance.
44       * @return the unique instance
45       */
46      public static FractionField getInstance() {
47          return LazyHolder.INSTANCE;
48      }
49  
50      /** {@inheritDoc} */
51      @Override
52      public Fraction getOne() {
53          return Fraction.ONE;
54      }
55  
56      /** {@inheritDoc} */
57      @Override
58      public Fraction getZero() {
59          return Fraction.ZERO;
60      }
61  
62      /** {@inheritDoc} */
63      @Override
64      public Class<? extends FieldElement<Fraction>> getRuntimeClass() {
65          return Fraction.class;
66      }
67      // CHECKSTYLE: stop HideUtilityClassConstructor
68      /** Holder for the instance.
69       * <p>We use here the Initialization On Demand Holder Idiom.</p>
70       */
71      private static class LazyHolder {
72          /** Cached field instance. */
73          private static final FractionField INSTANCE = new FractionField();
74      }
75      // CHECKSTYLE: resume HideUtilityClassConstructor
76  
77      /** Handle deserialization of the singleton.
78       * @return the singleton instance
79       */
80      private Object readResolve() {
81          // return the singleton instance
82          return LazyHolder.INSTANCE;
83      }
84  
85  }
