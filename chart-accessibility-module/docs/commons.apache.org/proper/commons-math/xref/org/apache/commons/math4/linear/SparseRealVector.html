17  package org.apache.commons.math4.linear;
18  
19  /**
20   * Marker class for RealVectors that require sparse backing storage
21   * <p>
22   *  Caveat: Implementation are allowed to assume that, for any {@code x},
23   *  the equality {@code x * 0d == 0d} holds. But it is is not true for
24   *  {@code NaN}. Moreover, zero entries will lose their sign.
25   *  Some operations (that involve {@code NaN} and/or infinities) may
26   *  thus give incorrect results, like multiplications, divisions or
27   *  functions mapping.
28   * </p>
29   * @since 2.0
30   */
31  public abstract class SparseRealVector extends RealVector {}
