18  package org.apache.commons.math4.analysis.differentiation;
19  
20  import org.apache.commons.math4.analysis.MultivariateVectorFunction;
21  import org.apache.commons.math4.exception.MathIllegalArgumentException;
22  
23  
24  /**
25   * Extension of {@link MultivariateVectorFunction} representing a
26   * multivariate differentiable vectorial function.
27   * @since 3.1
28   */
29  public interface MultivariateDifferentiableVectorFunction
30      extends MultivariateVectorFunction {
31  
32      /**
33       * Compute the value for the function at the given point.
34       * @param point point at which the function must be evaluated
35       * @return function value for the given point
36       * @exception MathIllegalArgumentException if {@code point} does not
37       * satisfy the function's constraints (wrong dimension, argument out of bound,
38       * or unsupported derivative order for example)
39       */
40      DerivativeStructure[] value(DerivativeStructure[] point)
41          throws MathIllegalArgumentException;
42  
43  }
