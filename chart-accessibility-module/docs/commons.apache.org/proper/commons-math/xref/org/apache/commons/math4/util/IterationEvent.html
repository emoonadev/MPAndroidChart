17  package org.apache.commons.math4.util;
18  
19  import java.util.EventObject;
20  
21  /**
22   * The root class from which all events occurring while running an
23   * {@link IterationManager} should be derived.
24   *
25   */
26  public class IterationEvent extends EventObject {
27      /** */
28      private static final long serialVersionUID = 20120128L;
29  
30      /** The number of iterations performed so far. */
31      private final int iterations;
32  
33      /**
34       * Creates a new instance of this class.
35       *
36       * @param source the iterative algorithm on which the event initially
37       * occurred
38       * @param iterations the number of iterations performed at the time
39       * {@code this} event is created
40       */
41      public IterationEvent(final Object source, final int iterations) {
42          super(source);
43          this.iterations = iterations;
44      }
45  
46      /**
47       * Returns the number of iterations performed at the time {@code this} event
48       * is created.
49       *
50       * @return the number of iterations performed
51       */
52      public int getIterations() {
53          return iterations;
54      }
55   }
