17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.LocalizedFormats;
20  
21  /**
22   * Exception to be thrown when the maximal number of iterations is exceeded.
23   *
24   * @since 3.1
25   */
26  public class TooManyIterationsException extends MaxCountExceededException {
27      /** Serializable version Id. */
28      private static final long serialVersionUID = 20121211L;
29  
30      /**
31       * Construct the exception.
32       *
33       * @param max Maximum number of evaluations.
34       */
35      public TooManyIterationsException(Number max) {
36          super(max);
37          getContext().addMessage(LocalizedFormats.ITERATIONS);
38      }
39  }
