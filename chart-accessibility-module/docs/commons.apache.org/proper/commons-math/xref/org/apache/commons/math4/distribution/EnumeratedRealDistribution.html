17  package org.apache.commons.math4.distribution;
18  
19  import java.util.ArrayList;
20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Map.Entry;
24  
25  import org.apache.commons.math4.exception.DimensionMismatchException;
26  import org.apache.commons.math4.exception.MathArithmeticException;
27  import org.apache.commons.math4.exception.NotANumberException;
28  import org.apache.commons.math4.exception.NotFiniteNumberException;
29  import org.apache.commons.math4.exception.NotPositiveException;
30  import org.apache.commons.math4.exception.OutOfRangeException;
31  import org.apache.commons.rng.UniformRandomProvider;
32  import org.apache.commons.math4.util.Pair;
33  
34  /**
35   * <p>Implementation of a real-valued {@link EnumeratedDistribution}.
36   *
37   * <p>Values with zero-probability are allowed but they do not extend the
38   * support.<br>
39   * Duplicate values are allowed. Probabilities of duplicate values are combined
40   * when computing cumulative probabilities and statistics.</p>
41   *
42   * @since 3.2
43   */
44  public class EnumeratedRealDistribution extends AbstractRealDistribution {
45  
46      /** Serializable UID. */
47      private static final long serialVersionUID = 20160311L;
48  
49      /**
50       * {@link EnumeratedDistribution} (using the {@link Double} wrapper)
51       * used to generate the pmf.
52       */
53      protected final EnumeratedDistribution<Double> innerDistribution;
54  
55      /**
56       * Create a discrete real-valued distribution using the given random number generator
57       * and probability mass function enumeration.
58       *
59       * @param singletons array of random variable values.
60       * @param probabilities array of probabilities.
61       * @throws DimensionMismatchException if
62       * {@code singletons.length != probabilities.length}
63       * @throws NotPositiveException if any of the probabilities are negative.
64       * @throws NotFiniteNumberException if any of the probabilities are infinite.
65       * @throws NotANumberException if any of the probabilities are NaN.
66       * @throws MathArithmeticException all of the probabilities are 0.
67       */
68      public EnumeratedRealDistribution(final double[] singletons,
69                                        final double[] probabilities)
70          throws DimensionMismatchException,
71                 NotPositiveException,
72                 MathArithmeticException,
73                 NotFiniteNumberException,
74                 NotANumberException {
75          innerDistribution = new EnumeratedDistribution<Double>(createDistribution(singletons, probabilities));
76      }
77  
78      /**
79       * Creates a discrete real-valued distribution from the input data.
80       * Values are assigned mass based on their frequency.
81       *
82       * @param data input dataset
83       */
84      public EnumeratedRealDistribution(final double[] data) {
85          final Map<Double, Integer> dataMap = new HashMap<Double, Integer>();
86          for (double value : data) {
87              Integer count = dataMap.get(value);
88              if (count == null) {
89                  count = 0;
90              }
91              dataMap.put(value, ++count);
92          }
93          final int massPoints = dataMap.size();
94          final double denom = data.length;
95          final double[] values = new double[massPoints];
96          final double[] probabilities = new double[massPoints];
97          int index = 0;
98          for (Entry<Double, Integer> entry : dataMap.entrySet()) {
99              values[index] = entry.getKey();
100             probabilities[index] = entry.getValue().intValue() / denom;
101             index++;
102         }
103         innerDistribution = new EnumeratedDistribution<Double>(createDistribution(values, probabilities));
104     }
105 
106     /**
107      * Create the list of Pairs representing the distribution from singletons and probabilities.
108      *
109      * @param singletons values
110      * @param probabilities probabilities
111      * @return list of value/probability pairs
112      */
113     private static List<Pair<Double, Double>>  createDistribution(double[] singletons, double[] probabilities) {
114         if (singletons.length != probabilities.length) {
115             throw new DimensionMismatchException(probabilities.length, singletons.length);
116         }
117 
118         final List<Pair<Double, Double>> samples = new ArrayList<Pair<Double, Double>>(singletons.length);
119 
120         for (int i = 0; i < singletons.length; i++) {
121             samples.add(new Pair<Double, Double>(singletons[i], probabilities[i]));
122         }
123         return samples;
124 
125     }
126 
127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public double probability(final double x) {
132         return innerDistribution.probability(x);
133     }
134 
135     /**
136      * For a random variable {@code X} whose values are distributed according to
137      * this distribution, this method returns {@code P(X = x)}. In other words,
138      * this method represents the probability mass function (PMF) for the
139      * distribution.
140      *
141      * @param x the point at which the PMF is evaluated
142      * @return the value of the probability mass function at point {@code x}
143      */
144     @Override
145     public double density(final double x) {
146         return probability(x);
147     }
148 
149     /**
150      * {@inheritDoc}
151      */
152     @Override
153     public double cumulativeProbability(final double x) {
154         double probability = 0;
155 
156         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
157             if (sample.getKey() <= x) {
158                 probability += sample.getValue();
159             }
160         }
161 
162         return probability;
163     }
164 
165     /**
166      * {@inheritDoc}
167      */
168     @Override
169     public double inverseCumulativeProbability(final double p) throws OutOfRangeException {
170         if (p < 0.0 || p > 1.0) {
171             throw new OutOfRangeException(p, 0, 1);
172         }
173 
174         double probability = 0;
175         double x = getSupportLowerBound();
176         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
177             if (sample.getValue() == 0.0) {
178                 continue;
179             }
180 
181             probability += sample.getValue();
182             x = sample.getKey();
183 
184             if (probability >= p) {
185                 break;
186             }
187         }
188 
189         return x;
190     }
191 
192     /**
193      * {@inheritDoc}
194      *
195      * @return {@code sum(singletons[i] * probabilities[i])}
196      */
197     @Override
198     public double getNumericalMean() {
199         double mean = 0;
200 
201         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
202             mean += sample.getValue() * sample.getKey();
203         }
204 
205         return mean;
206     }
207 
208     /**
209      * {@inheritDoc}
210      *
211      * @return {@code sum((singletons[i] - mean) ^ 2 * probabilities[i])}
212      */
213     @Override
214     public double getNumericalVariance() {
215         double mean = 0;
216         double meanOfSquares = 0;
217 
218         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
219             mean += sample.getValue() * sample.getKey();
220             meanOfSquares += sample.getValue() * sample.getKey() * sample.getKey();
221         }
222 
223         return meanOfSquares - mean * mean;
224     }
225 
226     /**
227      * {@inheritDoc}
228      *
229      * Returns the lowest value with non-zero probability.
230      *
231      * @return the lowest value with non-zero probability.
232      */
233     @Override
234     public double getSupportLowerBound() {
235         double min = Double.POSITIVE_INFINITY;
236         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
237             if (sample.getKey() < min && sample.getValue() > 0) {
238                 min = sample.getKey();
239             }
240         }
241 
242         return min;
243     }
244 
245     /**
246      * {@inheritDoc}
247      *
248      * Returns the highest value with non-zero probability.
249      *
250      * @return the highest value with non-zero probability.
251      */
252     @Override
253     public double getSupportUpperBound() {
254         double max = Double.NEGATIVE_INFINITY;
255         for (final Pair<Double, Double> sample : innerDistribution.getPmf()) {
256             if (sample.getKey() > max && sample.getValue() > 0) {
257                 max = sample.getKey();
258             }
259         }
260 
261         return max;
262     }
263 
264     /**
265      * {@inheritDoc}
266      *
267      * The support of this distribution is connected.
268      *
269      * @return {@code true}
270      */
271     @Override
272     public boolean isSupportConnected() {
273         return true;
274     }
275 
276     /** {@inheritDoc} */
277     @Override
278     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
279         return new RealDistribution.Sampler() {
280             /** Delegate. */
281             private final EnumeratedDistribution<Double>.Sampler inner =
282                 innerDistribution.createSampler(rng);
283 
284             /** {@inheritDoc} */
285             @Override
286             public double sample() {
287                 return inner.sample();
288             }
289         };
290     }
