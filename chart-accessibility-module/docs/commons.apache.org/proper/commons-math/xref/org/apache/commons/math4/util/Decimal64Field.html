17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.Field;
20  import org.apache.commons.math4.FieldElement;
21  
22  /**
23   * The field of double precision floating-point numbers.
24   *
25   * @since 3.1
26   * @see Decimal64
27   */
28  public class Decimal64Field implements Field<Decimal64> {
29  
30      /** The unique instance of this class. */
31      private static final Decimal64Field INSTANCE = new Decimal64Field();
32  
33      /** Default constructor. */
34      private Decimal64Field() {
35          // Do nothing
36      }
37  
38      /**
39       * Returns the unique instance of this class.
40       *
41       * @return the unique instance of this class
42       */
43      public static final Decimal64Field getInstance() {
44          return INSTANCE;
45      }
46  
47      /** {@inheritDoc} */
48      @Override
49      public Decimal64 getZero() {
50          return Decimal64.ZERO;
51      }
52  
53      /** {@inheritDoc} */
54      @Override
55      public Decimal64 getOne() {
56          return Decimal64.ONE;
57      }
58  
59      /** {@inheritDoc} */
60      @Override
61      public Class<? extends FieldElement<Decimal64>> getRuntimeClass() {
62          return Decimal64.class;
63      }
64  }
