18  package org.apache.commons.math4.util;
19  
20  import java.util.NoSuchElementException;
21  
22  import org.apache.commons.math4.exception.DimensionMismatchException;
23  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
24  import org.apache.commons.math4.exception.OutOfRangeException;
25  
26  /**
27   * Converter between unidimensional storage structure and multidimensional
28   * conceptual structure.
29   * This utility will convert from indices in a multidimensional structure
30   * to the corresponding index in a one-dimensional array. For example,
31   * assuming that the ranges (in 3 dimensions) of indices are 2, 4 and 3,
32   * the following correspondences, between 3-tuples indices and unidimensional
33   * indices, will hold:
34   * <ul>
35   *  <li>(0, 0, 0) corresponds to 0</li>
36   *  <li>(0, 0, 1) corresponds to 1</li>
37   *  <li>(0, 0, 2) corresponds to 2</li>
38   *  <li>(0, 1, 0) corresponds to 3</li>
39   *  <li>...</li>
40   *  <li>(1, 0, 0) corresponds to 12</li>
41   *  <li>...</li>
42   *  <li>(1, 3, 2) corresponds to 23</li>
43   * </ul>
44   *
45   * @since 2.2
46   */
47  public class MultidimensionalCounter implements Iterable<Integer> {
48      /**
49       * Number of dimensions.
50       */
51      private final int dimension;
52      /**
53       * Offset for each dimension.
54       */
55      private final int[] uniCounterOffset;
56      /**
57       * Counter sizes.
58       */
59      private final int[] size;
60      /**
61       * Total number of (one-dimensional) slots.
62       */
63      private final int totalSize;
64      /**
65       * Index of last dimension.
66       */
67      private final int last;
68  
69      /**
70       * Perform iteration over the multidimensional counter.
71       */
72      public class Iterator implements java.util.Iterator<Integer> {
73          /**
74           * Multidimensional counter.
75           */
76          private final int[] counter = new int[dimension];
77          /**
78           * Unidimensional counter.
79           */
80          private int count = -1;
81          /**
82           * Maximum value for {@link #count}.
83           */
84          private final int maxCount = totalSize - 1;
85  
86          /**
87           * Create an iterator
88           * @see #iterator()
89           */
90          Iterator() {
91              counter[last] = -1;
92          }
93  
94          /**
95           * {@inheritDoc}
96           */
97          @Override
98          public boolean hasNext() {
99              return count < maxCount;
100         }
101 
102         /**
103          * @return the unidimensional count after the counter has been
104          * incremented by {@code 1}.
105          * @throws NoSuchElementException if {@link #hasNext()} would have
106          * returned {@code false}.
107          */
108         @Override
109         public Integer next() {
110             if (!hasNext()) {
111                 throw new NoSuchElementException();
112             }
113 
114             for (int i = last; i >= 0; i--) {
115                 if (counter[i] == size[i] - 1) {
116                     counter[i] = 0;
117                 } else {
118                     ++counter[i];
119                     break;
120                 }
121             }
122 
123             return ++count;
124         }
125 
126         /**
127          * Get the current unidimensional counter slot.
128          *
129          * @return the index within the unidimensionl counter.
130          */
131         public int getCount() {
132             return count;
133         }
134         /**
135          * Get the current multidimensional counter slots.
136          *
137          * @return the indices within the multidimensional counter.
138          */
139         public int[] getCounts() {
140             return MathArrays.copyOf(counter);
141         }
142 
143         /**
144          * Get the current count in the selected dimension.
145          *
146          * @param dim Dimension index.
147          * @return the count at the corresponding index for the current state
148          * of the iterator.
149          * @throws IndexOutOfBoundsException if {@code index} is not in the
150          * correct interval (as defined by the length of the argument in the
151          * {@link MultidimensionalCounter#MultidimensionalCounter(int[])
152          * constructor of the enclosing class}).
153          */
154         public int getCount(int dim) {
155             return counter[dim];
156         }
157 
158         /**
159          * @throws UnsupportedOperationException
160          */
161         @Override
162         public void remove() {
163             throw new UnsupportedOperationException();
164         }
165     }
166 
167     /**
168      * Create a counter.
169      *
170      * @param size Counter sizes (number of slots in each dimension).
171      * @throws NotStrictlyPositiveException if one of the sizes is
172      * negative or zero.
173      */
174     public MultidimensionalCounter(int ... size) throws NotStrictlyPositiveException {
175         dimension = size.length;
176         this.size = MathArrays.copyOf(size);
177 
178         uniCounterOffset = new int[dimension];
179 
180         last = dimension - 1;
181         int tS = size[last];
182         for (int i = 0; i < last; i++) {
183             int count = 1;
184             for (int j = i + 1; j < dimension; j++) {
185                 count *= size[j];
186             }
187             uniCounterOffset[i] = count;
188             tS *= size[i];
189         }
190         uniCounterOffset[last] = 0;
191 
192         if (tS <= 0) {
193             throw new NotStrictlyPositiveException(tS);
194         }
195 
196         totalSize = tS;
197     }
198 
199     /**
200      * Create an iterator over this counter.
201      *
202      * @return the iterator.
203      */
204     @Override
205     public Iterator iterator() {
206         return new Iterator();
207     }
208 
209     /**
210      * Get the number of dimensions of the multidimensional counter.
211      *
212      * @return the number of dimensions.
213      */
214     public int getDimension() {
215         return dimension;
216     }
217 
218     /**
219      * Convert to multidimensional counter.
220      *
221      * @param index Index in unidimensional counter.
222      * @return the multidimensional counts.
223      * @throws OutOfRangeException if {@code index} is not between
224      * {@code 0} and the value returned by {@link #getSize()} (excluded).
225      */
226     public int[] getCounts(int index) throws OutOfRangeException {
227         if (index < 0 ||
228             index >= totalSize) {
229             throw new OutOfRangeException(index, 0, totalSize);
230         }
231 
232         final int[] indices = new int[dimension];
233 
234         int count = 0;
235         for (int i = 0; i < last; i++) {
236             int idx = 0;
237             final int offset = uniCounterOffset[i];
238             while (count <= index) {
239                 count += offset;
240                 ++idx;
241             }
242             --idx;
243             count -= offset;
244             indices[i] = idx;
245         }
246 
247         indices[last] = index - count;
248 
249         return indices;
250     }
251 
252     /**
253      * Convert to unidimensional counter.
254      *
255      * @param c Indices in multidimensional counter.
256      * @return the index within the unidimensionl counter.
257      * @throws DimensionMismatchException if the size of {@code c}
258      * does not match the size of the array given in the constructor.
259      * @throws OutOfRangeException if a value of {@code c} is not in
260      * the range of the corresponding dimension, as defined in the
261      * {@link MultidimensionalCounter#MultidimensionalCounter(int...) constructor}.
262      */
263     public int getCount(int ... c)
264         throws OutOfRangeException, DimensionMismatchException {
265         if (c.length != dimension) {
266             throw new DimensionMismatchException(c.length, dimension);
267         }
268         int count = 0;
269         for (int i = 0; i < dimension; i++) {
270             final int index = c[i];
271             if (index < 0 ||
272                 index >= size[i]) {
273                 throw new OutOfRangeException(index, 0, size[i] - 1);
274             }
275             count += uniCounterOffset[i] * c[i];
276         }
277         return count + c[last];
278     }
279 
280     /**
281      * Get the total number of elements.
282      *
283      * @return the total size of the unidimensional counter.
284      */
285     public int getSize() {
286         return totalSize;
287     }
288     /**
289      * Get the number of multidimensional counter slots in each dimension.
290      *
291      * @return the sizes of the multidimensional counter in each dimension.
292      */
293     public int[] getSizes() {
294         return MathArrays.copyOf(size);
295     }
296 
297     /**
298      * {@inheritDoc}
299      */
300     @Override
301     public String toString() {
302         final StringBuilder sb = new StringBuilder();
303         for (int i = 0; i < dimension; i++) {
304             sb.append("[").append(getCount(i)).append("]");
305         }
306         return sb.toString();
307     }
