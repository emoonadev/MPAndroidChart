17  package org.apache.commons.math4.exception.util;
18  
19  import java.util.List;
20  import java.util.ArrayList;
21  
22  /**
23   * Utility class for transforming the list of arguments passed to
24   * constructors of exceptions.
25   *
26   */
27  public class ArgUtils {
28      /**
29       * Class contains only static methods.
30       */
31      private ArgUtils() {}
32  
33      /**
34       * Transform a multidimensional array into a one-dimensional list.
35       *
36       * @param array Array (possibly multidimensional).
37       * @return a list of all the {@code Object} instances contained in
38       * {@code array}.
39       */
40      public static Object[] flatten(Object[] array) {
41          final List<Object> list = new ArrayList<Object>();
42          if (array != null) {
43              for (Object o : array) {
44                  if (o instanceof Object[]) {
45                      for (Object oR : flatten((Object[]) o)) {
46                          list.add(oR);
47                      }
48                  } else {
49                      list.add(o);
50                  }
51              }
52          }
53          return list.toArray();
54      }
55  }
