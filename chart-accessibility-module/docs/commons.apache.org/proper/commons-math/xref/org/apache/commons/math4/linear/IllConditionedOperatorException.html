17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * An exception to be thrown when the condition number of a
24   * {@link RealLinearOperator} is too high.
25   *
26   * @since 3.0
27   */
28  public class IllConditionedOperatorException
29      extends MathIllegalArgumentException {
30      /** Serializable version Id. */
31      private static final long serialVersionUID = -7883263944530490135L;
32  
33      /**
34       * Creates a new instance of this class.
35       *
36       * @param cond An estimate of the condition number of the offending linear
37       * operator.
38       */
39      public IllConditionedOperatorException(final double cond) {
40          super(LocalizedFormats.ILL_CONDITIONED_OPERATOR, cond);
41      }
42  }
