44  /**
45   * <p>Represents an <a href="http://en.wikipedia.org/wiki/Empirical_distribution_function">
46   * empirical probability distribution</a> -- a probability distribution derived
47   * from observed data without making any assumptions about the functional form
48   * of the population distribution that the data come from.</p>
49   *
50   * <p>An <code>EmpiricalDistribution</code> maintains data structures, called
51   * <i>distribution digests</i>, that describe empirical distributions and
52   * support the following operations: <ul>
53   * <li>loading the distribution from a file of observed data values</li>
54   * <li>dividing the input data into "bin ranges" and reporting bin frequency
55   *     counts (data for histogram)</li>
56   * <li>reporting univariate statistics describing the full set of data values
57   *     as well as the observations within each bin</li>
58   * <li>generating random values from the distribution</li>
59   * </ul>
60   * Applications can use <code>EmpiricalDistribution</code> to build grouped
61   * frequency histograms representing the input data or to generate random values
62   * "like" those in the input file -- i.e., the values generated will follow the
63   * distribution of the values in the file.</p>
64   *
65   * <p>The implementation uses what amounts to the
66   * <a href="http://nedwww.ipac.caltech.edu/level5/March02/Silverman/Silver2_6.html">
67   * Variable Kernel Method</a> with Gaussian smoothing:<p>
68   * <strong>Digesting the input file</strong>
69   * <ol><li>Pass the file once to compute min and max.</li>
70   * <li>Divide the range from min-max into <code>binCount</code> "bins."</li>
71   * <li>Pass the data file again, computing bin counts and univariate
72   *     statistics (mean, std dev.) for each of the bins </li>
73   * <li>Divide the interval (0,1) into subintervals associated with the bins,
74   *     with the length of a bin's subinterval proportional to its count.</li></ol>
75   * <strong>Generating random values from the distribution</strong><ol>
76   * <li>Generate a uniformly distributed value in (0,1) </li>
77   * <li>Select the subinterval to which the value belongs.
78   * <li>Generate a random Gaussian value with mean = mean of the associated
79   *     bin and std dev = std dev of associated bin.</li></ol></p>
80   *
81   * <p>EmpiricalDistribution implements the {@link RealDistribution} interface
82   * as follows.  Given x within the range of values in the dataset, let B
83   * be the bin containing x and let K be the within-bin kernel for B.  Let P(B-)
84   * be the sum of the probabilities of the bins below B and let K(B) be the
85   * mass of B under K (i.e., the integral of the kernel density over B).  Then
86   * set P(X < x) = P(B-) + P(B) * K(x) / K(B) where K(x) is the kernel distribution
87   * evaluated at x. This results in a cdf that matches the grouped frequency
88   * distribution at the bin endpoints and interpolates within bins using
89   * within-bin kernels.</p>
90   *
91   *<strong>USAGE NOTES:</strong><ul>
92   *<li>The <code>binCount</code> is set by default to 1000.  A good rule of thumb
93   *    is to set the bin count to approximately the length of the input file divided
94   *    by 10. </li>
95   *<li>The input file <i>must</i> be a plain text file containing one valid numeric
96   *    entry per line.</li>
97   * </ul></p>
98   *
99   */

156     /**
157      * Computes the empirical distribution from the provided
158      * array of numbers.
159      *
160      * @param in the input data array
161      * @exception NullArgumentException if in is null
162      */
163     public void load(double[] in) throws NullArgumentException {
164         DataAdapter da = new ArrayDataAdapter(in);
165         try {
166             da.computeStats();
167             // new adapter for the second pass
168             fillBinStats(new ArrayDataAdapter(in));
169         } catch (IOException ex) {
170             // Can't happen
171             throw new MathInternalError();
172         }
173         loaded = true;
174 
175     }

177     /**
178      * Computes the empirical distribution using data read from a URL.
179      *
180      * <p>The input file <i>must</i> be an ASCII text file containing one
181      * valid numeric entry per line.</p>
182      *
183      * @param url url of the input file
184      *
185      * @throws IOException if an IO error occurs
186      * @throws NullArgumentException if url is null
187      * @throws ZeroException if URL contains no data
188      */
189     public void load(URL url) throws IOException, NullArgumentException, ZeroException {
190         MathUtils.checkNotNull(url);
191         Charset charset = Charset.forName(FILE_CHARSET);
192         BufferedReader in =
193             new BufferedReader(new InputStreamReader(url.openStream(), charset));
194         try {
195             DataAdapter da = new StreamDataAdapter(in);
196             da.computeStats();
197             if (sampleStats.getN() == 0) {
198                 throw new ZeroException(LocalizedFormats.URL_CONTAINS_NO_DATA, url);
199             }
200             // new adapter for the second pass
201             in = new BufferedReader(new InputStreamReader(url.openStream(), charset));
202             fillBinStats(new StreamDataAdapter(in));
203             loaded = true;
204         } finally {
205            try {
206                in.close();
207            } catch (IOException ex) { //NOPMD
208                // ignore
209            }
210         }
211     }

214      * Computes the empirical distribution from the input file.
215      *
216      * <p>The input file <i>must</i> be an ASCII text file containing one
217      * valid numeric entry per line.</p>
218      *
219      * @param file the input file
220      * @throws IOException if an IO error occurs
221      * @throws NullArgumentException if file is null
222      */
223     public void load(File file) throws IOException, NullArgumentException {
224         MathUtils.checkNotNull(file);
225         Charset charset = Charset.forName(FILE_CHARSET);
226         InputStream is = new FileInputStream(file);
227         BufferedReader in = new BufferedReader(new InputStreamReader(is, charset));
228         try {
229             DataAdapter da = new StreamDataAdapter(in);
230             da.computeStats();
231             // new adapter for second pass
232             is = new FileInputStream(file);
233             in = new BufferedReader(new InputStreamReader(is, charset));
234             fillBinStats(new StreamDataAdapter(in));
235             loaded = true;
236         } finally {
237             try {
238                 in.close();
239             } catch (IOException ex) { //NOPMD
240                 // ignore
241             }
242         }
243     }

267     /**
268      * <code>DataAdapter</code> for data provided through some input stream
269      */
270     private class StreamDataAdapter extends DataAdapter{
271 
272         /** Input stream providing access to the data */
273         private BufferedReader inputStream;
274 
275         /**
276          * Create a StreamDataAdapter from a BufferedReader
277          *
278          * @param in BufferedReader input stream
279          */
280         StreamDataAdapter(BufferedReader in){
281             super();
282             inputStream = in;
283         }
284 
285         /** {@inheritDoc} */
286         @Override
287         public void computeBinStats() throws IOException {
288             String str = null;
289             double val = 0.0d;
290             while ((str = inputStream.readLine()) != null) {
291                 val = Double.parseDouble(str);
292                 SummaryStatistics stats = binStats.get(findBin(val));
293                 stats.addValue(val);
294             }
295 
296             inputStream.close();
297             inputStream = null;
298         }
299 
300         /** {@inheritDoc} */
301         @Override
302         public void computeStats() throws IOException {
303             String str = null;
304             double val = 0.0;
305             sampleStats = new SummaryStatistics();
306             while ((str = inputStream.readLine()) != null) {
307                 val = Double.parseDouble(str);
308                 sampleStats.addValue(val);
309             }
310             inputStream.close();
311             inputStream = null;
312         }
313     }

315     /**
316      * <code>DataAdapter</code> for data provided as array of doubles.
317      */
318     private class ArrayDataAdapter extends DataAdapter {
319 
320         /** Array of input  data values */
321         private final double[] inputArray;
322 
323         /**
324          * Construct an ArrayDataAdapter from a double[] array
325          *
326          * @param in double[] array holding the data
327          * @throws NullArgumentException if in is null
328          */
329         ArrayDataAdapter(double[] in) throws NullArgumentException {
330             super();
331             MathUtils.checkNotNull(in);
332             inputArray = in;
333         }
334 
335         /** {@inheritDoc} */
336         @Override
337         public void computeStats() throws IOException {
338             sampleStats = new SummaryStatistics();
339             for (int i = 0; i < inputArray.length; i++) {
340                 sampleStats.addValue(inputArray[i]);
341             }
342         }
343 
344         /** {@inheritDoc} */
345         @Override
346         public void computeBinStats() throws IOException {
347             for (int i = 0; i < inputArray.length; i++) {
348                 SummaryStatistics stats =
349                     binStats.get(findBin(inputArray[i]));
350                 stats.addValue(inputArray[i]);
351             }
352         }
353     }

355     /**
356      * Fills binStats array (second pass through data file).
357      *
358      * @param da object providing access to the data
359      * @throws IOException  if an IO error occurs
360      */
361     private void fillBinStats(final DataAdapter da)
362         throws IOException {
363         // Set up grid
364         min = sampleStats.getMin();
365         max = sampleStats.getMax();
366         delta = (max - min)/binCount;
367 
368         // Initialize binStats ArrayList
369         if (!binStats.isEmpty()) {
370             binStats.clear();
371         }
372         for (int i = 0; i < binCount; i++) {
373             SummaryStatistics stats = new SummaryStatistics();
374             binStats.add(i,stats);
375         }
376 
377         // Filling data in binStats Array
378         da.computeBinStats();
379 
380         // Assign upperBounds based on bin counts
381         upperBounds = new double[binCount];
382         upperBounds[0] =
383         ((double) binStats.get(0).getN()) / (double) sampleStats.getN();
384         for (int i = 1; i < binCount-1; i++) {
385             upperBounds[i] = upperBounds[i-1] +
386             ((double) binStats.get(i).getN()) / (double) sampleStats.getN();
387         }
388         upperBounds[binCount-1] = 1.0d;
389     }
390 
391     /**
392      * Returns the index of the bin to which the given value belongs
393      *
394      * @param value  the value whose bin we are trying to find
395      * @return the index of the bin containing the value
396      */
397     private int findBin(double value) {
398         return FastMath.min(
399                 FastMath.max((int) FastMath.ceil((value - min) / delta) - 1, 0),
400                 binCount - 1);
401     }

435     /**
436      * <p>Returns a fresh copy of the array of upper bounds for the bins.
437      * Bins are: <br/>
438      * [min,upperBounds[0]],(upperBounds[0],upperBounds[1]],...,
439      *  (upperBounds[binCount-2], upperBounds[binCount-1] = max].</p>
440      *
441      * <p>Note: In versions 1.0-2.0 of commons-math, this method
442      * incorrectly returned the array of probability generator upper
443      * bounds now returned by {@link #getGeneratorUpperBounds()}.</p>
444      *
445      * @return array of bin upper bounds
446      * @since 2.1
447      */
448     public double[] getUpperBounds() {
449         double[] binUpperBounds = new double[binCount];
450         for (int i = 0; i < binCount - 1; i++) {
451             binUpperBounds[i] = min + delta * (i + 1);
452         }
453         binUpperBounds[binCount - 1] = max;
454         return binUpperBounds;
455     }
456 
457     /**
458      * <p>Returns a fresh copy of the array of upper bounds of the subintervals
459      * of [0,1] used in generating data from the empirical distribution.
460      * Subintervals correspond to bins with lengths proportional to bin counts.</p>
461      *
462      * <strong>Preconditions:</strong><ul>
463      * <li>the distribution must be loaded before invoking this method</li></ul>
464      *
465      * <p>In versions 1.0-2.0 of commons-math, this array was (incorrectly) returned
466      * by {@link #getUpperBounds()}.</p>
467      *
468      * @since 2.1
469      * @return array of upper bounds of subintervals used in data generation
470      * @throws NullPointerException unless a {@code load} method has been
471      * called beforehand.
472      */
473     public double[] getGeneratorUpperBounds() {
474         int len = upperBounds.length;
475         double[] out = new double[len];
476         System.arraycopy(upperBounds, 0, out, 0, len);
477         return out;
478     }

497         return 0;

500     /**
501      * {@inheritDoc}
502      *
503      * <p>Returns the kernel density normalized so that its integral over each bin
504      * equals the bin mass.</p>
505      *
506      * <p>Algorithm description: <ol>
507      * <li>Find the bin B that x belongs to.</li>
508      * <li>Compute K(B) = the mass of B with respect to the within-bin kernel (i.e., the
509      * integral of the kernel density over B).</li>
510      * <li>Return k(x) * P(B) / K(B), where k is the within-bin kernel density
511      * and P(B) is the mass of B.</li></ol></p>
512      * @since 3.1
513      */
514     @Override
515     public double density(double x) {
516         if (x < min || x > max) {
517             return 0d;
518         }
519         final int binIndex = findBin(x);
520         final RealDistribution kernel = getKernel(binStats.get(binIndex));
521         return kernel.density(x) * pB(binIndex) / kB(binIndex);
522     }
523 
524     /**
525      * {@inheritDoc}
526      *
527      * <p>Algorithm description:<ol>
528      * <li>Find the bin B that x belongs to.</li>
529      * <li>Compute P(B) = the mass of B and P(B-) = the combined mass of the bins below B.</li>
530      * <li>Compute K(B) = the probability mass of B with respect to the within-bin kernel
531      * and K(B-) = the kernel distribution evaluated at the lower endpoint of B</li>
532      * <li>Return P(B-) + P(B) * [K(x) - K(B-)] / K(B) where
533      * K(x) is the within-bin kernel distribution function evaluated at x.</li></ol>
534      * If K is a constant distribution, we return P(B-) + P(B) (counting the full
535      * mass of B).</p>
536      *
537      * @since 3.1
538      */
539     @Override
540     public double cumulativeProbability(double x) {
541         if (x < min) {
542             return 0d;
543         } else if (x >= max) {
544             return 1d;
545         }
546         final int binIndex = findBin(x);
547         final double pBminus = pBminus(binIndex);
548         final double pB = pB(binIndex);
549         final RealDistribution kernel = k(x);
550         if (kernel instanceof ConstantRealDistribution) {
551             if (x < kernel.getNumericalMean()) {
552                 return pBminus;
553             } else {
554                 return pBminus + pB;
555             }
556         }
557         final double[] binBounds = getUpperBounds();
558         final double kB = kB(binIndex);
559         final double lower = binIndex == 0 ? min : binBounds[binIndex - 1];
560         final double withinBinCum =
561             (kernel.cumulativeProbability(x) -  kernel.cumulativeProbability(lower)) / kB;
562         return pBminus + pB * withinBinCum;
563     }
564 
565     /**
566      * {@inheritDoc}
567      *
568      * <p>Algorithm description:<ol>
569      * <li>Find the smallest i such that the sum of the masses of the bins
570      *  through i is at least p.</li>
571      * <li>
572      *   Let K be the within-bin kernel distribution for bin i.</br>
573      *   Let K(B) be the mass of B under K. <br/>
574      *   Let K(B-) be K evaluated at the lower endpoint of B (the combined
575      *   mass of the bins below B under K).<br/>
576      *   Let P(B) be the probability of bin i.<br/>
577      *   Let P(B-) be the sum of the bin masses below bin i. <br/>
578      *   Let pCrit = p - P(B-)<br/>
579      * <li>Return the inverse of K evaluated at <br/>
580      *    K(B-) + pCrit * K(B) / P(B) </li>
581      *  </ol></p>
582      *
583      * @since 3.1
584      */
585     @Override
586     public double inverseCumulativeProbability(final double p) throws OutOfRangeException {
587         if (p < 0.0 || p > 1.0) {
588             throw new OutOfRangeException(p, 0, 1);
589         }
590 
591         if (p == 0.0) {
592             return getSupportLowerBound();
593         }
594 
595         if (p == 1.0) {
596             return getSupportUpperBound();
597         }
598 
599         int i = 0;
600         while (cumBinP(i) < p) {
601             i++;
602         }
603 
604         final RealDistribution kernel = getKernel(binStats.get(i));
605         final double kB = kB(i);
606         final double[] binBounds = getUpperBounds();
607         final double lower = i == 0 ? min : binBounds[i - 1];
608         final double kBminus = kernel.cumulativeProbability(lower);
609         final double pB = pB(i);
610         final double pBminus = pBminus(i);
611         final double pCrit = p - pBminus;
612         if (pCrit <= 0) {
613             return lower;
614         }
615         return kernel.inverseCumulativeProbability(kBminus + pCrit * kB / pB);
616     }

672     /**
673      * The probability of bin i.
674      *
675      * @param i the index of the bin
676      * @return the probability that selection begins in bin i
677      */
678     private double pB(int i) {
679         return i == 0 ? upperBounds[0] :
680             upperBounds[i] - upperBounds[i - 1];
681     }
682 
683     /**
684      * The combined probability of the bins up to but not including bin i.
685      *
686      * @param i the index of the bin
687      * @return the probability that selection begins in a bin below bin i.
688      */
689     private double pBminus(int i) {
690         return i == 0 ? 0 : upperBounds[i - 1];
691     }
692 
693     /**
694      * Mass of bin i under the within-bin kernel of the bin.
695      *
696      * @param i index of the bin
697      * @return the difference in the within-bin kernel cdf between the
698      * upper and lower endpoints of bin i
699      */
700     private double kB(int i) {
701         final double[] binBounds = getUpperBounds();
702         final RealDistribution kernel = getKernel(binStats.get(i));
703         return i == 0 ? kernel.probability(min, binBounds[0]) :
704             kernel.probability(binBounds[i - 1], binBounds[i]);
705     }
706 
707     /**
708      * The within-bin kernel of the bin that x belongs to.
709      *
710      * @param x the value to locate within a bin
711      * @return the within-bin kernel of the bin containing x
712      */
713     private RealDistribution k(double x) {
714         final int binIndex = findBin(x);
715         return getKernel(binStats.get(binIndex));
716     }
717 
718     /**
719      * The combined probability of the bins up to and including binIndex.
720      *
721      * @param binIndex maximum bin index
722      * @return sum of the probabilities of bins through binIndex
723      */
724     private double cumBinP(int binIndex) {
725         return upperBounds[binIndex];
726     }
727 
728     /**
729      * The within-bin smoothing kernel. Returns a Gaussian distribution
730      * parameterized by {@code bStats}, unless the bin contains only one
731      * observation, in which case a constant distribution is returned.
732      *
733      * @param bStats summary statistics for the bin
734      * @return within-bin kernel parameterized by bStats
735      */
736     protected RealDistribution getKernel(SummaryStatistics bStats) {
737         if (bStats.getN() == 1 || bStats.getVariance() == 0) {
738             return new ConstantRealDistribution(bStats.getMean());
739         } else {
740             return new NormalDistribution(bStats.getMean(), bStats.getStandardDeviation(),
741                                           NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
742         }
743     }
