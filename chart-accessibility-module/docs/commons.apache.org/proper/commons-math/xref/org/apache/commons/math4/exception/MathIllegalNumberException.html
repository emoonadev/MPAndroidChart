17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  
21  /**
22   * Base class for exceptions raised by a wrong number.
23   * This class is not intended to be instantiated directly: it should serve
24   * as a base class to create all the exceptions that are raised because some
25   * precondition is violated by a number argument.
26   *
27   * @since 2.2
28   */
29  public class MathIllegalNumberException extends MathIllegalArgumentException {
30  
31      /** Helper to avoid boxing warnings. @since 3.3 */
32      protected static final Integer INTEGER_ZERO = Integer.valueOf(0);
33  
34      /** Serializable version Id. */
35      private static final long serialVersionUID = -7447085893598031110L;
36  
37      /** Requested. */
38      private final Number argument;
39  
40      /**
41       * Construct an exception.
42       *
43       * @param pattern Localizable pattern.
44       * @param wrong Wrong number.
45       * @param arguments Arguments.
46       */
47      protected MathIllegalNumberException(Localizable pattern,
48                                           Number wrong,
49                                           Object ... arguments) {
50          super(pattern, wrong, arguments);
51          argument = wrong;
52      }
53  
54      /**
55       * @return the requested value.
56       */
57      public Number getArgument() {
58          return argument;
59      }
60  }
