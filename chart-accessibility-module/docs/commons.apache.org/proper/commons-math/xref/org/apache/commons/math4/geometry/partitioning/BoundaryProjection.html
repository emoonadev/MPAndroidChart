17  package org.apache.commons.math4.geometry.partitioning;
18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.geometry.Space;
21  
22  /** Class holding the result of point projection on region boundary.
23   * <p>This class is a simple placeholder, it does not provide any
24   * processing methods.</p>
25   * <p>Instances of this class are guaranteed to be immutable</p>
26   * @param <S> Type of the space.
27   * @since 3.3
28   * @see AbstractRegion#projectToBoundary(Point)
29   */
30  public class BoundaryProjection<S extends Space> {
31  
32      /** Original point. */
33      private final Point<S> original;
34  
35      /** Projected point. */
36      private final Point<S> projected;
37  
38      /** Offset of the point with respect to the boundary it is projected on. */
39      private final double offset;
40  
41      /** Constructor from raw elements.
42       * @param original original point
43       * @param projected projected point
44       * @param offset offset of the point with respect to the boundary it is projected on
45       */
46      public BoundaryProjection(final Point<S> original, final Point<S> projected, final double offset) {
47          this.original  = original;
48          this.projected = projected;
49          this.offset    = offset;
50      }
51  
52      /** Get the original point.
53       * @return original point
54       */
55      public Point<S> getOriginal() {
56          return original;
57      }
58  
59      /** Projected point.
60       * @return projected point, or null if there are no boundary
61       */
62      public Point<S> getProjected() {
63          return projected;
64      }
65  
66      /** Offset of the point with respect to the boundary it is projected on.
67       * <p>
68       * The offset with respect to the boundary is negative if the {@link
69       * #getOriginal() original point} is inside the region, and positive otherwise.
70       * </p>
71       * <p>
72       * If there are no boundary, the value is set to either {@code
73       * Double.POSITIVE_INFINITY} if the region is empty (i.e. all points are
74       * outside of the region) or {@code Double.NEGATIVE_INFINITY} if the region
75       * covers the whole space (i.e. all points are inside of the region).
76       * </p>
77       * @return offset of the point with respect to the boundary it is projected on
78       */
79      public double getOffset() {
80          return offset;
81      }
82  
83  }
