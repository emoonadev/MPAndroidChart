17  package org.apache.commons.math4.exception.util;
18  
19  import java.util.Locale;
20  
21  /**
22   * Dummy implementation of the {@link Localizable} interface, without localization.
23   *
24   * @since 2.2
25   */
26  public class DummyLocalizable implements Localizable {
27  
28      /** Serializable version identifier. */
29      private static final long serialVersionUID = 8843275624471387299L;
30  
31      /** Source string. */
32      private final String source;
33  
34      /** Simple constructor.
35       * @param source source text
36       */
37      public DummyLocalizable(final String source) {
38          this.source = source;
39      }
40  
41      /** {@inheritDoc} */
42      @Override
43      public String getSourceString() {
44          return source;
45      }
46  
47      /** {@inheritDoc} */
48      @Override
49      public String getLocalizedString(Locale locale) {
50          return source;
51      }
52  
53      /** {@inheritDoc} */
54      @Override
55      public String toString() {
56          return source;
57      }
58  
59  }
