35  /**
36   * This class implements the common part of all fixed step Runge-Kutta
37   * integrators for Ordinary Differential Equations.
38   *
39   * <p>These methods are explicit Runge-Kutta methods, their Butcher
40   * arrays are as follows :
41   * <pre>
42   *    0  |
43   *   c2  | a21
44   *   c3  | a31  a32
45   *   ... |        ...
46   *   cs  | as1  as2  ...  ass-1
47   *       |--------------------------
48   *       |  b1   b2  ...   bs-1  bs
49   * </pre>
50   * </p>
51   *
52   * @see EulerFieldIntegrator
53   * @see ClassicalRungeKuttaFieldIntegrator
54   * @see GillFieldIntegrator
55   * @see MidpointFieldIntegrator
56   * @param <T> the type of the field elements
57   * @since 3.6
58   */
59  
60  public abstract class RungeKuttaFieldIntegrator<T extends RealFieldElement<T>>
61      extends AbstractFieldIntegrator<T>
62      implements FieldButcherArrayProvider<T> {
63  
64      /** Time steps from Butcher array (without the first zero). */
65      private final T[] c;
66  
67      /** Internal weights from Butcher array (without the first empty row). */
68      private final T[][] a;
69  
70      /** External weights for the high order method from Butcher array. */
71      private final T[] b;
72  
73      /** Integration step. */
74      private final T step;
75  
76      /** Simple constructor.
77       * Build a Runge-Kutta integrator with the given
78       * step. The default step handler does nothing.
79       * @param field field to which the time and state vector elements belong
80       * @param name name of the method
81       * @param step integration step
82       */
83      protected RungeKuttaFieldIntegrator(final Field<T> field, final String name, final T step) {
84          super(field, name);
85          this.c    = getC();
86          this.a    = getA();
87          this.b    = getB();
88          this.step = step.abs();
89      }
90  
91      /** Create a fraction.
92       * @param p numerator
93       * @param q denominator
94       * @return p/q computed in the instance field
95       */
96      protected T fraction(final int p, final int q) {
97          return getField().getZero().add(p).divide(q);
98      }
99  
100     /** Create an interpolator.
101      * @param forward integration direction indicator
102      * @param yDotK slopes at the intermediate points
103      * @param globalPreviousState start of the global step
104      * @param globalCurrentState end of the global step
105      * @param mapper equations mapper for the all equations
106      * @return external weights for the high order method from Butcher array
107      */
108     protected abstract RungeKuttaFieldStepInterpolator<T> createInterpolator(boolean forward, T[][] yDotK,
109                                                                              final FieldODEStateAndDerivative<T> globalPreviousState,
110                                                                              final FieldODEStateAndDerivative<T> globalCurrentState,
111                                                                              FieldEquationsMapper<T> mapper);
112 
113     /** {@inheritDoc} */
114     @Override
115     public FieldODEStateAndDerivative<T> integrate(final FieldExpandableODE<T> equations,
116                                                    final FieldODEState<T> initialState, final T finalTime)
117         throws NumberIsTooSmallException, DimensionMismatchException,
118         MaxCountExceededException, NoBracketingException {
119 
120         sanityChecks(initialState, finalTime);
121         final T   t0 = initialState.getTime();
122         final T[] y0 = equations.getMapper().mapState(initialState);
123         setStepStart(initIntegration(equations, t0, y0, finalTime));
124         final boolean forward = finalTime.subtract(initialState.getTime()).getReal() > 0;
125 
126         // create some internal working arrays
127         final int   stages = c.length + 1;
128         T[]         y      = y0;
129         final T[][] yDotK  = MathArrays.buildArray(getField(), stages, -1);
130         final T[]   yTmp   = MathArrays.buildArray(getField(), y0.length);
131 
132         // set up integration control objects
133         if (forward) {
134             if (getStepStart().getTime().add(step).subtract(finalTime).getReal() >= 0) {
135                 setStepSize(finalTime.subtract(getStepStart().getTime()));
136             } else {
137                 setStepSize(step);
138             }
139         } else {
140             if (getStepStart().getTime().subtract(step).subtract(finalTime).getReal() <= 0) {
141                 setStepSize(finalTime.subtract(getStepStart().getTime()));
142             } else {
143                 setStepSize(step.negate());
144             }
145         }
146 
147         // main integration loop
148         setIsLastStep(false);
149         do {
150 
151             // first stage
152             y        = equations.getMapper().mapState(getStepStart());
153             yDotK[0] = equations.getMapper().mapDerivative(getStepStart());
154 
155             // next stages
156             for (int k = 1; k < stages; ++k) {
157 
158                 for (int j = 0; j < y0.length; ++j) {
159                     T sum = yDotK[0][j].multiply(a[k-1][0]);
160                     for (int l = 1; l < k; ++l) {
161                         sum = sum.add(yDotK[l][j].multiply(a[k-1][l]));
162                     }
163                     yTmp[j] = y[j].add(getStepSize().multiply(sum));
164                 }
165 
166                 yDotK[k] = computeDerivatives(getStepStart().getTime().add(getStepSize().multiply(c[k-1])), yTmp);
167 
168             }
169 
170             // estimate the state at the end of the step
171             for (int j = 0; j < y0.length; ++j) {
172                 T sum = yDotK[0][j].multiply(b[0]);
173                 for (int l = 1; l < stages; ++l) {
174                     sum = sum.add(yDotK[l][j].multiply(b[l]));
175                 }
176                 yTmp[j] = y[j].add(getStepSize().multiply(sum));
177             }
178             final T stepEnd   = getStepStart().getTime().add(getStepSize());
179             final T[] yDotTmp = computeDerivatives(stepEnd, yTmp);
180             final FieldODEStateAndDerivative<T> stateTmp = new FieldODEStateAndDerivative<T>(stepEnd, yTmp, yDotTmp);
181 
182             // discrete events handling
183             System.arraycopy(yTmp, 0, y, 0, y0.length);
184             setStepStart(acceptStep(createInterpolator(forward, yDotK, getStepStart(), stateTmp, equations.getMapper()),
185                                     finalTime));
186 
187             if (!isLastStep()) {
188 
189                 // stepsize control for next step
190                 final T  nextT      = getStepStart().getTime().add(getStepSize());
191                 final boolean nextIsLast = forward ?
192                                            (nextT.subtract(finalTime).getReal() >= 0) :
193                                            (nextT.subtract(finalTime).getReal() <= 0);
194                 if (nextIsLast) {
195                     setStepSize(finalTime.subtract(getStepStart().getTime()));
196                 }
197             }
198 
199         } while (!isLastStep());
200 
201         final FieldODEStateAndDerivative<T> finalState = getStepStart();
202         setStepStart(null);
203         setStepSize(null);
204         return finalState;
205 
206     }

208     /** Fast computation of a single step of ODE integration.
209      * <p>This method is intended for the limited use case of
210      * very fast computation of only one step without using any of the
211      * rich features of general integrators that may take some time
212      * to set up (i.e. no step handlers, no events handlers, no additional
213      * states, no interpolators, no error control, no evaluations count,
214      * no sanity checks ...). It handles the strict minimum of computation,
215      * so it can be embedded in outer loops.</p>
216      * <p>
217      * This method is <em>not</em> used at all by the {@link #integrate(FieldExpandableODE,
218      * FieldODEState, RealFieldElement)} method. It also completely ignores the step set at
219      * construction time, and uses only a single step to go from {@code t0} to {@code t}.
220      * </p>
221      * <p>
222      * As this method does not use any of the state-dependent features of the integrator,
223      * it should be reasonably thread-safe <em>if and only if</em> the provided differential
224      * equations are themselves thread-safe.
225      * </p>
226      * @param equations differential equations to integrate
227      * @param t0 initial time
228      * @param y0 initial value of the state vector at t0
229      * @param t target time for the integration
230      * (can be set to a value smaller than {@code t0} for backward integration)
231      * @return state vector at {@code t}
232      */
233     public T[] singleStep(final FirstOrderFieldDifferentialEquations<T> equations,
234                           final T t0, final T[] y0, final T t) {
235 
236         // create some internal working arrays
237         final T[] y       = y0.clone();
238         final int stages  = c.length + 1;
239         final T[][] yDotK = MathArrays.buildArray(getField(), stages, -1);
240         final T[] yTmp    = y0.clone();
241 
242         // first stage
243         final T h = t.subtract(t0);
244         yDotK[0] = equations.computeDerivatives(t0, y);
245 
246         // next stages
247         for (int k = 1; k < stages; ++k) {
248 
249             for (int j = 0; j < y0.length; ++j) {
250                 T sum = yDotK[0][j].multiply(a[k-1][0]);
251                 for (int l = 1; l < k; ++l) {
252                     sum = sum.add(yDotK[l][j].multiply(a[k-1][l]));
253                 }
254                 yTmp[j] = y[j].add(h.multiply(sum));
255             }
256 
257             yDotK[k] = equations.computeDerivatives(t0.add(h.multiply(c[k-1])), yTmp);
258 
259         }
260 
261         // estimate the state at the end of the step
262         for (int j = 0; j < y0.length; ++j) {
263             T sum = yDotK[0][j].multiply(b[0]);
264             for (int l = 1; l < stages; ++l) {
265                 sum = sum.add(yDotK[l][j].multiply(b[l]));
266             }
267             y[j] = y[j].add(h.multiply(sum));
268         }
269 
270         return y;
271 
272     }
