18  package org.apache.commons.math4.analysis.differentiation;
19  
20  import org.apache.commons.math4.analysis.MultivariateFunction;
21  import org.apache.commons.math4.exception.MathIllegalArgumentException;
22  
23  /**
24   * Extension of {@link MultivariateFunction} representing a
25   * multivariate differentiable real function.
26   * @since 3.1
27   */
28  public interface MultivariateDifferentiableFunction extends MultivariateFunction {
29  
30      /**
31       * Compute the value for the function at the given point.
32       *
33       * @param point Point at which the function must be evaluated.
34       * @return the function value for the given point.
35       * @exception MathIllegalArgumentException if {@code point} does not
36       * satisfy the function's constraints (wrong dimension, argument out of bound,
37       * or unsupported derivative order for example)
38       */
39      DerivativeStructure value(DerivativeStructure[] point)
40          throws MathIllegalArgumentException;
41  
42  }
