17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Base class for all unsupported features.
24   * It is used for all the exceptions that have the semantics of the standard
25   * {@link UnsupportedOperationException}, but must also provide a localized
26   * message.
27   *
28   * @since 2.2
29   */
30  public class MathUnsupportedOperationException extends MathRuntimeException {
31      /** Serializable version Id. */
32      private static final long serialVersionUID = -6024911025449780478L;
33  
34      /**
35       * Default constructor.
36       */
37      public MathUnsupportedOperationException() {
38          this(LocalizedFormats.UNSUPPORTED_OPERATION);
39      }
40      /**
41       * @param pattern Message pattern providing the specific context of
42       * the error.
43       * @param args Arguments.
44       */
45      public MathUnsupportedOperationException(Localizable pattern, Object ... args) {
46          super(pattern, args);
47      }
48  
49  }
