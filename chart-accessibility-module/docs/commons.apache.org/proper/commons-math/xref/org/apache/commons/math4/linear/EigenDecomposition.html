29  /**
30   * Calculates the eigen decomposition of a real matrix.
31   * <p>
32   * The eigen decomposition of matrix A is a set of two matrices:
33   * V and D such that A = V &times; D &times; V<sup>T</sup>.
34   * A, V and D are all m &times; m matrices.
35   * <p>
36   * This class is similar in spirit to the {@code EigenvalueDecomposition}
37   * class from the <a href="http://math.nist.gov/javanumerics/jama/">JAMA</a>
38   * library, with the following changes:
39   * <ul>
40   *   <li>a {@link #getVT() getVt} method has been added,</li>
41   *   <li>two {@link #getRealEigenvalue(int) getRealEigenvalue} and
42   *       {@link #getImagEigenvalue(int) getImagEigenvalue} methods to pick up a
43   *       single eigenvalue have been added,</li>
44   *   <li>a {@link #getEigenvector(int) getEigenvector} method to pick up a
45   *       single eigenvector has been added,</li>
46   *   <li>a {@link #getDeterminant() getDeterminant} method has been added.</li>
47   *   <li>a {@link #getSolver() getSolver} method has been added.</li>
48   * </ul>
49   * <p>
50   * As of 3.1, this class supports general real matrices (both symmetric and non-symmetric):
51   * <p>
52   * If A is symmetric, then A = V*D*V' where the eigenvalue matrix D is diagonal
53   * and the eigenvector matrix V is orthogonal, i.e.
54   * {@code A = V.multiply(D.multiply(V.transpose()))} and
55   * {@code V.multiply(V.transpose())} equals the identity matrix.
56   * </p>
57   * <p>
58   * If A is not symmetric, then the eigenvalue matrix D is block diagonal with the real
59   * eigenvalues in 1-by-1 blocks and any complex eigenvalues, lambda + i*mu, in 2-by-2
60   * blocks:
61   * <pre>
62   *    [lambda, mu    ]
63   *    [   -mu, lambda]
64   * </pre>
65   * The columns of V represent the eigenvectors in the sense that {@code A*V = V*D},
66   * i.e. A.multiply(V) equals V.multiply(D).
67   * The matrix V may be badly conditioned, or even singular, so the validity of the
68   * equation {@code A = V*D*inverse(V)} depends upon the condition of V.
69   * <p>
70   * This implementation is based on the paper by A. Drubrulle, R.S. Martin and
71   * J.H. Wilkinson "The Implicit QL Algorithm" in Wilksinson and Reinsch (1971)
72   * Handbook for automatic computation, vol. 2, Linear algebra, Springer-Verlag,
73   * New-York.
74   *
75   * @see <a href="http://mathworld.wolfram.com/EigenDecomposition.html">MathWorld</a>
76   * @see <a href="http://en.wikipedia.org/wiki/Eigendecomposition_of_a_matrix">Wikipedia</a>
77   * @since 2.0 (changed to concrete class in 3.0)
78   */
79  public class EigenDecomposition {
80      /** Internally used epsilon criteria. */
81      private static final double EPSILON = 1e-12;
82      /** Maximum number of iterations accepted in the implicit QL transformation */
83      private static final byte MAX_ITER = 30;
84      /** Main diagonal of the tridiagonal matrix. */
85      private double[] main;
86      /** Secondary diagonal of the tridiagonal matrix. */
87      private double[] secondary;
88      /**
89       * Transformer to tridiagonal (may be null if matrix is already
90       * tridiagonal).
91       */
92      private TriDiagonalTransformer transformer;
93      /** Real part of the realEigenvalues. */
94      private double[] realEigenvalues;
95      /** Imaginary part of the realEigenvalues. */
96      private double[] imagEigenvalues;
97      /** Eigenvectors. */
98      private ArrayRealVector[] eigenvectors;
99      /** Cached value of V. */
100     private RealMatrix cachedV;
101     /** Cached value of D. */
102     private RealMatrix cachedD;
103     /** Cached value of Vt. */
104     private RealMatrix cachedVt;
105     /** Whether the matrix is symmetric. */
106     private final boolean isSymmetric;

108     /**
109      * Calculates the eigen decomposition of the given real matrix.
110      * <p>
111      * Supports decomposition of a general matrix since 3.1.
112      *
113      * @param matrix Matrix to decompose.
114      * @throws MaxCountExceededException if the algorithm fails to converge.
115      * @throws MathArithmeticException if the decomposition of a general matrix
116      * results in a matrix with zero norm
117      * @since 3.1
118      */
119     public EigenDecomposition(final RealMatrix matrix)
120         throws MathArithmeticException {
121         final double symTol = 10 * matrix.getRowDimension() * matrix.getColumnDimension() * Precision.EPSILON;
122         isSymmetric = MatrixUtils.isSymmetric(matrix, symTol);
123         if (isSymmetric) {
124             transformToTridiagonal(matrix);
125             findEigenVectors(transformer.getQ().getData());
126         } else {
127             final SchurTransformer t = transformToSchur(matrix);
128             findEigenVectorsFromSchur(t);
129         }
130     }

132     /**
133      * Calculates the eigen decomposition of the symmetric tridiagonal
134      * matrix.  The Householder matrix is assumed to be the identity matrix.
135      *
136      * @param main Main diagonal of the symmetric tridiagonal form.
137      * @param secondary Secondary of the tridiagonal form.
138      * @throws MaxCountExceededException if the algorithm fails to converge.
139      * @since 3.1
140      */
141     public EigenDecomposition(final double[] main, final double[] secondary) {
142         isSymmetric = true;
143         this.main      = main.clone();
144         this.secondary = secondary.clone();
145         transformer    = null;
146         final int size = main.length;
147         final double[][] z = new double[size][size];
148         for (int i = 0; i < size; i++) {
149             z[i][i] = 1.0;
150         }
151         findEigenVectors(z);
152     }

154     /**
155      * Gets the matrix V of the decomposition.
156      * V is an orthogonal matrix, i.e. its transpose is also its inverse.
157      * The columns of V are the eigenvectors of the original matrix.
158      * No assumption is made about the orientation of the system axes formed
159      * by the columns of V (e.g. in a 3-dimension space, V can form a left-
160      * or right-handed system).
161      *
162      * @return the V matrix.
163      */
164     public RealMatrix getV() {
165 
166         if (cachedV == null) {
167             final int m = eigenvectors.length;
168             cachedV = MatrixUtils.createRealMatrix(m, m);
169             for (int k = 0; k < m; ++k) {
170                 cachedV.setColumnVector(k, eigenvectors[k]);
171             }
172         }
173         // return the cached matrix
174         return cachedV;
175     }
176 
177     /**
178      * Gets the block diagonal matrix D of the decomposition.
179      * D is a block diagonal matrix.
180      * Real eigenvalues are on the diagonal while complex values are on
181      * 2x2 blocks { {real +imaginary}, {-imaginary, real} }.
182      *
183      * @return the D matrix.
184      *
185      * @see #getRealEigenvalues()
186      * @see #getImagEigenvalues()
187      */
188     public RealMatrix getD() {
189 
190         if (cachedD == null) {
191             // cache the matrix for subsequent calls
192             cachedD = MatrixUtils.createRealMatrixWithDiagonal(realEigenvalues);
193 
194             for (int i = 0; i < imagEigenvalues.length; i++) {
195                 if (Precision.compareTo(imagEigenvalues[i], 0.0, EPSILON) > 0) {
196                     cachedD.setEntry(i, i+1, imagEigenvalues[i]);
197                 } else if (Precision.compareTo(imagEigenvalues[i], 0.0, EPSILON) < 0) {
198                     cachedD.setEntry(i, i-1, imagEigenvalues[i]);
199                 }
200             }
201         }
202         return cachedD;
203     }

205     /**
206      * Gets the transpose of the matrix V of the decomposition.
207      * V is an orthogonal matrix, i.e. its transpose is also its inverse.
208      * The columns of V are the eigenvectors of the original matrix.
209      * No assumption is made about the orientation of the system axes formed
210      * by the columns of V (e.g. in a 3-dimension space, V can form a left-
211      * or right-handed system).
212      *
213      * @return the transpose of the V matrix.
214      */
215     public RealMatrix getVT() {
216 
217         if (cachedVt == null) {
218             final int m = eigenvectors.length;
219             cachedVt = MatrixUtils.createRealMatrix(m, m);
220             for (int k = 0; k < m; ++k) {
221                 cachedVt.setRowVector(k, eigenvectors[k]);
222             }
223         }
224 
225         // return the cached matrix
226         return cachedVt;
227     }

230      * Returns whether the calculated eigen values are complex or real.
231      * <p>The method performs a zero check for each element of the
232      * {@link #getImagEigenvalues()} array and returns {@code true} if any
233      * element is not equal to zero.
234      *
235      * @return {@code true} if the eigen values are complex, {@code false} otherwise

239         for (int i = 0; i < imagEigenvalues.length; i++) {

242             }
243         }
244         return false;

331     /**
332      * Computes the square-root of the matrix.
333      * This implementation assumes that the matrix is symmetric and positive
334      * definite.
335      *
336      * @return the square-root of the matrix.
337      * @throws MathUnsupportedOperationException if the matrix is not
338      * symmetric or not positive definite.
339      * @since 3.1
340      */
341     public RealMatrix getSquareRoot() {
342         if (!isSymmetric) {
343             throw new MathUnsupportedOperationException();
344         }
345 
346         final double[] sqrtEigenValues = new double[realEigenvalues.length];
347         for (int i = 0; i < realEigenvalues.length; i++) {
348             final double eigen = realEigenvalues[i];
349             if (eigen <= 0) {
350                 throw new MathUnsupportedOperationException();
351             }
352             sqrtEigenValues[i] = FastMath.sqrt(eigen);
353         }
354         final RealMatrix sqrtEigen = MatrixUtils.createRealDiagonalMatrix(sqrtEigenValues);
355         final RealMatrix v = getV();
356         final RealMatrix vT = getVT();
357 
358         return v.multiply(sqrtEigen).multiply(vT);
359     }

362      * Gets a solver for finding the A &times; X = B solution in exact
363      * linear sense.
364      * <p>
365      * Since 3.1, eigen decomposition of a general matrix is supported,
366      * but the {@link DecompositionSolver} only supports real eigenvalues.
367      *
368      * @return a solver
369      * @throws MathUnsupportedOperationException if the decomposition resulted in

374             throw new MathUnsupportedOperationException();
375         }
376         return new Solver(realEigenvalues, imagEigenvalues, eigenvectors);

379     /** Specialized solver. */
380     private static class Solver implements DecompositionSolver {
381         /** Real part of the realEigenvalues. */
382         private final double[] realEigenvalues;
383         /** Imaginary part of the realEigenvalues. */
384         private final double[] imagEigenvalues;
385         /** Eigenvectors. */
386         private final ArrayRealVector[] eigenvectors;
387 
388         /**
389          * Builds a solver from decomposed matrix.
390          *
391          * @param realEigenvalues Real parts of the eigenvalues.
392          * @param imagEigenvalues Imaginary parts of the eigenvalues.
393          * @param eigenvectors Eigenvectors.
394          */
395         private Solver(final double[] realEigenvalues,
396                 final double[] imagEigenvalues,
397                 final ArrayRealVector[] eigenvectors) {
398             this.realEigenvalues = realEigenvalues;
399             this.imagEigenvalues = imagEigenvalues;
400             this.eigenvectors = eigenvectors;
401         }
402 
403         /**
404          * Solves the linear equation A &times; X = B for symmetric matrices A.
405          * <p>
406          * This method only finds exact linear solutions, i.e. solutions for
407          * which ||A &times; X - B|| is exactly 0.
408          * </p>
409          *
410          * @param b Right-hand side of the equation A &times; X = B.
411          * @return a Vector X that minimizes the two norm of A &times; X - B.
412          *
413          * @throws DimensionMismatchException if the matrices dimensions do not match.
414          * @throws SingularMatrixException if the decomposed matrix is singular.
415          */
416         @Override
417         public RealVector solve(final RealVector b) {
418             if (!isNonSingular()) {
419                 throw new SingularMatrixException();
420             }
421 
422             final int m = realEigenvalues.length;
423             if (b.getDimension() != m) {
424                 throw new DimensionMismatchException(b.getDimension(), m);
425             }
426 
427             final double[] bp = new double[m];
428             for (int i = 0; i < m; ++i) {
429                 final ArrayRealVector v = eigenvectors[i];
430                 final double[] vData = v.getDataRef();
431                 final double s = v.dotProduct(b) / realEigenvalues[i];
432                 for (int j = 0; j < m; ++j) {
433                     bp[j] += s * vData[j];
434                 }
435             }
436 
437             return new ArrayRealVector(bp, false);
438         }
439 
440         /** {@inheritDoc} */
441         @Override
442         public RealMatrix solve(RealMatrix b) {
443 
444             if (!isNonSingular()) {
445                 throw new SingularMatrixException();
446             }
447 
448             final int m = realEigenvalues.length;
449             if (b.getRowDimension() != m) {
450                 throw new DimensionMismatchException(b.getRowDimension(), m);
451             }
452 
453             final int nColB = b.getColumnDimension();
454             final double[][] bp = new double[m][nColB];
455             final double[] tmpCol = new double[m];
456             for (int k = 0; k < nColB; ++k) {
457                 for (int i = 0; i < m; ++i) {
458                     tmpCol[i] = b.getEntry(i, k);
459                     bp[i][k]  = 0;
460                 }
461                 for (int i = 0; i < m; ++i) {
462                     final ArrayRealVector v = eigenvectors[i];
463                     final double[] vData = v.getDataRef();
464                     double s = 0;
465                     for (int j = 0; j < m; ++j) {
466                         s += v.getEntry(j) * tmpCol[j];
467                     }
468                     s /= realEigenvalues[i];
469                     for (int j = 0; j < m; ++j) {
470                         bp[j][k] += s * vData[j];
471                     }
472                 }
473             }
474 
475             return new Array2DRowRealMatrix(bp, false);
476 
477         }

479         /**
480          * Checks whether the decomposed matrix is non-singular.
481          *
482          * @return true if the decomposed matrix is non-singular.
483          */
484         @Override
485         public boolean isNonSingular() {
486             double largestEigenvalueNorm = 0.0;
487             // Looping over all values (in case they are not sorted in decreasing
488             // order of their norm).
489             for (int i = 0; i < realEigenvalues.length; ++i) {
490                 largestEigenvalueNorm = FastMath.max(largestEigenvalueNorm, eigenvalueNorm(i));
491             }
492             // Corner case: zero matrix, all exactly 0 eigenvalues
493             if (largestEigenvalueNorm == 0.0) {
494                 return false;
495             }
496             for (int i = 0; i < realEigenvalues.length; ++i) {
497                 // Looking for eigenvalues that are 0, where we consider anything much much smaller
498                 // than the largest eigenvalue to be effectively 0.
499                 if (Precision.equals(eigenvalueNorm(i) / largestEigenvalueNorm, 0, EPSILON)) {
500                     return false;
501                 }
502             }
503             return true;
504         }
505 
506         /**
507          * @param i which eigenvalue to find the norm of
508          * @return the norm of ith (complex) eigenvalue.
509          */
510         private double eigenvalueNorm(int i) {
511             final double re = realEigenvalues[i];
512             final double im = imagEigenvalues[i];
513             return FastMath.sqrt(re * re + im * im);
514         }
515 
516         /**
517          * Get the inverse of the decomposed matrix.
518          *
519          * @return the inverse matrix.
520          * @throws SingularMatrixException if the decomposed matrix is singular.
521          */
522         @Override
523         public RealMatrix getInverse() {
524             if (!isNonSingular()) {
525                 throw new SingularMatrixException();
526             }
527 
528             final int m = realEigenvalues.length;
529             final double[][] invData = new double[m][m];
530 
531             for (int i = 0; i < m; ++i) {
532                 final double[] invI = invData[i];
533                 for (int j = 0; j < m; ++j) {
534                     double invIJ = 0;
535                     for (int k = 0; k < m; ++k) {
536                         final double[] vK = eigenvectors[k].getDataRef();
537                         invIJ += vK[i] * vK[j] / realEigenvalues[k];
538                     }
539                     invI[j] = invIJ;
540                 }
541             }
542             return MatrixUtils.createRealMatrix(invData);
543         }
544     }

546     /**
547      * Transforms the matrix to tridiagonal form.
548      *
549      * @param matrix Matrix to transform.
550      */
551     private void transformToTridiagonal(final RealMatrix matrix) {
552         // transform the matrix to tridiagonal
553         transformer = new TriDiagonalTransformer(matrix);
554         main = transformer.getMainDiagonalRef();
555         secondary = transformer.getSecondaryDiagonalRef();
556     }

558     /**
559      * Find eigenvalues and eigenvectors (Dubrulle et al., 1971)
560      *
561      * @param householderMatrix Householder matrix of the transformation
562      * to tridiagonal form.
563      */
564     private void findEigenVectors(final double[][] householderMatrix) {
565         final double[][]z = householderMatrix.clone();
566         final int n = main.length;
567         realEigenvalues = new double[n];
568         imagEigenvalues = new double[n];
569         final double[] e = new double[n];
570         for (int i = 0; i < n - 1; i++) {
571             realEigenvalues[i] = main[i];
572             e[i] = secondary[i];
573         }
574         realEigenvalues[n - 1] = main[n - 1];
575         e[n - 1] = 0;
576 
577         // Determine the largest main and secondary value in absolute term.
578         double maxAbsoluteValue = 0;
579         for (int i = 0; i < n; i++) {
580             if (FastMath.abs(realEigenvalues[i]) > maxAbsoluteValue) {
581                 maxAbsoluteValue = FastMath.abs(realEigenvalues[i]);
582             }
583             if (FastMath.abs(e[i]) > maxAbsoluteValue) {
584                 maxAbsoluteValue = FastMath.abs(e[i]);
585             }
586         }
587         // Make null any main and secondary value too small to be significant
588         if (maxAbsoluteValue != 0) {
589             for (int i=0; i < n; i++) {
590                 if (FastMath.abs(realEigenvalues[i]) <= Precision.EPSILON * maxAbsoluteValue) {
591                     realEigenvalues[i] = 0;
592                 }
593                 if (FastMath.abs(e[i]) <= Precision.EPSILON * maxAbsoluteValue) {
594                     e[i]=0;
595                 }
596             }
597         }
598 
599         for (int j = 0; j < n; j++) {
600             int its = 0;
601             int m;
602             do {
603                 for (m = j; m < n - 1; m++) {
604                     double delta = FastMath.abs(realEigenvalues[m]) +
605                         FastMath.abs(realEigenvalues[m + 1]);
606                     if (FastMath.abs(e[m]) + delta == delta) {
607                         break;
608                     }
609                 }
610                 if (m != j) {
611                     if (its == MAX_ITER) {
612                         throw new MaxCountExceededException(LocalizedFormats.CONVERGENCE_FAILED,
613                                                             MAX_ITER);
614                     }
615                     its++;
616                     double q = (realEigenvalues[j + 1] - realEigenvalues[j]) / (2 * e[j]);
617                     double t = FastMath.sqrt(1 + q * q);
618                     if (q < 0.0) {
619                         q = realEigenvalues[m] - realEigenvalues[j] + e[j] / (q - t);
620                     } else {
621                         q = realEigenvalues[m] - realEigenvalues[j] + e[j] / (q + t);
622                     }
623                     double u = 0.0;
624                     double s = 1.0;
625                     double c = 1.0;
626                     int i;
627                     for (i = m - 1; i >= j; i--) {
628                         double p = s * e[i];
629                         double h = c * e[i];
630                         if (FastMath.abs(p) >= FastMath.abs(q)) {
631                             c = q / p;
632                             t = FastMath.sqrt(c * c + 1.0);
633                             e[i + 1] = p * t;
634                             s = 1.0 / t;
635                             c *= s;
636                         } else {
637                             s = p / q;
638                             t = FastMath.sqrt(s * s + 1.0);
639                             e[i + 1] = q * t;
640                             c = 1.0 / t;
641                             s *= c;
642                         }
643                         if (e[i + 1] == 0.0) {
644                             realEigenvalues[i + 1] -= u;
645                             e[m] = 0.0;
646                             break;
647                         }
648                         q = realEigenvalues[i + 1] - u;
649                         t = (realEigenvalues[i] - q) * s + 2.0 * c * h;
650                         u = s * t;
651                         realEigenvalues[i + 1] = q + u;
652                         q = c * t - h;
653                         for (int ia = 0; ia < n; ia++) {
654                             p = z[ia][i + 1];
655                             z[ia][i + 1] = s * z[ia][i] + c * p;
656                             z[ia][i] = c * z[ia][i] - s * p;
657                         }
658                     }
659                     if (t == 0.0 && i >= j) {
660                         continue;
661                     }
662                     realEigenvalues[j] -= u;
663                     e[j] = q;
664                     e[m] = 0.0;
665                 }
666             } while (m != j);
667         }
668 
669         //Sort the eigen values (and vectors) in increase order
670         for (int i = 0; i < n; i++) {
671             int k = i;
672             double p = realEigenvalues[i];
673             for (int j = i + 1; j < n; j++) {
674                 if (realEigenvalues[j] > p) {
675                     k = j;
676                     p = realEigenvalues[j];
677                 }
678             }
679             if (k != i) {
680                 realEigenvalues[k] = realEigenvalues[i];
681                 realEigenvalues[i] = p;
682                 for (int j = 0; j < n; j++) {
683                     p = z[j][i];
684                     z[j][i] = z[j][k];
685                     z[j][k] = p;
686                 }
687             }
688         }
689 
690         // Determine the largest eigen value in absolute term.
691         maxAbsoluteValue = 0;
692         for (int i = 0; i < n; i++) {
693             if (FastMath.abs(realEigenvalues[i]) > maxAbsoluteValue) {
694                 maxAbsoluteValue=FastMath.abs(realEigenvalues[i]);
695             }
696         }
697         // Make null any eigen value too small to be significant
698         if (maxAbsoluteValue != 0.0) {
699             for (int i=0; i < n; i++) {
700                 if (FastMath.abs(realEigenvalues[i]) < Precision.EPSILON * maxAbsoluteValue) {
701                     realEigenvalues[i] = 0;
702                 }
703             }
704         }
705         eigenvectors = new ArrayRealVector[n];
706         final double[] tmp = new double[n];
707         for (int i = 0; i < n; i++) {
708             for (int j = 0; j < n; j++) {
709                 tmp[j] = z[j][i];
710             }
711             eigenvectors[i] = new ArrayRealVector(tmp);
712         }
713     }

715     /**
716      * Transforms the matrix to Schur form and calculates the eigenvalues.
717      *
718      * @param matrix Matrix to transform.
719      * @return the {@link SchurTransformer Shur transform} for this matrix
720      */
721     private SchurTransformer transformToSchur(final RealMatrix matrix) {
722         final SchurTransformer schurTransform = new SchurTransformer(matrix);
723         final double[][] matT = schurTransform.getT().getData();
724 
725         realEigenvalues = new double[matT.length];
726         imagEigenvalues = new double[matT.length];
727 
728         for (int i = 0; i < realEigenvalues.length; i++) {
729             if (i == (realEigenvalues.length - 1) ||
730                 Precision.equals(matT[i + 1][i], 0.0, EPSILON)) {
731                 realEigenvalues[i] = matT[i][i];
732             } else {
733                 final double x = matT[i + 1][i + 1];
734                 final double p = 0.5 * (matT[i][i] - x);
735                 final double z = FastMath.sqrt(FastMath.abs(p * p + matT[i + 1][i] * matT[i][i + 1]));
736                 realEigenvalues[i] = x + p;
737                 imagEigenvalues[i] = z;
738                 realEigenvalues[i + 1] = x + p;
739                 imagEigenvalues[i + 1] = -z;
740                 i++;
741             }
742         }
743         return schurTransform;
744     }

746     /**
747      * Performs a division of two complex numbers.
748      *
749      * @param xr real part of the first number
750      * @param xi imaginary part of the first number
751      * @param yr real part of the second number
752      * @param yi imaginary part of the second number
753      * @return result of the complex division
754      */
755     private Complex cdiv(final double xr, final double xi,
756                          final double yr, final double yi) {
757         return new Complex(xr, xi).divide(new Complex(yr, yi));
758     }

760     /**
761      * Find eigenvectors from a matrix transformed to Schur form.
762      *
763      * @param schur the schur transformation of the matrix
764      * @throws MathArithmeticException if the Schur form has a norm of zero
765      */
766     private void findEigenVectorsFromSchur(final SchurTransformer schur)
767         throws MathArithmeticException {
768         final double[][] matrixT = schur.getT().getData();
769         final double[][] matrixP = schur.getP().getData();
770 
771         final int n = matrixT.length;
772 
773         // compute matrix norm
774         double norm = 0.0;
775         for (int i = 0; i < n; i++) {
776            for (int j = FastMath.max(i - 1, 0); j < n; j++) {
777                norm += FastMath.abs(matrixT[i][j]);
778            }
779         }
780 
781         // we can not handle a matrix with zero norm
782         if (Precision.equals(norm, 0.0, EPSILON)) {
783            throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
784         }
785 
786         // Backsubstitute to find vectors of upper triangular form
787 
788         double r = 0.0;
789         double s = 0.0;
790         double z = 0.0;
791 
792         for (int idx = n - 1; idx >= 0; idx--) {
793             double p = realEigenvalues[idx];
794             double q = imagEigenvalues[idx];
795 
796             if (Precision.equals(q, 0.0)) {
797                 // Real vector
798                 int l = idx;
799                 matrixT[idx][idx] = 1.0;
800                 for (int i = idx - 1; i >= 0; i--) {
801                     double w = matrixT[i][i] - p;
802                     r = 0.0;
803                     for (int j = l; j <= idx; j++) {
804                         r += matrixT[i][j] * matrixT[j][idx];
805                     }
806                     if (Precision.compareTo(imagEigenvalues[i], 0.0, EPSILON) < 0) {
807                         z = w;
808                         s = r;
809                     } else {
810                         l = i;
811                         if (Precision.equals(imagEigenvalues[i], 0.0)) {
812                             if (w != 0.0) {
813                                 matrixT[i][idx] = -r / w;
814                             } else {
815                                 matrixT[i][idx] = -r / (Precision.EPSILON * norm);
816                             }
817                         } else {
818                             // Solve real equations
819                             double x = matrixT[i][i + 1];
820                             double y = matrixT[i + 1][i];
821                             q = (realEigenvalues[i] - p) * (realEigenvalues[i] - p) +
822                                 imagEigenvalues[i] * imagEigenvalues[i];
823                             double t = (x * s - z * r) / q;
824                             matrixT[i][idx] = t;
825                             if (FastMath.abs(x) > FastMath.abs(z)) {
826                                 matrixT[i + 1][idx] = (-r - w * t) / x;
827                             } else {
828                                 matrixT[i + 1][idx] = (-s - y * t) / z;
829                             }
830                         }
831 
832                         // Overflow control
833                         double t = FastMath.abs(matrixT[i][idx]);
834                         if ((Precision.EPSILON * t) * t > 1) {
835                             for (int j = i; j <= idx; j++) {
836                                 matrixT[j][idx] /= t;
837                             }
838                         }
839                     }
840                 }
841             } else if (q < 0.0) {
842                 // Complex vector
843                 int l = idx - 1;
844 
845                 // Last vector component imaginary so matrix is triangular
846                 if (FastMath.abs(matrixT[idx][idx - 1]) > FastMath.abs(matrixT[idx - 1][idx])) {
847                     matrixT[idx - 1][idx - 1] = q / matrixT[idx][idx - 1];
848                     matrixT[idx - 1][idx]     = -(matrixT[idx][idx] - p) / matrixT[idx][idx - 1];
849                 } else {
850                     final Complex result = cdiv(0.0, -matrixT[idx - 1][idx],
851                                                 matrixT[idx - 1][idx - 1] - p, q);
852                     matrixT[idx - 1][idx - 1] = result.getReal();
853                     matrixT[idx - 1][idx]     = result.getImaginary();
854                 }
855 
856                 matrixT[idx][idx - 1] = 0.0;
857                 matrixT[idx][idx]     = 1.0;
858 
859                 for (int i = idx - 2; i >= 0; i--) {
860                     double ra = 0.0;
861                     double sa = 0.0;
862                     for (int j = l; j <= idx; j++) {
863                         ra += matrixT[i][j] * matrixT[j][idx - 1];
864                         sa += matrixT[i][j] * matrixT[j][idx];
865                     }
866                     double w = matrixT[i][i] - p;
867 
868                     if (Precision.compareTo(imagEigenvalues[i], 0.0, EPSILON) < 0) {
869                         z = w;
870                         r = ra;
871                         s = sa;
872                     } else {
873                         l = i;
874                         if (Precision.equals(imagEigenvalues[i], 0.0)) {
875                             final Complex c = cdiv(-ra, -sa, w, q);
876                             matrixT[i][idx - 1] = c.getReal();
877                             matrixT[i][idx] = c.getImaginary();
878                         } else {
879                             // Solve complex equations
880                             double x = matrixT[i][i + 1];
881                             double y = matrixT[i + 1][i];
882                             double vr = (realEigenvalues[i] - p) * (realEigenvalues[i] - p) +
883                                         imagEigenvalues[i] * imagEigenvalues[i] - q * q;
884                             final double vi = (realEigenvalues[i] - p) * 2.0 * q;
885                             if (Precision.equals(vr, 0.0) && Precision.equals(vi, 0.0)) {
886                                 vr = Precision.EPSILON * norm *
887                                      (FastMath.abs(w) + FastMath.abs(q) + FastMath.abs(x) +
888                                       FastMath.abs(y) + FastMath.abs(z));
889                             }
890                             final Complex c     = cdiv(x * r - z * ra + q * sa,
891                                                        x * s - z * sa - q * ra, vr, vi);
892                             matrixT[i][idx - 1] = c.getReal();
893                             matrixT[i][idx]     = c.getImaginary();
894 
895                             if (FastMath.abs(x) > (FastMath.abs(z) + FastMath.abs(q))) {
896                                 matrixT[i + 1][idx - 1] = (-ra - w * matrixT[i][idx - 1] +
897                                                            q * matrixT[i][idx]) / x;
898                                 matrixT[i + 1][idx]     = (-sa - w * matrixT[i][idx] -
899                                                            q * matrixT[i][idx - 1]) / x;
900                             } else {
901                                 final Complex c2        = cdiv(-r - y * matrixT[i][idx - 1],
902                                                                -s - y * matrixT[i][idx], z, q);
903                                 matrixT[i + 1][idx - 1] = c2.getReal();
904                                 matrixT[i + 1][idx]     = c2.getImaginary();
905                             }
906                         }
907 
908                         // Overflow control
909                         double t = FastMath.max(FastMath.abs(matrixT[i][idx - 1]),
910                                                 FastMath.abs(matrixT[i][idx]));
911                         if ((Precision.EPSILON * t) * t > 1) {
912                             for (int j = i; j <= idx; j++) {
913                                 matrixT[j][idx - 1] /= t;
914                                 matrixT[j][idx] /= t;
915                             }
916                         }
917                     }
918                 }
919             }
920         }
921 
922         // Back transformation to get eigenvectors of original matrix
923         for (int j = n - 1; j >= 0; j--) {
924             for (int i = 0; i <= n - 1; i++) {
925                 z = 0.0;
926                 for (int k = 0; k <= FastMath.min(j, n - 1); k++) {
927                     z += matrixP[i][k] * matrixT[k][j];
928                 }
929                 matrixP[i][j] = z;
930             }
931         }
932 
933         eigenvectors = new ArrayRealVector[n];
934         final double[] tmp = new double[n];
935         for (int i = 0; i < n; i++) {
936             for (int j = 0; j < n; j++) {
937                 tmp[j] = matrixP[j][i];
938             }
939             eigenvectors[i] = new ArrayRealVector(tmp);
940         }
941     }
