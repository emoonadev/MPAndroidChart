42  /**
43   * A {@link StorelessUnivariateStatistic} estimating percentiles using the
44   * <ahref=http://www.cs.wustl.edu/~jain/papers/ftp/psqr.pdf>P<SUP>2</SUP></a>;
45   * Algorithm as explained by <a href=http://www.cse.wustl.edu/~jain/>Raj
46   * Jain</a> and Imrich Chlamtac in
47   * <a href=http://www.cse.wustl.edu/~jain/papers/psqr.htm>P<SUP>2</SUP> Algorithm
48   * for Dynamic Calculation of Quantiles and Histogram Without Storing
49   * Observations</a>.
50   * <p>
51   * Note: This implementation is not synchronized and produces an approximate
52   * result. For small samples, where data can be stored and processed in memory,
53   * {@link Percentile} should be used.</p>
54   *
55   */
56  public class PSquarePercentile extends AbstractStorelessUnivariateStatistic
57          implements StorelessUnivariateStatistic, Serializable {

115     /**
116      * Constructs a PSquarePercentile with the specific percentile value.
117      * @param p the percentile
118      * @throws OutOfRangeException  if p is not greater than 0 and less
119      * than or equal to 100
120      */
121     public PSquarePercentile(final double p) {
122         if (p > 100 || p < 0) {
123             throw new OutOfRangeException(LocalizedFormats.OUT_OF_RANGE, p, 0, 100);
124         }
125         this.quantile = p / 100d;// always set it within (0,1]
126     }

148     /**
149      * Returns true iff {@code o} is a {@code PSquarePercentile} returning the
150      * same values as this for {@code getResult()} and {@code getN()} and also
151      * having equal markers
152      *
153      * @param o object to compare
154      * @return true if {@code o} is a {@code PSquarePercentile} with
155      * equivalent internal state
156      */
157     @Override
158     public boolean equals(Object o) {
159         boolean result = false;
160         if (this == o) {
161             result = true;
162         } else if (o instanceof PSquarePercentile) {
163             PSquarePercentile that = (PSquarePercentile) o;
164             boolean isNotNull = markers != null && that.markers != null;
165             boolean isNull = markers == null && that.markers == null;
166             result = isNotNull ? markers.equals(that.markers) : isNull;
167             // markers as in the case of first
168             // five observations
169             result = result && getN() == that.getN();
170         }
171         return result;
172     }

174     /**
175      * {@inheritDoc}The internal state updated due to the new value in this
176      * context is basically of the marker positions and computation of the
177      * approximate quantile.
178      *
179      * @param observation the observation currently being added.
180      */
181     @Override
182     public void increment(final double observation) {
183         // Increment counter
184         countOfObservations++;
185 
186         // Store last observation
187         this.lastObservation = observation;
188 
189         // 0. Use Brute force for <5
190         if (markers == null) {
191             if (initialFive.add(observation)) {
192                 Collections.sort(initialFive);
193                 pValue =
194                         initialFive
195                                 .get((int) (quantile * (initialFive.size() - 1)));
196                 return;
197             }
198             // 1. Initialize once after 5th observation
199             markers = newMarkers(initialFive, quantile);
200         }
201         // 2. process a Data Point and return pValue
202         pValue = markers.processDataPoint(observation);
203     }

271     /**
272      * {@inheritDoc}
273      */
274     @Override
275     public double getResult() {
276         if (Double.compare(quantile, 1d) == 0) {
277             pValue = maximum();
278         } else if (Double.compare(quantile, 0d) == 0) {
279             pValue = minimum();
280         }
281         return pValue;
282     }

284     /**
285      * @return maximum in the data set added to this statistic
286      */
287     private double maximum() {
288         double val = Double.NaN;
289         if (markers != null) {
290             val = markers.height(PSQUARE_CONSTANT);
291         } else if (!initialFive.isEmpty()) {
292             val = initialFive.get(initialFive.size() - 1);
293         }
294         return val;
295     }
296 
297     /**
298      * @return minimum in the data set added to this statistic
299      */
300     private double minimum() {
301         double val = Double.NaN;
302         if (markers != null) {
303             val = markers.height(1);
304         } else if (!initialFive.isEmpty()) {
305             val = initialFive.get(0);
306         }
307         return val;
308     }

310     /**
311      * Markers is an encapsulation of the five markers/buckets as indicated in
312      * the original works.
313      */
314     private static class Markers implements PSquareMarkers, Serializable {
315         /**
316          * Serial version id
317          */
318         private static final long serialVersionUID = 1L;
319 
320         /** Low marker index */
321         private static final int LOW = 2;
322 
323         /** High marker index */
324         private static final int HIGH = 4;
325 
326         /**
327          * Array of 5+1 Markers (The first marker is dummy just so we
328          * can match the rest of indexes [1-5] indicated in the original works
329          * which follows unit based index)
330          */
331         private final Marker[] markerArray;
332 
333         /**
334          * Kth cell belonging to [1-5] of the markerArray. No need for
335          * this to be serialized
336          */
337         private transient int k = -1;
338 
339         /**
340          * Constructor
341          *
342          * @param theMarkerArray marker array to be used
343          */
344         private Markers(final Marker[] theMarkerArray) {
345             MathUtils.checkNotNull(theMarkerArray);
346             markerArray = theMarkerArray;
347             for (int i = 1; i < PSQUARE_CONSTANT; i++) {
348                 markerArray[i].previous(markerArray[i - 1])
349                         .next(markerArray[i + 1]).index(i);
350             }
351             markerArray[0].previous(markerArray[0]).next(markerArray[1])
352                     .index(0);
353             markerArray[5].previous(markerArray[4]).next(markerArray[5])
354                     .index(5);
355         }
356 
357         /**
358          * Constructor
359          *
360          * @param initialFive elements required to build Marker
361          * @param p quantile required to be computed
362          */
363         private Markers(final List<Double> initialFive, final double p) {
364             this(createMarkerArray(initialFive, p));
365         }
366 
367         /**
368          * Creates a marker array using initial five elements and a quantile
369          *
370          * @param initialFive list of initial five elements
371          * @param p the pth quantile
372          * @return Marker array
373          */
374         private static Marker[] createMarkerArray(
375                 final List<Double> initialFive, final double p) {
376             final int countObserved =
377                     initialFive == null ? -1 : initialFive.size();
378             if (countObserved < PSQUARE_CONSTANT) {
379                 throw new InsufficientDataException(
380                         LocalizedFormats.INSUFFICIENT_OBSERVED_POINTS_IN_SAMPLE,
381                         countObserved, PSQUARE_CONSTANT);
382             }
383             Collections.sort(initialFive);
384             return new Marker[] {
385                     new Marker(),// Null Marker
386                     new Marker(initialFive.get(0), 1, 0, 1),
387                     new Marker(initialFive.get(1), 1 + 2 * p, p / 2, 2),
388                     new Marker(initialFive.get(2), 1 + 4 * p, p, 3),
389                     new Marker(initialFive.get(3), 3 + 2 * p, (1 + p) / 2, 4),
390                     new Marker(initialFive.get(4), 5, 1, 5) };
391         }
392 
393         /**
394          * {@inheritDoc}
395          */
396         @Override
397         public int hashCode() {
398             return Arrays.deepHashCode(markerArray);
399         }
400 
401         /**
402          * {@inheritDoc}.This equals method basically checks for marker array to
403          * be deep equals.
404          *
405          * @param o is the other object
406          * @return true if the object compares with this object are equivalent
407          */
408         @Override
409         public boolean equals(Object o) {
410             boolean result = false;
411             if (this == o) {
412                 result = true;
413             } else if (o instanceof Markers) {
414                 Markers that = (Markers) o;
415                 result = Arrays.deepEquals(markerArray, that.markerArray);
416             }
417             return result;

420         /**
421          * Process a data point
422          *
423          * @param inputDataPoint is the data point passed
424          * @return computed percentile
425          */
426         @Override
427         public double processDataPoint(final double inputDataPoint) {
428 
429             // 1. Find cell and update minima and maxima
430             final int kthCell = findCellAndUpdateMinMax(inputDataPoint);
431 
432             // 2. Increment positions
433             incrementPositions(1, kthCell + 1, 5);
434 
435             // 2a. Update desired position with increments
436             updateDesiredPositions();
437 
438             // 3. Adjust heights of m[2-4] if necessary
439             adjustHeightsOfMarkers();
440 
441             // 4. Return percentile
442             return getPercentileValue();
443         }
444 
445         /**
446          * Returns the percentile computed thus far.
447          *
448          * @return height of mid point marker
449          */
450         @Override
451         public double getPercentileValue() {
452             return height(3);
453         }
454 
455         /**
456          * Finds the cell where the input observation / value fits.
457          *
458          * @param observation the input value to be checked for
459          * @return kth cell (of the markers ranging from 1-5) where observed
460          *         sample fits
461          */
462         private int findCellAndUpdateMinMax(final double observation) {
463             k = -1;
464             if (observation < height(1)) {
465                 markerArray[1].markerHeight = observation;
466                 k = 1;
467             } else if (observation < height(2)) {
468                 k = 1;
469             } else if (observation < height(3)) {
470                 k = 2;
471             } else if (observation < height(4)) {
472                 k = 3;
473             } else if (observation <= height(5)) {
474                 k = 4;
475             } else {
476                 markerArray[5].markerHeight = observation;
477                 k = 4;
478             }
479             return k;
480         }
481 
482         /**
483          * Adjust marker heights by setting quantile estimates to middle markers.
484          */
485         private void adjustHeightsOfMarkers() {
486             for (int i = LOW; i <= HIGH; i++) {
487                 estimate(i);
488             }
489         }
490 
491         /**
492          * {@inheritDoc}
493          */
494         @Override
495         public double estimate(final int index) {
496             if (index < LOW || index > HIGH) {
497                 throw new OutOfRangeException(index, LOW, HIGH);
498             }
499             return markerArray[index].estimate();
500         }
501 
502         /**
503          * Increment positions by d. Refer to algorithm paper for the
504          * definition of d.
505          *
506          * @param d The increment value for the position
507          * @param startIndex start index of the marker array
508          * @param endIndex end index of the marker array
509          */
510         private void incrementPositions(final int d, final int startIndex,
511                 final int endIndex) {
512             for (int i = startIndex; i <= endIndex; i++) {
513                 markerArray[i].incrementPosition(d);
514             }
515         }
516 
517         /**
518          * Desired positions incremented by bucket width. The bucket width is
519          * basically the desired increments.
520          */
521         private void updateDesiredPositions() {
522             for (int i = 1; i < markerArray.length; i++) {
523                 markerArray[i].updateDesiredPosition();
524             }
525         }

527         /**
528          * Sets previous and next markers after default read is done.
529          *
530          * @param anInputStream the input stream to be deserialized
531          * @throws ClassNotFoundException thrown when a desired class not found
532          * @throws IOException thrown due to any io errors
533          */
534         private void readObject(ObjectInputStream anInputStream)
535                 throws ClassNotFoundException, IOException {
536             // always perform the default de-serialization first
537             anInputStream.defaultReadObject();
538             // Build links
539             for (int i = 1; i < PSQUARE_CONSTANT; i++) {
540                 markerArray[i].previous(markerArray[i - 1]).next(markerArray[i + 1]).index(i);
541             }
542             markerArray[0].previous(markerArray[0]).next(markerArray[1]).index(0);
543             markerArray[5].previous(markerArray[4]).next(markerArray[5]).index(5);
544         }

546         /**
547          * Return marker height given index
548          *
549          * @param markerIndex index of marker within (1,6)

552         @Override
553         public double height(final int markerIndex) {
554             if (markerIndex >= markerArray.length || markerIndex <= 0) {

556             }
557             return markerArray[markerIndex].markerHeight;

729         /**
730          * Difference between desired and actual position
731          *
732          * @return difference between desired and actual position
733          */
734         private double difference() {
735             return desiredMarkerPosition - intMarkerPosition;
736         }
737 
738         /**
739          * Estimate the quantile for the current marker.
740          *
741          * @return estimated quantile
742          */
743         private double estimate() {
744             final double di = difference();
745             final boolean isNextHigher =
746                     next.intMarkerPosition - intMarkerPosition > 1;
747             final boolean isPreviousLower =
748                     previous.intMarkerPosition - intMarkerPosition < -1;
749 
750             if (di >= 1 && isNextHigher || di <= -1 && isPreviousLower) {
751                 final int d = di >= 0 ? 1 : -1;
752                 final double[] xval =
753                         new double[] { previous.intMarkerPosition,
754                                 intMarkerPosition, next.intMarkerPosition };
755                 final double[] yval =
756                         new double[] { previous.markerHeight, markerHeight,
757                                 next.markerHeight };
758                 final double xD = intMarkerPosition + d;
759 
760                 UnivariateFunction univariateFunction =
761                         nonLinear.interpolate(xval, yval);
762                 markerHeight = univariateFunction.value(xD);
763 
764                 // If parabolic estimate is bad then turn linear
765                 if (isEstimateBad(yval, markerHeight)) {
766                     int delta = xD - xval[1] > 0 ? 1 : -1;
767                     final double[] xBad =
768                             new double[] { xval[1], xval[1 + delta] };
769                     final double[] yBad =
770                             new double[] { yval[1], yval[1 + delta] };
771                     MathArrays.sortInPlace(xBad, yBad);// since d can be +/- 1
772                     univariateFunction = linear.interpolate(xBad, yBad);
773                     markerHeight = univariateFunction.value(xD);
774                 }
775                 incrementPosition(d);
776             }
777             return markerHeight;
778         }

780         /**
781          * Check if parabolic/nonlinear estimate is bad by checking if the
782          * ordinate found is beyond the y[0] and y[2].
783          *
784          * @param y the array to get the bounds
785          * @param yD the estimate
786          * @return true if yD is a bad estimate
787          */
788         private boolean isEstimateBad(final double[] y, final double yD) {
789             return yD <= y[0] || yD >= y[2];
790         }

795          * between this and passed in object</i>

798          * @return true if this equals passed in other object o

802             boolean result = false;
803             if (this == o) {
804                 result = true;
805             } else if (o instanceof Marker) {
806                 Marker that = (Marker) o;
807 
808                 result = Double.compare(markerHeight, that.markerHeight) == 0;
809                 result =
810                         result &&
811                                 Double.compare(intMarkerPosition,
812                                         that.intMarkerPosition) == 0;
813                 result =
814                         result &&
815                                 Double.compare(desiredMarkerPosition,
816                                         that.desiredMarkerPosition) == 0;
817                 result =
818                         result &&
819                                 Double.compare(desiredMarkerIncrement,
820                                         that.desiredMarkerIncrement) == 0;
821 
822                 result = result && next.index == that.next.index;
823                 result = result && previous.index == that.previous.index;
824             }
825             return result;
826         }

874     /**
875      * A simple fixed capacity list that has an upper bound to growth.
876      * Once its capacity is reached, {@code add} is a no-op, returning
877      * {@code false}.
878      *
879      * @param <E>
880      */
881     private static class FixedCapacityList<E> extends ArrayList<E> implements Serializable {
882 
883         /**
884          * Serialization Version Id
885          */
886         private static final long serialVersionUID = 2283952083075725479L;
887         /**
888          * Capacity of the list
889          */
890         private final int capacity;
891 
892         /**
893          * This constructor constructs the list with given capacity and as well
894          * as stores the capacity
895          *
896          * @param fixedCapacity the capacity to be fixed for this list
897          */
898         FixedCapacityList(final int fixedCapacity) {
899             super(fixedCapacity);
900             this.capacity = fixedCapacity;
901         }
902 
903         /**
904          * {@inheritDoc} In addition it checks if the {@link #size()} returns a
905          * size that is within capacity and if true it adds; otherwise the list
906          * contents are unchanged and {@code false} is returned.
907          *
908          * @return true if addition is successful and false otherwise
909          */
910         @Override
911         public boolean add(final E e) {
912             return size() < capacity ? super.add(e) : false;
913         }
914 
915         /**
916          * {@inheritDoc} In addition it checks if the sum of Collection size and
917          * this instance's {@link #size()} returns a value that is within
918          * capacity and if true it adds the collection; otherwise the list
919          * contents are unchanged and {@code false} is returned.
920          *
921          * @return true if addition is successful and false otherwise
922          */
923         @Override
924         public boolean addAll(Collection<? extends E> collection) {
925             boolean isCollectionLess =
926                     collection != null &&
927                             collection.size() + size() <= capacity;
928             return isCollectionLess ? super.addAll(collection) : false;
929         }
930     }
