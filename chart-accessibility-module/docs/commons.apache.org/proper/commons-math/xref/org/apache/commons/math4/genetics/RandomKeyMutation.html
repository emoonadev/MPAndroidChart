17  package org.apache.commons.math4.genetics;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import org.apache.commons.math4.exception.MathIllegalArgumentException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  
25  /**
26   * Mutation operator for {@link RandomKey}s. Changes a randomly chosen element
27   * of the array representation to a random value uniformly distributed in [0,1].
28   *
29   * @since 2.0
30   */
31  public class RandomKeyMutation implements MutationPolicy {
32  
33      /**
34       * {@inheritDoc}
35       *
36       * @throws MathIllegalArgumentException if <code>original</code> is not a {@link RandomKey} instance
37       */
38      @Override
39      public Chromosome mutate(final Chromosome original) throws MathIllegalArgumentException {
40          if (!(original instanceof RandomKey<?>)) {
41              throw new MathIllegalArgumentException(LocalizedFormats.RANDOMKEY_MUTATION_WRONG_CLASS,
42                                                     original.getClass().getSimpleName());
43          }
44  
45          RandomKey<?> originalRk = (RandomKey<?>) original;
46          List<Double> repr = originalRk.getRepresentation();
47          int rInd = GeneticAlgorithm.getRandomGenerator().nextInt(repr.size());
48  
49          List<Double> newRepr = new ArrayList<Double> (repr);
50          newRepr.set(rInd, GeneticAlgorithm.getRandomGenerator().nextDouble());
51  
52          return originalRk.newFixedLengthChromosome(newRepr);
53      }
54  
55  }
