18  package org.apache.commons.math4.linear;
19  
20  /**
21   * Default implementation of the {@link RealMatrixPreservingVisitor} interface.
22   * <p>
23   * This class is a convenience to create custom visitors without defining all
24   * methods. This class provides default implementations that do nothing.
25   * </p>
26   *
27   * @since 2.0
28   */
29  public class DefaultRealMatrixPreservingVisitor implements RealMatrixPreservingVisitor {
30      /** {@inheritDoc} */
31      @Override
32      public void start(int rows, int columns,
33                        int startRow, int endRow, int startColumn, int endColumn) {
34      }
35  
36      /** {@inheritDoc} */
37      @Override
38      public void visit(int row, int column, double value) {}
39  
40      /** {@inheritDoc} */
41      @Override
42      public double end() {
43          return 0;
44      }
45  }
