17  package org.apache.commons.math4.ode;
18  
19  import org.apache.commons.math4.exception.DimensionMismatchException;
20  import org.apache.commons.math4.exception.MaxCountExceededException;
21  
22  /** Interface expanding {@link FirstOrderDifferentialEquations first order
23   *  differential equations} in order to compute exactly the main state jacobian
24   *  matrix for {@link JacobianMatrices partial derivatives equations}.
25   *
26   * @since 3.0
27   */
28  public interface MainStateJacobianProvider extends FirstOrderDifferentialEquations {
29  
30      /** Compute the jacobian matrix of ODE with respect to main state.
31       * @param t current value of the independent <I>time</I> variable
32       * @param y array containing the current value of the main state vector
33       * @param yDot array containing the current value of the time derivative of the main state vector
34       * @param dFdY placeholder array where to put the jacobian matrix of the ODE w.r.t. the main state vector
35       * @exception MaxCountExceededException if the number of functions evaluations is exceeded
36       * @exception DimensionMismatchException if arrays dimensions do not match equations settings
37       */
38      void computeMainStateJacobian(double t, double[] y, double[] yDot, double[][] dFdY)
39          throws MaxCountExceededException, DimensionMismatchException;
40  
41  }
