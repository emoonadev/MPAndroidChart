18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.exception.util.LocalizedFormats;
23  import org.apache.commons.rng.UniformRandomProvider;
24  import org.apache.commons.math4.special.Erf;
25  import org.apache.commons.math4.util.FastMath;
26  
27  /**
28   * Implementation of the log-normal (gaussian) distribution.
29   *
30   * <p>
31   * <strong>Parameters:</strong>
32   * {@code X} is log-normally distributed if its natural logarithm {@code log(X)}
33   * is normally distributed. The probability distribution function of {@code X}
34   * is given by (for {@code x > 0})
35   * </p>
36   * <p>
37   * {@code exp(-0.5 * ((ln(x) - m) / s)^2) / (s * sqrt(2 * pi) * x)}
38   * </p>
39   * <ul>
40   * <li>{@code m} is the <em>scale</em> parameter: this is the mean of the
41   * normally distributed natural logarithm of this distribution,</li>
42   * <li>{@code s} is the <em>shape</em> parameter: this is the standard
43   * deviation of the normally distributed natural logarithm of this
44   * distribution.
45   * </ul>
46   *
47   * @see <a href="http://en.wikipedia.org/wiki/Log-normal_distribution">
48   * Log-normal distribution (Wikipedia)</a>
49   * @see <a href="http://mathworld.wolfram.com/LogNormalDistribution.html">
50   * Log Normal distribution (MathWorld)</a>
51   *
52   * @since 3.0
53   */
54  public class LogNormalDistribution extends AbstractRealDistribution {
55      /** Default inverse cumulative probability accuracy. */
56      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
57  
58      /** Serializable version identifier. */
59      private static final long serialVersionUID = 20120112;
60  
61      /** &radic;(2 &pi;) */
62      private static final double SQRT2PI = FastMath.sqrt(2 * FastMath.PI);
63  
64      /** &radic;(2) */
65      private static final double SQRT2 = FastMath.sqrt(2.0);
66  
67      /** The scale parameter of this distribution. */
68      private final double scale;
69  
70      /** The shape parameter of this distribution. */
71      private final double shape;
72      /** The value of {@code log(shape) + 0.5 * log(2*PI)} stored for faster computation. */
73      private final double logShapePlusHalfLog2Pi;
74  
75      /** Inverse cumulative probability accuracy. */
76      private final double solverAbsoluteAccuracy;
77  
78      /**
79       * Creates a log-normal distribution, where the mean and standard deviation
80       * of the {@link NormalDistribution normally distributed} natural
81       * logarithm of the log-normal distribution are equal to zero and one
82       * respectively. In other words, the scale of the returned distribution is
83       * {@code 0}, while its shape is {@code 1}.
84       */
85      public LogNormalDistribution() {
86          this(0, 1);
87      }
88  
89      /**
90       * Creates a log-normal distribution.
91       *
92       * @param scale Scale parameter of this distribution.
93       * @param shape Shape parameter of this distribution.
94       * @throws NotStrictlyPositiveException if {@code shape <= 0}.
95       */
96      public LogNormalDistribution(double scale, double shape)
97          throws NotStrictlyPositiveException {
98          this(scale, shape, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
99      }
100 
101     /**
102      * Creates a log-normal distribution.
103      *
104      * @param scale Scale parameter of this distribution.
105      * @param shape Shape parameter of this distribution.
106      * @param inverseCumAccuracy Inverse cumulative probability accuracy.
107      * @throws NotStrictlyPositiveException if {@code shape <= 0}.
108      */
109     public LogNormalDistribution(double scale,
110                                  double shape,
111                                  double inverseCumAccuracy)
112         throws NotStrictlyPositiveException {
113         if (shape <= 0) {
114             throw new NotStrictlyPositiveException(LocalizedFormats.SHAPE, shape);
115         }
116 
117         this.scale = scale;
118         this.shape = shape;
119         this.logShapePlusHalfLog2Pi = FastMath.log(shape) + 0.5 * FastMath.log(2 * FastMath.PI);
120         this.solverAbsoluteAccuracy = inverseCumAccuracy;
121     }
122 
123     /**
124      * Returns the scale parameter of this distribution.
125      *
126      * @return the scale parameter
127      */
128     public double getScale() {
129         return scale;
130     }
131 
132     /**
133      * Returns the shape parameter of this distribution.
134      *
135      * @return the shape parameter
136      */
137     public double getShape() {
138         return shape;
139     }
140 
141     /**
142      * {@inheritDoc}
143      *
144      * For scale {@code m}, and shape {@code s} of this distribution, the PDF
145      * is given by
146      * <ul>
147      * <li>{@code 0} if {@code x <= 0},</li>
148      * <li>{@code exp(-0.5 * ((ln(x) - m) / s)^2) / (s * sqrt(2 * pi) * x)}
149      * otherwise.</li>
150      * </ul>
151      */
152     @Override
153     public double density(double x) {
154         if (x <= 0) {
155             return 0;
156         }
157         final double x0 = FastMath.log(x) - scale;
158         final double x1 = x0 / shape;
159         return FastMath.exp(-0.5 * x1 * x1) / (shape * SQRT2PI * x);
160     }
161 
162     /** {@inheritDoc}
163      *
164      * See documentation of {@link #density(double)} for computation details.
165      */
166     @Override
167     public double logDensity(double x) {
168         if (x <= 0) {
169             return Double.NEGATIVE_INFINITY;
170         }
171         final double logX = FastMath.log(x);
172         final double x0 = logX - scale;
173         final double x1 = x0 / shape;
174         return -0.5 * x1 * x1 - (logShapePlusHalfLog2Pi + logX);
175     }
176 
177     /**
178      * {@inheritDoc}
179      *
180      * For scale {@code m}, and shape {@code s} of this distribution, the CDF
181      * is given by
182      * <ul>
183      * <li>{@code 0} if {@code x <= 0},</li>
184      * <li>{@code 0} if {@code ln(x) - m < 0} and {@code m - ln(x) > 40 * s}, as
185      * in these cases the actual value is within {@code Double.MIN_VALUE} of 0,
186      * <li>{@code 1} if {@code ln(x) - m >= 0} and {@code ln(x) - m > 40 * s},
187      * as in these cases the actual value is within {@code Double.MIN_VALUE} of
188      * 1,</li>
189      * <li>{@code 0.5 + 0.5 * erf((ln(x) - m) / (s * sqrt(2))} otherwise.</li>
190      * </ul>
191      */
192     @Override
193     public double cumulativeProbability(double x)  {
194         if (x <= 0) {
195             return 0;
196         }
197         final double dev = FastMath.log(x) - scale;
198         if (FastMath.abs(dev) > 40 * shape) {
199             return dev < 0 ? 0.0d : 1.0d;
200         }
201         return 0.5 + 0.5 * Erf.erf(dev / (shape * SQRT2));
202     }
203 
204     /** {@inheritDoc} */
205     @Override
206     public double probability(double x0,
207                               double x1)
208         throws NumberIsTooLargeException {
209         if (x0 > x1) {
210             throw new NumberIsTooLargeException(LocalizedFormats.LOWER_ENDPOINT_ABOVE_UPPER_ENDPOINT,
211                                                 x0, x1, true);
212         }
213         if (x0 <= 0 || x1 <= 0) {
214             return super.probability(x0, x1);
215         }
216         final double denom = shape * SQRT2;
217         final double v0 = (FastMath.log(x0) - scale) / denom;
218         final double v1 = (FastMath.log(x1) - scale) / denom;
219         return 0.5 * Erf.erf(v0, v1);
220     }
221 
222     /** {@inheritDoc} */
223     @Override
224     protected double getSolverAbsoluteAccuracy() {
225         return solverAbsoluteAccuracy;
226     }
227 
228     /**
229      * {@inheritDoc}
230      *
231      * For scale {@code m} and shape {@code s}, the mean is
232      * {@code exp(m + s^2 / 2)}.
233      */
234     @Override
235     public double getNumericalMean() {
236         double s = shape;
237         return FastMath.exp(scale + (s * s / 2));
238     }
239 
240     /**
241      * {@inheritDoc}
242      *
243      * For scale {@code m} and shape {@code s}, the variance is
244      * {@code (exp(s^2) - 1) * exp(2 * m + s^2)}.
245      */
246     @Override
247     public double getNumericalVariance() {
248         final double s = shape;
249         final double ss = s * s;
250         return (FastMath.expm1(ss)) * FastMath.exp(2 * scale + ss);
251     }
252 
253     /**
254      * {@inheritDoc}
255      *
256      * The lower bound of the support is always 0 no matter the parameters.
257      *
258      * @return lower bound of the support (always 0)
259      */
260     @Override
261     public double getSupportLowerBound() {
262         return 0;
263     }
264 
265     /**
266      * {@inheritDoc}
267      *
268      * The upper bound of the support is always positive infinity
269      * no matter the parameters.
270      *
271      * @return upper bound of the support (always
272      * {@code Double.POSITIVE_INFINITY})
273      */
274     @Override
275     public double getSupportUpperBound() {
276         return Double.POSITIVE_INFINITY;
277     }
278 
279     /**
280      * {@inheritDoc}
281      *
282      * The support of this distribution is connected.
283      *
284      * @return {@code true}
285      */
286     @Override
287     public boolean isSupportConnected() {
288         return true;
289     }
290 
291     /** {@inheritDoc} */
292     @Override
293     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
294         return new RealDistribution.Sampler() {
295             /** Gaussian sampling. */
296             private final RealDistribution.Sampler gaussian = new NormalDistribution().createSampler(rng);
297 
298             /** {@inheritDoc} */
299             @Override
300             public double sample() {
301                 final double n = gaussian.sample();
302                 return FastMath.exp(scale + shape * n);
303             }
304         };
305     }
