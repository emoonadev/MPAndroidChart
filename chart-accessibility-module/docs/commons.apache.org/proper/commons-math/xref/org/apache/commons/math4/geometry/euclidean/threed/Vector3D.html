101     /** Simple constructor.
102      * Build a vector from its coordinates
103      * @param v coordinates array
104      * @exception DimensionMismatchException if array does not have 3 elements
105      * @see #toArray()
106      */
107     public Vector3D(double[] v) throws DimensionMismatchException {
108         if (v.length != 3) {
109             throw new DimensionMismatchException(v.length, 3);
110         }
111         this.x = v[0];
112         this.y = v[1];
113         this.z = v[2];
114     }

116     /** Simple constructor.
117      * Build a vector from its azimuthal coordinates
118      * @param alpha azimuth (&alpha;) around Z
119      *              (0 is +X, &pi;/2 is +Y, &pi; is -X and 3&pi;/2 is -Y)
120      * @param delta elevation (&delta;) above (XY) plane, from -&pi;/2 to +&pi;/2
121      * @see #getAlpha()
122      * @see #getDelta()
123      */
124     public Vector3D(double alpha, double delta) {
125         double cosDelta = FastMath.cos(delta);
126         this.x = FastMath.cos(alpha) * cosDelta;
127         this.y = FastMath.sin(alpha) * cosDelta;
128         this.z = FastMath.sin(delta);
129     }

134      * @param a scale factor
135      * @param u base (unscaled) vector
136      */
137     public Vector3D(double a, Vector3D u) {
138         this.x = a * u.x;
139         this.y = a * u.y;

143     /** Linear constructor
144      * Build a vector from two other ones and corresponding scale factors.
145      * The vector built will be a1 * u1 + a2 * u2
146      * @param a1 first scale factor
147      * @param u1 first base (unscaled) vector
148      * @param a2 second scale factor
149      * @param u2 second base (unscaled) vector
150      */
151     public Vector3D(double a1, Vector3D u1, double a2, Vector3D u2) {
152         this.x = MathArrays.linearCombination(a1, u1.x, a2, u2.x);
153         this.y = MathArrays.linearCombination(a1, u1.y, a2, u2.y);
154         this.z = MathArrays.linearCombination(a1, u1.z, a2, u2.z);
155     }

305     /** {@inheritDoc} */
306     @Override
307     public Vector3D normalize() throws MathArithmeticException {
308         double s = getNorm();
309         if (s == 0) {
310             throw new MathArithmeticException(LocalizedFormats.CANNOT_NORMALIZE_A_ZERO_NORM_VECTOR);
311         }
312         return scalarMultiply(1 / s);
313     }

315     /** Get a vector orthogonal to the instance.
316      * <p>There are an infinite number of normalized vectors orthogonal
317      * to the instance. This method picks up one of them almost
318      * arbitrarily. It is useful when one needs to compute a reference
319      * frame with one of the axes in a predefined direction. The
320      * following example shows how to build a frame having the k axis
321      * aligned with the known vector u :
322      * <pre><code>
323      *   Vector3D k = u.normalize();
324      *   Vector3D i = k.orthogonal();
325      *   Vector3D j = Vector3D.crossProduct(k, i);
326      * </code></pre></p>
327      * @return a new normalized vector orthogonal to the instance
328      * @exception MathArithmeticException if the norm of the instance is null
329      */
330     public Vector3D orthogonal() throws MathArithmeticException {
331 
332         double threshold = 0.6 * getNorm();
333         if (threshold == 0) {
334             throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
335         }
336 
337         if (FastMath.abs(x) <= threshold) {
338             double inverse  = 1 / FastMath.sqrt(y * y + z * z);
339             return new Vector3D(0, inverse * z, -inverse * y);
340         } else if (FastMath.abs(y) <= threshold) {
341             double inverse  = 1 / FastMath.sqrt(x * x + z * z);
342             return new Vector3D(-inverse * z, 0, inverse * x);
343         }
344         double inverse  = 1 / FastMath.sqrt(x * x + y * y);
345         return new Vector3D(inverse * y, -inverse * x, 0);
346 
347     }

349     /** Compute the angular separation between two vectors.
350      * <p>This method computes the angular separation between two
351      * vectors using the dot product for well separated vectors and the
352      * cross product for almost aligned vectors. This allows to have a
353      * good accuracy in all cases, even for vectors very close to each
354      * other.</p>
355      * @param v1 first vector
356      * @param v2 second vector
357      * @return angular separation between v1 and v2
358      * @exception MathArithmeticException if either vector has a null norm
359      */
360     public static double angle(Vector3D v1, Vector3D v2) throws MathArithmeticException {
361 
362         double normProduct = v1.getNorm() * v2.getNorm();
363         if (normProduct == 0) {
364             throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
365         }
366 
367         double dot = v1.dotProduct(v2);
368         double threshold = normProduct * 0.9999;
369         if ((dot < -threshold) || (dot > threshold)) {
370             // the vectors are almost aligned, compute using the sine
371             Vector3D v3 = crossProduct(v1, v2);
372             if (dot >= 0) {
373                 return FastMath.asin(v3.getNorm() / normProduct);
374             }
375             return FastMath.PI - FastMath.asin(v3.getNorm() / normProduct);
376         }
377 
378         // the vectors are sufficiently separated to use the cosine
379         return FastMath.acos(dot / normProduct);
380 
381     }

407     /**
408      * Test for the equality of two 3D vectors.
409      * <p>
410      * If all coordinates of two 3D vectors are exactly the same, and none are
411      * <code>Double.NaN</code>, the two 3D vectors are considered to be equal.
412      * </p>
413      * <p>
414      * <code>NaN</code> coordinates are considered to affect globally the vector
415      * and be equals to each other - i.e, if either (or all) coordinates of the
416      * 3D vector are equal to <code>Double.NaN</code>, the 3D vector is equal to
417      * {@link #NaN}.
418      * </p>
419      *
420      * @param other Object to test for equality to this
421      * @return true if two 3D vector objects are equal, false if
422      *         object is null, not an instance of Vector3D, or
423      *         not equal to this Vector3D instance
424      *
425      */
426     @Override
427     public boolean equals(Object other) {
428 
429         if (this == other) {
430             return true;
431         }
432 
433         if (other instanceof Vector3D) {
434             final Vector3D rhs = (Vector3D)other;
435             if (rhs.isNaN()) {
436                 return this.isNaN();
437             }
438 
439             return (x == rhs.x) && (y == rhs.y) && (z == rhs.z);
440         }
441         return false;
442     }

444     /**
445      * Get a hashCode for the 3D vector.
446      * <p>
447      * All NaN values have the same hash code.</p>
448      *
449      * @return a hash code value for this object
450      */
451     @Override
452     public int hashCode() {
453         if (isNaN()) {
454             return 642;
455         }
456         return 643 * (164 * MathUtils.hash(x) +  3 * MathUtils.hash(y) +  MathUtils.hash(z));
457     }

462      * algorithms to preserve accuracy and reduce cancellation effects.

470         return MathArrays.linearCombination(x, v3.x, y, v3.y, z, v3.z);

473     /** Compute the cross-product of the instance with another vector.
474      * @param v other vector
475      * @return the cross product this ^ v as a new Vector3D
476      */
477     public Vector3D crossProduct(final Vector<Euclidean3D> v) {
478         final Vector3D v3 = (Vector3D) v;
479         return new Vector3D(MathArrays.linearCombination(y, v3.z, -z, v3.y),
480                             MathArrays.linearCombination(z, v3.x, -x, v3.z),
481                             MathArrays.linearCombination(x, v3.y, -y, v3.x));
482     }

488         final double dx = FastMath.abs(v3.x - x);
489         final double dy = FastMath.abs(v3.y - y);
490         final double dz = FastMath.abs(v3.z - z);
491         return dx + dy + dz;
