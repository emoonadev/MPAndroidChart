25  /**
26   * Implementation of the Gamma distribution.
27   *
28   * @see <a href="http://en.wikipedia.org/wiki/Gamma_distribution">Gamma distribution (Wikipedia)</a>
29   * @see <a href="http://mathworld.wolfram.com/GammaDistribution.html">Gamma distribution (MathWorld)</a>
30   */
31  public class GammaDistribution extends AbstractRealDistribution {
32      /**
33       * Default inverse cumulative probability accuracy.
34       * @since 2.1
35       */
36      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
37      /** Serializable version identifier. */
38      private static final long serialVersionUID = 20160311L;
39      /** The shape parameter. */
40      private final double shape;
41      /** The scale parameter. */
42      private final double scale;
43      /**
44       * The constant value of {@code shape + g + 0.5}, where {@code g} is the
45       * Lanczos constant {@link Gamma#LANCZOS_G}.
46       */
47      private final double shiftedShape;
48      /**
49       * The constant value of
50       * {@code shape / scale * sqrt(e / (2 * pi * (shape + g + 0.5))) / L(shape)},
51       * where {@code L(shape)} is the Lanczos approximation returned by
52       * {@link Gamma#lanczos(double)}. This prefactor is used in
53       * {@link #density(double)}, when no overflow occurs with the natural
54       * calculation.
55       */
56      private final double densityPrefactor1;
57      /**
58       * The constant value of
59       * {@code log(shape / scale * sqrt(e / (2 * pi * (shape + g + 0.5))) / L(shape))},
60       * where {@code L(shape)} is the Lanczos approximation returned by
61       * {@link Gamma#lanczos(double)}. This prefactor is used in
62       * {@link #logDensity(double)}, when no overflow occurs with the natural
63       * calculation.
64       */
65      private final double logDensityPrefactor1;
66      /**
67       * The constant value of
68       * {@code shape * sqrt(e / (2 * pi * (shape + g + 0.5))) / L(shape)},
69       * where {@code L(shape)} is the Lanczos approximation returned by
70       * {@link Gamma#lanczos(double)}. This prefactor is used in
71       * {@link #density(double)}, when overflow occurs with the natural
72       * calculation.
73       */
74      private final double densityPrefactor2;
75      /**
76       * The constant value of
77       * {@code log(shape * sqrt(e / (2 * pi * (shape + g + 0.5))) / L(shape))},
78       * where {@code L(shape)} is the Lanczos approximation returned by
79       * {@link Gamma#lanczos(double)}. This prefactor is used in
80       * {@link #logDensity(double)}, when overflow occurs with the natural
81       * calculation.
82       */
83      private final double logDensityPrefactor2;
84      /**
85       * Lower bound on {@code y = x / scale} for the selection of the computation
86       * method in {@link #density(double)}. For {@code y <= minY}, the natural
87       * calculation overflows.
88       */
89      private final double minY;
90      /**
91       * Upper bound on {@code log(y)} ({@code y = x / scale}) for the selection
92       * of the computation method in {@link #density(double)}. For
93       * {@code log(y) >= maxLogY}, the natural calculation overflows.
94       */
95      private final double maxLogY;
96      /** Inverse cumulative probability accuracy. */
97      private final double solverAbsoluteAccuracy;
98  
99      /**
100      * Creates a distribution.
101      *
102      * @param shape the shape parameter
103      * @param scale the scale parameter
104      * @throws NotStrictlyPositiveException if {@code shape <= 0} or
105      * {@code scale <= 0}.
106      */
107     public GammaDistribution(double shape, double scale)
108         throws NotStrictlyPositiveException {
109         this(shape, scale, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
110     }
111 
112     /**
113      * Creates a distribution.
114      *
115      * @param shape the shape parameter
116      * @param scale the scale parameter
117      * @param inverseCumAccuracy the maximum absolute error in inverse
118      * cumulative probability estimates (defaults to
119      * {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
120      * @throws NotStrictlyPositiveException if {@code shape <= 0} or
121      * {@code scale <= 0}.
122      */
123     public GammaDistribution(double shape,
124                              double scale,
125                              double inverseCumAccuracy)
126         throws NotStrictlyPositiveException {
127         if (shape <= 0) {
128             throw new NotStrictlyPositiveException(LocalizedFormats.SHAPE, shape);
129         }
130         if (scale <= 0) {
131             throw new NotStrictlyPositiveException(LocalizedFormats.SCALE, scale);
132         }
133 
134         this.shape = shape;
135         this.scale = scale;
136         this.solverAbsoluteAccuracy = inverseCumAccuracy;
137         this.shiftedShape = shape + Gamma.LANCZOS_G + 0.5;
138         final double aux = FastMath.E / (2.0 * FastMath.PI * shiftedShape);
139         this.densityPrefactor2 = shape * FastMath.sqrt(aux) / Gamma.lanczos(shape);
140         this.logDensityPrefactor2 = FastMath.log(shape) + 0.5 * FastMath.log(aux) -
141                                     FastMath.log(Gamma.lanczos(shape));
142         this.densityPrefactor1 = this.densityPrefactor2 / scale *
143                 FastMath.pow(shiftedShape, -shape) *
144                 FastMath.exp(shape + Gamma.LANCZOS_G);
145         this.logDensityPrefactor1 = this.logDensityPrefactor2 - FastMath.log(scale) -
146                 FastMath.log(shiftedShape) * shape +
147                 shape + Gamma.LANCZOS_G;
148         this.minY = shape + Gamma.LANCZOS_G - FastMath.log(Double.MAX_VALUE);
149         this.maxLogY = FastMath.log(Double.MAX_VALUE) / (shape - 1.0);
150     }

172     /** {@inheritDoc} */
173     @Override
174     public double density(double x) {
175        /* The present method must return the value of
176         *
177         *     1       x a     - x
178         * ---------- (-)  exp(---)
179         * x Gamma(a)  b        b
180         *
181         * where a is the shape parameter, and b the scale parameter.
182         * Substituting the Lanczos approximation of Gamma(a) leads to the
183         * following expression of the density
184         *
185         * a              e            1         y      a
186         * - sqrt(------------------) ---- (-----------)  exp(a - y + g),
187         * x      2 pi (a + g + 0.5)  L(a)  a + g + 0.5
188         *
189         * where y = x / b. The above formula is the "natural" computation, which
190         * is implemented when no overflow is likely to occur. If overflow occurs
191         * with the natural computation, the following identity is used. It is
192         * based on the BOOST library
193         * http://www.boost.org/doc/libs/1_35_0/libs/math/doc/sf_and_dist/html/math_toolkit/special/sf_gamma/igamma.html
194         * Formula (15) needs adaptations, which are detailed below.
195         *
196         *       y      a
197         * (-----------)  exp(a - y + g)
198         *  a + g + 0.5
199         *                              y - a - g - 0.5    y (g + 0.5)
200         *               = exp(a log1pm(---------------) - ----------- + g),
201         *                                a + g + 0.5      a + g + 0.5
202         *
203         *  where log1pm(z) = log(1 + z) - z. Therefore, the value to be
204         *  returned is
205         *
206         * a              e            1
207         * - sqrt(------------------) ----
208         * x      2 pi (a + g + 0.5)  L(a)
209         *                              y - a - g - 0.5    y (g + 0.5)
210         *               * exp(a log1pm(---------------) - ----------- + g).
211         *                                a + g + 0.5      a + g + 0.5
212         */
213         if (x < 0) {
214             return 0;
215         }
216         final double y = x / scale;
217         if ((y <= minY) || (FastMath.log(y) >= maxLogY)) {
218             /*
219              * Overflow.
220              */
221             final double aux1 = (y - shiftedShape) / shiftedShape;
222             final double aux2 = shape * (FastMath.log1p(aux1) - aux1);
223             final double aux3 = -y * (Gamma.LANCZOS_G + 0.5) / shiftedShape +
224                     Gamma.LANCZOS_G + aux2;
225             return densityPrefactor2 / x * FastMath.exp(aux3);
226         }
227         /*
228          * Natural calculation.
229          */
230         return densityPrefactor1 * FastMath.exp(-y) * FastMath.pow(y, shape - 1);
231     }

233     /** {@inheritDoc} **/
234     @Override
235     public double logDensity(double x) {
236         /*
237          * see the comment in {@link #density(double)} for computation details
238          */
239         if (x < 0) {
240             return Double.NEGATIVE_INFINITY;
241         }
242         final double y = x / scale;
243         if ((y <= minY) || (FastMath.log(y) >= maxLogY)) {
244             /*
245              * Overflow.
246              */
247             final double aux1 = (y - shiftedShape) / shiftedShape;
248             final double aux2 = shape * (FastMath.log1p(aux1) - aux1);
249             final double aux3 = -y * (Gamma.LANCZOS_G + 0.5) / shiftedShape +
250                     Gamma.LANCZOS_G + aux2;
251             return logDensityPrefactor2 - FastMath.log(x) + aux3;
252         }
253         /*
254          * Natural calculation.
255          */
256         return logDensityPrefactor1 - y + FastMath.log(y) * (shape - 1);
257     }

259     /**
260      * {@inheritDoc}
261      *
262      * The implementation of this method is based on:
263      * <ul>
264      *  <li>
265      *   <a href="http://mathworld.wolfram.com/Chi-SquaredDistribution.html">
266      *    Chi-Squared Distribution</a>, equation (9).
267      *  </li>
268      *  <li>Casella, G., & Berger, R. (1990). <i>Statistical Inference</i>.
269      *    Belmont, CA: Duxbury Press.
270      *  </li>
271      * </ul>
272      */
273     @Override
274     public double cumulativeProbability(double x) {
275         double ret;
276 
277         if (x <= 0) {
278             ret = 0;
279         } else {
280             ret = Gamma.regularizedGammaP(shape, x / scale);
281         }
282 
283         return ret;
284     }

353     /**
354      * {@inheritDoc}
355      *
356      * <p>
357      * Sampling algorithms:
358      * <ul>
359      *  <li>
360      *   For {@code 0 < shape < 1}:
361      *   <blockquote>
362      *    Ahrens, J. H. and Dieter, U.,
363      *    <i>Computer methods for sampling from gamma, beta, Poisson and binomial distributions,</i>
364      *    Computing, 12, 223-246, 1974.
365      *   </blockquote>
366      *  </li>
367      *  <li>
368      *  For {@code shape >= 1}:
369      *   <blockquote>
370      *   Marsaglia and Tsang, <i>A Simple Method for Generating
371      *   Gamma Variables.</i> ACM Transactions on Mathematical Software,
372      *   Volume 26 Issue 3, September, 2000.
373      *   </blockquote>
374      *  </li>
375      * </ul>
376      * </p>
377      */
378     @Override
379     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
380         return new RealDistribution.Sampler() {
381             /** Gaussian sampling. */
382             private final RealDistribution.Sampler gaussian = new NormalDistribution().createSampler(rng);
383 
384             /** {@inheritDoc} */
385             @Override
386             public double sample() {
387                 if (shape < 1) {
388                     // [1]: p. 228, Algorithm GS
389 
390                     while (true) {
391                         // Step 1:
392                         final double u = rng.nextDouble();
393                         final double bGS = 1 + shape / FastMath.E;
394                         final double p = bGS * u;
395 
396                         if (p <= 1) {
397                             // Step 2:
398 
399                             final double x = FastMath.pow(p, 1 / shape);
400                             final double u2 = rng.nextDouble();
401 
402                             if (u2 > FastMath.exp(-x)) {
403                                 // Reject
404                                 continue;
405                             } else {
406                                 return scale * x;
407                             }
408                         } else {
409                             // Step 3:
410 
411                             final double x = -1 * FastMath.log((bGS - p) / shape);
412                             final double u2 = rng.nextDouble();
413 
414                             if (u2 > FastMath.pow(x, shape - 1)) {
415                                 // Reject
416                                 continue;
417                             } else {
418                                 return scale * x;
419                             }
420                         }
421                     }
422                 }
423 
424                 // Now shape >= 1
425 
426                 final double d = shape - 0.333333333333333333;
427                 final double c = 1 / (3 * FastMath.sqrt(d));
428 
429                 while (true) {
430                     final double x = gaussian.sample();
431                     final double v = (1 + c * x) * (1 + c * x) * (1 + c * x);
432 
433                     if (v <= 0) {
434                         continue;
435                     }
436 
437                     final double x2 = x * x;
438                     final double u = rng.nextDouble();
439 
440                     // Squeeze
441                     if (u < 1 - 0.0331 * x2 * x2) {
442                         return scale * d * v;
443                     }
444 
445                     if (FastMath.log(u) < 0.5 * x2 + d * (1 - v + FastMath.log(v))) {
446                         return scale * d * v;
447                     }
448                 }
449             }
450         };
451     }
