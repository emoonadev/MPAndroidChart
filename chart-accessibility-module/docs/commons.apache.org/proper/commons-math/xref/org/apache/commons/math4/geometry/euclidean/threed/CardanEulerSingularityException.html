18  package org.apache.commons.math4.geometry.euclidean.threed;
19  
20  import org.apache.commons.math4.exception.MathIllegalStateException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  
23  /** This class represents exceptions thrown while extractiong Cardan
24   * or Euler angles from a rotation.
25  
26   * @since 1.2
27   */
28  public class CardanEulerSingularityException
29    extends MathIllegalStateException {
30  
31      /** Serializable version identifier */
32      private static final long serialVersionUID = -1360952845582206770L;
33  
34      /**
35       * Simple constructor.
36       * build an exception with a default message.
37       * @param isCardan if true, the rotation is related to Cardan angles,
38       * if false it is related to EulerAngles
39       */
40      public CardanEulerSingularityException(boolean isCardan) {
41          super(isCardan ? LocalizedFormats.CARDAN_ANGLES_SINGULARITY : LocalizedFormats.EULER_ANGLES_SINGULARITY);
42      }
43  
44  }
