37  /**
38   * <p>
39   * Computes summary statistics for a stream of data values added using the
40   * {@link #addValue(double) addValue} method. The data values are not stored in
41   * memory, so this class can be used to compute statistics for very large data
42   * streams.
43   * </p>
44   * <p>
45   * The {@link StorelessUnivariateStatistic} instances used to maintain summary
46   * state and compute statistics are configurable via setters. For example, the
47   * default implementation for the variance can be overridden by calling
48   * {@link #setVarianceImpl(StorelessUnivariateStatistic)}. Actual parameters to
49   * these methods must implement the {@link StorelessUnivariateStatistic}
50   * interface and configuration must be completed before <code>addValue</code>
51   * is called. No configuration is necessary to use the default, commons-math
52   * provided implementations.
53   * </p>
54   * <p>
55   * Note: This class is not thread-safe. Use
56   * {@link SynchronizedSummaryStatistics} if concurrent access from multiple
57   * threads is required.
58   * </p>
59   */
60  public class SummaryStatistics implements StatisticalSummary, Serializable {
61

65      /** count of values that have been added */
66      private long n = 0;

68      /** SecondMoment is used to compute the mean and variance */
69      private SecondMoment secondMoment = new SecondMoment();

71      /** sum of values that have been added */
72      private Sum sum = new Sum();
73  
74      /** sum of the square of each value that has been added */
75      private SumOfSquares sumsq = new SumOfSquares();
76  
77      /** min of values that have been added */
78      private Min min = new Min();

80      /** max of values that have been added */
81      private Max max = new Max();

145     /**
146      * Add a value to the data
147      * @param value the value to add
148      */
149     public void addValue(double value) {
150         sumImpl.increment(value);
151         sumsqImpl.increment(value);
152         minImpl.increment(value);
153         maxImpl.increment(value);
154         sumLogImpl.increment(value);
155         secondMoment.increment(value);
156         // If mean, variance or geomean have been overridden,
157         // need to increment these
158         if (meanImpl != mean) {
159             meanImpl.increment(value);
160         }
161         if (varianceImpl != variance) {
162             varianceImpl.increment(value);
163         }
164         if (geoMeanImpl != geoMean) {
165             geoMeanImpl.increment(value);
166         }
167         n++;
168     }

211     /**
212      * Returns the standard deviation of the values that have been added.
213      * <p>
214      * Double.NaN is returned if no values have been added.
215      * </p>
216      * @return the standard deviation
217      */
218     @Override
219     public double getStandardDeviation() {
220         double stdDev = Double.NaN;
221         if (getN() > 0) {
222             if (getN() > 1) {
223                 stdDev = FastMath.sqrt(getVariance());
224             } else {
225                 stdDev = 0.0;
226             }
227         }
228         return stdDev;
229     }
230 
231     /**
232      * Returns the quadratic mean, a.k.a.
233      * <a href="http://mathworld.wolfram.com/Root-Mean-Square.html">
234      * root-mean-square</a> of the available values
235      * @return The quadratic mean or {@code Double.NaN} if no values
236      * have been added.
237      */
238     public double getQuadraticMean() {
239         final long size = getN();
240         return size > 0 ? FastMath.sqrt(getSumsq() / size) : Double.NaN;
241     }

335     /**
336      * Generates a text report displaying summary statistics from values that
337      * have been added.
338      * @return String with line feeds displaying statistics
339      * @since 1.2
340      */
341     @Override
342     public String toString() {
343         StringBuilder outBuffer = new StringBuilder();
344         String endl = "\n";
345         outBuffer.append("SummaryStatistics:").append(endl);
346         outBuffer.append("n: ").append(getN()).append(endl);
347         outBuffer.append("min: ").append(getMin()).append(endl);
348         outBuffer.append("max: ").append(getMax()).append(endl);
349         outBuffer.append("sum: ").append(getSum()).append(endl);
350         outBuffer.append("mean: ").append(getMean()).append(endl);
351         outBuffer.append("geometric mean: ").append(getGeometricMean())
352             .append(endl);
353         outBuffer.append("variance: ").append(getVariance()).append(endl);
354         outBuffer.append("population variance: ").append(getPopulationVariance()).append(endl);
355         outBuffer.append("second moment: ").append(getSecondMoment()).append(endl);
356         outBuffer.append("sum of squares: ").append(getSumsq()).append(endl);
357         outBuffer.append("standard deviation: ").append(getStandardDeviation())
358             .append(endl);
359         outBuffer.append("sum of logs: ").append(getSumOfLogs()).append(endl);
360         return outBuffer.toString();
361     }

363     /**
364      * Resets all statistics and storage
365      */
366     public void clear() {
367         this.n = 0;
368         minImpl.clear();
369         maxImpl.clear();
370         sumImpl.clear();
371         sumLogImpl.clear();
372         sumsqImpl.clear();
373         geoMeanImpl.clear();
374         secondMoment.clear();
375         if (meanImpl != mean) {
376             meanImpl.clear();
377         }
378         if (varianceImpl != variance) {
379             varianceImpl.clear();
380         }
381     }

383     /**
384      * Returns true iff <code>object</code> is a
385      * <code>SummaryStatistics</code> instance and all statistics have the
386      * same values as this.
387      * @param object the object to test equality against.
388      * @return true if object equals this
389      */
390     @Override
391     public boolean equals(Object object) {
392         if (object == this) {
393             return true;
394         }
395         if (object instanceof SummaryStatistics == false) {
396             return false;
397         }
398         SummaryStatistics stat = (SummaryStatistics)object;
399         return Precision.equalsIncludingNaN(stat.getGeometricMean(), getGeometricMean()) &&
400                Precision.equalsIncludingNaN(stat.getMax(),           getMax())           &&
401                Precision.equalsIncludingNaN(stat.getMean(),          getMean())          &&
402                Precision.equalsIncludingNaN(stat.getMin(),           getMin())           &&
403                Precision.equalsIncludingNaN(stat.getN(),             getN())             &&
404                Precision.equalsIncludingNaN(stat.getSum(),           getSum())           &&
405                Precision.equalsIncludingNaN(stat.getSumsq(),         getSumsq())         &&
406                Precision.equalsIncludingNaN(stat.getVariance(),      getVariance());
407     }

409     /**
410      * Returns hash code based on values of statistics
411      * @return hash code
412      */
413     @Override
414     public int hashCode() {
415         int result = 31 + MathUtils.hash(getGeometricMean());
416         result = result * 31 + MathUtils.hash(getGeometricMean());
417         result = result * 31 + MathUtils.hash(getMax());
418         result = result * 31 + MathUtils.hash(getMean());
419         result = result * 31 + MathUtils.hash(getMin());
420         result = result * 31 + MathUtils.hash(getN());
421         result = result * 31 + MathUtils.hash(getSum());
422         result = result * 31 + MathUtils.hash(getSumsq());
423         result = result * 31 + MathUtils.hash(getVariance());
424         return result;
425     }

692     /**
693      * Copies source to dest.
694      * <p>Neither source nor dest can be null.</p>
695      *
696      * @param source SummaryStatistics to copy
697      * @param dest SummaryStatistics to copy to
698      * @throws NullArgumentException if either source or dest is null
699      */
700     public static void copy(SummaryStatistics source, SummaryStatistics dest)
701         throws NullArgumentException {
702         MathUtils.checkNotNull(source);
703         MathUtils.checkNotNull(dest);
704         dest.maxImpl = source.maxImpl.copy();
705         dest.minImpl = source.minImpl.copy();
706         dest.sumImpl = source.sumImpl.copy();
707         dest.sumLogImpl = source.sumLogImpl.copy();
708         dest.sumsqImpl = source.sumsqImpl.copy();
709         dest.secondMoment = source.secondMoment.copy();
710         dest.n = source.n;
711 
712         // Keep commons-math supplied statistics with embedded moments in synch
713         if (source.getVarianceImpl() instanceof Variance) {
714             dest.varianceImpl = new Variance(dest.secondMoment);
715         } else {
716             dest.varianceImpl = source.varianceImpl.copy();
717         }
718         if (source.meanImpl instanceof Mean) {
719             dest.meanImpl = new Mean(dest.secondMoment);
720         } else {
721             dest.meanImpl = source.meanImpl.copy();
722         }
723         if (source.getGeoMeanImpl() instanceof GeometricMean) {
724             dest.geoMeanImpl = new GeometricMean((SumOfLogs) dest.sumLogImpl);
725         } else {
726             dest.geoMeanImpl = source.geoMeanImpl.copy();
727         }
728 
729         // Make sure that if stat == statImpl in source, same
730         // holds in dest; otherwise copy stat
731         if (source.geoMean == source.geoMeanImpl) {
732             dest.geoMean = (GeometricMean) dest.geoMeanImpl;
733         } else {
734             GeometricMean.copy(source.geoMean, dest.geoMean);
735         }
736         if (source.max == source.maxImpl) {
737             dest.max = (Max) dest.maxImpl;
738         } else {
739             Max.copy(source.max, dest.max);
740         }
741         if (source.mean == source.meanImpl) {
742             dest.mean = (Mean) dest.meanImpl;
743         } else {
744             Mean.copy(source.mean, dest.mean);
745         }
746         if (source.min == source.minImpl) {
747             dest.min = (Min) dest.minImpl;
748         } else {
749             Min.copy(source.min, dest.min);
750         }
751         if (source.sum == source.sumImpl) {
752             dest.sum = (Sum) dest.sumImpl;
753         } else {
754             Sum.copy(source.sum, dest.sum);
755         }
756         if (source.variance == source.varianceImpl) {
757             dest.variance = (Variance) dest.varianceImpl;
758         } else {
759             Variance.copy(source.variance, dest.variance);
760         }
761         if (source.sumLog == source.sumLogImpl) {
762             dest.sumLog = (SumOfLogs) dest.sumLogImpl;
763         } else {
764             SumOfLogs.copy(source.sumLog, dest.sumLog);
765         }
766         if (source.sumsq == source.sumsqImpl) {
767             dest.sumsq = (SumOfSquares) dest.sumsqImpl;
768         } else {
769             SumOfSquares.copy(source.sumsq, dest.sumsq);
770         }
771     }
