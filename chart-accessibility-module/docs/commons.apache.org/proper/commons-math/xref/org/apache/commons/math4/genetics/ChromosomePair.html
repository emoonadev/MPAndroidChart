17  package org.apache.commons.math4.genetics;
18  
19  /**
20   * A pair of {@link Chromosome} objects.
21   * @since 2.0
22   *
23   */
24  public class ChromosomePair {
25      /** the first chromosome in the pair. */
26      private final Chromosome first;
27  
28      /** the second chromosome in the pair. */
29      private final Chromosome second;
30  
31      /**
32       * Create a chromosome pair.
33       *
34       * @param c1 the first chromosome.
35       * @param c2 the second chromosome.
36       */
37      public ChromosomePair(final Chromosome c1, final Chromosome c2) {
38          super();
39          first = c1;
40          second = c2;
41      }
42  
43      /**
44       * Access the first chromosome.
45       *
46       * @return the first chromosome.
47       */
48      public Chromosome getFirst() {
49          return first;
50      }
51  
52      /**
53       * Access the second chromosome.
54       *
55       * @return the second chromosome.
56       */
57      public Chromosome getSecond() {
58          return second;
59      }
60  
61      /** {@inheritDoc} */
62      @Override
63      public String toString() {
64          return String.format("(%s,%s)", getFirst(), getSecond());
65      }
66  }
