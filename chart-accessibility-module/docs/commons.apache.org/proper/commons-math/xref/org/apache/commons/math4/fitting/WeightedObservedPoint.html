17  package org.apache.commons.math4.fitting;
18  
19  import java.io.Serializable;
20  
21  /**
22   * This class is a simple container for weighted observed point in
23   * {@link AbstractCurveFitter curve fitting}.
24   * <p>Instances of this class are guaranteed to be immutable.</p>
25   * @since 2.0
26   */
27  public class WeightedObservedPoint implements Serializable {
28      /** Serializable version id. */
29      private static final long serialVersionUID = 5306874947404636157L;
30      /** Weight of the measurement in the fitting process. */
31      private final double weight;
32      /** Abscissa of the point. */
33      private final double x;
34      /** Observed value of the function at x. */
35      private final double y;
36  
37      /**
38       * Simple constructor.
39       *
40       * @param weight Weight of the measurement in the fitting process.
41       * @param x Abscissa of the measurement.
42       * @param y Ordinate of the measurement.
43       */
44      public WeightedObservedPoint(final double weight, final double x, final double y) {
45          this.weight = weight;
46          this.x      = x;
47          this.y      = y;
48      }
49  
50      /**
51       * Gets the weight of the measurement in the fitting process.
52       *
53       * @return the weight of the measurement in the fitting process.
54       */
55      public double getWeight() {
56          return weight;
57      }
58  
59      /**
60       * Gets the abscissa of the point.
61       *
62       * @return the abscissa of the point.
63       */
64      public double getX() {
65          return x;
66      }
67  
68      /**
69       * Gets the observed value of the function at x.
70       *
71       * @return the observed value of the function at x.
72       */
73      public double getY() {
74          return y;
75      }
76  
77  }
