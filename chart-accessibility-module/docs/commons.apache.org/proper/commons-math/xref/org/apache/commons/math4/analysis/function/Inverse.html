18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.differentiation.DerivativeStructure;
21  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
22  
23  /**
24   * Inverse function.
25   *
26   * @since 3.0
27   */
28  public class Inverse implements UnivariateDifferentiableFunction {
29      /** {@inheritDoc} */
30      @Override
31      public double value(double x) {
32          return 1 / x;
33      }
34  
35      /** {@inheritDoc}
36       * @since 3.1
37       */
38      @Override
39      public DerivativeStructure value(final DerivativeStructure t) {
40          return t.reciprocal();
41      }
42  
43  }
