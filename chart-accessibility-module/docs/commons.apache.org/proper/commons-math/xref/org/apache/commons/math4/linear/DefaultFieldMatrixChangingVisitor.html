18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.FieldElement;
21  
22  /**
23   * Default implementation of the {@link FieldMatrixChangingVisitor} interface.
24   * <p>
25   * This class is a convenience to create custom visitors without defining all
26   * methods. This class provides default implementations that do nothing.
27   * </p>
28   *
29   * @param <T> the type of the field elements
30   * @since 2.0
31   */
32  public class DefaultFieldMatrixChangingVisitor<T extends FieldElement<T>>
33      implements FieldMatrixChangingVisitor<T> {
34      /** Zero element of the field. */
35      private final T zero;
36  
37      /** Build a new instance.
38       * @param zero additive identity of the field
39       */
40      public DefaultFieldMatrixChangingVisitor(final T zero) {
41          this.zero = zero;
42      }
43  
44      /** {@inheritDoc} */
45      @Override
46      public void start(int rows, int columns,
47                        int startRow, int endRow, int startColumn, int endColumn) {
48      }
49  
50      /** {@inheritDoc} */
51      @Override
52      public T visit(int row, int column, T value) {
53          return value;
54      }
55  
56      /** {@inheritDoc} */
57      @Override
58      public T end() {
59          return zero;
60      }
61  }
