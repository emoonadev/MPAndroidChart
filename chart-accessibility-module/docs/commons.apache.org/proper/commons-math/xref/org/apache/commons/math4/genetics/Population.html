17  package org.apache.commons.math4.genetics;
18  
19  import org.apache.commons.math4.exception.NumberIsTooLargeException;
20  
21  
22  /**
23   * A collection of chromosomes that facilitates generational evolution.
24   *
25   * @since 2.0
26   */
27  public interface Population extends Iterable<Chromosome> {
28      /**
29       * Access the current population size.
30       * @return the current population size.
31       */
32      int getPopulationSize();
33  
34      /**
35       * Access the maximum population size.
36       * @return the maximum population size.
37       */
38      int getPopulationLimit();
39  
40      /**
41       * Start the population for the next generation.
42       * @return the beginnings of the next generation.
43       */
44      Population nextGeneration();
45  
46      /**
47       * Add the given chromosome to the population.
48       * @param chromosome the chromosome to add.
49       * @throws NumberIsTooLargeException if the population would exceed the population limit when adding
50       *   this chromosome
51       */
52      void addChromosome(Chromosome chromosome) throws NumberIsTooLargeException;
53  
54      /**
55       * Access the fittest chromosome in this population.
56       * @return the fittest chromosome.
57       */
58      Chromosome getFittestChromosome();
59  }
