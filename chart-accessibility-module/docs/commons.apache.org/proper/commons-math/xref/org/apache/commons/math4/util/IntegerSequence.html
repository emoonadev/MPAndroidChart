17  package org.apache.commons.math4.util;
18  
19  import java.util.Iterator;
20  import org.apache.commons.math4.exception.MaxCountExceededException;
21  import org.apache.commons.math4.exception.NullArgumentException;
22  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
23  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
24  import org.apache.commons.math4.exception.ZeroException;
25  
26  /**
27   * Provides a sequence of integers.
28   *
29   * @since 3.6
30   */
31  public class IntegerSequence {
32      /**
33       * Utility class contains only static methods.
34       */
35      private IntegerSequence() {}
36  
37      /**
38       * Creates a sequence {@code [start .. end]}.
39       * It calls {@link #range(int,int,int) range(start, end, 1)}.
40       *
41       * @param start First value of the range.
42       * @param end Last value of the range.
43       * @return a range.
44       */
45      public static Range range(int start,
46                                int end) {
47          return range(start, end, 1);
48      }
49  
50      /**
51       * Creates a sequence <code>a<sub>i</sub>, i < 0 < n</code>
52       * where <code>a<sub>i</sub> = start + i * step</code>
53       * and {@code n} is such that <code>a<sub>n</sub> <= max</code>
54       * and  <code>a<sub>n+1</sub> > max</code>.
55       *
56       * @param start First value of the range.
57       * @param max Last value of the range that satisfies the above
58       * construction rule.
59       * @param step Increment.
60       * @return a range.
61       */
62      public static Range range(final int start,
63                                final int max,
64                                final int step) {
65          return new Range(start, max, step);
66      }
67  
68      /**
69       * Generates a sequence of integers.
70       */
71      public static class Range implements Iterable<Integer> {
72          /** Number of integers contained in this range. */
73          private final int size;
74          /** First value. */
75          private final int start;
76          /** Final value. */
77          private final int max;
78          /** Increment. */
79          private final int step;
80  
81          /**
82           * Creates a sequence <code>a<sub>i</sub>, i < 0 < n</code>
83           * where <code>a<sub>i</sub> = start + i * step</code>
84           * and {@code n} is such that <code>a<sub>n</sub> <= max</code>
85           * and  <code>a<sub>n+1</sub> > max</code>.
86           *
87           * @param start First value of the range.
88           * @param max Last value of the range that satisfies the above
89           * construction rule.
90           * @param step Increment.
91           */
92          public Range(int start,
93                       int max,
94                       int step) {
95              this.start = start;
96              this.max = max;
97              this.step = step;
98  
99              final int s = (max - start) / step + 1;
100             this.size = s < 0 ? 0 : s;
101         }
102 
103         /**
104          * Gets the number of elements contained in the range.
105          *
106          * @return the size of the range.
107          */
108         public int size() {
109             return size;
110         }
111 
112         /** {@inheritDoc} */
113         @Override
114         public Iterator<Integer> iterator() {
115             return Incrementor.create()
116                 .withStart(start)
117                 .withMaximalCount(max + (step > 0 ? 1 : -1))
118                 .withIncrement(step);
119         }
120     }
121 
122     /**
123      * Utility that increments a counter until a maximum is reached, at
124      * which point, the instance will by default throw a
125      * {@link MaxCountExceededException}.
126      * However, the user is able to override this behaviour by defining a
127      * custom {@link MaxCountExceededCallback callback}, in order to e.g.
128      * select which exception must be thrown.
129      */
130     public static class Incrementor implements Iterator<Integer> {
131         /** Default callback. */
132         private static final MaxCountExceededCallback CALLBACK
133             = new MaxCountExceededCallback() {
134                     /** {@inheritDoc} */
135                     @Override
136                     public void trigger(int max) throws MaxCountExceededException {
137                         throw new MaxCountExceededException(max);
138                     }
139                 };
140 
141         /** Initial value the counter. */
142         private final int init;
143         /** Upper limit for the counter. */
144         private final int maximalCount;
145         /** Increment. */
146         private final int increment;
147         /** Function called at counter exhaustion. */
148         private final MaxCountExceededCallback maxCountCallback;
149         /** Current count. */
150         private int count = 0;
151 
152         /**
153          * Defines a method to be called at counter exhaustion.
154          * The {@link #trigger(int) trigger} method should usually throw an exception.
155          */
156         public interface MaxCountExceededCallback {
157             /**
158              * Function called when the maximal count has been reached.
159              *
160              * @param maximalCount Maximal count.
161              * @throws MaxCountExceededException at counter exhaustion
162              */
163             void trigger(int maximalCount) throws MaxCountExceededException;
164         }
165 
166         /**
167          * Creates an incrementor.
168          * The counter will be exhausted either when {@code max} is reached
169          * or when {@code nTimes} increments have been performed.
170          *
171          * @param start Initial value.
172          * @param max Maximal count.
173          * @param step Increment.
174          * @param cb Function to be called when the maximal count has been reached.
175          * @throws NullArgumentException if {@code cb} is {@code null}.
176          */
177         private Incrementor(int start,
178                             int max,
179                             int step,
180                             MaxCountExceededCallback cb)
181             throws NullArgumentException {
182             if (cb == null) {
183                 throw new NullArgumentException();
184             }
185             this.init = start;
186             this.maximalCount = max;
187             this.increment = step;
188             this.maxCountCallback = cb;
189             this.count = start;
190         }
191 
192         /**
193          * Factory method that creates a default instance.
194          * The initial and maximal values are set to 0.
195          * For the new instance to be useful, the maximal count must be set
196          * by calling {@link #withMaximalCount(int) withMaximalCount}.
197          *
198          * @return an new instance.
199          */
200         public static Incrementor create() {
201             return new Incrementor(0, 0, 1, CALLBACK);
202         }
203 
204         /**
205          * Creates a new instance with a given initial value.
206          * The counter is reset to the initial value.
207          *
208          * @param start Initial value of the counter.
209          * @return a new instance.
210          */
211         public Incrementor withStart(int start) {
212             return new Incrementor(start,
213                                    this.maximalCount,
214                                    this.increment,
215                                    this.maxCountCallback);
216         }
217 
218         /**
219          * Creates a new instance with a given maximal count.
220          * The counter is reset to the initial value.
221          *
222          * @param max Maximal count.
223          * @return a new instance.
224          */
225         public Incrementor withMaximalCount(int max) {
226             return new Incrementor(this.init,
227                                    max,
228                                    this.increment,
229                                    this.maxCountCallback);
230         }
231 
232         /**
233          * Creates a new instance with a given increment.
234          * The counter is reset to the initial value.
235          *
236          * @param step Increment.
237          * @return a new instance.
238          */
239         public Incrementor withIncrement(int step) {
240             if (step == 0) {
241                 throw new ZeroException();
242             }
243             return new Incrementor(this.init,
244                                    this.maximalCount,
245                                    step,
246                                    this.maxCountCallback);
247         }
248 
249         /**
250          * Creates a new instance with a given callback.
251          * The counter is reset to the initial value.
252          *
253          * @param cb Callback to be called at counter exhaustion.
254          * @return a new instance.
255          */
256         public Incrementor withCallback(MaxCountExceededCallback cb) {
257             return new Incrementor(this.init,
258                                    this.maximalCount,
259                                    this.increment,
260                                    cb);
261         }
262 
263         /**
264          * Gets the upper limit of the counter.
265          *
266          * @return the counter upper limit.
267          */
268         public int getMaximalCount() {
269             return maximalCount;
270         }
271 
272         /**
273          * Gets the current count.
274          *
275          * @return the current count.
276          */
277         public int getCount() {
278             return count;
279         }
280 
281         /**
282          * Checks whether incrementing the counter {@code nTimes} is allowed.
283          *
284          * @return {@code false} if calling {@link #increment()}
285          * will trigger a {@code MaxCountExceededException},
286          * {@code true} otherwise.
287          */
288         public boolean canIncrement() {
289             return canIncrement(1);
290         }
291 
292         /**
293          * Checks whether incrementing the counter several times is allowed.
294          *
295          * @param nTimes Number of increments.
296          * @return {@code false} if calling {@link #increment(int)
297          * increment(nTimes)} would call the {@link MaxCountExceededCallback callback}
298          * {@code true} otherwise.
299          */
300         public boolean canIncrement(int nTimes) {
301             final int finalCount = count + nTimes * increment;
302             return increment < 0 ?
303                 finalCount > maximalCount :
304                 finalCount < maximalCount;
305         }
306 
307         /**
308          * Performs multiple increments.
309          *
310          * @param nTimes Number of increments.
311          * @throws MaxCountExceededException at counter exhaustion.
312          * @throws NotStrictlyPositiveException if {@code nTimes <= 0}.
313          *
314          * @see #increment()
315          */
316         public void increment(int nTimes) throws MaxCountExceededException {
317             if (nTimes <= 0) {
318                 throw new NotStrictlyPositiveException(nTimes);
319             }
320 
321             if (!canIncrement(0)) {
322                 maxCountCallback.trigger(maximalCount);
323             }
324             count += nTimes * increment;
325         }
326 
327         /**
328          * Adds the increment value to the current iteration count.
329          * At counter exhaustion, this method will call the
330          * {@link MaxCountExceededCallback#trigger(int) trigger} method of the
331          * callback object passed to the
332          * {@link #withCallback(MaxCountExceededCallback)} method.
333          * If not explictly set, a default callback is used that will throw
334          * a {@code MaxCountExceededException}.
335          *
336          * @throws MaxCountExceededException at counter exhaustion, unless a
337          * custom {@link MaxCountExceededCallback callback} has been set.
338          *
339          * @see #increment(int)
340          */
341         public void increment() throws MaxCountExceededException {
342             increment(1);
343         }
344 
345         /** {@inheritDoc} */
346         @Override
347         public boolean hasNext() {
348             return canIncrement(0);
349         }
350 
351         /** {@inheritDoc} */
352         @Override
353         public Integer next() {
354             final int value = count;
355             increment();
356             return value;
357         }
358 
359         /**
360          * Not applicable.
361          *
362          * @throws MathUnsupportedOperationException
363          */
364         @Override
365         public void remove() {
366             throw new MathUnsupportedOperationException();
367         }
368     }
