18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.OutOfRangeException;
22  import org.apache.commons.math4.exception.util.LocalizedFormats;
23  import org.apache.commons.math4.special.Gamma;
24  import org.apache.commons.math4.util.FastMath;
25  
26  /**
27   * Implementation of the Weibull distribution. This implementation uses the
28   * two parameter form of the distribution defined by
29   * <a href="http://mathworld.wolfram.com/WeibullDistribution.html">
30   * Weibull Distribution</a>, equations (1) and (2).
31   *
32   * @see <a href="http://en.wikipedia.org/wiki/Weibull_distribution">Weibull distribution (Wikipedia)</a>
33   * @see <a href="http://mathworld.wolfram.com/WeibullDistribution.html">Weibull distribution (MathWorld)</a>
34   *
35   * @since 1.1
36   */
37  public class WeibullDistribution extends AbstractRealDistribution {
38      /**
39       * Default inverse cumulative probability accuracy.
40       * @since 2.1
41       */
42      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
43      /** Serializable version identifier. */
44      private static final long serialVersionUID = 20160311L;
45      /** The shape parameter. */
46      private final double shape;
47      /** The scale parameter. */
48      private final double scale;
49      /** Inverse cumulative probability accuracy. */
50      private final double solverAbsoluteAccuracy;
51      /** Cached numerical mean */
52      private double numericalMean = Double.NaN;
53      /** Whether or not the numerical mean has been calculated */
54      private boolean numericalMeanIsCalculated = false;
55      /** Cached numerical variance */
56      private double numericalVariance = Double.NaN;
57      /** Whether or not the numerical variance has been calculated */
58      private boolean numericalVarianceIsCalculated = false;
59  
60      /**
61       * Creates a distribution.
62       *
63       * @param alpha Shape parameter.
64       * @param beta Scale parameter.
65       * @throws NotStrictlyPositiveException if {@code alpha <= 0} or
66       * {@code beta <= 0}.
67       */
68      public WeibullDistribution(double alpha, double beta)
69          throws NotStrictlyPositiveException {
70          this(alpha, beta, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
71      }
72  
73      /**
74       * Creates a distribution.
75       *
76       * @param alpha Shape parameter.
77       * @param beta Scale parameter.
78       * @param inverseCumAccuracy Maximum absolute error in inverse
79       * cumulative probability estimates
80       * (defaults to {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
81       * @throws NotStrictlyPositiveException if {@code alpha <= 0} or {@code beta <= 0}.
82       */
83      public WeibullDistribution(double alpha,
84                                 double beta,
85                                 double inverseCumAccuracy)
86          throws NotStrictlyPositiveException {
87          if (alpha <= 0) {
88              throw new NotStrictlyPositiveException(LocalizedFormats.SHAPE,
89                                                     alpha);
90          }
91          if (beta <= 0) {
92              throw new NotStrictlyPositiveException(LocalizedFormats.SCALE,
93                                                     beta);
94          }
95          scale = beta;
96          shape = alpha;
97          solverAbsoluteAccuracy = inverseCumAccuracy;
98      }
99  
100     /**
101      * Access the shape parameter, {@code alpha}.
102      *
103      * @return the shape parameter, {@code alpha}.
104      */
105     public double getShape() {
106         return shape;
107     }
108 
109     /**
110      * Access the scale parameter, {@code beta}.
111      *
112      * @return the scale parameter, {@code beta}.
113      */
114     public double getScale() {
115         return scale;
116     }
117 
118     /** {@inheritDoc} */
119     @Override
120     public double density(double x) {
121         if (x < 0) {
122             return 0;
123         }
124 
125         final double xscale = x / scale;
126         final double xscalepow = FastMath.pow(xscale, shape - 1);
127 
128         /*
129          * FastMath.pow(x / scale, shape) =
130          * FastMath.pow(xscale, shape) =
131          * FastMath.pow(xscale, shape - 1) * xscale
132          */
133         final double xscalepowshape = xscalepow * xscale;
134 
135         return (shape / scale) * xscalepow * FastMath.exp(-xscalepowshape);
136     }
137 
138     /** {@inheritDoc} */
139     @Override
140     public double logDensity(double x) {
141         if (x < 0) {
142             return Double.NEGATIVE_INFINITY;
143         }
144 
145         final double xscale = x / scale;
146         final double logxscalepow = FastMath.log(xscale) * (shape - 1);
147 
148         /*
149          * FastMath.pow(x / scale, shape) =
150          * FastMath.pow(xscale, shape) =
151          * FastMath.pow(xscale, shape - 1) * xscale
152          */
153         final double xscalepowshape = FastMath.exp(logxscalepow) * xscale;
154 
155         return FastMath.log(shape / scale) + logxscalepow - xscalepowshape;
156     }
157 
158     /** {@inheritDoc} */
159     @Override
160     public double cumulativeProbability(double x) {
161         double ret;
162         if (x <= 0.0) {
163             ret = 0.0;
164         } else {
165             ret = 1.0 - FastMath.exp(-FastMath.pow(x / scale, shape));
166         }
167         return ret;
168     }
169 
170     /**
171      * {@inheritDoc}
172      *
173      * Returns {@code 0} when {@code p == 0} and
174      * {@code Double.POSITIVE_INFINITY} when {@code p == 1}.
175      */
176     @Override
177     public double inverseCumulativeProbability(double p) {
178         double ret;
179         if (p < 0.0 || p > 1.0) {
180             throw new OutOfRangeException(p, 0.0, 1.0);
181         } else if (p == 0) {
182             ret = 0.0;
183         } else  if (p == 1) {
184             ret = Double.POSITIVE_INFINITY;
185         } else {
186             ret = scale * FastMath.pow(-FastMath.log1p(-p), 1.0 / shape);
187         }
188         return ret;
189     }
190 
191     /**
192      * Return the absolute accuracy setting of the solver used to estimate
193      * inverse cumulative probabilities.
194      *
195      * @return the solver absolute accuracy.
196      * @since 2.1
197      */
198     @Override
199     protected double getSolverAbsoluteAccuracy() {
200         return solverAbsoluteAccuracy;
201     }
202 
203     /**
204      * {@inheritDoc}
205      *
206      * The mean is {@code scale * Gamma(1 + (1 / shape))}, where {@code Gamma()}
207      * is the Gamma-function.
208      */
209     @Override
210     public double getNumericalMean() {
211         if (!numericalMeanIsCalculated) {
212             numericalMean = calculateNumericalMean();
213             numericalMeanIsCalculated = true;
214         }
215         return numericalMean;
216     }
217 
218     /**
219      * used by {@link #getNumericalMean()}
220      *
221      * @return the mean of this distribution
222      */
223     protected double calculateNumericalMean() {
224         final double sh = getShape();
225         final double sc = getScale();
226 
227         return sc * FastMath.exp(Gamma.logGamma(1 + (1 / sh)));
228     }
229 
230     /**
231      * {@inheritDoc}
232      *
233      * The variance is {@code scale^2 * Gamma(1 + (2 / shape)) - mean^2}
234      * where {@code Gamma()} is the Gamma-function.
235      */
236     @Override
237     public double getNumericalVariance() {
238         if (!numericalVarianceIsCalculated) {
239             numericalVariance = calculateNumericalVariance();
240             numericalVarianceIsCalculated = true;
241         }
242         return numericalVariance;
243     }
244 
245     /**
246      * used by {@link #getNumericalVariance()}
247      *
248      * @return the variance of this distribution
249      */
250     protected double calculateNumericalVariance() {
251         final double sh = getShape();
252         final double sc = getScale();
253         final double mn = getNumericalMean();
254 
255         return (sc * sc) * FastMath.exp(Gamma.logGamma(1 + (2 / sh))) -
256                (mn * mn);
257     }
258 
259     /**
260      * {@inheritDoc}
261      *
262      * The lower bound of the support is always 0 no matter the parameters.
263      *
264      * @return lower bound of the support (always 0)
265      */
266     @Override
267     public double getSupportLowerBound() {
268         return 0;
269     }
270 
271     /**
272      * {@inheritDoc}
273      *
274      * The upper bound of the support is always positive infinity
275      * no matter the parameters.
276      *
277      * @return upper bound of the support (always
278      * {@code Double.POSITIVE_INFINITY})
279      */
280     @Override
281     public double getSupportUpperBound() {
282         return Double.POSITIVE_INFINITY;
283     }
284 
285     /**
286      * {@inheritDoc}
287      *
288      * The support of this distribution is connected.
289      *
290      * @return {@code true}
291      */
292     @Override
293     public boolean isSupportConnected() {
294         return true;
295     }
296 }
