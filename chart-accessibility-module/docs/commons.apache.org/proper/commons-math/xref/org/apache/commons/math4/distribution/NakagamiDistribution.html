17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.exception.NumberIsTooSmallException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.math4.special.Gamma;
23  import org.apache.commons.math4.util.FastMath;
24  
25  /**
26   * This class implements the Nakagami distribution.
27   *
28   * @see <a href="http://en.wikipedia.org/wiki/Nakagami_distribution">Nakagami Distribution (Wikipedia)</a>
29   *
30   * @since 3.4
31   */
32  public class NakagamiDistribution extends AbstractRealDistribution {
33  
34      /** Default inverse cumulative probability accuracy. */
35      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
36  
37      /** Serializable version identifier. */
38      private static final long serialVersionUID = 20160311L;
39  
40      /** The shape parameter. */
41      private final double mu;
42      /** The scale parameter. */
43      private final double omega;
44      /** Inverse cumulative probability accuracy. */
45      private final double inverseAbsoluteAccuracy;
46  
47      /**
48       * Creates a distribution.
49       *
50       * @param mu shape parameter
51       * @param omega scale parameter (must be positive)
52       * @throws NumberIsTooSmallException if {@code mu < 0.5}
53       * @throws NotStrictlyPositiveException if {@code omega <= 0}
54       */
55      public NakagamiDistribution(double mu,
56                                  double omega) {
57          this(mu, omega, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
58      }
59  
60      /**
61       * Creates a distribution.
62       *
63       * @param mu shape parameter
64       * @param omega scale parameter (must be positive)
65       * @param inverseAbsoluteAccuracy the maximum absolute error in inverse
66       * cumulative probability estimates (defaults to {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
67       * @throws NumberIsTooSmallException if {@code mu < 0.5}
68       * @throws NotStrictlyPositiveException if {@code omega <= 0}
69       */
70      public NakagamiDistribution(double mu,
71                                  double omega,
72                                  double inverseAbsoluteAccuracy) {
73          if (mu < 0.5) {
74              throw new NumberIsTooSmallException(mu, 0.5, true);
75          }
76          if (omega <= 0) {
77              throw new NotStrictlyPositiveException(LocalizedFormats.NOT_POSITIVE_SCALE, omega);
78          }
79  
80          this.mu = mu;
81          this.omega = omega;
82          this.inverseAbsoluteAccuracy = inverseAbsoluteAccuracy;
83      }
84  
85      /**
86       * Access the shape parameter, {@code mu}.
87       *
88       * @return the shape parameter.
89       */
90      public double getShape() {
91          return mu;
92      }
93  
94      /**
95       * Access the scale parameter, {@code omega}.
96       *
97       * @return the scale parameter.
98       */
99      public double getScale() {
100         return omega;
101     }
102 
103     /** {@inheritDoc} */
104     @Override
105     protected double getSolverAbsoluteAccuracy() {
106         return inverseAbsoluteAccuracy;
107     }
108 
109     /** {@inheritDoc} */
110     @Override
111     public double density(double x) {
112         if (x <= 0) {
113             return 0.0;
114         }
115         return 2.0 * FastMath.pow(mu, mu) / (Gamma.gamma(mu) * FastMath.pow(omega, mu)) *
116                      FastMath.pow(x, 2 * mu - 1) * FastMath.exp(-mu * x * x / omega);
117     }
118 
119     /** {@inheritDoc} */
120     @Override
121     public double cumulativeProbability(double x) {
122         return Gamma.regularizedGammaP(mu, mu * x * x / omega);
123     }
124 
125     /** {@inheritDoc} */
126     @Override
127     public double getNumericalMean() {
128         return Gamma.gamma(mu + 0.5) / Gamma.gamma(mu) * FastMath.sqrt(omega / mu);
129     }
130 
131     /** {@inheritDoc} */
132     @Override
133     public double getNumericalVariance() {
134         double v = Gamma.gamma(mu + 0.5) / Gamma.gamma(mu);
135         return omega * (1 - 1 / mu * v * v);
136     }
137 
138     /** {@inheritDoc} */
139     @Override
140     public double getSupportLowerBound() {
141         return 0;
142     }
143 
144     /** {@inheritDoc} */
145     @Override
146     public double getSupportUpperBound() {
147         return Double.POSITIVE_INFINITY;
148     }
149 
150     /** {@inheritDoc} */
151     @Override
152     public boolean isSupportConnected() {
153         return true;
154     }
155 
156 }
