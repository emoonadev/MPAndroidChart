17  package org.apache.commons.math4.transform;
18  
19  /**
20   * This enumeration defines the various types of normalizations that can be
21   * applied to discrete Fourier transforms (DFT). The exact definition of these
22   * normalizations is detailed below.
23   *
24   * @see FastFourierTransformer
25   * @since 3.0
26   */
27  public enum DftNormalization {
28      /**
29       * Should be passed to the constructor of {@link FastFourierTransformer}
30       * to use the <em>standard</em> normalization convention. This normalization
31       * convention is defined as follows
32       * <ul>
33       * <li>forward transform: y<sub>n</sub> = &sum;<sub>k=0</sub><sup>N-1</sup>
34       * x<sub>k</sub> exp(-2&pi;i n k / N),</li>
35       * <li>inverse transform: x<sub>k</sub> = N<sup>-1</sup>
36       * &sum;<sub>n=0</sub><sup>N-1</sup> y<sub>n</sub> exp(2&pi;i n k / N),</li>
37       * </ul>
38       * where N is the size of the data sample.
39       */
40      STANDARD,
41  
42      /**
43       * Should be passed to the constructor of {@link FastFourierTransformer}
44       * to use the <em>unitary</em> normalization convention. This normalization
45       * convention is defined as follows
46       * <ul>
47       * <li>forward transform: y<sub>n</sub> = (1 / &radic;N)
48       * &sum;<sub>k=0</sub><sup>N-1</sup> x<sub>k</sub>
49       * exp(-2&pi;i n k / N),</li>
50       * <li>inverse transform: x<sub>k</sub> = (1 / &radic;N)
51       * &sum;<sub>n=0</sub><sup>N-1</sup> y<sub>n</sub> exp(2&pi;i n k / N),</li>
52       * </ul>
53       * which makes the transform unitary. N is the size of the data sample.
54       */
55      UNITARY;
56  }
