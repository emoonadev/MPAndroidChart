17  package org.apache.commons.math4.analysis.interpolation;
18  
19  import org.apache.commons.math4.analysis.BivariateFunction;
20  import org.apache.commons.math4.exception.DimensionMismatchException;
21  import org.apache.commons.math4.exception.NoDataException;
22  import org.apache.commons.math4.exception.NonMonotonicSequenceException;
23  import org.apache.commons.math4.exception.NumberIsTooSmallException;
24  
25  /**
26   * Interface representing a bivariate real interpolating function where the
27   * sample points must be specified on a regular grid.
28   *
29   */
30  public interface BivariateGridInterpolator {
31      /**
32       * Compute an interpolating function for the dataset.
33       *
34       * @param xval All the x-coordinates of the interpolation points, sorted
35       * in increasing order.
36       * @param yval All the y-coordinates of the interpolation points, sorted
37       * in increasing order.
38       * @param fval The values of the interpolation points on all the grid knots:
39       * {@code fval[i][j] = f(xval[i], yval[j])}.
40       * @return a function which interpolates the dataset.
41       * @throws NoDataException if any of the arrays has zero length.
42       * @throws DimensionMismatchException if the array lengths are inconsistent.
43       * @throws NonMonotonicSequenceException if the array is not sorted.
44       * @throws NumberIsTooSmallException if the number of points is too small for
45       * the order of the interpolation
46       */
47      BivariateFunction interpolate(double[] xval, double[] yval,
48                                    double[][] fval)
49          throws NoDataException, DimensionMismatchException,
50                 NonMonotonicSequenceException, NumberIsTooSmallException;
51  }
