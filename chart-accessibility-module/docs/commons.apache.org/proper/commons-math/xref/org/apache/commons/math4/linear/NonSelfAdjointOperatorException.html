18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.exception.MathIllegalArgumentException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  
23  /**
24   * Exception to be thrown when a self-adjoint {@link RealLinearOperator}
25   * is expected.
26   * Since the coefficients of the matrix are not accessible, the most
27   * general definition is used to check that A is not self-adjoint, i.e.
28   * there exist x and y such as {@code | x' A y - y' A x | >= eps},
29   * where {@code eps} is a user-specified tolerance, and {@code x'}
30   * denotes the transpose of {@code x}.
31   * In the terminology of this exception, {@code A} is the "offending"
32   * linear operator, {@code x} and {@code y} are the first and second
33   * "offending" vectors, respectively.
34   *
35   * @since 3.0
36   */
37  public class NonSelfAdjointOperatorException
38      extends MathIllegalArgumentException {
39      /** Serializable version Id. */
40      private static final long serialVersionUID = 1784999305030258247L;
41  
42      /** Creates a new instance of this class. */
43      public NonSelfAdjointOperatorException() {
44          super(LocalizedFormats.NON_SELF_ADJOINT_OPERATOR);
45      }
46  }
