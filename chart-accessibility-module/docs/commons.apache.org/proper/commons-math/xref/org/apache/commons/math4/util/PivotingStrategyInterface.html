17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  
21  
22  /**
23   * A strategy to pick a pivoting index of an array for doing partitioning.
24   * @see MedianOf3PivotingStrategy
25   * @see RandomPivotingStrategy
26   * @see CentralPivotingStrategy
27   * @since 3.4
28   */
29  public interface PivotingStrategyInterface {
30  
31      /**
32       * Find pivot index of the array so that partition and K<sup>th</sup>
33       * element selection can be made
34       * @param work data array
35       * @param begin index of the first element of the slice
36       * @param end index after the last element of the slice
37       * @return the index of the pivot element chosen between the
38       * first and the last element of the array slice
39       * @throws MathIllegalArgumentException when indices exceeds range
40       */
41      int pivotIndex(double[] work, int begin, int end)
42          throws MathIllegalArgumentException;
43  
44  }
