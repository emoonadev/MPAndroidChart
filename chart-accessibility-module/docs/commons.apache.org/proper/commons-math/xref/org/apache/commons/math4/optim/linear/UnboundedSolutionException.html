19  import org.apache.commons.math4.exception.MathIllegalStateException;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;

22  /**
23   * This class represents exceptions thrown by optimizers when a solution escapes to infinity.
24   *
25   * @since 2.0
26   */
27  public class UnboundedSolutionException extends MathIllegalStateException {
28      /** Serializable version identifier. */
29      private static final long serialVersionUID = 940539497277290619L;
30  
31      /**
32       * Simple constructor using a default message.
33       */
34      public UnboundedSolutionException() {
35          super(LocalizedFormats.UNBOUNDED_SOLUTION);
36      }
37  }
