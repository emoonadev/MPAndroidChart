18  package org.apache.commons.math4.linear;
19  
20  /**
21   * Interface defining a visitor for matrix entries.
22   *
23   * @see DefaultRealMatrixPreservingVisitor
24   * @since 2.0
25   */
26  public interface RealMatrixPreservingVisitor {
27      /**
28       * Start visiting a matrix.
29       * <p>This method is called once before any entry of the matrix is visited.</p>
30       * @param rows number of rows of the matrix
31       * @param columns number of columns of the matrix
32       * @param startRow Initial row index
33       * @param endRow Final row index (inclusive)
34       * @param startColumn Initial column index
35       * @param endColumn Final column index (inclusive)
36       */
37      void start(int rows, int columns,
38                 int startRow, int endRow, int startColumn, int endColumn);
39  
40      /**
41       * Visit one matrix entry.
42       * @param row row index of the entry
43       * @param column column index of the entry
44       * @param value current value of the entry
45       */
46      void visit(int row, int column, double value);
47  
48      /**
49       * End visiting a matrix.
50       * <p>This method is called once after all entries of the matrix have been visited.</p>
51       * @return the value that the <code>walkInXxxOrder</code> must return
52       */
53      double end();
54  }
