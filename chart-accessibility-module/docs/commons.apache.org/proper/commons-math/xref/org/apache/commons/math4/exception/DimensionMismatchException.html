17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when two dimensions differ.
24   *
25   * @since 2.2
26   */
27  public class DimensionMismatchException extends MathIllegalNumberException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -8415396756375798143L;
30      /** Correct dimension. */
31      private final int dimension;
32  
33      /**
34       * Construct an exception from the mismatched dimensions.
35       *
36       * @param specific Specific context information pattern.
37       * @param wrong Wrong dimension.
38       * @param expected Expected dimension.
39       */
40      public DimensionMismatchException(Localizable specific,
41                                        int wrong,
42                                        int expected) {
43          super(specific, Integer.valueOf(wrong), Integer.valueOf(expected));
44          dimension = expected;
45      }
46  
47      /**
48       * Construct an exception from the mismatched dimensions.
49       *
50       * @param wrong Wrong dimension.
51       * @param expected Expected dimension.
52       */
53      public DimensionMismatchException(int wrong,
54                                        int expected) {
55          this(LocalizedFormats.DIMENSIONS_MISMATCH_SIMPLE, wrong, expected);
56      }
57  
58      /**
59       * @return the expected dimension.
60       */
61      public int getDimension() {
62          return dimension;
63      }
64  }
