18  package org.apache.commons.math4.analysis;
19  
20  /**
21   * An interface representing a multivariate real function.
22   *
23   * @since 2.0
24   */
25  public interface MultivariateFunction {
26  
27      /**
28       * Compute the value for the function at the given point.
29       *
30       * @param point Point at which the function must be evaluated.
31       * @return the function value for the given point.
32       * @throws org.apache.commons.math4.exception.DimensionMismatchException
33       * if the parameter's dimension is wrong for the function being evaluated.
34       * @throws  org.apache.commons.math4.exception.MathIllegalArgumentException
35       * when the activated method itself can ascertain that preconditions,
36       * specified in the API expressed at the level of the activated method,
37       * have been violated.  In the vast majority of cases where Commons Math
38       * throws this exception, it is the result of argument checking of actual
39       * parameters immediately passed to a method.
40       */
41      double value(double[] point);
42  }
