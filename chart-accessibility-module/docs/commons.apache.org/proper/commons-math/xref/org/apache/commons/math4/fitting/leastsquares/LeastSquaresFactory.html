47      /**
48       * Create a {@link org.apache.commons.math4.fitting.leastsquares.LeastSquaresProblem}
49       * from the given elements. There will be no weights applied (unit weights).
50       *
51       * @param model          the model function. Produces the computed values.
52       * @param observed       the observed (target) values
53       * @param start          the initial guess.
54       * @param weight         the weight matrix
55       * @param checker        convergence checker
56       * @param maxEvaluations the maximum number of times to evaluate the model
57       * @param maxIterations  the maximum number to times to iterate in the algorithm
58       * @param lazyEvaluation Whether the call to {@link Evaluation#evaluate(RealVector)}
59       * will defer the evaluation until access to the value is requested.
60       * @param paramValidator Model parameters validator.
61       * @return the specified General Least Squares problem.
62       *
63       * @since 3.4
64       */
65      public static LeastSquaresProblem create(final MultivariateJacobianFunction model,
66                                               final RealVector observed,
67                                               final RealVector start,
68                                               final RealMatrix weight,
69                                               final ConvergenceChecker<Evaluation> checker,
70                                               final int maxEvaluations,
71                                               final int maxIterations,
72                                               final boolean lazyEvaluation,
73                                               final ParameterValidator paramValidator) {
74          final LeastSquaresProblem p = new LocalLeastSquaresProblem(model,
75                                                                     observed,
76                                                                     start,
77                                                                     checker,
78                                                                     maxEvaluations,
79                                                                     maxIterations,
80                                                                     lazyEvaluation,
81                                                                     paramValidator);
82          if (weight != null) {
83              return weightMatrix(p, weight);
84          } else {
85              return p;
86          }
87      }

148      * Create a {@link org.apache.commons.math4.fitting.leastsquares.LeastSquaresProblem}
149      * from the given elements.
150      * <p>
151      * This factory method is provided for continuity with previous interfaces. Newer
152      * applications should use {@link #create(MultivariateJacobianFunction, RealVector,

154      * RealVector, RealVector, RealMatrix, ConvergenceChecker, int, int)}.
155      *
156      * @param model          the model function. Produces the computed values.
157      * @param jacobian       the jacobian of the model with respect to the parameters
158      * @param observed       the observed (target) values
159      * @param start          the initial guess.
160      * @param weight         the weight matrix
161      * @param checker        convergence checker
162      * @param maxEvaluations the maximum number of times to evaluate the model
163      * @param maxIterations  the maximum number to times to iterate in the algorithm
164      * @return the specified General Least Squares problem.

166     public static LeastSquaresProblem create(final MultivariateVectorFunction model,
167                                              final MultivariateMatrixFunction jacobian,
168                                              final double[] observed,
169                                              final double[] start,
170                                              final RealMatrix weight,
171                                              final ConvergenceChecker<Evaluation> checker,
172                                              final int maxEvaluations,
173                                              final int maxIterations) {

176                       new ArrayRealVector(start, false),
177                       weight,
178                       checker,

180                       maxIterations);

183     /**
184      * Apply a dense weight matrix to the {@link LeastSquaresProblem}.
185      *
186      * @param problem the unweighted problem
187      * @param weights the matrix of weights
188      * @return a new {@link LeastSquaresProblem} with the weights applied. The original
189      *         {@code problem} is not modified.
190      */
191     public static LeastSquaresProblem weightMatrix(final LeastSquaresProblem problem,
192                                                    final RealMatrix weights) {
193         final RealMatrix weightSquareRoot = squareRoot(weights);
194         return new LeastSquaresAdapter(problem) {
195             /** {@inheritDoc} */
196             @Override
197             public Evaluation evaluate(final RealVector point) {
198                 return new DenseWeightedEvaluation(super.evaluate(point), weightSquareRoot);
199             }
200         };
201     }

204      * Apply a diagonal weight matrix to the {@link LeastSquaresProblem}.
205      *
206      * @param problem the unweighted problem
207      * @param weights the diagonal of the weight matrix
208      * @return a new {@link LeastSquaresProblem} with the weights applied. The original
209      *         {@code problem} is not modified.

211     public static LeastSquaresProblem weightDiagonal(final LeastSquaresProblem problem,

213         // TODO more efficient implementation
214         return weightMatrix(problem, new DiagonalMatrix(weights.toArray()));

217     /**
218      * Count the evaluations of a particular problem. The {@code counter} will be
219      * incremented every time {@link LeastSquaresProblem#evaluate(RealVector)} is called on
220      * the <em>returned</em> problem.
221      *
222      * @param problem the problem to track.
223      * @param counter the counter to increment.
224      * @return a least squares problem that tracks evaluations
225      */
226     public static LeastSquaresProblem countEvaluations(final LeastSquaresProblem problem,
227                                                        final Incrementor counter) {
228         return new LeastSquaresAdapter(problem) {
229 
230             /** {@inheritDoc} */
231             @Override
232             public Evaluation evaluate(final RealVector point) {
233                 counter.incrementCount();
234                 return super.evaluate(point);
235             }
236 
237             // Delegate the rest.
238         };
239     }

241     /**
242      * View a convergence checker specified for a {@link PointVectorValuePair} as one
243      * specified for an {@link Evaluation}.
244      *
245      * @param checker the convergence checker to adapt.
246      * @return a convergence checker that delegates to {@code checker}.
247      */
248     public static ConvergenceChecker<Evaluation> evaluationChecker(final ConvergenceChecker<PointVectorValuePair> checker) {
249         return new ConvergenceChecker<Evaluation>() {
250             /** {@inheritDoc} */
251             @Override
252             public boolean converged(final int iteration,
253                                      final Evaluation previous,
254                                      final Evaluation current) {
255                 return checker.converged(
256                         iteration,
257                         new PointVectorValuePair(
258                                 previous.getPoint().toArray(),
259                                 previous.getResiduals().toArray(),
260                                 false),
261                         new PointVectorValuePair(
262                                 current.getPoint().toArray(),
263                                 current.getResiduals().toArray(),
264                                 false)
265                 );
266             }
267         };
268     }

270     /**
271      * Computes the square-root of the weight matrix.
272      *
273      * @param m Symmetric, positive-definite (weight) matrix.
274      * @return the square-root of the weight matrix.
275      */
276     private static RealMatrix squareRoot(final RealMatrix m) {
277         if (m instanceof DiagonalMatrix) {
278             final int dim = m.getRowDimension();
279             final RealMatrix sqrtM = new DiagonalMatrix(dim);
280             for (int i = 0; i < dim; i++) {
281                 sqrtM.setEntry(i, i, FastMath.sqrt(m.getEntry(i, i)));
282             }
283             return sqrtM;
284         } else {
285             final EigenDecomposition dec = new EigenDecomposition(m);
286             return dec.getSquareRoot();
287         }
288     }

303     /**
304      * Combine a {@link MultivariateVectorFunction} with a {@link
305      * MultivariateMatrixFunction} to produce a {@link MultivariateJacobianFunction}.
306      *
307      * @param value    the vector value function
308      * @param jacobian the Jacobian function
309      * @return a function that computes both at the same time
310      */
311     private static class LocalValueAndJacobianFunction
312         implements ValueAndJacobianFunction {
313         /** Model. */
314         private final MultivariateVectorFunction value;
315         /** Model's Jacobian. */
316         private final MultivariateMatrixFunction jacobian;
317 
318         /**
319          * @param value Model function.
320          * @param jacobian Model's Jacobian function.
321          */
322         LocalValueAndJacobianFunction(final MultivariateVectorFunction value,
323                                       final MultivariateMatrixFunction jacobian) {
324             this.value = value;
325             this.jacobian = jacobian;
326         }
327 
328         /** {@inheritDoc} */
329         @Override
330         public Pair<RealVector, RealMatrix> value(final RealVector point) {
331             //TODO get array from RealVector without copying?
332             final double[] p = point.toArray();
333 
334             // Evaluate.
335             return new Pair<RealVector, RealMatrix>(computeValue(p),
336                                                     computeJacobian(p));
337         }
338 
339         /** {@inheritDoc} */
340         @Override
341         public RealVector computeValue(final double[] params) {
342             return new ArrayRealVector(value.value(params), false);
343         }
344 
345         /** {@inheritDoc} */
346         @Override
347         public RealMatrix computeJacobian(final double[] params) {
348             return new Array2DRowRealMatrix(jacobian.value(params), false);
349         }
350     }

353     /**
354      * A private, "field" immutable (not "real" immutable) implementation of {@link
355      * LeastSquaresProblem}.
356      * @since 3.3
357      */
358     private static class LocalLeastSquaresProblem
359             extends AbstractOptimizationProblem<Evaluation>
360             implements LeastSquaresProblem {
361 
362         /** Target values for the model function at optimum. */
363         private final RealVector target;
364         /** Model function. */
365         private final MultivariateJacobianFunction model;
366         /** Initial guess. */
367         private final RealVector start;
368         /** Whether to use lazy evaluation. */
369         private final boolean lazyEvaluation;
370         /** Model parameters validator. */
371         private final ParameterValidator paramValidator;
372 
373         /**
374          * Create a {@link LeastSquaresProblem} from the given data.
375          *
376          * @param model          the model function
377          * @param target         the observed data
378          * @param start          the initial guess
379          * @param checker        the convergence checker
380          * @param maxEvaluations the allowed evaluations
381          * @param maxIterations  the allowed iterations
382          * @param lazyEvaluation Whether the call to {@link Evaluation#evaluate(RealVector)}
383          * will defer the evaluation until access to the value is requested.
384          * @param paramValidator Model parameters validator.
385          */
386         LocalLeastSquaresProblem(final MultivariateJacobianFunction model,
387                                  final RealVector target,
388                                  final RealVector start,
389                                  final ConvergenceChecker<Evaluation> checker,
390                                  final int maxEvaluations,
391                                  final int maxIterations,
392                                  final boolean lazyEvaluation,
393                                  final ParameterValidator paramValidator) {
394             super(maxEvaluations, maxIterations, checker);
395             this.target = target;
396             this.model = model;
397             this.start = start;
398             this.lazyEvaluation = lazyEvaluation;
399             this.paramValidator = paramValidator;
400 
401             if (lazyEvaluation &&
402                 !(model instanceof ValueAndJacobianFunction)) {
403                 // Lazy evaluation requires that value and Jacobian
404                 // can be computed separately.
405                 throw new MathIllegalStateException(LocalizedFormats.INVALID_IMPLEMENTATION,
406                                                     model.getClass().getName());
407             }
408         }
409 
410         /** {@inheritDoc} */
411         @Override
412         public int getObservationSize() {
413             return target.getDimension();
414         }
415 
416         /** {@inheritDoc} */
417         @Override
418         public int getParameterSize() {
419             return start.getDimension();
420         }
421 
422         /** {@inheritDoc} */
423         @Override
424         public RealVector getStart() {
425             return start == null ? null : start.copy();
426         }
427 
428         /** {@inheritDoc} */
429         @Override
430         public Evaluation evaluate(final RealVector point) {
431             // Copy so optimizer can change point without changing our instance.
432             final RealVector p = paramValidator == null ?
433                 point.copy() :
434                 paramValidator.validate(point.copy());
435 
436             if (lazyEvaluation) {
437                 return new LazyUnweightedEvaluation((ValueAndJacobianFunction) model,
438                                                     target,
439                                                     p);
440             } else {
441                 // Evaluate value and jacobian in one function call.
442                 final Pair<RealVector, RealMatrix> value = model.value(p);
443                 return new UnweightedEvaluation(value.getFirst(),
444                                                 value.getSecond(),
445                                                 target,
446                                                 p);
447             }
448         }

450         /**
451          * Container with the model evaluation at a particular point.
452          */
453         private static class UnweightedEvaluation extends AbstractEvaluation {
454             /** Point of evaluation. */
455             private final RealVector point;
456             /** Derivative at point. */
457             private final RealMatrix jacobian;
458             /** Computed residuals. */
459             private final RealVector residuals;
460 
461             /**
462              * Create an {@link Evaluation} with no weights.
463              *
464              * @param values   the computed function values
465              * @param jacobian the computed function Jacobian
466              * @param target   the observed values
467              * @param point    the abscissa
468              */
469             private UnweightedEvaluation(final RealVector values,
470                                          final RealMatrix jacobian,
471                                          final RealVector target,
472                                          final RealVector point) {
473                 super(target.getDimension());
474                 this.jacobian = jacobian;
475                 this.point = point;
476                 this.residuals = target.subtract(values);
477             }
478 
479             /** {@inheritDoc} */
480             @Override
481             public RealMatrix getJacobian() {
482                 return jacobian;
483             }
484 
485             /** {@inheritDoc} */
486             @Override
487             public RealVector getPoint() {
488                 return point;
489             }
490 
491             /** {@inheritDoc} */
492             @Override
493             public RealVector getResiduals() {
494                 return residuals;
495             }
496         }
497 
498         /**
499          * Container with the model <em>lazy</em> evaluation at a particular point.
500          */
501         private static class LazyUnweightedEvaluation extends AbstractEvaluation {
502             /** Point of evaluation. */
503             private final RealVector point;
504             /** Model and Jacobian functions. */
505             private final ValueAndJacobianFunction model;
506             /** Target values for the model function at optimum. */
507             private final RealVector target;
508 
509             /**
510              * Create an {@link Evaluation} with no weights.
511              *
512              * @param model  the model function
513              * @param target the observed values
514              * @param point  the abscissa
515              */
516             private LazyUnweightedEvaluation(final ValueAndJacobianFunction model,
517                                              final RealVector target,
518                                              final RealVector point) {
519                 super(target.getDimension());
520                 // Safe to cast as long as we control usage of this class.
521                 this.model = model;
522                 this.point = point;
523                 this.target = target;
524             }
525 
526             /** {@inheritDoc} */
527             @Override
528             public RealMatrix getJacobian() {
529                 return model.computeJacobian(point.toArray());
530             }
531 
532             /** {@inheritDoc} */
533             @Override
534             public RealVector getPoint() {
535                 return point;
536             }
537 
538             /** {@inheritDoc} */
539             @Override
540             public RealVector getResiduals() {
541                 return target.subtract(model.computeValue(point.toArray()));
542             }
543         }
