18  package org.apache.commons.math4.optim.univariate;
19  
20  import java.io.Serializable;
21  
22  /**
23   * This class holds a point and the value of an objective function at this
24   * point.
25   * This is a simple immutable container.
26   *
27   * @since 3.0
28   */
29  public class UnivariatePointValuePair implements Serializable {
30      /** Serializable version identifier. */
31      private static final long serialVersionUID = 1003888396256744753L;
32      /** Point. */
33      private final double point;
34      /** Value of the objective function at the point. */
35      private final double value;
36  
37      /**
38       * Build a point/objective function value pair.
39       *
40       * @param point Point.
41       * @param value Value of an objective function at the point
42       */
43      public UnivariatePointValuePair(final double point,
44                                      final double value) {
45          this.point = point;
46          this.value = value;
47      }
48  
49      /**
50       * Get the point.
51       *
52       * @return the point.
53       */
54      public double getPoint() {
55          return point;
56      }
57  
58      /**
59       * Get the value of the objective function.
60       *
61       * @return the stored value of the objective function.
62       */
63      public double getValue() {
64          return value;
65      }
66  }
