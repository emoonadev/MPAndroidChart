17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when some counter maximum value is exceeded.
24   *
25   * @since 3.0
26   */
27  public class MaxCountExceededException extends MathIllegalStateException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = 4330003017885151975L;
30      /**
31       * Maximum number of evaluations.
32       */
33      private final Number max;
34  
35      /**
36       * Construct the exception.
37       *
38       * @param max Maximum.
39       */
40      public MaxCountExceededException(Number max) {
41          this(LocalizedFormats.MAX_COUNT_EXCEEDED, max);
42      }
43      /**
44       * Construct the exception with a specific context.
45       *
46       * @param specific Specific context pattern.
47       * @param max Maximum.
48       * @param args Additional arguments.
49       */
50      public MaxCountExceededException(Localizable specific,
51                                       Number max,
52                                       Object ... args) {
53          getContext().addMessage(specific, max, args);
54          this.max = max;
55      }
56  
57      /**
58       * @return the maximum number of evaluations.
59       */
60      public Number getMax() {
61          return max;
62      }
63  }
