17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import java.io.Serializable;
20  import java.util.Arrays;
21  
22  import org.apache.commons.math4.exception.DimensionMismatchException;
23  import org.apache.commons.math4.linear.MatrixUtils;
24  import org.apache.commons.math4.linear.RealMatrix;
25  
26  /**
27   * Returns the covariance matrix of the available vectors.
28   * @since 1.2
29   */
30  public class VectorialCovariance implements Serializable {
31  
32      /** Serializable version identifier */
33      private static final long serialVersionUID = 4118372414238930270L;
34  
35      /** Sums for each component. */
36      private final double[] sums;
37  
38      /** Sums of products for each component. */
39      private final double[] productsSums;
40  
41      /** Indicator for bias correction. */
42      private final boolean isBiasCorrected;
43  
44      /** Number of vectors in the sample. */
45      private long n;
46  
47      /** Constructs a VectorialCovariance.
48       * @param dimension vectors dimension
49       * @param isBiasCorrected if true, computed the unbiased sample covariance,
50       * otherwise computes the biased population covariance
51       */
52      public VectorialCovariance(int dimension, boolean isBiasCorrected) {
53          sums         = new double[dimension];
54          productsSums = new double[dimension * (dimension + 1) / 2];
55          n            = 0;
56          this.isBiasCorrected = isBiasCorrected;
57      }
58  
59      /**
60       * Add a new vector to the sample.
61       * @param v vector to add
62       * @throws DimensionMismatchException if the vector does not have the right dimension
63       */
64      public void increment(double[] v) throws DimensionMismatchException {
65          if (v.length != sums.length) {
66              throw new DimensionMismatchException(v.length, sums.length);
67          }
68          int k = 0;
69          for (int i = 0; i < v.length; ++i) {
70              sums[i] += v[i];
71              for (int j = 0; j <= i; ++j) {
72                  productsSums[k++] += v[i] * v[j];
73              }
74          }
75          n++;
76      }
77  
78      /**
79       * Get the covariance matrix.
80       * @return covariance matrix
81       */
82      public RealMatrix getResult() {
83  
84          int dimension = sums.length;
85          RealMatrix result = MatrixUtils.createRealMatrix(dimension, dimension);
86  
87          if (n > 1) {
88              double c = 1.0 / (n * (isBiasCorrected ? (n - 1) : n));
89              int k = 0;
90              for (int i = 0; i < dimension; ++i) {
91                  for (int j = 0; j <= i; ++j) {
92                      double e = c * (n * productsSums[k++] - sums[i] * sums[j]);
93                      result.setEntry(i, j, e);
94                      result.setEntry(j, i, e);
95                  }
96              }
97          }
98  
99          return result;
100 
101     }
102 
103     /**
104      * Get the number of vectors in the sample.
105      * @return number of vectors in the sample
106      */
107     public long getN() {
108         return n;
109     }
110 
111     /**
112      * Clears the internal state of the Statistic
113      */
114     public void clear() {
115         n = 0;
116         Arrays.fill(sums, 0.0);
117         Arrays.fill(productsSums, 0.0);
118     }
119 
120     /** {@inheritDoc} */
121     @Override
122     public int hashCode() {
123         final int prime = 31;
124         int result = 1;
125         result = prime * result + (isBiasCorrected ? 1231 : 1237);
126         result = prime * result + (int) (n ^ (n >>> 32));
127         result = prime * result + Arrays.hashCode(productsSums);
128         result = prime * result + Arrays.hashCode(sums);
129         return result;
130     }
131 
132     /** {@inheritDoc} */
133     @Override
134     public boolean equals(Object obj) {
135         if (this == obj) {
136             return true;
137         }
138         if (!(obj instanceof VectorialCovariance)) {
139             return false;
140         }
141         VectorialCovariance other = (VectorialCovariance) obj;
142         if (isBiasCorrected != other.isBiasCorrected) {
143             return false;
144         }
145         if (n != other.n) {
146             return false;
147         }
148         if (!Arrays.equals(productsSums, other.productsSums)) {
149             return false;
150         }
151         if (!Arrays.equals(sums, other.sums)) {
152             return false;
153         }
154         return true;
155     }
156 
157 }
