33  /**
34   * This class implements the common part of all embedded Runge-Kutta
35   * integrators for Ordinary Differential Equations.
36   *
37   * <p>These methods are embedded explicit Runge-Kutta methods with two
38   * sets of coefficients allowing to estimate the error, their Butcher
39   * arrays are as follows :
40   * <pre>
41   *    0  |
42   *   c2  | a21
43   *   c3  | a31  a32
44   *   ... |        ...
45   *   cs  | as1  as2  ...  ass-1
46   *       |--------------------------
47   *       |  b1   b2  ...   bs-1  bs
48   *       |  b'1  b'2 ...   b's-1 b's
49   * </pre>
50   * </p>
51   *
52   * <p>In fact, we rather use the array defined by ej = bj - b'j to
53   * compute directly the error rather than computing two estimates and
54   * then comparing them.</p>
55   *
56   * <p>Some methods are qualified as <i>fsal</i> (first same as last)
57   * methods. This means the last evaluation of the derivatives in one
58   * step is the same as the first in the next step. Then, this
59   * evaluation can be reused from one step to the next one and the cost
60   * of such a method is really s-1 evaluations despite the method still
61   * has s stages. This behaviour is true only for successful steps, if
62   * the step is rejected after the error estimation phase, no
63   * evaluation is saved. For an <i>fsal</i> method, we have cs = 1 and
64   * asi = bi for all i.</p>
65   *
66   * @param <T> the type of the field elements
67   * @since 3.6
68   */
69  
70  public abstract class EmbeddedRungeKuttaFieldIntegrator<T extends RealFieldElement<T>>
71      extends AdaptiveStepsizeFieldIntegrator<T>
72      implements FieldButcherArrayProvider<T> {
73  
74      /** Index of the pre-computed derivative for <i>fsal</i> methods. */
75      private final int fsal;

77      /** Time steps from Butcher array (without the first zero). */
78      private final T[] c;
79  
80      /** Internal weights from Butcher array (without the first empty row). */
81      private final T[][] a;

83      /** External weights for the high order method from Butcher array. */
84      private final T[] b;
85  
86      /** Stepsize control exponent. */
87      private final T exp;
88  
89      /** Safety factor for stepsize control. */
90      private T safety;

93      private T minReduction;
94  
95      /** Maximal growth factor for stepsize control. */
96      private T maxGrowth;

98      /** Build a Runge-Kutta integrator with the given Butcher array.
99       * @param field field to which the time and state vector elements belong
100      * @param name name of the method
101      * @param fsal index of the pre-computed derivative for <i>fsal</i> methods
102      * or -1 if method is not <i>fsal</i>
103      * @param minStep minimal step (sign is irrelevant, regardless of
104      * integration direction, forward or backward), the last step can
105      * be smaller than this
106      * @param maxStep maximal step (sign is irrelevant, regardless of
107      * integration direction, forward or backward), the last step can
108      * be smaller than this
109      * @param scalAbsoluteTolerance allowed absolute error
110      * @param scalRelativeTolerance allowed relative error
111      */
112     protected EmbeddedRungeKuttaFieldIntegrator(final Field<T> field, final String name, final int fsal,
113                                                 final double minStep, final double maxStep,
114                                                 final double scalAbsoluteTolerance,
115                                                 final double scalRelativeTolerance) {
116 
117         super(field, name, minStep, maxStep, scalAbsoluteTolerance, scalRelativeTolerance);
118 
119         this.fsal = fsal;
120         this.c    = getC();
121         this.a    = getA();
122         this.b    = getB();
123 
124         exp = field.getOne().divide(-getOrder());
125 
126         // set the default values of the algorithm control parameters
127         setSafety(field.getZero().add(0.9));
128         setMinReduction(field.getZero().add(0.2));
129         setMaxGrowth(field.getZero().add(10.0));
130 
131     }

215     /** {@inheritDoc} */
216     @Override
217     public FieldODEStateAndDerivative<T> integrate(final FieldExpandableODE<T> equations,
218                                                    final FieldODEState<T> initialState, final T finalTime)
219         throws NumberIsTooSmallException, DimensionMismatchException,
220         MaxCountExceededException, NoBracketingException {
221 
222         sanityChecks(initialState, finalTime);
223         final T   t0 = initialState.getTime();
224         final T[] y0 = equations.getMapper().mapState(initialState);
225         setStepStart(initIntegration(equations, t0, y0, finalTime));
226         final boolean forward = finalTime.subtract(initialState.getTime()).getReal() > 0;
227 
228         // create some internal working arrays
229         final int   stages = c.length + 1;
230         T[]         y      = y0;
231         final T[][] yDotK  = MathArrays.buildArray(getField(), stages, -1);
232         final T[]   yTmp   = MathArrays.buildArray(getField(), y0.length);
233 
234         // set up integration control objects
235         T  hNew           = getField().getZero();
236         boolean firstTime = true;
237 
238         // main integration loop
239         setIsLastStep(false);
240         do {
241 
242             // iterate over step size, ensuring local normalized error is smaller than 1
243             T error = getField().getZero().add(10);
244             while (error.subtract(1.0).getReal() >= 0) {
245 
246                 // first stage
247                 y        = equations.getMapper().mapState(getStepStart());
248                 yDotK[0] = equations.getMapper().mapDerivative(getStepStart());
249 
250                 if (firstTime) {
251                     final T[] scale = MathArrays.buildArray(getField(), mainSetDimension);
252                     if (vecAbsoluteTolerance == null) {
253                         for (int i = 0; i < scale.length; ++i) {
254                             scale[i] = y[i].abs().multiply(scalRelativeTolerance).add(scalAbsoluteTolerance);
255                         }
256                     } else {
257                         for (int i = 0; i < scale.length; ++i) {
258                             scale[i] = y[i].abs().multiply(vecRelativeTolerance[i]).add(vecAbsoluteTolerance[i]);
259                         }
260                     }
261                     hNew = initializeStep(forward, getOrder(), scale, getStepStart(), equations.getMapper());
262                     firstTime = false;
263                 }
264 
265                 setStepSize(hNew);
266                 if (forward) {
267                     if (getStepStart().getTime().add(getStepSize()).subtract(finalTime).getReal() >= 0) {
268                         setStepSize(finalTime.subtract(getStepStart().getTime()));
269                     }
270                 } else {
271                     if (getStepStart().getTime().add(getStepSize()).subtract(finalTime).getReal() <= 0) {
272                         setStepSize(finalTime.subtract(getStepStart().getTime()));
273                     }
274                 }
275 
276                 // next stages
277                 for (int k = 1; k < stages; ++k) {
278 
279                     for (int j = 0; j < y0.length; ++j) {
280                         T sum = yDotK[0][j].multiply(a[k-1][0]);
281                         for (int l = 1; l < k; ++l) {
282                             sum = sum.add(yDotK[l][j].multiply(a[k-1][l]));
283                         }
284                         yTmp[j] = y[j].add(getStepSize().multiply(sum));
285                     }
286 
287                     yDotK[k] = computeDerivatives(getStepStart().getTime().add(getStepSize().multiply(c[k-1])), yTmp);
288 
289                 }
290 
291                 // estimate the state at the end of the step
292                 for (int j = 0; j < y0.length; ++j) {
293                     T sum    = yDotK[0][j].multiply(b[0]);
294                     for (int l = 1; l < stages; ++l) {
295                         sum = sum.add(yDotK[l][j].multiply(b[l]));
296                     }
297                     yTmp[j] = y[j].add(getStepSize().multiply(sum));
298                 }
299 
300                 // estimate the error at the end of the step
301                 error = estimateError(yDotK, y, yTmp, getStepSize());
302                 if (error.subtract(1.0).getReal() >= 0) {
303                     // reject the step and attempt to reduce error by stepsize control
304                     final T factor = MathUtils.min(maxGrowth,
305                                                    MathUtils.max(minReduction, safety.multiply(error.pow(exp))));
306                     hNew = filterStep(getStepSize().multiply(factor), forward, false);
307                 }
308 
309             }
310             final T   stepEnd = getStepStart().getTime().add(getStepSize());
311             final T[] yDotTmp = (fsal >= 0) ? yDotK[fsal] : computeDerivatives(stepEnd, yTmp);
312             final FieldODEStateAndDerivative<T> stateTmp = new FieldODEStateAndDerivative<T>(stepEnd, yTmp, yDotTmp);
313 
314             // local error is small enough: accept the step, trigger events and step handlers
315             System.arraycopy(yTmp, 0, y, 0, y0.length);
316             setStepStart(acceptStep(createInterpolator(forward, yDotK, getStepStart(), stateTmp, equations.getMapper()),
317                                     finalTime));
318 
319             if (!isLastStep()) {
320 
321                 // stepsize control for next step
322                 final T factor = MathUtils.min(maxGrowth,
323                                                MathUtils.max(minReduction, safety.multiply(error.pow(exp))));
324                 final T  scaledH    = getStepSize().multiply(factor);
325                 final T  nextT      = getStepStart().getTime().add(scaledH);
326                 final boolean nextIsLast = forward ?
327                                            nextT.subtract(finalTime).getReal() >= 0 :
328                                            nextT.subtract(finalTime).getReal() <= 0;
329                 hNew = filterStep(scaledH, forward, nextIsLast);
330 
331                 final T  filteredNextT      = getStepStart().getTime().add(hNew);
332                 final boolean filteredNextIsLast = forward ?
333                                                    filteredNextT.subtract(finalTime).getReal() >= 0 :
334                                                    filteredNextT.subtract(finalTime).getReal() <= 0;
335                 if (filteredNextIsLast) {
336                     hNew = finalTime.subtract(getStepStart().getTime());
337                 }
338 
339             }
340 
341         } while (!isLastStep());
342 
343         final FieldODEStateAndDerivative<T> finalState = getStepStart();
344         resetInternalState();
345         return finalState;
346 
347     }
