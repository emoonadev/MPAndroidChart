23  /**
24   * This class wraps a {@code double} value in an object. It is similar to the
25   * standard class {@link Double}, while also implementing the
26   * {@link RealFieldElement} interface.
27   *
28   * @since 3.1
29   */
30  public class Decimal64 extends Number
31                         implements RealFieldElement<Decimal64>, Comparable<Decimal64> {
32  
33      /** The constant value of {@code 0d} as a {@code Decimal64}. */
34      public static final Decimal64 ZERO;

36      /** The constant value of {@code 1d} as a {@code Decimal64}. */
37      public static final Decimal64 ONE;
38

40       * The constant value of {@link Double#NEGATIVE_INFINITY} as a

252     /** {@inheritDoc} */
253     @Override
254     public boolean equals(final Object obj) {
255         if (obj instanceof Decimal64) {
256             final Decimal64 that = (Decimal64) obj;
257             return Double.doubleToLongBits(this.value) == Double
258                     .doubleToLongBits(that.value);
259         }
260         return false;
261     }
262 
263     /**
264      * {@inheritDoc}
265      *
266      * The current implementation returns the same value as
267      * {@code new Double(this.doubleValue()).hashCode()}
268      *
269      * @see Double#hashCode()
270      */
271     @Override
272     public int hashCode() {
273         long v = Double.doubleToLongBits(value);
274         return (int) (v ^ (v >>> 32));
275     }

647     /** {@inheritDoc}
648      * @since 3.2
649      */
650     @Override
651     public Decimal64 linearCombination(final Decimal64[] a, final Decimal64[] b)
652         throws DimensionMismatchException {
653         if (a.length != b.length) {
654             throw new DimensionMismatchException(a.length, b.length);
655         }
656         final double[] aDouble = new double[a.length];
657         final double[] bDouble = new double[b.length];
658         for (int i = 0; i < a.length; ++i) {
659             aDouble[i] = a[i].value;
660             bDouble[i] = b[i].value;
661         }
662         return new Decimal64(MathArrays.linearCombination(aDouble, bDouble));
663     }
