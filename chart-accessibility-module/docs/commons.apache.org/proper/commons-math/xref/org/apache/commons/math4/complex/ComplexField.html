18  package org.apache.commons.math4.complex;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.Field;
23  import org.apache.commons.math4.FieldElement;
24  
25  /**
26   * Representation of the complex numbers field.
27   * <p>
28   * This class is a singleton.
29   * </p>
30   * @see Complex
31   * @since 2.0
32   */
33  public class ComplexField implements Field<Complex>, Serializable {
34  
35      /** Serializable version identifier. */
36      private static final long serialVersionUID = -6130362688700788798L;
37  
38      /** Private constructor for the singleton.
39       */
40      private ComplexField() {
41      }
42  
43      /** Get the unique instance.
44       * @return the unique instance
45       */
46      public static ComplexField getInstance() {
47          return LazyHolder.INSTANCE;
48      }
49  
50      /** {@inheritDoc} */
51      @Override
52      public Complex getOne() {
53          return Complex.ONE;
54      }
55  
56      /** {@inheritDoc} */
57      @Override
58      public Complex getZero() {
59          return Complex.ZERO;
60      }
61  
62      /** {@inheritDoc} */
63      @Override
64      public Class<? extends FieldElement<Complex>> getRuntimeClass() {
65          return Complex.class;
66      }
67  
68      // CHECKSTYLE: stop HideUtilityClassConstructor
69      /** Holder for the instance.
70       * <p>We use here the Initialization On Demand Holder Idiom.</p>
71       */
72      private static class LazyHolder {
73          /** Cached field instance. */
74          private static final ComplexField INSTANCE = new ComplexField();
75      }
76      // CHECKSTYLE: resume HideUtilityClassConstructor
77  
78      /** Handle deserialization of the singleton.
79       * @return the singleton instance
80       */
81      private Object readResolve() {
82          // return the singleton instance
83          return LazyHolder.INSTANCE;
84      }
85  
86  }
