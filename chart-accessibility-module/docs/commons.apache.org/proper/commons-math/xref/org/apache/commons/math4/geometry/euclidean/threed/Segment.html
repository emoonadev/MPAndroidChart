17  package org.apache.commons.math4.geometry.euclidean.threed;
18  
19  
20  /** Simple container for a two-points segment.
21   * @since 3.0
22   */
23  public class Segment {
24  
25      /** Start point of the segment. */
26      private final Vector3D start;
27  
28      /** End point of the segments. */
29      private final Vector3D end;
30  
31      /** Line containing the segment. */
32      private final Line     line;
33  
34      /** Build a segment.
35       * @param start start point of the segment
36       * @param end end point of the segment
37       * @param line line containing the segment
38       */
39      public Segment(final Vector3D start, final Vector3D end, final Line line) {
40          this.start  = start;
41          this.end    = end;
42          this.line   = line;
43      }
44  
45      /** Get the start point of the segment.
46       * @return start point of the segment
47       */
48      public Vector3D getStart() {
49          return start;
50      }
51  
52      /** Get the end point of the segment.
53       * @return end point of the segment
54       */
55      public Vector3D getEnd() {
56          return end;
57      }
58  
59      /** Get the line containing the segment.
60       * @return line containing the segment
61       */
62      public Line getLine() {
63          return line;
64      }
65  
66  }
