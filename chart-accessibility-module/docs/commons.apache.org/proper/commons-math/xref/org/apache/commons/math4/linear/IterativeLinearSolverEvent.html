17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
20  import org.apache.commons.math4.util.IterationEvent;
21  
22  /**
23   * This is the base class for all events occurring during the iterations of a
24   * {@link IterativeLinearSolver}.
25   *
26   * @since 3.0
27   */
28  public abstract class IterativeLinearSolverEvent
29      extends IterationEvent {
30      /** Serialization identifier. */
31      private static final long serialVersionUID = 20120129L;
32  
33      /**
34       * Creates a new instance of this class.
35       *
36       * @param source the iterative algorithm on which the event initially
37       * occurred
38       * @param iterations the number of iterations performed at the time
39       * {@code this} event is created
40       */
41      public IterativeLinearSolverEvent(final Object source, final int iterations) {
42          super(source, iterations);
43      }
44  
45      /**
46       * Returns the current right-hand side of the linear system to be solved.
47       * This method should return an unmodifiable view, or a deep copy of the
48       * actual right-hand side vector, in order not to compromise subsequent
49       * iterations of the source {@link IterativeLinearSolver}.
50       *
51       * @return the right-hand side vector, b
52       */
53      public abstract RealVector getRightHandSideVector();
54  
55      /**
56       * Returns the norm of the residual. The returned value is not required to
57       * be <em>exact</em>. Instead, the norm of the so-called <em>updated</em>
58       * residual (if available) should be returned. For example, the
59       * {@link ConjugateGradient conjugate gradient} method computes a sequence
60       * of residuals, the norm of which is cheap to compute. However, due to
61       * accumulation of round-off errors, this residual might differ from the
62       * true residual after some iterations. See e.g. A. Greenbaum and
63       * Z. Strakos, <em>Predicting the Behavior of Finite Precision Lanzos and
64       * Conjugate Gradient Computations</em>, Technical Report 538, Department of
65       * Computer Science, New York University, 1991 (available
66       * <a href="http://www.archive.org/details/predictingbehavi00gree">here</a>).
67       *
68       * @return the norm of the residual, ||r||
69       */
70      public abstract double getNormOfResidual();
71  
72      /**
73       * <p>
74       * Returns the residual. This is an optional operation, as all iterative
75       * linear solvers do not provide cheap estimate of the updated residual
76       * vector, in which case
77       * </p>
78       * <ul>
79       * <li>this method should throw a
80       * {@link MathUnsupportedOperationException},</li>
81       * <li>{@link #providesResidual()} returns {@code false}.</li>
82       * </ul>
83       * <p>
84       * The default implementation throws a
85       * {@link MathUnsupportedOperationException}. If this method is overriden,
86       * then {@link #providesResidual()} should be overriden as well.
87       * </p>
88       *
89       * @return the updated residual, r
90       */
91      public RealVector getResidual() {
92          throw new MathUnsupportedOperationException();
93      }
94  
95      /**
96       * Returns the current estimate of the solution to the linear system to be
97       * solved. This method should return an unmodifiable view, or a deep copy of
98       * the actual current solution, in order not to compromise subsequent
99       * iterations of the source {@link IterativeLinearSolver}.
100      *
101      * @return the solution, x
102      */
103     public abstract RealVector getSolution();
104 
105     /**
106      * Returns {@code true} if {@link #getResidual()} is supported. The default
107      * implementation returns {@code false}.
108      *
109      * @return {@code false} if {@link #getResidual()} throws a
110      * {@link MathUnsupportedOperationException}
111      */
112     public boolean providesResidual() {
113         return false;
114     }
115 }
