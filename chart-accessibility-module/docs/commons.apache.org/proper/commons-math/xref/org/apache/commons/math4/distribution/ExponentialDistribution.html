17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.rng.UniformRandomProvider;
23  import org.apache.commons.math4.util.CombinatoricsUtils;
24  import org.apache.commons.math4.util.FastMath;
25  import org.apache.commons.math4.util.ResizableDoubleArray;
26  
27  /**
28   * Implementation of the exponential distribution.
29   *
30   * @see <a href="http://en.wikipedia.org/wiki/Exponential_distribution">Exponential distribution (Wikipedia)</a>
31   * @see <a href="http://mathworld.wolfram.com/ExponentialDistribution.html">Exponential distribution (MathWorld)</a>
32   */
33  public class ExponentialDistribution extends AbstractRealDistribution {
34      /**
35       * Default inverse cumulative probability accuracy.
36       * @since 2.1
37       */
38      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
39      /** Serializable version identifier */
40      private static final long serialVersionUID = 20160311L;
41      /**
42       * Used when generating Exponential samples.
43       * Table containing the constants
44       * q_i = sum_{j=1}^i (ln 2)^j/j! = ln 2 + (ln 2)^2/2 + ... + (ln 2)^i/i!
45       * until the largest representable fraction below 1 is exceeded.
46       *
47       * Note that
48       * 1 = 2 - 1 = exp(ln 2) - 1 = sum_{n=1}^infty (ln 2)^n / n!
49       * thus q_i -> 1 as i -> +inf,
50       * so the higher i, the closer to one we get (the series is not alternating).
51       *
52       * By trying, n = 16 in Java is enough to reach 1.0.
53       */
54      private static final double[] EXPONENTIAL_SA_QI;
55      /** The mean of this distribution. */
56      private final double mean;
57      /** The logarithm of the mean, stored to reduce computing time. **/
58      private final double logMean;
59      /** Inverse cumulative probability accuracy. */
60      private final double solverAbsoluteAccuracy;
61  
62      /**
63       * Initialize tables.
64       */
65      static {
66          /**
67           * Filling EXPONENTIAL_SA_QI table.
68           * Note that we don't want qi = 0 in the table.
69           */
70          final double LN2 = FastMath.log(2);
71          double qi = 0;
72          int i = 1;
73  
74          /**
75           * ArithmeticUtils provides factorials up to 20, so let's use that
76           * limit together with Precision.EPSILON to generate the following
77           * code (a priori, we know that there will be 16 elements, but it is
78           * better to not hardcode it).
79           */
80          final ResizableDoubleArray ra = new ResizableDoubleArray(20);
81  
82          while (qi < 1) {
83              qi += FastMath.pow(LN2, i) / CombinatoricsUtils.factorial(i);
84              ra.addElement(qi);
85              ++i;
86          }
87  
88          EXPONENTIAL_SA_QI = ra.getElements();
89      }
90  
91      /**
92       * Creates a distribution.
93       *
94       * @param mean mean of this distribution.
95       */
96      public ExponentialDistribution(double mean) {
97          this(mean, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
98      }
99  
100     /**
101      * Creates a distribution.
102      *
103      * @param mean Mean of this distribution.
104      * @param inverseCumAccuracy Maximum absolute error in inverse
105      * cumulative probability estimates (defaults to
106      * {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
107      * @throws NotStrictlyPositiveException if {@code mean <= 0}.
108      *
109      * @since 2.1
110      */
111     public ExponentialDistribution(double mean,
112                                    double inverseCumAccuracy)
113         throws NotStrictlyPositiveException {
114         if (mean <= 0) {
115             throw new NotStrictlyPositiveException(LocalizedFormats.MEAN, mean);
116         }
117         this.mean = mean;
118         logMean = FastMath.log(mean);
119         solverAbsoluteAccuracy = inverseCumAccuracy;
120     }
121 
122     /**
123      * Access the mean.
124      *
125      * @return the mean.
126      */
127     public double getMean() {
128         return mean;
129     }
130 
131     /** {@inheritDoc} */
132     @Override
133     public double density(double x) {
134         final double logDensity = logDensity(x);
135         return logDensity == Double.NEGATIVE_INFINITY ? 0 : FastMath.exp(logDensity);
136     }
137 
138     /** {@inheritDoc} **/
139     @Override
140     public double logDensity(double x) {
141         if (x < 0) {
142             return Double.NEGATIVE_INFINITY;
143         }
144         return -x / mean - logMean;
145     }
146 
147     /**
148      * {@inheritDoc}
149      *
150      * The implementation of this method is based on:
151      * <ul>
152      * <li>
153      * <a href="http://mathworld.wolfram.com/ExponentialDistribution.html">
154      * Exponential Distribution</a>, equation (1).</li>
155      * </ul>
156      */
157     @Override
158     public double cumulativeProbability(double x)  {
159         double ret;
160         if (x <= 0.0) {
161             ret = 0.0;
162         } else {
163             ret = 1.0 - FastMath.exp(-x / mean);
164         }
165         return ret;
166     }
167 
168     /**
169      * {@inheritDoc}
170      *
171      * Returns {@code 0} when {@code p= = 0} and
172      * {@code Double.POSITIVE_INFINITY} when {@code p == 1}.
173      */
174     @Override
175     public double inverseCumulativeProbability(double p) throws OutOfRangeException {
176         double ret;
177 
178         if (p < 0.0 || p > 1.0) {
179             throw new OutOfRangeException(p, 0.0, 1.0);
180         } else if (p == 1.0) {
181             ret = Double.POSITIVE_INFINITY;
182         } else {
183             ret = -mean * FastMath.log(1.0 - p);
184         }
185 
186         return ret;
187     }
188 
189     /** {@inheritDoc} */
190     @Override
191     protected double getSolverAbsoluteAccuracy() {
192         return solverAbsoluteAccuracy;
193     }
194 
195     /**
196      * {@inheritDoc}
197      *
198      * For mean parameter {@code k}, the mean is {@code k}.
199      */
200     @Override
201     public double getNumericalMean() {
202         return getMean();
203     }
204 
205     /**
206      * {@inheritDoc}
207      *
208      * For mean parameter {@code k}, the variance is {@code k^2}.
209      */
210     @Override
211     public double getNumericalVariance() {
212         final double m = getMean();
213         return m * m;
214     }

253     /**
254      * {@inheritDoc}
255      *
256      * <p>Sampling algorithm uses the
257      *  <a href="http://www.jesus.ox.ac.uk/~clifford/a5/chap1/node5.html">
258      *   inversion method</a> to generate exponentially distributed
259      *  random values from uniform deviates.
260      * </p>
261      */
262     @Override
263     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
264         return new RealDistribution.Sampler() {
265             /** {@inheritDoc} */
266             @Override
267             public double sample() {
268                 // Step 1:
269                 double a = 0;
270                 double u = rng.nextDouble();
271 
272                 // Step 2 and 3:
273                 while (u < 0.5) {
274                     a += EXPONENTIAL_SA_QI[0];
275                     u *= 2;
276                 }
277 
278                 // Step 4 (now u >= 0.5):
279                 u += u - 1;
280 
281                 // Step 5:
282                 if (u <= EXPONENTIAL_SA_QI[0]) {
283                     return mean * (a + u);
284                 }
285 
286                 // Step 6:
287                 int i = 0; // Should be 1, be we iterate before it in while using 0
288                 double u2 = rng.nextDouble();
289                 double umin = u2;
290 
291                 // Step 7 and 8:
292                 do {
293                     ++i;
294                     u2 = rng.nextDouble();
295 
296                     if (u2 < umin) {
297                         umin = u2;
298                     }
299 
300                     // Step 8:
301                 } while (u > EXPONENTIAL_SA_QI[i]); // Ensured to exit since EXPONENTIAL_SA_QI[MAX] = 1
302 
303                 return mean * (a + umin * EXPONENTIAL_SA_QI[0]);
304             }
305         };
306     }
