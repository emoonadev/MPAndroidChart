17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  
21  /**
22   * Exception to be thrown when the argument is not greater than 0.
23   *
24   * @since 2.2
25   */
26  public class NotStrictlyPositiveException extends NumberIsTooSmallException {
27  
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -7824848630829852237L;
30  
31      /**
32       * Construct the exception.
33       *
34       * @param value Argument.
35       */
36      public NotStrictlyPositiveException(Number value) {
37          super(value, INTEGER_ZERO, false);
38      }
39      /**
40       * Construct the exception with a specific context.
41       *
42       * @param specific Specific context where the error occurred.
43       * @param value Argument.
44       */
45      public NotStrictlyPositiveException(Localizable specific,
46                                          Number value) {
47          super(specific, value, INTEGER_ZERO, false);
48      }
49  }
