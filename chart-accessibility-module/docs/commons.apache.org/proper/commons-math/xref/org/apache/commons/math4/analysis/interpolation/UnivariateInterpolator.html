17  package org.apache.commons.math4.analysis.interpolation;
18  
19  import org.apache.commons.math4.analysis.UnivariateFunction;
20  import org.apache.commons.math4.exception.DimensionMismatchException;
21  import org.apache.commons.math4.exception.MathIllegalArgumentException;
22  
23  /**
24   * Interface representing a univariate real interpolating function.
25   *
26   */
27  public interface UnivariateInterpolator {
28      /**
29       * Compute an interpolating function for the dataset.
30       *
31       * @param xval Arguments for the interpolation points.
32       * @param yval Values for the interpolation points.
33       * @return a function which interpolates the dataset.
34       * @throws MathIllegalArgumentException
35       * if the arguments violate assumptions made by the interpolation
36       * algorithm.
37       * @throws DimensionMismatchException if arrays lengthes do not match
38       */
39      UnivariateFunction interpolate(double xval[], double yval[])
40          throws MathIllegalArgumentException, DimensionMismatchException;
41  }
