18  package org.apache.commons.math4.random;
19  
20  import java.util.Arrays;
21  
22  import org.apache.commons.math4.exception.DimensionMismatchException;
23  
24  /**
25   * A {@link RandomVectorGenerator} that generates vectors with uncorrelated
26   * components. Components of generated vectors follow (independent) Gaussian
27   * distributions, with parameters supplied in the constructor.
28   *
29   * @since 1.2
30   */
31  public class UncorrelatedRandomVectorGenerator
32    implements RandomVectorGenerator {
33  
34      /** Underlying scalar generator. */
35      private final NormalizedRandomGenerator generator;
36  
37      /** Mean vector. */
38      private final double[] mean;
39  
40      /** Standard deviation vector. */
41      private final double[] standardDeviation;
42  
43    /** Simple constructor.
44     * <p>Build an uncorrelated random vector generator from
45     * its mean and standard deviation vectors.</p>
46     * @param mean expected mean values for each component
47     * @param standardDeviation standard deviation for each component
48     * @param generator underlying generator for uncorrelated normalized
49     * components
50     */
51    public UncorrelatedRandomVectorGenerator(double[] mean,
52                                             double[] standardDeviation,
53                                             NormalizedRandomGenerator generator) {
54      if (mean.length != standardDeviation.length) {
55          throw new DimensionMismatchException(mean.length, standardDeviation.length);
56      }
57      this.mean              = mean.clone();
58      this.standardDeviation = standardDeviation.clone();
59      this.generator = generator;
60    }
61  
62    /** Simple constructor.
63     * <p>Build a null mean random and unit standard deviation
64     * uncorrelated vector generator</p>
65     * @param dimension dimension of the vectors to generate
66     * @param generator underlying generator for uncorrelated normalized
67     * components
68     */
69    public UncorrelatedRandomVectorGenerator(int dimension,
70                                             NormalizedRandomGenerator generator) {
71      mean              = new double[dimension];
72      standardDeviation = new double[dimension];
73      Arrays.fill(standardDeviation, 1.0);
74      this.generator = generator;
75    }
76  
77    /** Generate an uncorrelated random vector.
78     * @return a random vector as a newly built array of double
79     */
80    @Override
81  public double[] nextVector() {
82  
83      double[] random = new double[mean.length];
84      for (int i = 0; i < random.length; ++i) {
85        random[i] = mean[i] + standardDeviation[i] * generator.nextNormalizedDouble();
86      }
87  
88      return random;
89  
90    }
91  
92  }
