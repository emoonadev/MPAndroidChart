17  package org.apache.commons.math4.analysis;
18  
19  import org.apache.commons.math4.RealFieldElement;
20  
21  /**
22   * An interface representing a univariate real function.
23   * <p>
24   * When a <em>user-defined</em> function encounters an error during
25   * evaluation, the {@link #value(RealFieldElement) value} method should throw a
26   * <em>user-defined</em> unchecked exception.</p>
27   * <p>
28   * The following code excerpt shows the recommended way to do that using
29   * a root solver as an example, but the same construct is applicable to
30   * ODE integrators or optimizers.
31   *
32   * <pre>
33   * private static class LocalException extends RuntimeException {
34   *     // The x value that caused the problem.
35   *     private final SomeFieldType x;
36   *
37   *     public LocalException(SomeFieldType x) {
38   *         this.x = x;
39   *     }
40   *
41   *     public double getX() {
42   *         return x;
43   *     }
44   * }
45   *
46   * private static class MyFunction implements FieldUnivariateFunction&lt;SomeFieldType&gt; {
47   *     public SomeFieldType value(SomeFieldType x) {
48   *         SomeFieldType y = hugeFormula(x);
49   *         if (somethingBadHappens) {
50   *           throw new LocalException(x);
51   *         }
52   *         return y;
53   *     }
54   * }
55   *
56   * public void compute() {
57   *     try {
58   *         solver.solve(maxEval, new MyFunction(a, b, c), min, max);
59   *     } catch (LocalException le) {
60   *         // Retrieve the x value.
61   *     }
62   * }
63   * </pre>
64   *<p>
65   * As shown, the exception is local to the user's code and it is guaranteed
66   * that Apache Commons Math will not catch it.</p>
67   *
68   * @param <T> the type of the field elements
69   * @since 3.6
70   * @see UnivariateFunction
71   */
72  public interface RealFieldUnivariateFunction<T extends RealFieldElement<T>> {
73      /**
74       * Compute the value of the function.
75       *
76       * @param x Point at which the function value should be computed.
77       * @return the value of the function.
78       * @throws IllegalArgumentException when the activated method itself can
79       * ascertain that a precondition, specified in the API expressed at the
80       * level of the activated method, has been violated.
81       * When Commons Math throws an {@code IllegalArgumentException}, it is
82       * usually the consequence of checking the actual parameters passed to
83       * the method.
84       */
85      T value(T x);
86  }
