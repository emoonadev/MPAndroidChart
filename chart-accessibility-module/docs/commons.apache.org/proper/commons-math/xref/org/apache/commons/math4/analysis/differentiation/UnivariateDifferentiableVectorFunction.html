17  package org.apache.commons.math4.analysis.differentiation;
18  
19  import org.apache.commons.math4.analysis.UnivariateVectorFunction;
20  import org.apache.commons.math4.exception.MathIllegalArgumentException;
21  
22  /**
23   * Extension of {@link UnivariateVectorFunction} representing a univariate differentiable vectorial function.
24   *
25   * @since 3.1
26   */
27  public interface UnivariateDifferentiableVectorFunction
28      extends UnivariateVectorFunction {
29  
30      /**
31       * Compute the value for the function.
32       * @param x the point for which the function value should be computed
33       * @return the value
34       * @exception MathIllegalArgumentException if {@code x} does not
35       * satisfy the function's constraints (argument out of bound, or unsupported
36       * derivative order for example)
37       */
38      DerivativeStructure[] value(DerivativeStructure x) throws MathIllegalArgumentException;
39  
40  }
