17  package org.apache.commons.math4.ode;
18  
19  
20  /** Interface to compute by finite difference Jacobian matrix for some parameter
21   *  when computing {@link JacobianMatrices partial derivatives equations}.
22   *
23   * @since 3.0
24   */
25  
26  public interface ParameterizedODE extends Parameterizable {
27  
28      /** Get parameter value from its name.
29       * @param name parameter name
30       * @return parameter value
31       * @exception UnknownParameterException if parameter is not supported
32       */
33      double getParameter(String name) throws UnknownParameterException;
34  
35      /** Set the value for a given parameter.
36       * @param name parameter name
37       * @param value parameter value
38       * @exception UnknownParameterException if parameter is not supported
39       */
40      void setParameter(String name, double value) throws UnknownParameterException;
41  
42  }
