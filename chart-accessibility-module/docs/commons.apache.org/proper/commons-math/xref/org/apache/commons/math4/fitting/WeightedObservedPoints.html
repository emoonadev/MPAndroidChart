17  package org.apache.commons.math4.fitting;
18  
19  import java.util.List;
20  import java.util.ArrayList;
21  import java.io.Serializable;
22  
23  /**
24   * Simple container for weighted observed points used
25   * in {@link AbstractCurveFitter curve fitting} algorithms.
26   *
27   * @since 3.3
28   */
29  public class WeightedObservedPoints implements Serializable {
30      /** Serializable version id. */
31      private static final long serialVersionUID = 20130813L;
32  
33      /** Observed points. */
34      private final List<WeightedObservedPoint> observations
35          = new ArrayList<WeightedObservedPoint>();
36  
37      /**
38       * Adds a point to the sample.
39       * Calling this method is equivalent to calling
40       * {@code add(1.0, x, y)}.
41       *
42       * @param x Abscissa of the point.
43       * @param y Observed value  at {@code x}. After fitting we should
44       * have {@code f(x)} as close as possible to this value.
45       *
46       * @see #add(double, double, double)
47       * @see #add(WeightedObservedPoint)
48       * @see #toList()
49       */
50      public void add(double x, double y) {
51          add(1d, x, y);
52      }
53  
54      /**
55       * Adds a point to the sample.
56       *
57       * @param weight Weight of the observed point.
58       * @param x Abscissa of the point.
59       * @param y Observed value  at {@code x}. After fitting we should
60       * have {@code f(x)} as close as possible to this value.
61       *
62       * @see #add(double, double)
63       * @see #add(WeightedObservedPoint)
64       * @see #toList()
65       */
66      public void add(double weight, double x, double y) {
67          observations.add(new WeightedObservedPoint(weight, x, y));
68      }
69  
70      /**
71       * Adds a point to the sample.
72       *
73       * @param observed Observed point to add.
74       *
75       * @see #add(double, double)
76       * @see #add(double, double, double)
77       * @see #toList()
78       */
79      public void add(WeightedObservedPoint observed) {
80          observations.add(observed);
81      }
82  
83      /**
84       * Gets a <em>snapshot</em> of the observed points.
85       * The list of stored points is copied in order to ensure that
86       * modification of the returned instance does not affect this
87       * container.
88       * Conversely, further modification of this container (through
89       * the {@code add} or {@code clear} methods) will not affect the
90       * returned list.
91       *
92       * @return the observed points, in the order they were added to this
93       * container.
94       *
95       * @see #add(double, double)
96       * @see #add(double, double, double)
97       * @see #add(WeightedObservedPoint)
98       */
99      public List<WeightedObservedPoint> toList() {
100         // The copy is necessary to ensure thread-safety because of the
101         // "clear" method (which otherwise would be able to empty the
102         // list of points while it is being used by another thread).
103         return new ArrayList<WeightedObservedPoint>(observations);
104     }
105 
106     /**
107      * Removes all observations from this container.
108      */
109     public void clear() {
110         observations.clear();
111     }
112 }
