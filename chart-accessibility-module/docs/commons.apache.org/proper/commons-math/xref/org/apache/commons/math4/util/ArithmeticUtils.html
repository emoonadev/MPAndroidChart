40       *
41       * @param x an addend
42       * @param y an addend
43       * @return the sum {@code x+y}
44       * @throws MathArithmeticException if the result can not be represented
45       * as an {@code int}.
46       * @since 1.1

48      public static int addAndCheck(int x, int y)
49              throws MathArithmeticException {

70      /**
71       * Computes the greatest common divisor of the absolute value of two
72       * numbers, using a modified version of the "binary gcd" method.
73       * See Knuth 4.5.2 algorithm B.
74       * The algorithm is due to Josef Stein (1961).
75       * <br/>
76       * Special cases:
77       * <ul>
78       *  <li>The invocations
79       *   {@code gcd(Integer.MIN_VALUE, Integer.MIN_VALUE)},
80       *   {@code gcd(Integer.MIN_VALUE, 0)} and
81       *   {@code gcd(0, Integer.MIN_VALUE)} throw an
82       *   {@code ArithmeticException}, because the result would be 2^31, which
83       *   is too large for an int value.</li>
84       *  <li>The result of {@code gcd(x, x)}, {@code gcd(0, x)} and
85       *   {@code gcd(x, 0)} is the absolute value of {@code x}, except
86       *   for the special cases above.</li>
87       *  <li>The invocation {@code gcd(0, 0)} is the only one which returns
88       *   {@code 0}.</li>
89       * </ul>
90       *
91       * @param p Number.
92       * @param q Number.
93       * @return the greatest common divisor (never negative).
94       * @throws MathArithmeticException if the result cannot be represented as
95       * a non-negative {@code int} value.
96       * @since 1.1
97       */
98      public static int gcd(int p, int q) throws MathArithmeticException {
99          int a = p;
100         int b = q;
101         if (a == 0 ||
102             b == 0) {
103             if (a == Integer.MIN_VALUE ||
104                 b == Integer.MIN_VALUE) {
105                 throw new MathArithmeticException(LocalizedFormats.GCD_OVERFLOW_32_BITS,
106                                                   p, q);
107             }
108             return FastMath.abs(a + b);
109         }
110 
111         long al = a;
112         long bl = b;
113         boolean useLong = false;
114         if (a < 0) {
115             if(Integer.MIN_VALUE == a) {
116                 useLong = true;
117             } else {
118                 a = -a;
119             }
120             al = -al;
121         }
122         if (b < 0) {
123             if (Integer.MIN_VALUE == b) {
124                 useLong = true;
125             } else {
126                 b = -b;
127             }
128             bl = -bl;
129         }
130         if (useLong) {
131             if(al == bl) {
132                 throw new MathArithmeticException(LocalizedFormats.GCD_OVERFLOW_32_BITS,
133                                                   p, q);
134             }
135             long blbu = bl;
136             bl = al;
137             al = blbu % al;
138             if (al == 0) {
139                 if (bl > Integer.MAX_VALUE) {
140                     throw new MathArithmeticException(LocalizedFormats.GCD_OVERFLOW_32_BITS,
141                                                       p, q);
142                 }
143                 return (int) bl;
144             }
145             blbu = bl;
146 
147             // Now "al" and "bl" fit in an "int".
148             b = (int) al;
149             a = (int) (blbu % al);
150         }
151 
152         return gcdPositive(a, b);
153     }

155     /**
156      * Computes the greatest common divisor of two <em>positive</em> numbers
157      * (this precondition is <em>not</em> checked and the result is undefined
158      * if not fulfilled) using the "binary gcd" method which avoids division
159      * and modulo operations.
160      * See Knuth 4.5.2 algorithm B.
161      * The algorithm is due to Josef Stein (1961).
162      * <br/>
163      * Special cases:
164      * <ul>
165      *  <li>The result of {@code gcd(x, x)}, {@code gcd(0, x)} and
166      *   {@code gcd(x, 0)} is the value of {@code x}.</li>
167      *  <li>The invocation {@code gcd(0, 0)} is the only one which returns
168      *   {@code 0}.</li>
169      * </ul>
170      *
171      * @param a Positive number.
172      * @param b Positive number.
173      * @return the greatest common divisor.
174      */
175     private static int gcdPositive(int a, int b) {
176         if (a == 0) {
177             return b;
178         }
179         else if (b == 0) {
180             return a;
181         }
182 
183         // Make "a" and "b" odd, keeping track of common power of 2.
184         final int aTwos = Integer.numberOfTrailingZeros(a);
185         a >>= aTwos;
186         final int bTwos = Integer.numberOfTrailingZeros(b);
187         b >>= bTwos;
188         final int shift = FastMath.min(aTwos, bTwos);
189 
190         // "a" and "b" are positive.
191         // If a > b then "gdc(a, b)" is equal to "gcd(a - b, b)".
192         // If a < b then "gcd(a, b)" is equal to "gcd(b - a, a)".
193         // Hence, in the successive iterations:
194         //  "a" becomes the absolute difference of the current values,
195         //  "b" becomes the minimum of the current values.
196         while (a != b) {
197             final int delta = a - b;
198             b = Math.min(a, b);
199             a = Math.abs(delta);
200 
201             // Remove any power of 2 in "a" ("b" is guaranteed to be odd).
202             a >>= Integer.numberOfTrailingZeros(a);
203         }
204 
205         // Recover the common power of 2.
206         return a << shift;
207     }

209     /**
210      * <p>
211      * Gets the greatest common divisor of the absolute value of two numbers,
212      * using the "binary gcd" method which avoids division and modulo
213      * operations. See Knuth 4.5.2 algorithm B. This algorithm is due to Josef
214      * Stein (1961).
215      * </p>
216      * Special cases:
217      * <ul>
218      * <li>The invocations
219      * {@code gcd(Long.MIN_VALUE, Long.MIN_VALUE)},
220      * {@code gcd(Long.MIN_VALUE, 0L)} and
221      * {@code gcd(0L, Long.MIN_VALUE)} throw an
222      * {@code ArithmeticException}, because the result would be 2^63, which
223      * is too large for a long value.</li>
224      * <li>The result of {@code gcd(x, x)}, {@code gcd(0L, x)} and
225      * {@code gcd(x, 0L)} is the absolute value of {@code x}, except
226      * for the special cases above.
227      * <li>The invocation {@code gcd(0L, 0L)} is the only one which returns
228      * {@code 0L}.</li>
229      * </ul>
230      *
231      * @param p Number.
232      * @param q Number.
233      * @return the greatest common divisor, never negative.
234      * @throws MathArithmeticException if the result cannot be represented as
235      * a non-negative {@code long} value.
236      * @since 2.1
237      */
238     public static long gcd(final long p, final long q) throws MathArithmeticException {
239         long u = p;
240         long v = q;
241         if ((u == 0) || (v == 0)) {
242             if ((u == Long.MIN_VALUE) || (v == Long.MIN_VALUE)){
243                 throw new MathArithmeticException(LocalizedFormats.GCD_OVERFLOW_64_BITS,
244                                                   p, q);
245             }
246             return FastMath.abs(u) + FastMath.abs(v);
247         }
248         // keep u and v negative, as negative integers range down to
249         // -2^63, while positive numbers can only be as large as 2^63-1
250         // (i.e. we can't necessarily negate a negative number without
251         // overflow)
252         /* assert u!=0 && v!=0; */
253         if (u > 0) {
254             u = -u;
255         } // make u negative
256         if (v > 0) {
257             v = -v;
258         } // make v negative
259         // B1. [Find power of 2]
260         int k = 0;
261         while ((u & 1) == 0 && (v & 1) == 0 && k < 63) { // while u and v are
262                                                             // both even...
263             u /= 2;
264             v /= 2;
265             k++; // cast out twos.
266         }
267         if (k == 63) {
268             throw new MathArithmeticException(LocalizedFormats.GCD_OVERFLOW_64_BITS,
269                                               p, q);
270         }
271         // B2. Initialize: u and v have been divided by 2^k and at least
272         // one is odd.
273         long t = ((u & 1) == 1) ? v : -(u / 2)/* B3 */;
274         // t negative: u was odd, v may be even (t replaces v)
275         // t positive: u was even, v is odd (t replaces u)
276         do {
277             /* assert u<0 && v<0; */
278             // B4/B3: cast out twos from t.
279             while ((t & 1) == 0) { // while t is even..
280                 t /= 2; // cast out twos
281             }
282             // B5 [reset max(u,v)]
283             if (t > 0) {
284                 u = -t;
285             } else {
286                 v = t;
287             }
288             // B6/B3. at this point both u and v should be odd.
289             t = (v - u) / 2;
290             // |u| larger: t positive (replace u)
291             // |v| larger: t negative (replace v)
292         } while (t != 0);
293         return -u * (1L << k); // gcd is u*2^k
294     }

296     /**
297      * <p>
298      * Returns the least common multiple of the absolute value of two numbers,
299      * using the formula {@code lcm(a,b) = (a / gcd(a,b)) * b}.
300      * </p>
301      * Special cases:
302      * <ul>
303      * <li>The invocations {@code lcm(Integer.MIN_VALUE, n)} and
304      * {@code lcm(n, Integer.MIN_VALUE)}, where {@code abs(n)} is a
305      * power of 2, throw an {@code ArithmeticException}, because the result
306      * would be 2^31, which is too large for an int value.</li>
307      * <li>The result of {@code lcm(0, x)} and {@code lcm(x, 0)} is
308      * {@code 0} for any {@code x}.
309      * </ul>
310      *
311      * @param a Number.
312      * @param b Number.
313      * @return the least common multiple, never negative.
314      * @throws MathArithmeticException if the result cannot be represented as
315      * a non-negative {@code int} value.
316      * @since 1.1
317      */
318     public static int lcm(int a, int b) throws MathArithmeticException {
319         if (a == 0 || b == 0){
320             return 0;
321         }
322         int lcm = FastMath.abs(ArithmeticUtils.mulAndCheck(a / gcd(a, b), b));
323         if (lcm == Integer.MIN_VALUE) {
324             throw new MathArithmeticException(LocalizedFormats.LCM_OVERFLOW_32_BITS,
325                                               a, b);
326         }
327         return lcm;
328     }

343      * </ul>

382     /**
383      * Multiply two long integers, checking for overflow.
384      *
385      * @param a Factor.
386      * @param b Factor.
387      * @return the product {@code a * b}.
388      * @throws MathArithmeticException if the result can not be represented
389      * as a {@code long}.
390      * @since 1.2
391      */
392     public static long mulAndCheck(long a, long b) throws MathArithmeticException {
393         long ret;
394         if (a > b) {
395             // use symmetry to reduce boundary cases
396             ret = mulAndCheck(b, a);
397         } else {
398             if (a < 0) {
399                 if (b < 0) {
400                     // check for positive overflow with negative a, negative b
401                     if (a >= Long.MAX_VALUE / b) {
402                         ret = a * b;
403                     } else {
404                         throw new MathArithmeticException();
405                     }
406                 } else if (b > 0) {
407                     // check for negative overflow with negative a, positive b
408                     if (Long.MIN_VALUE / b <= a) {
409                         ret = a * b;
410                     } else {
411                         throw new MathArithmeticException();
412 
413                     }
414                 } else {
415                     // assert b == 0
416                     ret = 0;
417                 }
418             } else if (a > 0) {
419                 // assert a > 0
420                 // assert b > 0
421 
422                 // check for positive overflow with positive a, positive b
423                 if (a <= Long.MAX_VALUE / b) {
424                     ret = a * b;
425                 } else {
426                     throw new MathArithmeticException();
427                 }
428             } else {
429                 // assert a == 0
430                 ret = 0;
431             }
432         }
433         return ret;
434     }

454     /**
455      * Subtract two long integers, checking for overflow.
456      *
457      * @param a Value.
458      * @param b Value.
459      * @return the difference {@code a - b}.
460      * @throws MathArithmeticException if the result can not be represented as a
461      * {@code long}.
462      * @since 1.2
463      */
464     public static long subAndCheck(long a, long b) throws MathArithmeticException {
465         long ret;
466         if (b == Long.MIN_VALUE) {
467             if (a < 0) {
468                 ret = a - b;
469             } else {
470                 throw new MathArithmeticException(LocalizedFormats.OVERFLOW_IN_ADDITION, a, -b);
471             }
472         } else {
473             // use additive inverse
474             ret = addAndCheck(a, -b, LocalizedFormats.OVERFLOW_IN_ADDITION);
475         }
476         return ret;
477     }

479     /**
480      * Raise an int to an int power.
481      *
482      * @param k Number to raise.
483      * @param e Exponent (must be positive or zero).
484      * @return \( k^e \)
485      * @throws NotPositiveException if {@code e < 0}.
486      * @throws MathArithmeticException if the result would overflow.
487      */
488     public static int pow(final int k,
489                           final int e)
490         throws NotPositiveException,
491                MathArithmeticException {
492         if (e < 0) {
493             throw new NotPositiveException(LocalizedFormats.EXPONENT, e);
494         }
495 
496         try {
497             int exp = e;
498             int result = 1;
499             int k2p    = k;
500             while (true) {
501                 if ((exp & 0x1) != 0) {
502                     result = mulAndCheck(result, k2p);
503                 }
504 
505                 exp >>= 1;
506                 if (exp == 0) {
507                     break;
508                 }
509 
510                 k2p = mulAndCheck(k2p, k2p);
511             }
512 
513             return result;
514         } catch (MathArithmeticException mae) {
515             // Add context information.
516             mae.getContext().addMessage(LocalizedFormats.OVERFLOW);
517             mae.getContext().addMessage(LocalizedFormats.BASE, k);
518             mae.getContext().addMessage(LocalizedFormats.EXPONENT, e);
519 
520             // Rethrow.
521             throw mae;
522         }
523     }

587     /**
588      * Raise a BigInteger to a long power.
589      *
590      * @param k Number to raise.
591      * @param e Exponent (must be positive or zero).
592      * @return k<sup>e</sup>
593      * @throws NotPositiveException if {@code e < 0}.
594      */
595     public static BigInteger pow(final BigInteger k, long e) throws NotPositiveException {
596         if (e < 0) {
597             throw new NotPositiveException(LocalizedFormats.EXPONENT, e);
598         }
599 
600         BigInteger result = BigInteger.ONE;
601         BigInteger k2p    = k;
602         while (e != 0) {
603             if ((e & 0x1) != 0) {
604                 result = result.multiply(k2p);
605             }
606             k2p = k2p.multiply(k2p);
607             e >>= 1;
608         }
609 
610         return result;
611 
612     }

614     /**
615      * Raise a BigInteger to a BigInteger power.
616      *
617      * @param k Number to raise.
618      * @param e Exponent (must be positive or zero).
619      * @return k<sup>e</sup>
620      * @throws NotPositiveException if {@code e < 0}.
621      */
622     public static BigInteger pow(final BigInteger k, BigInteger e) throws NotPositiveException {
623         if (e.compareTo(BigInteger.ZERO) < 0) {
624             throw new NotPositiveException(LocalizedFormats.EXPONENT, e);
625         }
626 
627         BigInteger result = BigInteger.ONE;
628         BigInteger k2p    = k;
629         while (!BigInteger.ZERO.equals(e)) {
630             if (e.testBit(0)) {
631                 result = result.multiply(k2p);
632             }
633             k2p = k2p.multiply(k2p);
634             e = e.shiftRight(1);
635         }
636 
637         return result;
638     }

640     /**
641      * Add two long integers, checking for overflow.
642      *
643      * @param a Addend.
644      * @param b Addend.
645      * @param pattern Pattern to use for any thrown exception.
646      * @return the sum {@code a + b}.
647      * @throws MathArithmeticException if the result cannot be represented
648      * as a {@code long}.
649      * @since 1.2
650      */
651      private static long addAndCheck(long a, long b, Localizable pattern) throws MathArithmeticException {
652          final long result = a + b;
653          if (!((a ^ b) < 0 || (a ^ result) >= 0)) {
654              throw new MathArithmeticException(pattern, a, b);
655          }
656          return result;
657     }

669     /**
670      * Returns the unsigned remainder from dividing the first argument
671      * by the second where each argument and the result is interpreted
672      * as an unsigned value.
673      * <p>This method does not use the {@code long} datatype.</p>
674      *
675      * @param dividend the value to be divided
676      * @param divisor the value doing the dividing
677      * @return the unsigned remainder of the first argument divided by
678      * the second argument.
679      *
680      * @since 4.0
681      */
682     public static int remainderUnsigned(int dividend, int divisor) {
683         if (divisor >= 0) {
684             if (dividend >= 0) {
685                 return dividend % divisor;
686             }
687             // The implementation is a Java port of algorithm described in the book
688             // "Hacker's Delight" (section "Unsigned short division from signed division").
689             int q = ((dividend >>> 1) / divisor) << 1;
690             dividend -= q * divisor;
691             if (dividend < 0 || dividend >= divisor) {
692                 dividend -= divisor;
693             }
694             return dividend;
695         }
696         return dividend >= 0 || dividend < divisor ? dividend : dividend - divisor;
697     }

746      */
747     public static int divideUnsigned(int dividend, int divisor) {

750                 return dividend / divisor;
751             }

754             int q = ((dividend >>> 1) / divisor) << 1;
