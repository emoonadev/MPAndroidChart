18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.geometry.euclidean.threed.Rotation;
21  import org.apache.commons.math4.geometry.euclidean.threed.Vector3D;
22  import org.apache.commons.math4.geometry.partitioning.Embedding;
23  import org.apache.commons.math4.geometry.partitioning.Hyperplane;
24  import org.apache.commons.math4.geometry.partitioning.SubHyperplane;
25  import org.apache.commons.math4.geometry.partitioning.Transform;
26  import org.apache.commons.math4.geometry.spherical.oned.Arc;
27  import org.apache.commons.math4.geometry.spherical.oned.ArcsSet;
28  import org.apache.commons.math4.geometry.spherical.oned.S1Point;
29  import org.apache.commons.math4.geometry.spherical.oned.Sphere1D;
30  import org.apache.commons.math4.util.FastMath;

32  /** This class represents an oriented great circle on the 2-sphere.
33  
34   * <p>An oriented circle can be defined by a center point. The circle
35   * is the the set of points that are in the normal plan the center.</p>
36  
37   * <p>Since it is oriented the two spherical caps at its two sides are
38   * unambiguously identified as a left cap and a right cap. This can be
39   * used to identify the interior and the exterior in a simple way by
40   * local properties only when part of a line is used to define part of
41   * a spherical polygon boundary.</p>
42  
43   * @since 3.3
44   */
45  public class Circle implements Hyperplane<Sphere2D>, Embedding<Sphere2D, Sphere1D> {
46  
47      /** Pole or circle center. */
48      private Vector3D pole;
49  
50      /** First axis in the equator plane, origin of the phase angles. */
51      private Vector3D x;
52  
53      /** Second axis in the equator plane, in quadrature with respect to x. */
54      private Vector3D y;
55  
56      /** Tolerance below which close sub-arcs are merged together. */
57      private final double tolerance;
58  
59      /** Build a great circle from its pole.
60       * <p>The circle is oriented in the trigonometric direction around pole.</p>
61       * @param pole circle pole
62       * @param tolerance tolerance below which close sub-arcs are merged together
63       */
64      public Circle(final Vector3D pole, final double tolerance) {
65          reset(pole);
66          this.tolerance = tolerance;
67      }
68  
69      /** Build a great circle from two non-aligned points.
70       * <p>The circle is oriented from first to second point using the path smaller than \( \pi \).</p>
71       * @param first first point contained in the great circle
72       * @param second second point contained in the great circle
73       * @param tolerance tolerance below which close sub-arcs are merged together
74       */
75      public Circle(final S2Point first, final S2Point second, final double tolerance) {
76          reset(first.getVector().crossProduct(second.getVector()));
77          this.tolerance = tolerance;
78      }
79  
80      /** Build a circle from its internal components.
81       * <p>The circle is oriented in the trigonometric direction around center.</p>
82       * @param pole circle pole
83       * @param x first axis in the equator plane
84       * @param y second axis in the equator plane
85       * @param tolerance tolerance below which close sub-arcs are merged together
86       */
87      private Circle(final Vector3D pole, final Vector3D x, final Vector3D y,
88                     final double tolerance) {
89          this.pole      = pole;
90          this.x         = x;
91          this.y         = y;
92          this.tolerance = tolerance;
93      }
94  
95      /** Copy constructor.
96       * <p>The created instance is completely independent from the
97       * original instance, it is a deep copy.</p>
98       * @param circle circle to copy
99       */
100     public Circle(final Circle circle) {
101         this(circle.pole, circle.x, circle.y, circle.tolerance);
102     }
103 
104     /** {@inheritDoc} */
105     @Override
106     public Circle copySelf() {
107         return new Circle(this);
108     }
109 
110     /** Reset the instance as if built from a pole.
111      * <p>The circle is oriented in the trigonometric direction around pole.</p>
112      * @param newPole circle pole
113      */
114     public void reset(final Vector3D newPole) {
115         this.pole = newPole.normalize();
116         this.x    = newPole.orthogonal();
117         this.y    = Vector3D.crossProduct(newPole, x).normalize();
118     }
119 
120     /** Revert the instance.
121      */
122     public void revertSelf() {
123         // x remains the same
124         y    = y.negate();
125         pole = pole.negate();
126     }
127 
128     /** Get the reverse of the instance.
129      * <p>Get a circle with reversed orientation with respect to the
130      * instance. A new object is built, the instance is untouched.</p>
131      * @return a new circle, with orientation opposite to the instance orientation
132      */
133     public Circle getReverse() {
134         return new Circle(pole.negate(), x, y.negate(), tolerance);
135     }
136 
137     /** {@inheritDoc} */
138     @Override
139     public Point<Sphere2D> project(Point<Sphere2D> point) {
140         return toSpace(toSubSpace(point));
141     }
142 
143     /** {@inheritDoc} */
144     @Override
145     public double getTolerance() {
146         return tolerance;
147     }
148 
149     /** {@inheritDoc}
150      * @see #getPhase(Vector3D)
151      */
152     @Override
153     public S1Point toSubSpace(final Point<Sphere2D> point) {
154         return new S1Point(getPhase(((S2Point) point).getVector()));
155     }
156 
157     /** Get the phase angle of a direction.
158      * <p>
159      * The direction may not belong to the circle as the
160      * phase is computed for the meridian plane between the circle
161      * pole and the direction.
162      * </p>
163      * @param direction direction for which phase is requested
164      * @return phase angle of the direction around the circle
165      * @see #toSubSpace(Point)
166      */
167     public double getPhase(final Vector3D direction) {
168         return FastMath.PI + FastMath.atan2(-direction.dotProduct(y), -direction.dotProduct(x));
169     }
170 
171     /** {@inheritDoc}
172      * @see #getPointAt(double)
173      */
174     @Override
175     public S2Point toSpace(final Point<Sphere1D> point) {
176         return new S2Point(getPointAt(((S1Point) point).getAlpha()));
177     }
178 
179     /** Get a circle point from its phase around the circle.
180      * @param alpha phase around the circle
181      * @return circle point on the sphere
182      * @see #toSpace(Point)
183      * @see #getXAxis()
184      * @see #getYAxis()
185      */
186     public Vector3D getPointAt(final double alpha) {
187         return new Vector3D(FastMath.cos(alpha), x, FastMath.sin(alpha), y);
188     }
189 
190     /** Get the X axis of the circle.
191      * <p>
192      * This method returns the same value as {@link #getPointAt(double)
193      * getPointAt(0.0)} but it does not do any computation and always
194      * return the same instance.
195      * </p>
196      * @return an arbitrary x axis on the circle
197      * @see #getPointAt(double)
198      * @see #getYAxis()
199      * @see #getPole()
200      */
201     public Vector3D getXAxis() {
202         return x;
203     }
204 
205     /** Get the Y axis of the circle.
206      * <p>
207      * This method returns the same value as {@link #getPointAt(double)
208      * getPointAt(0.5 * FastMath.PI)} but it does not do any computation and always
209      * return the same instance.
210      * </p>
211      * @return an arbitrary y axis point on the circle
212      * @see #getPointAt(double)
213      * @see #getXAxis()
214      * @see #getPole()
215      */
216     public Vector3D getYAxis() {
217         return y;
218     }
219 
220     /** Get the pole of the circle.
221      * <p>
222      * As the circle is a great circle, the pole does <em>not</em>
223      * belong to it.
224      * </p>
225      * @return pole of the circle
226      * @see #getXAxis()
227      * @see #getYAxis()
228      */
229     public Vector3D getPole() {
230         return pole;
231     }
232 
233     /** Get the arc of the instance that lies inside the other circle.
234      * @param other other circle
235      * @return arc of the instance that lies inside the other circle
236      */
237     public Arc getInsideArc(final Circle other) {
238         final double alpha  = getPhase(other.pole);
239         final double halfPi = 0.5 * FastMath.PI;
240         return new Arc(alpha - halfPi, alpha + halfPi, tolerance);
241     }
242 
243     /** {@inheritDoc} */
244     @Override
245     public SubCircle wholeHyperplane() {
246         return new SubCircle(this, new ArcsSet(tolerance));
247     }
248 
249     /** Build a region covering the whole space.
250      * @return a region containing the instance (really a {@link
251      * SphericalPolygonsSet SphericalPolygonsSet} instance)
252      */
253     @Override
254     public SphericalPolygonsSet wholeSpace() {
255         return new SphericalPolygonsSet(tolerance);
256     }
257 
258     /** {@inheritDoc}
259      * @see #getOffset(Vector3D)
260      */
261     @Override
262     public double getOffset(final Point<Sphere2D> point) {
263         return getOffset(((S2Point) point).getVector());
264     }
265 
266     /** Get the offset (oriented distance) of a direction.
267      * <p>The offset is defined as the angular distance between the
268      * circle center and the direction minus the circle radius. It
269      * is therefore 0 on the circle, positive for directions outside of
270      * the cone delimited by the circle, and negative inside the cone.</p>
271      * @param direction direction to check
272      * @return offset of the direction
273      * @see #getOffset(Point)
274      */
275     public double getOffset(final Vector3D direction) {
276         return Vector3D.angle(pole, direction) - 0.5 * FastMath.PI;
277     }
278 
279     /** {@inheritDoc} */
280     @Override
281     public boolean sameOrientationAs(final Hyperplane<Sphere2D> other) {
282         final Circle otherC = (Circle) other;
283         return Vector3D.dotProduct(pole, otherC.pole) >= 0.0;
284     }
285 
286     /** Get a {@link org.apache.commons.math4.geometry.partitioning.Transform
287      * Transform} embedding a 3D rotation.
288      * @param rotation rotation to use
289      * @return a new transform that can be applied to either {@link
290      * Point Point}, {@link Circle Line} or {@link
291      * org.apache.commons.math4.geometry.partitioning.SubHyperplane
292      * SubHyperplane} instances
293      */
294     public static Transform<Sphere2D, Sphere1D> getTransform(final Rotation rotation) {
295         return new CircleTransform(rotation);
296     }
297 
298     /** Class embedding a 3D rotation. */
299     private static class CircleTransform implements Transform<Sphere2D, Sphere1D> {
300 
301         /** Underlying rotation. */
302         private final Rotation rotation;
303 
304         /** Build a transform from a {@code Rotation}.
305          * @param rotation rotation to use
306          */
307         CircleTransform(final Rotation rotation) {
308             this.rotation = rotation;
309         }
310 
311         /** {@inheritDoc} */
312         @Override
313         public S2Point apply(final Point<Sphere2D> point) {
314             return new S2Point(rotation.applyTo(((S2Point) point).getVector()));
315         }
316 
317         /** {@inheritDoc} */
318         @Override
319         public Circle apply(final Hyperplane<Sphere2D> hyperplane) {
320             final Circle circle = (Circle) hyperplane;
321             return new Circle(rotation.applyTo(circle.pole),
322                               rotation.applyTo(circle.x),
323                               rotation.applyTo(circle.y),
324                               circle.tolerance);
325         }
326 
327         /** {@inheritDoc} */
328         @Override
329         public SubHyperplane<Sphere1D> apply(final SubHyperplane<Sphere1D> sub,
330                                              final Hyperplane<Sphere2D> original,
331                                              final Hyperplane<Sphere2D> transformed) {
332             // as the circle is rotated, the limit angles are rotated too
333             return sub;
334         }
335 
336     }
