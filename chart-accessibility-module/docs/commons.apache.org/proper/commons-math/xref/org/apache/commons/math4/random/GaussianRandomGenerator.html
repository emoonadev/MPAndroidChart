18  package org.apache.commons.math4.random;
19  
20  import org.apache.commons.rng.UniformRandomProvider;
21  import org.apache.commons.math4.distribution.RealDistribution;
22  import org.apache.commons.math4.distribution.NormalDistribution;
23  
24  /**
25   * Random generator that generates normally distributed samples.
26   *
27   * @since 1.2
28   */
29  public class GaussianRandomGenerator implements NormalizedRandomGenerator {
30      /** Gaussian distribution sampler. */
31      private final RealDistribution.Sampler sampler;
32  
33      /**
34       * Creates a new generator.
35       *
36       * @param generator Underlying random generator.
37       */
38      public GaussianRandomGenerator(final UniformRandomProvider generator) {
39          sampler = new NormalDistribution().createSampler(generator);
40      }
41  
42      /**
43       * Generates a random scalar with zero mean and unit standard deviation.
44       *
45       * @return a random value sampled from a normal distribution.
46       */
47      @Override
48      public double nextNormalizedDouble() {
49          return sampler.sample();
50      }
51  }
