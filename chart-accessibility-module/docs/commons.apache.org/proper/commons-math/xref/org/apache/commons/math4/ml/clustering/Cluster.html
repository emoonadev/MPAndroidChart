18  package org.apache.commons.math4.ml.clustering;
19  
20  import java.io.Serializable;
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  /**
25   * Cluster holding a set of {@link Clusterable} points.
26   * @param <T> the type of points that can be clustered
27   * @since 3.2
28   */
29  public class Cluster<T extends Clusterable> implements Serializable {
30  
31      /** Serializable version identifier. */
32      private static final long serialVersionUID = -3442297081515880464L;
33  
34      /** The points contained in this cluster. */
35      private final List<T> points;
36  
37      /**
38       * Build a cluster centered at a specified point.
39       */
40      public Cluster() {
41          points = new ArrayList<T>();
42      }
43  
44      /**
45       * Add a point to this cluster.
46       * @param point point to add
47       */
48      public void addPoint(final T point) {
49          points.add(point);
50      }
51  
52      /**
53       * Get the points contained in the cluster.
54       * @return points contained in the cluster
55       */
56      public List<T> getPoints() {
57          return points;
58      }
59  
60  }
