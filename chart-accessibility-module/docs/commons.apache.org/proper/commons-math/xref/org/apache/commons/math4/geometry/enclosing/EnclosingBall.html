17  package org.apache.commons.math4.geometry.enclosing;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.geometry.Point;
22  import org.apache.commons.math4.geometry.Space;
23  
24  /** This class represents a ball enclosing some points.
25   * @param <S> Space type.
26   * @param <P> Point type.
27   * @see Space
28   * @see Point
29   * @see Encloser
30   * @since 3.3
31   */
32  public class EnclosingBall<S extends Space, P extends Point<S>> implements Serializable {
33  
34      /** Serializable UID. */
35      private static final long serialVersionUID = 20140126L;
36  
37      /** Center of the ball. */
38      private final P center;
39  
40      /** Radius of the ball. */
41      private final double radius;
42  
43      /** Support points used to define the ball. */
44      private final P[] support;
45  
46      /** Simple constructor.
47       * @param center center of the ball
48       * @param radius radius of the ball
49       * @param support support points used to define the ball
50       */
51      @SafeVarargs
52      public EnclosingBall(final P center, final double radius, final P ... support) {
53          this.center  = center;
54          this.radius  = radius;
55          this.support = support.clone();
56      }
57  
58      /** Get the center of the ball.
59       * @return center of the ball
60       */
61      public P getCenter() {
62          return center;
63      }
64  
65      /** Get the radius of the ball.
66       * @return radius of the ball (can be negative if the ball is empty)
67       */
68      public double getRadius() {
69          return radius;
70      }
71  
72      /** Get the support points used to define the ball.
73       * @return support points used to define the ball
74       */
75      public P[] getSupport() {
76          return support.clone();
77      }
78  
79      /** Get the number of support points used to define the ball.
80       * @return number of support points used to define the ball
81       */
82      public int getSupportSize() {
83          return support.length;
84      }
85  
86      /** Check if a point is within the ball or at boundary.
87       * @param point point to test
88       * @return true if the point is within the ball or at boundary
89       */
90      public boolean contains(final P point) {
91          return point.distance(center) <= radius;
92      }
93  
94      /** Check if a point is within an enlarged ball or at boundary.
95       * @param point point to test
96       * @param margin margin to consider
97       * @return true if the point is within the ball enlarged
98       * by the margin or at boundary
99       */
100     public boolean contains(final P point, final double margin) {
101         return point.distance(center) <= radius + margin;
102     }
103 
104 }
