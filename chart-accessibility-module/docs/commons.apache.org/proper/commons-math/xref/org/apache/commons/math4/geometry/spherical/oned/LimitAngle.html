17  package org.apache.commons.math4.geometry.spherical.oned;
18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.geometry.partitioning.Hyperplane;
21  
22  /** This class represents a 1D oriented hyperplane on the circle.
23   * <p>An hyperplane on the 1-sphere is an angle with an orientation.</p>
24   * <p>Instances of this class are guaranteed to be immutable.</p>
25   * @since 3.3
26   */
27  public class LimitAngle implements Hyperplane<Sphere1D> {
28  
29      /** Angle location. */
30      private final S1Point location;
31  
32      /** Orientation. */
33      private final boolean direct;
34  
35      /** Tolerance below which angles are considered identical. */
36      private final double tolerance;
37  
38      /** Simple constructor.
39       * @param location location of the hyperplane
40       * @param direct if true, the plus side of the hyperplane is towards
41       * angles greater than {@code location}
42       * @param tolerance tolerance below which angles are considered identical
43       */
44      public LimitAngle(final S1Point location, final boolean direct, final double tolerance) {
45          this.location  = location;
46          this.direct    = direct;
47          this.tolerance = tolerance;
48      }
49  
50      /** Copy the instance.
51       * <p>Since instances are immutable, this method directly returns
52       * the instance.</p>
53       * @return the instance itself
54       */
55      @Override
56      public LimitAngle copySelf() {
57          return this;
58      }
59  
60      /** {@inheritDoc} */
61      @Override
62      public double getOffset(final Point<Sphere1D> point) {
63          final double delta = ((S1Point) point).getAlpha() - location.getAlpha();
64          return direct ? delta : -delta;
65      }
66  
67      /** Check if the hyperplane orientation is direct.
68       * @return true if the plus side of the hyperplane is towards
69       * angles greater than hyperplane location
70       */
71      public boolean isDirect() {
72          return direct;
73      }
74  
75      /** Get the reverse of the instance.
76       * <p>Get a limit angle with reversed orientation with respect to the
77       * instance. A new object is built, the instance is untouched.</p>
78       * @return a new limit angle, with orientation opposite to the instance orientation
79       */
80      public LimitAngle getReverse() {
81          return new LimitAngle(location, !direct, tolerance);
82      }
83  
84      /** Build a region covering the whole hyperplane.
85       * <p>Since this class represent zero dimension spaces which does
86       * not have lower dimension sub-spaces, this method returns a dummy
87       * implementation of a {@link
88       * org.apache.commons.math4.geometry.partitioning.SubHyperplane SubHyperplane}.
89       * This implementation is only used to allow the {@link
90       * org.apache.commons.math4.geometry.partitioning.SubHyperplane
91       * SubHyperplane} class implementation to work properly, it should
92       * <em>not</em> be used otherwise.</p>
93       * @return a dummy sub hyperplane
94       */
95      @Override
96      public SubLimitAngle wholeHyperplane() {
97          return new SubLimitAngle(this, null);
98      }
99  
100     /** Build a region covering the whole space.
101      * @return a region containing the instance (really an {@link
102      * ArcsSet IntervalsSet} instance)
103      */
104     @Override
105     public ArcsSet wholeSpace() {
106         return new ArcsSet(tolerance);
107     }
108 
109     /** {@inheritDoc} */
110     @Override
111     public boolean sameOrientationAs(final Hyperplane<Sphere1D> other) {
112         return !(direct ^ ((LimitAngle) other).direct);
113     }
114 
115     /** Get the hyperplane location on the circle.
116      * @return the hyperplane location
117      */
118     public S1Point getLocation() {
119         return location;
120     }
121 
122     /** {@inheritDoc} */
123     @Override
124     public Point<Sphere1D> project(Point<Sphere1D> point) {
125         return location;
126     }
127 
128     /** {@inheritDoc} */
129     @Override
130     public double getTolerance() {
131         return tolerance;
132     }
133 
134 }
