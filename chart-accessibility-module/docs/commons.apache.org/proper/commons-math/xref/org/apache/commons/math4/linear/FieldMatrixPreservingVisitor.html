18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.FieldElement;
21  
22  /**
23   * Interface defining a visitor for matrix entries.
24   *
25   * @param <T> the type of the field elements
26   * @since 2.0
27   */
28  public interface FieldMatrixPreservingVisitor<T extends FieldElement<?>> {
29      /**
30       * Start visiting a matrix.
31       * <p>This method is called once before any entry of the matrix is visited.</p>
32       * @param rows number of rows of the matrix
33       * @param columns number of columns of the matrix
34       * @param startRow Initial row index
35       * @param endRow Final row index (inclusive)
36       * @param startColumn Initial column index
37       * @param endColumn Final column index (inclusive)
38       */
39      void start(int rows, int columns,
40                 int startRow, int endRow, int startColumn, int endColumn);
41  
42      /**
43       * Visit one matrix entry.
44       * @param row row index of the entry
45       * @param column column index of the entry
46       * @param value current value of the entry
47       */
48      void visit(int row, int column, T value);
49  
50      /**
51       * End visiting a matrix.
52       * <p>This method is called once after all entries of the matrix have been visited.</p>
53       * @return the value that the <code>walkInXxxOrder</code> must return
54       */
55      T end();
56  }
