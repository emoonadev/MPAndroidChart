18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.exception.MathIllegalArgumentException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  
23  /**
24   * Exception to be thrown when a symmetric, definite positive
25   * {@link RealLinearOperator} is expected.
26   * Since the coefficients of the matrix are not accessible, the most
27   * general definition is used to check that {@code A} is not positive
28   * definite, i.e.  there exists {@code x} such that {@code x' A x <= 0}.
29   * In the terminology of this exception, {@code A} is the "offending"
30   * linear operator and {@code x} the "offending" vector.
31   *
32   * @since 3.0
33   */
34  public class NonPositiveDefiniteOperatorException
35      extends MathIllegalArgumentException {
36      /** Serializable version Id. */
37      private static final long serialVersionUID = 917034489420549847L;
38  
39      /** Creates a new instance of this class. */
40      public NonPositiveDefiniteOperatorException() {
41          super(LocalizedFormats.NON_POSITIVE_DEFINITE_OPERATOR);
42      }
43  }
