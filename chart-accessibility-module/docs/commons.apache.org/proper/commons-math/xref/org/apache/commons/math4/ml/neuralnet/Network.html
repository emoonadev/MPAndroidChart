38  /**
39   * Neural network, composed of {@link Neuron} instances and the links
40   * between them.
41   *
42   * Although updating a neuron's state is thread-safe, modifying the
43   * network's topology (adding or removing links) is not.
44   *
45   * @since 3.3
46   */
47  public class Network
48      implements Iterable<Neuron>,
49                 Serializable {
50      /** Serializable. */
51      private static final long serialVersionUID = 20130207L;
52      /** Neurons. */
53      private final ConcurrentHashMap<Long, Neuron> neuronMap
54          = new ConcurrentHashMap<Long, Neuron>();
55      /** Next available neuron identifier. */
56      private final AtomicLong nextId;
57      /** Neuron's features set size. */
58      private final int featureSize;
59      /** Links. */
60      private final ConcurrentHashMap<Long, Set<Long>> linkMap
61          = new ConcurrentHashMap<Long, Set<Long>>();
62  
63      /**
64       * Comparator that prescribes an order of the neurons according
65       * to the increasing order of their identifier.
66       */
67      public static class NeuronIdentifierComparator
68          implements Comparator<Neuron>,
69                     Serializable {
70          /** Version identifier. */
71          private static final long serialVersionUID = 20130207L;
72  
73          /** {@inheritDoc} */
74          @Override
75          public int compare(Neuron a,
76                             Neuron b) {
77              final long aId = a.getIdentifier();
78              final long bId = b.getIdentifier();
79              return aId < bId ? -1 :
80                  aId > bId ? 1 : 0;
81          }
82      }

84      /**
85       * Constructor with restricted access, solely used for deserialization.
86       *
87       * @param nextId Next available identifier.
88       * @param featureSize Number of features.
89       * @param neuronList Neurons.
90       * @param neighbourIdList Links associated to each of the neurons in
91       * {@code neuronList}.
92       * @throws MathIllegalStateException if an inconsistency is detected
93       * (which probably means that the serialized form has been corrupted).
94       */
95      Network(long nextId,
96              int featureSize,
97              Neuron[] neuronList,
98              long[][] neighbourIdList) {
99          final int numNeurons = neuronList.length;
100         if (numNeurons != neighbourIdList.length) {
101             throw new MathIllegalStateException();
102         }
103 
104         for (int i = 0; i < numNeurons; i++) {
105             final Neuron n = neuronList[i];
106             final long id = n.getIdentifier();
107             if (id >= nextId) {
108                 throw new MathIllegalStateException();
109             }
110             neuronMap.put(id, n);
111             linkMap.put(id, new HashSet<Long>());
112         }
113 
114         for (int i = 0; i < numNeurons; i++) {
115             final long aId = neuronList[i].getIdentifier();
116             final Set<Long> aLinks = linkMap.get(aId);
117             for (Long bId : neighbourIdList[i]) {
118                 if (neuronMap.get(bId) == null) {
119                     throw new MathIllegalStateException();
120                 }
121                 addLinkToLinkSet(aLinks, bId);
122             }
123         }
124 
125         this.nextId = new AtomicLong(nextId);
126         this.featureSize = featureSize;
127     }
128 
129     /**
130      * @param initialIdentifier Identifier for the first neuron that
131      * will be added to this network.
132      * @param featureSize Size of the neuron's features.
133      */
134     public Network(long initialIdentifier,
135                    int featureSize) {
136         nextId = new AtomicLong(initialIdentifier);
137         this.featureSize = featureSize;
138     }

140     /**
141      * Performs a deep copy of this instance.
142      * Upon return, the copied and original instances will be independent:
143      * Updating one will not affect the other.
144      *
145      * @return a new instance with the same state as this instance.
146      * @since 3.6
147      */
148     public synchronized Network copy() {
149         final Network copy = new Network(nextId.get(),
150                                          featureSize);
151 
152 
153         for (Map.Entry<Long, Neuron> e : neuronMap.entrySet()) {
154             copy.neuronMap.put(e.getKey(), e.getValue().copy());
155         }
156 
157         for (Map.Entry<Long, Set<Long>> e : linkMap.entrySet()) {
158             copy.linkMap.put(e.getKey(), new HashSet<Long>(e.getValue()));
159         }
160 
161         return copy;
162     }

166      */
167     @Override
168     public Iterator<Neuron> iterator() {
169         return neuronMap.values().iterator();

172     /**
173      * Creates a list of the neurons, sorted in a custom order.
174      *
175      * @param comparator {@link Comparator} used for sorting the neurons.
176      * @return a list of neurons, sorted in the order prescribed by the
177      * given {@code comparator}.
178      * @see NeuronIdentifierComparator
179      */
180     public Collection<Neuron> getNeurons(Comparator<Neuron> comparator) {
181         final List<Neuron> neurons = new ArrayList<Neuron>();
182         neurons.addAll(neuronMap.values());
183 
184         Collections.sort(neurons, comparator);
185 
186         return neurons;
187     }

189     /**
190      * Creates a neuron and assigns it a unique identifier.
191      *
192      * @param features Initial values for the neuron's features.
193      * @return the neuron's identifier.
194      * @throws DimensionMismatchException if the length of {@code features}
195      * is different from the expected size (as set by the
196      * {@link #Network(long,int) constructor}).
197      */
198     public long createNeuron(double[] features) {
199         if (features.length != featureSize) {
200             throw new DimensionMismatchException(features.length, featureSize);
201         }
202 
203         final long id = createNextId();
204         neuronMap.put(id, new Neuron(id, features));
205         linkMap.put(id, new HashSet<Long>());
206         return id;
207     }
208 
209     /**
210      * Deletes a neuron.
211      * Links from all neighbours to the removed neuron will also be
212      * {@link #deleteLink(Neuron,Neuron) deleted}.
213      *
214      * @param neuron Neuron to be removed from this network.
215      * @throws NoSuchElementException if {@code n} does not belong to
216      * this network.
217      */
218     public void deleteNeuron(Neuron neuron) {
219         final Collection<Neuron> neighbours = getNeighbours(neuron);
220 
221         // Delete links to from neighbours.
222         for (Neuron n : neighbours) {
223             deleteLink(n, neuron);
224         }
225 
226         // Remove neuron.
227         neuronMap.remove(neuron.getIdentifier());
228     }

239     /**
240      * Adds a link from neuron {@code a} to neuron {@code b}.
241      * Note: the link is not bi-directional; if a bi-directional link is
242      * required, an additional call must be made with {@code a} and
243      * {@code b} exchanged in the argument list.
244      *
245      * @param a Neuron.
246      * @param b Neuron.
247      * @throws NoSuchElementException if the neurons do not exist in the
248      * network.
249      */
250     public void addLink(Neuron a,
251                         Neuron b) {
252         final long aId = a.getIdentifier();
253         final long bId = b.getIdentifier();
254 
255         // Check that the neurons belong to this network.
256         if (a != getNeuron(aId)) {
257             throw new NoSuchElementException(Long.toString(aId));
258         }
259         if (b != getNeuron(bId)) {
260             throw new NoSuchElementException(Long.toString(bId));
261         }
262 
263         // Add link from "a" to "b".
264         addLinkToLinkSet(linkMap.get(aId), bId);
265     }
266 
267     /**
268      * Adds a link to neuron {@code id} in given {@code linkSet}.
269      * Note: no check verifies that the identifier indeed belongs
270      * to this network.
271      *
272      * @param linkSet Neuron identifier.
273      * @param id Neuron identifier.
274      */
275     private void addLinkToLinkSet(Set<Long> linkSet,
276                                   long id) {
277         linkSet.add(id);
278     }

280     /**
281      * Deletes the link between neurons {@code a} and {@code b}.
282      *
283      * @param a Neuron.
284      * @param b Neuron.
285      * @throws NoSuchElementException if the neurons do not exist in the
286      * network.
287      */
288     public void deleteLink(Neuron a,
289                            Neuron b) {
290         final long aId = a.getIdentifier();
291         final long bId = b.getIdentifier();
292 
293         // Check that the neurons belong to this network.
294         if (a != getNeuron(aId)) {
295             throw new NoSuchElementException(Long.toString(aId));
296         }
297         if (b != getNeuron(bId)) {
298             throw new NoSuchElementException(Long.toString(bId));
299         }
300 
301         // Delete link from "a" to "b".
302         deleteLinkFromLinkSet(linkMap.get(aId), bId);
303     }
304 
305     /**
306      * Deletes a link to neuron {@code id} in given {@code linkSet}.
307      * Note: no check verifies that the identifier indeed belongs
308      * to this network.
309      *
310      * @param linkSet Neuron identifier.
311      * @param id Neuron identifier.
312      */
313     private void deleteLinkFromLinkSet(Set<Long> linkSet,
314                                        long id) {
315         linkSet.remove(id);
316     }

318     /**
319      * Retrieves the neuron with the given (unique) {@code id}.
320      *
321      * @param id Identifier.
322      * @return the neuron associated with the given {@code id}.
323      * @throws NoSuchElementException if the neuron does not exist in the
324      * network.
325      */
326     public Neuron getNeuron(long id) {
327         final Neuron n = neuronMap.get(id);
328         if (n == null) {
329             throw new NoSuchElementException(Long.toString(id));
330         }
331         return n;
332     }

334     /**
335      * Retrieves the neurons in the neighbourhood of any neuron in the
336      * {@code neurons} list.
337      * @param neurons Neurons for which to retrieve the neighbours.
338      * @return the list of neighbours.
339      * @see #getNeighbours(Iterable,Iterable)
340      */
341     public Collection<Neuron> getNeighbours(Iterable<Neuron> neurons) {
342         return getNeighbours(neurons, null);
343     }

345     /**
346      * Retrieves the neurons in the neighbourhood of any neuron in the
347      * {@code neurons} list.
348      * The {@code exclude} list allows to retrieve the "concentric"
349      * neighbourhoods by removing the neurons that belong to the inner
350      * "circles".
351      *
352      * @param neurons Neurons for which to retrieve the neighbours.
353      * @param exclude Neurons to exclude from the returned list.
354      * Can be {@code null}.
355      * @return the list of neighbours.
356      */
357     public Collection<Neuron> getNeighbours(Iterable<Neuron> neurons,
358                                             Iterable<Neuron> exclude) {
359         final Set<Long> idList = new HashSet<Long>();
360 
361         for (Neuron n : neurons) {
362             idList.addAll(linkMap.get(n.getIdentifier()));
363         }
364         if (exclude != null) {
365             for (Neuron n : exclude) {
366                 idList.remove(n.getIdentifier());
367             }
368         }
369 
370         final List<Neuron> neuronList = new ArrayList<Neuron>();
371         for (Long id : idList) {
372             neuronList.add(getNeuron(id));
373         }
374 
375         return neuronList;
376     }

389     /**
390      * Retrieves the neighbours of the given neuron.
391      *
392      * @param neuron Neuron for which to retrieve the neighbours.
393      * @param exclude Neurons to exclude from the returned list.
394      * Can be {@code null}.
395      * @return the list of neighbours.
396      */
397     public Collection<Neuron> getNeighbours(Neuron neuron,
398                                             Iterable<Neuron> exclude) {
399         final Set<Long> idList = linkMap.get(neuron.getIdentifier());
400         if (exclude != null) {
401             for (Neuron n : exclude) {
402                 idList.remove(n.getIdentifier());
403             }
404         }
405 
406         final List<Neuron> neuronList = new ArrayList<Neuron>();
407         for (Long id : idList) {
408             neuronList.add(getNeuron(id));
409         }
410 
411         return neuronList;
412     }

417      * @return a value that will serve as a unique identifier.

420         return nextId.getAndIncrement();

432     /**
433      * Custom serialization.
434      *
435      * @return the proxy instance that will be actually serialized.
436      */
437     private Object writeReplace() {
438         final Neuron[] neuronList = neuronMap.values().toArray(new Neuron[0]);
439         final long[][] neighbourIdList = new long[neuronList.length][];
440 
441         for (int i = 0; i < neuronList.length; i++) {
442             final Collection<Neuron> neighbours = getNeighbours(neuronList[i]);
443             final long[] neighboursId = new long[neighbours.size()];
444             int count = 0;
445             for (Neuron n : neighbours) {
446                 neighboursId[count] = n.getIdentifier();
447                 ++count;
448             }
449             neighbourIdList[i] = neighboursId;
450         }
451 
452         return new SerializationProxy(nextId.get(),
453                                       featureSize,
454                                       neuronList,
455                                       neighbourIdList);
456     }

458     /**
459      * Serialization.
460      */
461     private static class SerializationProxy implements Serializable {
462         /** Serializable. */
463         private static final long serialVersionUID = 20130207L;
464         /** Next identifier. */
465         private final long nextId;
466         /** Number of features. */
467         private final int featureSize;
468         /** Neurons. */
469         private final Neuron[] neuronList;
470         /** Links. */
471         private final long[][] neighbourIdList;
472 
473         /**
474          * @param nextId Next available identifier.
475          * @param featureSize Number of features.
476          * @param neuronList Neurons.
477          * @param neighbourIdList Links associated to each of the neurons in
478          * {@code neuronList}.
479          */
480         SerializationProxy(long nextId,
481                            int featureSize,
482                            Neuron[] neuronList,
483                            long[][] neighbourIdList) {
484             this.nextId = nextId;
485             this.featureSize = featureSize;
486             this.neuronList = neuronList;
487             this.neighbourIdList = neighbourIdList;
488         }
489 
490         /**
491          * Custom serialization.
492          *
493          * @return the {@link Network} for which this instance is the proxy.
494          */
495         private Object readResolve() {
496             return new Network(nextId,
497                                featureSize,
498                                neuronList,
499                                neighbourIdList);
500         }
