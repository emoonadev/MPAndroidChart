17  package org.apache.commons.math4.genetics;
18  
19  import java.util.List;
20  
21  /**
22   * Interface indicating that the chromosome represents a permutation of objects.
23   *
24   * @param <T> type of the permuted objects
25   * @since 2.0
26   */
27  public interface PermutationChromosome<T> {
28  
29      /**
30       * Permutes the <code>sequence</code> of objects of type T according to the
31       * permutation this chromosome represents. For example, if this chromosome
32       * represents a permutation (3,0,1,2), and the unpermuted sequence is
33       * (a,b,c,d), this yields (d,a,b,c).
34       *
35       * @param sequence the unpermuted (original) sequence of objects
36       * @return permutation of <code>sequence</code> represented by this permutation
37       */
38      List<T> decode(List<T> sequence);
39  
40  }
