37      /**
38       * Creates a complex number from the given polar representation.
39       * <p>
40       * If either {@code r} or {@code theta} is NaN, or {@code theta} is
41       * infinite, {@link Complex#NaN} is returned.
42       * <p>
43       * If {@code r} is infinite and {@code theta} is finite, infinite or NaN
44       * values may be returned in parts of the result, following the rules for
45       * double arithmetic.
46       *
47       * <pre>
48       * Examples:
49       * {@code
50       * polar2Complex(INFINITY, \(\pi\)) = INFINITY + INFINITY i
51       * polar2Complex(INFINITY, 0) = INFINITY + NaN i
52       * polar2Complex(INFINITY, \(-\frac{\pi}{4}\)) = INFINITY - INFINITY i
53       * polar2Complex(INFINITY, \(5\frac{\pi}{4}\)) = -INFINITY - INFINITY i }
54       * </pre>
55       *
56       * @param r the modulus of the complex number to create
57       * @param theta the argument of the complex number to create
58       * @return {@code Complex}
59       * @throws MathIllegalArgumentException  if {@code r} is negative.
60       * @since 1.1
61       */
62      public static Complex polar2Complex(double r, double theta) throws MathIllegalArgumentException {
63          if (r < 0) {
64              throw new MathIllegalArgumentException(LocalizedFormats.NEGATIVE_COMPLEX_MODULE, r);
65          }
66          return new Complex(r * FastMath.cos(theta), r * FastMath.sin(theta));
67      }

69      /**
70       * Creates {@code Complex[]} array given {@code double[]} arrays of r and
71       * theta.
72       *
73       * @param r {@code double[]} of moduli
74       * @param theta {@code double[]} of arguments
75       * @return {@code Complex[]}
76       * @throws MathIllegalArgumentException
77       *             if {@code r} is negative.
78       * @since 4.0
79       */
80      public static Complex[] polar2Complex(double[] r, double[] theta) throws MathIllegalArgumentException {
81          final int length = r.length;
82          final Complex[] c = new Complex[length];
83          for (int x = 0; x < length; x++) {
84              if (r[x] < 0) {
85                  throw new MathIllegalArgumentException(LocalizedFormats.NEGATIVE_COMPLEX_MODULE, r[x]);
86              }
87              c[x] = new Complex(r[x] * FastMath.cos(theta[x]), r[x] * FastMath.sin(theta[x]));
88          }
89          return c;
90      }

97       * @param theta {@code double[]} of arguments
