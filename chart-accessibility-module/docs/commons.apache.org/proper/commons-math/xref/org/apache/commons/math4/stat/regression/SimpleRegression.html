29  /**
30   * Estimates an ordinary least squares regression model
31   * with one independent variable.
32   * <p>
33   * <code> y = intercept + slope * x  </code></p>
34   * <p>
35   * Standard errors for <code>intercept</code> and <code>slope</code> are
36   * available as well as ANOVA, r-square and Pearson's r statistics.</p>
37   * <p>
38   * Observations (x,y pairs) can be added to the model one at a time or they
39   * can be provided in a 2-dimensional array.  The observations are not stored
40   * in memory, so there is no limit to the number of observations that can be
41   * added to the model.</p>
42   * <p>
43   * <strong>Usage Notes</strong>: <ul>
44   * <li> When there are fewer than two observations in the model, or when
45   * there is no variation in the x values (i.e. all x values are the same)
46   * all statistics return <code>NaN</code>. At least two observations with
47   * different x coordinates are required to estimate a bivariate regression
48   * model.
49   * </li>
50   * <li> Getters for the statistics always compute values based on the current
51   * set of observations -- i.e., you can get statistics, then add more data
52   * and get updated statistics without using a new instance.  There is no
53   * "compute" method that updates all statistics.  Each of the getters performs
54   * the necessary computations to return the requested statistic.
55   * </li>
56   * <li> The intercept term may be suppressed by passing {@code false} to
57   * the {@link #SimpleRegression(boolean)} constructor.  When the
58   * {@code hasIntercept} property is false, the model is estimated without a
59   * constant term and {@link #getIntercept()} returns {@code 0}.</li>
60   * </ul></p>
61   *
62   */
63  public class SimpleRegression implements Serializable, UpdatingMultipleLinearRegression {

118     /**
119      * Adds the observation (x,y) to the regression data set.
120      * <p>
121      * Uses updating formulas for means and sums of squares defined in
122      * "Algorithms for Computing the Sample Variance: Analysis and
123      * Recommendations", Chan, T.F., Golub, G.H., and LeVeque, R.J.
124      * 1983, American Statistician, vol. 37, pp. 242-247, referenced in
125      * Weisberg, S. "Applied Linear Regression". 2nd Ed. 1985.</p>
126      *
127      *
128      * @param x independent variable value
129      * @param y dependent variable value
130      */
131     public void addData(final double x,final double y) {
132         if (n == 0) {
133             xbar = x;
134             ybar = y;
135         } else {
136             if( hasIntercept ){
137                 final double fact1 = 1.0 + n;
138                 final double fact2 = n / (1.0 + n);
139                 final double dx = x - xbar;
140                 final double dy = y - ybar;
141                 sumXX += dx * dx * fact2;
142                 sumYY += dy * dy * fact2;
143                 sumXY += dx * dy * fact2;
144                 xbar += dx / fact1;
145                 ybar += dy / fact1;
146             }
147          }
148         if( !hasIntercept ){
149             sumXX += x * x ;
150             sumYY += y * y ;
151             sumXY += x * y ;
152         }
153         sumX += x;
154         sumY += y;
155         n++;
156     }

158     /**
159      * Appends data from another regression calculation to this one.
160      *
161      * <p>The mean update formulae are based on a paper written by Philippe
162      * P&eacute;bay:
163      * <a
164      * href="http://prod.sandia.gov/techlib/access-control.cgi/2008/086212.pdf">
165      * Formulas for Robust, One-Pass Parallel Computation of Covariances and
166      * Arbitrary-Order Statistical Moments</a>, 2008, Technical Report
167      * SAND2008-6212, Sandia National Laboratories.</p>
168      *
169      * @param reg model to append data from
170      * @since 3.3
171      */
172     public void append(SimpleRegression reg) {
173         if (n == 0) {
174             xbar = reg.xbar;
175             ybar = reg.ybar;
176             sumXX = reg.sumXX;
177             sumYY = reg.sumYY;
178             sumXY = reg.sumXY;
179         } else {
180             if (hasIntercept) {
181                 final double fact1 = reg.n / (double) (reg.n + n);
182                 final double fact2 = n * reg.n / (double) (reg.n + n);
183                 final double dx = reg.xbar - xbar;
184                 final double dy = reg.ybar - ybar;
185                 sumXX += reg.sumXX + dx * dx * fact2;
186                 sumYY += reg.sumYY + dy * dy * fact2;
187                 sumXY += reg.sumXY + dx * dy * fact2;
188                 xbar += dx * fact1;
189                 ybar += dy * fact1;
190             }else{
191                 sumXX += reg.sumXX;
192                 sumYY += reg.sumYY;
193                 sumXY += reg.sumXY;
194             }
195         }
196         sumX += reg.sumX;
197         sumY += reg.sumY;
198         n += reg.n;
199     }

201     /**
202      * Removes the observation (x,y) from the regression data set.
203      * <p>
204      * Mirrors the addData method.  This method permits the use of
205      * SimpleRegression instances in streaming mode where the regression
206      * is applied to a sliding "window" of observations, however the caller is
207      * responsible for maintaining the set of observations in the window.</p>
208      *
209      * The method has no effect if there are no points of data (i.e. n=0)
210      *
211      * @param x independent variable value
212      * @param y dependent variable value
213      */
214     public void removeData(final double x,final double y) {
215         if (n > 0) {
216             if (hasIntercept) {
217                 final double fact1 = n - 1.0;
218                 final double fact2 = n / (n - 1.0);
219                 final double dx = x - xbar;
220                 final double dy = y - ybar;
221                 sumXX -= dx * dx * fact2;
222                 sumYY -= dy * dy * fact2;
223                 sumXY -= dx * dy * fact2;
224                 xbar -= dx / fact1;
225                 ybar -= dy / fact1;
226             } else {
227                 final double fact1 = n - 1.0;
228                 sumXX -= x * x;
229                 sumYY -= y * y;
230                 sumXY -= x * y;
231                 xbar -= x / fact1;
232                 ybar -= y / fact1;
233             }
234              sumX -= x;
235              sumY -= y;
236              n--;
237         }
238     }

240     /**
241      * Adds the observations represented by the elements in
242      * <code>data</code>.
243      * <p>
244      * <code>(data[0][0],data[0][1])</code> will be the first observation, then
245      * <code>(data[1][0],data[1][1])</code>, etc.</p>
246      * <p>
247      * This method does not replace data that has already been added.  The
248      * observations represented by <code>data</code> are added to the existing
249      * dataset.</p>
250      * <p>
251      * To replace all data, use <code>clear()</code> before adding the new
252      * data.</p>
253      *
254      * @param data array of observations to be added
255      * @throws ModelSpecificationException if the length of {@code data[i]} is not
256      * greater than or equal to 2
257      */
258     public void addData(final double[][] data) throws ModelSpecificationException {
259         for (int i = 0; i < data.length; i++) {
260             if( data[i].length < 2 ){
261                throw new ModelSpecificationException(LocalizedFormats.INVALID_REGRESSION_OBSERVATION,
262                     data[i].length, 2);
263             }
264             addData(data[i][0], data[i][1]);
265         }
266     }

268     /**
269      * Adds one observation to the regression model.
270      *
271      * @param x the independent variables which form the design matrix
272      * @param y the dependent or response variable
273      * @throws ModelSpecificationException if the length of {@code x} does not equal
274      * the number of independent variables in the model
275      */
276     @Override
277     public void addObservation(final double[] x,final double y)
278             throws ModelSpecificationException {
279         if( x == null || x.length == 0 ){
280             throw new ModelSpecificationException(LocalizedFormats.INVALID_REGRESSION_OBSERVATION,x!=null?x.length:0, 1);
281         }
282         addData( x[0], y );
283     }

285     /**
286      * Adds a series of observations to the regression model. The lengths of
287      * x and y must be the same and x must be rectangular.
288      *
289      * @param x a series of observations on the independent variables
290      * @param y a series of observations on the dependent variable
291      * The length of x and y must be the same
292      * @throws ModelSpecificationException if {@code x} is not rectangular, does not match
293      * the length of {@code y} or does not contain sufficient data to estimate the model
294      */
295     @Override
296     public void addObservations(final double[][] x,final double[] y) throws ModelSpecificationException {
297         if ((x == null) || (y == null) || (x.length != y.length)) {
298             throw new ModelSpecificationException(
299                   LocalizedFormats.DIMENSIONS_MISMATCH_SIMPLE,
300                   (x == null) ? 0 : x.length,
301                   (y == null) ? 0 : y.length);
302         }
303         boolean obsOk=true;
304         for( int i = 0 ; i < x.length; i++){
305             if( x[i] == null || x[i].length == 0 ){
306                 obsOk = false;
307             }
308         }
309         if( !obsOk ){
310             throw new ModelSpecificationException(
311                   LocalizedFormats.NOT_ENOUGH_DATA_FOR_NUMBER_OF_PREDICTORS,
312                   0, 1);
313         }
314         for( int i = 0 ; i < x.length ; i++){
315             addData( x[i][0], y[i] );
316         }
317     }

361     /**
362      * Returns the "predicted" <code>y</code> value associated with the
363      * supplied <code>x</code> value,  based on the data that has been
364      * added to the model when this method is activated.
365      * <p>
366      * <code> predict(x) = intercept + slope * x </code></p>
367      * <p>
368      * <strong>Preconditions</strong>: <ul>
369      * <li>At least two observations (with at least two different x values)
370      * must have been added before invoking this method. If this method is
371      * invoked before a model can be estimated, <code>Double,NaN</code> is
372      * returned.
373      * </li></ul></p>
374      *
375      * @param x input <code>x</code> value
376      * @return predicted <code>y</code> value
377      */
378     public double predict(final double x) {
379         final double b1 = getSlope();
380         if (hasIntercept) {
381             return getIntercept(b1) + b1 * x;
382         }
383         return b1 * x;
384     }

389      * <p>
390      * The least squares estimate of the intercept is computed using the

392      * The intercept is sometimes denoted b0.</p>
393      * <p>
394      * <strong>Preconditions</strong>: <ul>
395      * <li>At least two observations (with at least two different x values)
396      * must have been added before invoking this method. If this method is
397      * invoked before a model can be estimated, <code>Double,NaN</code> is
398      * returned.
399      * </li></ul></p>

401      * @return the intercept of the regression line if the model includes an
402      * intercept; 0 otherwise

420     /**
421     * Returns the slope of the estimated regression line.
422     * <p>
423     * The least squares estimate of the slope is computed using the
424     * <a href="http://www.xycoon.com/estimation4.htm">normal equations</a>.
425     * The slope is sometimes denoted b1.</p>
426     * <p>
427     * <strong>Preconditions</strong>: <ul>
428     * <li>At least two observations (with at least two different x values)
429     * must have been added before invoking this method. If this method is
430     * invoked before a model can be estimated, <code>Double.NaN</code> is
431     * returned.
432     * </li></ul></p>
433     *
434     * @return the slope of the regression line
435     */
436     public double getSlope() {
437         if (n < 2) {
438             return Double.NaN; //not enough data
439         }
440         if (FastMath.abs(sumXX) < 10 * Double.MIN_VALUE) {
441             return Double.NaN; //not enough variation in x
442         }
443         return sumXY / sumXX;
444     }

446     /**
447      * Returns the <a href="http://www.xycoon.com/SumOfSquares.htm">
448      * sum of squared errors</a> (SSE) associated with the regression
449      * model.
450      * <p>
451      * The sum is computed using the computational formula</p>
452      * <p>
453      * <code>SSE = SYY - (SXY * SXY / SXX)</code></p>
454      * <p>
455      * where <code>SYY</code> is the sum of the squared deviations of the y
456      * values about their mean, <code>SXX</code> is similarly defined and
457      * <code>SXY</code> is the sum of the products of x and y mean deviations.
458      * </p><p>
459      * The sums are accumulated using the updating algorithm referenced in
460      * {@link #addData}.</p>
461      * <p>
462      * The return value is constrained to be non-negative - i.e., if due to
463      * rounding errors the computational formula returns a negative result,
464      * 0 is returned.</p>
465      * <p>
466      * <strong>Preconditions</strong>: <ul>
467      * <li>At least two observations (with at least two different x values)
468      * must have been added before invoking this method. If this method is
469      * invoked before a model can be estimated, <code>Double,NaN</code> is
470      * returned.
471      * </li></ul></p>
472      *
473      * @return sum of squared errors associated with the regression model
474      */
475     public double getSumSquaredErrors() {
476         return FastMath.max(0d, sumYY - sumXY * sumXY / sumXX);
477     }
478 
479     /**
480      * Returns the sum of squared deviations of the y values about their mean.
481      * <p>
482      * This is defined as SSTO
483      * <a href="http://www.xycoon.com/SumOfSquares.htm">here</a>.</p>
484      * <p>
485      * If <code>n < 2</code>, this returns <code>Double.NaN</code>.</p>
486      *
487      * @return sum of squared deviations of y values
488      */
489     public double getTotalSumSquares() {
490         if (n < 2) {
491             return Double.NaN;
492         }
493         return sumYY;
494     }

519     /**
520      * Returns the sum of squared deviations of the predicted y values about

522      * <p>
523      * This is usually abbreviated SSR or SSM.  It is defined as SSM
524      * <a href="http://www.xycoon.com/SumOfSquares.htm">here</a></p>
525      * <p>
526      * <strong>Preconditions</strong>: <ul>
527      * <li>At least two observations (with at least two different x values)
528      * must have been added before invoking this method. If this method is
529      * invoked before a model can be estimated, <code>Double.NaN</code> is

531      * </li></ul></p>

536         return getRegressionSumSquares(getSlope());
537     }

539     /**
540      * Returns the sum of squared errors divided by the degrees of freedom,
541      * usually abbreviated MSE.
542      * <p>
543      * If there are fewer than <strong>three</strong> data pairs in the model,
544      * or if there is no variation in <code>x</code>, this returns
545      * <code>Double.NaN</code>.</p>
546      *
547      * @return sum of squared deviations of y values
548      */
549     public double getMeanSquareError() {
550         if (n < 3) {
551             return Double.NaN;
552         }
553         return hasIntercept ? (getSumSquaredErrors() / (n - 2)) : (getSumSquaredErrors() / (n - 1));
554     }

556     /**
557      * Returns <a href="http://mathworld.wolfram.com/CorrelationCoefficient.html">
558      * Pearson's product moment correlation coefficient</a>,
559      * usually denoted r.
560      * <p>
561      * <strong>Preconditions</strong>: <ul>
562      * <li>At least two observations (with at least two different x values)
563      * must have been added before invoking this method. If this method is
564      * invoked before a model can be estimated, <code>Double,NaN</code> is
565      * returned.
566      * </li></ul></p>
567      *
568      * @return Pearson's r
569      */
570     public double getR() {
571         double b1 = getSlope();
572         double result = FastMath.sqrt(getRSquare());
573         if (b1 < 0) {
574             result = -result;
575         }
576         return result;
577     }

579     /**
580      * Returns the <a href="http://www.xycoon.com/coefficient1.htm">
581      * coefficient of determination</a>,
582      * usually denoted r-square.
583      * <p>
584      * <strong>Preconditions</strong>: <ul>
585      * <li>At least two observations (with at least two different x values)
586      * must have been added before invoking this method. If this method is
587      * invoked before a model can be estimated, <code>Double,NaN</code> is
588      * returned.
589      * </li></ul></p>
590      *
591      * @return r-square
592      */
593     public double getRSquare() {
594         double ssto = getTotalSumSquares();
595         return (ssto - getSumSquaredErrors()) / ssto;
596     }

598     /**
599      * Returns the <a href="http://www.xycoon.com/standarderrorb0.htm">
600      * standard error of the intercept estimate</a>,
601      * usually denoted s(b0).
602      * <p>
603      * If there are fewer that <strong>three</strong> observations in the
604      * model, or if there is no variation in x, this returns
605      * <code>Double.NaN</code>.</p> Additionally, a <code>Double.NaN</code> is
606      * returned when the intercept is constrained to be zero
607      *
608      * @return standard error associated with intercept estimate
609      */
610     public double getInterceptStdErr() {
611         if( !hasIntercept ){
612             return Double.NaN;
613         }
614         return FastMath.sqrt(
615             getMeanSquareError() * ((1d / n) + (xbar * xbar) / sumXX));
616     }

618     /**
619      * Returns the <a href="http://www.xycoon.com/standerrorb(1).htm">standard
620      * error of the slope estimate</a>,
621      * usually denoted s(b1).
622      * <p>
623      * If there are fewer that <strong>three</strong> data pairs in the model,
624      * or if there is no variation in x, this returns <code>Double.NaN</code>.
625      * </p>
626      *
627      * @return standard error associated with slope estimate
628      */
629     public double getSlopeStdErr() {
630         return FastMath.sqrt(getMeanSquareError() / sumXX);
631     }

659     /**
660      * Returns the half-width of a (100-100*alpha)% confidence interval for
661      * the slope estimate.
662      * <p>
663      * The (100-100*alpha)% confidence interval is </p>
664      * <p>
665      * <code>(getSlope() - getSlopeConfidenceInterval(),
666      * getSlope() + getSlopeConfidenceInterval())</code></p>
667      * <p>
668      * To request, for example, a 99% confidence interval, use
669      * <code>alpha = .01</code></p>
670      * <p>
671      * <strong>Usage Note</strong>:<br>
672      * The validity of this statistic depends on the assumption that the
673      * observations included in the model are drawn from a
674      * <a href="http://mathworld.wolfram.com/BivariateNormalDistribution.html">
675      * Bivariate Normal Distribution</a>.</p>
676      * <p>
677      * <strong> Preconditions:</strong><ul>
678      * <li>If there are fewer that <strong>three</strong> observations in the
679      * model, or if there is no variation in x, this returns
680      * <code>Double.NaN</code>.
681      * </li>
682      * <li><code>(0 < alpha < 1)</code>; otherwise an
683      * <code>OutOfRangeException</code> is thrown.
684      * </li></ul></p>
685      *
686      * @param alpha the desired significance level
687      * @return half-width of 95% confidence interval for the slope estimate
688      * @throws OutOfRangeException if the confidence interval can not be computed.
689      */
690     public double getSlopeConfidenceInterval(final double alpha)
691     throws OutOfRangeException {
692         if (n < 3) {
693             return Double.NaN;
694         }
695         if (alpha >= 1 || alpha <= 0) {
696             throw new OutOfRangeException(LocalizedFormats.SIGNIFICANCE_LEVEL,
697                                           alpha, 0, 1);
698         }
699         // No advertised NotStrictlyPositiveException here - will return NaN above
700         TDistribution distribution = new TDistribution(n - 2);
701         return getSlopeStdErr() *
702             distribution.inverseCumulativeProbability(1d - alpha / 2d);
703     }
704 
705     /**
706      * Returns the significance level of the slope (equiv) correlation.
707      * <p>
708      * Specifically, the returned value is the smallest <code>alpha</code>
709      * such that the slope confidence interval with significance level
710      * equal to <code>alpha</code> does not include <code>0</code>.
711      * On regression output, this is often denoted <code>Prob(|t| > 0)</code>
712      * </p><p>
713      * <strong>Usage Note</strong>:<br>
714      * The validity of this statistic depends on the assumption that the
715      * observations included in the model are drawn from a
716      * <a href="http://mathworld.wolfram.com/BivariateNormalDistribution.html">
717      * Bivariate Normal Distribution</a>.</p>
718      * <p>
719      * If there are fewer that <strong>three</strong> observations in the
720      * model, or if there is no variation in x, this returns
721      * <code>Double.NaN</code>.</p>
722      *
723      * @return significance level for slope/correlation
724      * @throws org.apache.commons.math4.exception.MaxCountExceededException
725      * if the significance level can not be computed.
726      */
727     public double getSignificance() {
728         if (n < 3) {
729             return Double.NaN;
730         }
731         // No advertised NotStrictlyPositiveException here - will return NaN above
732         TDistribution distribution = new TDistribution(n - 2);
733         return 2d * (1.0 - distribution.cumulativeProbability(
734                     FastMath.abs(getSlope()) / getSlopeStdErr()));
735     }

739     /**
740     * Returns the intercept of the estimated regression line, given the slope.
741     * <p>
742     * Will return <code>NaN</code> if slope is <code>NaN</code>.</p>
743     *
744     * @param slope current slope
745     * @return the intercept of the regression line
746     */
747     private double getIntercept(final double slope) {
748       if( hasIntercept){
749         return (sumY - slope * sumX) / n;
750       }
751       return 0.0;
752     }

754     /**
755      * Computes SSR from b1.
756      *
757      * @param slope regression slope estimate
758      * @return sum of squared deviations of predicted y values
759      */
760     private double getRegressionSumSquares(final double slope) {
761         return slope * slope * sumXX;

764     /**
765      * Performs a regression on data present in buffers and outputs a RegressionResults object.
766      *
767      * <p>If there are fewer than 3 observations in the model and {@code hasIntercept} is true
768      * a {@code NoDataException} is thrown.  If there is no intercept term, the model must
769      * contain at least 2 observations.</p>
770      *
771      * @return RegressionResults acts as a container of regression output
772      * @throws ModelSpecificationException if the model is not correctly specified
773      * @throws NoDataException if there is not sufficient data in the model to
774      * estimate the regression parameters
775      */
776     @Override
777     public RegressionResults regress() throws ModelSpecificationException, NoDataException {
778         if (hasIntercept) {
779             if (n < 3) {
780                 throw new NoDataException(LocalizedFormats.NOT_ENOUGH_DATA_REGRESSION);
781             }
782             if (FastMath.abs(sumXX) > Precision.SAFE_MIN) {
783                 final double[] params = new double[] { getIntercept(), getSlope() };
784                 final double mse = getMeanSquareError();
785                 final double _syy = sumYY + sumY * sumY / n;
786                 final double[] vcv = new double[] { mse * (xbar * xbar / sumXX + 1.0 / n), -xbar * mse / sumXX, mse / sumXX };
787                 return new RegressionResults(params, new double[][] { vcv }, true, n, 2, sumY, _syy, getSumSquaredErrors(), true,
788                         false);
789             } else {
790                 final double[] params = new double[] { sumY / n, Double.NaN };
791                 // final double mse = getMeanSquareError();
792                 final double[] vcv = new double[] { ybar / (n - 1.0), Double.NaN, Double.NaN };
793                 return new RegressionResults(params, new double[][] { vcv }, true, n, 1, sumY, sumYY, getSumSquaredErrors(), true,
794                         false);
795             }
796         } else {
797             if (n < 2) {
798                 throw new NoDataException(LocalizedFormats.NOT_ENOUGH_DATA_REGRESSION);
799             }
800             if (!Double.isNaN(sumXX)) {
801                 final double[] vcv = new double[] { getMeanSquareError() / sumXX };
802                 final double[] params = new double[] { sumXY / sumXX };
803                 return new RegressionResults(params, new double[][] { vcv }, true, n, 1, sumY, sumYY, getSumSquaredErrors(), false,
804                         false);
805             } else {
806                 final double[] vcv = new double[] { Double.NaN };
807                 final double[] params = new double[] { Double.NaN };
808                 return new RegressionResults(params, new double[][] { vcv }, true, n, 1, Double.NaN, Double.NaN, Double.NaN, false,
809                         false);
810             }
811         }
812     }

814     /**
815      * Performs a regression on data present in buffers including only regressors
816      * indexed in variablesToInclude and outputs a RegressionResults object
817      * @param variablesToInclude an array of indices of regressors to include
818      * @return RegressionResults acts as a container of regression output
819      * @throws MathIllegalArgumentException if the variablesToInclude array is null or zero length
820      * @throws OutOfRangeException if a requested variable is not present in model
821      */
822     @Override
823     public RegressionResults regress(int[] variablesToInclude) throws MathIllegalArgumentException{
824         if( variablesToInclude == null || variablesToInclude.length == 0){
825           throw new MathIllegalArgumentException(LocalizedFormats.ARRAY_ZERO_LENGTH_OR_NULL_NOT_ALLOWED);
826         }
827         if( variablesToInclude.length > 2 || (variablesToInclude.length > 1 && !hasIntercept) ){
828             throw new ModelSpecificationException(
829                     LocalizedFormats.ARRAY_SIZE_EXCEEDS_MAX_VARIABLES,
830                     (variablesToInclude.length > 1 && !hasIntercept) ? 1 : 2);
831         }
832 
833         if( hasIntercept ){
834             if( variablesToInclude.length == 2 ){
835                 if( variablesToInclude[0] == 1 ){
836                     throw new ModelSpecificationException(LocalizedFormats.NOT_INCREASING_SEQUENCE);
837                 }else if( variablesToInclude[0] != 0 ){
838                     throw new OutOfRangeException( variablesToInclude[0], 0,1 );
839                 }
840                 if( variablesToInclude[1] != 1){
841                      throw new OutOfRangeException( variablesToInclude[0], 0,1 );
842                 }
843                 return regress();
844             }else{
845                 if( variablesToInclude[0] != 1 && variablesToInclude[0] != 0 ){
846                      throw new OutOfRangeException( variablesToInclude[0],0,1 );
847                 }
848                 final double _mean = sumY * sumY / n;
849                 final double _syy = sumYY + _mean;
850                 if( variablesToInclude[0] == 0 ){
851                     //just the mean
852                     final double[] vcv = new double[]{ sumYY/(((n-1)*n)) };
853                     final double[] params = new double[]{ ybar };
854                     return new RegressionResults(
855                       params, new double[][]{vcv}, true, n, 1,
856                       sumY, _syy+_mean, sumYY,true,false);
857 
858                 }else if( variablesToInclude[0] == 1){
859                     //final double _syy = sumYY + sumY * sumY / ((double) n);
860                     final double _sxx = sumXX + sumX * sumX / n;
861                     final double _sxy = sumXY + sumX * sumY / n;
862                     final double _sse = FastMath.max(0d, _syy - _sxy * _sxy / _sxx);
863                     final double _mse = _sse/((n-1));
864                     if( !Double.isNaN(_sxx) ){
865                         final double[] vcv = new double[]{ _mse / _sxx };
866                         final double[] params = new double[]{ _sxy/_sxx };
867                         return new RegressionResults(
868                                     params, new double[][]{vcv}, true, n, 1,
869                                     sumY, _syy, _sse,false,false);
870                     }else{
871                         final double[] vcv = new double[]{Double.NaN };
872                         final double[] params = new double[]{ Double.NaN };
873                         return new RegressionResults(
874                                     params, new double[][]{vcv}, true, n, 1,
875                                     Double.NaN, Double.NaN, Double.NaN,false,false);
876                     }
877                 }
878             }
879         }else{
880             if( variablesToInclude[0] != 0 ){
881                 throw new OutOfRangeException(variablesToInclude[0],0,0);
882             }
883             return regress();
884         }
885 
886         return null;
887     }
