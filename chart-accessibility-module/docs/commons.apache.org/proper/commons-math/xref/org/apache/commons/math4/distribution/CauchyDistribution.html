17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  import org.apache.commons.math4.util.FastMath;
23  
24  /**
25   * Implementation of the Cauchy distribution.
26   *
27   * @see <a href="http://en.wikipedia.org/wiki/Cauchy_distribution">Cauchy distribution (Wikipedia)</a>
28   * @see <a href="http://mathworld.wolfram.com/CauchyDistribution.html">Cauchy Distribution (MathWorld)</a>
29   * @since 1.1 (changed to concrete class in 3.0)
30   */
31  public class CauchyDistribution extends AbstractRealDistribution {
32      /**
33       * Default inverse cumulative probability accuracy.
34       * @since 2.1
35       */
36      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
37      /** Serializable version identifier */
38      private static final long serialVersionUID = 20160311L;
39      /** The median of this distribution. */
40      private final double median;
41      /** The scale of this distribution. */
42      private final double scale;
43      /** Inverse cumulative probability accuracy */
44      private final double solverAbsoluteAccuracy;
45  
46      /**
47       * Creates a Cauchy distribution with the median equal to zero and scale
48       * equal to one.
49       */
50      public CauchyDistribution() {
51          this(0, 1);
52      }
53  
54      /**
55       * Creates a distribution.
56       *
57       * @param median Median for this distribution.
58       * @param scale Scale parameter for this distribution.
59       * @throws NotStrictlyPositiveException if {@code scale <= 0}.
60       */
61      public CauchyDistribution(double median,
62                                double scale) {
63          this(median, scale, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
64      }
65  
66      /**
67       * Creates a distribution.
68       *
69       * @param median Median for this distribution.
70       * @param scale Scale parameter for this distribution.
71       * @param inverseCumAccuracy Maximum absolute error in inverse
72       * cumulative probability estimates
73       * (defaults to {@link #DEFAULT_INVERSE_ABSOLUTE_ACCURACY}).
74       * @throws NotStrictlyPositiveException if {@code scale <= 0}.
75       */
76      public CauchyDistribution(double median,
77                                double scale,
78                                double inverseCumAccuracy) {
79          if (scale <= 0) {
80              throw new NotStrictlyPositiveException(LocalizedFormats.SCALE, scale);
81          }
82          this.scale = scale;
83          this.median = median;
84          solverAbsoluteAccuracy = inverseCumAccuracy;
85      }
86  
87      /** {@inheritDoc} */
88      @Override
89      public double cumulativeProbability(double x) {
90          return 0.5 + (FastMath.atan((x - median) / scale) / FastMath.PI);
91      }
92  
93      /**
94       * Access the median.
95       *
96       * @return the median for this distribution.
97       */
98      public double getMedian() {
99          return median;
100     }
101 
102     /**
103      * Access the scale parameter.
104      *
105      * @return the scale parameter for this distribution.
106      */
107     public double getScale() {
108         return scale;
109     }
110 
111     /** {@inheritDoc} */
112     @Override
113     public double density(double x) {
114         final double dev = x - median;
115         return (1 / FastMath.PI) * (scale / (dev * dev + scale * scale));
116     }
117 
118     /**
119      * {@inheritDoc}
120      *
121      * Returns {@code Double.NEGATIVE_INFINITY} when {@code p == 0}
122      * and {@code Double.POSITIVE_INFINITY} when {@code p == 1}.
123      */
124     @Override
125     public double inverseCumulativeProbability(double p) throws OutOfRangeException {
126         double ret;
127         if (p < 0 || p > 1) {
128             throw new OutOfRangeException(p, 0, 1);
129         } else if (p == 0) {
130             ret = Double.NEGATIVE_INFINITY;
131         } else  if (p == 1) {
132             ret = Double.POSITIVE_INFINITY;
133         } else {
134             ret = median + scale * FastMath.tan(FastMath.PI * (p - .5));
135         }
136         return ret;
137     }
138 
139     /** {@inheritDoc} */
140     @Override
141     protected double getSolverAbsoluteAccuracy() {
142         return solverAbsoluteAccuracy;
143     }
144 
145     /**
146      * {@inheritDoc}
147      *
148      * The mean is always undefined no matter the parameters.
149      *
150      * @return mean (always Double.NaN)
151      */
152     @Override
153     public double getNumericalMean() {
154         return Double.NaN;
155     }
156 
157     /**
158      * {@inheritDoc}
159      *
160      * The variance is always undefined no matter the parameters.
161      *
162      * @return variance (always Double.NaN)
163      */
164     @Override
165     public double getNumericalVariance() {
166         return Double.NaN;
167     }
168 
169     /**
170      * {@inheritDoc}
171      *
172      * The lower bound of the support is always negative infinity no matter
173      * the parameters.
174      *
175      * @return lower bound of the support (always Double.NEGATIVE_INFINITY)
176      */
177     @Override
178     public double getSupportLowerBound() {
179         return Double.NEGATIVE_INFINITY;
180     }
181 
182     /**
183      * {@inheritDoc}
184      *
185      * The upper bound of the support is always positive infinity no matter
186      * the parameters.
187      *
188      * @return upper bound of the support (always Double.POSITIVE_INFINITY)
189      */
190     @Override
191     public double getSupportUpperBound() {
192         return Double.POSITIVE_INFINITY;
193     }
194 
195     /**
196      * {@inheritDoc}
197      *
198      * The support of this distribution is connected.
199      *
200      * @return {@code true}
201      */
202     @Override
203     public boolean isSupportConnected() {
204         return true;
205     }
206 }
