18  package org.apache.commons.math4.geometry.spherical.oned;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  import org.apache.commons.math4.geometry.Space;
25  
26  /**
27   * This class implements a one-dimensional sphere (i.e. a circle).
28   * <p>
29   * We use here the topologists definition of the 1-sphere (see
30   * <a href="http://mathworld.wolfram.com/Sphere.html">Sphere</a> on
31   * MathWorld), i.e. the 1-sphere is the one-dimensional closed curve
32   * defined in 2D as x<sup>2</sup>+y<sup>2</sup>=1.
33   * </p>
34   * @since 3.3
35   */
36  public class Sphere1D implements Serializable, Space {
37  
38      /** Serializable version identifier. */
39      private static final long serialVersionUID = 20131218L;
40  
41      /** Private constructor for the singleton.
42       */
43      private Sphere1D() {
44      }
45  
46      /** Get the unique instance.
47       * @return the unique instance
48       */
49      public static Sphere1D getInstance() {
50          return LazyHolder.INSTANCE;
51      }
52  
53      /** {@inheritDoc} */
54      @Override
55      public int getDimension() {
56          return 1;
57      }
58  
59      /** {@inheritDoc}
60       * <p>
61       * As the 1-dimension sphere does not have proper sub-spaces,
62       * this method always throws a {@link NoSubSpaceException}
63       * </p>
64       * @return nothing
65       * @throws NoSubSpaceException in all cases
66       */
67      @Override
68      public Space getSubSpace() throws NoSubSpaceException {
69          throw new NoSubSpaceException();
70      }
71  
72      // CHECKSTYLE: stop HideUtilityClassConstructor
73      /** Holder for the instance.
74       * <p>We use here the Initialization On Demand Holder Idiom.</p>
75       */
76      private static class LazyHolder {
77          /** Cached field instance. */
78          private static final Sphere1D INSTANCE = new Sphere1D();
79      }
80      // CHECKSTYLE: resume HideUtilityClassConstructor
81  
82      /** Handle deserialization of the singleton.
83       * @return the singleton instance
84       */
85      private Object readResolve() {
86          // return the singleton instance
87          return LazyHolder.INSTANCE;
88      }
89  
90      /** Specialized exception for inexistent sub-space.
91       * <p>
92       * This exception is thrown when attempting to get the sub-space of a one-dimensional space
93       * </p>
94       */
95      public static class NoSubSpaceException extends MathUnsupportedOperationException {
96  
97          /** Serializable UID. */
98          private static final long serialVersionUID = 20140225L;
99  
100         /** Simple constructor.
101          */
102         public NoSubSpaceException() {
103             super(LocalizedFormats.NOT_SUPPORTED_IN_DIMENSION_N, 1);
104         }
105 
106     }
107 
108 }
