221     /**
222      * Computes the function \( \ln \Gamma(x) \) for \( x > 0 \).
223      *
224      * <p>
225      * For \( x \leq 8 \), the implementation is based on the double precision
226      * implementation in the <em>NSWC Library of Mathematics Subroutines</em>,
227      * {@code DGAMLN}. For \( x \geq 8 \), the implementation is based on
228      * </p>
229      *
230      * <ul>
231      * <li><a href="http://mathworld.wolfram.com/GammaFunction.html">Gamma
232      *     Function</a>, equation (28).</li>
233      * <li><a href="http://mathworld.wolfram.com/LanczosApproximation.html">
234      *     Lanczos Approximation</a>, equations (1) through (5).</li>
235      * <li><a href="http://my.fit.edu/~gabdo/gamma.txt">Paul Godfrey, A note on
236      *     the computation of the convergent Lanczos complex Gamma
237      *     approximation</a></li>
238      * </ul>
239      *
240      * @param x Argument.
241      * @return \( \ln \Gamma(x) \), or {@code NaN} if {@code x <= 0}.
242      */
243     public static double logGamma(double x) {
244         double ret;
245 
246         if (Double.isNaN(x) || (x <= 0.0)) {
247             ret = Double.NaN;
248         } else if (x < 0.5) {
249             return logGamma1p(x) - FastMath.log(x);
250         } else if (x <= 2.5) {
251             return logGamma1p((x - 0.5) - 0.5);
252         } else if (x <= 8.0) {
253             final int n = (int) FastMath.floor(x - 1.5);
254             double prod = 1.0;
255             for (int i = 1; i <= n; i++) {
256                 prod *= x - i;
257             }
258             return logGamma1p(x - (n + 1)) + FastMath.log(prod);
259         } else {
260             double sum = lanczos(x);
261             double tmp = x + LANCZOS_G + .5;
262             ret = ((x + .5) * FastMath.log(tmp)) - tmp +
263                 HALF_LOG_2_PI + FastMath.log(sum / x);
264         }
265 
266         return ret;
267     }

281     /**
282      * Computes the regularized gamma function \( P(a, x) \).
283      *
284      * The implementation of this method is based on:
285      * <ul>
286      *  <li>
287      *   <a href="http://mathworld.wolfram.com/RegularizedGammaFunction.html">
288      *   Regularized Gamma Function</a>, equation (1)
289      *  </li>
290      *  <li>
291      *   <a href="http://mathworld.wolfram.com/IncompleteGammaFunction.html">
292      *   Incomplete Gamma Function</a>, equation (4).
293      *  </li>
294      *  <li>
295      *   <a href="http://mathworld.wolfram.com/ConfluentHypergeometricFunctionoftheFirstKind.html">
296      *   Confluent Hypergeometric Function of the First Kind</a>, equation (1).
297      *  </li>
298      * </ul>
299      *
300      * @param a Parameter \( a \).
301      * @param x Argument.
302      * @param epsilon When the absolute value of the nth item in the
303      * series is less than epsilon the approximation ceases to calculate
304      * further elements in the series.
305      * @param maxIterations Maximum number of "iterations" to complete.
306      * @return \( P(a, x) \)
307      * @throws MaxCountExceededException if the algorithm fails to converge.
308      */
309     public static double regularizedGammaP(double a,
310                                            double x,
311                                            double epsilon,
312                                            int maxIterations) {
313         double ret;
314 
315         if (Double.isNaN(a) || Double.isNaN(x) || (a <= 0.0) || (x < 0.0)) {
316             ret = Double.NaN;
317         } else if (x == 0.0) {
318             ret = 0.0;
319         } else if (x >= a + 1) {
320             // use regularizedGammaQ because it should converge faster in this
321             // case.
322             ret = 1.0 - regularizedGammaQ(a, x, epsilon, maxIterations);
323         } else {
324             // calculate series
325             double n = 0.0; // current element index
326             double an = 1.0 / a; // n-th element in the series
327             double sum = an; // partial sum
328             while (FastMath.abs(an/sum) > epsilon &&
329                    n < maxIterations &&
330                    sum < Double.POSITIVE_INFINITY) {
331                 // compute next element in the series
332                 n += 1.0;
333                 an *= x / (a + n);
334 
335                 // update partial sum
336                 sum += an;
337             }
338             if (n >= maxIterations) {
339                 throw new MaxCountExceededException(maxIterations);
340             } else if (Double.isInfinite(sum)) {
341                 ret = 1.0;
342             } else {
343                 ret = FastMath.exp(-x + (a * FastMath.log(x)) - logGamma(a)) * sum;
344             }
345         }
346 
347         return ret;
348     }

362     /**
363      * Computes the regularized gamma function \( Q(a, x) = 1 - P(a, x) \).
364      *
365      * The implementation of this method is based on:
366      * <ul>
367      *  <li>
368      *   <a href="http://mathworld.wolfram.com/RegularizedGammaFunction.html">
369      *   Regularized Gamma Function</a>, equation (1).
370      *  </li>
371      *  <li>
372      *   <a href="http://functions.wolfram.com/GammaBetaErf/GammaRegularized/10/0003/">
373      *   Regularized incomplete gamma function: Continued fraction representations
374      *   (formula 06.08.10.0003)</a>
375      *  </li>
376      * </ul>
377      *
378      * @param a Parameter \( a \).
379      * @param x Argument.
380      * @param epsilon When the absolute value of the nth item in the
381      * series is less than epsilon the approximation ceases to calculate
382      * further elements in the series.
383      * @param maxIterations Maximum number of "iterations" to complete.
384      * @return \( Q(a, x) \)
385      * @throws MaxCountExceededException if the algorithm fails to converge.
386      */
387     public static double regularizedGammaQ(final double a,
388                                            double x,
389                                            double epsilon,
390                                            int maxIterations) {
391         double ret;
392 
393         if (Double.isNaN(a) || Double.isNaN(x) || (a <= 0.0) || (x < 0.0)) {
394             ret = Double.NaN;
395         } else if (x == 0.0) {
396             ret = 1.0;
397         } else if (x < a + 1.0) {
398             // use regularizedGammaP because it should converge faster in this
399             // case.
400             ret = 1.0 - regularizedGammaP(a, x, epsilon, maxIterations);
401         } else {
402             // create continued fraction
403             ContinuedFraction cf = new ContinuedFraction() {
404 
405                 /** {@inheritDoc} */
406                 @Override
407                 protected double getA(int n, double x) {
408                     return ((2.0 * n) + 1.0) - a + x;
409                 }
410 
411                 /** {@inheritDoc} */
412                 @Override
413                 protected double getB(int n, double x) {
414                     return n * (a - n);
415                 }
416             };
417 
418             ret = 1.0 / cf.evaluate(x, epsilon, maxIterations);
419             ret = FastMath.exp(-x + (a * FastMath.log(x)) - logGamma(a)) * ret;
420         }
421 
422         return ret;
423     }

426     /**
427      * Computes the digamma function, defined as the logarithmic derivative
428      * of the \( \Gamma \) function:
429      * \( \frac{d}{dx}(\ln \Gamma(x)) = \frac{\Gamma^\prime(x)}{\Gamma(x)} \).
430      *
431      * <p>This is an independently written implementation of the algorithm described in
432      * Jose Bernardo, Algorithm AS 103: Psi (Digamma) Function, Applied Statistics, 1976.
433      * A <a href="https://en.wikipedia.org/wiki/Digamma_function#Reflection_formula">
434      * reflection formula</a> is incorporated to improve performance on negative values.</p>
435      *
436      * <p>Some of the constants have been changed to increase accuracy at the moderate
437      * expense of run-time.  The result should be accurate to within \( 10^{-8} \)
438      * relative tolerance for \( 0 < x < 10^{-5} \) and within \( 10^{-8} \) absolute
439      * tolerance otherwise.</p>
440      *
441      * @param x Argument.
442      * @return digamma(x) to within \( 10^{-8} \) relative or absolute error whichever is larger.
443      *
444      * @see <a href="http://en.wikipedia.org/wiki/Digamma_function">Digamma</a>
445      * @see <a href="http://www.uv.es/~bernardo/1976AppStatist.pdf">Bernardo's original article</a>
446      *
447      * @since 2.0
448      */
449     public static double digamma(double x) {
450         if (Double.isNaN(x) || Double.isInfinite(x)) {
451             return x;
452         }
453 
454         double digamma = 0.0;
455         if (x < 0) {
456             // use reflection formula to fall back into positive values
457             digamma -= FastMath.PI / FastMath.tan(FastMath.PI * x);
458             x =  1 - x;
459         }
460 
461         if (x > 0 && x <= S_LIMIT) {
462             // use method 5 from Bernardo AS103
463             // accurate to O(x)
464             return digamma -GAMMA - 1 / x;
465         }
466 
467         while (x < C_LIMIT) {
468             digamma -= 1.0 / x;
469             x += 1.0;
470         }
471 
472         // use method 4 (accurate to O(1/x^8)
473         double inv = 1 / (x * x);
474         //            1       1        1         1
475         // log(x) -  --- - ------ + ------- - -------
476         //           2 x   12 x^2   120 x^4   252 x^6
477         digamma += FastMath.log(x) - 0.5 / x - inv * ((1.0 / 12) + inv * (1.0 / 120 - inv / 252));
478 
479         return digamma;
480     }

482     /**
483      * Computes the trigamma function \( \psi_1(x) = \frac{d^2}{dx^2} (\ln \Gamma(x)) \).
484      * <p>
485      * This function is the derivative of the {@link #digamma(double) digamma function}.
486      * </p>
487      *
488      * @param x Argument.
489      * @return \( \psi_1(x) \) to within \( 10^{-8} \) relative or absolute
490      * error whichever is smaller
491      *
492      * @see <a href="http://en.wikipedia.org/wiki/Trigamma_function">Trigamma</a>
493      * @see #digamma(double)
494      *
495      * @since 2.0
496      */
497     public static double trigamma(double x) {
498         if (Double.isNaN(x) || Double.isInfinite(x)) {
499             return x;
500         }
501 
502         if (x > 0 && x <= S_LIMIT) {
503             return 1 / (x * x);
504         }
505 
506         if (x >= C_LIMIT) {
507             double inv = 1 / (x * x);
508             //  1    1      1       1       1
509             //  - + ---- + ---- - ----- + -----
510             //  x      2      3       5       7
511             //      2 x    6 x    30 x    42 x
512             return 1 / x + inv / 2 + inv / x * (1.0 / 6 - inv * (1.0 / 30 + inv / 42));
513         }
514 
515         return trigamma(x + 1) + 1 / (x * x);
516     }

518     /**
519      * Computes the Lanczos approximation used to compute the gamma function.
520      *
521      * <p>
522      * The Lanczos approximation is related to the Gamma function by the
523      * following equation
524      * \[
525      * \Gamma(x) = \sqrt{2\pi} \, \frac{(g + x + \frac{1}{2})^{x + \frac{1}{2}} \, e^{-(g + x + \frac{1}{2})} \, \mathrm{lanczos}(x)}
526      *                                 {x}
527      * \]
528      * where \(g\) is the Lanczos constant.
529      * </p>
530      *
531      * @param x Argument.
532      * @return The Lanczos approximation.
533      *
534      * @see <a href="http://mathworld.wolfram.com/LanczosApproximation.html">Lanczos Approximation</a>
535      * equations (1) through (5), and Paul Godfrey's
536      * <a href="http://my.fit.edu/~gabdo/gamma.txt">Note on the computation
537      * of the convergent Lanczos complex Gamma approximation</a>
538      *
539      * @since 3.1
540      */
541     public static double lanczos(final double x) {
542         double sum = 0.0;
543         for (int i = LANCZOS.length - 1; i > 0; --i) {
544             sum += LANCZOS[i] / (x + i);
545         }
546         return sum + LANCZOS[0];
547     }

549     /**
550      * Computes the function \( \frac{1}{\Gamma(1 + x)} - 1 \) for \( -0.5 \leq x \leq 1.5 \).
551      * <p>
552      * This implementation is based on the double precision implementation in
553      * the <em>NSWC Library of Mathematics Subroutines</em>, {@code DGAM1}.
554      * </p>
555      *
556      * @param x Argument.
557      * @return \( \frac{1}{\Gamma(1 + x)} - 1 \)
558      * @throws NumberIsTooSmallException if {@code x < -0.5}
559      * @throws NumberIsTooLargeException if {@code x > 1.5}
560      *
561      * @since 3.1
562      */
563     public static double invGamma1pm1(final double x) {
564 
565         if (x < -0.5) {
566             throw new NumberIsTooSmallException(x, -0.5, true);
567         }
568         if (x > 1.5) {
569             throw new NumberIsTooLargeException(x, 1.5, true);
570         }
571 
572         final double ret;
573         final double t = x <= 0.5 ? x : (x - 0.5) - 0.5;
574         if (t < 0.0) {
575             final double a = INV_GAMMA1P_M1_A0 + t * INV_GAMMA1P_M1_A1;
576             double b = INV_GAMMA1P_M1_B8;
577             b = INV_GAMMA1P_M1_B7 + t * b;
578             b = INV_GAMMA1P_M1_B6 + t * b;
579             b = INV_GAMMA1P_M1_B5 + t * b;
580             b = INV_GAMMA1P_M1_B4 + t * b;
581             b = INV_GAMMA1P_M1_B3 + t * b;
582             b = INV_GAMMA1P_M1_B2 + t * b;
583             b = INV_GAMMA1P_M1_B1 + t * b;
584             b = 1.0 + t * b;
585 
586             double c = INV_GAMMA1P_M1_C13 + t * (a / b);
587             c = INV_GAMMA1P_M1_C12 + t * c;
588             c = INV_GAMMA1P_M1_C11 + t * c;
589             c = INV_GAMMA1P_M1_C10 + t * c;
590             c = INV_GAMMA1P_M1_C9 + t * c;
591             c = INV_GAMMA1P_M1_C8 + t * c;
592             c = INV_GAMMA1P_M1_C7 + t * c;
593             c = INV_GAMMA1P_M1_C6 + t * c;
594             c = INV_GAMMA1P_M1_C5 + t * c;
595             c = INV_GAMMA1P_M1_C4 + t * c;
596             c = INV_GAMMA1P_M1_C3 + t * c;
597             c = INV_GAMMA1P_M1_C2 + t * c;
598             c = INV_GAMMA1P_M1_C1 + t * c;
599             c = INV_GAMMA1P_M1_C + t * c;
600             if (x > 0.5) {
601                 ret = t * c / x;
602             } else {
603                 ret = x * ((c + 0.5) + 0.5);
604             }
605         } else {
606             double p = INV_GAMMA1P_M1_P6;
607             p = INV_GAMMA1P_M1_P5 + t * p;
608             p = INV_GAMMA1P_M1_P4 + t * p;
609             p = INV_GAMMA1P_M1_P3 + t * p;
610             p = INV_GAMMA1P_M1_P2 + t * p;
611             p = INV_GAMMA1P_M1_P1 + t * p;
612             p = INV_GAMMA1P_M1_P0 + t * p;
613 
614             double q = INV_GAMMA1P_M1_Q4;
615             q = INV_GAMMA1P_M1_Q3 + t * q;
616             q = INV_GAMMA1P_M1_Q2 + t * q;
617             q = INV_GAMMA1P_M1_Q1 + t * q;
618             q = 1.0 + t * q;
619 
620             double c = INV_GAMMA1P_M1_C13 + (p / q) * t;
621             c = INV_GAMMA1P_M1_C12 + t * c;
622             c = INV_GAMMA1P_M1_C11 + t * c;
623             c = INV_GAMMA1P_M1_C10 + t * c;
624             c = INV_GAMMA1P_M1_C9 + t * c;
625             c = INV_GAMMA1P_M1_C8 + t * c;
626             c = INV_GAMMA1P_M1_C7 + t * c;
627             c = INV_GAMMA1P_M1_C6 + t * c;
628             c = INV_GAMMA1P_M1_C5 + t * c;
629             c = INV_GAMMA1P_M1_C4 + t * c;
630             c = INV_GAMMA1P_M1_C3 + t * c;
631             c = INV_GAMMA1P_M1_C2 + t * c;
632             c = INV_GAMMA1P_M1_C1 + t * c;
633             c = INV_GAMMA1P_M1_C0 + t * c;
634 
635             if (x > 0.5) {
636                 ret = (t / x) * ((c - 0.5) - 0.5);
637             } else {
638                 ret = x * c;
639             }
640         }
641 
642         return ret;
643     }

646      * Computes the function \( \ln \Gamma(1 + x) \) for \( -0.5 \leq x \leq 1.5 \).
647      * <p>
648      * This implementation is based on the double precision implementation in
649      * the <em>NSWC Library of Mathematics Subroutines</em>, {@code DGMLN1}.

672     /**
673      * Computes the value of \( \Gamma(x) \).
674      * <p>
675      * Based on the <em>NSWC Library of Mathematics Subroutines</em> double
676      * precision implementation, {@code DGAMMA}.
677      * </p>
678      *
679      * @param x Argument.
680      * @return \( \Gamma(x) \)
681      *
682      * @since 3.1
683      */
684     public static double gamma(final double x) {
685 
686         if ((x == FastMath.rint(x)) && (x <= 0.0)) {
687             return Double.NaN;
688         }
689 
690         final double ret;
691         final double absX = FastMath.abs(x);
692         if (absX <= 20.0) {
693             if (x >= 1.0) {
694                 /*
695                  * From the recurrence relation
696                  * Gamma(x) = (x - 1) * ... * (x - n) * Gamma(x - n),
697                  * then
698                  * Gamma(t) = 1 / [1 + invGamma1pm1(t - 1)],
699                  * where t = x - n. This means that t must satisfy
700                  * -0.5 <= t - 1 <= 1.5.
701                  */
702                 double prod = 1.0;
703                 double t = x;
704                 while (t > 2.5) {
705                     t -= 1.0;
706                     prod *= t;
707                 }
708                 ret = prod / (1.0 + invGamma1pm1(t - 1.0));
709             } else {
710                 /*
711                  * From the recurrence relation
712                  * Gamma(x) = Gamma(x + n + 1) / [x * (x + 1) * ... * (x + n)]
713                  * then
714                  * Gamma(x + n + 1) = 1 / [1 + invGamma1pm1(x + n)],
715                  * which requires -0.5 <= x + n <= 1.5.
716                  */
717                 double prod = x;
718                 double t = x;
719                 while (t < -0.5) {
720                     t += 1.0;
721                     prod *= t;
722                 }
723                 ret = 1.0 / (prod * (1.0 + invGamma1pm1(t)));
724             }
725         } else {
726             final double y = absX + LANCZOS_G + 0.5;
727             final double gammaAbs = SQRT_TWO_PI / absX *
728                                     FastMath.pow(y, absX + 0.5) *
729                                     FastMath.exp(-y) * lanczos(absX);
730             if (x > 0.0) {
731                 ret = gammaAbs;
732             } else {
733                 /*
734                  * From the reflection formula
735                  * Gamma(x) * Gamma(1 - x) * sin(pi * x) = pi,
736                  * and the recurrence relation
737                  * Gamma(1 - x) = -x * Gamma(-x),
738                  * it is found
739                  * Gamma(x) = -pi / [x * sin(pi * x) * Gamma(-x)].
740                  */
741                 ret = -FastMath.PI /
742                       (x * FastMath.sin(FastMath.PI * x) * gammaAbs);
743             }
744         }
745         return ret;
746     }
