18  package org.apache.commons.math4.random;
19  
20  /**
21   * This interface represent a normalized random generator for
22   * scalars.
23   * Normalized generator provide null mean and unit standard deviation scalars.
24   * @since 1.2
25   */
26  public interface NormalizedRandomGenerator {
27  
28    /** Generate a random scalar with null mean and unit standard deviation.
29     * <p>This method does <strong>not</strong> specify the shape of the
30     * distribution, it is the implementing class that provides it. The
31     * only contract here is to generate numbers with null mean and unit
32     * standard deviation.</p>
33     * @return a random scalar with null mean and unit standard deviation
34     */
35    double nextNormalizedDouble();
36  
37  }
