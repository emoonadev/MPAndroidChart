18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.differentiation.DerivativeStructure;
21  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
22  import org.apache.commons.math4.util.FastMath;
23  
24  /**
25   * Square-root function.
26   *
27   * @since 3.0
28   */
29  public class Sqrt implements UnivariateDifferentiableFunction {
30      /** {@inheritDoc} */
31      @Override
32      public double value(double x) {
33          return FastMath.sqrt(x);
34      }
35  
36      /** {@inheritDoc}
37       * @since 3.1
38       */
39      @Override
40      public DerivativeStructure value(final DerivativeStructure t) {
41          return t.sqrt();
42      }
43  
44  }
