18  package org.apache.commons.math4.ml.neuralnet.sofm.util;
19  
20  import org.apache.commons.math4.analysis.function.Logistic;
21  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
22  import org.apache.commons.math4.exception.NumberIsTooLargeException;
23  
24  /**
25   * Decay function whose shape is similar to a sigmoid.
26   * <br/>
27   * Class is immutable.
28   *
29   * @since 3.3
30   */
31  public class QuasiSigmoidDecayFunction {
32      /** Sigmoid. */
33      private final Logistic sigmoid;
34      /** See {@link #value(long)}. */
35      private final double scale;
36  
37      /**
38       * Creates an instance.
39       * The function {@code f} will have the following properties:
40       * <ul>
41       *  <li>{@code f(0) = initValue}</li>
42       *  <li>{@code numCall} is the inflexion point</li>
43       *  <li>{@code slope = f'(numCall)}</li>
44       * </ul>
45       *
46       * @param initValue Initial value, i.e. {@link #value(long) value(0)}.
47       * @param slope Value of the function derivative at {@code numCall}.
48       * @param numCall Inflexion point.
49       * @throws NotStrictlyPositiveException if {@code initValue <= 0}.
50       * @throws NumberIsTooLargeException if {@code slope >= 0}.
51       * @throws NotStrictlyPositiveException if {@code numCall <= 0}.
52       */
53      public QuasiSigmoidDecayFunction(double initValue,
54                                       double slope,
55                                       long numCall) {
56          if (initValue <= 0) {
57              throw new NotStrictlyPositiveException(initValue);
58          }
59          if (slope >= 0) {
60              throw new NumberIsTooLargeException(slope, 0, false);
61          }
62          if (numCall <= 1) {
63              throw new NotStrictlyPositiveException(numCall);
64          }
65  
66          final double k = initValue;
67          final double m = numCall;
68          final double b = 4 * slope / initValue;
69          final double q = 1;
70          final double a = 0;
71          final double n = 1;
72          sigmoid = new Logistic(k, m, b, q, a, n);
73  
74          final double y0 = sigmoid.value(0);
75          scale = k / y0;
76      }
77  
78      /**
79       * Computes the value of the learning factor.
80       *
81       * @param numCall Current step of the training task.
82       * @return the value of the function at {@code numCall}.
83       */
84      public double value(long numCall) {
85          return scale * sigmoid.value(numCall);
86      }
87  }
