17  package org.apache.commons.math4.ode;
18  
19  import org.apache.commons.math4.exception.DimensionMismatchException;
20  import org.apache.commons.math4.exception.MaxCountExceededException;
21  
22  /** Interface to compute exactly Jacobian matrix for some parameter
23   *  when computing {@link JacobianMatrices partial derivatives equations}.
24   *
25   * @since 3.0
26   */
27  public interface ParameterJacobianProvider extends Parameterizable {
28  
29      /** Compute the Jacobian matrix of ODE with respect to one parameter.
30       * <p>If the parameter does not belong to the collection returned by
31       * {@link #getParametersNames()}, the Jacobian will be set to 0,
32       * but no errors will be triggered.</p>
33       * @param t current value of the independent <I>time</I> variable
34       * @param y array containing the current value of the main state vector
35       * @param yDot array containing the current value of the time derivative
36       * of the main state vector
37       * @param paramName name of the parameter to consider
38       * @param dFdP placeholder array where to put the Jacobian matrix of the
39       * ODE with respect to the parameter
40       * @exception MaxCountExceededException if the number of functions evaluations is exceeded
41       * @exception DimensionMismatchException if arrays dimensions do not match equations settings
42       * @exception UnknownParameterException if the parameter is not supported
43       */
44      void computeParameterJacobian(double t, double[] y, double[] yDot,
45                                    String paramName, double[] dFdP)
46          throws DimensionMismatchException, MaxCountExceededException, UnknownParameterException;
47  
48  }
