17  package org.apache.commons.math4.stat.descriptive;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.util.FastMath;
22  import org.apache.commons.math4.util.MathUtils;
23  import org.apache.commons.math4.util.Precision;
24  
25  /**
26   *  Value object representing the results of a univariate statistical summary.
27   *
28   */
29  public class StatisticalSummaryValues implements Serializable,
30      StatisticalSummary {
31  
32      /** Serialization id */
33      private static final long serialVersionUID = -5108854841843722536L;
34  
35      /** The sample mean */
36      private final double mean;
37  
38      /** The sample variance */
39      private final double variance;
40  
41      /** The number of observations in the sample */
42      private final long n;
43  
44      /** The maximum value */
45      private final double max;
46  
47      /** The minimum value */
48      private final double min;
49  
50      /** The sum of the sample values */
51      private final double sum;
52  
53      /**
54        * Constructor
55        *
56        * @param mean  the sample mean
57        * @param variance  the sample variance
58        * @param n  the number of observations in the sample
59        * @param max  the maximum value
60        * @param min  the minimum value
61        * @param sum  the sum of the values
62       */
63      public StatisticalSummaryValues(double mean, double variance, long n,
64          double max, double min, double sum) {
65          super();
66          this.mean = mean;
67          this.variance = variance;
68          this.n = n;
69          this.max = max;
70          this.min = min;
71          this.sum = sum;
72      }
73  
74      /**
75       * @return Returns the max.
76       */
77      @Override
78      public double getMax() {
79          return max;
80      }
81  
82      /**
83       * @return Returns the mean.
84       */
85      @Override
86      public double getMean() {
87          return mean;
88      }
89  
90      /**
91       * @return Returns the min.
92       */
93      @Override
94      public double getMin() {
95          return min;
96      }
97  
98      /**
99       * @return Returns the number of values.
100      */
101     @Override
102     public long getN() {
103         return n;
104     }
105 
106     /**
107      * @return Returns the sum.
108      */
109     @Override
110     public double getSum() {
111         return sum;
112     }
113 
114     /**
115      * @return Returns the standard deviation
116      */
117     @Override
118     public double getStandardDeviation() {
119         return FastMath.sqrt(variance);
120     }
121 
122     /**
123      * @return Returns the variance.
124      */
125     @Override
126     public double getVariance() {
127         return variance;
128     }
129 
130     /**
131      * Returns true iff <code>object</code> is a
132      * <code>StatisticalSummaryValues</code> instance and all statistics have
133      *  the same values as this.
134      *
135      * @param object the object to test equality against.
136      * @return true if object equals this
137      */
138     @Override
139     public boolean equals(Object object) {
140         if (object == this ) {
141             return true;
142         }
143         if (object instanceof StatisticalSummaryValues == false) {
144             return false;
145         }
146         StatisticalSummaryValues stat = (StatisticalSummaryValues) object;
147         return Precision.equalsIncludingNaN(stat.getMax(),      getMax())  &&
148                Precision.equalsIncludingNaN(stat.getMean(),     getMean()) &&
149                Precision.equalsIncludingNaN(stat.getMin(),      getMin())  &&
150                Precision.equalsIncludingNaN(stat.getN(),        getN())    &&
151                Precision.equalsIncludingNaN(stat.getSum(),      getSum())  &&
152                Precision.equalsIncludingNaN(stat.getVariance(), getVariance());
153     }
154 
155     /**
156      * Returns hash code based on values of statistics
157      *
158      * @return hash code
159      */
160     @Override
161     public int hashCode() {
162         int result = 31 + MathUtils.hash(getMax());
163         result = result * 31 + MathUtils.hash(getMean());
164         result = result * 31 + MathUtils.hash(getMin());
165         result = result * 31 + MathUtils.hash(getN());
166         result = result * 31 + MathUtils.hash(getSum());
167         result = result * 31 + MathUtils.hash(getVariance());
168         return result;
169     }
170 
171     /**
172      * Generates a text report displaying values of statistics.
173      * Each statistic is displayed on a separate line.
174      *
175      * @return String with line feeds displaying statistics
176      */
177     @Override
178     public String toString() {
179         StringBuffer outBuffer = new StringBuffer();
180         String endl = "\n";
181         outBuffer.append("StatisticalSummaryValues:").append(endl);
182         outBuffer.append("n: ").append(getN()).append(endl);
183         outBuffer.append("min: ").append(getMin()).append(endl);
184         outBuffer.append("max: ").append(getMax()).append(endl);
185         outBuffer.append("mean: ").append(getMean()).append(endl);
186         outBuffer.append("std dev: ").append(getStandardDeviation())
187             .append(endl);
188         outBuffer.append("variance: ").append(getVariance()).append(endl);
189         outBuffer.append("sum: ").append(getSum()).append(endl);
190         return outBuffer.toString();
191     }
192 
193 }
