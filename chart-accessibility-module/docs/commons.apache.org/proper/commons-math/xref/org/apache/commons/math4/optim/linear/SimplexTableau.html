39  /**
40   * A tableau for use in the Simplex method.
41   *
42   * <p>
43   * Example:
44   * <pre>
45   *   W |  Z |  x1 |  x2 |  x- | s1 |  s2 |  a1 |  RHS
46   * ---------------------------------------------------
47   *  -1    0    0     0     0     0     0     1     0   &lt;= phase 1 objective
48   *   0    1   -15   -10    0     0     0     0     0   &lt;= phase 2 objective
49   *   0    0    1     0     0     1     0     0     2   &lt;= constraint 1
50   *   0    0    0     1     0     0     1     0     3   &lt;= constraint 2
51   *   0    0    1     1     0     0     0     1     4   &lt;= constraint 3
52   * </pre>
53   * W: Phase 1 objective function</br>
54   * Z: Phase 2 objective function</br>
55   * x1 &amp; x2: Decision variables</br>
56   * x-: Extra decision variable to allow for negative values</br>
57   * s1 &amp; s2: Slack/Surplus variables</br>
58   * a1: Artificial variable</br>
59   * RHS: Right hand side</br>
60   * </p>
61   * @since 2.0
62   */

127     /**
128      * Build a tableau for a linear problem.
129      * @param f linear objective function
130      * @param constraints linear constraints
131      * @param goalType type of optimization goal: either {@link GoalType#MAXIMIZE} or {@link GoalType#MINIMIZE}
132      * @param restrictToNonNegative whether to restrict the variables to non-negative values
133      * @param epsilon amount of error to accept when checking for optimality
134      * @param maxUlps amount of error to accept in floating point comparisons
135      * @throws DimensionMismatchException if the dimension of the constraints does not match the
136      *   dimension of the objective function
137      */
138     SimplexTableau(final LinearObjectiveFunction f,
139                    final Collection<LinearConstraint> constraints,
140                    final GoalType goalType,
141                    final boolean restrictToNonNegative,
142                    final double epsilon,
143                    final int maxUlps) throws DimensionMismatchException {
144         checkDimensions(f, constraints);
145         this.f                      = f;
146         this.constraints            = normalizeConstraints(constraints);
147         this.restrictToNonNegative  = restrictToNonNegative;
148         this.epsilon                = epsilon;
149         this.maxUlps                = maxUlps;
150         this.numDecisionVariables   = f.getCoefficients().getDimension() + (restrictToNonNegative ? 0 : 1);
151         this.numSlackVariables      = getConstraintTypeCounts(Relationship.LEQ) +
152                                       getConstraintTypeCounts(Relationship.GEQ);
153         this.numArtificialVariables = getConstraintTypeCounts(Relationship.EQ) +
154                                       getConstraintTypeCounts(Relationship.GEQ);
155         this.tableau = createTableau(goalType == GoalType.MAXIMIZE);
156         // initialize the basic variables for phase 1:
157         //   we know that only slack or artificial variables can be basic
158         initializeBasicVariables(getSlackVariableOffset());
159         initializeColumnLabels();
160     }

162     /**
163      * Checks that the dimensions of the objective function and the constraints match.
164      * @param objectiveFunction the objective function
165      * @param c the set of constraints
166      * @throws DimensionMismatchException if the constraint dimensions do not match with the
167      *   dimension of the objective function
168      */
169     private void checkDimensions(final LinearObjectiveFunction objectiveFunction,
170                                  final Collection<LinearConstraint> c) {
171         final int dimension = objectiveFunction.getCoefficients().getDimension();
172         for (final LinearConstraint constraint : c) {
173             final int constraintDimension = constraint.getCoefficients().getDimension();
174             if (constraintDimension != dimension) {
175                 throw new DimensionMismatchException(constraintDimension, dimension);
176             }
177         }
178     }
179     /**
180      * Initialize the labels for the columns.
181      */
182     protected void initializeColumnLabels() {
183       if (getNumObjectiveFunctions() == 2) {
184         columnLabels.add("W");
185       }
186       columnLabels.add("Z");
187       for (int i = 0; i < getOriginalNumDecisionVariables(); i++) {
188         columnLabels.add("x" + i);
189       }
190       if (!restrictToNonNegative) {
191         columnLabels.add(NEGATIVE_VAR_COLUMN_LABEL);
192       }
193       for (int i = 0; i < getNumSlackVariables(); i++) {
194         columnLabels.add("s" + i);
195       }
196       for (int i = 0; i < getNumArtificialVariables(); i++) {
197         columnLabels.add("a" + i);
198       }
199       columnLabels.add("RHS");
200     }

202     /**
203      * Create the tableau by itself.
204      * @param maximize if true, goal is to maximize the objective function
205      * @return created tableau
206      */
207     protected Array2DRowRealMatrix createTableau(final boolean maximize) {
208 
209         // create a matrix of the correct size
210         int width = numDecisionVariables + numSlackVariables +
211         numArtificialVariables + getNumObjectiveFunctions() + 1; // + 1 is for RHS
212         int height = constraints.size() + getNumObjectiveFunctions();
213         Array2DRowRealMatrix matrix = new Array2DRowRealMatrix(height, width);
214 
215         // initialize the objective function rows
216         if (getNumObjectiveFunctions() == 2) {
217             matrix.setEntry(0, 0, -1);
218         }
219 
220         int zIndex = (getNumObjectiveFunctions() == 1) ? 0 : 1;
221         matrix.setEntry(zIndex, zIndex, maximize ? 1 : -1);
222         RealVector objectiveCoefficients = maximize ? f.getCoefficients().mapMultiply(-1) : f.getCoefficients();
223         copyArray(objectiveCoefficients.toArray(), matrix.getDataRef()[zIndex]);
224         matrix.setEntry(zIndex, width - 1, maximize ? f.getConstantTerm() : -1 * f.getConstantTerm());
225 
226         if (!restrictToNonNegative) {
227             matrix.setEntry(zIndex, getSlackVariableOffset() - 1,
228                             getInvertedCoefficientSum(objectiveCoefficients));
229         }
230 
231         // initialize the constraint rows
232         int slackVar = 0;
233         int artificialVar = 0;
234         for (int i = 0; i < constraints.size(); i++) {
235             LinearConstraint constraint = constraints.get(i);
236             int row = getNumObjectiveFunctions() + i;
237 
238             // decision variable coefficients
239             copyArray(constraint.getCoefficients().toArray(), matrix.getDataRef()[row]);
240 
241             // x-
242             if (!restrictToNonNegative) {
243                 matrix.setEntry(row, getSlackVariableOffset() - 1,
244                                 getInvertedCoefficientSum(constraint.getCoefficients()));
245             }
246 
247             // RHS
248             matrix.setEntry(row, width - 1, constraint.getValue());
249 
250             // slack variables
251             if (constraint.getRelationship() == Relationship.LEQ) {
252                 matrix.setEntry(row, getSlackVariableOffset() + slackVar++, 1);  // slack
253             } else if (constraint.getRelationship() == Relationship.GEQ) {
254                 matrix.setEntry(row, getSlackVariableOffset() + slackVar++, -1); // excess
255             }
256 
257             // artificial variables
258             if ((constraint.getRelationship() == Relationship.EQ) ||
259                 (constraint.getRelationship() == Relationship.GEQ)) {
260                 matrix.setEntry(0, getArtificialVariableOffset() + artificialVar, 1);
261                 matrix.setEntry(row, getArtificialVariableOffset() + artificialVar++, 1);
262                 matrix.setRowVector(0, matrix.getRowVector(0).subtract(matrix.getRowVector(row)));
263             }
264         }
265 
266         return matrix;
267     }

269     /**
270      * Get new versions of the constraints which have positive right hand sides.
271      * @param originalConstraints original (not normalized) constraints
272      * @return new versions of the constraints
273      */
274     public List<LinearConstraint> normalizeConstraints(Collection<LinearConstraint> originalConstraints) {
275         List<LinearConstraint> normalized = new ArrayList<LinearConstraint>(originalConstraints.size());
276         for (LinearConstraint constraint : originalConstraints) {
277             normalized.add(normalize(constraint));
278         }
279         return normalized;
280     }

282     /**
283      * Get a new equation equivalent to this one with a positive right hand side.
284      * @param constraint reference constraint
285      * @return new equation
286      */
287     private LinearConstraint normalize(final LinearConstraint constraint) {
288         if (constraint.getValue() < 0) {
289             return new LinearConstraint(constraint.getCoefficients().mapMultiply(-1),
290                                         constraint.getRelationship().oppositeRelationship(),
291                                         -1 * constraint.getValue());
292         }
293         return new LinearConstraint(constraint.getCoefficients(),
294                                     constraint.getRelationship(), constraint.getValue());
295     }

305     /**
306      * Get a count of constraints corresponding to a specified relationship.
307      * @param relationship relationship to count
308      * @return number of constraint with the specified relationship
309      */
310     private int getConstraintTypeCounts(final Relationship relationship) {
311         int count = 0;
312         for (final LinearConstraint constraint : constraints) {
313             if (constraint.getRelationship() == relationship) {
314                 ++count;
315             }
316         }
317         return count;
318     }

320     /**
321      * Get the -1 times the sum of all coefficients in the given array.
322      * @param coefficients coefficients to sum
323      * @return the -1 times the sum of all coefficients in the given array.
324      */
325     protected static double getInvertedCoefficientSum(final RealVector coefficients) {
326         double sum = 0;
327         for (double coefficient : coefficients.toArray()) {
328             sum -= coefficient;
329         }
330         return sum;
331     }

352     /**
353      * Initializes the basic variable / row mapping.
354      * @param startColumn the column to start
355      */
356     private void initializeBasicVariables(final int startColumn) {
357         basicVariables = new int[getWidth() - 1];
358         basicRows = new int[getHeight()];
359 
360         Arrays.fill(basicVariables, -1);
361 
362         for (int i = startColumn; i < getWidth() - 1; i++) {
363             Integer row = findBasicRow(i);
364             if (row != null) {
365                 basicVariables[i] = row;
366                 basicRows[row] = i;
367             }
368         }
369     }
370 
371     /**
372      * Returns the row in which the given column is basic.
373      * @param col index of the column
374      * @return the row that the variable is basic in, or {@code null} if the variable is not basic.
375      */
376     private Integer findBasicRow(final int col) {
377         Integer row = null;
378         for (int i = 0; i < getHeight(); i++) {
379             final double entry = getEntry(i, col);
380             if (Precision.equals(entry, 1d, maxUlps) && (row == null)) {
381                 row = i;
382             } else if (!Precision.equals(entry, 0d, maxUlps)) {
383                 return null;
384             }
385         }
386         return row;
387     }

389     /**
390      * Removes the phase 1 objective function, positive cost non-artificial variables,
391      * and the non-basic artificial variables from this tableau.
392      */
393     protected void dropPhase1Objective() {
394         if (getNumObjectiveFunctions() == 1) {
395             return;
396         }
397 
398         final Set<Integer> columnsToDrop = new TreeSet<Integer>();
399         columnsToDrop.add(0);
400 
401         // positive cost non-artificial variables
402         for (int i = getNumObjectiveFunctions(); i < getArtificialVariableOffset(); i++) {
403             final double entry = getEntry(0, i);
404             if (Precision.compareTo(entry, 0d, epsilon) > 0) {
405                 columnsToDrop.add(i);
406             }
407         }
408 
409         // non-basic artificial variables
410         for (int i = 0; i < getNumArtificialVariables(); i++) {
411             int col = i + getArtificialVariableOffset();
412             if (getBasicRow(col) == null) {
413                 columnsToDrop.add(col);
414             }
415         }
416 
417         final double[][] matrix = new double[getHeight() - 1][getWidth() - columnsToDrop.size()];
418         for (int i = 1; i < getHeight(); i++) {
419             int col = 0;
420             for (int j = 0; j < getWidth(); j++) {
421                 if (!columnsToDrop.contains(j)) {
422                     matrix[i - 1][col++] = getEntry(i, j);
423                 }
424             }
425         }
426 
427         // remove the columns in reverse order so the indices are correct
428         Integer[] drop = columnsToDrop.toArray(new Integer[columnsToDrop.size()]);
429         for (int i = drop.length - 1; i >= 0; i--) {
430             columnLabels.remove((int) drop[i]);
431         }
432 
433         this.tableau = new Array2DRowRealMatrix(matrix);
434         this.numArtificialVariables = 0;
435         // need to update the basic variable mappings as row/columns have been dropped
436         initializeBasicVariables(getNumObjectiveFunctions());
437     }

447     /**
448      * Returns whether the problem is at an optimal state.
449      * @return whether the model has been solved
450      */
451     boolean isOptimal() {
452         final double[] objectiveFunctionRow = getRow(0);
453         final int end = getRhsOffset();
454         for (int i = getNumObjectiveFunctions(); i < end; i++) {
455             final double entry = objectiveFunctionRow[i];
456             if (Precision.compareTo(entry, 0d, epsilon) < 0) {
457                 return false;
458             }
459         }
460         return true;
461     }

463     /**
464      * Get the current solution.
465      * @return current solution
466      */
467     protected PointValuePair getSolution() {
468         int negativeVarColumn = columnLabels.indexOf(NEGATIVE_VAR_COLUMN_LABEL);
469         Integer negativeVarBasicRow = negativeVarColumn > 0 ? getBasicRow(negativeVarColumn) : null;
470         double mostNegative = negativeVarBasicRow == null ? 0 : getEntry(negativeVarBasicRow, getRhsOffset());
471 
472         final Set<Integer> usedBasicRows = new HashSet<Integer>();
473         final double[] coefficients = new double[getOriginalNumDecisionVariables()];
474         for (int i = 0; i < coefficients.length; i++) {
475             int colIndex = columnLabels.indexOf("x" + i);
476             if (colIndex < 0) {
477                 coefficients[i] = 0;
478                 continue;
479             }
480             Integer basicRow = getBasicRow(colIndex);
481             if (basicRow != null && basicRow == 0) {
482                 // if the basic row is found to be the objective function row
483                 // set the coefficient to 0 -> this case handles unconstrained
484                 // variables that are still part of the objective function
485                 coefficients[i] = 0;
486             } else if (usedBasicRows.contains(basicRow)) {
487                 // if multiple variables can take a given value
488                 // then we choose the first and set the rest equal to 0
489                 coefficients[i] = 0 - (restrictToNonNegative ? 0 : mostNegative);
490             } else {
491                 usedBasicRows.add(basicRow);
492                 coefficients[i] =
493                     (basicRow == null ? 0 : getEntry(basicRow, getRhsOffset())) -
494                     (restrictToNonNegative ? 0 : mostNegative);
495             }
496         }
497         return new PointValuePair(coefficients, f.value(coefficients));
498     }

500     /**
501      * Perform the row operations of the simplex algorithm with the selected
502      * pivot column and row.
503      * @param pivotCol the pivot column
504      * @param pivotRow the pivot row
505      */
506     protected void performRowOperations(int pivotCol, int pivotRow) {
507         // set the pivot element to 1
508         final double pivotVal = getEntry(pivotRow, pivotCol);
509         divideRow(pivotRow, pivotVal);
510 
511         // set the rest of the pivot column to 0
512         for (int i = 0; i < getHeight(); i++) {
513             if (i != pivotRow) {
514                 final double multiplier = getEntry(i, pivotCol);
515                 if (multiplier != 0.0) {
516                     subtractRow(i, pivotRow, multiplier);
517                 }
518             }
519         }
520 
521         // update the basic variable mappings
522         final int previousBasicVariable = getBasicVariable(pivotRow);
523         basicVariables[previousBasicVariable] = -1;
524         basicVariables[pivotCol] = pivotRow;
525         basicRows[pivotRow] = pivotCol;
526     }

528     /**
529      * Divides one row by a given divisor.
530      * <p>
531      * After application of this operation, the following will hold:
532      * <pre>dividendRow = dividendRow / divisor</pre>
533      *
534      * @param dividendRowIndex index of the row
535      * @param divisor value of the divisor
536      */
537     protected void divideRow(final int dividendRowIndex, final double divisor) {
538         final double[] dividendRow = getRow(dividendRowIndex);
539         for (int j = 0; j < getWidth(); j++) {
540             dividendRow[j] /= divisor;
541         }
542     }
543 
544     /**
545      * Subtracts a multiple of one row from another.
546      * <p>
547      * After application of this operation, the following will hold:
548      * <pre>minuendRow = minuendRow - multiple * subtrahendRow</pre>
549      *
550      * @param minuendRowIndex row index
551      * @param subtrahendRowIndex row index
552      * @param multiplier multiplication factor
553      */
554     protected void subtractRow(final int minuendRowIndex, final int subtrahendRowIndex, final double multiplier) {
555         final double[] minuendRow = getRow(minuendRowIndex);
556         final double[] subtrahendRow = getRow(subtrahendRowIndex);
557         for (int i = 0; i < getWidth(); i++) {
558             minuendRow[i] -= subtrahendRow[i] * multiplier;
559         }
560     }

677     /** {@inheritDoc} */
678     @Override
679     public boolean equals(Object other) {
680 
681       if (this == other) {
682         return true;
683       }
684 
685       if (other instanceof SimplexTableau) {
686           SimplexTableau rhs = (SimplexTableau) other;
687           return (restrictToNonNegative  == rhs.restrictToNonNegative) &&
688                  (numDecisionVariables   == rhs.numDecisionVariables) &&
689                  (numSlackVariables      == rhs.numSlackVariables) &&
