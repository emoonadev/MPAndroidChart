17  package org.apache.commons.math4.geometry.enclosing;
18  
19  import java.util.List;
20  
21  import org.apache.commons.math4.geometry.Point;
22  import org.apache.commons.math4.geometry.Space;
23  
24  /** Interface for generating balls based on support points.
25   * <p>
26   * This generator is used in the {@link WelzlEncloser Emo Welzl} algorithm
27   * and its derivatives.
28   * </p>
29   * @param <S> Space type.
30   * @param <P> Point type.
31   * @see EnclosingBall
32   * @since 3.3
33   */
34  public interface SupportBallGenerator<S extends Space, P extends Point<S>> {
35  
36      /** Create a ball whose boundary lies on prescribed support points.
37       * @param support support points (may be empty)
38       * @return ball whose boundary lies on the prescribed support points
39       */
40      EnclosingBall<S, P> ballOnSupport(List<P> support);
41  
42  }
