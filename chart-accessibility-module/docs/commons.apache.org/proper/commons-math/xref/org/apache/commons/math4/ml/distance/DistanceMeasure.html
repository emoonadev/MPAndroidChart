17  package org.apache.commons.math4.ml.distance;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.exception.DimensionMismatchException;
22  
23  /**
24   * Interface for distance measures of n-dimensional vectors.
25   *
26   * @since 3.2
27   */
28  public interface DistanceMeasure extends Serializable {
29  
30      /**
31       * Compute the distance between two n-dimensional vectors.
32       * <p>
33       * The two vectors are required to have the same dimension.
34       *
35       * @param a the first vector
36       * @param b the second vector
37       * @return the distance between the two vectors
38       * @throws DimensionMismatchException if the array lengths differ.
39       */
40      double compute(double[] a, double[] b) throws DimensionMismatchException;
41  }
