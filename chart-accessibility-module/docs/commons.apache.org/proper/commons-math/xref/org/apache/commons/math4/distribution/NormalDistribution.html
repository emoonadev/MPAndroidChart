18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.exception.OutOfRangeException;
23  import org.apache.commons.math4.exception.util.LocalizedFormats;
24  import org.apache.commons.rng.UniformRandomProvider;
25  import org.apache.commons.math4.special.Erf;
26  import org.apache.commons.math4.util.FastMath;
27  
28  /**
29   * Implementation of the normal (gaussian) distribution.
30   *
31   * @see <a href="http://en.wikipedia.org/wiki/Normal_distribution">Normal distribution (Wikipedia)</a>
32   * @see <a href="http://mathworld.wolfram.com/NormalDistribution.html">Normal distribution (MathWorld)</a>
33   */
34  public class NormalDistribution extends AbstractRealDistribution {
35      /**
36       * Default inverse cumulative probability accuracy.
37       * @since 2.1
38       */
39      public static final double DEFAULT_INVERSE_ABSOLUTE_ACCURACY = 1e-9;
40      /** Serializable version identifier. */
41      private static final long serialVersionUID = 8589540077390120676L;
42      /** &radic;(2) */
43      private static final double SQRT2 = FastMath.sqrt(2.0);
44      /** Mean of this distribution. */
45      private final double mean;
46      /** Standard deviation of this distribution. */
47      private final double standardDeviation;
48      /** The value of {@code log(sd) + 0.5*log(2*pi)} stored for faster computation. */
49      private final double logStandardDeviationPlusHalfLog2Pi;
50      /** Inverse cumulative probability accuracy. */
51      private final double solverAbsoluteAccuracy;
52  
53      /**
54       * Create a normal distribution with mean equal to zero and standard
55       * deviation equal to one.
56       */
57      public NormalDistribution() {
58          this(0, 1);
59      }
60  
61      /**
62       * Creates a distribution.
63       *
64       * @param mean Mean for this distribution.
65       * @param sd Standard deviation for this distribution.
66       * @throws NotStrictlyPositiveException if {@code sd <= 0}.
67       */
68      public NormalDistribution(double mean,
69                                double sd)
70          throws NotStrictlyPositiveException {
71          this(mean, sd, DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
72      }
73  
74  
75      /**
76       * Creates a distribution.
77       *
78       * @param mean Mean for this distribution.
79       * @param sd Standard deviation for this distribution.
80       * @param inverseCumAccuracy Inverse cumulative probability accuracy.
81       * @throws NotStrictlyPositiveException if {@code sd <= 0}.
82       */
83      public NormalDistribution(double mean,
84                                double sd,
85                                double inverseCumAccuracy)
86          throws NotStrictlyPositiveException {
87          if (sd <= 0) {
88              throw new NotStrictlyPositiveException(LocalizedFormats.STANDARD_DEVIATION, sd);
89          }
90  
91          this.mean = mean;
92          standardDeviation = sd;
93          logStandardDeviationPlusHalfLog2Pi = FastMath.log(sd) + 0.5 * FastMath.log(2 * FastMath.PI);
94          solverAbsoluteAccuracy = inverseCumAccuracy;
95      }
96  
97      /**
98       * Access the mean.
99       *
100      * @return the mean for this distribution.
101      */
102     public double getMean() {
103         return mean;
104     }
105 
106     /**
107      * Access the standard deviation.
108      *
109      * @return the standard deviation for this distribution.
110      */
111     public double getStandardDeviation() {
112         return standardDeviation;
113     }
114 
115     /** {@inheritDoc} */
116     @Override
117     public double density(double x) {
118         return FastMath.exp(logDensity(x));
119     }
120 
121     /** {@inheritDoc} */
122     @Override
123     public double logDensity(double x) {
124         final double x0 = x - mean;
125         final double x1 = x0 / standardDeviation;
126         return -0.5 * x1 * x1 - logStandardDeviationPlusHalfLog2Pi;
127     }
128 
129     /**
130      * {@inheritDoc}
131      *
132      * If {@code x} is more than 40 standard deviations from the mean, 0 or 1
133      * is returned, as in these cases the actual value is within
134      * {@code Double.MIN_VALUE} of 0 or 1.
135      */
136     @Override
137     public double cumulativeProbability(double x)  {
138         final double dev = x - mean;
139         if (FastMath.abs(dev) > 40 * standardDeviation) {
140             return dev < 0 ? 0.0d : 1.0d;
141         }
142         return 0.5 * Erf.erfc(-dev / (standardDeviation * SQRT2));
143     }
144 
145     /** {@inheritDoc}
146      * @since 3.2
147      */
148     @Override
149     public double inverseCumulativeProbability(final double p) throws OutOfRangeException {
150         if (p < 0.0 || p > 1.0) {
151             throw new OutOfRangeException(p, 0, 1);
152         }
153         return mean + standardDeviation * SQRT2 * Erf.erfInv(2 * p - 1);
154     }
155 
156     /** {@inheritDoc} */
157     @Override
158     public double probability(double x0,
159                               double x1)
160         throws NumberIsTooLargeException {
161         if (x0 > x1) {
162             throw new NumberIsTooLargeException(LocalizedFormats.LOWER_ENDPOINT_ABOVE_UPPER_ENDPOINT,
163                                                 x0, x1, true);
164         }
165         final double denom = standardDeviation * SQRT2;
166         final double v0 = (x0 - mean) / denom;
167         final double v1 = (x1 - mean) / denom;
168         return 0.5 * Erf.erf(v0, v1);
169     }
170 
171     /** {@inheritDoc} */
172     @Override
173     protected double getSolverAbsoluteAccuracy() {
174         return solverAbsoluteAccuracy;
175     }
176 
177     /**
178      * {@inheritDoc}
179      *
180      * For mean parameter {@code mu}, the mean is {@code mu}.
181      */
182     @Override
183     public double getNumericalMean() {
184         return getMean();
185     }
186 
187     /**
188      * {@inheritDoc}
189      *
190      * For standard deviation parameter {@code s}, the variance is {@code s^2}.
191      */
192     @Override
193     public double getNumericalVariance() {
194         final double s = getStandardDeviation();
195         return s * s;
196     }
197 
198     /**
199      * {@inheritDoc}
200      *
201      * The lower bound of the support is always negative infinity
202      * no matter the parameters.
203      *
204      * @return lower bound of the support (always
205      * {@code Double.NEGATIVE_INFINITY})
206      */
207     @Override
208     public double getSupportLowerBound() {
209         return Double.NEGATIVE_INFINITY;
210     }
211 
212     /**
213      * {@inheritDoc}
214      *
215      * The upper bound of the support is always positive infinity
216      * no matter the parameters.
217      *
218      * @return upper bound of the support (always
219      * {@code Double.POSITIVE_INFINITY})
220      */
221     @Override
222     public double getSupportUpperBound() {
223         return Double.POSITIVE_INFINITY;
224     }
225 
226     /**
227      * {@inheritDoc}
228      *
229      * The support of this distribution is connected.
230      *
231      * @return {@code true}
232      */
233     @Override
234     public boolean isSupportConnected() {
235         return true;
236     }
237 
238     /** {@inheritDoc} */
239     @Override
240     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
241         return new RealDistribution.Sampler() {
242             /** Next gaussian. */
243             private double nextGaussian = Double.NaN;
244 
245             /** {@inheritDoc} */
246             @Override
247             public double sample() {
248                 final double random;
249                 if (Double.isNaN(nextGaussian)) {
250                     // Generate a pair of Gaussian numbers.
251 
252                     final double x = rng.nextDouble();
253                     final double y = rng.nextDouble();
254                     final double alpha = 2 * FastMath.PI * x;
255                     final double r = FastMath.sqrt(-2 * FastMath.log(y));
256 
257                     // Return the first element of the generated pair.
258                     random = r * FastMath.cos(alpha);
259 
260                     // Keep second element of the pair for next invocation.
261                     nextGaussian = r * FastMath.sin(alpha);
262                 } else {
263                     // Use the second element of the pair (generated at the
264                     // previous invocation).
265                     random = nextGaussian;
266 
267                     // Both elements of the pair have been used.
268                     nextGaussian = Double.NaN;
269                 }
270 
271                 return standardDeviation * random + mean;
272             }
273         };
274     }
