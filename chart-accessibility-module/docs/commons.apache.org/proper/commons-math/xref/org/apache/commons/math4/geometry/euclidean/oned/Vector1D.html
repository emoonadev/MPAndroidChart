19  import java.text.NumberFormat;

21  import org.apache.commons.math4.exception.MathArithmeticException;

29  /** This class represents a 1D vector.
30   * <p>Instances of this class are guaranteed to be immutable.</p>
31   * @since 3.0
32   */
33  public class Vector1D implements Vector<Euclidean1D> {
34  
35      /** Origin (coordinates: 0). */
36      public static final Vector1D ZERO = new Vector1D(0.0);
37  
38      /** Unit (coordinates: 1). */
39      public static final Vector1D ONE  = new Vector1D(1.0);
40  
41      // CHECKSTYLE: stop ConstantName
42      /** A vector with all coordinates set to NaN. */
43      public static final Vector1D NaN = new Vector1D(Double.NaN);
44      // CHECKSTYLE: resume ConstantName
45  
46      /** A vector with all coordinates set to positive infinity. */
47      public static final Vector1D POSITIVE_INFINITY =
48          new Vector1D(Double.POSITIVE_INFINITY);
49  
50      /** A vector with all coordinates set to negative infinity. */
51      public static final Vector1D NEGATIVE_INFINITY =
52          new Vector1D(Double.NEGATIVE_INFINITY);
53  
54      /** Serializable UID. */
55      private static final long serialVersionUID = 7556674948671647925L;
56  
57      /** Abscissa. */
58      private final double x;
59  
60      /** Simple constructor.
61       * Build a vector from its coordinates
62       * @param x abscissa
63       * @see #getX()
64       */
65      public Vector1D(double x) {
66          this.x = x;
67      }
68  
69      /** Multiplicative constructor
70       * Build a vector from another one and a scale factor.
71       * The vector built will be a * u
72       * @param a scale factor
73       * @param u base (unscaled) vector
74       */
75      public Vector1D(double a, Vector1D u) {
76          this.x = a * u.x;
77      }
78  
79      /** Linear constructor
80       * Build a vector from two other ones and corresponding scale factors.
81       * The vector built will be a1 * u1 + a2 * u2
82       * @param a1 first scale factor
83       * @param u1 first base (unscaled) vector
84       * @param a2 second scale factor
85       * @param u2 second base (unscaled) vector
86       */
87      public Vector1D(double a1, Vector1D u1, double a2, Vector1D u2) {
88          this.x = a1 * u1.x + a2 * u2.x;
89      }
90  
91      /** Linear constructor
92       * Build a vector from three other ones and corresponding scale factors.
93       * The vector built will be a1 * u1 + a2 * u2 + a3 * u3
94       * @param a1 first scale factor
95       * @param u1 first base (unscaled) vector
96       * @param a2 second scale factor
97       * @param u2 second base (unscaled) vector
98       * @param a3 third scale factor
99       * @param u3 third base (unscaled) vector
100      */
101     public Vector1D(double a1, Vector1D u1, double a2, Vector1D u2,
102                    double a3, Vector1D u3) {
103         this.x = a1 * u1.x + a2 * u2.x + a3 * u3.x;
104     }
105 
106     /** Linear constructor
107      * Build a vector from four other ones and corresponding scale factors.
108      * The vector built will be a1 * u1 + a2 * u2 + a3 * u3 + a4 * u4
109      * @param a1 first scale factor
110      * @param u1 first base (unscaled) vector
111      * @param a2 second scale factor
112      * @param u2 second base (unscaled) vector
113      * @param a3 third scale factor
114      * @param u3 third base (unscaled) vector
115      * @param a4 fourth scale factor
116      * @param u4 fourth base (unscaled) vector
117      */
118     public Vector1D(double a1, Vector1D u1, double a2, Vector1D u2,
119                    double a3, Vector1D u3, double a4, Vector1D u4) {
120         this.x = a1 * u1.x + a2 * u2.x + a3 * u3.x + a4 * u4.x;
121     }
122 
123     /** Get the abscissa of the vector.
124      * @return abscissa of the vector
125      * @see #Vector1D(double)
126      */
127     public double getX() {
128         return x;
129     }
130 
131     /** {@inheritDoc} */
132     @Override
133     public Space getSpace() {
134         return Euclidean1D.getInstance();
135     }
136 
137     /** {@inheritDoc} */
138     @Override
139     public Vector1D getZero() {
140         return ZERO;
141     }
142 
143     /** {@inheritDoc} */
144     @Override
145     public double getNorm1() {
146         return FastMath.abs(x);
147     }
148 
149     /** {@inheritDoc} */
150     @Override
151     public double getNorm() {
152         return FastMath.abs(x);
153     }
154 
155     /** {@inheritDoc} */
156     @Override
157     public double getNormSq() {
158         return x * x;
159     }
160 
161     /** {@inheritDoc} */
162     @Override
163     public double getNormInf() {
164         return FastMath.abs(x);
165     }
166 
167     /** {@inheritDoc} */
168     @Override
169     public Vector1D add(Vector<Euclidean1D> v) {
170         Vector1D v1 = (Vector1D) v;
171         return new Vector1D(x + v1.getX());
172     }
173 
174     /** {@inheritDoc} */
175     @Override
176     public Vector1D add(double factor, Vector<Euclidean1D> v) {
177         Vector1D v1 = (Vector1D) v;
178         return new Vector1D(x + factor * v1.getX());
179     }
180 
181     /** {@inheritDoc} */
182     @Override
183     public Vector1D subtract(Vector<Euclidean1D> p) {
184         Vector1D p3 = (Vector1D) p;
185         return new Vector1D(x - p3.x);
186     }
187 
188     /** {@inheritDoc} */
189     @Override
190     public Vector1D subtract(double factor, Vector<Euclidean1D> v) {
191         Vector1D v1 = (Vector1D) v;
192         return new Vector1D(x - factor * v1.getX());
193     }
194 
195     /** {@inheritDoc} */
196     @Override
197     public Vector1D normalize() throws MathArithmeticException {
198         double s = getNorm();
199         if (s == 0) {
200             throw new MathArithmeticException(LocalizedFormats.CANNOT_NORMALIZE_A_ZERO_NORM_VECTOR);
201         }
202         return scalarMultiply(1 / s);
203     }
204     /** {@inheritDoc} */
205     @Override
206     public Vector1D negate() {
207         return new Vector1D(-x);
208     }
209 
210     /** {@inheritDoc} */
211     @Override
212     public Vector1D scalarMultiply(double a) {
213         return new Vector1D(a * x);
214     }
215 
216     /** {@inheritDoc} */
217     @Override
218     public boolean isNaN() {
219         return Double.isNaN(x);
220     }
221 
222     /** {@inheritDoc} */
223     @Override
224     public boolean isInfinite() {
225         return !isNaN() && Double.isInfinite(x);
226     }
227 
228     /** {@inheritDoc} */
229     @Override
230     public double distance1(Vector<Euclidean1D> p) {
231         Vector1D p3 = (Vector1D) p;
232         final double dx = FastMath.abs(p3.x - x);
233         return dx;
234     }
235 
236     /** {@inheritDoc} */
237     @Override
238     public double distance(Point<Euclidean1D> p) {
239         Vector1D p3 = (Vector1D) p;
240         final double dx = p3.x - x;
241         return FastMath.abs(dx);
242     }
243 
244     /** {@inheritDoc} */
245     @Override
246     public double distanceInf(Vector<Euclidean1D> p) {
247         Vector1D p3 = (Vector1D) p;
248         final double dx = FastMath.abs(p3.x - x);
249         return dx;
250     }
251 
252     /** {@inheritDoc} */
253     @Override
254     public double distanceSq(Vector<Euclidean1D> p) {
255         Vector1D p3 = (Vector1D) p;
256         final double dx = p3.x - x;
257         return dx * dx;
258     }
259 
260     /** {@inheritDoc} */
261     @Override
262     public double dotProduct(final Vector<Euclidean1D> v) {
263         final Vector1D v1 = (Vector1D) v;
264         return x * v1.x;
265     }

267     /** Compute the distance between two vectors according to the L<sub>2</sub> norm.
268      * <p>Calling this method is equivalent to calling:
269      * <code>p1.subtract(p2).getNorm()</code> except that no intermediate
270      * vector is built</p>
271      * @param p1 first vector
272      * @param p2 second vector
273      * @return the distance between p1 and p2 according to the L<sub>2</sub> norm
274      */
275     public static double distance(Vector1D p1, Vector1D p2) {
276         return p1.distance(p2);
277     }
278 
279     /** Compute the distance between two vectors according to the L<sub>&infin;</sub> norm.
280      * <p>Calling this method is equivalent to calling:
281      * <code>p1.subtract(p2).getNormInf()</code> except that no intermediate
282      * vector is built</p>
283      * @param p1 first vector
284      * @param p2 second vector
285      * @return the distance between p1 and p2 according to the L<sub>&infin;</sub> norm
286      */
287     public static double distanceInf(Vector1D p1, Vector1D p2) {
288         return p1.distanceInf(p2);
289     }
290 
291     /** Compute the square of the distance between two vectors.
292      * <p>Calling this method is equivalent to calling:
293      * <code>p1.subtract(p2).getNormSq()</code> except that no intermediate
294      * vector is built</p>
295      * @param p1 first vector
296      * @param p2 second vector
297      * @return the square of the distance between p1 and p2
298      */
299     public static double distanceSq(Vector1D p1, Vector1D p2) {
300         return p1.distanceSq(p2);
301     }
302 
303     /**
304      * Test for the equality of two 1D vectors.
305      * <p>
306      * If all coordinates of two 1D vectors are exactly the same, and none are
307      * <code>Double.NaN</code>, the two 1D vectors are considered to be equal.
308      * </p>
309      * <p>
310      * <code>NaN</code> coordinates are considered to affect globally the vector
311      * and be equals to each other - i.e, if either (or all) coordinates of the
312      * 1D vector are equal to <code>Double.NaN</code>, the 1D vector is equal to
313      * {@link #NaN}.
314      * </p>
315      *
316      * @param other Object to test for equality to this
317      * @return true if two 1D vector objects are equal, false if
318      *         object is null, not an instance of Vector1D, or
319      *         not equal to this Vector1D instance
320      *
321      */
322     @Override
323     public boolean equals(Object other) {
324 
325         if (this == other) {
326             return true;
327         }
328 
329         if (other instanceof Vector1D) {
330             final Vector1D rhs = (Vector1D)other;
331             if (rhs.isNaN()) {
332                 return this.isNaN();
333             }
334 
335             return x == rhs.x;
336         }
337         return false;
338     }
339 
340     /**
341      * Get a hashCode for the 1D vector.
342      * <p>
343      * All NaN values have the same hash code.</p>
344      *
345      * @return a hash code value for this object
346      */
347     @Override
348     public int hashCode() {
349         if (isNaN()) {
350             return 7785;
351         }
352         return 997 * MathUtils.hash(x);
353     }

361     }
