146     /**
147      * Create a fraction given the double value and either the maximum error
148      * allowed or the maximum number of denominator digits.
149      * <p>
150      *
151      * NOTE: This constructor is called with EITHER
152      *   - a valid epsilon value and the maxDenominator set to Integer.MAX_VALUE
153      *     (that way the maxDenominator has no effect).
154      * OR
155      *   - a valid maxDenominator value and the epsilon value set to zero
156      *     (that way epsilon only has effect if there is an exact match before
157      *     the maxDenominator value is reached).
158      * </p><p>
159      *
160      * It has been done this way so that the same code can be (re)used for both
161      * scenarios. However this could be confusing to users if it were part of
162      * the public API and this constructor should therefore remain PRIVATE.
163      * </p>
164      *
165      * See JIRA issue ticket MATH-181 for more details:
166      *
167      *     https://issues.apache.org/jira/browse/MATH-181
168      *
169      * @param value the double value to convert to a fraction.
170      * @param epsilon maximum error allowed.  The resulting fraction is within
171      *        {@code epsilon} of {@code value}, in absolute terms.
172      * @param maxDenominator maximum denominator value allowed.
173      * @param maxIterations maximum number of convergents
174      * @throws FractionConversionException if the continued fraction failed to
175      *         converge.
176      */
177     private Fraction(double value, double epsilon, int maxDenominator, int maxIterations)
178         throws FractionConversionException
179     {
180         long overflow = Integer.MAX_VALUE;
181         double r0 = value;
182         long a0 = (long)FastMath.floor(r0);
183         if (FastMath.abs(a0) > overflow) {
184             throw new FractionConversionException(value, a0, 1l);
185         }
186 
187         // check for (almost) integer arguments, which should not go to iterations.
188         if (FastMath.abs(a0 - value) < epsilon) {
189             this.numerator = (int) a0;
190             this.denominator = 1;
191             return;
192         }
193 
194         long p0 = 1;
195         long q0 = 0;
196         long p1 = a0;
197         long q1 = 1;
198 
199         long p2 = 0;
200         long q2 = 1;
201 
202         int n = 0;
203         boolean stop = false;
204         do {
205             ++n;
206             double r1 = 1.0 / (r0 - a0);
207             long a1 = (long)FastMath.floor(r1);
208             p2 = (a1 * p1) + p0;
209             q2 = (a1 * q1) + q0;
210 
211             if ((FastMath.abs(p2) > overflow) || (FastMath.abs(q2) > overflow)) {
212                 // in maxDenominator mode, if the last fraction was very close to the actual value
213                 // q2 may overflow in the next iteration; in this case return the last one.
214                 if (epsilon == 0.0 && FastMath.abs(q1) < maxDenominator) {
215                     break;
216                 }
217                 throw new FractionConversionException(value, p2, q2);
218             }
219 
220             double convergent = (double)p2 / (double)q2;
221             if (n < maxIterations && FastMath.abs(convergent - value) > epsilon && q2 < maxDenominator) {
222                 p0 = p1;
223                 p1 = p2;
224                 q0 = q1;
225                 q1 = q2;
226                 a0 = a1;
227                 r0 = r1;
228             } else {
229                 stop = true;
230             }
231         } while (!stop);
232 
233         if (n >= maxIterations) {
234             throw new FractionConversionException(value, maxIterations);
235         }
236 
237         if (q2 < maxDenominator) {
238             this.numerator = (int) p2;
239             this.denominator = (int) q2;
240         } else {
241             this.numerator = (int) p1;
242             this.denominator = (int) q1;
243         }
244 
245     }

256     /**
257      * Create a fraction given the numerator and denominator.  The fraction is
258      * reduced to lowest terms.
259      * @param num the numerator.
260      * @param den the denominator.
261      * @throws MathArithmeticException if the denominator is {@code zero}
262      */
263     public Fraction(int num, int den) {
264         if (den == 0) {
265             throw new MathArithmeticException(LocalizedFormats.ZERO_DENOMINATOR_IN_FRACTION,
266                                               num, den);
267         }
268         if (den < 0) {
269             if (num == Integer.MIN_VALUE ||
270                 den == Integer.MIN_VALUE) {
271                 throw new MathArithmeticException(LocalizedFormats.OVERFLOW_IN_FRACTION,
272                                                   num, den);
273             }
274             num = -num;
275             den = -den;
276         }
277         // reduce numerator and denominator by greatest common denominator.
278         final int d = ArithmeticUtils.gcd(num, den);
279         if (d > 1) {
280             num /= d;
281             den /= d;
282         }
283 
284         // move sign to numerator.
285         if (den < 0) {
286             num = -num;
287             den = -den;
288         }
289         this.numerator   = num;
290         this.denominator = den;
291     }

293     /**
294      * Returns the absolute value of this fraction.
295      * @return the absolute value.
296      */
297     public Fraction abs() {
298         Fraction ret;
299         if (numerator >= 0) {
300             ret = this;
301         } else {
302             ret = negate();
303         }
304         return ret;
305     }

307     /**
308      * Compares this object to another based on size.
309      * @param object the object to compare to
310      * @return -1 if this is less than {@code object}, +1 if this is greater
311      *         than {@code object}, 0 if they are equal.
312      */
313     @Override
314     public int compareTo(Fraction object) {
315         long nOd = ((long) numerator) * object.denominator;
316         long dOn = ((long) denominator) * object.numerator;
317         return (nOd < dOn) ? -1 : ((nOd > dOn) ? +1 : 0);
318     }

330     /**
331      * Test for the equality of two fractions.  If the lowest term
332      * numerator and denominators are the same for both fractions, the two
333      * fractions are considered to be equal.
334      * @param other fraction to test for equality to this fraction
335      * @return true if two fractions are equal, false if object is
336      *         {@code null}, not an instance of {@link Fraction}, or not equal
337      *         to this fraction instance.
338      */
339     @Override
340     public boolean equals(Object other) {
341         if (this == other) {
342             return true;
343         }
344         if (other instanceof Fraction) {
345             // since fractions are always in lowest terms, numerators and
346             // denominators can be compared directly for equality.
347             Fraction rhs = (Fraction)other;
348             return (numerator == rhs.numerator) &&
349                 (denominator == rhs.denominator);
350         }
351         return false;
352     }

478     /**
479      * Implement add and subtract using algorithm described in Knuth 4.5.1.
480      *
481      * @param fraction the fraction to subtract, must not be {@code null}
482      * @param isAdd true to add, false to subtract
483      * @return a {@code Fraction} instance with the resulting values
484      * @throws NullArgumentException if the fraction is {@code null}
485      * @throws MathArithmeticException if the resulting numerator or denominator
486      *   cannot be represented in an {@code int}.
487      */
488     private Fraction addSub(Fraction fraction, boolean isAdd) {
489         if (fraction == null) {
490             throw new NullArgumentException(LocalizedFormats.FRACTION);
491         }
492         // zero is identity for addition.
493         if (numerator == 0) {
494             return isAdd ? fraction : fraction.negate();
495         }
496         if (fraction.numerator == 0) {
497             return this;
498         }
499         // if denominators are randomly distributed, d1 will be 1 about 61%
500         // of the time.
501         int d1 = ArithmeticUtils.gcd(denominator, fraction.denominator);
502         if (d1==1) {
503             // result is ( (u*v' +/- u'v) / u'v')
504             int uvp = ArithmeticUtils.mulAndCheck(numerator, fraction.denominator);
505             int upv = ArithmeticUtils.mulAndCheck(fraction.numerator, denominator);
506             return new Fraction
507                 (isAdd ? ArithmeticUtils.addAndCheck(uvp, upv) :
508                  ArithmeticUtils.subAndCheck(uvp, upv),
509                  ArithmeticUtils.mulAndCheck(denominator, fraction.denominator));
510         }
511         // the quantity 't' requires 65 bits of precision; see knuth 4.5.1
512         // exercise 7.  we're going to use a BigInteger.
513         // t = u(v'/d1) +/- v(u'/d1)
514         BigInteger uvp = BigInteger.valueOf(numerator)
515         .multiply(BigInteger.valueOf(fraction.denominator/d1));
516         BigInteger upv = BigInteger.valueOf(fraction.numerator)
517         .multiply(BigInteger.valueOf(denominator/d1));
518         BigInteger t = isAdd ? uvp.add(upv) : uvp.subtract(upv);
519         // but d2 doesn't need extra precision because
520         // d2 = gcd(t,d1) = gcd(t mod d1, d1)
521         int tmodd1 = t.mod(BigInteger.valueOf(d1)).intValue();
522         int d2 = (tmodd1==0)?d1:ArithmeticUtils.gcd(tmodd1, d1);
523 
524         // result is (t/d2) / (u'/d1)(v'/d2)
525         BigInteger w = t.divide(BigInteger.valueOf(d2));
526         if (w.bitLength() > 31) {
527             throw new MathArithmeticException(LocalizedFormats.NUMERATOR_OVERFLOW_AFTER_MULTIPLY,
528                                               w);
529         }
530         return new Fraction (w.intValue(),
531                 ArithmeticUtils.mulAndCheck(denominator/d1,
532                         fraction.denominator/d2));
533     }

535     /**
536      * <p>Multiplies the value of this fraction by another, returning the
537      * result in reduced form.</p>
538      *
539      * @param fraction  the fraction to multiply by, must not be {@code null}
540      * @return a {@code Fraction} instance with the resulting values
541      * @throws NullArgumentException if the fraction is {@code null}
542      * @throws MathArithmeticException if the resulting numerator or denominator exceeds
543      *  {@code Integer.MAX_VALUE}
544      */
545     @Override
546     public Fraction multiply(Fraction fraction) {
547         if (fraction == null) {
548             throw new NullArgumentException(LocalizedFormats.FRACTION);
549         }
550         if (numerator == 0 || fraction.numerator == 0) {
551             return ZERO;
552         }
553         // knuth 4.5.1
554         // make sure we don't overflow unless the result *must* overflow.
555         int d1 = ArithmeticUtils.gcd(numerator, fraction.denominator);
556         int d2 = ArithmeticUtils.gcd(fraction.numerator, denominator);
557         return getReducedFraction
558         (ArithmeticUtils.mulAndCheck(numerator/d1, fraction.numerator/d2),
559                 ArithmeticUtils.mulAndCheck(denominator/d2, fraction.denominator/d1));
560     }

574      *
575      * @param fraction  the fraction to divide by, must not be {@code null}
576      * @return a {@code Fraction} instance with the resulting values
577      * @throws IllegalArgumentException if the fraction is {@code null}
578      * @throws MathArithmeticException if the fraction to divide by is zero
579      * @throws MathArithmeticException if the resulting numerator or denominator exceeds
580      *  {@code Integer.MAX_VALUE}
581      */
582     @Override
583     public Fraction divide(Fraction fraction) {
584         if (fraction == null) {
585             throw new NullArgumentException(LocalizedFormats.FRACTION);
586         }
587         if (fraction.numerator == 0) {
588             throw new MathArithmeticException(LocalizedFormats.ZERO_FRACTION_TO_DIVIDE_BY,
589                                               fraction.numerator, fraction.denominator);
590         }
591         return multiply(fraction.reciprocal());
592     }

615     /**
616      * <p>Creates a {@code Fraction} instance with the 2 parts
617      * of a fraction Y/Z.</p>
618      *
619      * <p>Any negative signs are resolved to be on the numerator.</p>
620      *
621      * @param numerator  the numerator, for example the three in 'three sevenths'
622      * @param denominator  the denominator, for example the seven in 'three sevenths'
623      * @return a new fraction instance, with the numerator and denominator reduced
624      * @throws MathArithmeticException if the denominator is {@code zero}
625      */
626     public static Fraction getReducedFraction(int numerator, int denominator) {
627         if (denominator == 0) {
628             throw new MathArithmeticException(LocalizedFormats.ZERO_DENOMINATOR_IN_FRACTION,
629                                               numerator, denominator);
630         }
631         if (numerator==0) {
632             return ZERO; // normalize zero.
633         }
634         // allow 2^k/-2^31 as a valid fraction (where k>0)
635         if (denominator==Integer.MIN_VALUE && (numerator&1)==0) {
636             numerator/=2; denominator/=2;
637         }
638         if (denominator < 0) {
639             if (numerator==Integer.MIN_VALUE ||
640                     denominator==Integer.MIN_VALUE) {
641                 throw new MathArithmeticException(LocalizedFormats.OVERFLOW_IN_FRACTION,
642                                                   numerator, denominator);
643             }
644             numerator = -numerator;
645             denominator = -denominator;
646         }
647         // simplify fraction.
648         int gcd = ArithmeticUtils.gcd(numerator, denominator);
649         numerator /= gcd;
650         denominator /= gcd;
651         return new Fraction(numerator, denominator);
652     }

655      * <p>
656      * Returns the {@code String} representing this fraction, ie
657      * "num / dem" or just "num" if the denominator is one.
658      * </p>
659      *
660      * @return a string representation of the fraction.
661      * @see java.lang.Object#toString()
662      */
663     @Override
664     public String toString() {
665         String str = null;
666         if (denominator == 1) {
667             str = Integer.toString(numerator);
668         } else if (numerator == 0) {
669             str = "0";
670         } else {
671             str = numerator + " / " + denominator;
672         }
673         return str;
674     }
