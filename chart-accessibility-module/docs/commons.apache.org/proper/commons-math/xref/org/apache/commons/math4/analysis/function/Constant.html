18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.differentiation.DerivativeStructure;
21  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
22  
23  /**
24   * Constant function.
25   *
26   * @since 3.0
27   */
28  public class Constant implements UnivariateDifferentiableFunction {
29      /** Constant. */
30      private final double c;
31  
32      /**
33       * @param c Constant.
34       */
35      public Constant(double c) {
36          this.c = c;
37      }
38  
39      /** {@inheritDoc} */
40      @Override
41      public double value(double x) {
42          return c;
43      }
44  
45      /** {@inheritDoc}
46       * @since 3.1
47       */
48      @Override
49      public DerivativeStructure value(final DerivativeStructure t) {
50          return new DerivativeStructure(t.getFreeParameters(), t.getOrder(), c);
51      }
52  
53  }
