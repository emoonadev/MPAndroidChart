31  /**
32   * Immutable representation of a real polynomial function with real coefficients.
33   * <p>
34   * <a href="http://mathworld.wolfram.com/HornersMethod.html">Horner's Method</a>
35   * is used to evaluate the function.</p>
36   *
37   */
38  public class PolynomialFunction implements UnivariateDifferentiableFunction, Serializable {
39      /**
40       * Serialization identifier
41       */
42      private static final long serialVersionUID = -7726511984200295583L;
43      /**
44       * The coefficients of the polynomial, ordered by degree -- i.e.,
45       * coefficients[0] is the constant term and coefficients[n] is the
46       * coefficient of x^n where n is the degree of the polynomial.
47       */
48      private final double coefficients[];
49  
50      /**
51       * Construct a polynomial with the given coefficients.  The first element
52       * of the coefficients array is the constant term.  Higher degree
53       * coefficients follow in sequence.  The degree of the resulting polynomial
54       * is the index of the last non-null element of the array, or 0 if all elements
55       * are null.
56       * <p>
57       * The constructor makes a copy of the input array and assigns the copy to
58       * the coefficients property.</p>
59       *
60       * @param c Polynomial coefficients.
61       * @throws NullArgumentException if {@code c} is {@code null}.
62       * @throws NoDataException if {@code c} is empty.
63       */
64      public PolynomialFunction(double c[])
65          throws NullArgumentException, NoDataException {
66          super();
67          MathUtils.checkNotNull(c);
68          int n = c.length;
69          if (n == 0) {
70              throw new NoDataException(LocalizedFormats.EMPTY_POLYNOMIALS_COEFFICIENTS_ARRAY);
71          }
72          while ((n > 1) && (c[n - 1] == 0)) {
73              --n;
74          }
75          this.coefficients = new double[n];
76          System.arraycopy(c, 0, this.coefficients, 0, n);
77      }
78  
79      /**
80       * Compute the value of the function for the given argument.
81       * <p>
82       *  The value returned is </p><p>
83       *  {@code coefficients[n] * x^n + ... + coefficients[1] * x  + coefficients[0]}
84       * </p>
85       *
86       * @param x Argument for which the function value should be computed.
87       * @return the value of the polynomial at the given point.
88       *
89       * @see org.apache.commons.math4.analysis.UnivariateFunction#value(double)
90       */
91      @Override
92      public double value(double x) {
93         return evaluate(coefficients, x);
94      }
95  
96      /**
97       * Returns the degree of the polynomial.
98       *
99       * @return the degree of the polynomial.
100      */
101     public int degree() {
102         return coefficients.length - 1;
103     }
104 
105     /**
106      * Returns a copy of the coefficients array.
107      * <p>
108      * Changes made to the returned copy will not affect the coefficients of
109      * the polynomial.</p>
110      *
111      * @return a fresh copy of the coefficients array.
112      */
113     public double[] getCoefficients() {
114         return coefficients.clone();
115     }
116 
117     /**
118      * Uses Horner's Method to evaluate the polynomial with the given coefficients at
119      * the argument.
120      *
121      * @param coefficients Coefficients of the polynomial to evaluate.
122      * @param argument Input value.
123      * @return the value of the polynomial.
124      * @throws NoDataException if {@code coefficients} is empty.
125      * @throws NullArgumentException if {@code coefficients} is {@code null}.
126      */
127     protected static double evaluate(double[] coefficients, double argument)
128         throws NullArgumentException, NoDataException {
129         MathUtils.checkNotNull(coefficients);
130         int n = coefficients.length;
131         if (n == 0) {
132             throw new NoDataException(LocalizedFormats.EMPTY_POLYNOMIALS_COEFFICIENTS_ARRAY);
133         }
134         double result = coefficients[n - 1];
135         for (int j = n - 2; j >= 0; j--) {
136             result = argument * result + coefficients[j];
137         }
138         return result;
139     }

142     /** {@inheritDoc}
143      * @since 3.1
144      * @throws NoDataException if {@code coefficients} is empty.
145      * @throws NullArgumentException if {@code coefficients} is {@code null}.
146      */
147     @Override
148     public DerivativeStructure value(final DerivativeStructure t)
149         throws NullArgumentException, NoDataException {
150         MathUtils.checkNotNull(coefficients);
151         int n = coefficients.length;
152         if (n == 0) {
153             throw new NoDataException(LocalizedFormats.EMPTY_POLYNOMIALS_COEFFICIENTS_ARRAY);
154         }
155         DerivativeStructure result =
156                 new DerivativeStructure(t.getFreeParameters(), t.getOrder(), coefficients[n - 1]);
157         for (int j = n - 2; j >= 0; j--) {
158             result = result.multiply(t).add(coefficients[j]);
159         }
160         return result;
161     }

163     /**
164      * Add a polynomial to the instance.
165      *
166      * @param p Polynomial to add.
167      * @return a new polynomial which is the sum of the instance and {@code p}.
168      */
169     public PolynomialFunction add(final PolynomialFunction p) {
170         // identify the lowest degree polynomial
171         final int lowLength  = FastMath.min(coefficients.length, p.coefficients.length);
172         final int highLength = FastMath.max(coefficients.length, p.coefficients.length);
173 
174         // build the coefficients array
175         double[] newCoefficients = new double[highLength];
176         for (int i = 0; i < lowLength; ++i) {
177             newCoefficients[i] = coefficients[i] + p.coefficients[i];
178         }
179         System.arraycopy((coefficients.length < p.coefficients.length) ?
180                          p.coefficients : coefficients,
181                          lowLength,
182                          newCoefficients, lowLength,
183                          highLength - lowLength);
184 
185         return new PolynomialFunction(newCoefficients);
186     }
187 
188     /**
189      * Subtract a polynomial from the instance.
190      *
191      * @param p Polynomial to subtract.
192      * @return a new polynomial which is the instance minus {@code p}.
193      */
194     public PolynomialFunction subtract(final PolynomialFunction p) {
195         // identify the lowest degree polynomial
196         int lowLength  = FastMath.min(coefficients.length, p.coefficients.length);
197         int highLength = FastMath.max(coefficients.length, p.coefficients.length);
198 
199         // build the coefficients array
200         double[] newCoefficients = new double[highLength];
201         for (int i = 0; i < lowLength; ++i) {
202             newCoefficients[i] = coefficients[i] - p.coefficients[i];
203         }
204         if (coefficients.length < p.coefficients.length) {
205             for (int i = lowLength; i < highLength; ++i) {
206                 newCoefficients[i] = -p.coefficients[i];
207             }
208         } else {
209             System.arraycopy(coefficients, lowLength, newCoefficients, lowLength,
210                              highLength - lowLength);
211         }
212 
213         return new PolynomialFunction(newCoefficients);
214     }
215 
216     /**
217      * Negate the instance.
218      *
219      * @return a new polynomial with all coefficients negated
220      */
221     public PolynomialFunction negate() {
222         double[] newCoefficients = new double[coefficients.length];
223         for (int i = 0; i < coefficients.length; ++i) {
224             newCoefficients[i] = -coefficients[i];
225         }
226         return new PolynomialFunction(newCoefficients);
227     }
228 
229     /**
230      * Multiply the instance by a polynomial.
231      *
232      * @param p Polynomial to multiply by.
233      * @return a new polynomial equal to this times {@code p}
234      */
235     public PolynomialFunction multiply(final PolynomialFunction p) {
236         double[] newCoefficients = new double[coefficients.length + p.coefficients.length - 1];
237 
238         for (int i = 0; i < newCoefficients.length; ++i) {
239             newCoefficients[i] = 0.0;
240             for (int j = FastMath.max(0, i + 1 - p.coefficients.length);
241                  j < FastMath.min(coefficients.length, i + 1);
242                  ++j) {
243                 newCoefficients[i] += coefficients[j] * p.coefficients[i-j];
244             }
245         }
246 
247         return new PolynomialFunction(newCoefficients);
248     }

250     /**
251      * Returns the coefficients of the derivative of the polynomial with the given coefficients.
252      *
253      * @param coefficients Coefficients of the polynomial to differentiate.
254      * @return the coefficients of the derivative or {@code null} if coefficients has length 1.
255      * @throws NoDataException if {@code coefficients} is empty.
256      * @throws NullArgumentException if {@code coefficients} is {@code null}.
257      */
258     protected static double[] differentiate(double[] coefficients)
259         throws NullArgumentException, NoDataException {
260         MathUtils.checkNotNull(coefficients);
261         int n = coefficients.length;
262         if (n == 0) {
263             throw new NoDataException(LocalizedFormats.EMPTY_POLYNOMIALS_COEFFICIENTS_ARRAY);
264         }
265         if (n == 1) {
266             return new double[]{0};
267         }
268         double[] result = new double[n - 1];
269         for (int i = n - 1; i > 0; i--) {
270             result[i - 1] = i * coefficients[i];
271         }
272         return result;
273     }
274 
275     /**
276      * Returns the derivative as a {@link PolynomialFunction}.
277      *
278      * @return the derivative polynomial.
279      */
280     public PolynomialFunction polynomialDerivative() {
281         return new PolynomialFunction(differentiate(coefficients));
282     }

284     /**
285      * Returns a string representation of the polynomial.
286      *
287      * <p>The representation is user oriented. Terms are displayed lowest
288      * degrees first. The multiplications signs, coefficients equals to
289      * one and null terms are not displayed (except if the polynomial is 0,
290      * in which case the 0 constant term is displayed). Addition of terms
291      * with negative coefficients are replaced by subtraction of terms
292      * with positive coefficients except for the first displayed term
293      * (i.e. we display <code>-3</code> for a constant negative polynomial,
294      * but <code>1 - 3 x + x^2</code> if the negative coefficient is not
295      * the first one displayed).</p>
296      *
297      * @return a string representation of the polynomial.
298      */
299     @Override
300     public String toString() {
301         StringBuilder s = new StringBuilder();
302         if (coefficients[0] == 0.0) {
303             if (coefficients.length == 1) {
304                 return "0";
305             }
306         } else {
307             s.append(toString(coefficients[0]));
308         }
309 
310         for (int i = 1; i < coefficients.length; ++i) {
311             if (coefficients[i] != 0) {
312                 if (s.length() > 0) {
313                     if (coefficients[i] < 0) {
314                         s.append(" - ");
315                     } else {
316                         s.append(" + ");
317                     }
318                 } else {
319                     if (coefficients[i] < 0) {
320                         s.append("-");
321                     }
322                 }
323 
324                 double absAi = FastMath.abs(coefficients[i]);
325                 if ((absAi - 1) != 0) {
326                     s.append(toString(absAi));
327                     s.append(' ');
328                 }
329 
330                 s.append("x");
331                 if (i > 1) {
332                     s.append('^');
333                     s.append(Integer.toString(i));
334                 }
335             }
336         }
337 
338         return s.toString();
339     }

341     /**
342      * Creates a string representing a coefficient, removing ".0" endings.
343      *
344      * @param coeff Coefficient.
345      * @return a string representation of {@code coeff}.
346      */
347     private static String toString(double coeff) {
348         final String c = Double.toString(coeff);
349         if (c.endsWith(".0")) {
350             return c.substring(0, c.length() - 2);
351         } else {
352             return c;
353         }
354     }

356     /** {@inheritDoc} */
357     @Override
358     public int hashCode() {
359         final int prime = 31;
360         int result = 1;
361         result = prime * result + Arrays.hashCode(coefficients);
362         return result;
363     }
364 
365     /** {@inheritDoc} */
366     @Override
367     public boolean equals(Object obj) {
368         if (this == obj) {
369             return true;
370         }
371         if (!(obj instanceof PolynomialFunction)) {
372             return false;
373         }
374         PolynomialFunction other = (PolynomialFunction) obj;
375         if (!Arrays.equals(coefficients, other.coefficients)) {
376             return false;
377         }
378         return true;
379     }

381     /**
382      * Dedicated parametric polynomial class.
383      *
384      * @since 3.0
385      */
386     public static class Parametric implements ParametricUnivariateFunction {
387         /** {@inheritDoc} */
388         @Override
389         public double[] gradient(double x, double ... parameters) {
390             final double[] gradient = new double[parameters.length];
391             double xn = 1.0;
392             for (int i = 0; i < parameters.length; ++i) {
393                 gradient[i] = xn;
394                 xn *= x;
395             }
396             return gradient;
397         }
398 
399         /** {@inheritDoc} */
400         @Override
401         public double value(final double x, final double ... parameters)
402             throws NoDataException {
403             return PolynomialFunction.evaluate(parameters, x);
404         }
405     }
