17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import java.io.Serializable;
20  
21  import org.apache.commons.math4.exception.NullArgumentException;
22  import org.apache.commons.math4.util.MathUtils;
23  
24  /**
25   * Computes a statistic related to the Fourth Central Moment.  Specifically,
26   * what is computed is the sum of
27   * <p>
28   * (x_i - xbar) ^ 4, </p>
29   * <p>
30   * where the x_i are the
31   * sample observations and xbar is the sample mean. </p>
32   * <p>
33   * The following recursive updating formula is used: </p>
34   * <p>
35   * Let <ul>
36   * <li> dev = (current obs - previous mean) </li>
37   * <li> m2 = previous value of {@link SecondMoment} </li>
38   * <li> m2 = previous value of {@link ThirdMoment} </li>
39   * <li> n = number of observations (including current obs) </li>
40   * </ul>
41   * Then </p>
42   * <p>
43   * new value = old value - 4 * (dev/n) * m3 + 6 * (dev/n)^2 * m2 + <br>
44   * [n^2 - 3 * (n-1)] * dev^4 * (n-1) / n^3 </p>
45   * <p>
46   * Returns <code>Double.NaN</code> if no data values have been added and
47   * returns <code>0</code> if there is just one value in the data set. Note that
48   * Double.NaN may also be returned if the input includes NaN and / or infinite
49   * values. </p>
50   * <p>
51   * <strong>Note that this implementation is not synchronized.</strong> If
52   * multiple threads access an instance of this class concurrently, and at least
53   * one of the threads invokes the <code>increment()</code> or
54   * <code>clear()</code> method, it must be synchronized externally. </p>
55   */
56  class FourthMoment extends ThirdMoment implements Serializable{
57  
58      /** Serializable version identifier */
59      private static final long serialVersionUID = 20150412L;
60  
61      /** fourth moment of values that have been added */
62      private double m4;
63  
64      /**
65       * Create a FourthMoment instance
66       */
67      FourthMoment() {
68          super();
69          m4 = Double.NaN;
70      }
71  
72      /**
73       * Copy constructor, creates a new {@code FourthMoment} identical
74       * to the {@code original}.
75       *
76       * @param original the {@code FourthMoment} instance to copy
77       * @throws NullArgumentException if original is null
78       */
79       FourthMoment(FourthMoment original) throws NullArgumentException {
80           super();
81           copy(original, this);
82       }
83  
84      /**
85       * {@inheritDoc}
86       */
87       @Override
88      public void increment(final double d) {
89          if (n < 1) {
90              m4 = 0.0;
91              m3 = 0.0;
92              m2 = 0.0;
93              m1 = 0.0;
94          }
95  
96          double prevM3 = m3;
97          double prevM2 = m2;
98  
99          super.increment(d);
100 
101         double n0 = n;
102 
103         m4 = m4 - 4.0 * nDev * prevM3 + 6.0 * nDevSq * prevM2 +
104             ((n0 * n0) - 3 * (n0 -1)) * (nDevSq * nDevSq * (n0 - 1) * n0);
105     }
106 
107     /**
108      * {@inheritDoc}
109      */
110     @Override
111     public double getResult() {
112         return m4;
113     }
114 
115     /**
116      * {@inheritDoc}
117      */
118     @Override
119     public void clear() {
120         super.clear();
121         m4 = Double.NaN;
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     @Override
128     public FourthMoment copy() {
129         FourthMoment result = new FourthMoment();
130         // No try-catch or advertised exception because args are guaranteed non-null
131         copy(this, result);
132         return result;
133     }
134 
135     /**
136      * Copies source to dest.
137      * <p>Neither source nor dest can be null.</p>
138      *
139      * @param source FourthMoment to copy
140      * @param dest FourthMoment to copy to
141      * @throws NullArgumentException if either source or dest is null
142      */
143     public static void copy(FourthMoment source, FourthMoment dest)
144         throws NullArgumentException {
145         MathUtils.checkNotNull(source);
146         MathUtils.checkNotNull(dest);
147         ThirdMoment.copy(source, dest);
148         dest.m4 = source.m4;
149     }
150 }
