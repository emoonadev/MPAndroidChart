18  package org.apache.commons.math4.ml.clustering;
19  
20  import java.io.Serializable;
21  import java.util.Arrays;
22  
23  /**
24   * A simple implementation of {@link Clusterable} for points with double coordinates.
25   * @since 3.2
26   */
27  public class DoublePoint implements Clusterable, Serializable {
28  
29      /** Serializable version identifier. */
30      private static final long serialVersionUID = 3946024775784901369L;
31  
32      /** Point coordinates. */
33      private final double[] point;
34  
35      /**
36       * Build an instance wrapping an double array.
37       * <p>
38       * The wrapped array is referenced, it is <em>not</em> copied.
39       *
40       * @param point the n-dimensional point in double space
41       */
42      public DoublePoint(final double[] point) {
43          this.point = point;
44      }
45  
46      /**
47       * Build an instance wrapping an integer array.
48       * <p>
49       * The wrapped array is copied to an internal double array.
50       *
51       * @param point the n-dimensional point in integer space
52       */
53      public DoublePoint(final int[] point) {
54          this.point = new double[point.length];
55          for ( int i = 0; i < point.length; i++) {
56              this.point[i] = point[i];
57          }
58      }
59  
60      /** {@inheritDoc} */
61      @Override
62      public double[] getPoint() {
63          return point;
64      }
65  
66      /** {@inheritDoc} */
67      @Override
68      public boolean equals(final Object other) {
69          if (!(other instanceof DoublePoint)) {
70              return false;
71          }
72          return Arrays.equals(point, ((DoublePoint) other).point);
73      }
74  
75      /** {@inheritDoc} */
76      @Override
77      public int hashCode() {
78          return Arrays.hashCode(point);
79      }
80  
81      /** {@inheritDoc} */
82      @Override
83      public String toString() {
84          return Arrays.toString(point);
85      }
86  
87  }
