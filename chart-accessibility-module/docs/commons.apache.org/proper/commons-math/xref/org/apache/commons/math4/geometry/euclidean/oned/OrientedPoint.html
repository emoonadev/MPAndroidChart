17  package org.apache.commons.math4.geometry.euclidean.oned;
18  
19  import org.apache.commons.math4.geometry.Point;
20  import org.apache.commons.math4.geometry.Vector;
21  import org.apache.commons.math4.geometry.partitioning.Hyperplane;
22  
23  /** This class represents a 1D oriented hyperplane.
24   * <p>An hyperplane in 1D is a simple point, its orientation being a
25   * boolean.</p>
26   * <p>Instances of this class are guaranteed to be immutable.</p>
27   * @since 3.0
28   */
29  public class OrientedPoint implements Hyperplane<Euclidean1D> {
30  
31      /** Vector location. */
32      private final Vector1D location;
33  
34      /** Orientation. */
35      private boolean direct;
36  
37      /** Tolerance below which points are considered to belong to the hyperplane. */
38      private final double tolerance;
39  
40      /** Simple constructor.
41       * @param location location of the hyperplane
42       * @param direct if true, the plus side of the hyperplane is towards
43       * abscissas greater than {@code location}
44       * @param tolerance tolerance below which points are considered to belong to the hyperplane
45       * @since 3.3
46       */
47      public OrientedPoint(final Vector1D location, final boolean direct, final double tolerance) {
48          this.location  = location;
49          this.direct    = direct;
50          this.tolerance = tolerance;
51      }
52  
53      /** Copy the instance.
54       * <p>Since instances are immutable, this method directly returns
55       * the instance.</p>
56       * @return the instance itself
57       */
58      @Override
59      public OrientedPoint copySelf() {
60          return this;
61      }
62  
63      /** Get the offset (oriented distance) of a vector.
64       * @param vector vector to check
65       * @return offset of the vector
66       */
67      public double getOffset(Vector<Euclidean1D> vector) {
68          return getOffset((Point<Euclidean1D>) vector);
69      }
70  
71      /** {@inheritDoc} */
72      @Override
73      public double getOffset(final Point<Euclidean1D> point) {
74          final double delta = ((Vector1D) point).getX() - location.getX();
75          return direct ? delta : -delta;
76      }
77  
78      /** Build a region covering the whole hyperplane.
79       * <p>Since this class represent zero dimension spaces which does
80       * not have lower dimension sub-spaces, this method returns a dummy
81       * implementation of a {@link
82       * org.apache.commons.math4.geometry.partitioning.SubHyperplane SubHyperplane}.
83       * This implementation is only used to allow the {@link
84       * org.apache.commons.math4.geometry.partitioning.SubHyperplane
85       * SubHyperplane} class implementation to work properly, it should
86       * <em>not</em> be used otherwise.</p>
87       * @return a dummy sub hyperplane
88       */
89      @Override
90      public SubOrientedPoint wholeHyperplane() {
91          return new SubOrientedPoint(this, null);
92      }
93  
94      /** Build a region covering the whole space.
95       * @return a region containing the instance (really an {@link
96       * IntervalsSet IntervalsSet} instance)
97       */
98      @Override
99      public IntervalsSet wholeSpace() {
100         return new IntervalsSet(tolerance);
101     }
102 
103     /** {@inheritDoc} */
104     @Override
105     public boolean sameOrientationAs(final Hyperplane<Euclidean1D> other) {
106         return !(direct ^ ((OrientedPoint) other).direct);
107     }
108 
109     /** {@inheritDoc}
110      * @since 3.3
111      */
112     @Override
113     public Point<Euclidean1D> project(Point<Euclidean1D> point) {
114         return location;
115     }
116 
117     /** {@inheritDoc}
118      * @since 3.3
119      */
120     @Override
121     public double getTolerance() {
122         return tolerance;
123     }
124 
125     /** Get the hyperplane location on the real line.
126      * @return the hyperplane location
127      */
128     public Vector1D getLocation() {
129         return location;
130     }
131 
132     /** Check if the hyperplane orientation is direct.
133      * @return true if the plus side of the hyperplane is towards
134      * abscissae greater than hyperplane location
135      */
136     public boolean isDirect() {
137         return direct;
138     }
139 
140     /** Revert the instance.
141      */
142     public void revertSelf() {
143         direct = !direct;
144     }
145 
146 }
