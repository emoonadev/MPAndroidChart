18  package org.apache.commons.math4.analysis;
19  
20  /**
21   * An interface representing a trivariate real function.
22   *
23   * @since 2.2
24   */
25  public interface TrivariateFunction {
26      /**
27       * Compute the value for the function.
28       *
29       * @param x x-coordinate for which the function value should be computed.
30       * @param y y-coordinate for which the function value should be computed.
31       * @param z z-coordinate for which the function value should be computed.
32       * @return the value.
33       */
34      double value(double x, double y, double z);
35  }
