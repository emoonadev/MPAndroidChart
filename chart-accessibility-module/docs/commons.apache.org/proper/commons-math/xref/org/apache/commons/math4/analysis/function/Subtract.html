18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.BivariateFunction;
21  
22  /**
23   * Subtract the second operand from the first.
24   *
25   * @since 3.0
26   */
27  public class Subtract implements BivariateFunction {
28      /** {@inheritDoc} */
29      @Override
30      public double value(double x, double y) {
31          return x - y;
32      }
33  }
