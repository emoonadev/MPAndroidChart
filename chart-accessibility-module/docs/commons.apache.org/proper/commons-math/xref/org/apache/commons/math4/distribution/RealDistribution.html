17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.NumberIsTooLargeException;
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.rng.UniformRandomProvider;
22  
23  /**
24   * Base interface for distributions on the reals.
25   *
26   * @since 3.0
27   */
28  public interface RealDistribution {
29      /**
30       * For a random variable {@code X} whose values are distributed according
31       * to this distribution, this method returns {@code P(X = x)}. In other
32       * words, this method represents the probability mass function (PMF)
33       * for the distribution.
34       *
35       * @param x the point at which the PMF is evaluated
36       * @return the value of the probability mass function at point {@code x}
37       */
38      double probability(double x);
39  
40      /**
41       * For a random variable {@code X} whose values are distributed according
42       * to this distribution, this method returns {@code P(x0 < X <= x1)}.
43       *
44       * @param x0 the exclusive lower bound
45       * @param x1 the inclusive upper bound
46       * @return the probability that a random variable with this distribution
47       * takes a value between {@code x0} and {@code x1},
48       * excluding the lower and including the upper endpoint
49       * @throws NumberIsTooLargeException if {@code x0 > x1}
50       *
51       * @since 4.0, was previously named cumulativeProbability
52       */
53      double probability(double x0, double x1) throws NumberIsTooLargeException;
54  
55      /**
56       * Returns the probability density function (PDF) of this distribution
57       * evaluated at the specified point {@code x}. In general, the PDF is
58       * the derivative of the {@link #cumulativeProbability(double) CDF}.
59       * If the derivative does not exist at {@code x}, then an appropriate
60       * replacement should be returned, e.g. {@code Double.POSITIVE_INFINITY},
61       * {@code Double.NaN}, or  the limit inferior or limit superior of the
62       * difference quotient.
63       *
64       * @param x the point at which the PDF is evaluated
65       * @return the value of the probability density function at point {@code x}
66       */
67      double density(double x);
68  
69      /**
70       * Returns the natural logarithm of the probability density function
71       * (PDF) of this distribution evaluated at the specified point {@code x}.
72       * In general, the PDF is the derivative of the {@link #cumulativeProbability(double) CDF}.
73       * If the derivative does not exist at {@code x}, then an appropriate replacement
74       * should be returned, e.g. {@code Double.POSITIVE_INFINITY}, {@code Double.NaN},
75       * or the limit inferior or limit superior of the difference quotient. Note that
76       * due to the floating point precision and under/overflow issues, this method will
77       * for some distributions be more precise and faster than computing the logarithm of
78       * {@link #density(double)}.
79       *
80       * @param x the point at which the PDF is evaluated
81       * @return the logarithm of the value of the probability density function at point {@code x}
82       * @since 4.0
83       */
84      double logDensity(double x);
85  
86      /**
87       * For a random variable {@code X} whose values are distributed according
88       * to this distribution, this method returns {@code P(X <= x)}. In other
89       * words, this method represents the (cumulative) distribution function
90       * (CDF) for this distribution.
91       *
92       * @param x the point at which the CDF is evaluated
93       * @return the probability that a random variable with this
94       * distribution takes a value less than or equal to {@code x}
95       */
96      double cumulativeProbability(double x);
97  
98      /**
99       * Computes the quantile function of this distribution. For a random
100      * variable {@code X} distributed according to this distribution, the
101      * returned value is
102      * <ul>
103      * <li><code>inf{x in R | P(X<=x) >= p}</code> for {@code 0 < p <= 1},</li>
104      * <li><code>inf{x in R | P(X<=x) > 0}</code> for {@code p = 0}.</li>
105      * </ul>
106      *
107      * @param p the cumulative probability
108      * @return the smallest {@code p}-quantile of this distribution
109      * (largest 0-quantile for {@code p = 0})
110      * @throws OutOfRangeException if {@code p < 0} or {@code p > 1}
111      */
112     double inverseCumulativeProbability(double p) throws OutOfRangeException;
113 
114     /**
115      * Use this method to get the numerical value of the mean of this
116      * distribution.
117      *
118      * @return the mean or {@code Double.NaN} if it is not defined
119      */
120     double getNumericalMean();
121 
122     /**
123      * Use this method to get the numerical value of the variance of this
124      * distribution.
125      *
126      * @return the variance (possibly {@code Double.POSITIVE_INFINITY} as
127      * for certain cases in {@link TDistribution}) or {@code Double.NaN} if it
128      * is not defined
129      */
130     double getNumericalVariance();
131 
132     /**
133      * Access the lower bound of the support. This method must return the same
134      * value as {@code inverseCumulativeProbability(0)}. In other words, this
135      * method must return
136      * <p><code>inf {x in R | P(X <= x) > 0}</code>.</p>
137      *
138      * @return lower bound of the support (might be
139      * {@code Double.NEGATIVE_INFINITY})
140      */
141     double getSupportLowerBound();
142 
143     /**
144      * Access the upper bound of the support. This method must return the same
145      * value as {@code inverseCumulativeProbability(1)}. In other words, this
146      * method must return
147      * <p><code>inf {x in R | P(X <= x) = 1}</code>.</p>
148      *
149      * @return upper bound of the support (might be
150      * {@code Double.POSITIVE_INFINITY})
151      */
152     double getSupportUpperBound();
153 
154     /**
155      * Use this method to get information about whether the support is connected,
156      * i.e. whether all values between the lower and upper bound of the support
157      * are included in the support.
158      *
159      * @return whether the support is connected or not
160      */
161     boolean isSupportConnected();
162 
163     /**
164      * Creates a sampler.
165      *
166      * @param rng Generator of uniformly distributed numbers.
167      * @return a sampler that produces random numbers according this
168      * distribution.
169      */
170     Sampler createSampler(UniformRandomProvider rng);
171 
172     /**
173      * Sampling functionality.
174      */
175     interface Sampler {
176         /**
177          * Generates a random value sampled from this distribution.
178          *
179          * @return a random value.
180          */
181         double sample();
182     }
183 }
