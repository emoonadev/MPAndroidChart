32  /** The class represent planes in a three dimensional space.
33   * @since 3.0
34   */
35  public class Plane implements Hyperplane<Euclidean3D>, Embedding<Euclidean3D, Euclidean2D> {
36  
37      /** Offset of the origin with respect to the plane. */
38      private double originOffset;
39  
40      /** Origin of the plane frame. */
41      private Vector3D origin;
42  
43      /** First vector of the plane frame (in plane). */
44      private Vector3D u;
45  
46      /** Second vector of the plane frame (in plane). */
47      private Vector3D v;
48  
49      /** Third vector of the plane frame (plane normal). */
50      private Vector3D w;
51  
52      /** Tolerance below which points are considered identical. */
53      private final double tolerance;
54  
55      /** Build a plane normal to a given direction and containing the origin.
56       * @param normal normal direction to the plane
57       * @param tolerance tolerance below which points are considered identical
58       * @exception MathArithmeticException if the normal norm is too small
59       * @since 3.3
60       */
61      public Plane(final Vector3D normal, final double tolerance)
62          throws MathArithmeticException {
63          setNormal(normal);
64          this.tolerance = tolerance;
65          originOffset = 0;
66          setFrame();
67      }
68  
69      /** Build a plane from a point and a normal.
70       * @param p point belonging to the plane
71       * @param normal normal direction to the plane
72       * @param tolerance tolerance below which points are considered identical
73       * @exception MathArithmeticException if the normal norm is too small
74       * @since 3.3
75       */
76      public Plane(final Vector3D p, final Vector3D normal, final double tolerance)
77          throws MathArithmeticException {
78          setNormal(normal);
79          this.tolerance = tolerance;
80          originOffset = -p.dotProduct(w);
81          setFrame();
82      }
83  
84      /** Build a plane from three points.
85       * <p>The plane is oriented in the direction of
86       * {@code (p2-p1) ^ (p3-p1)}</p>
87       * @param p1 first point belonging to the plane
88       * @param p2 second point belonging to the plane
89       * @param p3 third point belonging to the plane
90       * @param tolerance tolerance below which points are considered identical
91       * @exception MathArithmeticException if the points do not constitute a plane
92       * @since 3.3
93       */
94      public Plane(final Vector3D p1, final Vector3D p2, final Vector3D p3, final double tolerance)
95          throws MathArithmeticException {
96          this(p1, p2.subtract(p1).crossProduct(p3.subtract(p1)), tolerance);
97      }
98  
99      /** Copy constructor.
100      * <p>The instance created is completely independent of the original
101      * one. A deep copy is used, none of the underlying object are
102      * shared.</p>
103      * @param plane plane to copy
104      */
105     public Plane(final Plane plane) {
106         originOffset = plane.originOffset;
107         origin       = plane.origin;
108         u            = plane.u;
109         v            = plane.v;
110         w            = plane.w;
111         tolerance    = plane.tolerance;
112     }
113 
114     /** Copy the instance.
115      * <p>The instance created is completely independant of the original
116      * one. A deep copy is used, none of the underlying objects are
117      * shared (except for immutable objects).</p>
118      * @return a new hyperplane, copy of the instance
119      */
120     @Override
121     public Plane copySelf() {
122         return new Plane(this);
123     }
124 
125     /** Reset the instance as if built from a point and a normal.
126      * @param p point belonging to the plane
127      * @param normal normal direction to the plane
128      * @exception MathArithmeticException if the normal norm is too small
129      */
130     public void reset(final Vector3D p, final Vector3D normal) throws MathArithmeticException {
131         setNormal(normal);
132         originOffset = -p.dotProduct(w);
133         setFrame();
134     }
135 
136     /** Reset the instance from another one.
137      * <p>The updated instance is completely independant of the original
138      * one. A deep reset is used none of the underlying object is
139      * shared.</p>
140      * @param original plane to reset from
141      */
142     public void reset(final Plane original) {
143         originOffset = original.originOffset;
144         origin       = original.origin;
145         u            = original.u;
146         v            = original.v;
147         w            = original.w;
148     }
149 
150     /** Set the normal vactor.
151      * @param normal normal direction to the plane (will be copied)
152      * @exception MathArithmeticException if the normal norm is too small
153      */
154     private void setNormal(final Vector3D normal) throws MathArithmeticException {
155         final double norm = normal.getNorm();
156         if (norm < 1.0e-10) {
157             throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
158         }
159         w = new Vector3D(1.0 / norm, normal);
160     }
161 
162     /** Reset the plane frame.
163      */
164     private void setFrame() {
165         origin = new Vector3D(-originOffset, w);
166         u = w.orthogonal();
167         v = Vector3D.crossProduct(w, u);
168     }
169 
170     /** Get the origin point of the plane frame.
171      * <p>The point returned is the orthogonal projection of the
172      * 3D-space origin in the plane.</p>
173      * @return the origin point of the plane frame (point closest to the
174      * 3D-space origin)
175      */
176     public Vector3D getOrigin() {
177         return origin;
178     }
179 
180     /** Get the normalized normal vector.
181      * <p>The frame defined by ({@link #getU getU}, {@link #getV getV},
182      * {@link #getNormal getNormal}) is a rigth-handed orthonormalized
183      * frame).</p>
184      * @return normalized normal vector
185      * @see #getU
186      * @see #getV
187      */
188     public Vector3D getNormal() {
189         return w;
190     }
191 
192     /** Get the plane first canonical vector.
193      * <p>The frame defined by ({@link #getU getU}, {@link #getV getV},
194      * {@link #getNormal getNormal}) is a rigth-handed orthonormalized
195      * frame).</p>
196      * @return normalized first canonical vector
197      * @see #getV
198      * @see #getNormal
199      */
200     public Vector3D getU() {
201         return u;
202     }
203 
204     /** Get the plane second canonical vector.
205      * <p>The frame defined by ({@link #getU getU}, {@link #getV getV},
206      * {@link #getNormal getNormal}) is a rigth-handed orthonormalized
207      * frame).</p>
208      * @return normalized second canonical vector
209      * @see #getU
210      * @see #getNormal
211      */
212     public Vector3D getV() {
213         return v;
214     }

221         return toSpace(toSubSpace(point));
222     }

228     public double getTolerance() {
229         return tolerance;

232     /** Revert the plane.
233      * <p>Replace the instance by a similar plane with opposite orientation.</p>
234      * <p>The new plane frame is chosen in such a way that a 3D point that had
235      * {@code (x, y)} in-plane coordinates and {@code z} offset with
236      * respect to the plane and is unaffected by the change will have
237      * {@code (y, x)} in-plane coordinates and {@code -z} offset with
238      * respect to the new plane. This means that the {@code u} and {@code v}
239      * vectors returned by the {@link #getU} and {@link #getV} methods are exchanged,
240      * and the {@code w} vector returned by the {@link #getNormal} method is
241      * reversed.</p>
242      */
243     public void revertSelf() {
244         final Vector3D tmp = u;
245         u = v;
246         v = tmp;
247         w = w.negate();
248         originOffset = -originOffset;
249     }

251     /** Transform a space point into a sub-space point.
252      * @param vector n-dimension point of the space
253      * @return (n-1)-dimension point of the sub-space corresponding to
254      * the specified space point
255      */
256     public Vector2D toSubSpace(Vector<Euclidean3D> vector) {
257         return toSubSpace((Point<Euclidean3D>) vector);
258     }

260     /** Transform a sub-space point into a space point.
261      * @param vector (n-1)-dimension point of the sub-space
262      * @return n-dimension point of the space corresponding to the
263      * specified sub-space point
264      */
265     public Vector3D toSpace(Vector<Euclidean2D> vector) {
266         return toSpace((Point<Euclidean2D>) vector);
267     }
268 
269     /** Transform a 3D space point into an in-plane point.
270      * @param point point of the space (must be a {@link Vector3D
271      * Vector3D} instance)
272      * @return in-plane point (really a {@link
273      * org.apache.commons.math4.geometry.euclidean.twod.Vector2D Vector2D} instance)
274      * @see #toSpace
275      */
276     @Override
277     public Vector2D toSubSpace(final Point<Euclidean3D> point) {
278         final Vector3D p3D = (Vector3D) point;
279         return new Vector2D(p3D.dotProduct(u), p3D.dotProduct(v));
280     }
281 
282     /** Transform an in-plane point into a 3D space point.
283      * @param point in-plane point (must be a {@link
284      * org.apache.commons.math4.geometry.euclidean.twod.Vector2D Vector2D} instance)
285      * @return 3D space point (really a {@link Vector3D Vector3D} instance)
286      * @see #toSubSpace
287      */
288     @Override
289     public Vector3D toSpace(final Point<Euclidean2D> point) {
290         final Vector2D p2D = (Vector2D) point;
291         return new Vector3D(p2D.getX(), u, p2D.getY(), v, -originOffset, w);
292     }
293 
294     /** Get one point from the 3D-space.
295      * @param inPlane desired in-plane coordinates for the point in the
296      * plane
297      * @param offset desired offset for the point
298      * @return one point in the 3D-space, with given coordinates and offset
299      * relative to the plane
300      */
301     public Vector3D getPointAt(final Vector2D inPlane, final double offset) {
302         return new Vector3D(inPlane.getX(), u, inPlane.getY(), v, offset - originOffset, w);
303     }

305     /** Check if the instance is similar to another plane.
306      * <p>Planes are considered similar if they contain the same
307      * points. This does not mean they are equal since they can have
308      * opposite normals.</p>
309      * @param plane plane to which the instance is compared
310      * @return true if the planes are similar
311      */
312     public boolean isSimilarTo(final Plane plane) {
313         final double angle = Vector3D.angle(w, plane.w);
314         return ((angle < 1.0e-10) && (FastMath.abs(originOffset - plane.originOffset) < tolerance)) ||
315                ((angle > (FastMath.PI - 1.0e-10)) && (FastMath.abs(originOffset + plane.originOffset) < tolerance));
316     }
317 
318     /** Rotate the plane around the specified point.
319      * <p>The instance is not modified, a new instance is created.</p>
320      * @param center rotation center
321      * @param rotation vectorial rotation operator
322      * @return a new plane
323      */
324     public Plane rotate(final Vector3D center, final Rotation rotation) {
325 
326         final Vector3D delta = origin.subtract(center);
327         final Plane plane = new Plane(center.add(rotation.applyTo(delta)),
328                                       rotation.applyTo(w), tolerance);
329 
330         // make sure the frame is transformed as desired
331         plane.u = rotation.applyTo(u);
332         plane.v = rotation.applyTo(v);
333 
334         return plane;
335 
336     }
337 
338     /** Translate the plane by the specified amount.
339      * <p>The instance is not modified, a new instance is created.</p>
340      * @param translation translation to apply
341      * @return a new plane
342      */
343     public Plane translate(final Vector3D translation) {
344 
345         final Plane plane = new Plane(origin.add(translation), w, tolerance);
346 
347         // make sure the frame is transformed as desired
348         plane.u = u;
349         plane.v = v;
350 
351         return plane;
352 
353     }

355     /** Get the intersection of a line with the instance.
356      * @param line line intersecting the instance
357      * @return intersection point between between the line and the
358      * instance (null if the line is parallel to the instance)
359      */
360     public Vector3D intersection(final Line line) {
361         final Vector3D direction = line.getDirection();
362         final double   dot       = w.dotProduct(direction);
363         if (FastMath.abs(dot) < 1.0e-10) {
364             return null;
365         }
366         final Vector3D point = line.toSpace((Point<Euclidean1D>) Vector1D.ZERO);
367         final double   k     = -(originOffset + w.dotProduct(point)) / dot;
368         return new Vector3D(1.0, point, k, direction);
369     }
370 
371     /** Build the line shared by the instance and another plane.
372      * @param other other plane
373      * @return line at the intersection of the instance and the
374      * other plane (really a {@link Line Line} instance)
375      */
376     public Line intersection(final Plane other) {
377         final Vector3D direction = Vector3D.crossProduct(w, other.w);
378         if (direction.getNorm() < tolerance) {
379             return null;
380         }
381         final Vector3D point = intersection(this, other, new Plane(direction, tolerance));
382         return new Line(point, point.add(direction), tolerance);
383     }

385     /** Get the intersection point of three planes.
386      * @param plane1 first plane1
387      * @param plane2 second plane2
388      * @param plane3 third plane2
389      * @return intersection point of three planes, null if some planes are parallel
390      */
391     public static Vector3D intersection(final Plane plane1, final Plane plane2, final Plane plane3) {
392 
393         // coefficients of the three planes linear equations
394         final double a1 = plane1.w.getX();
395         final double b1 = plane1.w.getY();
396         final double c1 = plane1.w.getZ();
397         final double d1 = plane1.originOffset;
398 
399         final double a2 = plane2.w.getX();
400         final double b2 = plane2.w.getY();
401         final double c2 = plane2.w.getZ();
402         final double d2 = plane2.originOffset;
403 
404         final double a3 = plane3.w.getX();
405         final double b3 = plane3.w.getY();
406         final double c3 = plane3.w.getZ();
407         final double d3 = plane3.originOffset;
408 
409         // direct Cramer resolution of the linear system
410         // (this is still feasible for a 3x3 system)
411         final double a23         = b2 * c3 - b3 * c2;
412         final double b23         = c2 * a3 - c3 * a2;
413         final double c23         = a2 * b3 - a3 * b2;
414         final double determinant = a1 * a23 + b1 * b23 + c1 * c23;
415         if (FastMath.abs(determinant) < 1.0e-10) {
416             return null;
417         }
418 
419         final double r = 1.0 / determinant;
420         return new Vector3D(
421                             (-a23 * d1 - (c1 * b3 - c3 * b1) * d2 - (c2 * b1 - c1 * b2) * d3) * r,
422                             (-b23 * d1 - (c3 * a1 - c1 * a3) * d2 - (c1 * a2 - c2 * a1) * d3) * r,
423                             (-c23 * d1 - (b1 * a3 - b3 * a1) * d2 - (b2 * a1 - b1 * a2) * d3) * r);
424 
425     }

444     /** Check if the instance contains a point.
445      * @param p point to check
446      * @return true if p belongs to the plane
447      */
448     public boolean contains(final Vector3D p) {
449         return FastMath.abs(getOffset(p)) < tolerance;
450     }

452     /** Get the offset (oriented distance) of a parallel plane.
453      * <p>This method should be called only for parallel planes otherwise
454      * the result is not meaningful.</p>
455      * <p>The offset is 0 if both planes are the same, it is
456      * positive if the plane is on the plus side of the instance and
457      * negative if it is on the minus side, according to its natural
458      * orientation.</p>
459      * @param plane plane to check
460      * @return offset of the plane
461      */
462     public double getOffset(final Plane plane) {
463         return originOffset + (sameOrientationAs(plane) ? -plane.originOffset : plane.originOffset);
464     }

466     /** Get the offset (oriented distance) of a vector.

470     public double getOffset(Vector<Euclidean3D> vector) {
471         return getOffset((Point<Euclidean3D>) vector);
472     }

474     /** Get the offset (oriented distance) of a point.
475      * <p>The offset is 0 if the point is on the underlying hyperplane,
476      * it is positive if the point is on one particular side of the
477      * hyperplane, and it is negative if the point is on the other side,
478      * according to the hyperplane natural orientation.</p>
479      * @param point point to check
480      * @return offset of the point
481      */
482     @Override
483     public double getOffset(final Point<Euclidean3D> point) {
484         return ((Vector3D) point).dotProduct(w) + originOffset;
485     }

487     /** Check if the instance has the same orientation as another hyperplane.
488      * @param other other hyperplane to check against the instance
489      * @return true if the instance and the other hyperplane have
490      * the same orientation
491      */
492     @Override
493     public boolean sameOrientationAs(final Hyperplane<Euclidean3D> other) {
494         return (((Plane) other).w).dotProduct(w) > 0.0;
495     }
