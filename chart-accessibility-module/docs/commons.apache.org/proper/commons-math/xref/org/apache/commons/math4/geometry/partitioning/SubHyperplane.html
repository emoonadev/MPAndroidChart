17  package org.apache.commons.math4.geometry.partitioning;
18  
19  import org.apache.commons.math4.geometry.Space;
20  
21  /** This interface represents the remaining parts of an hyperplane after
22   * other parts have been chopped off.
23  
24   * <p>sub-hyperplanes are obtained when parts of an {@link
25   * Hyperplane hyperplane} are chopped off by other hyperplanes that
26   * intersect it. The remaining part is a convex region. Such objects
27   * appear in {@link BSPTree BSP trees} as the intersection of a cut
28   * hyperplane with the convex region which it splits, the chopping
29   * hyperplanes are the cut hyperplanes closer to the tree root.</p>
30  
31   * <p>
32   * Note that this interface is <em>not</em> intended to be implemented
33   * by Apache Commons Math users, it is only intended to be implemented
34   * within the library itself. New methods may be added even for minor
35   * versions, which breaks compatibility for external implementations.
36   * </p>
37  
38   * @param <S> Type of the embedding space.
39  
40   * @since 3.0
41   */
42  public interface SubHyperplane<S extends Space> {
43  
44      /** Copy the instance.
45       * <p>The instance created is completely independent of the original
46       * one. A deep copy is used, none of the underlying objects are
47       * shared (except for the nodes attributes and immutable
48       * objects).</p>
49       * @return a new sub-hyperplane, copy of the instance
50       */
51      SubHyperplane<S> copySelf();
52  
53      /** Get the underlying hyperplane.
54       * @return underlying hyperplane
55       */
56      Hyperplane<S> getHyperplane();
57  
58      /** Check if the instance is empty.
59       * @return true if the instance is empty
60       */
61      boolean isEmpty();
62  
63      /** Get the size of the instance.
64       * @return the size of the instance (this is a length in 1D, an area
65       * in 2D, a volume in 3D ...)
66       */
67      double getSize();
68  
69      /** Split the instance in two parts by an hyperplane.
70       * @param hyperplane splitting hyperplane
71       * @return an object containing both the part of the instance
72       * on the plus side of the hyperplane and the part of the
73       * instance on the minus side of the hyperplane
74       */
75      SplitSubHyperplane<S> split(Hyperplane<S> hyperplane);
76  
77      /** Compute the union of the instance and another sub-hyperplane.
78       * @param other other sub-hyperplane to union (<em>must</em> be in the
79       * same hyperplane as the instance)
80       * @return a new sub-hyperplane, union of the instance and other
81       */
82      SubHyperplane<S> reunite(SubHyperplane<S> other);
83  
84      /** Class holding the results of the {@link #split split} method.
85       * @param <U> Type of the embedding space.
86       */
87      class SplitSubHyperplane<U extends Space> {
88  
89          /** Part of the sub-hyperplane on the plus side of the splitting hyperplane. */
90          private final SubHyperplane<U> plus;
91  
92          /** Part of the sub-hyperplane on the minus side of the splitting hyperplane. */
93          private final SubHyperplane<U> minus;
94  
95          /** Build a SplitSubHyperplane from its parts.
96           * @param plus part of the sub-hyperplane on the plus side of the
97           * splitting hyperplane
98           * @param minus part of the sub-hyperplane on the minus side of the
99           * splitting hyperplane
100          */
101         public SplitSubHyperplane(final SubHyperplane<U> plus,
102                                   final SubHyperplane<U> minus) {
103             this.plus  = plus;
104             this.minus = minus;
105         }
106 
107         /** Get the part of the sub-hyperplane on the plus side of the splitting hyperplane.
108          * @return part of the sub-hyperplane on the plus side of the splitting hyperplane
109          */
110         public SubHyperplane<U> getPlus() {
111             return plus;
112         }
113 
114         /** Get the part of the sub-hyperplane on the minus side of the splitting hyperplane.
115          * @return part of the sub-hyperplane on the minus side of the splitting hyperplane
116          */
117         public SubHyperplane<U> getMinus() {
118             return minus;
119         }
120 
121         /** Get the side of the split sub-hyperplane with respect to its splitter.
122          * @return {@link Side#PLUS} if only {@link #getPlus()} is neither null nor empty,
123          * {@link Side#MINUS} if only {@link #getMinus()} is neither null nor empty,
124          * {@link Side#BOTH} if both {@link #getPlus()} and {@link #getMinus()}
125          * are neither null nor empty or {@link Side#HYPER} if both {@link #getPlus()} and
126          * {@link #getMinus()} are either null or empty
127          * @since 3.6
128          */
129         public Side getSide() {
130             if (plus != null && !plus.isEmpty()) {
131                 if (minus != null && !minus.isEmpty()) {
132                     return Side.BOTH;
133                 } else {
134                     return Side.PLUS;
135                 }
136             } else if (minus != null && !minus.isEmpty()) {
137                 return Side.MINUS;
138             } else {
139                 return Side.HYPER;
140             }
141         }
142 
143     }
144 
145 }
