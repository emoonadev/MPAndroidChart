27  /** This abstract class represents an interpolator over the last step
28   * during an ODE integration.
29   *
30   * <p>The various ODE integrators provide objects extending this class
31   * to the step handlers. The handlers can use these objects to
32   * retrieve the state vector at intermediate times between the
33   * previous and the current grid points (dense output).</p>
34   *
35   * @see org.apache.commons.math4.ode.FirstOrderIntegrator
36   * @see org.apache.commons.math4.ode.SecondOrderIntegrator
37   * @see StepHandler
38   *
39   * @since 1.2
40   *
41   */
42

100   /** Simple constructor.
101    * This constructor builds an instance that is not usable yet, the
102    * {@link #reinitialize} method should be called before using the
103    * instance in order to initialize the internal arrays. This
104    * constructor is used only in order to delay the initialization in
105    * some cases. As an example, the {@link
106    * org.apache.commons.math4.ode.nonstiff.EmbeddedRungeKuttaIntegrator}
107    * class uses the prototyping design pattern to create the step
108    * interpolators by cloning an uninitialized model and latter
109    * initializing the copy.
110    */
111   protected AbstractStepInterpolator() {
112     globalPreviousTime = Double.NaN;
113     globalCurrentTime  = Double.NaN;
114     softPreviousTime   = Double.NaN;
115     softCurrentTime    = Double.NaN;
116     h                  = Double.NaN;
117     interpolatedTime   = Double.NaN;
118     currentState       = null;
119     finalized          = false;
120     this.forward       = true;
121     this.dirtyState    = true;
122     primaryMapper      = null;
123     secondaryMappers   = null;
124     allocateInterpolatedArrays(-1);
125   }

127   /** Simple constructor.
128    * @param y reference to the integrator array holding the state at
129    * the end of the step
130    * @param forward integration direction indicator
131    * @param primaryMapper equations mapper for the primary equations set
132    * @param secondaryMappers equations mappers for the secondary equations sets
133    */
134   protected AbstractStepInterpolator(final double[] y, final boolean forward,
135                                      final EquationsMapper primaryMapper,
136                                      final EquationsMapper[] secondaryMappers) {
137 
138     globalPreviousTime    = Double.NaN;
139     globalCurrentTime     = Double.NaN;
140     softPreviousTime      = Double.NaN;
141     softCurrentTime       = Double.NaN;
142     h                     = Double.NaN;
143     interpolatedTime      = Double.NaN;
144     currentState          = y;
145     finalized             = false;
146     this.forward          = forward;
147     this.dirtyState       = true;
148     this.primaryMapper    = primaryMapper;
149     this.secondaryMappers = (secondaryMappers == null) ? null : secondaryMappers.clone();
150     allocateInterpolatedArrays(y.length);
151 
152   }
153 
154   /** Copy constructor.
155 
156    * <p>The copied interpolator should have been finalized before the
157    * copy, otherwise the copy will not be able to perform correctly
158    * any derivative computation and will throw a {@link
159    * NullPointerException} later. Since we don't want this constructor
160    * to throw the exceptions finalization may involve and since we
161    * don't want this method to modify the state of the copied
162    * interpolator, finalization is <strong>not</strong> done
163    * automatically, it remains under user control.</p>
164 
165    * <p>The copy is a deep copy: its arrays are separated from the
166    * original arrays of the instance.</p>
167 
168    * @param interpolator interpolator to copy from.
169 
170    */
171   protected AbstractStepInterpolator(final AbstractStepInterpolator interpolator) {
172 
173     globalPreviousTime = interpolator.globalPreviousTime;
174     globalCurrentTime  = interpolator.globalCurrentTime;
175     softPreviousTime   = interpolator.softPreviousTime;
176     softCurrentTime    = interpolator.softCurrentTime;
177     h                  = interpolator.h;
178     interpolatedTime   = interpolator.interpolatedTime;
179 
180     if (interpolator.currentState == null) {
181         currentState     = null;
182         primaryMapper    = null;
183         secondaryMappers = null;
184         allocateInterpolatedArrays(-1);
185     } else {
186       currentState                     = interpolator.currentState.clone();
187       interpolatedState                = interpolator.interpolatedState.clone();
188       interpolatedDerivatives          = interpolator.interpolatedDerivatives.clone();
189       interpolatedPrimaryState         = interpolator.interpolatedPrimaryState.clone();
190       interpolatedPrimaryDerivatives   = interpolator.interpolatedPrimaryDerivatives.clone();
191       interpolatedSecondaryState       = new double[interpolator.interpolatedSecondaryState.length][];
192       interpolatedSecondaryDerivatives = new double[interpolator.interpolatedSecondaryDerivatives.length][];
193       for (int i = 0; i < interpolatedSecondaryState.length; ++i) {
194           interpolatedSecondaryState[i]       = interpolator.interpolatedSecondaryState[i].clone();
195           interpolatedSecondaryDerivatives[i] = interpolator.interpolatedSecondaryDerivatives[i].clone();
196       }
197     }
198 
199     finalized        = interpolator.finalized;
200     forward          = interpolator.forward;
201     dirtyState       = interpolator.dirtyState;
202     primaryMapper    = interpolator.primaryMapper;
203     secondaryMappers = (interpolator.secondaryMappers == null) ?
204                        null : interpolator.secondaryMappers.clone();
205 
206   }
207 
208   /** Allocate the various interpolated states arrays.
209    * @param dimension total dimension (negative if arrays should be set to null)
210    */
211   private void allocateInterpolatedArrays(final int dimension) {
212       if (dimension < 0) {
213           interpolatedState                = null;
214           interpolatedDerivatives          = null;
215           interpolatedPrimaryState         = null;
216           interpolatedPrimaryDerivatives   = null;
217           interpolatedSecondaryState       = null;
218           interpolatedSecondaryDerivatives = null;
219       } else {
220           interpolatedState                = new double[dimension];
221           interpolatedDerivatives          = new double[dimension];
222           interpolatedPrimaryState         = new double[primaryMapper.getDimension()];
223           interpolatedPrimaryDerivatives   = new double[primaryMapper.getDimension()];
224           if (secondaryMappers == null) {
225               interpolatedSecondaryState       = null;
226               interpolatedSecondaryDerivatives = null;
227           } else {
228               interpolatedSecondaryState       = new double[secondaryMappers.length][];
229               interpolatedSecondaryDerivatives = new double[secondaryMappers.length][];
230               for (int i = 0; i < secondaryMappers.length; ++i) {
231                   interpolatedSecondaryState[i]       = new double[secondaryMappers[i].getDimension()];
232                   interpolatedSecondaryDerivatives[i] = new double[secondaryMappers[i].getDimension()];
233               }
234           }
235       }
236   }

238   /** Reinitialize the instance
239    * @param y reference to the integrator array holding the state at the end of the step
240    * @param isForward integration direction indicator
241    * @param primary equations mapper for the primary equations set
242    * @param secondary equations mappers for the secondary equations sets
243    */
244   protected void reinitialize(final double[] y, final boolean isForward,
245                               final EquationsMapper primary,
246                               final EquationsMapper[] secondary) {
247 
248     globalPreviousTime    = Double.NaN;
249     globalCurrentTime     = Double.NaN;
250     softPreviousTime      = Double.NaN;
251     softCurrentTime       = Double.NaN;
252     h                     = Double.NaN;
253     interpolatedTime      = Double.NaN;
254     currentState          = y;
255     finalized             = false;
256     this.forward          = isForward;
257     this.dirtyState       = true;
258     this.primaryMapper    = primary;
259     this.secondaryMappers = secondary.clone();
260     allocateInterpolatedArrays(y.length);
261 
262   }

264   /** {@inheritDoc} */
265   @Override
266    public StepInterpolator copy() throws MaxCountExceededException {
267 
268      // finalize the step before performing copy
269      finalizeStep();
270 
271      // create the new independent instance
272      return doCopy();
273 
274    }

285   /** Shift one step forward.
286    * Copy the current time into the previous time, hence preparing the
287    * interpolator for future calls to {@link #storeTime storeTime}
288    */
289   public void shift() {
290     globalPreviousTime = globalCurrentTime;
291     softPreviousTime   = globalPreviousTime;
292     softCurrentTime    = globalCurrentTime;
293   }

295   /** Store the current step time.
296    * @param t current time
297    */
298   public void storeTime(final double t) {
299 
300     globalCurrentTime = t;
301     softCurrentTime   = globalCurrentTime;
302     h                 = globalCurrentTime - globalPreviousTime;
303     setInterpolatedTime(t);
304 
305     // the step is not finalized anymore
306     finalized  = false;
307 
308   }

391   /** Compute the state and derivatives at the interpolated time.
392    * This is the main processing method that should be implemented by
393    * the derived classes to perform the interpolation.
394    * @param theta normalized interpolation abscissa within the step
395    * (theta is zero at the previous time step and one at the current time step)
396    * @param oneMinusThetaH time gap between the interpolated time and
397    * the current time
398    * @exception MaxCountExceededException if the number of functions evaluations is exceeded
399    */
400   protected abstract void computeInterpolatedStateAndDerivatives(double theta,
401                                                                  double oneMinusThetaH)
402       throws MaxCountExceededException;
403 
404   /** Lazy evaluation of complete interpolated state.
405    * @exception MaxCountExceededException if the number of functions evaluations is exceeded
406    */
407   private void evaluateCompleteInterpolatedState()
408       throws MaxCountExceededException {
409       // lazy evaluation of the state
410       if (dirtyState) {
411           final double oneMinusThetaH = globalCurrentTime - interpolatedTime;
412           final double theta = (h == 0) ? 0 : (h - oneMinusThetaH) / h;
413           computeInterpolatedStateAndDerivatives(theta, oneMinusThetaH);
414           dirtyState = false;
415       }
416   }
417 
418   /** {@inheritDoc} */
419   @Override
420   public double[] getInterpolatedState() throws MaxCountExceededException {
421       evaluateCompleteInterpolatedState();
422       primaryMapper.extractEquationData(interpolatedState,
423                                         interpolatedPrimaryState);
424       return interpolatedPrimaryState;
425   }
426 
427   /** {@inheritDoc} */
428   @Override
429   public double[] getInterpolatedDerivatives() throws MaxCountExceededException {
430       evaluateCompleteInterpolatedState();
431       primaryMapper.extractEquationData(interpolatedDerivatives,
432                                         interpolatedPrimaryDerivatives);
433       return interpolatedPrimaryDerivatives;
434   }

454   /**
455    * Finalize the step.
456 
457    * <p>Some embedded Runge-Kutta integrators need fewer functions
458    * evaluations than their counterpart step interpolators. These
459    * interpolators should perform the last evaluations they need by
460    * themselves only if they need them. This method triggers these
461    * extra evaluations. It can be called directly by the user step
462    * handler and it is called automatically if {@link
463    * #setInterpolatedTime} is called.</p>
464 
465    * <p>Once this method has been called, <strong>no</strong> other
466    * evaluation will be performed on this step. If there is a need to
467    * have some side effects between the step handler and the
468    * differential equations (for example update some data in the
469    * equations once the step has been done), it is advised to call
470    * this method explicitly from the step handler before these side
471    * effects are set up. If the step handler induces no side effect,
472    * then this method can safely be ignored, it will be called
473    * transparently as needed.</p>
474 
475    * <p><strong>Warning</strong>: since the step interpolator provided
476    * to the step handler as a parameter of the {@link
477    * StepHandler#handleStep handleStep} is valid only for the duration
478    * of the {@link StepHandler#handleStep handleStep} call, one cannot
479    * simply store a reference and reuse it later. One should first
480    * finalize the instance, then copy this finalized instance into a
481    * new object that can be kept.</p>
482 
483    * <p>This method calls the protected <code>doFinalize</code> method
484    * if it has never been called during this step and set a flag
485    * indicating that it has been called once. It is the <code>
486    * doFinalize</code> method which should perform the evaluations.
487    * This wrapping prevents from calling <code>doFinalize</code> several
488    * times and hence evaluating the differential equations too often.
489    * Therefore, subclasses are not allowed not reimplement it, they
490    * should rather reimplement <code>doFinalize</code>.</p>
491 
492    * @exception MaxCountExceededException if the number of functions evaluations is exceeded
493 
494    */
495   public final void finalizeStep() throws MaxCountExceededException {
496     if (! finalized) {
497       doFinalize();
498       finalized = true;
499     }
500   }
501 
502   /**
503    * Really finalize the step.
504    * The default implementation of this method does nothing.
505    * @exception MaxCountExceededException if the number of functions evaluations is exceeded
506    */
507   protected void doFinalize() throws MaxCountExceededException {
508   }

520   /** Save the base state of the instance.
521    * This method performs step finalization if it has not been done
522    * before.
523    * @param out stream where to save the state
524    * @exception IOException in case of write error
525    */
526   protected void writeBaseExternal(final ObjectOutput out)
527     throws IOException {
528 
529     if (currentState == null) {
530         out.writeInt(-1);
531     } else {
532         out.writeInt(currentState.length);
533     }
534     out.writeDouble(globalPreviousTime);
535     out.writeDouble(globalCurrentTime);
536     out.writeDouble(softPreviousTime);
537     out.writeDouble(softCurrentTime);
538     out.writeDouble(h);
539     out.writeBoolean(forward);
540     out.writeObject(primaryMapper);
541     out.write(secondaryMappers.length);
542     for (final EquationsMapper  mapper : secondaryMappers) {
543         out.writeObject(mapper);
544     }
545 
546     if (currentState != null) {
547         for (int i = 0; i < currentState.length; ++i) {
548             out.writeDouble(currentState[i]);
549         }
550     }
551 
552     out.writeDouble(interpolatedTime);
553 
554     // we do not store the interpolated state,
555     // it will be recomputed as needed after reading
556 
557     try {
558         // finalize the step (and don't bother saving the now true flag)
559         finalizeStep();
560     } catch (MaxCountExceededException mcee) {
561         final IOException ioe = new IOException(mcee.getLocalizedMessage());
562         ioe.initCause(mcee);
563         throw ioe;
564     }
565 
566   }

568   /** Read the base state of the instance.
569    * This method does <strong>neither</strong> set the interpolated
570    * time nor state. It is up to the derived class to reset it
571    * properly calling the {@link #setInterpolatedTime} method later,
572    * once all rest of the object state has been set up properly.
573    * @param in stream where to read the state from
574    * @return interpolated time to be set later by the caller
575    * @exception IOException in case of read error
576    * @exception ClassNotFoundException if an equation mapper class
577    * cannot be found
578    */
579   protected double readBaseExternal(final ObjectInput in)
580     throws IOException, ClassNotFoundException {
581 
582     final int dimension = in.readInt();
583     globalPreviousTime  = in.readDouble();
584     globalCurrentTime   = in.readDouble();
585     softPreviousTime    = in.readDouble();
586     softCurrentTime     = in.readDouble();
587     h                   = in.readDouble();
588     forward             = in.readBoolean();
589     primaryMapper       = (EquationsMapper) in.readObject();
590     secondaryMappers    = new EquationsMapper[in.read()];
591     for (int i = 0; i < secondaryMappers.length; ++i) {
592         secondaryMappers[i] = (EquationsMapper) in.readObject();
593     }
594     dirtyState          = true;
595 
596     if (dimension < 0) {
597         currentState = null;
598     } else {
599         currentState  = new double[dimension];
600         for (int i = 0; i < currentState.length; ++i) {
601             currentState[i] = in.readDouble();
602         }
603     }
604 
605     // we do NOT handle the interpolated time and state here
606     interpolatedTime = Double.NaN;
607     allocateInterpolatedArrays(dimension);
608 
609     finalized = true;
610 
611     return in.readDouble();
612 
613   }
