17  package org.apache.commons.math4.optim.linear;
18  
19  import java.util.LinkedHashSet;
20  import java.util.Set;
21  import java.util.Collection;
22  import java.util.Collections;
23  
24  import org.apache.commons.math4.optim.OptimizationData;
25  
26  /**
27   * Class that represents a set of {@link LinearConstraint linear constraints}.
28   *
29   * @since 3.1
30   */
31  public class LinearConstraintSet implements OptimizationData {
32      /** Set of constraints. */
33      private final Set<LinearConstraint> linearConstraints = new LinkedHashSet<LinearConstraint>();
34  
35      /**
36       * Creates a set containing the given constraints.
37       *
38       * @param constraints Constraints.
39       */
40      public LinearConstraintSet(LinearConstraint... constraints) {
41          for (LinearConstraint c : constraints) {
42              linearConstraints.add(c);
43          }
44      }
45  
46      /**
47       * Creates a set containing the given constraints.
48       *
49       * @param constraints Constraints.
50       */
51      public LinearConstraintSet(Collection<LinearConstraint> constraints) {
52          linearConstraints.addAll(constraints);
53      }
54  
55      /**
56       * Gets the set of linear constraints.
57       *
58       * @return the constraints.
59       */
60      public Collection<LinearConstraint> getConstraints() {
61          return Collections.unmodifiableSet(linearConstraints);
62      }
63  }
