34  /**
35   * Basic implementation of RealMatrix methods regardless of the underlying storage.
36   * <p>All the methods implemented here use {@link #getEntry(int, int)} to access

39   * @since 2.0

42      extends RealLinearOperator

58       * Create a new RealMatrix with the supplied row and column dimensions.
59       *
60       * @param rowDimension  the number of rows in the new matrix
61       * @param columnDimension  the number of columns in the new matrix
62       * @throws NotStrictlyPositiveException if row or column dimension is not positive
63       */
64      protected AbstractRealMatrix(final int rowDimension,
65          final int columnDimension)
66          throws NotStrictlyPositiveException {
67          if (rowDimension < 1) {
68              throw new NotStrictlyPositiveException(rowDimension);

70          if (columnDimension < 1) {
71              throw new NotStrictlyPositiveException(columnDimension);

75      /** {@inheritDoc} */
76      @Override
77      public RealMatrix add(RealMatrix m)
78          throws MatrixDimensionMismatchException {
79          MatrixUtils.checkAdditionCompatible(this, m);
80  
81          final int rowCount    = getRowDimension();
82          final int columnCount = getColumnDimension();
83          final RealMatrix out = createMatrix(rowCount, columnCount);
84          for (int row = 0; row < rowCount; ++row) {
85              for (int col = 0; col < columnCount; ++col) {
86                  out.setEntry(row, col, getEntry(row, col) + m.getEntry(row, col));
87              }
88          }
89  
90          return out;
91      }

141     /** {@inheritDoc} */
142     @Override
143     public RealMatrix multiply(final RealMatrix m)
144         throws DimensionMismatchException {
145         MatrixUtils.checkMultiplicationCompatible(this, m);
146 
147         final int nRows = getRowDimension();
148         final int nCols = m.getColumnDimension();
149         final int nSum  = getColumnDimension();
150         final RealMatrix out = createMatrix(nRows, nCols);
151         for (int row = 0; row < nRows; ++row) {
152             for (int col = 0; col < nCols; ++col) {
153                 double sum = 0;
154                 for (int i = 0; i < nSum; ++i) {
155                     sum += getEntry(row, i) * m.getEntry(i, col);
156                 }
157                 out.setEntry(row, col, sum);
158             }
159         }
160 
161         return out;
162     }

171     /** {@inheritDoc} */
172     @Override
173     public RealMatrix power(final int p)
174         throws NotPositiveException, NonSquareMatrixException {
175         if (p < 0) {
176             throw new NotPositiveException(LocalizedFormats.NOT_POSITIVE_EXPONENT, p);
177         }
178 
179         if (!isSquare()) {
180             throw new NonSquareMatrixException(getRowDimension(), getColumnDimension());
181         }
182 
183         if (p == 0) {
184             return MatrixUtils.createRealIdentityMatrix(this.getRowDimension());
185         }
186 
187         if (p == 1) {
188             return this.copy();
189         }
190 
191         final int power = p - 1;
192 
193         /*
194          * Only log_2(p) operations is used by doing as follows:
195          * 5^214 = 5^128 * 5^64 * 5^16 * 5^4 * 5^2
196          *
197          * In general, the same approach is used for A^p.
198          */
199 
200         final char[] binaryRepresentation = Integer.toBinaryString(power).toCharArray();
201         final ArrayList<Integer> nonZeroPositions = new ArrayList<Integer>();
202         int maxI = -1;
203 
204         for (int i = 0; i < binaryRepresentation.length; ++i) {
205             if (binaryRepresentation[i] == '1') {
206                 final int pos = binaryRepresentation.length - i - 1;
207                 nonZeroPositions.add(pos);
208 
209                 // The positions are taken in turn, so maxI is only changed once
210                 if (maxI == -1) {
211                     maxI = pos;
212                 }
213             }
214         }
215 
216         RealMatrix[] results = new RealMatrix[maxI + 1];
217         results[0] = this.copy();
218 
219         for (int i = 1; i <= maxI; ++i) {
220             results[i] = results[i-1].multiply(results[i-1]);
221         }
222 
223         RealMatrix result = this.copy();
224 
225         for (Integer i : nonZeroPositions) {
226             result = result.multiply(results[i]);
227         }
228 
229         return result;
230     }

247     /** {@inheritDoc} */
248     @Override
249     public double getNorm() {
250         return walkInColumnOrder(new RealMatrixPreservingVisitor() {
251 
252             /** Last row index. */
253             private double endRow;
254 
255             /** Sum of absolute values on one column. */
256             private double columnSum;
257 
258             /** Maximal sum across all columns. */
259             private double maxColSum;
260 
261             /** {@inheritDoc} */
262             @Override
263             public void start(final int rows, final int columns,
264                               final int startRow, final int endRow,
265                               final int startColumn, final int endColumn) {
266                 this.endRow = endRow;
267                 columnSum   = 0;
268                 maxColSum   = 0;
269             }
270 
271             /** {@inheritDoc} */
272             @Override
273             public void visit(final int row, final int column, final double value) {
274                 columnSum += FastMath.abs(value);
275                 if (row == endRow) {
276                     maxColSum = FastMath.max(maxColSum, columnSum);
277                     columnSum = 0;
278                 }
279             }
280 
281             /** {@inheritDoc} */
282             @Override
283             public double end() {
284                 return maxColSum;
285             }
286         });
287     }

289     /** {@inheritDoc} */
290     @Override
291     public double getFrobeniusNorm() {
292         return walkInOptimizedOrder(new RealMatrixPreservingVisitor() {
293 
294             /** Sum of squared entries. */
295             private double sum;
296 
297             /** {@inheritDoc} */
298             @Override
299             public void start(final int rows, final int columns,
300                               final int startRow, final int endRow,
301                               final int startColumn, final int endColumn) {
302                 sum = 0;
303             }
304 
305             /** {@inheritDoc} */
306             @Override
307             public void visit(final int row, final int column, final double value) {
308                 sum += value * value;
309             }
310 
311             /** {@inheritDoc} */
312             @Override
313             public double end() {
314                 return FastMath.sqrt(sum);
315             }
316         });
317     }

320     @Override
321     public RealMatrix getSubMatrix(final int startRow, final int endRow,
322                                    final int startColumn, final int endColumn)
323         throws OutOfRangeException, NumberIsTooSmallException {
324         MatrixUtils.checkSubMatrixIndex(this, startRow, endRow, startColumn, endColumn);
325 
326         final RealMatrix subMatrix =
327             createMatrix(endRow - startRow + 1, endColumn - startColumn + 1);
328         for (int i = startRow; i <= endRow; ++i) {
329             for (int j = startColumn; j <= endColumn; ++j) {
330                 subMatrix.setEntry(i - startRow, j - startColumn, getEntry(i, j));
331             }
332         }
333 
334         return subMatrix;
335     }

359     /** {@inheritDoc} */
360     @Override
361     public void copySubMatrix(final int startRow, final int endRow,
362                               final int startColumn, final int endColumn,
363                               final double[][] destination)
364         throws OutOfRangeException, NumberIsTooSmallException,
365         MatrixDimensionMismatchException {
366         MatrixUtils.checkSubMatrixIndex(this, startRow, endRow, startColumn, endColumn);
367         final int rowsCount    = endRow + 1 - startRow;
368         final int columnsCount = endColumn + 1 - startColumn;
369         if ((destination.length < rowsCount) || (destination[0].length < columnsCount)) {
370             throw new MatrixDimensionMismatchException(destination.length, destination[0].length,
371                                                        rowsCount, columnsCount);
372         }
373 
374         for (int i = 1; i < rowsCount; i++) {
375             if (destination[i].length < columnsCount) {
376                 throw new MatrixDimensionMismatchException(destination.length, destination[i].length,
377                                                            rowsCount, columnsCount);
378             }
379         }
380 
381         walkInOptimizedOrder(new DefaultRealMatrixPreservingVisitor() {
382 
383             /** Initial row index. */
384             private int startRow;
385 
386             /** Initial column index. */
387             private int startColumn;
388 
389             /** {@inheritDoc} */
390             @Override
391             public void start(final int rows, final int columns,
392                               final int startRow, final int endRow,
393                               final int startColumn, final int endColumn) {
394                 this.startRow    = startRow;
395                 this.startColumn = startColumn;
396             }
397 
398             /** {@inheritDoc} */
399             @Override
400             public void visit(final int row, final int column, final double value) {
401                 destination[row - startRow][column - startColumn] = value;
402             }
403 
404         }, startRow, endRow, startColumn, endColumn);
405     }

407     /** {@inheritDoc} */
408     @Override
409     public void copySubMatrix(int[] selectedRows, int[] selectedColumns,
410                               double[][] destination)
411         throws OutOfRangeException, NullArgumentException, NoDataException,
412         MatrixDimensionMismatchException {
413         MatrixUtils.checkSubMatrixIndex(this, selectedRows, selectedColumns);
414         final int nCols = selectedColumns.length;
415         if ((destination.length < selectedRows.length) ||
416             (destination[0].length < nCols)) {
417             throw new MatrixDimensionMismatchException(destination.length, destination[0].length,
418                                                        selectedRows.length, selectedColumns.length);
419         }
420 
421         for (int i = 0; i < selectedRows.length; i++) {
422             final double[] destinationI = destination[i];
423             if (destinationI.length < nCols) {
424                 throw new MatrixDimensionMismatchException(destination.length, destinationI.length,
425                                                            selectedRows.length, selectedColumns.length);
426             }
427             for (int j = 0; j < selectedColumns.length; j++) {
428                 destinationI[j] = getEntry(selectedRows[i], selectedColumns[j]);
429             }
430         }
431     }

433     /** {@inheritDoc} */
434     @Override
435     public void setSubMatrix(final double[][] subMatrix, final int row, final int column)
436         throws NoDataException, OutOfRangeException,
437         DimensionMismatchException, NullArgumentException {
438         MathUtils.checkNotNull(subMatrix);
439         final int nRows = subMatrix.length;
440         if (nRows == 0) {
441             throw new NoDataException(LocalizedFormats.AT_LEAST_ONE_ROW);
442         }
443 
444         final int nCols = subMatrix[0].length;
445         if (nCols == 0) {
446             throw new NoDataException(LocalizedFormats.AT_LEAST_ONE_COLUMN);
447         }
448 
449         for (int r = 1; r < nRows; ++r) {
450             if (subMatrix[r].length != nCols) {
451                 throw new DimensionMismatchException(nCols, subMatrix[r].length);
452             }
453         }
454 
455         MatrixUtils.checkRowIndex(this, row);
456         MatrixUtils.checkColumnIndex(this, column);
457         MatrixUtils.checkRowIndex(this, nRows + row - 1);
458         MatrixUtils.checkColumnIndex(this, nCols + column - 1);
459 
460         for (int i = 0; i < nRows; ++i) {
461             for (int j = 0; j < nCols; ++j) {
462                 setEntry(row + i, column + j, subMatrix[i][j]);
463             }
464         }
465     }

698     /** {@inheritDoc} */
699     @Override
700     public double[] operate(final double[] v)
701         throws DimensionMismatchException {
702         final int nRows = getRowDimension();
703         final int nCols = getColumnDimension();
704         if (v.length != nCols) {
705             throw new DimensionMismatchException(v.length, nCols);
706         }
707 
708         final double[] out = new double[nRows];
709         for (int row = 0; row < nRows; ++row) {
710             double sum = 0;
711             for (int i = 0; i < nCols; ++i) {
712                 sum += getEntry(row, i) * v[i];
713             }
714             out[row] = sum;
715         }
716 
717         return out;
718     }

720     /** {@inheritDoc} */
721     @Override
722     public RealVector operate(final RealVector v)
723         throws DimensionMismatchException {
724         try {
725             return new ArrayRealVector(operate(((ArrayRealVector) v).getDataRef()), false);
726         } catch (ClassCastException cce) {

734             for (int row = 0; row < nRows; ++row) {

736                 for (int i = 0; i < nCols; ++i) {

972     /**
973      * Returns true iff <code>object</code> is a
974      * <code>RealMatrix</code> instance with the same dimensions as this
975      * and all corresponding matrix entries are equal.
976      *
977      * @param object the object to test equality against.
978      * @return true if object equals this
979      */
980     @Override
981     public boolean equals(final Object object) {
982         if (object == this ) {
983             return true;
984         }
985         if (object instanceof RealMatrix == false) {
986             return false;
987         }
988         RealMatrix m = (RealMatrix) object;
989         final int nRows = getRowDimension();
990         final int nCols = getColumnDimension();
991         if (m.getColumnDimension() != nCols || m.getRowDimension() != nRows) {
992             return false;
993         }
994         for (int row = 0; row < nRows; ++row) {
995             for (int col = 0; col < nCols; ++col) {
996                 if (getEntry(row, col) != m.getEntry(row, col)) {
997                     return false;
998                 }
999             }
1000         }
1001         return true;
1002     }

1004     /**
1005      * Computes a hashcode for the matrix.
1006      *
1007      * @return hashcode for matrix
1008      */
1009     @Override
1010     public int hashCode() {
1011         int ret = 7;
1012         final int nRows = getRowDimension();
1013         final int nCols = getColumnDimension();
1014         ret = ret * 31 + nRows;
1015         ret = ret * 31 + nCols;
1016         for (int row = 0; row < nRows; ++row) {
1017             for (int col = 0; col < nCols; ++col) {
1018                ret = ret * 31 + (11 * (row+1) + 17 * (col+1)) *
1019                    MathUtils.hash(getEntry(row, col));
1020            }
1021         }
1022         return ret;
1023     }
