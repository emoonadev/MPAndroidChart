17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when there is insufficient data to perform a computation.
24   *
25   * @since 3.3
26   */
27  public class InsufficientDataException
28      extends MathIllegalArgumentException {
29  
30      /** Serializable version Id. */
31      private static final long serialVersionUID = -2629324471511903359L;
32  
33      /**
34       * Construct the exception.
35       */
36      public InsufficientDataException() {
37          this(LocalizedFormats.INSUFFICIENT_DATA);
38      }
39  
40      /**
41       * Construct the exception with a specific context.
42       *
43       * @param pattern Message pattern providing the specific context of the error.
44       * @param arguments Values for replacing the placeholders in {@code pattern}.
45       */
46      public InsufficientDataException(Localizable pattern, Object... arguments) {
47          super(pattern, arguments);
48      }
49  }
