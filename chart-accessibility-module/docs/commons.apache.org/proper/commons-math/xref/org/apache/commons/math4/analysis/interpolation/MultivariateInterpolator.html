17  package org.apache.commons.math4.analysis.interpolation;
18  
19  import org.apache.commons.math4.analysis.MultivariateFunction;
20  import org.apache.commons.math4.exception.DimensionMismatchException;
21  import org.apache.commons.math4.exception.MathIllegalArgumentException;
22  import org.apache.commons.math4.exception.NoDataException;
23  import org.apache.commons.math4.exception.NullArgumentException;
24  
25  /**
26   * Interface representing a univariate real interpolating function.
27   *
28   * @since 2.1
29   */
30  public interface MultivariateInterpolator {
31  
32      /**
33       * Computes an interpolating function for the data set.
34       *
35       * @param xval the arguments for the interpolation points.
36       * {@code xval[i][0]} is the first component of interpolation point
37       * {@code i}, {@code xval[i][1]} is the second component, and so on
38       * until {@code xval[i][d-1]}, the last component of that interpolation
39       * point (where {@code d} is thus the dimension of the space).
40       * @param yval the values for the interpolation points
41       * @return a function which interpolates the data set
42       * @throws MathIllegalArgumentException if the arguments violate assumptions
43       * made by the interpolation algorithm.
44       * @throws DimensionMismatchException when the array dimensions are not consistent.
45       * @throws NoDataException if an array has zero-length.
46       * @throws NullArgumentException if the arguments are {@code null}.
47       */
48      MultivariateFunction interpolate(double[][] xval, double[] yval)
49          throws MathIllegalArgumentException, DimensionMismatchException,
50                 NoDataException, NullArgumentException;
51  }
