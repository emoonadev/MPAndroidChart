17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.OutOfRangeException;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  import org.apache.commons.math4.util.FastMath;
22  
23  /**
24   * Implementation of the geometric distribution.
25   *
26   * @see <a href="http://en.wikipedia.org/wiki/Geometric_distribution">Geometric distribution (Wikipedia)</a>
27   * @see <a href="http://mathworld.wolfram.com/GeometricDistribution.html">Geometric Distribution (MathWorld)</a>
28   * @since 3.3
29   */
30  public class GeometricDistribution extends AbstractIntegerDistribution {
31  
32      /** Serializable version identifier. */
33      private static final long serialVersionUID = 20160318L;
34      /** The probability of success. */
35      private final double probabilityOfSuccess;
36      /** {@code log(p)} where p is the probability of success. */
37      private final double logProbabilityOfSuccess;
38      /** {@code log(1 - p)} where p is the probability of success. */
39      private final double log1mProbabilityOfSuccess;
40  
41      /**
42       * Creates a geometric distribution.
43       *
44       * @param p Probability of success.
45       * @throws OutOfRangeException if {@code p <= 0} or {@code p > 1}.
46       */
47      public GeometricDistribution(double p) {
48          if (p <= 0 || p > 1) {
49              throw new OutOfRangeException(LocalizedFormats.OUT_OF_RANGE_LEFT, p, 0, 1);
50          }
51  
52          probabilityOfSuccess = p;
53          logProbabilityOfSuccess = FastMath.log(p);
54          log1mProbabilityOfSuccess = FastMath.log1p(-p);
55      }
56  
57      /**
58       * Access the probability of success for this distribution.
59       *
60       * @return the probability of success.
61       */
62      public double getProbabilityOfSuccess() {
63          return probabilityOfSuccess;
64      }
65  
66      /** {@inheritDoc} */
67      @Override
68      public double probability(int x) {
69          if (x < 0) {
70              return 0.0;
71          } else {
72              return FastMath.exp(log1mProbabilityOfSuccess * x) * probabilityOfSuccess;
73          }
74      }
75  
76      /** {@inheritDoc} */
77      @Override
78      public double logProbability(int x) {
79          if (x < 0) {
80              return Double.NEGATIVE_INFINITY;
81          } else {
82              return x * log1mProbabilityOfSuccess + logProbabilityOfSuccess;
83          }
84      }
85  
86      /** {@inheritDoc} */
87      @Override
88      public double cumulativeProbability(int x) {
89          if (x < 0) {
90              return 0.0;
91          } else {
92              return -FastMath.expm1(log1mProbabilityOfSuccess * (x + 1));
93          }
94      }
95  
96      /**
97       * {@inheritDoc}
98       *
99       * For probability parameter {@code p}, the mean is {@code (1 - p) / p}.
100      */
101     @Override
102     public double getNumericalMean() {
103         return (1 - probabilityOfSuccess) / probabilityOfSuccess;
104     }
105 
106     /**
107      * {@inheritDoc}
108      *
109      * For probability parameter {@code p}, the variance is
110      * {@code (1 - p) / (p * p)}.
111      */
112     @Override
113     public double getNumericalVariance() {
114         return (1 - probabilityOfSuccess) / (probabilityOfSuccess * probabilityOfSuccess);
115     }
116 
117     /**
118      * {@inheritDoc}
119      *
120      * The lower bound of the support is always 0.
121      *
122      * @return lower bound of the support (always 0)
123      */
124     @Override
125     public int getSupportLowerBound() {
126         return 0;
127     }
128 
129     /**
130      * {@inheritDoc}
131      *
132      * The upper bound of the support is infinite (which we approximate as
133      * {@code Integer.MAX_VALUE}).
134      *
135      * @return upper bound of the support (always Integer.MAX_VALUE)
136      */
137     @Override
138     public int getSupportUpperBound() {
139         return Integer.MAX_VALUE;
140     }
141 
142     /**
143      * {@inheritDoc}
144      *
145      * The support of this distribution is connected.
146      *
147      * @return {@code true}
148      */
149     @Override
150     public boolean isSupportConnected() {
151         return true;
152     }
153 
154     /**
155      * {@inheritDoc}
156      */
157     @Override
158     public int inverseCumulativeProbability(double p) throws OutOfRangeException {
159         if (p < 0 || p > 1) {
160             throw new OutOfRangeException(p, 0, 1);
161         }
162         if (p == 1) {
163             return Integer.MAX_VALUE;
164         }
165         if (p == 0) {
166             return 0;
167         }
168         return Math.max(0, (int) Math.ceil(FastMath.log1p(-p)/log1mProbabilityOfSuccess-1));
169     }
170 }
