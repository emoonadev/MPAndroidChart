17  package org.apache.commons.math4.random;
18  
19  /**
20   * Any {@link RandomGenerator} implementation can be thread-safe if it
21   * is used through an instance of this class.
22   * This is achieved by enclosing calls to the methods of the actual
23   * generator inside the overridden {@code synchronized} methods of this
24   * class.
25   *
26   * @since 3.1
27   * @deprecated As of 4.0. No replacement.
28   */
29  @Deprecated
30  public class SynchronizedRandomGenerator implements RandomGenerator {
31      /** Object to which all calls will be delegated. */
32      private final RandomGenerator wrapped;
33  
34      /**
35       * Creates a synchronized wrapper for the given {@code RandomGenerator}
36       * instance.
37       *
38       * @param rng Generator whose methods will be called through
39       * their corresponding overridden synchronized version.
40       * To ensure thread-safety, the wrapped generator <em>must</em>
41       * not be used directly.
42       */
43      public SynchronizedRandomGenerator(RandomGenerator rng) {
44          wrapped = rng;
45      }
46  
47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      public synchronized void setSeed(int seed) {
52          wrapped.setSeed(seed);
53      }
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public synchronized void setSeed(int[] seed) {
60          wrapped.setSeed(seed);
61      }
62  
63      /**
64       * {@inheritDoc}
65       */
66      @Override
67      public synchronized void setSeed(long seed) {
68          wrapped.setSeed(seed);
69      }
70  
71      /**
72       * {@inheritDoc}
73       */
74      @Override
75      public synchronized void nextBytes(byte[] bytes) {
76          wrapped.nextBytes(bytes);
77      }
78  
79      /**
80       * {@inheritDoc}
81       */
82      @Override
83      public synchronized int nextInt() {
84          return wrapped.nextInt();
85      }
86  
87      /**
88       * {@inheritDoc}
89       */
90      @Override
91      public synchronized int nextInt(int n) {
92          return wrapped.nextInt(n);
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public synchronized long nextLong() {
100         return wrapped.nextLong();
101     }
102 
103     /**
104      * {@inheritDoc}
105      */
106     @Override
107     public synchronized boolean nextBoolean() {
108         return wrapped.nextBoolean();
109     }
110 
111     /**
112      * {@inheritDoc}
113      */
114     @Override
115     public synchronized float nextFloat() {
116         return wrapped.nextFloat();
117     }
118 
119     /**
120      * {@inheritDoc}
121      */
122     @Override
123     public synchronized double nextDouble() {
124         return wrapped.nextDouble();
125     }
126 
127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public synchronized double nextGaussian() {
132         return wrapped.nextGaussian();
133     }
134 }
