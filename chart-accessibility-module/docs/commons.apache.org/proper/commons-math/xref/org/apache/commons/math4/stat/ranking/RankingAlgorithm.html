18  package org.apache.commons.math4.stat.ranking;
19  
20  /**
21   * Interface representing a rank transformation.
22   *
23   * @since 2.0
24   */
25  public interface RankingAlgorithm {
26      /**
27       * <p>Performs a rank transformation on the input data, returning an array
28       * of ranks.</p>
29       *
30       * <p>Ranks should be 1-based - that is, the smallest value
31       * returned in an array of ranks should be greater than or equal to one,
32       * rather than 0. Ranks should in general take integer values, though
33       * implementations may return averages or other floating point values
34       * to resolve ties in the input data.</p>
35       *
36       * @param data array of data to be ranked
37       * @return an array of ranks corresponding to the elements of the input array
38       */
39      double[] rank (double[] data);
40  }
