17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.exception.OutOfRangeException;
20  import org.apache.commons.math4.special.Erf;
21  import org.apache.commons.math4.util.FastMath;
22  
23  /**
24   * This class implements the <a href="http://en.wikipedia.org/wiki/L%C3%A9vy_distribution">
25   * L&eacute;vy distribution</a>.
26   *
27   * @since 3.2
28   */
29  public class LevyDistribution extends AbstractRealDistribution {
30  
31      /** Serializable UID. */
32      private static final long serialVersionUID = 20630311L;
33  
34      /** Location parameter. */
35      private final double mu;
36  
37      /** Scale parameter. */
38      private final double c;  // Setting this to 1 returns a cumProb of 1.0
39  
40      /** Half of c (for calculations). */
41      private final double halfC;
42  
43      /**
44       * Creates a distribution.
45       *
46       * @param mu location
47       * @param c scale parameter
48       */
49      public LevyDistribution(final double mu, final double c) {
50          this.mu = mu;
51          this.c = c;
52          this.halfC = 0.5 * c;
53      }
54  
55      /** {@inheritDoc}
56      * <p>
57      * From Wikipedia: The probability density function of the L&eacute;vy distribution
58      * over the domain is
59      * </p>
60      * <pre>
61      * f(x; &mu;, c) = &radic;(c / 2&pi;) * e<sup>-c / 2 (x - &mu;)</sup> / (x - &mu;)<sup>3/2</sup>
62      * </pre>
63      * <p>
64      * For this distribution, {@code X}, this method returns {@code P(X < x)}.
65      * If {@code x} is less than location parameter &mu;, {@code Double.NaN} is
66      * returned, as in these cases the distribution is not defined.
67      * </p>
68      */
69      @Override
70      public double density(final double x) {
71          if (x < mu) {
72              return Double.NaN;
73          }
74  
75          final double delta = x - mu;
76          final double f     = halfC / delta;
77          return FastMath.sqrt(f / FastMath.PI) * FastMath.exp(-f) /delta;
78      }
79  
80      /** {@inheritDoc}
81       *
82       * See documentation of {@link #density(double)} for computation details.
83       */
84      @Override
85      public double logDensity(double x) {
86          if (x < mu) {
87              return Double.NaN;
88          }
89  
90          final double delta = x - mu;
91          final double f     = halfC / delta;
92          return 0.5 * FastMath.log(f / FastMath.PI) - f - FastMath.log(delta);
93      }
94  
95      /** {@inheritDoc}
96       * <p>
97       * From Wikipedia: the cumulative distribution function is
98       * </p>
99       * <pre>
100      * f(x; u, c) = erfc (&radic; (c / 2 (x - u )))
101      * </pre>
102      */
103     @Override
104     public double cumulativeProbability(final double x) {
105         if (x < mu) {
106             return Double.NaN;
107         }
108         return Erf.erfc(FastMath.sqrt(halfC / (x - mu)));
109     }
110 
111     /** {@inheritDoc} */
112     @Override
113     public double inverseCumulativeProbability(final double p) throws OutOfRangeException {
114         if (p < 0.0 || p > 1.0) {
115             throw new OutOfRangeException(p, 0, 1);
116         }
117         final double t = Erf.erfcInv(p);
118         return mu + halfC / (t * t);
119     }
120 
121     /** Get the scale parameter of the distribution.
122      * @return scale parameter of the distribution
123      */
124     public double getScale() {
125         return c;
126     }
127 
128     /** Get the location parameter of the distribution.
129      * @return location parameter of the distribution
130      */
131     public double getLocation() {
132         return mu;
133     }
134 
135     /** {@inheritDoc} */
136     @Override
137     public double getNumericalMean() {
138         return Double.POSITIVE_INFINITY;
139     }
140 
141     /** {@inheritDoc} */
142     @Override
143     public double getNumericalVariance() {
144         return Double.POSITIVE_INFINITY;
145     }
146 
147     /** {@inheritDoc} */
148     @Override
149     public double getSupportLowerBound() {
150         return mu;
151     }
152 
153     /** {@inheritDoc} */
154     @Override
155     public double getSupportUpperBound() {
156         return Double.POSITIVE_INFINITY;
157     }
158 
159     /** {@inheritDoc} */
160     @Override
161     public boolean isSupportConnected() {
162         return true;
163     }
164 
165 }
