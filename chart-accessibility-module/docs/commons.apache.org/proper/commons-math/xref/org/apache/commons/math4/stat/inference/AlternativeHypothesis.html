17  package org.apache.commons.math4.stat.inference;
18  
19  /**
20   * Represents an alternative hypothesis for a hypothesis test.
21   *
22   * @since 3.3
23   */
24  public enum AlternativeHypothesis {
25  
26      /**
27       * Represents a two-sided test. H0: p=p0, H1: p &ne; p0
28       */
29      TWO_SIDED,
30  
31      /**
32       * Represents a right-sided test. H0: p &le; p0, H1: p &gt; p0.
33       */
34      GREATER_THAN,
35  
36      /**
37       * Represents a left-sided test. H0: p &ge; p0, H1: p &lt; p0.
38       */
39      LESS_THAN
40  }
