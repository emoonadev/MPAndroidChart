19  /**
20   * Algorithm used to determine when to stop evolution.
21   *
22   * @since 2.0
23   */
24  public interface StoppingCondition {
25      /**
26       * Determine whether or not the given population satisfies the stopping condition.
27       *
28       * @param population the population to test.
29       * @return <code>true</code> if this stopping condition is met by the given population,
30       *   <code>false</code> otherwise.
31       */
32      boolean isSatisfied(Population population);
33  }
