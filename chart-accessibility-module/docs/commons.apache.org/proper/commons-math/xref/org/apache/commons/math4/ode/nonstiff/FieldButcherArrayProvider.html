18  package org.apache.commons.math4.ode.nonstiff;
19  
20  import org.apache.commons.math4.RealFieldElement;
21  
22  /** This interface represents an integrator  based on Butcher arrays.
23   * @see RungeKuttaFieldIntegrator
24   * @see EmbeddedRungeKuttaFieldIntegrator
25   * @param <T> the type of the field elements
26   * @since 3.6
27   */
28  
29  public interface FieldButcherArrayProvider<T extends RealFieldElement<T>> {
30  
31      /** Get the time steps from Butcher array (without the first zero).
32       * @return time steps from Butcher array (without the first zero
33       */
34      T[] getC();
35  
36      /** Get the internal weights from Butcher array (without the first empty row).
37       * @return internal weights from Butcher array (without the first empty row)
38       */
39      T[][] getA();
40  
41      /** Get the external weights for the high order method from Butcher array.
42       * @return external weights for the high order method from Butcher array
43       */
44      T[] getB();
45  
46  }
