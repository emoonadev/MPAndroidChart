17  package org.apache.commons.math4.genetics;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  
21  /**
22   * Policy used to create a pair of new chromosomes by performing a crossover
23   * operation on a source pair of chromosomes.
24   *
25   * @since 2.0
26   */
27  public interface CrossoverPolicy {
28  
29      /**
30       * Perform a crossover operation on the given chromosomes.
31       *
32       * @param first the first chromosome.
33       * @param second the second chromosome.
34       * @return the pair of new chromosomes that resulted from the crossover.
35       * @throws MathIllegalArgumentException if the given chromosomes are not compatible with this {@link CrossoverPolicy}
36       */
37      ChromosomePair crossover(Chromosome first, Chromosome second) throws MathIllegalArgumentException;
38  }
