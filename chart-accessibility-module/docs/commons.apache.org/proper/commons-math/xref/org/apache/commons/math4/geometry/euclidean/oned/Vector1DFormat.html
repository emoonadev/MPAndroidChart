18  package org.apache.commons.math4.geometry.euclidean.oned;
19  
20  import java.text.FieldPosition;
21  import java.text.NumberFormat;
22  import java.text.ParsePosition;
23  import java.util.Locale;
24  
25  import org.apache.commons.math4.exception.MathParseException;
26  import org.apache.commons.math4.geometry.Vector;
27  import org.apache.commons.math4.geometry.VectorFormat;
28  import org.apache.commons.math4.util.CompositeFormat;
29  
30  /**
31   * Formats a 1D vector in components list format "{x}".
32   * <p>The prefix and suffix "{" and "}" can be replaced by
33   * any user-defined strings. The number format for components can be configured.</p>
34   * <p>White space is ignored at parse time, even if it is in the prefix, suffix
35   * or separator specifications. So even if the default separator does include a space
36   * character that is used at format time, both input string "{1}" and
37   * " { 1 } " will be parsed without error and the same vector will be
38   * returned. In the second case, however, the parse position after parsing will be
39   * just after the closing curly brace, i.e. just before the trailing space.</p>
40   * <p><b>Note:</b> using "," as a separator may interfere with the grouping separator
41   * of the default {@link NumberFormat} for the current locale. Thus it is advised
42   * to use a {@link NumberFormat} instance with disabled grouping in such a case.</p>
43   *
44   * @since 3.0
45   */
46  public class Vector1DFormat extends VectorFormat<Euclidean1D> {
47  
48      /**
49       * Create an instance with default settings.
50       * <p>The instance uses the default prefix, suffix and separator:
51       * "{", "}", and "; " and the default number format for components.</p>
52       */
53      public Vector1DFormat() {
54          super(DEFAULT_PREFIX, DEFAULT_SUFFIX, DEFAULT_SEPARATOR,
55                CompositeFormat.getDefaultNumberFormat());
56      }
57  
58      /**
59       * Create an instance with a custom number format for components.
60       * @param format the custom format for components.
61       */
62      public Vector1DFormat(final NumberFormat format) {
63          super(DEFAULT_PREFIX, DEFAULT_SUFFIX, DEFAULT_SEPARATOR, format);
64      }
65  
66      /**
67       * Create an instance with custom prefix, suffix and separator.
68       * @param prefix prefix to use instead of the default "{"
69       * @param suffix suffix to use instead of the default "}"
70       */
71      public Vector1DFormat(final String prefix, final String suffix) {
72          super(prefix, suffix, DEFAULT_SEPARATOR, CompositeFormat.getDefaultNumberFormat());
73      }
74  
75      /**
76       * Create an instance with custom prefix, suffix, separator and format
77       * for components.
78       * @param prefix prefix to use instead of the default "{"
79       * @param suffix suffix to use instead of the default "}"
80       * @param format the custom format for components.
81       */
82      public Vector1DFormat(final String prefix, final String suffix,
83                           final NumberFormat format) {
84          super(prefix, suffix, DEFAULT_SEPARATOR, format);
85      }
86  
87      /**
88       * Returns the default 1D vector format for the current locale.
89       * @return the default 1D vector format.
90       */
91      public static Vector1DFormat getInstance() {
92          return getInstance(Locale.getDefault());
93      }
94  
95      /**
96       * Returns the default 1D vector format for the given locale.
97       * @param locale the specific locale used by the format.
98       * @return the 1D vector format specific to the given locale.
99       */
100     public static Vector1DFormat getInstance(final Locale locale) {
101         return new Vector1DFormat(CompositeFormat.getDefaultNumberFormat(locale));
102     }
103 
104     /** {@inheritDoc} */
105     @Override
106     public StringBuffer format(final Vector<Euclidean1D> vector, final StringBuffer toAppendTo,
107                                final FieldPosition pos) {
108         final Vector1D p1 = (Vector1D) vector;
109         return format(toAppendTo, pos, p1.getX());
110     }
111 
112     /** {@inheritDoc} */
113     @Override
114     public Vector1D parse(final String source) throws MathParseException {
115         ParsePosition parsePosition = new ParsePosition(0);
116         Vector1D result = parse(source, parsePosition);
117         if (parsePosition.getIndex() == 0) {
118             throw new MathParseException(source,
119                                          parsePosition.getErrorIndex(),
120                                          Vector1D.class);
121         }
122         return result;
123     }
124 
125     /** {@inheritDoc} */
126     @Override
127     public Vector1D parse(final String source, final ParsePosition pos) {
128         final double[] coordinates = parseCoordinates(1, source, pos);
129         if (coordinates == null) {
130             return null;
131         }
132         return new Vector1D(coordinates[0]);
133     }
134 
135 }
