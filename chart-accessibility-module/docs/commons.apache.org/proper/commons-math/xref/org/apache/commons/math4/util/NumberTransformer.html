17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  
21  /**
22   * Subclasses implementing this interface can transform Objects to doubles.
23   *
24   * No longer extends Serializable since 2.0
25   *
26   */
27  public interface NumberTransformer {
28  
29      /**
30       * Implementing this interface provides a facility to transform
31       * from Object to Double.
32       *
33       * @param o the Object to be transformed.
34       * @return the double value of the Object.
35       * @throws MathIllegalArgumentException if the Object can not be transformed into a Double.
36       */
37      double transform(Object o) throws MathIllegalArgumentException;
38  }
