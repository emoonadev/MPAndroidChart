17  package org.apache.commons.math4.geometry.euclidean.oned;
18  
19  import org.apache.commons.math4.geometry.partitioning.Region.Location;
20  import org.apache.commons.math4.exception.NumberIsTooSmallException;
21  import org.apache.commons.math4.exception.util.LocalizedFormats;
22  
23  
24  /** This class represents a 1D interval.
25   * @see IntervalsSet
26   * @since 3.0
27   */
28  public class Interval {
29  
30      /** The lower bound of the interval. */
31      private final double lower;
32  
33      /** The upper bound of the interval. */
34      private final double upper;
35  
36      /** Simple constructor.
37       * @param lower lower bound of the interval
38       * @param upper upper bound of the interval
39       */
40      public Interval(final double lower, final double upper) {
41          if (upper < lower) {
42              throw new NumberIsTooSmallException(LocalizedFormats.ENDPOINTS_NOT_AN_INTERVAL,
43                                                  upper, lower, true);
44          }
45          this.lower = lower;
46          this.upper = upper;
47      }
48  
49      /** Get the lower bound of the interval.
50       * @return lower bound of the interval
51       * @since 3.1
52       */
53      public double getInf() {
54          return lower;
55      }
56  
57      /** Get the upper bound of the interval.
58       * @return upper bound of the interval
59       * @since 3.1
60       */
61      public double getSup() {
62          return upper;
63      }
64  
65      /** Get the size of the interval.
66       * @return size of the interval
67       * @since 3.1
68       */
69      public double getSize() {
70          return upper - lower;
71      }
72  
73      /** Get the barycenter of the interval.
74       * @return barycenter of the interval
75       * @since 3.1
76       */
77      public double getBarycenter() {
78          return 0.5 * (lower + upper);
79      }
80  
81      /** Check a point with respect to the interval.
82       * @param point point to check
83       * @param tolerance tolerance below which points are considered to
84       * belong to the boundary
85       * @return a code representing the point status: either {@link
86       * Location#INSIDE}, {@link Location#OUTSIDE} or {@link Location#BOUNDARY}
87       * @since 3.1
88       */
89      public Location checkPoint(final double point, final double tolerance) {
90          if (point < lower - tolerance || point > upper + tolerance) {
91              return Location.OUTSIDE;
92          } else if (point > lower + tolerance && point < upper - tolerance) {
93              return Location.INSIDE;
94          } else {
95              return Location.BOUNDARY;
96          }
97      }
98  
99  }
