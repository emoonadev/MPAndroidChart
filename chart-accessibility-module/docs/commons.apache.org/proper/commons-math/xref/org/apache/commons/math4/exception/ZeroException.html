17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when zero is provided where it is not allowed.
24   *
25   * @since 2.2
26   */
27  public class ZeroException extends MathIllegalNumberException {
28  
29      /** Serializable version identifier */
30      private static final long serialVersionUID = -1960874856936000015L;
31  
32      /**
33       * Construct the exception.
34       */
35      public ZeroException() {
36          this(LocalizedFormats.ZERO_NOT_ALLOWED);
37      }
38  
39      /**
40       * Construct the exception with a specific context.
41       *
42       * @param specific Specific context pattern.
43       * @param arguments Arguments.
44       */
45      public ZeroException(Localizable specific, Object ... arguments) {
46          super(specific, INTEGER_ZERO, arguments);
47      }
48  }
