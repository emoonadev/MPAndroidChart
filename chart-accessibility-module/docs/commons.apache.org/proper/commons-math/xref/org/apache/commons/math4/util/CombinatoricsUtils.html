61      /**
62       * Returns an exact representation of the <a
63       * href="http://mathworld.wolfram.com/BinomialCoefficient.html"> Binomial
64       * Coefficient</a>, "{@code n choose k}", the number of
65       * {@code k}-element subsets that can be selected from an
66       * {@code n}-element set.
67       * <p>
68       * <Strong>Preconditions</strong>:
69       * <ul>
70       * <li> {@code 0 <= k <= n } (otherwise
71       * {@code MathIllegalArgumentException} is thrown)</li>
72       * <li> The result is small enough to fit into a {@code long}. The
73       * largest value of {@code n} for which all coefficients are
74       * {@code  < Long.MAX_VALUE} is 66. If the computed value exceeds
75       * {@code Long.MAX_VALUE} a {@code MathArithMeticException} is
76       * thrown.</li>
77       * </ul></p>
78       *
79       * @param n the size of the set
80       * @param k the size of the subsets to be counted
81       * @return {@code n choose k}
82       * @throws NotPositiveException if {@code n < 0}.
83       * @throws NumberIsTooLargeException if {@code k > n}.
84       * @throws MathArithmeticException if the result is too large to be
85       * represented by a long integer.
86       */
87      public static long binomialCoefficient(final int n, final int k)
88          throws NotPositiveException, NumberIsTooLargeException, MathArithmeticException {
89          CombinatoricsUtils.checkBinomial(n, k);
90          if ((n == k) || (k == 0)) {
91              return 1;
92          }
93          if ((k == 1) || (k == n - 1)) {
94              return n;
95          }
96          // Use symmetry for large k
97          if (k > n / 2) {
98              return binomialCoefficient(n, n - k);
99          }
100 
101         // We use the formula
102         // (n choose k) = n! / (n-k)! / k!
103         // (n choose k) == ((n-k+1)*...*n) / (1*...*k)
104         // which could be written
105         // (n choose k) == (n-1 choose k-1) * n / k
106         long result = 1;
107         if (n <= 61) {
108             // For n <= 61, the naive implementation cannot overflow.
109             int i = n - k + 1;
110             for (int j = 1; j <= k; j++) {
111                 result = result * i / j;
112                 i++;
113             }
114         } else if (n <= 66) {
115             // For n > 61 but n <= 66, the result cannot overflow,
116             // but we must take care not to overflow intermediate values.
117             int i = n - k + 1;
118             for (int j = 1; j <= k; j++) {
119                 // We know that (result * i) is divisible by j,
120                 // but (result * i) may overflow, so we split j:
121                 // Filter out the gcd, d, so j/d and i/d are integer.
122                 // result is divisible by (j/d) because (j/d)
123                 // is relative prime to (i/d) and is a divisor of
124                 // result * (i/d).
125                 final long d = ArithmeticUtils.gcd(i, j);
126                 result = (result / (j / d)) * (i / d);
127                 i++;
128             }
129         } else {
130             // For n > 66, a result overflow might occur, so we check
131             // the multiplication, taking care to not overflow
132             // unnecessary.
133             int i = n - k + 1;
134             for (int j = 1; j <= k; j++) {
135                 final long d = ArithmeticUtils.gcd(i, j);
136                 result = ArithmeticUtils.mulAndCheck(result / (j / d), i / d);
137                 i++;
138             }
139         }
140         return result;
141     }

143     /**
144      * Returns a {@code double} representation of the <a
145      * href="http://mathworld.wolfram.com/BinomialCoefficient.html"> Binomial
146      * Coefficient</a>, "{@code n choose k}", the number of
147      * {@code k}-element subsets that can be selected from an
148      * {@code n}-element set.
149      * <p>
150      * <Strong>Preconditions</strong>:
151      * <ul>
152      * <li> {@code 0 <= k <= n } (otherwise
153      * {@code IllegalArgumentException} is thrown)</li>
154      * <li> The result is small enough to fit into a {@code double}. The
155      * largest value of {@code n} for which all coefficients are <
156      * Double.MAX_VALUE is 1029. If the computed value exceeds Double.MAX_VALUE,
157      * Double.POSITIVE_INFINITY is returned</li>
158      * </ul></p>
159      *
160      * @param n the size of the set
161      * @param k the size of the subsets to be counted
162      * @return {@code n choose k}
163      * @throws NotPositiveException if {@code n < 0}.
164      * @throws NumberIsTooLargeException if {@code k > n}.
165      * @throws MathArithmeticException if the result is too large to be
166      * represented by a long integer.
167      */
168     public static double binomialCoefficientDouble(final int n, final int k)
169         throws NotPositiveException, NumberIsTooLargeException, MathArithmeticException {
170         CombinatoricsUtils.checkBinomial(n, k);
171         if ((n == k) || (k == 0)) {
172             return 1d;
173         }
174         if ((k == 1) || (k == n - 1)) {
175             return n;
176         }
177         if (k > n/2) {
178             return binomialCoefficientDouble(n, n - k);
179         }
180         if (n < 67) {
181             return binomialCoefficient(n,k);
182         }
183 
184         double result = 1d;
185         for (int i = 1; i <= k; i++) {
186              result *= (double)(n - k + i) / (double)i;
187         }
188 
189         return FastMath.floor(result + 0.5);
190     }

192     /**
193      * Returns the natural {@code log} of the <a
194      * href="http://mathworld.wolfram.com/BinomialCoefficient.html"> Binomial
195      * Coefficient</a>, "{@code n choose k}", the number of
196      * {@code k}-element subsets that can be selected from an
197      * {@code n}-element set.
198      * <p>
199      * <Strong>Preconditions</strong>:
200      * <ul>
201      * <li> {@code 0 <= k <= n } (otherwise
202      * {@code MathIllegalArgumentException} is thrown)</li>
203      * </ul></p>
204      *
205      * @param n the size of the set
206      * @param k the size of the subsets to be counted
207      * @return {@code n choose k}
208      * @throws NotPositiveException if {@code n < 0}.
209      * @throws NumberIsTooLargeException if {@code k > n}.
210      * @throws MathArithmeticException if the result is too large to be
211      * represented by a long integer.
212      */
213     public static double binomialCoefficientLog(final int n, final int k)
214         throws NotPositiveException, NumberIsTooLargeException, MathArithmeticException {
215         CombinatoricsUtils.checkBinomial(n, k);
216         if ((n == k) || (k == 0)) {
217             return 0;
218         }
219         if ((k == 1) || (k == n - 1)) {
220             return FastMath.log(n);
221         }
222 
223         /*
224          * For values small enough to do exact integer computation,
225          * return the log of the exact value
226          */
227         if (n < 67) {
228             return FastMath.log(binomialCoefficient(n,k));
229         }
230 
231         /*
232          * Return the log of binomialCoefficientDouble for values that will not
233          * overflow binomialCoefficientDouble
234          */
235         if (n < 1030) {
236             return FastMath.log(binomialCoefficientDouble(n, k));
237         }
238 
239         if (k > n / 2) {
240             return binomialCoefficientLog(n, n - k);
241         }
242 
243         /*
244          * Sum logs for values that could overflow
245          */
246         double logSum = 0;
247 
248         // n!/(n-k)!
249         for (int i = n - k + 1; i <= n; i++) {
250             logSum += FastMath.log(i);
251         }
252 
253         // divide by k!
254         for (int i = 2; i <= k; i++) {
255             logSum -= FastMath.log(i);
256         }
257 
258         return logSum;
259     }

261     /**
262      * Returns n!. Shorthand for {@code n} <a
263      * href="http://mathworld.wolfram.com/Factorial.html"> Factorial</a>, the
264      * product of the numbers {@code 1,...,n}.
265      * <p>
266      * <Strong>Preconditions</strong>:
267      * <ul>
268      * <li> {@code n >= 0} (otherwise
269      * {@code MathIllegalArgumentException} is thrown)</li>
270      * <li> The result is small enough to fit into a {@code long}. The
271      * largest value of {@code n} for which {@code n!} does not exceed
272      * Long.MAX_VALUE} is 20. If the computed value exceeds {@code Long.MAX_VALUE}
273      * an {@code MathArithMeticException } is thrown.</li>
274      * </ul>
275      * </p>
276      *
277      * @param n argument
278      * @return {@code n!}
279      * @throws MathArithmeticException if the result is too large to be represented
280      * by a {@code long}.
281      * @throws NotPositiveException if {@code n < 0}.
282      * @throws MathArithmeticException if {@code n > 20}: The factorial value is too
283      * large to fit in a {@code long}.
284      */
285     public static long factorial(final int n) throws NotPositiveException, MathArithmeticException {
286         if (n < 0) {
287             throw new NotPositiveException(LocalizedFormats.FACTORIAL_NEGATIVE_PARAMETER,
288                                            n);
289         }
290         if (n > 20) {
291             throw new MathArithmeticException();
292         }
293         return FACTORIALS[n];
294     }

296     /**
297      * Compute n!, the<a href="http://mathworld.wolfram.com/Factorial.html">
298      * factorial</a> of {@code n} (the product of the numbers 1 to n), as a
299      * {@code double}.
300      * The result should be small enough to fit into a {@code double}: The
301      * largest {@code n} for which {@code n!} does not exceed
302      * {@code Double.MAX_VALUE} is 170. If the computed value exceeds
303      * {@code Double.MAX_VALUE}, {@code Double.POSITIVE_INFINITY} is returned.
304      *
305      * @param n Argument.
306      * @return {@code n!}
307      * @throws NotPositiveException if {@code n < 0}.
308      */
309     public static double factorialDouble(final int n) throws NotPositiveException {
310         if (n < 0) {
311             throw new NotPositiveException(LocalizedFormats.FACTORIAL_NEGATIVE_PARAMETER,
312                                            n);
313         }
314         if (n < 21) {
315             return FACTORIALS[n];
316         }
317         return FastMath.floor(FastMath.exp(CombinatoricsUtils.factorialLog(n)) + 0.5);
318     }

331     /**
332      * Returns the <a
333      * href="http://mathworld.wolfram.com/StirlingNumberoftheSecondKind.html">
334      * Stirling number of the second kind</a>, "{@code S(n,k)}", the number of
335      * ways of partitioning an {@code n}-element set into {@code k} non-empty
336      * subsets.
337      * <p>
338      * The preconditions are {@code 0 <= k <= n } (otherwise
339      * {@code NotPositiveException} is thrown)
340      * </p>
341      * @param n the size of the set
342      * @param k the number of non-empty subsets
343      * @return {@code S(n,k)}
344      * @throws NotPositiveException if {@code k < 0}.
345      * @throws NumberIsTooLargeException if {@code k > n}.
346      * @throws MathArithmeticException if some overflow happens, typically for n exceeding 25 and
347      * k between 20 and n-2 (S(n,n-1) is handled specifically and does not overflow)
348      * @since 3.1
349      */
350     public static long stirlingS2(final int n, final int k)
351         throws NotPositiveException, NumberIsTooLargeException, MathArithmeticException {
352         if (k < 0) {
353             throw new NotPositiveException(k);
354         }
355         if (k > n) {
356             throw new NumberIsTooLargeException(k, n, true);
357         }
358 
359         long[][] stirlingS2 = STIRLING_S2.get();
360 
361         if (stirlingS2 == null) {
362             // the cache has never been initialized, compute the first numbers
363             // by direct recurrence relation
364 
365             // as S(26,9) = 11201516780955125625 is larger than Long.MAX_VALUE
366             // we must stop computation at row 26
367             final int maxIndex = 26;
368             stirlingS2 = new long[maxIndex][];
369             stirlingS2[0] = new long[] { 1l };
370             for (int i = 1; i < stirlingS2.length; ++i) {
371                 stirlingS2[i] = new long[i + 1];
372                 stirlingS2[i][0] = 0;
373                 stirlingS2[i][1] = 1;
374                 stirlingS2[i][i] = 1;
375                 for (int j = 2; j < i; ++j) {
376                     stirlingS2[i][j] = j * stirlingS2[i - 1][j] + stirlingS2[i - 1][j - 1];
377                 }
378             }
379 
380             // atomically save the cache
381             STIRLING_S2.compareAndSet(null, stirlingS2);
382 
383         }
384 
385         if (n < stirlingS2.length) {
386             // the number is in the small cache
387             return stirlingS2[n][k];
388         } else {
389             // use explicit formula to compute the number without caching it
390             if (k == 0) {
391                 return 0;
392             } else if (k == 1 || k == n) {
393                 return 1;
394             } else if (k == 2) {
395                 return (1l << (n - 1)) - 1l;
396             } else if (k == n - 1) {
397                 return binomialCoefficient(n, 2);
398             } else {
399                 // definition formula: note that this may trigger some overflow
400                 long sum = 0;
401                 long sign = ((k & 0x1) == 0) ? 1 : -1;
402                 for (int j = 1; j <= k; ++j) {
403                     sign = -sign;
404                     sum += sign * binomialCoefficient(k, j) * ArithmeticUtils.pow(j, n);
405                     if (sum < 0) {
406                         // there was an overflow somewhere
407                         throw new MathArithmeticException(LocalizedFormats.ARGUMENT_OUTSIDE_DOMAIN,
408                                                           n, 0, stirlingS2.length - 1);
409                     }
410                 }
411                 return sum / factorial(k);
412             }
413         }
414 
415     }

417     /**
418      * Returns an iterator whose range is the k-element subsets of {0, ..., n - 1}
419      * represented as {@code int[]} arrays.
420      * <p>
421      * The arrays returned by the iterator are sorted in descending order and
422      * they are visited in lexicographic order with significance from right to
423      * left. For example, combinationsIterator(4, 2) returns an Iterator that
424      * will generate the following sequence of arrays on successive calls to
425      * {@code next()}:</p><p>
426      * {@code [0, 1], [0, 2], [1, 2], [0, 3], [1, 3], [2, 3]}
427      * </p><p>
428      * If {@code k == 0} an Iterator containing an empty array is returned and
429      * if {@code k == n} an Iterator containing [0, ..., n -1] is returned.</p>
430      *
431      * @param n Size of the set from which subsets are selected.
432      * @param k Size of the subsets to be enumerated.
433      * @return an {@link Iterator iterator} over the k-sets in n.
434      * @throws NotPositiveException if {@code n < 0}.
435      * @throws NumberIsTooLargeException if {@code k > n}.
436      */
437     public static Iterator<int[]> combinationsIterator(int n, int k) {
438         return new Combinations(n, k).iterator();
439     }

442      * Check binomial preconditions.
443      *
444      * @param n Size of the set.
445      * @param k Size of the subsets to be counted.
446      * @throws NotPositiveException if {@code n < 0}.
447      * @throws NumberIsTooLargeException if {@code k > n}.
448      */
449     public static void checkBinomial(final int n,
450                                      final int k)
451         throws NumberIsTooLargeException,
452                NotPositiveException {
453         if (n < k) {
454             throw new NumberIsTooLargeException(LocalizedFormats.BINOMIAL_INVALID_PARAMETERS_ORDER,
455                                                 k, n, true);
456         }
457         if (n < 0) {
458             throw new NotPositiveException(LocalizedFormats.BINOMIAL_NEGATIVE_PARAMETER, n);
459         }
460     }

462     /**
463      * Class for computing the natural logarithm of the factorial of {@code n}.
464      * It allows to allocate a cache of precomputed values.
465      * In case of cache miss, computation is preformed by a call to
466      * {@link Gamma#logGamma(double)}.
467      */
468     public static final class FactorialLog {
469         /**
470          * Precomputed values of the function:
471          * {@code LOG_FACTORIALS[i] = log(i!)}.
472          */
473         private final double[] LOG_FACTORIALS;
474 
475         /**
476          * Creates an instance, reusing the already computed values if available.
477          *
478          * @param numValues Number of values of the function to compute.
479          * @param cache Existing cache.
480          * @throw NotPositiveException if {@code n < 0}.
481          */
482         private FactorialLog(int numValues,
483                              double[] cache) {
484             if (numValues < 0) {
485                 throw new NotPositiveException(numValues);
486             }
487 
488             LOG_FACTORIALS = new double[numValues];
489 
490             final int beginCopy = 2;
491             final int endCopy = cache == null || cache.length <= beginCopy ?
492                 beginCopy : cache.length <= numValues ?
493                 cache.length : numValues;
494 
495             // Copy available values.
496             for (int i = beginCopy; i < endCopy; i++) {
497                 LOG_FACTORIALS[i] = cache[i];
498             }
499 
500             // Precompute.
501             for (int i = endCopy; i < numValues; i++) {
502                 LOG_FACTORIALS[i] = LOG_FACTORIALS[i - 1] + FastMath.log(i);
503             }
504         }
505 
506         /**
507          * Creates an instance with no precomputed values.
508          * @return instance with no precomputed values
509          */
510         public static FactorialLog create() {
511             return new FactorialLog(0, null);
512         }
513 
514         /**
515          * Creates an instance with the specified cache size.
516          *
517          * @param cacheSize Number of precomputed values of the function.
518          * @return a new instance where {@code cacheSize} values have been
519          * precomputed.
520          * @throws NotPositiveException if {@code n < 0}.
521          */
522         public FactorialLog withCache(final int cacheSize) {
523             return new FactorialLog(cacheSize, LOG_FACTORIALS);
524         }
525 
526         /**
527          * Computes {@code log(n!)}.
528          *
529          * @param n Argument.
530          * @return {@code log(n!)}.
531          * @throws NotPositiveException if {@code n < 0}.
532          */
533         public double value(final int n) {
534             if (n < 0) {
535                 throw new NotPositiveException(LocalizedFormats.FACTORIAL_NEGATIVE_PARAMETER,
536                                                n);
537             }
538 
539             // Use cache of precomputed values.
540             if (n < LOG_FACTORIALS.length) {
541                 return LOG_FACTORIALS[n];
542             }
543 
544             // Use cache of precomputed factorial values.
545             if (n < FACTORIALS.length) {
546                 return FastMath.log(FACTORIALS[n]);
547             }
548 
549             // Delegate.
550             return Gamma.logGamma(n + 1);
551         }
552     }
