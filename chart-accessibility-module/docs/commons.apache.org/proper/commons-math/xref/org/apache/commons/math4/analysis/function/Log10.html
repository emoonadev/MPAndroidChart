18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.analysis.differentiation.DerivativeStructure;
21  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
22  import org.apache.commons.math4.util.FastMath;
23  
24  /**
25   * Base 10 logarithm function.
26   *
27   * @since 3.0
28   */
29  public class Log10 implements UnivariateDifferentiableFunction {
30  
31      /** {@inheritDoc} */
32      @Override
33      public double value(double x) {
34          return FastMath.log10(x);
35      }
36  
37      /** {@inheritDoc}
38       * @since 3.1
39       */
40      @Override
41      public DerivativeStructure value(final DerivativeStructure t) {
42          return t.log10();
43      }
44  
45  }
