17  package org.apache.commons.math4.optim.nonlinear.scalar;
18  
19  import org.apache.commons.math4.analysis.MultivariateFunction;
20  import org.apache.commons.math4.analysis.UnivariateFunction;
21  import org.apache.commons.math4.analysis.function.Logit;
22  import org.apache.commons.math4.analysis.function.Sigmoid;
23  import org.apache.commons.math4.exception.DimensionMismatchException;
24  import org.apache.commons.math4.exception.NumberIsTooSmallException;
25  import org.apache.commons.math4.util.FastMath;
26  import org.apache.commons.math4.util.MathUtils;

28  /**
29   * <p>Adapter for mapping bounded {@link MultivariateFunction} to unbounded ones.</p>
30   *
31   * <p>
32   * This adapter can be used to wrap functions subject to simple bounds on
33   * parameters so they can be used by optimizers that do <em>not</em> directly
34   * support simple bounds.
35   * </p>
36   * <p>
37   * The principle is that the user function that will be wrapped will see its
38   * parameters bounded as required, i.e when its {@code value} method is called
39   * with argument array {@code point}, the elements array will fulfill requirement
40   * {@code lower[i] <= point[i] <= upper[i]} for all i. Some of the components
41   * may be unbounded or bounded only on one side if the corresponding bound is
42   * set to an infinite value. The optimizer will not manage the user function by
43   * itself, but it will handle this adapter and it is this adapter that will take
44   * care the bounds are fulfilled. The adapter {@link #value(double[])} method will
45   * be called by the optimizer with unbound parameters, and the adapter will map
46   * the unbounded value to the bounded range using appropriate functions like
47   * {@link Sigmoid} for double bounded elements for example.
48   * </p>
49   * <p>
50   * As the optimizer sees only unbounded parameters, it should be noted that the
51   * start point or simplex expected by the optimizer should be unbounded, so the
52   * user is responsible for converting his bounded point to unbounded by calling
53   * {@link #boundedToUnbounded(double[])} before providing them to the optimizer.
54   * For the same reason, the point returned by the {@link
55   * org.apache.commons.math4.optim.BaseMultivariateOptimizer#optimize(OptimizationData[])}
56   * method is unbounded. So to convert this point to bounded, users must call
57   * {@link #unboundedToBounded(double[])} by themselves!</p>
58   * <p>
59   * This adapter is only a poor man solution to simple bounds optimization constraints
60   * that can be used with simple optimizers like
61   * {@link org.apache.commons.math4.optim.nonlinear.scalar.noderiv.SimplexOptimizer
62   * SimplexOptimizer}.
63   * A better solution is to use an optimizer that directly supports simple bounds like
64   * {@link org.apache.commons.math4.optim.nonlinear.scalar.noderiv.CMAESOptimizer
65   * CMAESOptimizer} or
66   * {@link org.apache.commons.math4.optim.nonlinear.scalar.noderiv.BOBYQAOptimizer
67   * BOBYQAOptimizer}.
68   * One caveat of this poor-man's solution is that behavior near the bounds may be
69   * numerically unstable as bounds are mapped from infinite values.
70   * Another caveat is that convergence values are evaluated by the optimizer with
71   * respect to unbounded variables, so there will be scales differences when
72   * converted to bounded variables.
73   * </p>
74   *
75   * @see MultivariateFunctionPenaltyAdapter
76   *
77   * @since 3.0
78   */
79  public class MultivariateFunctionMappingAdapter
80      implements MultivariateFunction {
81      /** Underlying bounded function. */
82      private final MultivariateFunction bounded;
83      /** Mapping functions. */
84      private final Mapper[] mappers;
85  
86      /** Simple constructor.
87       * @param bounded bounded function
88       * @param lower lower bounds for each element of the input parameters array
89       * (some elements may be set to {@code Double.NEGATIVE_INFINITY} for
90       * unbounded values)
91       * @param upper upper bounds for each element of the input parameters array
92       * (some elements may be set to {@code Double.POSITIVE_INFINITY} for
93       * unbounded values)
94       * @exception DimensionMismatchException if lower and upper bounds are not
95       * consistent, either according to dimension or to values
96       */
97      public MultivariateFunctionMappingAdapter(final MultivariateFunction bounded,
98                                                final double[] lower, final double[] upper) {
99          // safety checks
100         MathUtils.checkNotNull(lower);
101         MathUtils.checkNotNull(upper);
102         if (lower.length != upper.length) {
103             throw new DimensionMismatchException(lower.length, upper.length);
104         }
105         for (int i = 0; i < lower.length; ++i) {
106             // note the following test is written in such a way it also fails for NaN
107             if (!(upper[i] >= lower[i])) {
108                 throw new NumberIsTooSmallException(upper[i], lower[i], true);
109             }
110         }
111 
112         this.bounded = bounded;
113         this.mappers = new Mapper[lower.length];
114         for (int i = 0; i < mappers.length; ++i) {
115             if (Double.isInfinite(lower[i])) {
116                 if (Double.isInfinite(upper[i])) {
117                     // element is unbounded, no transformation is needed
118                     mappers[i] = new NoBoundsMapper();
119                 } else {
120                     // element is simple-bounded on the upper side
121                     mappers[i] = new UpperBoundMapper(upper[i]);
122                 }
123             } else {
124                 if (Double.isInfinite(upper[i])) {
125                     // element is simple-bounded on the lower side
126                     mappers[i] = new LowerBoundMapper(lower[i]);
127                 } else {
128                     // element is double-bounded
129                     mappers[i] = new LowerUpperBoundMapper(lower[i], upper[i]);
130                 }
131             }
132         }
133     }
134 
135     /**
136      * Maps an array from unbounded to bounded.
137      *
138      * @param point Unbounded values.
139      * @return the bounded values.
140      */
141     public double[] unboundedToBounded(double[] point) {
142         // Map unbounded input point to bounded point.
143         final double[] mapped = new double[mappers.length];
144         for (int i = 0; i < mappers.length; ++i) {
145             mapped[i] = mappers[i].unboundedToBounded(point[i]);
146         }
147 
148         return mapped;
149     }
150 
151     /**
152      * Maps an array from bounded to unbounded.
153      *
154      * @param point Bounded values.
155      * @return the unbounded values.
156      */
157     public double[] boundedToUnbounded(double[] point) {
158         // Map bounded input point to unbounded point.
159         final double[] mapped = new double[mappers.length];
160         for (int i = 0; i < mappers.length; ++i) {
161             mapped[i] = mappers[i].boundedToUnbounded(point[i]);
162         }
163 
164         return mapped;
165     }
166 
167     /**
168      * Compute the underlying function value from an unbounded point.
169      * <p>
170      * This method simply bounds the unbounded point using the mappings
171      * set up at construction and calls the underlying function using
172      * the bounded point.
173      * </p>
174      * @param point unbounded value
175      * @return underlying function value
176      * @see #unboundedToBounded(double[])
177      */
178     @Override
179     public double value(double[] point) {
180         return bounded.value(unboundedToBounded(point));
181     }
182 
183     /** Mapping interface. */
184     private interface Mapper {
185         /**
186          * Maps a value from unbounded to bounded.
187          *
188          * @param y Unbounded value.
189          * @return the bounded value.
190          */
191         double unboundedToBounded(double y);
192 
193         /**
194          * Maps a value from bounded to unbounded.
195          *
196          * @param x Bounded value.
197          * @return the unbounded value.
198          */
199         double boundedToUnbounded(double x);
200     }
201 
202     /** Local class for no bounds mapping. */
203     private static class NoBoundsMapper implements Mapper {
204         /** {@inheritDoc} */
205         @Override
206         public double unboundedToBounded(final double y) {
207             return y;
208         }
209 
210         /** {@inheritDoc} */
211         @Override
212         public double boundedToUnbounded(final double x) {
213             return x;
214         }
215     }
216 
217     /** Local class for lower bounds mapping. */
218     private static class LowerBoundMapper implements Mapper {
219         /** Low bound. */
220         private final double lower;
221 
222         /**
223          * Simple constructor.
224          *
225          * @param lower lower bound
226          */
227         LowerBoundMapper(final double lower) {
228             this.lower = lower;
229         }
230 
231         /** {@inheritDoc} */
232         @Override
233         public double unboundedToBounded(final double y) {
234             return lower + FastMath.exp(y);
235         }
236 
237         /** {@inheritDoc} */
238         @Override
239         public double boundedToUnbounded(final double x) {
240             return FastMath.log(x - lower);
241         }
242 
243     }
244 
245     /** Local class for upper bounds mapping. */
246     private static class UpperBoundMapper implements Mapper {
247 
248         /** Upper bound. */
249         private final double upper;
250 
251         /** Simple constructor.
252          * @param upper upper bound
253          */
254         UpperBoundMapper(final double upper) {
255             this.upper = upper;
256         }
257 
258         /** {@inheritDoc} */
259         @Override
260         public double unboundedToBounded(final double y) {
261             return upper - FastMath.exp(-y);
262         }
263 
264         /** {@inheritDoc} */
265         @Override
266         public double boundedToUnbounded(final double x) {
267             return -FastMath.log(upper - x);
268         }
269 
270     }
271 
272     /** Local class for lower and bounds mapping. */
273     private static class LowerUpperBoundMapper implements Mapper {
274         /** Function from unbounded to bounded. */
275         private final UnivariateFunction boundingFunction;
276         /** Function from bounded to unbounded. */
277         private final UnivariateFunction unboundingFunction;
278 
279         /**
280          * Simple constructor.
281          *
282          * @param lower lower bound
283          * @param upper upper bound
284          */
285         LowerUpperBoundMapper(final double lower, final double upper) {
286             boundingFunction   = new Sigmoid(lower, upper);
287             unboundingFunction = new Logit(lower, upper);
288         }
289 
290         /** {@inheritDoc} */
291         @Override
292         public double unboundedToBounded(final double y) {
293             return boundingFunction.value(y);
294         }
295 
296         /** {@inheritDoc} */
297         @Override
298         public double boundedToUnbounded(final double x) {
299             return unboundingFunction.value(x);
300         }
301     }
