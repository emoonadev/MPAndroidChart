18  package org.apache.commons.math4.ml.neuralnet.twod.util;
19  
20  import org.apache.commons.math4.ml.neuralnet.twod.NeuronSquareMesh2D;
21  
22  /**
23   * Interface for algorithms that compute some metrics of the projection of
24   * data on a 2D-map.
25   * @since 3.6
26   */
27  public interface MapDataVisualization {
28      /**
29       * Creates an image of the {@code data} metrics when represented by the
30       * {@code map}.
31       *
32       * @param map Map.
33       * @param data Data.
34       * @return a 2D-array (in row major order) representing the metrics.
35       */
36      double[][] computeImage(NeuronSquareMesh2D map,
37                              Iterable<double[]> data);
38  }
