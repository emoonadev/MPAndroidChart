18  package org.apache.commons.math4.ml.neuralnet;
19  
20  /**
21   * Defines neighbourhood types.
22   *
23   * @since 3.3
24   */
25  public enum SquareNeighbourhood {
26      /**
27       * <a href="http://en.wikipedia.org/wiki/Von_Neumann_neighborhood"
28       *  Von Neumann neighbourhood</a>: in two dimensions, each (internal)
29       * neuron has four neighbours.
30       */
31      VON_NEUMANN,
32      /**
33       * <a href="http://en.wikipedia.org/wiki/Moore_neighborhood"
34       *  Moore neighbourhood</a>: in two dimensions, each (internal)
35       * neuron has eight neighbours.
36       */
37      MOORE,
38  }
