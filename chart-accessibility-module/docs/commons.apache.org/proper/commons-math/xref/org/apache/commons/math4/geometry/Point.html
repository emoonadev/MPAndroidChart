17  package org.apache.commons.math4.geometry;
18  
19  import java.io.Serializable;
20  
21  /** This interface represents a generic geometrical point.
22   * @param <S> Type of the space.
23   * @see Space
24   * @see Vector
25   * @since 3.3
26   */
27  public interface Point<S extends Space> extends Serializable {
28  
29      /** Get the space to which the point belongs.
30       * @return containing space
31       */
32      Space getSpace();
33  
34      /**
35       * Returns true if any coordinate of this point is NaN; false otherwise
36       * @return  true if any coordinate of this point is NaN; false otherwise
37       */
38      boolean isNaN();
39  
40      /** Compute the distance between the instance and another point.
41       * @param p second point
42       * @return the distance between the instance and p
43       */
44      double distance(Point<S> p);
45  
46  }
