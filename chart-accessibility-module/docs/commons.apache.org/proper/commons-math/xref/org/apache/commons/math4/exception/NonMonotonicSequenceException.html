17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.LocalizedFormats;
20  import org.apache.commons.math4.util.MathArrays;
21  
22  /**
23   * Exception to be thrown when the a sequence of values is not monotonically
24   * increasing or decreasing.
25   *
26   * @since 2.2 (name changed to "NonMonotonicSequenceException" in 3.0)
27   */
28  public class NonMonotonicSequenceException extends MathIllegalNumberException {
29      /** Serializable version Id. */
30      private static final long serialVersionUID = 3596849179428944575L;
31      /**
32       * Direction (positive for increasing, negative for decreasing).
33       */
34      private final MathArrays.OrderDirection direction;
35      /**
36       * Whether the sequence must be strictly increasing or decreasing.
37       */
38      private final boolean strict;
39      /**
40       * Index of the wrong value.
41       */
42      private final int index;
43      /**
44       * Previous value.
45       */
46      private final Number previous;
47  
48      /**
49       * Construct the exception.
50       * This constructor uses default values assuming that the sequence should
51       * have been strictly increasing.
52       *
53       * @param wrong Value that did not match the requirements.
54       * @param previous Previous value in the sequence.
55       * @param index Index of the value that did not match the requirements.
56       */
57      public NonMonotonicSequenceException(Number wrong,
58                                           Number previous,
59                                           int index) {
60          this(wrong, previous, index, MathArrays.OrderDirection.INCREASING, true);
61      }
62  
63      /**
64       * Construct the exception.
65       *
66       * @param wrong Value that did not match the requirements.
67       * @param previous Previous value in the sequence.
68       * @param index Index of the value that did not match the requirements.
69       * @param direction Strictly positive for a sequence required to be
70       * increasing, negative (or zero) for a decreasing sequence.
71       * @param strict Whether the sequence must be strictly increasing or
72       * decreasing.
73       */
74      public NonMonotonicSequenceException(Number wrong,
75                                           Number previous,
76                                           int index,
77                                           MathArrays.OrderDirection direction,
78                                           boolean strict) {
79          super(direction == MathArrays.OrderDirection.INCREASING ?
80                (strict ?
81                 LocalizedFormats.NOT_STRICTLY_INCREASING_SEQUENCE :
82                 LocalizedFormats.NOT_INCREASING_SEQUENCE) :
83                (strict ?
84                 LocalizedFormats.NOT_STRICTLY_DECREASING_SEQUENCE :
85                 LocalizedFormats.NOT_DECREASING_SEQUENCE),
86                wrong, previous, Integer.valueOf(index), Integer.valueOf(index - 1));
87  
88          this.direction = direction;
89          this.strict = strict;
90          this.index = index;
91          this.previous = previous;
92      }
93  
94      /**
95       * @return the order direction.
96       **/
97      public MathArrays.OrderDirection getDirection() {
98          return direction;
99      }
100     /**
101      * @return {@code true} is the sequence should be strictly monotonic.
102      **/
103     public boolean getStrict() {
104         return strict;
105     }
106     /**
107      * Get the index of the wrong value.
108      *
109      * @return the current index.
110      */
111     public int getIndex() {
112         return index;
113     }
114     /**
115      * @return the previous value.
116      */
117     public Number getPrevious() {
118         return previous;
119     }
120 }
