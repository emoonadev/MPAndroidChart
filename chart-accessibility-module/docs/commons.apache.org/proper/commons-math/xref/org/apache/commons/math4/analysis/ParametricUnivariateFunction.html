18  package org.apache.commons.math4.analysis;
19  
20  /**
21   * An interface representing a real function that depends on one independent
22   * variable plus some extra parameters.
23   *
24   * @since 3.0
25   */
26  public interface ParametricUnivariateFunction {
27      /**
28       * Compute the value of the function.
29       *
30       * @param x Point for which the function value should be computed.
31       * @param parameters Function parameters.
32       * @return the value.
33       */
34      double value(double x, double ... parameters);
35  
36      /**
37       * Compute the gradient of the function with respect to its parameters.
38       *
39       * @param x Point for which the function value should be computed.
40       * @param parameters Function parameters.
41       * @return the value.
42       */
43      double[] gradient(double x, double ... parameters);
44  }
