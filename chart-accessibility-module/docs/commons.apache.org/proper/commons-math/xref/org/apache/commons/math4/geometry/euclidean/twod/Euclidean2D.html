18  package org.apache.commons.math4.geometry.euclidean.twod;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.geometry.Space;
23  import org.apache.commons.math4.geometry.euclidean.oned.Euclidean1D;
24  
25  /**
26   * This class implements a two-dimensional space.
27   * @since 3.0
28   */
29  public class Euclidean2D implements Serializable, Space {
30  
31      /** Serializable version identifier. */
32      private static final long serialVersionUID = 4793432849757649566L;
33  
34      /** Private constructor for the singleton.
35       */
36      private Euclidean2D() {
37      }
38  
39      /** Get the unique instance.
40       * @return the unique instance
41       */
42      public static Euclidean2D getInstance() {
43          return LazyHolder.INSTANCE;
44      }
45  
46      /** {@inheritDoc} */
47      @Override
48      public int getDimension() {
49          return 2;
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      public Euclidean1D getSubSpace() {
55          return Euclidean1D.getInstance();
56      }
57  
58      // CHECKSTYLE: stop HideUtilityClassConstructor
59      /** Holder for the instance.
60       * <p>We use here the Initialization On Demand Holder Idiom.</p>
61       */
62      private static class LazyHolder {
63          /** Cached field instance. */
64          private static final Euclidean2D INSTANCE = new Euclidean2D();
65      }
66      // CHECKSTYLE: resume HideUtilityClassConstructor
67  
68      /** Handle deserialization of the singleton.
69       * @return the singleton instance
70       */
71      private Object readResolve() {
72          // return the singleton instance
73          return LazyHolder.INSTANCE;
74      }
75  
76  }
