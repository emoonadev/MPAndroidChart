17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  
21  /**
22   * Base class for all preconditions violation exceptions.
23   * In most cases, this class should not be instantiated directly: it should
24   * serve as a base class to create all the exceptions that have the semantics
25   * of the standard {@link IllegalArgumentException}.
26   *
27   * @since 2.2
28   */
29  public class MathIllegalArgumentException extends MathRuntimeException {
30      /** Serializable version Id. */
31      private static final long serialVersionUID = -6024911025449780478L;
32  
33      /**
34       * @param pattern Message pattern explaining the cause of the error.
35       * @param args Arguments.
36       */
37      public MathIllegalArgumentException(Localizable pattern, Object ... args) {
38          super(pattern, args);
39      }
40  
41  }
