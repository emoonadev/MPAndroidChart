18  package org.apache.commons.math4.analysis.solvers;
19  
20  import org.apache.commons.math4.analysis.UnivariateFunction;
21  import org.apache.commons.math4.exception.MaxCountExceededException;
22  import org.apache.commons.math4.exception.NoBracketingException;
23  import org.apache.commons.math4.exception.NullArgumentException;
24  import org.apache.commons.math4.exception.NumberIsTooLargeException;
25  import org.apache.commons.math4.exception.TooManyEvaluationsException;
26  import org.apache.commons.math4.util.Incrementor;
27  import org.apache.commons.math4.util.MathUtils;
28  
29  /**
30   * Provide a default implementation for several functions useful to generic
31   * solvers.
32   * The default values for relative and function tolerances are 1e-14
33   * and 1e-15, respectively. It is however highly recommended to not
34   * rely on the default, but rather carefully consider values that match
35   * user's expectations, as well as the specifics of each implementation.
36   *
37   * @param <FUNC> Type of function to solve.
38   *
39   * @since 2.0
40   */
41  public abstract class BaseAbstractUnivariateSolver<FUNC extends UnivariateFunction>
42      implements BaseUnivariateSolver<FUNC> {
43      /** Default relative accuracy. */
44      private static final double DEFAULT_RELATIVE_ACCURACY = 1e-14;
45      /** Default function value accuracy. */
46      private static final double DEFAULT_FUNCTION_VALUE_ACCURACY = 1e-15;
47      /** Function value accuracy. */
48      private final double functionValueAccuracy;
49      /** Absolute accuracy. */
50      private final double absoluteAccuracy;
51      /** Relative accuracy. */
52      private final double relativeAccuracy;
53      /** Evaluations counter. */
54      private final Incrementor evaluations = new Incrementor();
55      /** Lower end of search interval. */
56      private double searchMin;
57      /** Higher end of search interval. */
58      private double searchMax;
59      /** Initial guess. */
60      private double searchStart;
61      /** Function to solve. */
62      private FUNC function;
63  
64      /**
65       * Construct a solver with given absolute accuracy.
66       *
67       * @param absoluteAccuracy Maximum absolute error.
68       */
69      protected BaseAbstractUnivariateSolver(final double absoluteAccuracy) {
70          this(DEFAULT_RELATIVE_ACCURACY,
71               absoluteAccuracy,
72               DEFAULT_FUNCTION_VALUE_ACCURACY);
73      }
74  
75      /**
76       * Construct a solver with given accuracies.
77       *
78       * @param relativeAccuracy Maximum relative error.
79       * @param absoluteAccuracy Maximum absolute error.
80       */
81      protected BaseAbstractUnivariateSolver(final double relativeAccuracy,
82                                             final double absoluteAccuracy) {
83          this(relativeAccuracy,
84               absoluteAccuracy,
85               DEFAULT_FUNCTION_VALUE_ACCURACY);
86      }
87  
88      /**
89       * Construct a solver with given accuracies.
90       *
91       * @param relativeAccuracy Maximum relative error.
92       * @param absoluteAccuracy Maximum absolute error.
93       * @param functionValueAccuracy Maximum function value error.
94       */
95      protected BaseAbstractUnivariateSolver(final double relativeAccuracy,
96                                             final double absoluteAccuracy,
97                                             final double functionValueAccuracy) {
98          this.absoluteAccuracy = absoluteAccuracy;
99          this.relativeAccuracy = relativeAccuracy;
100         this.functionValueAccuracy = functionValueAccuracy;
101     }
102 
103     /** {@inheritDoc} */
104     @Override
105     public int getMaxEvaluations() {
106         return evaluations.getMaximalCount();
107     }
108     /** {@inheritDoc} */
109     @Override
110     public int getEvaluations() {
111         return evaluations.getCount();
112     }
113     /**
114      * @return the lower end of the search interval.
115      */
116     public double getMin() {
117         return searchMin;
118     }
119     /**
120      * @return the higher end of the search interval.
121      */
122     public double getMax() {
123         return searchMax;
124     }
125     /**
126      * @return the initial guess.
127      */
128     public double getStartValue() {
129         return searchStart;
130     }
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     public double getAbsoluteAccuracy() {
136         return absoluteAccuracy;
137     }
138     /**
139      * {@inheritDoc}
140      */
141     @Override
142     public double getRelativeAccuracy() {
143         return relativeAccuracy;
144     }
145     /**
146      * {@inheritDoc}
147      */
148     @Override
149     public double getFunctionValueAccuracy() {
150         return functionValueAccuracy;
151     }
152 
153     /**
154      * Compute the objective function value.
155      *
156      * @param point Point at which the objective function must be evaluated.
157      * @return the objective function value at specified point.
158      * @throws TooManyEvaluationsException if the maximal number of evaluations
159      * is exceeded.
160      */
161     protected double computeObjectiveValue(double point)
162         throws TooManyEvaluationsException {
163         incrementEvaluationCount();
164         return function.value(point);
165     }
166 
167     /**
168      * Prepare for computation.
169      * Subclasses must call this method if they override any of the
170      * {@code solve} methods.
171      *
172      * @param f Function to solve.
173      * @param min Lower bound for the interval.
174      * @param max Upper bound for the interval.
175      * @param startValue Start value to use.
176      * @param maxEval Maximum number of evaluations.
177      * @exception NullArgumentException if f is null
178      */
179     protected void setup(int maxEval,
180                          FUNC f,
181                          double min, double max,
182                          double startValue)
183         throws NullArgumentException {
184         // Checks.
185         MathUtils.checkNotNull(f);
186 
187         // Reset.
188         searchMin = min;
189         searchMax = max;
190         searchStart = startValue;
191         function = f;
192         evaluations.setMaximalCount(maxEval);
193         evaluations.resetCount();
194     }
195 
196     /** {@inheritDoc} */
197     @Override
198     public double solve(int maxEval, FUNC f, double min, double max, double startValue)
199         throws TooManyEvaluationsException,
200                NoBracketingException {
201         // Initialization.
202         setup(maxEval, f, min, max, startValue);
203 
204         // Perform computation.
205         return doSolve();
206     }
207 
208     /** {@inheritDoc} */
209     @Override
210     public double solve(int maxEval, FUNC f, double min, double max) {
211         return solve(maxEval, f, min, max, min + 0.5 * (max - min));
212     }
213 
214     /** {@inheritDoc} */
215     @Override
216     public double solve(int maxEval, FUNC f, double startValue)
217         throws TooManyEvaluationsException,
218                NoBracketingException {
219         return solve(maxEval, f, Double.NaN, Double.NaN, startValue);
220     }
221 
222     /**
223      * Method for implementing actual optimization algorithms in derived
224      * classes.
225      *
226      * @return the root.
227      * @throws TooManyEvaluationsException if the maximal number of evaluations
228      * is exceeded.
229      * @throws NoBracketingException if the initial search interval does not bracket
230      * a root and the solver requires it.
231      */
232     protected abstract double doSolve()
233         throws TooManyEvaluationsException, NoBracketingException;
234 
235     /**
236      * Check whether the function takes opposite signs at the endpoints.
237      *
238      * @param lower Lower endpoint.
239      * @param upper Upper endpoint.
240      * @return {@code true} if the function values have opposite signs at the
241      * given points.
242      */
243     protected boolean isBracketing(final double lower,
244                                    final double upper) {
245         return UnivariateSolverUtils.isBracketing(function, lower, upper);
246     }
247 
248     /**
249      * Check whether the arguments form a (strictly) increasing sequence.
250      *
251      * @param start First number.
252      * @param mid Second number.
253      * @param end Third number.
254      * @return {@code true} if the arguments form an increasing sequence.
255      */
256     protected boolean isSequence(final double start,
257                                  final double mid,
258                                  final double end) {
259         return UnivariateSolverUtils.isSequence(start, mid, end);
260     }
261 
262     /**
263      * Check that the endpoints specify an interval.
264      *
265      * @param lower Lower endpoint.
266      * @param upper Upper endpoint.
267      * @throws NumberIsTooLargeException if {@code lower >= upper}.
268      */
269     protected void verifyInterval(final double lower,
270                                   final double upper)
271         throws NumberIsTooLargeException {
272         UnivariateSolverUtils.verifyInterval(lower, upper);
273     }
274 
275     /**
276      * Check that {@code lower < initial < upper}.
277      *
278      * @param lower Lower endpoint.
279      * @param initial Initial value.
280      * @param upper Upper endpoint.
281      * @throws NumberIsTooLargeException if {@code lower >= initial} or
282      * {@code initial >= upper}.
283      */
284     protected void verifySequence(final double lower,
285                                   final double initial,
286                                   final double upper)
287         throws NumberIsTooLargeException {
288         UnivariateSolverUtils.verifySequence(lower, initial, upper);
289     }
290 
291     /**
292      * Check that the endpoints specify an interval and the function takes
293      * opposite signs at the endpoints.
294      *
295      * @param lower Lower endpoint.
296      * @param upper Upper endpoint.
297      * @throws NullArgumentException if the function has not been set.
298      * @throws NoBracketingException if the function has the same sign at
299      * the endpoints.
300      */
301     protected void verifyBracketing(final double lower,
302                                     final double upper)
303         throws NullArgumentException,
304                NoBracketingException {
305         UnivariateSolverUtils.verifyBracketing(function, lower, upper);
306     }
307 
308     /**
309      * Increment the evaluation count by one.
310      * Method {@link #computeObjectiveValue(double)} calls this method internally.
311      * It is provided for subclasses that do not exclusively use
312      * {@code computeObjectiveValue} to solve the function.
313      * See e.g. {@link AbstractUnivariateDifferentiableSolver}.
314      *
315      * @throws TooManyEvaluationsException when the allowed number of function
316      * evaluations has been exhausted.
317      */
318     protected void incrementEvaluationCount()
319         throws TooManyEvaluationsException {
320         try {
321             evaluations.incrementCount();
322         } catch (MaxCountExceededException e) {
323             throw new TooManyEvaluationsException(e.getMax());
324         }
325     }
