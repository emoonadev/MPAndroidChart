18  package org.apache.commons.math4.ml.neuralnet.sofm;
19  
20  import org.apache.commons.math4.exception.OutOfRangeException;
21  import org.apache.commons.math4.ml.neuralnet.sofm.util.ExponentialDecayFunction;
22  import org.apache.commons.math4.ml.neuralnet.sofm.util.QuasiSigmoidDecayFunction;
23  
24  /**
25   * Factory for creating instances of {@link LearningFactorFunction}.
26   *
27   * @since 3.3
28   */
29  public class LearningFactorFunctionFactory {
30      /** Class contains only static methods. */
31      private LearningFactorFunctionFactory() {}
32  
33      /**
34       * Creates an exponential decay {@link LearningFactorFunction function}.
35       * It will compute <code>a e<sup>-x / b</sup></code>,
36       * where {@code x} is the (integer) independent variable and
37       * <ul>
38       *  <li><code>a = initValue</code>
39       *  <li><code>b = -numCall / ln(valueAtNumCall / initValue)</code>
40       * </ul>
41       *
42       * @param initValue Initial value, i.e.
43       * {@link LearningFactorFunction#value(long) value(0)}.
44       * @param valueAtNumCall Value of the function at {@code numCall}.
45       * @param numCall Argument for which the function returns
46       * {@code valueAtNumCall}.
47       * @return the learning factor function.
48       * @throws org.apache.commons.math4.exception.OutOfRangeException
49       * if {@code initValue <= 0} or {@code initValue > 1}.
50       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
51       * if {@code valueAtNumCall <= 0}.
52       * @throws org.apache.commons.math4.exception.NumberIsTooLargeException
53       * if {@code valueAtNumCall >= initValue}.
54       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
55       * if {@code numCall <= 0}.
56       */
57      public static LearningFactorFunction exponentialDecay(final double initValue,
58                                                            final double valueAtNumCall,
59                                                            final long numCall) {
60          if (initValue <= 0 ||
61              initValue > 1) {
62              throw new OutOfRangeException(initValue, 0, 1);
63          }
64  
65          return new LearningFactorFunction() {
66              /** DecayFunction. */
67              private final ExponentialDecayFunction decay
68                  = new ExponentialDecayFunction(initValue, valueAtNumCall, numCall);
69  
70              /** {@inheritDoc} */
71              @Override
72              public double value(long n) {
73                  return decay.value(n);
74              }
75          };
76      }
77  
78      /**
79       * Creates an sigmoid-like {@code LearningFactorFunction function}.
80       * The function {@code f} will have the following properties:
81       * <ul>
82       *  <li>{@code f(0) = initValue}</li>
83       *  <li>{@code numCall} is the inflexion point</li>
84       *  <li>{@code slope = f'(numCall)}</li>
85       * </ul>
86       *
87       * @param initValue Initial value, i.e.
88       * {@link LearningFactorFunction#value(long) value(0)}.
89       * @param slope Value of the function derivative at {@code numCall}.
90       * @param numCall Inflexion point.
91       * @return the learning factor function.
92       * @throws org.apache.commons.math4.exception.OutOfRangeException
93       * if {@code initValue <= 0} or {@code initValue > 1}.
94       * @throws org.apache.commons.math4.exception.NumberIsTooLargeException
95       * if {@code slope >= 0}.
96       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
97       * if {@code numCall <= 0}.
98       */
99      public static LearningFactorFunction quasiSigmoidDecay(final double initValue,
100                                                            final double slope,
101                                                            final long numCall) {
102         if (initValue <= 0 ||
103             initValue > 1) {
104             throw new OutOfRangeException(initValue, 0, 1);
105         }
106 
107         return new LearningFactorFunction() {
108             /** DecayFunction. */
109             private final QuasiSigmoidDecayFunction decay
110                 = new QuasiSigmoidDecayFunction(initValue, slope, numCall);
111 
112             /** {@inheritDoc} */
113             @Override
114             public double value(long n) {
115                 return decay.value(n);
116             }
117         };
118     }
119 }
