17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when some argument is out of range.
24   *
25   * @since 2.2
26   */
27  public class OutOfRangeException extends MathIllegalNumberException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = 111601815794403609L;
30      /** Lower bound. */
31      private final Number lo;
32      /** Higher bound. */
33      private final Number hi;
34  
35      /**
36       * Construct an exception from the mismatched dimensions.
37       *
38       * @param wrong Requested value.
39       * @param lo Lower bound.
40       * @param hi Higher bound.
41       */
42      public OutOfRangeException(Number wrong,
43                                 Number lo,
44                                 Number hi) {
45          this(LocalizedFormats.OUT_OF_RANGE_SIMPLE, wrong, lo, hi);
46      }
47  
48      /**
49       * Construct an exception from the mismatched dimensions with a
50       * specific context information.
51       *
52       * @param specific Context information.
53       * @param wrong Requested value.
54       * @param lo Lower bound.
55       * @param hi Higher bound.
56       */
57      public OutOfRangeException(Localizable specific,
58                                 Number wrong,
59                                 Number lo,
60                                 Number hi) {
61          super(specific, wrong, lo, hi);
62          this.lo = lo;
63          this.hi = hi;
64      }
65  
66      /**
67       * @return the lower bound.
68       */
69      public Number getLo() {
70          return lo;
71      }
72      /**
73       * @return the higher bound.
74       */
75      public Number getHi() {
76          return hi;
77      }
78  }
