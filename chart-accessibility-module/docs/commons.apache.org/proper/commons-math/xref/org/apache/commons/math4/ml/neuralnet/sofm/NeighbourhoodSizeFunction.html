18  package org.apache.commons.math4.ml.neuralnet.sofm;
19  
20  /**
21   * Provides the network neighbourhood's size as a function of the
22   * number of calls already performed during the learning task.
23   * The "neighbourhood" is the set of neurons that can be reached
24   * by traversing at most the number of links returned by this
25   * function.
26   *
27   * @since 3.3
28   */
29  public interface NeighbourhoodSizeFunction {
30      /**
31       * Computes the neighbourhood size at the current call.
32       *
33       * @param numCall Current step of the training task.
34       * @return the value of the function at {@code numCall}.
35       */
36      int value(long numCall);
37  }
