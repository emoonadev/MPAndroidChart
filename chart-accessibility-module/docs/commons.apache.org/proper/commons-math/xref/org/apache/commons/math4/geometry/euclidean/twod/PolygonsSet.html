55      /** Build a polygons set from a BSP tree.
56       * <p>The leaf nodes of the BSP tree <em>must</em> have a
57       * {@code Boolean} attribute representing the inside status of
58       * the corresponding cell (true for inside cells, false for outside
59       * cells). In order to avoid building too many small objects, it is
60       * recommended to use the predefined constants
61       * {@code Boolean.TRUE} and {@code Boolean.FALSE}</p>
62       * <p>
63       * This constructor is aimed at expert use, as building the tree may
64       * be a difficult task. It is not intended for general use and for
65       * performances reasons does not check thoroughly its input, as this would
66       * require walking the full tree each time. Failing to provide a tree with
67       * the proper attributes, <em>will</em> therefore generate problems like
68       * {@link NullPointerException} or {@link ClassCastException} only later on.
69       * This limitation is known and explains why this constructor is for expert
70       * use only. The caller does have the responsibility to provided correct arguments.
71       * </p>
72       * @param tree inside/outside BSP tree representing the region
73       * @param tolerance tolerance below which points are considered identical
74       * @since 3.3
75       */
76      public PolygonsSet(final BSPTree<Euclidean2D> tree, final double tolerance) {
77          super(tree, tolerance);
78      }

80      /** Build a polygons set from a Boundary REPresentation (B-rep).
81       * <p>The boundary is provided as a collection of {@link
82       * SubHyperplane sub-hyperplanes}. Each sub-hyperplane has the
83       * interior part of the region on its minus side and the exterior on
84       * its plus side.</p>
85       * <p>The boundary elements can be in any order, and can form
86       * several non-connected sets (like for example polygons with holes
87       * or a set of disjoint polygons considered as a whole). In
88       * fact, the elements do not even need to be connected together
89       * (their topological connections are not used here). However, if the
90       * boundary does not really separate an inside open from an outside
91       * open (open having here its topological meaning), then subsequent
92       * calls to the {@link
93       * org.apache.commons.math4.geometry.partitioning.Region#checkPoint(org.apache.commons.math4.geometry.Point)
94       * checkPoint} method will not be meaningful anymore.</p>
95       * <p>If the boundary is empty, the region will represent the whole
96       * space.</p>
97       * @param boundary collection of boundary elements, as a
98       * collection of {@link SubHyperplane SubHyperplane} objects
99       * @param tolerance tolerance below which points are considered identical
100      * @since 3.3
101      */
102     public PolygonsSet(final Collection<SubHyperplane<Euclidean2D>> boundary, final double tolerance) {
103         super(boundary, tolerance);
104     }

120     /** Build a polygon from a simple list of vertices.
121      * <p>The boundary is provided as a list of points considering to
122      * represent the vertices of a simple loop. The interior part of the
123      * region is on the left side of this path and the exterior is on its
124      * right side.</p>
125      * <p>This constructor does not handle polygons with a boundary
126      * forming several disconnected paths (such as polygons with holes).</p>
127      * <p>For cases where this simple constructor applies, it is expected to
128      * be numerically more robust than the {@link #PolygonsSet(Collection,double) general
129      * constructor} using {@link SubHyperplane subhyperplanes}.</p>
130      * <p>If the list is empty, the region will represent the whole
131      * space.</p>
132      * <p>
133      * Polygons with thin pikes or dents are inherently difficult to handle because
134      * they involve lines with almost opposite directions at some vertices. Polygons
135      * whose vertices come from some physical measurement with noise are also
136      * difficult because an edge that should be straight may be broken in lots of
137      * different pieces with almost equal directions. In both cases, computing the
138      * lines intersections is not numerically robust due to the almost 0 or almost
139      * &pi; angle. Such cases need to carefully adjust the {@code hyperplaneThickness}
140      * parameter. A too small value would often lead to completely wrong polygons
141      * with large area wrongly identified as inside or outside. Large values are
142      * often much safer. As a rule of thumb, a value slightly below the size of the
143      * most accurate detail needed is a good value for the {@code hyperplaneThickness}
144      * parameter.
145      * </p>
146      * @param hyperplaneThickness tolerance below which points are considered to
147      * belong to the hyperplane (which is therefore more a slab)
148      * @param vertices vertices of the simple loop boundary
149      */
150     public PolygonsSet(final double hyperplaneThickness, final Vector2D ... vertices) {
151         super(verticesToTree(hyperplaneThickness, vertices), hyperplaneThickness);
152     }

154     /** Create a list of hyperplanes representing the boundary of a box.
155      * @param xMin low bound along the x direction
156      * @param xMax high bound along the x direction
157      * @param yMin low bound along the y direction
158      * @param yMax high bound along the y direction
159      * @param tolerance tolerance below which points are considered identical
160      * @return boundary of the box
161      */
162     private static Line[] boxBoundary(final double xMin, final double xMax,
163                                       final double yMin, final double yMax,
164                                       final double tolerance) {
165         if ((xMin >= xMax - tolerance) || (yMin >= yMax - tolerance)) {
166             // too thin box, build an empty polygons set
167             return null;
168         }
169         final Vector2D minMin = new Vector2D(xMin, yMin);
170         final Vector2D minMax = new Vector2D(xMin, yMax);
171         final Vector2D maxMin = new Vector2D(xMax, yMin);
172         final Vector2D maxMax = new Vector2D(xMax, yMax);
173         return new Line[] {
174             new Line(minMin, maxMin, tolerance),
175             new Line(maxMin, maxMax, tolerance),
176             new Line(maxMax, minMax, tolerance),
177             new Line(minMax, minMin, tolerance)
178         };
179     }

181     /** Build the BSP tree of a polygons set from a simple list of vertices.
182      * <p>The boundary is provided as a list of points considering to
183      * represent the vertices of a simple loop. The interior part of the
184      * region is on the left side of this path and the exterior is on its
185      * right side.</p>
186      * <p>This constructor does not handle polygons with a boundary
187      * forming several disconnected paths (such as polygons with holes).</p>
188      * <p>For cases where this simple constructor applies, it is expected to
189      * be numerically more robust than the {@link #PolygonsSet(Collection,double) general
190      * constructor} using {@link SubHyperplane subhyperplanes}.</p>
191      * @param hyperplaneThickness tolerance below which points are consider to
192      * belong to the hyperplane (which is therefore more a slab)
193      * @param vertices vertices of the simple loop boundary
194      * @return the BSP tree of the input vertices
195      */
196     private static BSPTree<Euclidean2D> verticesToTree(final double hyperplaneThickness,
197                                                        final Vector2D ... vertices) {
198 
199         final int n = vertices.length;
200         if (n == 0) {
201             // the tree represents the whole space
202             return new BSPTree<Euclidean2D>(Boolean.TRUE);
203         }
204 
205         // build the vertices
206         final Vertex[] vArray = new Vertex[n];
207         for (int i = 0; i < n; ++i) {
208             vArray[i] = new Vertex(vertices[i]);
209         }
210 
211         // build the edges
212         List<Edge> edges = new ArrayList<Edge>(n);
213         for (int i = 0; i < n; ++i) {
214 
215             // get the endpoints of the edge
216             final Vertex start = vArray[i];
217             final Vertex end   = vArray[(i + 1) % n];
218 
219             // get the line supporting the edge, taking care not to recreate it
220             // if it was already created earlier due to another edge being aligned
221             // with the current one
222             Line line = start.sharedLineWith(end);
223             if (line == null) {
224                 line = new Line(start.getLocation(), end.getLocation(), hyperplaneThickness);
225             }
226 
227             // create the edge and store it
228             edges.add(new Edge(start, end, line));
229 
230             // check if another vertex also happens to be on this line
231             for (final Vertex vertex : vArray) {
232                 if (vertex != start && vertex != end &&
233                     FastMath.abs(line.getOffset((Point<Euclidean2D>) vertex.getLocation())) <= hyperplaneThickness) {
234                     vertex.bindWith(line);
235                 }
236             }
237 
238         }
239 
240         // build the tree top-down
241         final BSPTree<Euclidean2D> tree = new BSPTree<Euclidean2D>();
242         insertEdges(hyperplaneThickness, tree, edges);
243 
244         return tree;
245 
246     }

248     /** Recursively build a tree by inserting cut sub-hyperplanes.
249      * @param hyperplaneThickness tolerance below which points are consider to
250      * belong to the hyperplane (which is therefore more a slab)
251      * @param node current tree node (it is a leaf node at the beginning
252      * of the call)
253      * @param edges list of edges to insert in the cell defined by this node
254      * (excluding edges not belonging to the cell defined by this node)
255      */
256     private static void insertEdges(final double hyperplaneThickness,
257                                     final BSPTree<Euclidean2D> node,
258                                     final List<Edge> edges) {
259 
260         // find an edge with an hyperplane that can be inserted in the node
261         int index = 0;
262         Edge inserted =null;
263         while (inserted == null && index < edges.size()) {
264             inserted = edges.get(index++);
265             if (inserted.getNode() == null) {
266                 if (node.insertCut(inserted.getLine())) {
267                     inserted.setNode(node);
268                 } else {
269                     inserted = null;
270                 }
271             } else {
272                 inserted = null;
273             }
274         }
275 
276         if (inserted == null) {
277             // no suitable edge was found, the node remains a leaf node
278             // we need to set its inside/outside boolean indicator
279             final BSPTree<Euclidean2D> parent = node.getParent();
280             if (parent == null || node == parent.getMinus()) {
281                 node.setAttribute(Boolean.TRUE);
282             } else {
283                 node.setAttribute(Boolean.FALSE);
284             }
285             return;
286         }
287 
288         // we have split the node by inserting an edge as a cut sub-hyperplane
289         // distribute the remaining edges in the two sub-trees
290         final List<Edge> plusList  = new ArrayList<Edge>();
291         final List<Edge> minusList = new ArrayList<Edge>();
292         for (final Edge edge : edges) {
293             if (edge != inserted) {
294                 final double startOffset = inserted.getLine().getOffset((Point<Euclidean2D>) edge.getStart().getLocation());
295                 final double endOffset   = inserted.getLine().getOffset((Point<Euclidean2D>) edge.getEnd().getLocation());
296                 Side startSide = (FastMath.abs(startOffset) <= hyperplaneThickness) ?
297                                  Side.HYPER : ((startOffset < 0) ? Side.MINUS : Side.PLUS);
298                 Side endSide   = (FastMath.abs(endOffset) <= hyperplaneThickness) ?
299                                  Side.HYPER : ((endOffset < 0) ? Side.MINUS : Side.PLUS);
300                 switch (startSide) {
301                     case PLUS:
302                         if (endSide == Side.MINUS) {
303                             // we need to insert a split point on the hyperplane
304                             final Vertex splitPoint = edge.split(inserted.getLine());
305                             minusList.add(splitPoint.getOutgoing());
306                             plusList.add(splitPoint.getIncoming());
307                         } else {
308                             plusList.add(edge);
309                         }
310                         break;
311                     case MINUS:
312                         if (endSide == Side.PLUS) {
313                             // we need to insert a split point on the hyperplane
314                             final Vertex splitPoint = edge.split(inserted.getLine());
315                             minusList.add(splitPoint.getIncoming());
316                             plusList.add(splitPoint.getOutgoing());
317                         } else {
318                             minusList.add(edge);
319                         }
320                         break;
321                     default:
322                         if (endSide == Side.PLUS) {
323                             plusList.add(edge);
324                         } else if (endSide == Side.MINUS) {
325                             minusList.add(edge);
326                         }
327                         break;
328                 }
329             }
330         }
331 
332         // recurse through lower levels
333         if (!plusList.isEmpty()) {
334             insertEdges(hyperplaneThickness, node.getPlus(),  plusList);
335         } else {
336             node.getPlus().setAttribute(Boolean.FALSE);
337         }
338         if (!minusList.isEmpty()) {
339             insertEdges(hyperplaneThickness, node.getMinus(), minusList);
340         } else {
341             node.getMinus().setAttribute(Boolean.TRUE);
342         }
343 
344     }

389          * to create a cut hyperplane for them.

396             for (final Line line1 : lines) {
397                 for (final Line line2 : vertex.lines) {
398                     if (line1 == line2) {

400                     }
401                 }
402             }
403             return null;
404         }

446     /** Internal class for holding edges while they are processed to build a BSP tree. */
447     private static class Edge {
448 
449         /** Start vertex. */
450         private final Vertex start;
451 
452         /** End vertex. */
453         private final Vertex end;
454 
455         /** Line supporting the edge. */
456         private final Line line;
457 
458         /** Node whose cut hyperplane contains this edge. */
459         private BSPTree<Euclidean2D> node;
460 
461         /** Build an edge not contained in any node yet.
462          * @param start start vertex
463          * @param end end vertex
464          * @param line line supporting the edge
465          */
466         Edge(final Vertex start, final Vertex end, final Line line) {
467 
468             this.start = start;
469             this.end   = end;
470             this.line  = line;
471             this.node  = null;
472 
473             // connect the vertices back to the edge
474             start.setOutgoing(this);
475             end.setIncoming(this);
476 
477         }
478 
479         /** Get start vertex.
480          * @return start vertex
481          */
482         public Vertex getStart() {
483             return start;
484         }
485 
486         /** Get end vertex.
487          * @return end vertex
488          */
489         public Vertex getEnd() {
490             return end;
491         }
492 
493         /** Get the line supporting this edge.
494          * @return line supporting this edge
495          */
496         public Line getLine() {
497             return line;
498         }
499 
500         /** Set the node whose cut hyperplane contains this edge.
501          * @param node node whose cut hyperplane contains this edge
502          */
503         public void setNode(final BSPTree<Euclidean2D> node) {
504             this.node = node;
505         }
506 
507         /** Get the node whose cut hyperplane contains this edge.
508          * @return node whose cut hyperplane contains this edge
509          * (null if edge has not yet been inserted into the BSP tree)
510          */
511         public BSPTree<Euclidean2D> getNode() {
512             return node;
513         }
514 
515         /** Split the edge.
516          * <p>
517          * Once split, this edge is not referenced anymore by the vertices,
518          * it is replaced by the two half-edges and an intermediate splitting
519          * vertex is introduced to connect these two halves.
520          * </p>
521          * @param splitLine line splitting the edge in two halves
522          * @return split vertex (its incoming and outgoing edges are the two halves)
523          */
524         public Vertex split(final Line splitLine) {
525             final Vertex splitVertex = new Vertex(line.intersection(splitLine));
526             splitVertex.bindWith(splitLine);
527             final Edge startHalf = new Edge(start, splitVertex, line);
528             final Edge endHalf   = new Edge(splitVertex, end, line);
529             startHalf.node = node;
530             endHalf.node   = node;
531             return splitVertex;
532         }
533 
534     }

542     /** {@inheritDoc} */
543     @Override
544     protected void computeGeometricalProperties() {
545 
546         final Vector2D[][] v = getVertices();
547 
548         if (v.length == 0) {
549             final BSPTree<Euclidean2D> tree = getTree(false);
550             if (tree.getCut() == null && (Boolean) tree.getAttribute()) {
551                 // the instance covers the whole space
552                 setSize(Double.POSITIVE_INFINITY);
553                 setBarycenter((Point<Euclidean2D>) Vector2D.NaN);
554             } else {
555                 setSize(0);
556                 setBarycenter((Point<Euclidean2D>) new Vector2D(0, 0));
557             }
558         } else if (v[0][0] == null) {
559             // there is at least one open-loop: the polygon is infinite
560             setSize(Double.POSITIVE_INFINITY);
561             setBarycenter((Point<Euclidean2D>) Vector2D.NaN);
562         } else {
563             // all loops are closed, we compute some integrals around the shape
564 
565             double sum  = 0;
566             double sumX = 0;
567             double sumY = 0;
568 
569             for (Vector2D[] loop : v) {
570                 double x1 = loop[loop.length - 1].getX();
571                 double y1 = loop[loop.length - 1].getY();
572                 for (final Vector2D point : loop) {
573                     final double x0 = x1;
574                     final double y0 = y1;
575                     x1 = point.getX();
576                     y1 = point.getY();
577                     final double factor = x0 * y1 - y0 * x1;
578                     sum  += factor;
579                     sumX += factor * (x0 + x1);
580                     sumY += factor * (y0 + y1);
581                 }
582             }
583 
584             if (sum < 0) {
585                 // the polygon as a finite outside surrounded by an infinite inside
586                 setSize(Double.POSITIVE_INFINITY);
587                 setBarycenter((Point<Euclidean2D>) Vector2D.NaN);
588             } else {
589                 setSize(sum / 2);
590                 setBarycenter((Point<Euclidean2D>) new Vector2D(sumX / (3 * sum), sumY / (3 * sum)));
591             }
592 
593         }
594 
595     }

597     /** Get the vertices of the polygon.
598      * <p>The polygon boundary can be represented as an array of loops,
599      * each loop being itself an array of vertices.</p>
600      * <p>In order to identify open loops which start and end by
601      * infinite edges, the open loops arrays start with a null point. In
602      * this case, the first non null point and the last point of the
603      * array do not represent real vertices, they are dummy points
604      * intended only to get the direction of the first and last edge. An
605      * open loop consisting of a single infinite line will therefore be
606      * represented by a three elements array with one null point
607      * followed by two dummy points. The open loops are always the first
608      * ones in the loops array.</p>
609      * <p>If the polygon has no boundary at all, a zero length loop
610      * array will be returned.</p>
611      * <p>All line segments in the various loops have the inside of the
612      * region on their left side and the outside on their right side
613      * when moving in the underlying line direction. This means that
614      * closed loops surrounding finite areas obey the direct
615      * trigonometric orientation.</p>
616      * @return vertices of the polygon, organized as oriented boundary
617      * loops with the open loops first (the returned value is guaranteed
618      * to be non-null)
619      */
620     public Vector2D[][] getVertices() {
621         if (vertices == null) {
622             if (getTree(false).getCut() == null) {
623                 vertices = new Vector2D[0][];
624             } else {
625 
626                 // build the unconnected segments
627                 final SegmentsBuilder visitor = new SegmentsBuilder(getTolerance());
628                 getTree(true).visit(visitor);
629                 final List<ConnectableSegment> segments = visitor.getSegments();
630 
631                 // connect all segments, using topological criteria first
632                 // and using Euclidean distance only as a last resort
633                 int pending = segments.size();
634                 pending -= naturalFollowerConnections(segments);
635                 if (pending > 0) {
636                     pending -= splitEdgeConnections(segments);
637                 }
638                 if (pending > 0) {
639                     pending -= closeVerticesConnections(segments);
640                 }
641 
642                 // create the segment loops
643                 final ArrayList<List<Segment>> loops = new ArrayList<List<Segment>>();
644                 for (ConnectableSegment s = getUnprocessed(segments); s != null; s = getUnprocessed(segments)) {
645                     final List<Segment> loop = followLoop(s);
646                     if (loop != null) {
647                         if (loop.get(0).getStart() == null) {
648                             // this is an open loop, we put it on the front
649                             loops.add(0, loop);
650                         } else {
651                             // this is a closed loop, we put it on the back
652                             loops.add(loop);
653                         }
654                     }
655                 }
656 
657                 // transform the loops in an array of arrays of points
658                 vertices = new Vector2D[loops.size()][];
659                 int i = 0;
660 
661                 for (final List<Segment> loop : loops) {
662                     if (loop.size() < 2 ||
663                         (loop.size() == 2 && loop.get(0).getStart() == null && loop.get(1).getEnd() == null)) {
664                         // single infinite line
665                         final Line line = loop.get(0).getLine();
666                         vertices[i++] = new Vector2D[] {
667                             null,
668                             line.toSpace((Point<Euclidean1D>) new Vector1D(-Float.MAX_VALUE)),
669                             line.toSpace((Point<Euclidean1D>) new Vector1D(+Float.MAX_VALUE))
670                         };
671                     } else if (loop.get(0).getStart() == null) {
672                         // open loop with at least one real point
673                         final Vector2D[] array = new Vector2D[loop.size() + 2];
674                         int j = 0;
675                         for (Segment segment : loop) {
676 
677                             if (j == 0) {
678                                 // null point and first dummy point
679                                 double x = segment.getLine().toSubSpace((Point<Euclidean2D>) segment.getEnd()).getX();
680                                 x -= FastMath.max(1.0, FastMath.abs(x / 2));
681                                 array[j++] = null;
682                                 array[j++] = segment.getLine().toSpace((Point<Euclidean1D>) new Vector1D(x));
683                             }
684 
685                             if (j < (array.length - 1)) {
686                                 // current point
687                                 array[j++] = segment.getEnd();
688                             }
689 
690                             if (j == (array.length - 1)) {
691                                 // last dummy point
692                                 double x = segment.getLine().toSubSpace((Point<Euclidean2D>) segment.getStart()).getX();
693                                 x += FastMath.max(1.0, FastMath.abs(x / 2));
694                                 array[j++] = segment.getLine().toSpace((Point<Euclidean1D>) new Vector1D(x));
695                             }
696 
697                         }
698                         vertices[i++] = array;
699                     } else {
700                         final Vector2D[] array = new Vector2D[loop.size()];
701                         int j = 0;
702                         for (Segment segment : loop) {
703                             array[j++] = segment.getStart();
704                         }
705                         vertices[i++] = array;
706                     }
707                 }
708 
709             }
710         }
711 
712         return vertices.clone();
713 
714     }

716     /** Connect the segments using only natural follower information.
717      * @param segments segments complete segments list
718      * @return number of connections performed
719      */
720     private int naturalFollowerConnections(final List<ConnectableSegment> segments) {
721         int connected = 0;
722         for (final ConnectableSegment segment : segments) {
723             if (segment.getNext() == null) {
724                 final BSPTree<Euclidean2D> node = segment.getNode();
725                 final BSPTree<Euclidean2D> end  = segment.getEndNode();
726                 for (final ConnectableSegment candidateNext : segments) {
727                     if (candidateNext.getPrevious()  == null &&
728                         candidateNext.getNode()      == end &&
729                         candidateNext.getStartNode() == node) {
730                         // connect the two segments
731                         segment.setNext(candidateNext);
732                         candidateNext.setPrevious(segment);
733                         ++connected;
734                         break;
735                     }
736                 }
737             }
738         }
739         return connected;
740     }

742     /** Connect the segments resulting from a line splitting a straight edge.
743      * @param segments segments complete segments list
744      * @return number of connections performed
745      */
746     private int splitEdgeConnections(final List<ConnectableSegment> segments) {
747         int connected = 0;
748         for (final ConnectableSegment segment : segments) {
749             if (segment.getNext() == null) {
750                 final Hyperplane<Euclidean2D> hyperplane = segment.getNode().getCut().getHyperplane();
751                 final BSPTree<Euclidean2D> end  = segment.getEndNode();
752                 for (final ConnectableSegment candidateNext : segments) {
753                     if (candidateNext.getPrevious()                      == null &&
754                         candidateNext.getNode().getCut().getHyperplane() == hyperplane &&
755                         candidateNext.getStartNode()                     == end) {
756                         // connect the two segments
757                         segment.setNext(candidateNext);
758                         candidateNext.setPrevious(segment);
759                         ++connected;
760                         break;
761                     }
762                 }
763             }
764         }
765         return connected;
766     }

768     /** Connect the segments using Euclidean distance.
769      * <p>
770      * This connection heuristic should be used last, as it relies
771      * only on a fuzzy distance criterion.
772      * </p>
773      * @param segments segments complete segments list
774      * @return number of connections performed
775      */
776     private int closeVerticesConnections(final List<ConnectableSegment> segments) {
777         int connected = 0;
778         for (final ConnectableSegment segment : segments) {
779             if (segment.getNext() == null && segment.getEnd() != null) {
780                 final Vector2D end = segment.getEnd();
781                 ConnectableSegment selectedNext = null;
782                 double min = Double.POSITIVE_INFINITY;
783                 for (final ConnectableSegment candidateNext : segments) {
784                     if (candidateNext.getPrevious() == null && candidateNext.getStart() != null) {
785                         final double distance = Vector2D.distance(end, candidateNext.getStart());
786                         if (distance < min) {
787                             selectedNext = candidateNext;
788                             min          = distance;
789                         }
790                     }
791                 }
792                 if (min <= getTolerance()) {
793                     // connect the two segments
794                     segment.setNext(selectedNext);
795                     selectedNext.setPrevious(segment);
796                     ++connected;
797                 }
798             }
799         }
800         return connected;
801     }

804      * @param segments segments list
805      * @return first segment that has not been processed yet
806      * or null if all segments have been processed
807      */
808     private ConnectableSegment getUnprocessed(final List<ConnectableSegment> segments) {
809         for (final ConnectableSegment segment : segments) {
810             if (!segment.isProcessed()) {
811                 return segment;
812             }
813         }
814         return null;

817     /** Build the loop containing a segment.
818      * <p>
819      * The segment put in the loop will be marked as processed.
820      * </p>
821      * @param defining segment used to define the loop
822      * @return loop containing the segment (may be null if the loop is a
823      * degenerated infinitely thin 2 points loop
824      */
825     private List<Segment> followLoop(final ConnectableSegment defining) {
826 
827         final List<Segment> loop = new ArrayList<Segment>();
828         loop.add(defining);
829         defining.setProcessed(true);
830 
831         // add segments in connection order
832         ConnectableSegment next = defining.getNext();
833         while (next != defining && next != null) {
834             loop.add(next);
835             next.setProcessed(true);
836             next = next.getNext();
837         }
838 
839         if (next == null) {
840             // the loop is open and we have found its end,
841             // we need to find its start too
842             ConnectableSegment previous = defining.getPrevious();
843             while (previous != null) {
844                 loop.add(0, previous);
845                 previous.setProcessed(true);
846                 previous = previous.getPrevious();
847             }
848         }
849 
850         // filter out spurious vertices
851         filterSpuriousVertices(loop);
852 
853         if (loop.size() == 2 && loop.get(0).getStart() != null) {
854             // this is a degenerated infinitely thin closed loop, we simply ignore it
855             return null;
856         } else {
857             return loop;
858         }
859 
860     }
861 
862     /** Filter out spurious vertices on straight lines (at machine precision).
863      * @param loop segments loop to filter (will be modified in-place)
864      */
865     private void filterSpuriousVertices(final List<Segment> loop) {
866         for (int i = 0; i < loop.size(); ++i) {
867             final Segment previous = loop.get(i);
868             int j = (i + 1) % loop.size();
869             final Segment next = loop.get(j);
870             if (next != null &&
871                 Precision.equals(previous.getLine().getAngle(), next.getLine().getAngle(), Precision.EPSILON)) {
872                 // the vertex between the two edges is a spurious one
873                 // replace the two segments by a single one
874                 loop.set(j, new Segment(previous.getStart(), next.getEnd(), previous.getLine()));
875                 loop.remove(i--);
876             }
877         }
878     }

987     /** Visitor building segments. */
988     private static class SegmentsBuilder implements BSPTreeVisitor<Euclidean2D> {
989 
990         /** Tolerance for close nodes connection. */
991         private final double tolerance;
992 
993         /** Built segments. */
994         private final List<ConnectableSegment> segments;
995 
996         /** Simple constructor.
997          * @param tolerance tolerance for close nodes connection
998          */
999         SegmentsBuilder(final double tolerance) {
1000             this.tolerance = tolerance;
1001             this.segments  = new ArrayList<ConnectableSegment>();
1002         }
1003 
1004         /** {@inheritDoc} */
1005         @Override
1006         public Order visitOrder(final BSPTree<Euclidean2D> node) {
1007             return Order.MINUS_SUB_PLUS;
1008         }
1009 
1010         /** {@inheritDoc} */
1011         @Override
1012         public void visitInternalNode(final BSPTree<Euclidean2D> node) {
1013             @SuppressWarnings("unchecked")
1014             final BoundaryAttribute<Euclidean2D> attribute = (BoundaryAttribute<Euclidean2D>) node.getAttribute();
1015             final Iterable<BSPTree<Euclidean2D>> splitters = attribute.getSplitters();
1016             if (attribute.getPlusOutside() != null) {
1017                 addContribution(attribute.getPlusOutside(), node, splitters, false);
1018             }
1019             if (attribute.getPlusInside() != null) {
1020                 addContribution(attribute.getPlusInside(), node, splitters, true);
1021             }
1022         }
1023 
1024         /** {@inheritDoc} */
1025         @Override
1026         public void visitLeafNode(final BSPTree<Euclidean2D> node) {
1027         }
1028 
1029         /** Add the contribution of a boundary facet.
1030          * @param sub boundary facet
1031          * @param node node containing segment
1032          * @param splitters splitters for the boundary facet
1033          * @param reversed if true, the facet has the inside on its plus side
1034          */
1035         private void addContribution(final SubHyperplane<Euclidean2D> sub,
1036                                      final BSPTree<Euclidean2D> node,
1037                                      final Iterable<BSPTree<Euclidean2D>> splitters,
1038                                      final boolean reversed) {
1039             @SuppressWarnings("unchecked")
1040             final AbstractSubHyperplane<Euclidean2D, Euclidean1D> absSub =
1041                 (AbstractSubHyperplane<Euclidean2D, Euclidean1D>) sub;
1042             final Line line      = (Line) sub.getHyperplane();
1043             final List<Interval> intervals = ((IntervalsSet) absSub.getRemainingRegion()).asList();
1044             for (final Interval i : intervals) {
1045 
1046                 // find the 2D points
1047                 final Vector2D startV = Double.isInfinite(i.getInf()) ?
1048                                         null : (Vector2D) line.toSpace((Point<Euclidean1D>) new Vector1D(i.getInf()));
1049                 final Vector2D endV   = Double.isInfinite(i.getSup()) ?
1050                                         null : (Vector2D) line.toSpace((Point<Euclidean1D>) new Vector1D(i.getSup()));
1051 
1052                 // recover the connectivity information
1053                 final BSPTree<Euclidean2D> startN = selectClosest(startV, splitters);
1054                 final BSPTree<Euclidean2D> endN   = selectClosest(endV, splitters);
1055 
1056                 if (reversed) {
1057                     segments.add(new ConnectableSegment(endV, startV, line.getReverse(),
1058                                                         node, endN, startN));
1059                 } else {
1060                     segments.add(new ConnectableSegment(startV, endV, line,
1061                                                         node, startN, endN));
1062                 }
1063 
1064             }
1065         }
1066 
1067         /** Select the node whose cut sub-hyperplane is closest to specified point.
1068          * @param point reference point
1069          * @param candidates candidate nodes
1070          * @return node closest to point, or null if no node is closer than tolerance
1071          */
1072         private BSPTree<Euclidean2D> selectClosest(final Vector2D point, final Iterable<BSPTree<Euclidean2D>> candidates) {
1073 
1074             BSPTree<Euclidean2D> selected = null;
1075             double min = Double.POSITIVE_INFINITY;
1076 
1077             for (final BSPTree<Euclidean2D> node : candidates) {
1078                 final double distance = FastMath.abs(node.getCut().getHyperplane().getOffset(point));
1079                 if (distance < min) {
1080                     selected = node;
1081                     min      = distance;
1082                 }
1083             }
1084 
1085             return min <= tolerance ? selected : null;
1086 
1087         }
1088 
1089         /** Get the segments.
1090          * @return built segments
1091          */
1092         public List<ConnectableSegment> getSegments() {
1093             return segments;
1094         }
