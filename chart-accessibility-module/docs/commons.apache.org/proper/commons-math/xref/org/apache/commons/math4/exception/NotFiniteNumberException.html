17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when a number is not finite.
24   *
25   * @since 3.0
26   */
27  public class NotFiniteNumberException extends MathIllegalNumberException {
28      /** Serializable version Id. */
29      private static final long serialVersionUID = -6100997100383932834L;
30  
31      /**
32       * Construct the exception.
33       *
34       * @param wrong Value that is infinite or NaN.
35       * @param args Optional arguments.
36       */
37      public NotFiniteNumberException(Number wrong,
38                                      Object ... args) {
39          this(LocalizedFormats.NOT_FINITE_NUMBER, wrong, args);
40      }
41  
42      /**
43       * Construct the exception with a specific context.
44       *
45       * @param specific Specific context pattern.
46       * @param wrong Value that is infinite or NaN.
47       * @param args Optional arguments.
48       */
49      public NotFiniteNumberException(Localizable specific,
50                                      Number wrong,
51                                      Object ... args) {
52          super(specific, wrong, args);
53      }
54  }
