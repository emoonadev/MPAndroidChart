18  package org.apache.commons.math4.ml.neuralnet.sofm;
19  
20  import org.apache.commons.math4.ml.neuralnet.sofm.util.ExponentialDecayFunction;
21  import org.apache.commons.math4.ml.neuralnet.sofm.util.QuasiSigmoidDecayFunction;
22  import org.apache.commons.math4.util.FastMath;
23  
24  /**
25   * Factory for creating instances of {@link NeighbourhoodSizeFunction}.
26   *
27   * @since 3.3
28   */
29  public class NeighbourhoodSizeFunctionFactory {
30      /** Class contains only static methods. */
31      private NeighbourhoodSizeFunctionFactory() {}
32  
33      /**
34       * Creates an exponential decay {@link NeighbourhoodSizeFunction function}.
35       * It will compute <code>a e<sup>-x / b</sup></code>,
36       * where {@code x} is the (integer) independent variable and
37       * <ul>
38       *  <li><code>a = initValue</code>
39       *  <li><code>b = -numCall / ln(valueAtNumCall / initValue)</code>
40       * </ul>
41       *
42       * @param initValue Initial value, i.e.
43       * {@link NeighbourhoodSizeFunction#value(long) value(0)}.
44       * @param valueAtNumCall Value of the function at {@code numCall}.
45       * @param numCall Argument for which the function returns
46       * {@code valueAtNumCall}.
47       * @return the neighbourhood size function.
48       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
49       * if {@code initValue <= 0}.
50       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
51       * if {@code valueAtNumCall <= 0}.
52       * @throws org.apache.commons.math4.exception.NumberIsTooLargeException
53       * if {@code valueAtNumCall >= initValue}.
54       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
55       * if {@code numCall <= 0}.
56       */
57      public static NeighbourhoodSizeFunction exponentialDecay(final double initValue,
58                                                               final double valueAtNumCall,
59                                                               final long numCall) {
60          return new NeighbourhoodSizeFunction() {
61              /** DecayFunction. */
62              private final ExponentialDecayFunction decay
63                  = new ExponentialDecayFunction(initValue, valueAtNumCall, numCall);
64  
65              /** {@inheritDoc} */
66              @Override
67              public int value(long n) {
68                  return (int) FastMath.rint(decay.value(n));
69              }
70          };
71      }
72  
73      /**
74       * Creates an sigmoid-like {@code NeighbourhoodSizeFunction function}.
75       * The function {@code f} will have the following properties:
76       * <ul>
77       *  <li>{@code f(0) = initValue}</li>
78       *  <li>{@code numCall} is the inflexion point</li>
79       *  <li>{@code slope = f'(numCall)}</li>
80       * </ul>
81       *
82       * @param initValue Initial value, i.e.
83       * {@link NeighbourhoodSizeFunction#value(long) value(0)}.
84       * @param slope Value of the function derivative at {@code numCall}.
85       * @param numCall Inflexion point.
86       * @return the neighbourhood size function.
87       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
88       * if {@code initValue <= 0}.
89       * @throws org.apache.commons.math4.exception.NumberIsTooLargeException
90       * if {@code slope >= 0}.
91       * @throws org.apache.commons.math4.exception.NotStrictlyPositiveException
92       * if {@code numCall <= 0}.
93       */
94      public static NeighbourhoodSizeFunction quasiSigmoidDecay(final double initValue,
95                                                                final double slope,
96                                                                final long numCall) {
97          return new NeighbourhoodSizeFunction() {
98              /** DecayFunction. */
99              private final QuasiSigmoidDecayFunction decay
100                 = new QuasiSigmoidDecayFunction(initValue, slope, numCall);
101 
102             /** {@inheritDoc} */
103             @Override
104             public int value(long n) {
105                 return (int) FastMath.rint(decay.value(n));
106             }
107         };
108     }
109 }
