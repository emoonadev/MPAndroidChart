47      /** Build a convex region from a collection of bounding hyperplanes.
48       * @param hyperplanes collection of bounding hyperplanes
49       * @return a new convex region, or null if the collection is empty
50       */
51      @SafeVarargs
52      public final Region<S> buildConvex(final Hyperplane<S> ... hyperplanes) {
53          if ((hyperplanes == null) || (hyperplanes.length == 0)) {
54              return null;
55          }
56  
57          // use the first hyperplane to build the right class
58          final Region<S> region = hyperplanes[0].wholeSpace();
59  
60          // chop off parts of the space
61          BSPTree<S> node = region.getTree(false);
62          node.setAttribute(Boolean.TRUE);
63          for (final Hyperplane<S> hyperplane : hyperplanes) {
64              if (node.insertCut(hyperplane)) {
65                  node.setAttribute(null);
66                  node.getPlus().setAttribute(Boolean.FALSE);
67                  node = node.getMinus();
68                  node.setAttribute(Boolean.TRUE);
69              } else {
70                  // the hyperplane could not be inserted in the current leaf node
71                  // either it is completely outside (which means the input hyperplanes
72                  // are wrong), or it is parallel to a previous hyperplane
73                  SubHyperplane<S> s = hyperplane.wholeHyperplane();
74                  for (BSPTree<S> tree = node; tree.getParent() != null && s != null; tree = tree.getParent()) {
75                      final Hyperplane<S>         other = tree.getParent().getCut().getHyperplane();
76                      final SplitSubHyperplane<S> split = s.split(other);
77                      switch (split.getSide()) {
78                          case HYPER :
79                              // the hyperplane is parallel to a previous hyperplane
80                              if (!hyperplane.sameOrientationAs(other)) {
81                                  // this hyperplane is opposite to the other one,
82                                  // the region is thinner than the tolerance, we consider it empty
83                                  return getComplement(hyperplanes[0].wholeSpace());
84                              }
85                              // the hyperplane is an extension of an already known hyperplane, we just ignore it
86                              break;
87                          case PLUS :
88                          // the hyperplane is outside of the current convex zone,
89                          // the input hyperplanes are inconsistent
90                          throw new MathIllegalArgumentException(LocalizedFormats.NOT_CONVEX_HYPERPLANES);
91                          default :
92                              s = split.getMinus();
93                      }
94                  }
95              }
96          }
97  
98          return region;
99  
100     }

102     /** Compute the union of two regions.
103      * @param region1 first region (will be unusable after the operation as
104      * parts of it will be reused in the new region)
105      * @param region2 second region (will be unusable after the operation as
106      * parts of it will be reused in the new region)
107      * @return a new region, result of {@code region1 union region2}
108      */
109     public Region<S> union(final Region<S> region1, final Region<S> region2) {
110         final BSPTree<S> tree =
111             region1.getTree(false).merge(region2.getTree(false), new UnionMerger());
112         tree.visit(nodeCleaner);
113         return region1.buildNew(tree);
114     }

144     /** Compute the difference of two regions.
145      * @param region1 first region (will be unusable after the operation as
146      * parts of it will be reused in the new region)
147      * @param region2 second region (will be unusable after the operation as
148      * parts of it will be reused in the new region)
149      * @return a new region, result of {@code region1 minus region2}

151     public Region<S> difference(final Region<S> region1, final Region<S> region2) {
152         final BSPTree<S> tree =
153             region1.getTree(false).merge(region2.getTree(false), new DifferenceMerger(region1, region2));
154         tree.visit(nodeCleaner);
155         return region1.buildNew(tree);

158     /** Get the complement of the region (exchanged interior/exterior).
159      * @param region region to complement, it will not modified, a new
160      * region independent region will be built
161      * @return a new region, complement of the specified one
162      */
163     /** Get the complement of the region (exchanged interior/exterior).
164      * @param region region to complement, it will not modified, a new
165      * region independent region will be built
166      * @return a new region, complement of the specified one
167      */
168     public Region<S> getComplement(final Region<S> region) {
169         return region.buildNew(recurseComplement(region.getTree(false)));
170     }

172     /** Recursively build the complement of a BSP tree.
173      * @param node current node of the original tree
174      * @return new tree, complement of the node
175      */
176     private BSPTree<S> recurseComplement(final BSPTree<S> node) {
177 
178         // transform the tree, except for boundary attribute splitters
179         final Map<BSPTree<S>, BSPTree<S>> map = new HashMap<BSPTree<S>, BSPTree<S>>();
180         final BSPTree<S> transformedTree = recurseComplement(node, map);
181 
182         // set up the boundary attributes splitters
183         for (final Map.Entry<BSPTree<S>, BSPTree<S>> entry : map.entrySet()) {
184             if (entry.getKey().getCut() != null) {
185                 @SuppressWarnings("unchecked")
186                 BoundaryAttribute<S> original = (BoundaryAttribute<S>) entry.getKey().getAttribute();
187                 if (original != null) {
188                     @SuppressWarnings("unchecked")
189                     BoundaryAttribute<S> transformed = (BoundaryAttribute<S>) entry.getValue().getAttribute();
190                     for (final BSPTree<S> splitter : original.getSplitters()) {
191                         transformed.getSplitters().add(map.get(splitter));
192                     }
193                 }
194             }
195         }
196 
197         return transformedTree;
198 
199     }

201     /** Recursively build the complement of a BSP tree.
202      * @param node current node of the original tree
203      * @param map transformed nodes map
204      * @return new tree, complement of the node
205      */
206     private BSPTree<S> recurseComplement(final BSPTree<S> node,
207                                          final Map<BSPTree<S>, BSPTree<S>> map) {
208 
209         final BSPTree<S> transformedNode;
210         if (node.getCut() == null) {
211             transformedNode = new BSPTree<S>(((Boolean) node.getAttribute()) ? Boolean.FALSE : Boolean.TRUE);
212         } else {
213 
214             @SuppressWarnings("unchecked")
215             BoundaryAttribute<S> attribute = (BoundaryAttribute<S>) node.getAttribute();
216             if (attribute != null) {
217                 final SubHyperplane<S> plusOutside =
218                         (attribute.getPlusInside() == null) ? null : attribute.getPlusInside().copySelf();
219                 final SubHyperplane<S> plusInside  =
220                         (attribute.getPlusOutside() == null) ? null : attribute.getPlusOutside().copySelf();
221                 // we start with an empty list of splitters, it will be filled in out of recursion
222                 attribute = new BoundaryAttribute<S>(plusOutside, plusInside, new NodesSet<S>());
223             }
224 
225             transformedNode = new BSPTree<S>(node.getCut().copySelf(),
226                                              recurseComplement(node.getPlus(),  map),
227                                              recurseComplement(node.getMinus(), map),
228                                              attribute);
229         }
230 
231         map.put(node, transformedNode);
232         return transformedNode;
233 
234     }

236     /** BSP tree leaf merger computing union of two regions. */
237     private class UnionMerger implements BSPTree.LeafMerger<S> {
238         /** {@inheritDoc} */
239         @Override
240         public BSPTree<S> merge(final BSPTree<S> leaf, final BSPTree<S> tree,
241                                 final BSPTree<S> parentTree,
242                                 final boolean isPlusChild, final boolean leafFromInstance) {
243             if ((Boolean) leaf.getAttribute()) {
244                 // the leaf node represents an inside cell
245                 leaf.insertInTree(parentTree, isPlusChild, new VanishingToLeaf(true));
246                 return leaf;
247             }
248             // the leaf node represents an outside cell
249             tree.insertInTree(parentTree, isPlusChild, new VanishingToLeaf(false));
250             return tree;
251         }
252     }

272     /** BSP tree leaf merger computing symmetric difference (exclusive or) of two regions. */
273     private class XorMerger implements BSPTree.LeafMerger<S> {
274         /** {@inheritDoc} */
275         @Override
276         public BSPTree<S> merge(final BSPTree<S> leaf, final BSPTree<S> tree,
277                                 final BSPTree<S> parentTree, final boolean isPlusChild,
278                                 final boolean leafFromInstance) {
279             BSPTree<S> t = tree;
280             if ((Boolean) leaf.getAttribute()) {
281                 // the leaf node represents an inside cell
282                 t = recurseComplement(t);
283             }
284             t.insertInTree(parentTree, isPlusChild, new VanishingToLeaf(true));
285             return t;
286         }
287     }

289     /** BSP tree leaf merger computing difference of two regions. */
290     private class DifferenceMerger implements BSPTree.LeafMerger<S>, VanishingCutHandler<S> {
291 
292         /** Region to subtract from. */
293         private final Region<S> region1;
294 
295         /** Region to subtract. */
296         private final Region<S> region2;
297 
298         /** Simple constructor.
299          * @param region1 region to subtract from
300          * @param region2 region to subtract
301          */
302         DifferenceMerger(final Region<S> region1, final Region<S> region2) {
303             this.region1 = region1.copySelf();
304             this.region2 = region2.copySelf();
305         }
306 
307         /** {@inheritDoc} */
308         @Override
309         public BSPTree<S> merge(final BSPTree<S> leaf, final BSPTree<S> tree,
310                                 final BSPTree<S> parentTree, final boolean isPlusChild,
311                                 final boolean leafFromInstance) {
312             if ((Boolean) leaf.getAttribute()) {
313                 // the leaf node represents an inside cell
314                 final BSPTree<S> argTree =
315                     recurseComplement(leafFromInstance ? tree : leaf);
316                 argTree.insertInTree(parentTree, isPlusChild, this);
317                 return argTree;
318             }
319             // the leaf node represents an outside cell
320             final BSPTree<S> instanceTree =
321                 leafFromInstance ? leaf : tree;
322             instanceTree.insertInTree(parentTree, isPlusChild, this);
323             return instanceTree;
324         }
325 
326         /** {@inheritDoc} */
327         @Override
328         public BSPTree<S> fixNode(final BSPTree<S> node) {
329             // get a representative point in the degenerate cell
330             final BSPTree<S> cell = node.pruneAroundConvexCell(Boolean.TRUE, Boolean.FALSE, null);
331             final Region<S> r = region1.buildNew(cell);
332             final Point<S> p = r.getBarycenter();
333             return new BSPTree<S>(region1.checkPoint(p) == Location.INSIDE &&
334                                   region2.checkPoint(p) == Location.OUTSIDE);
335         }
336 
337     }

339     /** Visitor removing internal nodes attributes. */
340     private class NodesCleaner implements  BSPTreeVisitor<S> {
341 
342         /** {@inheritDoc} */
343         @Override
344         public Order visitOrder(final BSPTree<S> node) {
345             return Order.PLUS_SUB_MINUS;
346         }
347 
348         /** {@inheritDoc} */
349         @Override
350         public void visitInternalNode(final BSPTree<S> node) {
351             node.setAttribute(null);
352         }
353 
354         /** {@inheritDoc} */
355         @Override
356         public void visitLeafNode(final BSPTree<S> node) {
357         }
358 
359     }

361     /** Handler replacing nodes with vanishing cuts with leaf nodes. */
362     private class VanishingToLeaf implements VanishingCutHandler<S> {
363 
364         /** Inside/outside indocator to use for ambiguous nodes. */
365         private final boolean inside;
366 
367         /** Simple constructor.
368          * @param inside inside/outside indicator to use for ambiguous nodes
369          */
370         VanishingToLeaf(final boolean inside) {
371             this.inside = inside;
372         }
373 
374         /** {@inheritDoc} */
375         @Override
376         public BSPTree<S> fixNode(final BSPTree<S> node) {
377             if (node.getPlus().getAttribute().equals(node.getMinus().getAttribute())) {
378                 // no ambiguity
379                 return new BSPTree<S>(node.getPlus().getAttribute());
380             } else {
381                 // ambiguous node
382                 return new BSPTree<S>(inside);
383             }
384         }
