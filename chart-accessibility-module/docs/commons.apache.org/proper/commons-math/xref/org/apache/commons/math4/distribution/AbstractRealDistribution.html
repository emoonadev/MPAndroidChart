19  import java.io.Serializable;

21  import org.apache.commons.math4.analysis.UnivariateFunction;
22  import org.apache.commons.math4.analysis.solvers.UnivariateSolverUtils;
23  import org.apache.commons.math4.exception.NumberIsTooLargeException;
24  import org.apache.commons.math4.exception.OutOfRangeException;
25  import org.apache.commons.math4.exception.util.LocalizedFormats;

29  /**
30   * Base class for probability distributions on the reals.
31   * Default implementations are provided for some of the methods
32   * that do not vary from distribution to distribution.
33   *
34   * <p>
35   * This base class provides a default factory method for creating
36   * a {@link RealDistribution.Sampler sampler instance} that uses the
37   * <a href="http://en.wikipedia.org/wiki/Inverse_transform_sampling">
38   * inversion method</a> for generating random samples that follow the
39   * distribution.
40   * </p>
41   *
42   * @since 3.0
43   */
44  public abstract class AbstractRealDistribution
45      implements RealDistribution, Serializable {
46      /** Default absolute accuracy for inverse cumulative computation. */
47      public static final double SOLVER_DEFAULT_ABSOLUTE_ACCURACY = 1e-6;
48      /** Serializable version identifier */
49      private static final long serialVersionUID = 20160311L;
50  
51      /**
52       * For a random variable {@code X} whose values are distributed according
53       * to this distribution, this method returns {@code P(x0 < X <= x1)}.
54       *
55       * @param x0 Lower bound (excluded).
56       * @param x1 Upper bound (included).
57       * @return the probability that a random variable with this distribution
58       * takes a value between {@code x0} and {@code x1}, excluding the lower
59       * and including the upper endpoint.
60       * @throws NumberIsTooLargeException if {@code x0 > x1}.
61       *
62       * The default implementation uses the identity
63       * {@code P(x0 < X <= x1) = P(X <= x1) - P(X <= x0)}
64       *
65       * @since 3.1
66       */
67      @Override
68      public double probability(double x0,
69                                double x1) {
70          if (x0 > x1) {
71              throw new NumberIsTooLargeException(LocalizedFormats.LOWER_ENDPOINT_ABOVE_UPPER_ENDPOINT,
72                                                  x0, x1, true);
73          }
74          return cumulativeProbability(x1) - cumulativeProbability(x0);
75      }
76  
77      /**
78       * {@inheritDoc}
79       *
80       * The default implementation returns
81       * <ul>
82       * <li>{@link #getSupportLowerBound()} for {@code p = 0},</li>
83       * <li>{@link #getSupportUpperBound()} for {@code p = 1}.</li>
84       * </ul>
85       */
86      @Override
87      public double inverseCumulativeProbability(final double p) throws OutOfRangeException {
88          /*
89           * IMPLEMENTATION NOTES
90           * --------------------
91           * Where applicable, use is made of the one-sided Chebyshev inequality
92           * to bracket the root. This inequality states that
93           * P(X - mu >= k * sig) <= 1 / (1 + k^2),
94           * mu: mean, sig: standard deviation. Equivalently
95           * 1 - P(X < mu + k * sig) <= 1 / (1 + k^2),
96           * F(mu + k * sig) >= k^2 / (1 + k^2).
97           *
98           * For k = sqrt(p / (1 - p)), we find
99           * F(mu + k * sig) >= p,
100          * and (mu + k * sig) is an upper-bound for the root.
101          *
102          * Then, introducing Y = -X, mean(Y) = -mu, sd(Y) = sig, and
103          * P(Y >= -mu + k * sig) <= 1 / (1 + k^2),
104          * P(-X >= -mu + k * sig) <= 1 / (1 + k^2),
105          * P(X <= mu - k * sig) <= 1 / (1 + k^2),
106          * F(mu - k * sig) <= 1 / (1 + k^2).
107          *
108          * For k = sqrt((1 - p) / p), we find
109          * F(mu - k * sig) <= p,
110          * and (mu - k * sig) is a lower-bound for the root.
111          *
112          * In cases where the Chebyshev inequality does not apply, geometric
113          * progressions 1, 2, 4, ... and -1, -2, -4, ... are used to bracket
114          * the root.
115          */
116         if (p < 0.0 || p > 1.0) {
117             throw new OutOfRangeException(p, 0, 1);
118         }
119 
120         double lowerBound = getSupportLowerBound();
121         if (p == 0.0) {
122             return lowerBound;
123         }
124 
125         double upperBound = getSupportUpperBound();
126         if (p == 1.0) {
127             return upperBound;
128         }
129 
130         final double mu = getNumericalMean();
131         final double sig = FastMath.sqrt(getNumericalVariance());
132         final boolean chebyshevApplies;
133         chebyshevApplies = !(Double.isInfinite(mu) || Double.isNaN(mu) ||
134                              Double.isInfinite(sig) || Double.isNaN(sig));
135 
136         if (lowerBound == Double.NEGATIVE_INFINITY) {
137             if (chebyshevApplies) {
138                 lowerBound = mu - sig * FastMath.sqrt((1. - p) / p);
139             } else {
140                 lowerBound = -1.0;
141                 while (cumulativeProbability(lowerBound) >= p) {
142                     lowerBound *= 2.0;
143                 }
144             }
145         }
146 
147         if (upperBound == Double.POSITIVE_INFINITY) {
148             if (chebyshevApplies) {
149                 upperBound = mu + sig * FastMath.sqrt(p / (1. - p));
150             } else {
151                 upperBound = 1.0;
152                 while (cumulativeProbability(upperBound) < p) {
153                     upperBound *= 2.0;
154                 }
155             }
156         }
157 
158         final UnivariateFunction toSolve = new UnivariateFunction() {
159             /** {@inheritDoc} */
160             @Override
161             public double value(final double x) {
162                 return cumulativeProbability(x) - p;
163             }
164         };
165 
166         double x = UnivariateSolverUtils.solve(toSolve,
167                                                    lowerBound,
168                                                    upperBound,
169                                                    getSolverAbsoluteAccuracy());
170 
171         if (!isSupportConnected()) {
172             /* Test for plateau. */
173             final double dx = getSolverAbsoluteAccuracy();
174             if (x - dx >= getSupportLowerBound()) {
175                 double px = cumulativeProbability(x);
176                 if (cumulativeProbability(x - dx) == px) {
177                     upperBound = x;
178                     while (upperBound - lowerBound > dx) {
179                         final double midPoint = 0.5 * (lowerBound + upperBound);
180                         if (cumulativeProbability(midPoint) < px) {
181                             lowerBound = midPoint;
182                         } else {
183                             upperBound = midPoint;
184                         }
185                     }
186                     return upperBound;
187                 }
188             }
189         }
190         return x;
191     }
192 
193     /**
194      * Returns the solver absolute accuracy for inverse cumulative computation.
195      * You can override this method in order to use a Brent solver with an
196      * absolute accuracy different from the default.
197      *
198      * @return the maximum absolute error in inverse cumulative probability estimates
199      */
200     protected double getSolverAbsoluteAccuracy() {
201         return SOLVER_DEFAULT_ABSOLUTE_ACCURACY;
202     }
203 
204     /**
205      * {@inheritDoc}
206      *
207      * @return zero.
208      * @since 3.1
209      */
210     @Override
211     public double probability(double x) {
212         return 0d;
213     }
214 
215     /**
216      * {@inheritDoc}
217      * <p>
218      * The default implementation simply computes the logarithm of {@code density(x)}.
219      */
220     @Override
221     public double logDensity(double x) {
222         return FastMath.log(density(x));
223     }
224 
225     /**
226      * Utility function for allocating an array and filling it with {@code n}
227      * samples generated by the given {@code sampler}.
228      *
229      * @param n Number of samples.
230      * @param sampler Sampler.
231      * @return an array of size {@code n}.
232      */
233     public static double[] sample(int n,
234                                   RealDistribution.Sampler sampler) {
235         final double[] samples = new double[n];
236         for (int i = 0; i < n; i++) {
237             samples[i] = sampler.sample();
238         }
239         return samples;
240     }
241 
242     /**{@inheritDoc} */
243     @Override
244     public RealDistribution.Sampler createSampler(final UniformRandomProvider rng) {
245         return new RealDistribution.Sampler() {
246             /** {@inheritDoc} */
247             @Override
248             public double sample() {
249                 return inverseCumulativeProbability(rng.nextDouble());
250             }
251         };
252     }
