44  /**
45   * An implementation of the active Covariance Matrix Adaptation Evolution Strategy (CMA-ES)
46   * for non-linear, non-convex, non-smooth, global function minimization.
47   * <p>
48   * The CMA-Evolution Strategy (CMA-ES) is a reliable stochastic optimization method
49   * which should be applied if derivative-based methods, e.g. quasi-Newton BFGS or
50   * conjugate gradient, fail due to a rugged search landscape (e.g. noise, local
51   * optima, outlier, etc.) of the objective function. Like a
52   * quasi-Newton method, the CMA-ES learns and applies a variable metric
53   * on the underlying search space. Unlike a quasi-Newton method, the
54   * CMA-ES neither estimates nor uses gradients, making it considerably more
55   * reliable in terms of finding a good, or even close to optimal, solution.
56   * <p>
57   * In general, on smooth objective functions the CMA-ES is roughly ten times
58   * slower than BFGS (counting objective function evaluations, no gradients provided).
59   * For up to <math>N=10</math> variables also the derivative-free simplex
60   * direct search method (Nelder and Mead) can be faster, but it is
61   * far less reliable than CMA-ES.
62   * <p>
63   * The CMA-ES is particularly well suited for non-separable
64   * and/or badly conditioned problems. To observe the advantage of CMA compared
65   * to a conventional evolution strategy, it will usually take about
66   * <math>30 N</math> function evaluations. On difficult problems the complete
67   * optimization (a single run) is expected to take <em>roughly</em> between
68   * <math>30 N</math> and <math>300 N<sup>2</sup></math>
69   * function evaluations.
70   * <p>
71   * This implementation is translated and adapted from the Matlab version
72   * of the CMA-ES algorithm as implemented in module {@code cmaes.m} version 3.51.
73   * <p>
74   * For more information, please refer to the following links:
75   * <ul>
76   *  <li><a href="http://www.lri.fr/~hansen/cmaes.m">Matlab code</a></li>
77   *  <li><a href="http://www.lri.fr/~hansen/cmaesintro.html">Introduction to CMA-ES</a></li>
78   *  <li><a href="http://en.wikipedia.org/wiki/CMA-ES">Wikipedia</a></li>
79   * </ul>
80   *
81   * @since 3.0
82   */

370     /** {@inheritDoc} */
371     @Override
372     protected PointValuePair doOptimize() {
373          // -------------------- Initialization --------------------------------
374         isMinimize = getGoalType().equals(GoalType.MINIMIZE);
375         final FitnessFunction fitfun = new FitnessFunction();
376         final double[] guess = getStartPoint();
377         // number of objective variables/problem dimension
378         dimension = guess.length;
379         initializeCMA(guess);
380         iterations = 0;
381         ValuePenaltyPair valuePenalty = fitfun.value(guess);
382         double bestValue = valuePenalty.value+valuePenalty.penalty;
383         push(fitnessHistory, bestValue);
384         PointValuePair optimum
385             = new PointValuePair(getStartPoint(),
386                                  isMinimize ? bestValue : -bestValue);
387         PointValuePair lastResult = null;
388 
389         // -------------------- Generation Loop --------------------------------
390 
391         generationLoop:
392         for (iterations = 1; iterations <= maxIterations; iterations++) {
393             incrementIterationCount();
394 
395             // Generate and evaluate lambda offspring
396             final RealMatrix arz = randn1(dimension, lambda);
397             final RealMatrix arx = zeros(dimension, lambda);
398             final double[] fitness = new double[lambda];
399             final ValuePenaltyPair[] valuePenaltyPairs = new ValuePenaltyPair[lambda];
400             // generate random offspring
401             for (int k = 0; k < lambda; k++) {
402                 RealMatrix arxk = null;
403                 for (int i = 0; i < checkFeasableCount + 1; i++) {
404                     if (diagonalOnly <= 0) {
405                         arxk = xmean.add(BD.multiply(arz.getColumnMatrix(k))
406                                          .scalarMultiply(sigma)); // m + sig * Normal(0,C)
407                     } else {
408                         arxk = xmean.add(times(diagD,arz.getColumnMatrix(k))
409                                          .scalarMultiply(sigma));
410                     }
411                     if (i >= checkFeasableCount ||
412                         fitfun.isFeasible(arxk.getColumn(0))) {
413                         break;
414                     }
415                     // regenerate random arguments for row
416                     arz.setColumn(k, randn(dimension));
417                 }
418                 copyColumn(arxk, 0, arx, k);
419                 try {
420                     valuePenaltyPairs[k] = fitfun.value(arx.getColumn(k)); // compute fitness
421                 } catch (TooManyEvaluationsException e) {
422                     break generationLoop;
423                 }
424             }
425 
426             // Compute fitnesses by adding value and penalty after scaling by value range.
427             double valueRange = valueRange(valuePenaltyPairs);
428             for (int iValue=0;iValue<valuePenaltyPairs.length;iValue++) {
429                  fitness[iValue] = valuePenaltyPairs[iValue].value + valuePenaltyPairs[iValue].penalty*valueRange;
430             }
431 
432             // Sort by fitness and compute weighted mean into xmean
433             final int[] arindex = sortedIndices(fitness);
434             // Calculate new xmean, this is selection and recombination
435             final RealMatrix xold = xmean; // for speed up of Eq. (2) and (3)
436             final RealMatrix bestArx = selectColumns(arx, MathArrays.copyOf(arindex, mu));
437             xmean = bestArx.multiply(weights);
438             final RealMatrix bestArz = selectColumns(arz, MathArrays.copyOf(arindex, mu));
439             final RealMatrix zmean = bestArz.multiply(weights);
440             final boolean hsig = updateEvolutionPaths(zmean, xold);
441             if (diagonalOnly <= 0) {
442                 updateCovariance(hsig, bestArx, arz, arindex, xold);
443             } else {
444                 updateCovarianceDiagonalOnly(hsig, bestArz);
445             }
446             // Adapt step size sigma - Eq. (5)
447             sigma *= FastMath.exp(FastMath.min(1, (normps/chiN - 1) * cs / damps));
448             final double bestFitness = fitness[arindex[0]];
449             final double worstFitness = fitness[arindex[arindex.length - 1]];
450             if (bestValue > bestFitness) {
451                 bestValue = bestFitness;
452                 lastResult = optimum;
453                 optimum = new PointValuePair(fitfun.repair(bestArx.getColumn(0)),
454                                              isMinimize ? bestFitness : -bestFitness);
455                 if (getConvergenceChecker() != null && lastResult != null &&
456                     getConvergenceChecker().converged(iterations, optimum, lastResult)) {
457                     break generationLoop;
458                 }
459             }
460             // handle termination criteria
461             // Break, if fitness is good enough
462             if (stopFitness != 0 && bestFitness < (isMinimize ? stopFitness : -stopFitness)) {
463                 break generationLoop;
464             }
465             final double[] sqrtDiagC = sqrt(diagC).getColumn(0);
466             final double[] pcCol = pc.getColumn(0);
467             for (int i = 0; i < dimension; i++) {
468                 if (sigma * FastMath.max(FastMath.abs(pcCol[i]), sqrtDiagC[i]) > stopTolX) {
469                     break;
470                 }
471                 if (i >= dimension - 1) {
472                     break generationLoop;
473                 }
474             }
475             for (int i = 0; i < dimension; i++) {
476                 if (sigma * sqrtDiagC[i] > stopTolUpX) {
477                     break generationLoop;
478                 }
479             }
480             final double historyBest = min(fitnessHistory);
481             final double historyWorst = max(fitnessHistory);
482             if (iterations > 2 &&
483                 FastMath.max(historyWorst, worstFitness) -
484                 FastMath.min(historyBest, bestFitness) < stopTolFun) {
485                 break generationLoop;
486             }
487             if (iterations > fitnessHistory.length &&
488                 historyWorst - historyBest < stopTolHistFun) {
489                 break generationLoop;
490             }
491             // condition number of the covariance matrix exceeds 1e14
492             if (max(diagD) / min(diagD) > 1e7) {
493                 break generationLoop;
494             }
495             // user defined termination
496             if (getConvergenceChecker() != null) {
497                 final PointValuePair current
498                     = new PointValuePair(bestArx.getColumn(0),
499                                          isMinimize ? bestFitness : -bestFitness);
500                 if (lastResult != null &&
501                     getConvergenceChecker().converged(iterations, current, lastResult)) {
502                     break generationLoop;
503                     }
504                 lastResult = current;
505             }
506             // Adjust step size in case of equal function values (flat fitness)
507             if (bestValue == fitness[arindex[(int)(0.1+lambda/4.)]]) {
508                 sigma *= FastMath.exp(0.2 + cs / damps);
509             }
510             if (iterations > 2 && FastMath.max(historyWorst, bestFitness) -
511                 FastMath.min(historyBest, bestFitness) == 0) {
512                 sigma *= FastMath.exp(0.2 + cs / damps);
513             }
514             // store best in history
515             push(fitnessHistory,bestFitness);
516             if (generateStatistics) {
517                 statisticsSigmaHistory.add(sigma);
518                 statisticsFitnessHistory.add(bestFitness);
519                 statisticsMeanHistory.add(xmean.transpose());
520                 statisticsDHistory.add(diagD.transpose().scalarMultiply(1E5));
521             }
522         }
523         return optimum;
524     }

526     /**
527      * Scans the list of (required and optional) optimization data that
528      * characterize the problem.
529      *
530      * @param optData Optimization data. The following data will be looked for:
531      * <ul>
532      *  <li>{@link Sigma}</li>
533      *  <li>{@link PopulationSize}</li>
534      * </ul>
535      */
536     @Override
537     protected void parseOptimizationData(OptimizationData... optData) {
538         // Allow base class to register its own data.
539         super.parseOptimizationData(optData);
540 
541         // The existing values (as set by the previous call) are reused if
542         // not provided in the argument list.
543         for (OptimizationData data : optData) {
544             if (data instanceof Sigma) {
545                 inputSigma = ((Sigma) data).getSigma();
546                 continue;
547             }
548             if (data instanceof PopulationSize) {
549                 lambda = ((PopulationSize) data).getPopulationSize();
550                 continue;
551             }
552         }
553 
554         checkParameters();
555     }

557     /**
558      * Checks dimensions and values of boundaries and inputSigma if defined.
559      */
560     private void checkParameters() {
561         if (inputSigma != null) {
562             final double[] init = getStartPoint();
563 
564             if (inputSigma.length != init.length) {
565                 throw new DimensionMismatchException(inputSigma.length, init.length);
566             }
567 
568             final double[] lB = getLowerBound();
569             final double[] uB = getUpperBound();
570 
571             for (int i = 0; i < init.length; i++) {
572                 if (inputSigma[i] > uB[i] - lB[i]) {
573                     throw new OutOfRangeException(inputSigma[i], 0, uB[i] - lB[i]);
574                 }
575             }
576         }
577     }

579     /**
580      * Initialization of the dynamic search parameters
581      *
582      * @param guess Initial guess for the arguments of the fitness function.
583      */
584     private void initializeCMA(double[] guess) {
585         if (lambda <= 0) {
586             throw new NotStrictlyPositiveException(lambda);
587         }
588         // initialize sigma
589         final double[][] sigmaArray = new double[guess.length][1];
590         for (int i = 0; i < guess.length; i++) {
591             sigmaArray[i][0] = inputSigma[i];
592         }
593         final RealMatrix insigma = new Array2DRowRealMatrix(sigmaArray, false);
594         sigma = max(insigma); // overall standard deviation
595 
596         // initialize termination criteria
597         stopTolUpX = 1e3 * max(insigma);
598         stopTolX = 1e-11 * max(insigma);
599         stopTolFun = 1e-12;
600         stopTolHistFun = 1e-13;
601 
602         // initialize selection strategy parameters
603         mu = lambda / 2; // number of parents/points for recombination
604         logMu2 = FastMath.log(mu + 0.5);
605         weights = log(sequence(1, mu, 1)).scalarMultiply(-1).scalarAdd(logMu2);
606         double sumw = 0;
607         double sumwq = 0;
608         for (int i = 0; i < mu; i++) {
609             double w = weights.getEntry(i, 0);
610             sumw += w;
611             sumwq += w * w;
612         }
613         weights = weights.scalarMultiply(1 / sumw);
614         mueff = sumw * sumw / sumwq; // variance-effectiveness of sum w_i x_i
615 
616         // initialize dynamic strategy parameters and constants
617         cc = (4 + mueff / dimension) /
618                 (dimension + 4 + 2 * mueff / dimension);
619         cs = (mueff + 2) / (dimension + mueff + 3.);
620         damps = (1 + 2 * FastMath.max(0, FastMath.sqrt((mueff - 1) /
621                                                        (dimension + 1)) - 1)) *
622             FastMath.max(0.3,
623                          1 - dimension / (1e-6 + maxIterations)) + cs; // minor increment
624         ccov1 = 2 / ((dimension + 1.3) * (dimension + 1.3) + mueff);
625         ccovmu = FastMath.min(1 - ccov1, 2 * (mueff - 2 + 1 / mueff) /
626                               ((dimension + 2) * (dimension + 2) + mueff));
627         ccov1Sep = FastMath.min(1, ccov1 * (dimension + 1.5) / 3);
628         ccovmuSep = FastMath.min(1 - ccov1, ccovmu * (dimension + 1.5) / 3);
629         chiN = FastMath.sqrt(dimension) *
630                 (1 - 1 / ((double) 4 * dimension) + 1 / ((double) 21 * dimension * dimension));
631         // intialize CMA internal values - updated each generation
632         xmean = MatrixUtils.createColumnRealMatrix(guess); // objective variables
633         diagD = insigma.scalarMultiply(1 / sigma);
634         diagC = square(diagD);
635         pc = zeros(dimension, 1); // evolution paths for C and sigma
636         ps = zeros(dimension, 1); // B defines the coordinate system
637         normps = ps.getFrobeniusNorm();
638 
639         B = eye(dimension, dimension);
640         D = ones(dimension, 1); // diagonal D defines the scaling
641         BD = times(B, repmat(diagD.transpose(), dimension, 1));
642         C = B.multiply(diag(square(D)).multiply(B.transpose())); // covariance
643         historySize = 10 + (int) (3 * 10 * dimension / (double) lambda);
644         fitnessHistory = new double[historySize]; // history of fitness values
645         for (int i = 0; i < historySize; i++) {
646             fitnessHistory[i] = Double.MAX_VALUE;
647         }
648     }

650     /**
651      * Update of the evolution paths ps and pc.
652      *
653      * @param zmean Weighted row matrix of the gaussian random numbers generating
654      * the current offspring.
655      * @param xold xmean matrix of the previous generation.
656      * @return hsig flag indicating a small correction.
657      */
658     private boolean updateEvolutionPaths(RealMatrix zmean, RealMatrix xold) {
659         ps = ps.scalarMultiply(1 - cs).add(
660                 B.multiply(zmean).scalarMultiply(
661                         FastMath.sqrt(cs * (2 - cs) * mueff)));
662         normps = ps.getFrobeniusNorm();
663         final boolean hsig = normps /
664             FastMath.sqrt(1 - FastMath.pow(1 - cs, 2 * iterations)) /
665             chiN < 1.4 + 2 / ((double) dimension + 1);
666         pc = pc.scalarMultiply(1 - cc);
667         if (hsig) {
668             pc = pc.add(xmean.subtract(xold).scalarMultiply(FastMath.sqrt(cc * (2 - cc) * mueff) / sigma));
669         }
670         return hsig;
671     }

673     /**
674      * Update of the covariance matrix C for diagonalOnly > 0
675      *
676      * @param hsig Flag indicating a small correction.
677      * @param bestArz Fitness-sorted matrix of the gaussian random values of the
678      * current offspring.
679      */
680     private void updateCovarianceDiagonalOnly(boolean hsig,
681                                               final RealMatrix bestArz) {
682         // minor correction if hsig==false
683         double oldFac = hsig ? 0 : ccov1Sep * cc * (2 - cc);
684         oldFac += 1 - ccov1Sep - ccovmuSep;
685         diagC = diagC.scalarMultiply(oldFac) // regard old matrix
686             .add(square(pc).scalarMultiply(ccov1Sep)) // plus rank one update
687             .add((times(diagC, square(bestArz).multiply(weights))) // plus rank mu update
688                  .scalarMultiply(ccovmuSep));
689         diagD = sqrt(diagC); // replaces eig(C)
690         if (diagonalOnly > 1 &&
691             iterations > diagonalOnly) {
692             // full covariance matrix from now on
693             diagonalOnly = 0;
694             B = eye(dimension, dimension);
695             BD = diag(diagD);
696             C = diag(diagC);
697         }
698     }

700     /**
701      * Update of the covariance matrix C.
702      *
703      * @param hsig Flag indicating a small correction.
704      * @param bestArx Fitness-sorted matrix of the argument vectors producing the
705      * current offspring.
706      * @param arz Unsorted matrix containing the gaussian random values of the
707      * current offspring.
708      * @param arindex Indices indicating the fitness-order of the current offspring.
709      * @param xold xmean matrix of the previous generation.
710      */
711     private void updateCovariance(boolean hsig, final RealMatrix bestArx,
712                                   final RealMatrix arz, final int[] arindex,
713                                   final RealMatrix xold) {
714         double negccov = 0;
715         if (ccov1 + ccovmu > 0) {
716             final RealMatrix arpos = bestArx.subtract(repmat(xold, 1, mu))
717                 .scalarMultiply(1 / sigma); // mu difference vectors
718             final RealMatrix roneu = pc.multiply(pc.transpose())
719                 .scalarMultiply(ccov1); // rank one update
720             // minor correction if hsig==false
721             double oldFac = hsig ? 0 : ccov1 * cc * (2 - cc);
722             oldFac += 1 - ccov1 - ccovmu;
723             if (isActiveCMA) {
724                 // Adapt covariance matrix C active CMA
725                 negccov = (1 - ccovmu) * 0.25 * mueff /
726                     (FastMath.pow(dimension + 2, 1.5) + 2 * mueff);
727                 // keep at least 0.66 in all directions, small popsize are most
728                 // critical
729                 final double negminresidualvariance = 0.66;
730                 // where to make up for the variance loss
731                 final double negalphaold = 0.5;
732                 // prepare vectors, compute negative updating matrix Cneg
733                 final int[] arReverseIndex = reverse(arindex);
734                 RealMatrix arzneg = selectColumns(arz, MathArrays.copyOf(arReverseIndex, mu));
735                 RealMatrix arnorms = sqrt(sumRows(square(arzneg)));
736                 final int[] idxnorms = sortedIndices(arnorms.getRow(0));
737                 final RealMatrix arnormsSorted = selectColumns(arnorms, idxnorms);
738                 final int[] idxReverse = reverse(idxnorms);
739                 final RealMatrix arnormsReverse = selectColumns(arnorms, idxReverse);
740                 arnorms = divide(arnormsReverse, arnormsSorted);
741                 final int[] idxInv = inverse(idxnorms);
742                 final RealMatrix arnormsInv = selectColumns(arnorms, idxInv);
743                 // check and set learning rate negccov
744                 final double negcovMax = (1 - negminresidualvariance) /
745                     square(arnormsInv).multiply(weights).getEntry(0, 0);
746                 if (negccov > negcovMax) {
747                     negccov = negcovMax;
748                 }
749                 arzneg = times(arzneg, repmat(arnormsInv, dimension, 1));
750                 final RealMatrix artmp = BD.multiply(arzneg);
751                 final RealMatrix Cneg = artmp.multiply(diag(weights)).multiply(artmp.transpose());
752                 oldFac += negalphaold * negccov;
753                 C = C.scalarMultiply(oldFac)
754                     .add(roneu) // regard old matrix
755                     .add(arpos.scalarMultiply( // plus rank one update
756                                               ccovmu + (1 - negalphaold) * negccov) // plus rank mu update
757                          .multiply(times(repmat(weights, 1, dimension),
758                                          arpos.transpose())))
759                     .subtract(Cneg.scalarMultiply(negccov));
760             } else {
761                 // Adapt covariance matrix C - nonactive
762                 C = C.scalarMultiply(oldFac) // regard old matrix
763                     .add(roneu) // plus rank one update
764                     .add(arpos.scalarMultiply(ccovmu) // plus rank mu update
765                          .multiply(times(repmat(weights, 1, dimension),
766                                          arpos.transpose())));
767             }
768         }
769         updateBD(negccov);
770     }

772     /**
773      * Update B and D from C.
774      *
775      * @param negccov Negative covariance factor.
776      */
777     private void updateBD(double negccov) {
778         if (ccov1 + ccovmu + negccov > 0 &&
779             (iterations % 1. / (ccov1 + ccovmu + negccov) / dimension / 10.) < 1) {
780             // to achieve O(N^2)
781             C = triu(C, 0).add(triu(C, 1).transpose());
782             // enforce symmetry to prevent complex numbers
783             final EigenDecomposition eig = new EigenDecomposition(C);
784             B = eig.getV(); // eigen decomposition, B==normalized eigenvectors
785             D = eig.getD();
786             diagD = diag(D);
787             if (min(diagD) <= 0) {
788                 for (int i = 0; i < dimension; i++) {
789                     if (diagD.getEntry(i, 0) < 0) {
790                         diagD.setEntry(i, 0, 0);
791                     }
792                 }
793                 final double tfac = max(diagD) / 1e14;
794                 C = C.add(eye(dimension, dimension).scalarMultiply(tfac));
795                 diagD = diagD.add(ones(dimension, 1).scalarMultiply(tfac));
796             }
797             if (max(diagD) > 1e14 * min(diagD)) {
798                 final double tfac = max(diagD) / 1e14 - min(diagD);
799                 C = C.add(eye(dimension, dimension).scalarMultiply(tfac));
800                 diagD = diagD.add(ones(dimension, 1).scalarMultiply(tfac));
801             }
802             diagC = diag(C);
803             diagD = sqrt(diagD); // D contains standard deviations now
804             BD = times(B, repmat(diagD.transpose(), dimension, 1)); // O(n^2)
805         }
806     }

926     /**
927      * Normalizes fitness values to the range [0,1]. Adds a penalty to the
928      * fitness value if out of range.
929      */
930     private class FitnessFunction {
931         /**
932          * Flag indicating whether the objective variables are forced into their
933          * bounds if defined
934          */
935         private final boolean isRepairMode;
936 
937         /** Simple constructor.
938          */
939         FitnessFunction() {
940             isRepairMode = true;
941         }
942 
943         /**
944          * @param point Normalized objective variables.
945          * @return the objective value + penalty for violated bounds.
946          */
947         public ValuePenaltyPair value(final double[] point) {
948             double value;
949             double penalty=0.0;
950             if (isRepairMode) {
951                 double[] repaired = repair(point);
952                 value = CMAESOptimizer.this.computeObjectiveValue(repaired);
953                 penalty =  penalty(point, repaired);
954             } else {
955                 value = CMAESOptimizer.this.computeObjectiveValue(point);
956             }
957             value = isMinimize ? value : -value;
958             penalty = isMinimize ? penalty : -penalty;
959             return new ValuePenaltyPair(value,penalty);
960         }
961 
962         /**
963          * @param x Normalized objective variables.
964          * @return {@code true} if in bounds.
965          */
966         public boolean isFeasible(final double[] x) {
967             final double[] lB = CMAESOptimizer.this.getLowerBound();
968             final double[] uB = CMAESOptimizer.this.getUpperBound();
969 
970             for (int i = 0; i < x.length; i++) {
971                 if (x[i] < lB[i]) {
972                     return false;
973                 }
974                 if (x[i] > uB[i]) {
975                     return false;
976                 }
977             }
978             return true;
979         }
980 
981         /**
982          * @param x Normalized objective variables.
983          * @return the repaired (i.e. all in bounds) objective variables.
984          */
985         private double[] repair(final double[] x) {
986             final double[] lB = CMAESOptimizer.this.getLowerBound();
987             final double[] uB = CMAESOptimizer.this.getUpperBound();
988 
989             final double[] repaired = new double[x.length];
990             for (int i = 0; i < x.length; i++) {
991                 if (x[i] < lB[i]) {
992                     repaired[i] = lB[i];
993                 } else if (x[i] > uB[i]) {
994                     repaired[i] = uB[i];
995                 } else {
996                     repaired[i] = x[i];
997                 }
998             }
999             return repaired;
1000         }
1001 
1002         /**
1003          * @param x Normalized objective variables.
1004          * @param repaired Repaired objective variables.
1005          * @return Penalty value according to the violation of the bounds.
1006          */
1007         private double penalty(final double[] x, final double[] repaired) {
1008             double penalty = 0;
1009             for (int i = 0; i < x.length; i++) {
1010                 double diff = FastMath.abs(x[i] - repaired[i]);
1011                 penalty += diff;
1012             }
1013             return isMinimize ? penalty : -penalty;
1014         }
1015     }
