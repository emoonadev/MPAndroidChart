17  package org.apache.commons.math4.stat.descriptive;
18  
19  /**
20   *  Reporting interface for basic univariate statistics.
21   *
22   */
23  public interface StatisticalSummary {
24  
25      /**
26       * Returns the <a href="http://www.xycoon.com/arithmetic_mean.htm">
27       * arithmetic mean </a> of the available values
28       * @return The mean or Double.NaN if no values have been added.
29       */
30      double getMean();
31      /**
32       * Returns the variance of the available values.
33       * @return The variance, Double.NaN if no values have been added
34       * or 0.0 for a single value set.
35       */
36      double getVariance();
37      /**
38       * Returns the standard deviation of the available values.
39       * @return The standard deviation, Double.NaN if no values have been added
40       * or 0.0 for a single value set.
41       */
42      double getStandardDeviation();
43      /**
44       * Returns the maximum of the available values
45       * @return The max or Double.NaN if no values have been added.
46       */
47      double getMax();
48      /**
49      * Returns the minimum of the available values
50      * @return The min or Double.NaN if no values have been added.
51      */
52      double getMin();
53      /**
54       * Returns the number of available values
55       * @return The number of available values
56       */
57      long getN();
58      /**
59       * Returns the sum of the values that have been added to Univariate.
60       * @return The sum or Double.NaN if no values have been added
61       */
62      double getSum();
63  
64  }
