18  package org.apache.commons.math4.geometry.spherical.twod;
19  
20  import java.io.Serializable;
21  
22  import org.apache.commons.math4.geometry.Space;
23  import org.apache.commons.math4.geometry.spherical.oned.Sphere1D;
24  
25  /**
26   * This class implements a two-dimensional sphere (i.e. the regular sphere).
27   * <p>
28   * We use here the topologists definition of the 2-sphere (see
29   * <a href="http://mathworld.wolfram.com/Sphere.html">Sphere</a> on
30   * MathWorld), i.e. the 2-sphere is the two-dimensional surface
31   * defined in 3D as x<sup>2</sup>+y<sup>2</sup>+z<sup>2</sup>=1.
32   * </p>
33   * @since 3.3
34   */
35  public class Sphere2D implements Serializable, Space {
36  
37      /** Serializable version identifier. */
38      private static final long serialVersionUID = 20131218L;
39  
40      /** Private constructor for the singleton.
41       */
42      private Sphere2D() {
43      }
44  
45      /** Get the unique instance.
46       * @return the unique instance
47       */
48      public static Sphere2D getInstance() {
49          return LazyHolder.INSTANCE;
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      public int getDimension() {
55          return 2;
56      }
57  
58      /** {@inheritDoc} */
59      @Override
60      public Sphere1D getSubSpace() {
61          return Sphere1D.getInstance();
62      }
63  
64      // CHECKSTYLE: stop HideUtilityClassConstructor
65      /** Holder for the instance.
66       * <p>We use here the Initialization On Demand Holder Idiom.</p>
67       */
68      private static class LazyHolder {
69          /** Cached field instance. */
70          private static final Sphere2D INSTANCE = new Sphere2D();
71      }
72      // CHECKSTYLE: resume HideUtilityClassConstructor
73  
74      /** Handle deserialization of the singleton.
75       * @return the singleton instance
76       */
77      private Object readResolve() {
78          // return the singleton instance
79          return LazyHolder.INSTANCE;
80      }
81  
82  }
