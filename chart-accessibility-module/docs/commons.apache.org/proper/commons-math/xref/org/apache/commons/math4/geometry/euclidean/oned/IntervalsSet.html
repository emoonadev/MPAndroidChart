54          super(buildTree(lower, upper, tolerance), tolerance);

57      /** Build an intervals set from an inside/outside BSP tree.
58       * <p>The leaf nodes of the BSP tree <em>must</em> have a
59       * {@code Boolean} attribute representing the inside status of
60       * the corresponding cell (true for inside cells, false for outside
61       * cells). In order to avoid building too many small objects, it is
62       * recommended to use the predefined constants
63       * {@code Boolean.TRUE} and {@code Boolean.FALSE}</p>
64       * @param tree inside/outside BSP tree representing the intervals set
65       * @param tolerance tolerance below which points are considered identical.
66       * @since 3.3
67       */
68      public IntervalsSet(final BSPTree<Euclidean1D> tree, final double tolerance) {
69          super(tree, tolerance);
70      }
71  
72      /** Build an intervals set from a Boundary REPresentation (B-rep).
73       * <p>The boundary is provided as a collection of {@link
74       * SubHyperplane sub-hyperplanes}. Each sub-hyperplane has the
75       * interior part of the region on its minus side and the exterior on
76       * its plus side.</p>
77       * <p>The boundary elements can be in any order, and can form
78       * several non-connected sets (like for example polygons with holes
79       * or a set of disjoints polyhedrons considered as a whole). In
80       * fact, the elements do not even need to be connected together
81       * (their topological connections are not used here). However, if the
82       * boundary does not really separate an inside open from an outside
83       * open (open having here its topological meaning), then subsequent
84       * calls to the {@link
85       * org.apache.commons.math4.geometry.partitioning.Region#checkPoint(org.apache.commons.math4.geometry.Point)
86       * checkPoint} method will not be meaningful anymore.</p>
87       * <p>If the boundary is empty, the region will represent the whole
88       * space.</p>
89       * @param boundary collection of boundary elements
90       * @param tolerance tolerance below which points are considered identical.
91       * @since 3.3
92       */
93      public IntervalsSet(final Collection<SubHyperplane<Euclidean1D>> boundary,
94                          final double tolerance) {
95          super(boundary, tolerance);
96      }
97  
98      /** Build an inside/outside tree representing a single interval.
99       * @param lower lower bound of the interval, must be lesser or equal
100      * to {@code upper} (may be {@code Double.NEGATIVE_INFINITY})
101      * @param upper upper bound of the interval, must be greater or equal
102      * to {@code lower} (may be {@code Double.POSITIVE_INFINITY})
103      * @param tolerance tolerance below which points are considered identical.
104      * @return the built tree
105      */
106     private static BSPTree<Euclidean1D> buildTree(final double lower, final double upper,
107                                                   final double tolerance) {
108         if (Double.isInfinite(lower) && (lower < 0)) {
109             if (Double.isInfinite(upper) && (upper > 0)) {
110                 // the tree must cover the whole real line
111                 return new BSPTree<Euclidean1D>(Boolean.TRUE);
112             }
113             // the tree must be open on the negative infinity side
114             final SubHyperplane<Euclidean1D> upperCut =
115                 new OrientedPoint(new Vector1D(upper), true, tolerance).wholeHyperplane();
116             return new BSPTree<Euclidean1D>(upperCut,
117                                new BSPTree<Euclidean1D>(Boolean.FALSE),
118                                new BSPTree<Euclidean1D>(Boolean.TRUE),
119                                null);
120         }
121         final SubHyperplane<Euclidean1D> lowerCut =
122             new OrientedPoint(new Vector1D(lower), false, tolerance).wholeHyperplane();
123         if (Double.isInfinite(upper) && (upper > 0)) {
124             // the tree must be open on the positive infinity side
125             return new BSPTree<Euclidean1D>(lowerCut,
126                                             new BSPTree<Euclidean1D>(Boolean.FALSE),
127                                             new BSPTree<Euclidean1D>(Boolean.TRUE),
128                                             null);
129         }
130 
131         // the tree must be bounded on the two sides
132         final SubHyperplane<Euclidean1D> upperCut =
133             new OrientedPoint(new Vector1D(upper), true, tolerance).wholeHyperplane();
134         return new BSPTree<Euclidean1D>(lowerCut,
135                                         new BSPTree<Euclidean1D>(Boolean.FALSE),
136                                         new BSPTree<Euclidean1D>(upperCut,
137                                                                  new BSPTree<Euclidean1D>(Boolean.FALSE),
138                                                                  new BSPTree<Euclidean1D>(Boolean.TRUE),
139                                                                  null),
140                                         null);
141 
142     }

150     /** {@inheritDoc} */
151     @Override
152     protected void computeGeometricalProperties() {
153         if (getTree(false).getCut() == null) {
154             setBarycenter((Point<Euclidean1D>) Vector1D.NaN);
155             setSize(((Boolean) getTree(false).getAttribute()) ? Double.POSITIVE_INFINITY : 0);
156         } else {
157             double size = 0.0;
158             double sum = 0.0;
159             for (final Interval interval : asList()) {
160                 size += interval.getSize();
161                 sum  += interval.getSize() * interval.getBarycenter();
162             }
163             setSize(size);
164             if (Double.isInfinite(size)) {
165                 setBarycenter((Point<Euclidean1D>) Vector1D.NaN);
166             } else if (size >= Precision.SAFE_MIN) {
167                 setBarycenter((Point<Euclidean1D>) new Vector1D(sum / size));
168             } else {
169                 setBarycenter((Point<Euclidean1D>) ((OrientedPoint) getTree(false).getCut().getHyperplane()).getLocation());
170             }
171         }
172     }

174     /** Get the lowest value belonging to the instance.
175      * @return lowest value belonging to the instance
176      * ({@code Double.NEGATIVE_INFINITY} if the instance doesn't
177      * have any low bound, {@code Double.POSITIVE_INFINITY} if the
178      * instance is empty)
179      */
180     public double getInf() {
181         BSPTree<Euclidean1D> node = getTree(false);
182         double  inf  = Double.POSITIVE_INFINITY;
183         while (node.getCut() != null) {
184             final OrientedPoint op = (OrientedPoint) node.getCut().getHyperplane();
185             inf  = op.getLocation().getX();
186             node = op.isDirect() ? node.getMinus() : node.getPlus();
187         }
188         return ((Boolean) node.getAttribute()) ? Double.NEGATIVE_INFINITY : inf;
189     }
190 
191     /** Get the highest value belonging to the instance.
192      * @return highest value belonging to the instance
193      * ({@code Double.POSITIVE_INFINITY} if the instance doesn't
194      * have any high bound, {@code Double.NEGATIVE_INFINITY} if the
195      * instance is empty)
196      */
197     public double getSup() {
198         BSPTree<Euclidean1D> node = getTree(false);
199         double  sup  = Double.NEGATIVE_INFINITY;
200         while (node.getCut() != null) {
201             final OrientedPoint op = (OrientedPoint) node.getCut().getHyperplane();
202             sup  = op.getLocation().getX();
203             node = op.isDirect() ? node.getPlus() : node.getMinus();
204         }
205         return ((Boolean) node.getAttribute()) ? Double.POSITIVE_INFINITY : sup;
206     }

208     /** {@inheritDoc}
209      * @since 3.3
210      */
211     @Override
212     public BoundaryProjection<Euclidean1D> projectToBoundary(final Point<Euclidean1D> point) {
213 
214         // get position of test point
215         final double x = ((Vector1D) point).getX();
216 
217         double previous = Double.NEGATIVE_INFINITY;
218         for (final double[] a : this) {
219             if (x < a[0]) {
220                 // the test point lies between the previous and the current intervals
221                 // offset will be positive
222                 final double previousOffset = x - previous;
223                 final double currentOffset  = a[0] - x;
224                 if (previousOffset < currentOffset) {
225                     return new BoundaryProjection<Euclidean1D>(point, finiteOrNullPoint(previous), previousOffset);
226                 } else {
227                     return new BoundaryProjection<Euclidean1D>(point, finiteOrNullPoint(a[0]), currentOffset);
228                 }
229             } else if (x <= a[1]) {
230                 // the test point lies within the current interval
231                 // offset will be negative
232                 final double offset0 = a[0] - x;
233                 final double offset1 = x - a[1];
234                 if (offset0 < offset1) {
235                     return new BoundaryProjection<Euclidean1D>(point, finiteOrNullPoint(a[1]), offset1);
236                 } else {
237                     return new BoundaryProjection<Euclidean1D>(point, finiteOrNullPoint(a[0]), offset0);
238                 }
239             }
240             previous = a[1];
241         }
242 
243         // the test point if past the last sub-interval
244         return new BoundaryProjection<Euclidean1D>(point, finiteOrNullPoint(previous), x - previous);
245 
246     }

248     /** Build a finite point.
249      * @param x abscissa of the point
250      * @return a new point for finite abscissa, null otherwise
251      */
252     private Vector1D finiteOrNullPoint(final double x) {
253         return Double.isInfinite(x) ? null : new Vector1D(x);
254     }

256     /** Build an ordered list of intervals representing the instance.
257      * <p>This method builds this intervals set as an ordered list of
258      * {@link Interval Interval} elements. If the intervals set has no
259      * lower limit, the first interval will have its low bound equal to
260      * {@code Double.NEGATIVE_INFINITY}. If the intervals set has
261      * no upper limit, the last interval will have its upper bound equal
262      * to {@code Double.POSITIVE_INFINITY}. An empty tree will
263      * build an empty list while a tree representing the whole real line
264      * will build a one element list with both bounds being
265      * infinite.</p>
266      * @return a new ordered list containing {@link Interval Interval}
267      * elements
268      */
269     public List<Interval> asList() {
270         final List<Interval> list = new ArrayList<Interval>();
271         for (final double[] a : this) {
272             list.add(new Interval(a[0], a[1]));
273         }
274         return list;
275     }

277     /** Get the first leaf node of a tree.
278      * @param root tree root
279      * @return first leaf node
280      */
281     private BSPTree<Euclidean1D> getFirstLeaf(final BSPTree<Euclidean1D> root) {
282 
283         if (root.getCut() == null) {
284             return root;
285         }
286 
287         // find the smallest internal node
288         BSPTree<Euclidean1D> smallest = null;
289         for (BSPTree<Euclidean1D> n = root; n != null; n = previousInternalNode(n)) {
290             smallest = n;
291         }
292 
293         return leafBefore(smallest);
294 
295     }
296 
297     /** Get the node corresponding to the first interval boundary.
298      * @return smallest internal node,
299      * or null if there are no internal nodes (i.e. the set is either empty or covers the real line)
300      */
301     private BSPTree<Euclidean1D> getFirstIntervalBoundary() {
302 
303         // start search at the tree root
304         BSPTree<Euclidean1D> node = getTree(false);
305         if (node.getCut() == null) {
306             return null;
307         }
308 
309         // walk tree until we find the smallest internal node
310         node = getFirstLeaf(node).getParent();
311 
312         // walk tree until we find an interval boundary
313         while (node != null && !(isIntervalStart(node) || isIntervalEnd(node))) {
314             node = nextInternalNode(node);
315         }
316 
317         return node;
318 
319     }

321     /** Check if an internal node corresponds to the start abscissa of an interval.
322      * @param node internal node to check
323      * @return true if the node corresponds to the start abscissa of an interval
324      */
325     private boolean isIntervalStart(final BSPTree<Euclidean1D> node) {
326 
327         if ((Boolean) leafBefore(node).getAttribute()) {
328             // it has an inside cell before it, it may end an interval but not start it
329             return false;
330         }
331 
332         if (!(Boolean) leafAfter(node).getAttribute()) {
333             // it has an outside cell after it, it is a dummy cut away from real intervals
334             return false;
335         }
336 
337         // the cell has an outside before and an inside after it
338         // it is the start of an interval
339         return true;
340 
341     }
342 
343     /** Check if an internal node corresponds to the end abscissa of an interval.
344      * @param node internal node to check
345      * @return true if the node corresponds to the end abscissa of an interval
346      */
347     private boolean isIntervalEnd(final BSPTree<Euclidean1D> node) {
348 
349         if (!(Boolean) leafBefore(node).getAttribute()) {
350             // it has an outside cell before it, it may start an interval but not end it
351             return false;
352         }
353 
354         if ((Boolean) leafAfter(node).getAttribute()) {
355             // it has an inside cell after it, it is a dummy cut in the middle of an interval
356             return false;
357         }
358 
359         // the cell has an inside before and an outside after it
360         // it is the end of an interval
361         return true;
362 
363     }

365     /** Get the next internal node.
366      * @param node current internal node
367      * @return next internal node in ascending order, or null
368      * if this is the last internal node
369      */
370     private BSPTree<Euclidean1D> nextInternalNode(BSPTree<Euclidean1D> node) {
371 
372         if (childAfter(node).getCut() != null) {
373             // the next node is in the sub-tree
374             return leafAfter(node).getParent();
375         }
376 
377         // there is nothing left deeper in the tree, we backtrack
378         while (isAfterParent(node)) {
379             node = node.getParent();
380         }
381         return node.getParent();
382 
383     }

385     /** Get the previous internal node.
386      * @param node current internal node
387      * @return previous internal node in ascending order, or null
388      * if this is the first internal node
389      */
390     private BSPTree<Euclidean1D> previousInternalNode(BSPTree<Euclidean1D> node) {
391 
392         if (childBefore(node).getCut() != null) {
393             // the next node is in the sub-tree
394             return leafBefore(node).getParent();
395         }
396 
397         // there is nothing left deeper in the tree, we backtrack
398         while (isBeforeParent(node)) {
399             node = node.getParent();
400         }
401         return node.getParent();
402 
403     }

405     /** Find the leaf node just before an internal node.
406      * @param node internal node at which the sub-tree starts
407      * @return leaf node just before the internal node
408      */
409     private BSPTree<Euclidean1D> leafBefore(BSPTree<Euclidean1D> node) {
410 
411         node = childBefore(node);
412         while (node.getCut() != null) {
413             node = childAfter(node);
414         }
415 
416         return node;
417 
418     }

420     /** Find the leaf node just after an internal node.
421      * @param node internal node at which the sub-tree starts

424     private BSPTree<Euclidean1D> leafAfter(BSPTree<Euclidean1D> node) {
425 
426         node = childAfter(node);
427         while (node.getCut() != null) {
428             node = childBefore(node);

435     /** Check if a node is the child before its parent in ascending order.
436      * @param node child node considered
437      * @return true is the node has a parent end is before it in ascending order
438      */
439     private boolean isBeforeParent(final BSPTree<Euclidean1D> node) {
440         final BSPTree<Euclidean1D> parent = node.getParent();
441         if (parent == null) {
442             return false;
443         } else {
444             return node == childBefore(parent);
445         }
446     }

448     /** Check if a node is the child after its parent in ascending order.
449      * @param node child node considered
450      * @return true is the node has a parent end is after it in ascending order
451      */
452     private boolean isAfterParent(final BSPTree<Euclidean1D> node) {
453         final BSPTree<Euclidean1D> parent = node.getParent();
454         if (parent == null) {
455             return false;
456         } else {
457             return node == childAfter(parent);
458         }
459     }

461     /** Find the child node just before an internal node.
462      * @param node internal node at which the sub-tree starts
463      * @return child node just before the internal node
464      */
465     private BSPTree<Euclidean1D> childBefore(BSPTree<Euclidean1D> node) {
466         if (isDirect(node)) {
467             // smaller abscissas are on minus side, larger abscissas are on plus side
468             return node.getMinus();
469         } else {
470             // smaller abscissas are on plus side, larger abscissas are on minus side
471             return node.getPlus();
472         }
473     }
474 
475     /** Find the child node just after an internal node.
476      * @param node internal node at which the sub-tree starts
477      * @return child node just after the internal node
478      */
479     private BSPTree<Euclidean1D> childAfter(BSPTree<Euclidean1D> node) {
480         if (isDirect(node)) {
481             // smaller abscissas are on minus side, larger abscissas are on plus side
482             return node.getPlus();
483         } else {
484             // smaller abscissas are on plus side, larger abscissas are on minus side
485             return node.getMinus();
486         }
487     }

505     /** {@inheritDoc}
506      * <p>
507      * The iterator returns the limit values of sub-intervals in ascending order.
508      * </p>
509      * <p>
510      * The iterator does <em>not</em> support the optional {@code remove} operation.
511      * </p>
512      * @since 3.3
513      */
514     @Override
515     public Iterator<double[]> iterator() {
516         return new SubIntervalsIterator();
517     }
518 
519     /** Local iterator for sub-intervals. */
520     private class SubIntervalsIterator implements Iterator<double[]> {
521 
522         /** Current node. */
523         private BSPTree<Euclidean1D> current;
524 
525         /** Sub-interval no yet returned. */
526         private double[] pending;
527 
528         /** Simple constructor.
529          */
530         SubIntervalsIterator() {
531 
532             current = getFirstIntervalBoundary();
533 
534             if (current == null) {
535                 // all the leaf tree nodes share the same inside/outside status
536                 if ((Boolean) getFirstLeaf(getTree(false)).getAttribute()) {
537                     // it is an inside node, it represents the full real line
538                     pending = new double[] {
539                         Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY
540                     };
541                 } else {
542                     pending = null;
543                 }
544             } else if (isIntervalEnd(current)) {
545                 // the first boundary is an interval end,
546                 // so the first interval starts at infinity
547                 pending = new double[] {
548                     Double.NEGATIVE_INFINITY, getAngle(current)
549                 };
550             } else {
551                 selectPending();
552             }
553         }
554 
555         /** Walk the tree to select the pending sub-interval.
556          */
557         private void selectPending() {
558 
559             // look for the start of the interval
560             BSPTree<Euclidean1D> start = current;
561             while (start != null && !isIntervalStart(start)) {
562                 start = nextInternalNode(start);
563             }
564 
565             if (start == null) {
566                 // we have exhausted the iterator
567                 current = null;
568                 pending = null;
569                 return;
570             }
571 
572             // look for the end of the interval
573             BSPTree<Euclidean1D> end = start;
574             while (end != null && !isIntervalEnd(end)) {
575                 end = nextInternalNode(end);
576             }
577 
578             if (end != null) {
579 
580                 // we have identified the interval
581                 pending = new double[] {
582                     getAngle(start), getAngle(end)
583                 };
584 
585                 // prepare search for next interval
586                 current = end;
587 
588             } else {
589 
590                 // the final interval is open toward infinity
591                 pending = new double[] {
592                     getAngle(start), Double.POSITIVE_INFINITY
593                 };
594 
595                 // there won't be any other intervals
596                 current = null;
597 
598             }
599 
600         }
601 
602         /** {@inheritDoc} */
603         @Override
604         public boolean hasNext() {
605             return pending != null;
606         }
607 
608         /** {@inheritDoc} */
609         @Override
610         public double[] next() {
611             if (pending == null) {
612                 throw new NoSuchElementException();
613             }
614             final double[] next = pending;
615             selectPending();
616             return next;
617         }
618 
619         /** {@inheritDoc} */
620         @Override
621         public void remove() {
622             throw new UnsupportedOperationException();
623         }
624 
625     }
