26  /**
27   * <p>
28   * An aggregator for {@code SummaryStatistics} from several data sets or
29   * data set partitions.  In its simplest usage mode, the client creates an
30   * instance via the zero-argument constructor, then uses
31   * {@link #createContributingStatistics()} to obtain a {@code SummaryStatistics}
32   * for each individual data set / partition.  The per-set statistics objects
33   * are used as normal, and at any time the aggregate statistics for all the
34   * contributors can be obtained from this object.
35   * </p><p>
36   * Clients with specialized requirements can use alternative constructors to
37   * control the statistics implementations and initial values used by the
38   * contributing and the internal aggregate {@code SummaryStatistics} objects.
39   * </p><p>
40   * A static {@link #aggregate(Collection)} method is also included that computes
41   * aggregate statistics directly from a Collection of SummaryStatistics instances.
42   * </p><p>
43   * When {@link #createContributingStatistics()} is used to create SummaryStatistics
44   * instances to be aggregated concurrently, the created instances'
45   * {@link SummaryStatistics#addValue(double)} methods must synchronize on the aggregating
46   * instance maintained by this class.  In multithreaded environments, if the functionality
47   * provided by {@link #aggregate(Collection)} is adequate, that method should be used
48   * to avoid unnecessary computation and synchronization delays.</p>
49   *
50   * @since 2.0
51   *
52   */

81      /**
82       * Initializes a new AggregateSummaryStatistics with the specified statistics
83       * object as a prototype for contributing statistics and for the internal
84       * aggregate statistics.  This provides for customized statistics implementations
85       * to be used by contributing and aggregate statistics.
86       *
87       * @param prototypeStatistics a {@code SummaryStatistics} serving as a
88       *      prototype both for the internal aggregate statistics and for
89       *      contributing statistics obtained via the
90       *      {@code createContributingStatistics()} method.  Being a prototype
91       *      means that other objects are initialized by copying this object's state.
92       *      If {@code null}, a new, default statistics object is used.  Any statistic
93       *      values in the prototype are propagated to contributing statistics
94       *      objects and (once) into these aggregate statistics.
95       * @throws NullArgumentException if prototypeStatistics is null
96       * @see #createContributingStatistics()
97       */
98      public AggregateSummaryStatistics(SummaryStatistics prototypeStatistics) throws NullArgumentException {
99          this(prototypeStatistics,
100              prototypeStatistics == null ? null : new SummaryStatistics(prototypeStatistics));
101     }

103     /**
104      * Initializes a new AggregateSummaryStatistics with the specified statistics
105      * object as a prototype for contributing statistics and for the internal
106      * aggregate statistics.  This provides for different statistics implementations
107      * to be used by contributing and aggregate statistics and for an initial
108      * state to be supplied for the aggregate statistics.
109      *
110      * @param prototypeStatistics a {@code SummaryStatistics} serving as a
111      *      prototype both for the internal aggregate statistics and for
112      *      contributing statistics obtained via the
113      *      {@code createContributingStatistics()} method.  Being a prototype
114      *      means that other objects are initialized by copying this object's state.
115      *      If {@code null}, a new, default statistics object is used.  Any statistic
116      *      values in the prototype are propagated to contributing statistics
117      *      objects, but not into these aggregate statistics.
118      * @param initialStatistics a {@code SummaryStatistics} to serve as the
119      *      internal aggregate statistics object.  If {@code null}, a new, default
120      *      statistics object is used.
121      * @see #createContributingStatistics()
122      */
123     public AggregateSummaryStatistics(SummaryStatistics prototypeStatistics,
124                                       SummaryStatistics initialStatistics) {
125         this.statisticsPrototype =
126             (prototypeStatistics == null) ? new SummaryStatistics() : prototypeStatistics;
127         this.statistics =
128             (initialStatistics == null) ? new SummaryStatistics() : initialStatistics;
129     }

282     /**
283      * Creates and returns a {@code SummaryStatistics} whose data will be
284      * aggregated with those of this {@code AggregateSummaryStatistics}.
285      *
286      * @return a {@code SummaryStatistics} whose data will be aggregated with
287      *      those of this {@code AggregateSummaryStatistics}.  The initial state
288      *      is a copy of the configured prototype statistics.
289      */
290     public SummaryStatistics createContributingStatistics() {
291         SummaryStatistics contributingStatistics
292                 = new AggregatingSummaryStatistics(statistics);
293 
294         // No try - catch or advertising NAE because neither argument will ever be null
295         SummaryStatistics.copy(statisticsPrototype, contributingStatistics);
296 
297         return contributingStatistics;
298     }

300     /**
301      * Computes aggregate summary statistics. This method can be used to combine statistics
302      * computed over partitions or subsamples - i.e., the StatisticalSummaryValues returned
303      * should contain the same values that would have been obtained by computing a single
304      * StatisticalSummary over the combined dataset.
305      * <p>
306      * Returns null if the collection is empty or null.
307      * </p>
308      *
309      * @param statistics collection of SummaryStatistics to aggregate
310      * @return summary statistics for the combined dataset
311      */
312     public static StatisticalSummaryValues aggregate(Collection<? extends StatisticalSummary> statistics) {
313         if (statistics == null) {
314             return null;
315         }
316         Iterator<? extends StatisticalSummary> iterator = statistics.iterator();
317         if (!iterator.hasNext()) {
318             return null;
319         }
320         StatisticalSummary current = iterator.next();
321         long n = current.getN();
322         double min = current.getMin();
323         double sum = current.getSum();
324         double max = current.getMax();
325         double var = current.getVariance();
326         double m2 = var * (n - 1d);
327         double mean = current.getMean();
328         while (iterator.hasNext()) {
329             current = iterator.next();
330             if (current.getMin() < min || Double.isNaN(min)) {
331                 min = current.getMin();
332             }
333             if (current.getMax() > max || Double.isNaN(max)) {
334                 max = current.getMax();
335             }
336             sum += current.getSum();
337             final double oldN = n;
338             final double curN = current.getN();
339             n += curN;
340             final double meanDiff = current.getMean() - mean;
341             mean = sum / n;
342             final double curM2 = current.getVariance() * (curN - 1d);
343             m2 = m2 + curM2 + meanDiff * meanDiff * oldN * curN / n;
344         }
345         final double variance;
346         if (n == 0) {
347             variance = Double.NaN;
348         } else if (n == 1) {
349             variance = 0d;
350         } else {
351             variance = m2 / (n - 1);
352         }
353         return new StatisticalSummaryValues(mean, variance, n, max, min, sum);
354     }

356     /**
357      * A SummaryStatistics that also forwards all values added to it to a second
358      * {@code SummaryStatistics} for aggregation.
359      *
360      * @since 2.0
361      */
362     private static class AggregatingSummaryStatistics extends SummaryStatistics {
363 
364         /**
365          * The serialization version of this class
366          */
367         private static final long serialVersionUID = 1L;
368 
369         /**
370          * An additional SummaryStatistics into which values added to these
371          * statistics (and possibly others) are aggregated
372          */
373         private final SummaryStatistics aggregateStatistics;
374 
375         /**
376          * Initializes a new AggregatingSummaryStatistics with the specified
377          * aggregate statistics object
378          *
379          * @param aggregateStatistics a {@code SummaryStatistics} into which
380          *      values added to this statistics object should be aggregated
381          */
382         AggregatingSummaryStatistics(SummaryStatistics aggregateStatistics) {
383             this.aggregateStatistics = aggregateStatistics;
384         }
385 
386         /**
387          * {@inheritDoc}.  This version adds the provided value to the configured
388          * aggregate after adding it to these statistics.
389          *
390          * @see SummaryStatistics#addValue(double)
391          */
392         @Override
393         public void addValue(double value) {
394             super.addValue(value);
395             synchronized (aggregateStatistics) {
396                 aggregateStatistics.addValue(value);
397             }
398         }
399 
400         /**
401          * Returns true iff <code>object</code> is a
402          * <code>SummaryStatistics</code> instance and all statistics have the
403          * same values as this.
404          * @param object the object to test equality against.
405          * @return true if object equals this
406          */
407         @Override
408         public boolean equals(Object object) {
409             if (object == this) {
410                 return true;
411             }
412             if (object instanceof AggregatingSummaryStatistics == false) {
413                 return false;
414             }
415             AggregatingSummaryStatistics stat = (AggregatingSummaryStatistics)object;
416             return super.equals(stat) &&
417                    aggregateStatistics.equals(stat.aggregateStatistics);
418         }
419 
420         /**
421          * Returns hash code based on values of statistics
422          * @return hash code
423          */
424         @Override
425         public int hashCode() {
426             return 123 + super.hashCode() + aggregateStatistics.hashCode();
427         }
428     }
