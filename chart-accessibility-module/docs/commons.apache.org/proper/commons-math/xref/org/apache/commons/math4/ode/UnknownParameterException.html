17  package org.apache.commons.math4.ode;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Exception to be thrown when a parameter is unknown.
24   *
25   * @since 3.1
26   */
27  public class UnknownParameterException extends MathIllegalArgumentException {
28  
29      /** Serializable version Id. */
30      private static final long serialVersionUID = 20120902L;
31  
32      /** Parameter name. */
33      private final String name;
34  
35      /**
36       * Construct an exception from the unknown parameter.
37       *
38       * @param name parameter name.
39       */
40      public UnknownParameterException(final String name) {
41          super(LocalizedFormats.UNKNOWN_PARAMETER, name);
42          this.name = name;
43      }
44  
45      /**
46       * @return the name of the unknown parameter.
47       */
48      public String getName() {
49          return name;
50      }
51  
52  }
