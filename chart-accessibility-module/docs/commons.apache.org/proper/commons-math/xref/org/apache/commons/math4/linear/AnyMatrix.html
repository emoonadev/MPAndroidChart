18  package org.apache.commons.math4.linear;
19  
20  
21  /**
22   * Interface defining very basic matrix operations.
23   * @since 2.0
24   */
25  public interface AnyMatrix {
26  
27      /**
28       * Is this a square matrix?
29       * @return true if the matrix is square (rowDimension = columnDimension)
30       */
31      boolean isSquare();
32  
33      /**
34       * Returns the number of rows in the matrix.
35       *
36       * @return rowDimension
37       */
38      int getRowDimension();
39  
40      /**
41       * Returns the number of columns in the matrix.
42       *
43       * @return columnDimension
44       */
45      int getColumnDimension();
46  
47  }
