17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.util.Localizable;
20  import org.apache.commons.math4.exception.util.LocalizedFormats;
21  
22  /**
23   * Base class for all exceptions that signal that the process
24   * throwing the exception is in a state that does not comply with
25   * the set of states that it is designed to be in.
26   *
27   * @since 2.2
28   */
29  public class MathIllegalStateException extends MathRuntimeException {
30      /** Serializable version Id. */
31      private static final long serialVersionUID = -6024911025449780478L;
32  
33      /**
34       * Simple constructor.
35       *
36       * @param pattern Message pattern explaining the cause of the error.
37       * @param args Arguments.
38       */
39      public MathIllegalStateException(Localizable pattern, Object ... args) {
40          super(pattern, args);
41      }
42  
43      /**
44       * Simple constructor.
45       *
46       * @param cause Root cause.
47       * @param pattern Message pattern explaining the cause of the error.
48       * @param args Arguments.
49       */
50      public MathIllegalStateException(Throwable cause,
51                                       Localizable pattern,
52                                       Object ... args) {
53          super(cause, pattern, args);
54      }
55  
56      /**
57       * Default constructor.
58       */
59      public MathIllegalStateException() {
60          this(LocalizedFormats.ILLEGAL_STATE);
61      }
62  
63  }
