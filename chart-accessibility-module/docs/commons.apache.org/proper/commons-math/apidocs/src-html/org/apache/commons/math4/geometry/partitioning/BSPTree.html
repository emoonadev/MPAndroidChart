027/** This class represent a Binary Space Partition tree.
028
029 * <p>BSP trees are an efficient way to represent space partitions and
030 * to associate attributes with each cell. Each node in a BSP tree
031 * represents a convex region which is partitioned in two convex
032 * sub-regions at each side of a cut hyperplane. The root tree
033 * contains the complete space.</p>
034
035 * <p>The main use of such partitions is to use a boolean attribute to
036 * define an inside/outside property, hence representing arbitrary
037 * polytopes (line segments in 1D, polygons in 2D and polyhedrons in
038 * 3D) and to operate on them.</p>
039
040 * <p>Another example would be to represent Voronoi tesselations, the
041 * attribute of each cell holding the defining point of the cell.</p>
042
043 * <p>The application-defined attributes are shared among copied
044 * instances and propagated to split parts. These attributes are not
045 * used by the BSP-tree algorithms themselves, so the application can
046 * use them for any purpose. Since the tree visiting method holds
047 * internal and leaf nodes differently, it is possible to use
048 * different classes for internal nodes attributes and leaf nodes
049 * attributes. This should be used with care, though, because if the
050 * tree is modified in any way after attributes have been set, some
051 * internal nodes may become leaf nodes and some leaf nodes may become
052 * internal nodes.</p>
053
054 * <p>One of the main sources for the development of this package was
055 * Bruce Naylor, John Amanatides and William Thibault paper <a
056 * href="http://www.cs.yorku.ca/~amana/research/bsptSetOp.pdf">Merging
057 * BSP Trees Yields Polyhedral Set Operations</a> Proc. Siggraph '90,
058 * Computer Graphics 24(4), August 1990, pp 115-124, published by the
059 * Association for Computing Machinery (ACM).</p>
060
061 * @param <S> Type of the space.
062
063 * @since 3.0
064 */
065public class BSPTree<S extends Space> {

103    /** Build a BSPTree from its underlying elements.
104     * <p>This method does <em>not</em> perform any verification on
105     * consistency of its arguments, it should therefore be used only
106     * when then caller knows what it is doing.</p>
107     * <p>This method is mainly useful to build trees
108     * bottom-up. Building trees top-down is realized with the help of
109     * method {@link #insertCut insertCut}.</p>
110     * @param cut cut sub-hyperplane for the tree
111     * @param plus plus side sub-tree
112     * @param minus minus side sub-tree
113     * @param attribute attribute associated with the node (may be null)
114     * @see #insertCut
115     */
116    public BSPTree(final SubHyperplane<S> cut, final BSPTree<S> plus, final BSPTree<S> minus,
117                   final Object attribute) {
118        this.cut       = cut;
119        this.plus      = plus;
120        this.minus     = minus;
121        this.parent    = null;
122        this.attribute = attribute;
123        plus.parent    = this;
124        minus.parent   = this;
125    }

127    /** Insert a cut sub-hyperplane in a node.
128     * <p>The sub-tree starting at this node will be completely
129     * overwritten. The new cut sub-hyperplane will be built from the
130     * intersection of the provided hyperplane with the cell. If the
131     * hyperplane does intersect the cell, the cell will have two
132     * children cells with {@code null} attributes on each side of
133     * the inserted cut sub-hyperplane. If the hyperplane does not
134     * intersect the cell then <em>no</em> cut hyperplane will be
135     * inserted and the cell will be changed to a leaf cell. The
136     * attribute of the node is never changed.</p>
137     * <p>This method is mainly useful when called on leaf nodes
138     * (i.e. nodes for which {@link #getCut getCut} returns
139     * {@code null}), in this case it provides a way to build a
140     * tree top-down (whereas the {@link #BSPTree(SubHyperplane,
141     * BSPTree, BSPTree, Object) 4 arguments constructor} is devoted to
142     * build trees bottom-up).</p>
143     * @param hyperplane hyperplane to insert, it will be chopped in
144     * order to fit in the cell defined by the parent nodes of the
145     * instance
146     * @return true if a cut sub-hyperplane has been inserted (i.e. if
147     * the cell now has two leaf child nodes)
148     * @see #BSPTree(SubHyperplane, BSPTree, BSPTree, Object)
149     */
150    public boolean insertCut(final Hyperplane<S> hyperplane) {
151
152        if (cut != null) {
153            plus.parent  = null;
154            minus.parent = null;
155        }
156
157        final SubHyperplane<S> chopped = fitToCell(hyperplane.wholeHyperplane());
158        if (chopped == null || chopped.isEmpty()) {
159            cut          = null;
160            plus         = null;
161            minus        = null;
162            return false;
163        }
164
165        cut          = chopped;
166        plus         = new BSPTree<S>();
167        plus.parent  = this;
168        minus        = new BSPTree<S>();
169        minus.parent = this;
170        return true;
171
172    }

182

184            return new BSPTree<S>(attribute);

187        return new BSPTree<S>(cut.copySelf(), plus.copySelf(), minus.copySelf(),

240    /** Visit the BSP tree nodes.
241     * @param visitor object visiting the tree nodes
242     */
243    public void visit(final BSPTreeVisitor<S> visitor) {
244        if (cut == null) {
245            visitor.visitLeafNode(this);
246        } else {
247            switch (visitor.visitOrder(this)) {
248            case PLUS_MINUS_SUB:
249                plus.visit(visitor);
250                minus.visit(visitor);
251                visitor.visitInternalNode(this);
252                break;
253            case PLUS_SUB_MINUS:
254                plus.visit(visitor);
255                visitor.visitInternalNode(this);
256                minus.visit(visitor);
257                break;
258            case MINUS_PLUS_SUB:
259                minus.visit(visitor);
260                plus.visit(visitor);
261                visitor.visitInternalNode(this);
262                break;
263            case MINUS_SUB_PLUS:
264                minus.visit(visitor);
265                visitor.visitInternalNode(this);
266                plus.visit(visitor);
267                break;
268            case SUB_PLUS_MINUS:
269                visitor.visitInternalNode(this);
270                plus.visit(visitor);
271                minus.visit(visitor);
272                break;
273            case SUB_MINUS_PLUS:
274                visitor.visitInternalNode(this);
275                minus.visit(visitor);
276                plus.visit(visitor);
277                break;
278            default:
279                throw new MathInternalError();
280            }
281
282        }
283    }

285    /** Fit a sub-hyperplane inside the cell defined by the instance.
286     * <p>Fitting is done by chopping off the parts of the
287     * sub-hyperplane that lie outside of the cell using the
288     * cut-hyperplanes of the parent nodes of the instance.</p>
289     * @param sub sub-hyperplane to fit
290     * @return a new sub-hyperplane, guaranteed to have no part outside
291     * of the instance cell
292     */
293    private SubHyperplane<S> fitToCell(final SubHyperplane<S> sub) {
294        SubHyperplane<S> s = sub;
295        for (BSPTree<S> tree = this; tree.parent != null && s != null; tree = tree.parent) {
296            if (tree == tree.parent.plus) {
297                s = s.split(tree.parent.cut.getHyperplane()).getPlus();
298            } else {
299                s = s.split(tree.parent.cut.getHyperplane()).getMinus();
300            }
301        }
302        return s;
303    }

305    /** Get the cell to which a point belongs.
306     * <p>If the returned cell is a leaf node the points belongs to the
307     * interior of the node, if the cell is an internal node the points
308     * belongs to the node cut sub-hyperplane.</p>
309     * @param point point to check
310     * @param tolerance tolerance below which points close to a cut hyperplane
311     * are considered to belong to the hyperplane itself
312     * @return the tree cell to which the point belongs
313     */
314    public BSPTree<S> getCell(final Point<S> point, final double tolerance) {
315
316        if (cut == null) {
317            return this;
318        }
319
320        // position of the point with respect to the cut hyperplane
321        final double offset = cut.getHyperplane().getOffset(point);
322
323        if (FastMath.abs(offset) < tolerance) {
324            return this;
325        } else if (offset <= 0) {
326            // point is on the minus side of the cut hyperplane
327            return minus.getCell(point, tolerance);
328        } else {
329            // point is on the plus side of the cut hyperplane
330            return plus.getCell(point, tolerance);
331        }
332
333    }
334
335    /** Get the cells whose cut sub-hyperplanes are close to the point.
336     * @param point point to check
337     * @param maxOffset offset below which a cut sub-hyperplane is considered
338     * close to the point (in absolute value)
339     * @return close cells (may be empty if all cut sub-hyperplanes are farther
340     * than maxOffset from the point)
341     */
342    public List<BSPTree<S>> getCloseCuts(final Point<S> point, final double maxOffset) {
343        final List<BSPTree<S>> close = new ArrayList<BSPTree<S>>();
344        recurseCloseCuts(point, maxOffset, close);
345        return close;
346    }
347
348    /** Get the cells whose cut sub-hyperplanes are close to the point.
349     * @param point point to check
350     * @param maxOffset offset below which a cut sub-hyperplane is considered
351     * close to the point (in absolute value)
352     * @param close list to fill
353     */
354    private void recurseCloseCuts(final Point<S> point, final double maxOffset,
355                                  final List<BSPTree<S>> close) {
356        if (cut != null) {
357
358            // position of the point with respect to the cut hyperplane
359            final double offset = cut.getHyperplane().getOffset(point);
360
361            if (offset < -maxOffset) {
362                // point is on the minus side of the cut hyperplane
363                minus.recurseCloseCuts(point, maxOffset, close);
364            } else if (offset > maxOffset) {
365                // point is on the plus side of the cut hyperplane
366                plus.recurseCloseCuts(point, maxOffset, close);
367            } else {
368                // point is close to the cut hyperplane
369                close.add(this);
370                minus.recurseCloseCuts(point, maxOffset, close);
371                plus.recurseCloseCuts(point, maxOffset, close);
372            }
373
374        }
375    }

377    /** Perform condensation on a tree.
378     * <p>The condensation operation is not recursive, it must be called
379     * explicitly from leaves to root.</p>
380     */
381    private void condense() {
382        if ((cut != null) && (plus.cut == null) && (minus.cut == null) &&
383            (((plus.attribute == null) && (minus.attribute == null)) ||
384             ((plus.attribute != null) && plus.attribute.equals(minus.attribute)))) {
385            attribute = (plus.attribute == null) ? minus.attribute : plus.attribute;
386            cut       = null;
387            plus      = null;
388            minus     = null;
389        }
390    }

392    /** Merge a BSP tree with the instance.
393     * <p>All trees are modified (parts of them are reused in the new
394     * tree), it is the responsibility of the caller to ensure a copy
395     * has been done before if any of the former tree should be
396     * preserved, <em>no</em> such copy is done here!</p>
397     * <p>The algorithm used here is directly derived from the one
398     * described in the Naylor, Amanatides and Thibault paper (section
399     * III, Binary Partitioning of a BSP Tree).</p>
400     * @param tree other tree to merge with the instance (will be
401     * <em>unusable</em> after the operation, as well as the
402     * instance itself)
403     * @param leafMerger object implementing the final merging phase
404     * (this is where the semantic of the operation occurs, generally
405     * depending on the attribute of the leaf node)
406     * @return a new tree, result of <code>instance &lt;op&gt;
407     * tree</code>, this value can be ignored if parentTree is not null
408     * since all connections have already been established
409     */
410    public BSPTree<S> merge(final BSPTree<S> tree, final LeafMerger<S> leafMerger) {
411        return merge(tree, leafMerger, null, false);
412    }
413
414    /** Merge a BSP tree with the instance.
415     * @param tree other tree to merge with the instance (will be
416     * <em>unusable</em> after the operation, as well as the
417     * instance itself)
418     * @param leafMerger object implementing the final merging phase
419     * (this is where the semantic of the operation occurs, generally
420     * depending on the attribute of the leaf node)
421     * @param parentTree parent tree to connect to (may be null)
422     * @param isPlusChild if true and if parentTree is not null, the
423     * resulting tree should be the plus child of its parent, ignored if
424     * parentTree is null
425     * @return a new tree, result of <code>instance &lt;op&gt;
426     * tree</code>, this value can be ignored if parentTree is not null
427     * since all connections have already been established
428     */
429    private BSPTree<S> merge(final BSPTree<S> tree, final LeafMerger<S> leafMerger,
430                             final BSPTree<S> parentTree, final boolean isPlusChild) {
431        if (cut == null) {
432            // cell/tree operation
433            return leafMerger.merge(this, tree, parentTree, isPlusChild, true);
434        } else if (tree.cut == null) {
435            // tree/cell operation
436            return leafMerger.merge(tree, this, parentTree, isPlusChild, false);
437        } else {
438            // tree/tree operation
439            final BSPTree<S> merged = tree.split(cut);
440            if (parentTree != null) {
441                merged.parent = parentTree;
442                if (isPlusChild) {
443                    parentTree.plus = merged;
444                } else {
445                    parentTree.minus = merged;
446                }
447            }
448
449            // merging phase
450            plus.merge(merged.plus, leafMerger, merged, true);
451            minus.merge(merged.minus, leafMerger, merged, false);
452            merged.condense();
453            if (merged.cut != null) {
454                merged.cut = merged.fitToCell(merged.cut.getHyperplane().wholeHyperplane());
455            }
456
457            return merged;
458
459        }
460    }

462    /** This interface gather the merging operations between a BSP tree
463     * leaf and another BSP tree.
464     * <p>As explained in Bruce Naylor, John Amanatides and William
465     * Thibault paper <a
466     * href="http://www.cs.yorku.ca/~amana/research/bsptSetOp.pdf">Merging
467     * BSP Trees Yields Polyhedral Set Operations</a>,
468     * the operations on {@link BSPTree BSP trees} can be expressed as a
469     * generic recursive merging operation where only the final part,
470     * when one of the operand is a leaf, is specific to the real
471     * operation semantics. For example, a tree representing a region
472     * using a boolean attribute to identify inside cells and outside
473     * cells would use four different objects to implement the final
474     * merging phase of the four set operations union, intersection,
475     * difference and symmetric difference (exclusive or).</p>
476     * @param <S> Type of the space.
477     */
478    public interface LeafMerger<S extends Space> {
479
480        /** Merge a leaf node and a tree node.
481         * <p>This method is called at the end of a recursive merging
482         * resulting from a {@code tree1.merge(tree2, leafMerger)}
483         * call, when one of the sub-trees involved is a leaf (i.e. when
484         * its cut-hyperplane is null). This is the only place where the
485         * precise semantics of the operation are required. For all upper
486         * level nodes in the tree, the merging operation is only a
487         * generic partitioning algorithm.</p>
488         * <p>Since the final operation may be non-commutative, it is
489         * important to know if the leaf node comes from the instance tree
490         * ({@code tree1}) or the argument tree
491         * ({@code tree2}). The third argument of the method is
492         * devoted to this. It can be ignored for commutative
493         * operations.</p>
494         * <p>The {@link BSPTree#insertInTree BSPTree.insertInTree} method
495         * may be useful to implement this method.</p>
496         * @param leaf leaf node (its cut hyperplane is guaranteed to be
497         * null)
498         * @param tree tree node (its cut hyperplane may be null or not)
499         * @param parentTree parent tree to connect to (may be null)
500         * @param isPlusChild if true and if parentTree is not null, the
501         * resulting tree should be the plus child of its parent, ignored if
502         * parentTree is null
503         * @param leafFromInstance if true, the leaf node comes from the
504         * instance tree ({@code tree1}) and the tree node comes from
505         * the argument tree ({@code tree2})
506         * @return the BSP tree resulting from the merging (may be one of
507         * the arguments)
508         */
509        BSPTree<S> merge(BSPTree<S> leaf, BSPTree<S> tree, BSPTree<S> parentTree,
510                         boolean isPlusChild, boolean leafFromInstance);
511
512    }

536    /** Split a BSP tree by an external sub-hyperplane.
537     * <p>Split a tree in two halves, on each side of the
538     * sub-hyperplane. The instance is not modified.</p>
539     * <p>The tree returned is not upward-consistent: despite all of its
540     * sub-trees cut sub-hyperplanes (including its own cut
541     * sub-hyperplane) are bounded to the current cell, it is <em>not</em>
542     * attached to any parent tree yet. This tree is intended to be
543     * later inserted into an higher level tree.</p>
544     * <p>The algorithm used here is the one given in Naylor, Amanatides
545     * and Thibault paper (section III, Binary Partitioning of a BSP
546     * Tree).</p>
547     * @param sub partitioning sub-hyperplane, must be already clipped
548     * to the convex region represented by the instance, will be used as
549     * the cut sub-hyperplane of the returned tree
550     * @return a tree having the specified sub-hyperplane as its cut
551     * sub-hyperplane, the two parts of the split instance as its two
552     * sub-trees and a null parent
553     */
554    public BSPTree<S> split(final SubHyperplane<S> sub) {
555
556        if (cut == null) {
557            return new BSPTree<S>(sub, copySelf(), new BSPTree<S>(attribute), null);
558        }
559
560        final Hyperplane<S> cHyperplane = cut.getHyperplane();
561        final Hyperplane<S> sHyperplane = sub.getHyperplane();
562        final SubHyperplane.SplitSubHyperplane<S> subParts = sub.split(cHyperplane);
563        switch (subParts.getSide()) {
564        case PLUS :
565        { // the partitioning sub-hyperplane is entirely in the plus sub-tree
566            final BSPTree<S> split = plus.split(sub);
567            if (cut.split(sHyperplane).getSide() == Side.PLUS) {
568                split.plus =
569                    new BSPTree<S>(cut.copySelf(), split.plus, minus.copySelf(), attribute);
570                split.plus.condense();
571                split.plus.parent = split;
572            } else {
573                split.minus =
574                    new BSPTree<S>(cut.copySelf(), split.minus, minus.copySelf(), attribute);
575                split.minus.condense();
576                split.minus.parent = split;
577            }
578            return split;
579        }
580        case MINUS :
581        { // the partitioning sub-hyperplane is entirely in the minus sub-tree
582            final BSPTree<S> split = minus.split(sub);
583            if (cut.split(sHyperplane).getSide() == Side.PLUS) {
584                split.plus =
585                    new BSPTree<S>(cut.copySelf(), plus.copySelf(), split.plus, attribute);
586                split.plus.condense();
587                split.plus.parent = split;
588            } else {
589                split.minus =
590                    new BSPTree<S>(cut.copySelf(), plus.copySelf(), split.minus, attribute);
591                split.minus.condense();
592                split.minus.parent = split;
593            }
594            return split;
595        }
596        case BOTH :
597        {
598            final SubHyperplane.SplitSubHyperplane<S> cutParts = cut.split(sHyperplane);
599            final BSPTree<S> split =
600                new BSPTree<S>(sub, plus.split(subParts.getPlus()), minus.split(subParts.getMinus()),
601                               null);
602            split.plus.cut          = cutParts.getPlus();
603            split.minus.cut         = cutParts.getMinus();
604            final BSPTree<S> tmp    = split.plus.minus;
605            split.plus.minus        = split.minus.plus;
606            split.plus.minus.parent = split.plus;
607            split.minus.plus        = tmp;
608            split.minus.plus.parent = split.minus;
609            split.plus.condense();
610            split.minus.condense();
611            return split;
612        }
613        default :
614            return cHyperplane.sameOrientationAs(sHyperplane) ?
615                   new BSPTree<S>(sub, plus.copySelf(),  minus.copySelf(), attribute) :
616                   new BSPTree<S>(sub, minus.copySelf(), plus.copySelf(),  attribute);
617        }
618
619    }

621    /** Insert the instance into another tree.
622     * <p>The instance itself is modified so its former parent should
623     * not be used anymore.</p>
624     * @param parentTree parent tree to connect to (may be null)
625     * @param isPlusChild if true and if parentTree is not null, the
626     * resulting tree should be the plus child of its parent, ignored if
627     * parentTree is null
628     * @param vanishingHandler handler to use for handling very rare corner
629     * cases of vanishing cut sub-hyperplanes in internal nodes during merging
630     * @see LeafMerger
631     * @since 3.4
632     */
633    public void insertInTree(final BSPTree<S> parentTree, final boolean isPlusChild,
634                             final VanishingCutHandler<S> vanishingHandler) {
635
636        // set up parent/child links
637        parent = parentTree;
638        if (parentTree != null) {
639            if (isPlusChild) {
640                parentTree.plus = this;
641            } else {
642                parentTree.minus = this;
643            }
644        }
645
646        // make sure the inserted tree lies in the cell defined by its parent nodes
647        if (cut != null) {
648
649            // explore the parent nodes from here towards tree root
650            for (BSPTree<S> tree = this; tree.parent != null; tree = tree.parent) {
651
652                // this is an hyperplane of some parent node
653                final Hyperplane<S> hyperplane = tree.parent.cut.getHyperplane();
654
655                // chop off the parts of the inserted tree that extend
656                // on the wrong side of this parent hyperplane
657                if (tree == tree.parent.plus) {
658                    cut = cut.split(hyperplane).getPlus();
659                    plus.chopOffMinus(hyperplane, vanishingHandler);
660                    minus.chopOffMinus(hyperplane, vanishingHandler);
661                } else {
662                    cut = cut.split(hyperplane).getMinus();
663                    plus.chopOffPlus(hyperplane, vanishingHandler);
664                    minus.chopOffPlus(hyperplane, vanishingHandler);
665                }
666
667                if (cut == null) {
668                    // the cut sub-hyperplane has vanished
669                    final BSPTree<S> fixed = vanishingHandler.fixNode(this);
670                    cut       = fixed.cut;
671                    plus      = fixed.plus;
672                    minus     = fixed.minus;
673                    attribute = fixed.attribute;
674                    if (cut == null) {
675                        break;
676                    }
677                }
678
679            }
680
681            // since we may have drop some parts of the inserted tree,
682            // perform a condensation pass to keep the tree structure simple
683            condense();
684
685        }
686
687    }

689    /** Prune a tree around a cell.
690     * <p>
691     * This method can be used to extract a convex cell from a tree.
692     * The original cell may either be a leaf node or an internal node.
693     * If it is an internal node, it's subtree will be ignored (i.e. the
694     * extracted cell will be a leaf node in all cases). The original
695     * tree to which the original cell belongs is not touched at all,
696     * a new independent tree will be built.
697     * </p>
698     * @param cellAttribute attribute to set for the leaf node
699     * corresponding to the initial instance cell
700     * @param otherLeafsAttributes attribute to set for the other leaf
701     * nodes
702     * @param internalAttributes attribute to set for the internal nodes
703     * @return a new tree (the original tree is left untouched) containing
704     * a single branch with the cell as a leaf node, and other leaf nodes
705     * as the remnants of the pruned branches
706     * @since 3.3
707     */
708    public BSPTree<S> pruneAroundConvexCell(final Object cellAttribute,
709                                            final Object otherLeafsAttributes,
710                                            final Object internalAttributes) {
711
712        // build the current cell leaf
713        BSPTree<S> tree = new BSPTree<S>(cellAttribute);
714
715        // build the pruned tree bottom-up
716        for (BSPTree<S> current = this; current.parent != null; current = current.parent) {
717            final SubHyperplane<S> parentCut = current.parent.cut.copySelf();
718            final BSPTree<S>       sibling   = new BSPTree<S>(otherLeafsAttributes);
719            if (current == current.parent.plus) {
720                tree = new BSPTree<S>(parentCut, tree, sibling, internalAttributes);
721            } else {
722                tree = new BSPTree<S>(parentCut, sibling, tree, internalAttributes);
723            }
724        }
725
726        return tree;
727
728    }

731     * <p>The instance is modified in place, all the parts that are on
732     * the minus side of the chopping hyperplane are discarded, only the
733     * parts on the plus side remain.</p>
734     * @param hyperplane chopping hyperplane
735     * @param vanishingHandler handler to use for handling very rare corner
736     * cases of vanishing cut sub-hyperplanes in internal nodes during merging
737     */

741            cut = cut.split(hyperplane).getPlus();

748                cut       = fixed.cut;
749                plus      = fixed.plus;
750                minus     = fixed.minus;
751                attribute = fixed.attribute;
752            }
753
754        }
755    }
