376    /** Source English format. */
377    private final String sourceFormat;
378
379    /** Simple constructor.
380     * @param sourceFormat source English format to use when no
381     * localized version is available
382     */
383    LocalizedFormats(final String sourceFormat) {
384        this.sourceFormat = sourceFormat;
385    }
386
387    /** {@inheritDoc} */
388    @Override
389    public String getSourceString() {
390        return sourceFormat;
391    }
392
393    /** {@inheritDoc} */
394    @Override
395    public String getLocalizedString(final Locale locale) {
396        try {
397            final String path = LocalizedFormats.class.getName().replaceAll("\\.", "/");
398            ResourceBundle bundle =
399                    ResourceBundle.getBundle("assets/" + path, locale);
400            if (bundle.getLocale().getLanguage().equals(locale.getLanguage())) {
401                // the value of the resource is the translated format
402                return bundle.getString(toString());
403            }
404
405        } catch (MissingResourceException mre) { // NOPMD
406            // do nothing here
407        }
408
409        // either the locale is not supported or the resource is unknown
410        // don't translate and fall back to using the source format
411        return sourceFormat;
412
413    }
