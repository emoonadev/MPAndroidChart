045/**
046 * Base class managing common boilerplate for all integrators.
047 * @param <T> the type of the field elements
048 * @since 3.6
049 */
050public abstract class AbstractFieldIntegrator<T extends RealFieldElement<T>> implements FirstOrderFieldIntegrator<T> {
051
052    /** Default relative accuracy. */
053    private static final double DEFAULT_RELATIVE_ACCURACY = 1e-14;
054
055    /** Default function value accuracy. */
056    private static final double DEFAULT_FUNCTION_VALUE_ACCURACY = 1e-15;
057
058    /** Step handler. */
059    private Collection<FieldStepHandler<T>> stepHandlers;
060
061    /** Current step start. */
062    private FieldODEStateAndDerivative<T> stepStart;
063
064    /** Current stepsize. */
065    private T stepSize;
066
067    /** Indicator for last step. */
068    private boolean isLastStep;
069
070    /** Indicator that a state or derivative reset was triggered by some event. */
071    private boolean resetOccurred;
072
073    /** Field to which the time and state vector elements belong. */
074    private final Field<T> field;
075
076    /** Events states. */
077    private Collection<FieldEventState<T>> eventsStates;
078
079    /** Initialization indicator of events states. */
080    private boolean statesInitialized;
081
082    /** Name of the method. */
083    private final String name;
084
085    /** Counter for number of evaluations. */
086    private IntegerSequence.Incrementor evaluations;

091    /** Build an instance.
092     * @param field field to which the time and state vector elements belong
093     * @param name name of the method
094     */
095    protected AbstractFieldIntegrator(final Field<T> field, final String name) {
096        this.field        = field;
097        this.name         = name;
098        stepHandlers      = new ArrayList<FieldStepHandler<T>>();
099        stepStart         = null;
100        stepSize          = null;
101        eventsStates      = new ArrayList<FieldEventState<T>>();
102        statesInitialized = false;
103        evaluations       = IntegerSequence.Incrementor.create().withMaximalCount(Integer.MAX_VALUE);
104    }

137    /** {@inheritDoc} */
138    @Override
139    public void addEventHandler(final FieldEventHandler<T> handler,
140                                final double maxCheckInterval,
141                                final double convergence,
142                                final int maxIterationCount) {
143        addEventHandler(handler, maxCheckInterval, convergence,
144                        maxIterationCount,

146                                                                  field.getZero().add(convergence),
147                                                                  field.getZero().add(DEFAULT_FUNCTION_VALUE_ACCURACY),
148                                                                  5));
149    }

151    /** {@inheritDoc} */
152    @Override
153    public void addEventHandler(final FieldEventHandler<T> handler,
154                                final double maxCheckInterval,
155                                final double convergence,
156                                final int maxIterationCount,
157                                final BracketedRealFieldUnivariateSolver<T> solver) {
158        eventsStates.add(new FieldEventState<T>(handler, maxCheckInterval, field.getZero().add(convergence),
159                                                maxIterationCount, solver));

162    /** {@inheritDoc} */
163    @Override

166        for (FieldEventState<T> state : eventsStates) {

208    /** Prepare the start of an integration.
209     * @param eqn equations to integrate
210     * @param t0 start value of the independent <i>time</i> variable
211     * @param y0 array containing the start value of the state vector
212     * @param t target time for the integration
213     * @return initial state with derivatives added
214     */
215    protected FieldODEStateAndDerivative<T> initIntegration(final FieldExpandableODE<T> eqn,
216                                                            final T t0, final T[] y0, final T t) {
217
218        this.equations = eqn;
219        evaluations    = evaluations.withStart(0);
220
221        // initialize ODE
222        eqn.init(t0, y0, t);
223
224        // set up derivatives of initial state
225        final T[] y0Dot = computeDerivatives(t0, y0);
226        final FieldODEStateAndDerivative<T> state0 = new FieldODEStateAndDerivative<T>(t0, y0, y0Dot);
227
228        // initialize event handlers
229        for (final FieldEventState<T> state : eventsStates) {
230            state.getEventHandler().init(state0, t);
231        }
232
233        // initialize step handlers
234        for (FieldStepHandler<T> handler : stepHandlers) {
235            handler.init(state0, t);
236        }
237
238        setStateInitialized(false);
239
240        return state0;
241
242    }
243
244    /** Get the differential equations to integrate.
245     * @return differential equations to integrate

247    protected FieldExpandableODE<T> getEquations() {
248        return equations;
249    }

255        return evaluations;

258    /** Compute the derivatives and check the number of evaluations.
259     * @param t current value of the independent <I>time</I> variable
260     * @param y array containing the current value of the state vector
261     * @return state completed with derivatives
262     * @exception DimensionMismatchException if arrays dimensions do not match equations settings
263     * @exception MaxCountExceededException if the number of functions evaluations is exceeded
264     * @exception NullPointerException if the ODE equations have not been set (i.e. if this method
265     * is called outside of a call to {@link #integrate(FieldExpandableODE, FieldODEState,
266     * RealFieldElement) integrate}
267     */
268    public T[] computeDerivatives(final T t, final T[] y)
269        throws DimensionMismatchException, MaxCountExceededException, NullPointerException {
270        evaluations.increment();
271        return equations.computeDerivatives(t, y);
272    }

274    /** Set the stateInitialized flag.
275     * <p>This method must be called by integrators with the value
276     * {@code false} before they start integration, so a proper lazy
277     * initialization is done automatically on the first step.</p>
278     * @param stateInitialized new value for the flag
279     */
280    protected void setStateInitialized(final boolean stateInitialized) {
281        this.statesInitialized = stateInitialized;
282    }

284    /** Accept a step, triggering events and step handlers.
285     * @param interpolator step interpolator
286     * @param tEnd final integration time
287     * @return state at end of step
288     * @exception MaxCountExceededException if the interpolator throws one because
289     * the number of functions evaluations is exceeded
290     * @exception NoBracketingException if the location of an event cannot be bracketed
291     * @exception DimensionMismatchException if arrays dimensions do not match equations settings
292     */
293    protected FieldODEStateAndDerivative<T> acceptStep(final AbstractFieldStepInterpolator<T> interpolator,
294                                                       final T tEnd)
295        throws MaxCountExceededException, DimensionMismatchException, NoBracketingException {
296
297            FieldODEStateAndDerivative<T> previousState = interpolator.getGlobalPreviousState();
298            final FieldODEStateAndDerivative<T> currentState = interpolator.getGlobalCurrentState();
299
300            // initialize the events states if needed
301            if (! statesInitialized) {
302                for (FieldEventState<T> state : eventsStates) {
303                    state.reinitializeBegin(interpolator);
304                }
305                statesInitialized = true;
306            }
307
308            // search for next events that may occur during the step
309            final int orderingSign = interpolator.isForward() ? +1 : -1;
310            SortedSet<FieldEventState<T>> occurringEvents = new TreeSet<FieldEventState<T>>(new Comparator<FieldEventState<T>>() {
311
312                /** {@inheritDoc} */
313                @Override
314                public int compare(FieldEventState<T> es0, FieldEventState<T> es1) {
315                    return orderingSign * Double.compare(es0.getEventTime().getReal(), es1.getEventTime().getReal());
316                }
317
318            });
319
320            for (final FieldEventState<T> state : eventsStates) {
321                if (state.evaluateStep(interpolator)) {
322                    // the event occurs during the current step
323                    occurringEvents.add(state);
324                }
325            }
326
327            AbstractFieldStepInterpolator<T> restricted = interpolator;
328            while (!occurringEvents.isEmpty()) {
329
330                // handle the chronologically first event
331                final Iterator<FieldEventState<T>> iterator = occurringEvents.iterator();
332                final FieldEventState<T> currentEvent = iterator.next();
333                iterator.remove();
334
335                // get state at event time
336                final FieldODEStateAndDerivative<T> eventState = restricted.getInterpolatedState(currentEvent.getEventTime());
337
338                // restrict the interpolator to the first part of the step, up to the event
339                restricted = restricted.restrictStep(previousState, eventState);
340
341                // advance all event states to current time
342                for (final FieldEventState<T> state : eventsStates) {
343                    state.stepAccepted(eventState);
344                    isLastStep = isLastStep || state.stop();
345                }
346
347                // handle the first part of the step, up to the event
348                for (final FieldStepHandler<T> handler : stepHandlers) {
349                    handler.handleStep(restricted, isLastStep);
350                }
351
352                if (isLastStep) {
353                    // the event asked to stop integration
354                    return eventState;
355                }
356
357                FieldODEState<T> newState = null;
358                resetOccurred = false;
359                for (final FieldEventState<T> state : eventsStates) {
360                    newState = state.reset(eventState);
361                    if (newState != null) {
362                        // some event handler has triggered changes that
363                        // invalidate the derivatives, we need to recompute them
364                        final T[] y    = equations.getMapper().mapState(newState);
365                        final T[] yDot = computeDerivatives(newState.getTime(), y);
366                        resetOccurred = true;
367                        return equations.getMapper().mapStateAndDerivative(newState.getTime(), y, yDot);
368                    }
369                }
370
371                // prepare handling of the remaining part of the step
372                previousState = eventState;
373                restricted = restricted.restrictStep(eventState, currentState);
374
375                // check if the same event occurs again in the remaining part of the step
376                if (currentEvent.evaluateStep(restricted)) {
377                    // the event occurs during the current step
378                    occurringEvents.add(currentEvent);
379                }
380
381            }
382
383            // last part of the step, after the last event
384            for (final FieldEventState<T> state : eventsStates) {
385                state.stepAccepted(currentState);
386                isLastStep = isLastStep || state.stop();
387            }
388            isLastStep = isLastStep || currentState.getTime().subtract(tEnd).abs().getReal() <= FastMath.ulp(tEnd.getReal());
389
390            // handle the remaining part of the step, after all events if any
391            for (FieldStepHandler<T> handler : stepHandlers) {
392                handler.handleStep(restricted, isLastStep);
393            }
394
395            return currentState;
396
397    }

399    /** Check the integration span.
400     * @param eqn set of differential equations
401     * @param t target time for the integration
402     * @exception NumberIsTooSmallException if integration span is too small
403     * @exception DimensionMismatchException if adaptive step size integrators
404     * tolerance arrays dimensions are not compatible with equations settings
405     */
406    protected void sanityChecks(final FieldODEState<T> eqn, final T t)
407        throws NumberIsTooSmallException, DimensionMismatchException {
408
409        final double threshold = 1000 * FastMath.ulp(FastMath.max(FastMath.abs(eqn.getTime().getReal()),
410                                                                  FastMath.abs(t.getReal())));
411        final double dt = eqn.getTime().subtract(t).abs().getReal();
412        if (dt <= threshold) {
413            throw new NumberIsTooSmallException(LocalizedFormats.TOO_SMALL_INTEGRATION_INTERVAL,
414                                                dt, threshold, false);
415        }
416
417    }
