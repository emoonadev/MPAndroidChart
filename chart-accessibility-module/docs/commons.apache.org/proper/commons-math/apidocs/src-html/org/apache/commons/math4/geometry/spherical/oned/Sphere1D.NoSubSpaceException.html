018package org.apache.commons.math4.geometry.spherical.oned;
019
020import java.io.Serializable;
021
022import org.apache.commons.math4.exception.MathUnsupportedOperationException;
023import org.apache.commons.math4.exception.util.LocalizedFormats;
024import org.apache.commons.math4.geometry.Space;
025
026/**
027 * This class implements a one-dimensional sphere (i.e. a circle).
028 * <p>
029 * We use here the topologists definition of the 1-sphere (see
030 * <a href="http://mathworld.wolfram.com/Sphere.html">Sphere</a> on
031 * MathWorld), i.e. the 1-sphere is the one-dimensional closed curve
032 * defined in 2D as x<sup>2</sup>+y<sup>2</sup>=1.
033 * </p>
034 * @since 3.3
035 */
036public class Sphere1D implements Serializable, Space {
037
038    /** Serializable version identifier. */
039    private static final long serialVersionUID = 20131218L;
040
041    /** Private constructor for the singleton.
042     */
043    private Sphere1D() {
044    }
045
046    /** Get the unique instance.
047     * @return the unique instance
048     */
049    public static Sphere1D getInstance() {
050        return LazyHolder.INSTANCE;
051    }
052
053    /** {@inheritDoc} */
054    @Override
055    public int getDimension() {
056        return 1;
057    }
058
059    /** {@inheritDoc}
060     * <p>
061     * As the 1-dimension sphere does not have proper sub-spaces,
062     * this method always throws a {@link NoSubSpaceException}
063     * </p>
064     * @return nothing
065     * @throws NoSubSpaceException in all cases
066     */
067    @Override
068    public Space getSubSpace() throws NoSubSpaceException {
069        throw new NoSubSpaceException();
070    }
071
072    // CHECKSTYLE: stop HideUtilityClassConstructor
073    /** Holder for the instance.
074     * <p>We use here the Initialization On Demand Holder Idiom.</p>
075     */
076    private static class LazyHolder {
077        /** Cached field instance. */
078        private static final Sphere1D INSTANCE = new Sphere1D();
079    }
080    // CHECKSTYLE: resume HideUtilityClassConstructor
081
082    /** Handle deserialization of the singleton.
083     * @return the singleton instance
084     */
085    private Object readResolve() {
086        // return the singleton instance
087        return LazyHolder.INSTANCE;
088    }
089
090    /** Specialized exception for inexistent sub-space.
091     * <p>
092     * This exception is thrown when attempting to get the sub-space of a one-dimensional space
093     * </p>
094     */
095    public static class NoSubSpaceException extends MathUnsupportedOperationException {
096
097        /** Serializable UID. */
098        private static final long serialVersionUID = 20140225L;
099
100        /** Simple constructor.
101         */
102        public NoSubSpaceException() {
103            super(LocalizedFormats.NOT_SUPPORTED_IN_DIMENSION_N, 1);
104        }
105
106    }
107
108}
