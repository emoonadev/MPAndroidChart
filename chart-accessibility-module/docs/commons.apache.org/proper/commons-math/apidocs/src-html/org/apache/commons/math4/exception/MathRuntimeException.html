017package org.apache.commons.math4.exception;
018
019import org.apache.commons.math4.exception.util.ExceptionContext;
020import org.apache.commons.math4.exception.util.ExceptionContextProvider;
021import org.apache.commons.math4.exception.util.Localizable;
022
023/**
024 * As of release 4.0, all exceptions thrown by the Commons Math code (except
025 * {@link NullArgumentException}) inherit from this class.
026 * In most cases, this class should not be instantiated directly: it should
027 * serve as a base class for implementing exception classes that describe a
028 * specific "problem".
029 *
030 * @since 3.1
031 */
032public class MathRuntimeException extends RuntimeException
033    implements ExceptionContextProvider {
034    /** Serializable version Id. */
035    private static final long serialVersionUID = 20120926L;
036    /** Context. */
037    private final ExceptionContext context;
038
039    /**
040     * @param pattern Message pattern explaining the cause of the error.
041     * @param args Arguments.
042     */
043    public MathRuntimeException(Localizable pattern,
044                                Object ... args) {
045        context = new ExceptionContext(this);
046        context.addMessage(pattern, args);
047    }
048
049    /**
050     * @param cause Root cause.
051     * @param pattern Message pattern explaining the cause of the error.
052     * @param args Arguments.
053     * @since 4.0
054     */
055    public MathRuntimeException(Throwable cause,
056                                Localizable pattern,
057                                Object ... args) {
058        super(cause);
059        context = new ExceptionContext(this);
060        context.addMessage(pattern, args);
061    }
062
063    /** {@inheritDoc} */
064    @Override
065    public ExceptionContext getContext() {
066        return context;
067    }
068
069    /** {@inheritDoc} */
070    @Override
071    public String getMessage() {
072        return context.getMessage();
073    }
074
075    /** {@inheritDoc} */
076    @Override
077    public String getLocalizedMessage() {
078        return context.getLocalizedMessage();
079    }
080}
