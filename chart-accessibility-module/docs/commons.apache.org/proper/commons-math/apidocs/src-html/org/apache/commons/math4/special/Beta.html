024/**
025 * <p>
026 * This is a utility class that provides computation methods related to the
027 * Beta family of functions.
028 * </p>
029 * <p>
030 * Implementation of {@link #logBeta(double, double)} is based on the
031 * algorithms described in
032 * <ul>
033 * <li><a href="http://dx.doi.org/10.1145/22721.23109">Didonato and Morris
034 *     (1986)</a>, <em>Computation of the Incomplete Gamma Function Ratios
035 *     and their Inverse</em>, TOMS 12(4), 377-393,</li>
036 * <li><a href="http://dx.doi.org/10.1145/131766.131776">Didonato and Morris
037 *     (1992)</a>, <em>Algorithm 708: Significant Digit Computation of the
038 *     Incomplete Beta Function Ratios</em>, TOMS 18(3), 360-373,</li>
039 * </ul>
040 * and implemented in the
041 * <a href="http://www.dtic.mil/docs/citations/ADA476840">NSWC Library of Mathematical Functions</a>,
042 * available
043 * <a href="http://www.ualberta.ca/CNS/RESEARCH/Software/NumericalNSWC/site.html">here</a>.
044 * This library is "approved for public release", and the
045 * <a href="http://www.dtic.mil/dtic/pdf/announcements/CopyrightGuidance.pdf">Copyright guidance</a>
046 * indicates that unless otherwise stated in the code, all FORTRAN functions in
047 * this library are license free. Since no such notice appears in the code these
048 * functions can safely be ported to Commons-Math.
049 * </p>
050 */
051public class Beta {
052    /** Maximum allowed numerical error. */
053    private static final double DEFAULT_EPSILON = 1E-14;
054
055    /** The constant value of ½log 2π. */
056    private static final double HALF_LOG_TWO_PI = .9189385332046727;

058    /**
059     * <p>
060     * The coefficients of the series expansion of the Δ function. This function
061     * is defined as follows
062     * </p>
063     * <center>Δ(x) = log Γ(x) - (x - 0.5) log a + a - 0.5 log 2π,</center>
064     * <p>
065     * see equation (23) in Didonato and Morris (1992). The series expansion,
066     * which applies for x ≥ 10, reads
067     * </p>
068     * <pre>
069     *                 14
070     *                ====
071     *             1  \                2 n
072     *     Δ(x) = ---  >    d  (10 / x)
073     *             x  /      n
074     *                ====
075     *                n = 0
076     * <pre>
077     */
078    private static final double[] DELTA = {
079        .833333333333333333333333333333E-01,
080        -.277777777777777777777777752282E-04,
081        .793650793650793650791732130419E-07,
082        -.595238095238095232389839236182E-09,
083        .841750841750832853294451671990E-11,
084        -.191752691751854612334149171243E-12,
085        .641025640510325475730918472625E-14,
086        -.295506514125338232839867823991E-15,
087        .179643716359402238723287696452E-16,
088        -.139228964661627791231203060395E-17,
089        .133802855014020915603275339093E-18,
090        -.154246009867966094273710216533E-19,
091        .197701992980957427278370133333E-20,
092        -.234065664793997056856992426667E-21,
093        .171348014966398575409015466667E-22
094    };

155    /**
156     * Returns the regularized beta function I(x, a, b).
157     *
158     * The implementation of this method is based on:
159     * <ul>
160     * <li>
161     * <a href="http://mathworld.wolfram.com/RegularizedBetaFunction.html">
162     * Regularized Beta Function</a>.</li>
163     * <li>
164     * <a href="http://functions.wolfram.com/06.21.10.0001.01">
165     * Regularized Beta Function</a>.</li>
166     * </ul>
167     *
168     * @param x the value.
169     * @param a Parameter {@code a}.
170     * @param b Parameter {@code b}.
171     * @param epsilon When the absolute value of the nth item in the
172     * series is less than epsilon the approximation ceases to calculate
173     * further elements in the series.
174     * @param maxIterations Maximum number of "iterations" to complete.
175     * @return the regularized beta function I(x, a, b)
176     * @throws org.apache.commons.math4.exception.MaxCountExceededException
177     * if the algorithm fails to converge.
178     */
179    public static double regularizedBeta(double x,
180                                         final double a, final double b,
181                                         double epsilon, int maxIterations) {
182        double ret;
183
184        if (Double.isNaN(x) ||
185            Double.isNaN(a) ||
186            Double.isNaN(b) ||
187            x < 0 ||
188            x > 1 ||
189            a <= 0 ||
190            b <= 0) {
191            ret = Double.NaN;
192        } else if (x > (a + 1) / (2 + b + a) &&
193                   1 - x <= (b + 1) / (2 + b + a)) {
194            ret = 1 - regularizedBeta(1 - x, b, a, epsilon, maxIterations);
195        } else {
196            ContinuedFraction fraction = new ContinuedFraction() {
197
198                /** {@inheritDoc} */
199                @Override
200                protected double getB(int n, double x) {
201                    double ret;
202                    double m;
203                    if (n % 2 == 0) { // even
204                        m = n / 2.0;
205                        ret = (m * (b - m) * x) /
206                            ((a + (2 * m) - 1) * (a + (2 * m)));
207                    } else {
208                        m = (n - 1.0) / 2.0;
209                        ret = -((a + m) * (a + b + m) * x) /
210                                ((a + (2 * m)) * (a + (2 * m) + 1.0));
211                    }
212                    return ret;
213                }
214
215                /** {@inheritDoc} */
216                @Override
217                protected double getA(int n, double x) {
218                    return 1.0;
219                }
220            };
221            ret = FastMath.exp((a * FastMath.log(x)) + (b * FastMath.log1p(-x)) -
222                FastMath.log(a) - logBeta(a, b)) *
223                1.0 / fraction.evaluate(x, epsilon, maxIterations);
224        }
225
226        return ret;
227    }

229    /**
230     * Returns the value of log Γ(a + b) for 1 ≤ a, b ≤ 2. Based on the
231     * <em>NSWC Library of Mathematics Subroutines</em> double precision
232     * implementation, {@code DGSMLN}. In {@code BetaTest.testLogGammaSum()},
233     * this private method is accessed through reflection.
234     *
235     * @param a First argument.
236     * @param b Second argument.
237     * @return the value of {@code log(Gamma(a + b))}.
238     * @throws OutOfRangeException if {@code a} or {@code b} is lower than
239     * {@code 1.0} or greater than {@code 2.0}.
240     */
241    private static double logGammaSum(final double a, final double b)
242        throws OutOfRangeException {
243
244        if ((a < 1.0) || (a > 2.0)) {
245            throw new OutOfRangeException(a, 1.0, 2.0);
246        }
247        if ((b < 1.0) || (b > 2.0)) {
248            throw new OutOfRangeException(b, 1.0, 2.0);
249        }
250
251        final double x = (a - 1.0) + (b - 1.0);
252        if (x <= 0.5) {
253            return Gamma.logGamma1p(1.0 + x);
254        } else if (x <= 1.5) {
255            return Gamma.logGamma1p(x) + FastMath.log1p(x);
256        } else {
257            return Gamma.logGamma1p(x - 1.0) + FastMath.log(x * (1.0 + x));
258        }
259    }

261    /**
262     * Returns the value of log[Γ(b) / Γ(a + b)] for a ≥ 0 and b ≥ 10. Based on
263     * the <em>NSWC Library of Mathematics Subroutines</em> double precision
264     * implementation, {@code DLGDIV}. In
265     * {@code BetaTest.testLogGammaMinusLogGammaSum()}, this private method is
266     * accessed through reflection.
267     *
268     * @param a First argument.
269     * @param b Second argument.
270     * @return the value of {@code log(Gamma(b) / Gamma(a + b))}.
271     * @throws NumberIsTooSmallException if {@code a < 0.0} or {@code b < 10.0}.
272     */
273    private static double logGammaMinusLogGammaSum(final double a,
274                                                   final double b)
275        throws NumberIsTooSmallException {
276
277        if (a < 0.0) {
278            throw new NumberIsTooSmallException(a, 0.0, true);
279        }
280        if (b < 10.0) {
281            throw new NumberIsTooSmallException(b, 10.0, true);
282        }
283
284        /*
285         * d = a + b - 0.5
286         */
287        final double d;
288        final double w;
289        if (a <= b) {
290            d = b + (a - 0.5);
291            w = deltaMinusDeltaSum(a, b);
292        } else {
293            d = a + (b - 0.5);
294            w = deltaMinusDeltaSum(b, a);
295        }
296
297        final double u = d * FastMath.log1p(a / b);
298        final double v = a * (FastMath.log(b) - 1.0);
299
300        return u <= v ? (w - u) - v : (w - v) - u;
301    }

303    /**
304     * Returns the value of Δ(b) - Δ(a + b), with 0 ≤ a ≤ b and b ≥ 10. Based
305     * on equations (26), (27) and (28) in Didonato and Morris (1992).
306     *
307     * @param a First argument.
308     * @param b Second argument.
309     * @return the value of {@code Delta(b) - Delta(a + b)}
310     * @throws OutOfRangeException if {@code a < 0} or {@code a > b}
311     * @throws NumberIsTooSmallException if {@code b < 10}
312     */
313    private static double deltaMinusDeltaSum(final double a,
314                                             final double b)
315        throws OutOfRangeException, NumberIsTooSmallException {
316
317        if ((a < 0) || (a > b)) {
318            throw new OutOfRangeException(a, 0, b);
319        }
320        if (b < 10) {
321            throw new NumberIsTooSmallException(b, 10, true);
322        }
323
324        final double h = a / b;
325        final double p = h / (1.0 + h);
326        final double q = 1.0 / (1.0 + h);
327        final double q2 = q * q;
328        /*
329         * s[i] = 1 + q + ... - q**(2 * i)
330         */
331        final double[] s = new double[DELTA.length];
332        s[0] = 1.0;
333        for (int i = 1; i < s.length; i++) {
334            s[i] = 1.0 + (q + q2 * s[i - 1]);
335        }
336        /*
337         * w = Delta(b) - Delta(a + b)
338         */
339        final double sqrtT = 10.0 / b;
340        final double t = sqrtT * sqrtT;
341        double w = DELTA[DELTA.length - 1] * s[s.length - 1];
342        for (int i = DELTA.length - 2; i >= 0; i--) {
343            w = t * w + DELTA[i] * s[i];
344        }
345        return w * p / b;
346    }

348    /**
349     * Returns the value of Δ(p) + Δ(q) - Δ(p + q), with p, q ≥ 10. Based on
350     * the <em>NSWC Library of Mathematics Subroutines</em> double precision
351     * implementation, {@code DBCORR}. In
352     * {@code BetaTest.testSumDeltaMinusDeltaSum()}, this private method is
353     * accessed through reflection.
354     *
355     * @param p First argument.
356     * @param q Second argument.
357     * @return the value of {@code Delta(p) + Delta(q) - Delta(p + q)}.
358     * @throws NumberIsTooSmallException if {@code p < 10.0} or {@code q < 10.0}.
359     */
360    private static double sumDeltaMinusDeltaSum(final double p,
361                                                final double q) {
362
363        if (p < 10.0) {
364            throw new NumberIsTooSmallException(p, 10.0, true);
365        }
366        if (q < 10.0) {
367            throw new NumberIsTooSmallException(q, 10.0, true);
368        }
369
370        final double a = FastMath.min(p, q);
371        final double b = FastMath.max(p, q);
372        final double sqrtT = 10.0 / a;
373        final double t = sqrtT * sqrtT;
374        double z = DELTA[DELTA.length - 1];
375        for (int i = DELTA.length - 2; i >= 0; i--) {
376            z = t * z + DELTA[i];
377        }
378        return z / a + deltaMinusDeltaSum(a, b);
379    }

381    /**
382     * Returns the value of log B(p, q) for 0 ≤ x ≤ 1 and p, q > 0. Based on the
383     * <em>NSWC Library of Mathematics Subroutines</em> implementation,
384     * {@code DBETLN}.
385     *
386     * @param p First argument.
387     * @param q Second argument.
388     * @return the value of {@code log(Beta(p, q))}, {@code NaN} if
389     * {@code p <= 0} or {@code q <= 0}.
390     */
391    public static double logBeta(final double p, final double q) {
392        if (Double.isNaN(p) || Double.isNaN(q) || (p <= 0.0) || (q <= 0.0)) {
393            return Double.NaN;
394        }
395
396        final double a = FastMath.min(p, q);
397        final double b = FastMath.max(p, q);
398        if (a >= 10.0) {
399            final double w = sumDeltaMinusDeltaSum(a, b);
400            final double h = a / b;
401            final double c = h / (1.0 + h);
402            final double u = -(a - 0.5) * FastMath.log(c);
403            final double v = b * FastMath.log1p(h);
404            if (u <= v) {
405                return (((-0.5 * FastMath.log(b) + HALF_LOG_TWO_PI) + w) - u) - v;
406            } else {
407                return (((-0.5 * FastMath.log(b) + HALF_LOG_TWO_PI) + w) - v) - u;
408            }
409        } else if (a > 2.0) {
410            if (b > 1000.0) {
411                final int n = (int) FastMath.floor(a - 1.0);
412                double prod = 1.0;
413                double ared = a;
414                for (int i = 0; i < n; i++) {
415                    ared -= 1.0;
416                    prod *= ared / (1.0 + ared / b);
417                }
418                return (FastMath.log(prod) - n * FastMath.log(b)) +
419                        (Gamma.logGamma(ared) +
420                         logGammaMinusLogGammaSum(ared, b));
421            } else {
422                double prod1 = 1.0;
423                double ared = a;
424                while (ared > 2.0) {
425                    ared -= 1.0;
426                    final double h = ared / b;
427                    prod1 *= h / (1.0 + h);
428                }
429                if (b < 10.0) {
430                    double prod2 = 1.0;
431                    double bred = b;
432                    while (bred > 2.0) {
433                        bred -= 1.0;
434                        prod2 *= bred / (ared + bred);
435                    }
436                    return FastMath.log(prod1) +
437                           FastMath.log(prod2) +
438                           (Gamma.logGamma(ared) +
439                           (Gamma.logGamma(bred) -
440                            logGammaSum(ared, bred)));
441                } else {
442                    return FastMath.log(prod1) +
443                           Gamma.logGamma(ared) +
444                           logGammaMinusLogGammaSum(ared, b);
445                }
446            }
447        } else if (a >= 1.0) {
448            if (b > 2.0) {
449                if (b < 10.0) {
450                    double prod = 1.0;
451                    double bred = b;
452                    while (bred > 2.0) {
453                        bred -= 1.0;
454                        prod *= bred / (a + bred);
455                    }
456                    return FastMath.log(prod) +
457                           (Gamma.logGamma(a) +
458                            (Gamma.logGamma(bred) -
459                             logGammaSum(a, bred)));
460                } else {
461                    return Gamma.logGamma(a) +
462                           logGammaMinusLogGammaSum(a, b);
463                }
464            } else {
465                return Gamma.logGamma(a) +
466                       Gamma.logGamma(b) -
467                       logGammaSum(a, b);
468            }
469        } else {
470            if (b >= 10.0) {
471                return Gamma.logGamma(a) +
472                       logGammaMinusLogGammaSum(a, b);
473            } else {
474                // The following command is the original NSWC implementation.
475                // return Gamma.logGamma(a) +
476                // (Gamma.logGamma(b) - Gamma.logGamma(a + b));
477                // The following command turns out to be more accurate.
478                return FastMath.log(Gamma.gamma(a) * Gamma.gamma(b) /
479                                    Gamma.gamma(a + b));
480            }
481        }
482    }
