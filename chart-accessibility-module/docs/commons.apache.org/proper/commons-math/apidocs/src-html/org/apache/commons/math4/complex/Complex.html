032/**
033 * Representation of a Complex number, i.e. a number which has both a
034 * real and imaginary part.
035 * <p>
036 * Implementations of arithmetic operations handle {@code NaN} and
037 * infinite values according to the rules for {@link java.lang.Double}, i.e.
038 * {@link #equals} is an equivalence relation for all instances that have
039 * a {@code NaN} in either real or imaginary part, e.g. the following are
040 * considered equal:
041 * <ul>
042 *  <li>{@code 1 + NaNi}</li>
043 *  <li>{@code NaN + i}</li>
044 *  <li>{@code NaN + NaNi}</li>
045 * </ul><p>
046 * Note that this contradicts the IEEE-754 standard for floating
047 * point numbers (according to which the test {@code x == x} must fail if
048 * {@code x} is {@code NaN}). The method
049 * {@link org.apache.commons.math4.util.Precision#equals(double,double,int)
050 * equals for primitive double} in {@link org.apache.commons.math4.util.Precision}
051 * conforms with IEEE-754 while this class conforms with the standard behavior
052 * for Java object types.</p>
053 *
054 */
055public class Complex implements FieldElement<Complex>, Serializable  {
056    /** The square root of -1. A number representing "0.0 + 1.0i" */
057    public static final Complex I = new Complex(0.0, 1.0);
058    // CHECKSTYLE: stop ConstantName

060    public static final Complex NaN = new Complex(Double.NaN, Double.NaN);
061    // CHECKSTYLE: resume ConstantName
062    /** A complex number representing "+INF + INFi" */
063    public static final Complex INF = new Complex(Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY);
064    /** A complex number representing "1.0 + 0.0i" */
065    public static final Complex ONE = new Complex(1.0, 0.0);
066    /** A complex number representing "0.0 + 0.0i" */
067    public static final Complex ZERO = new Complex(0.0, 0.0);

090    /**
091     * Create a complex number given the real and imaginary parts.
092     *
093     * @param real Real part.
094     * @param imaginary Imaginary part.
095     */
096    public Complex(double real, double imaginary) {
097        this.real = real;
098        this.imaginary = imaginary;
099
100        isNaN = Double.isNaN(real) || Double.isNaN(imaginary);
101        isInfinite = !isNaN &&
102            (Double.isInfinite(real) || Double.isInfinite(imaginary));
103    }

105    /**
106     * Return the absolute value of this complex number.
107     * Returns {@code NaN} if either real or imaginary part is {@code NaN}
108     * and {@code Double.POSITIVE_INFINITY} if neither part is {@code NaN},
109     * but at least one part is infinite.
110     *
111     * @return the absolute value.
112     */
113    public double abs() {
114        if (isNaN) {
115            return Double.NaN;
116        }
117        if (isInfinite()) {
118            return Double.POSITIVE_INFINITY;
119        }
120        if (FastMath.abs(real) < FastMath.abs(imaginary)) {
121            if (imaginary == 0.0) {
122                return FastMath.abs(real);
123            }
124            double q = real / imaginary;
125            return FastMath.abs(imaginary) * FastMath.sqrt(1 + q * q);
126        } else {
127            if (real == 0.0) {
128                return FastMath.abs(imaginary);
129            }
130            double q = imaginary / real;
131            return FastMath.abs(real) * FastMath.sqrt(1 + q * q);
132        }
133    }

135    /**
136     * Returns a {@code Complex} whose value is
137     * {@code (this + addend)}.
138     * Uses the definitional formula
139     * <p>
140     *   {@code (a + bi) + (c + di) = (a+c) + (b+d)i}
141     * </p>
142     * If either {@code this} or {@code addend} has a {@code NaN} value in
143     * either part, {@link #NaN} is returned; otherwise {@code Infinite}
144     * and {@code NaN} values are returned in the parts of the result
145     * according to the rules for {@link java.lang.Double} arithmetic.
146     *
147     * @param  addend Value to be added to this {@code Complex}.
148     * @return {@code this + addend}.
149     * @throws NullArgumentException if {@code addend} is {@code null}.
150     */
151    @Override
152    public Complex add(Complex addend) throws NullArgumentException {
153        MathUtils.checkNotNull(addend);
154        if (isNaN || addend.isNaN) {
155            return NaN;
156        }
157
158        return createComplex(real + addend.getReal(),
159                             imaginary + addend.getImaginary());
160    }

200    /**
201     * Returns a {@code Complex} whose value is
202     * {@code (this / divisor)}.
203     * Implements the definitional formula
204     * <pre>
205     *  <code>
206     *    a + bi          ac + bd + (bc - ad)i
207     *    ----------- = -------------------------
208     *    c + di         c<sup>2</sup> + d<sup>2</sup>
209     *  </code>
210     * </pre>
211     * but uses
212     * <a href="http://doi.acm.org/10.1145/1039813.1039814">
213     * prescaling of operands</a> to limit the effects of overflows and
214     * underflows in the computation.
215     * <p>
216     * {@code Infinite} and {@code NaN} values are handled according to the
217     * following rules, applied in the order presented:
218     * <ul>
219     *  <li>If either {@code this} or {@code divisor} has a {@code NaN} value
220     *   in either part, {@link #NaN} is returned.
221     *  </li>
222     *  <li>If {@code divisor} equals {@link #ZERO}, {@link #NaN} is returned.
223     *  </li>
224     *  <li>If {@code this} and {@code divisor} are both infinite,
225     *   {@link #NaN} is returned.
226     *  </li>
227     *  <li>If {@code this} is finite (i.e., has no {@code Infinite} or
228     *   {@code NaN} parts) and {@code divisor} is infinite (one or both parts
229     *   infinite), {@link #ZERO} is returned.
230     *  </li>
231     *  <li>If {@code this} is infinite and {@code divisor} is finite,
232     *   {@code NaN} values are returned in the parts of the result if the
233     *   {@link java.lang.Double} rules applied to the definitional formula
234     *   force {@code NaN} results.
235     *  </li>
236     * </ul>
237     *
238     * @param divisor Value by which this {@code Complex} is to be divided.
239     * @return {@code this / divisor}.
240     * @throws NullArgumentException if {@code divisor} is {@code null}.
241     */
242    @Override
243    public Complex divide(Complex divisor)
244        throws NullArgumentException {
245        MathUtils.checkNotNull(divisor);
246        if (isNaN || divisor.isNaN) {
247            return NaN;
248        }
249
250        final double c = divisor.getReal();
251        final double d = divisor.getImaginary();
252        if (c == 0.0 && d == 0.0) {
253            return NaN;
254        }
255
256        if (divisor.isInfinite() && !isInfinite()) {
257            return ZERO;
258        }
259
260        if (FastMath.abs(c) < FastMath.abs(d)) {
261            double q = c / d;
262            double denominator = c * q + d;
263            return createComplex((real * q + imaginary) / denominator,
264                (imaginary * q - real) / denominator);
265        } else {
266            double q = d / c;
267            double denominator = d * q + c;
268            return createComplex((imaginary * q + real) / denominator,
269                (imaginary - real * q) / denominator);
270        }
271    }

273    /**
274     * Returns a {@code Complex} whose value is {@code (this / divisor)},
275     * with {@code divisor} interpreted as a real number.
276     *
277     * @param  divisor Value by which this {@code Complex} is to be divided.
278     * @return {@code this / divisor}.
279     * @see #divide(Complex)
280     */
281    public Complex divide(double divisor) {
282        if (isNaN || Double.isNaN(divisor)) {
283            return NaN;
284        }
285        if (divisor == 0d) {
286            return NaN;
287        }
288        if (Double.isInfinite(divisor)) {
289            return !isInfinite() ? ZERO : NaN;
290        }
291        return createComplex(real / divisor,
292                             imaginary  / divisor);
293    }

295    /** {@inheritDoc} */
296    @Override
297    public Complex reciprocal() {
298        if (isNaN) {
299            return NaN;
300        }
301
302        if (real == 0.0 && imaginary == 0.0) {
303            return INF;
304        }
305
306        if (isInfinite) {
307            return ZERO;
308        }
309
310        if (FastMath.abs(real) < FastMath.abs(imaginary)) {
311            double q = real / imaginary;
312            double scale = 1. / (real * q + imaginary);
313            return createComplex(scale * q, -scale);
314        } else {
315            double q = imaginary / real;
316            double scale = 1. / (imaginary * q + real);
317            return createComplex(scale, -scale * q);
318        }
319    }

321    /**
322     * Test for equality with another object.
323     * If both the real and imaginary parts of two complex numbers
324     * are exactly the same, and neither is {@code Double.NaN}, the two
325     * Complex objects are considered to be equal.
326     * The behavior is the same as for JDK's {@link Double#equals(Object)
327     * Double}:
328     * <ul>
329     *  <li>All {@code NaN} values are considered to be equal,
330     *   i.e, if either (or both) real and imaginary parts of the complex
331     *   number are equal to {@code Double.NaN}, the complex number is equal
332     *   to {@code NaN}.
333     *  </li>
334     *  <li>
335     *   Instances constructed with different representations of zero (i.e.
336     *   either "0" or "-0") are <em>not</em> considered to be equal.
337     *  </li>
338     * </ul>
339     *
340     * @param other Object to test for equality with this instance.
341     * @return {@code true} if the objects are equal, {@code false} if object
342     * is {@code null}, not an instance of {@code Complex}, or not equal to
343     * this instance.
344     */
345    @Override
346    public boolean equals(Object other) {
347        if (this == other) {
348            return true;
349        }
350        if (other instanceof Complex){
351            Complex c = (Complex) other;
352            if (c.isNaN) {
353                return isNaN;
354            } else {
355                return MathUtils.equals(real, c.real) &&
356                    MathUtils.equals(imaginary, c.imaginary);
357            }
358        }
359        return false;
360    }

497    /**
498     * Returns a {@code Complex} whose value is {@code this * factor}.
499     * Implements preliminary checks for {@code NaN} and infinity followed by
500     * the definitional formula:
501     * <p>
502     *   {@code (a + bi)(c + di) = (ac - bd) + (ad + bc)i}
503     * </p>
504     * Returns {@link #NaN} if either {@code this} or {@code factor} has one or
505     * more {@code NaN} parts.
506     * <p>
507     * Returns {@link #INF} if neither {@code this} nor {@code factor} has one
508     * or more {@code NaN} parts and if either {@code this} or {@code factor}
509     * has one or more infinite parts (same result is returned regardless of
510     * the sign of the components).
511     * </p><p>
512     * Returns finite values in components of the result per the definitional
513     * formula in all remaining cases.</p>
514     *
515     * @param  factor value to be multiplied by this {@code Complex}.
516     * @return {@code this * factor}.
517     * @throws NullArgumentException if {@code factor} is {@code null}.
518     */
519    @Override
520    public Complex multiply(Complex factor)
521        throws NullArgumentException {
522        MathUtils.checkNotNull(factor);
523        if (isNaN || factor.isNaN) {
524            return NaN;
525        }
526        if (Double.isInfinite(real) ||
527            Double.isInfinite(imaginary) ||
528            Double.isInfinite(factor.real) ||
529            Double.isInfinite(factor.imaginary)) {
530            // we don't use isInfinite() to avoid testing for NaN again
531            return INF;
532        }
533        return createComplex(real * factor.real - imaginary * factor.imaginary,
534                             real * factor.imaginary + imaginary * factor.real);
535    }

986    /**
987     * Compute the
988     * <a href="http://mathworld.wolfram.com/SquareRoot.html" TARGET="_top">
989     * square root</a> of this complex number.
990     * Implements the following algorithm to compute {@code sqrt(a + bi)}:
991     * <ol><li>Let {@code t = sqrt((|a| + |a + bi|) / 2)}</li>
992     * <li><pre>if {@code  a &#8805; 0} return {@code t + (b/2t)i}
993     *  else return {@code |b|/2t + sign(b)t i }</pre></li>
994     * </ol>
995     * where <ul>
996     * <li>{@code |a| = }{@link FastMath#abs}(a)</li>
997     * <li>{@code |a + bi| = }{@link Complex#abs}(a + bi)</li>
998     * <li>{@code sign(b) =  }{@link FastMath#copySign(double,double) copySign(1d, b)}
999     * </ul>
1000     * <p>
1001     * Returns {@link Complex#NaN} if either real or imaginary part of the
1002     * input argument is {@code NaN}.
1003     * </p>
1004     * Infinite values in real or imaginary parts of the input may result in
1005     * infinite or NaN values returned in parts of the result.
1006     * <pre>
1007     *  Examples:
1008     *  <code>
1009     *   sqrt(1 &plusmn; INFINITY i) = INFINITY + NaN i
1010     *   sqrt(INFINITY + i) = INFINITY + 0i
1011     *   sqrt(-INFINITY + i) = 0 + INFINITY i
1012     *   sqrt(INFINITY &plusmn; INFINITY i) = INFINITY + NaN i
1013     *   sqrt(-INFINITY &plusmn; INFINITY i) = NaN &plusmn; INFINITY i
1014     *  </code>
1015     * </pre>
1016     *
1017     * @return the square root of {@code this}.
1018     * @since 1.2
1019     */
1020    public Complex sqrt() {
1021        if (isNaN) {
1022            return NaN;
1023        }
1024
1025        if (real == 0.0 && imaginary == 0.0) {
1026            return createComplex(0.0, 0.0);
1027        }
1028
1029        double t = FastMath.sqrt((FastMath.abs(real) + abs()) / 2.0);
1030        if (real >= 0.0) {
1031            return createComplex(t, imaginary / (2.0 * t));
1032        } else {
1033            return createComplex(FastMath.abs(imaginary) / (2.0 * t),
1034                                 FastMath.copySign(1d, imaginary) * t);
1035        }
1036    }

1183    /**
1184     * Computes the n-th roots of this complex number.
1185     * The nth roots are defined by the formula:
1186     * <pre>
1187     *  <code>
1188     *   z<sub>k</sub> = abs<sup>1/n</sup> (cos(phi + 2&pi;k/n) + i (sin(phi + 2&pi;k/n))
1189     *  </code>
1190     * </pre>
1191     * for <i>{@code k=0, 1, ..., n-1}</i>, where {@code abs} and {@code phi}
1192     * are respectively the {@link #abs() modulus} and
1193     * {@link #getArgument() argument} of this complex number.
1194     * <p>
1195     * If one or both parts of this complex number is NaN, a list with just
1196     * one element, {@link #NaN} is returned.
1197     * if neither part is NaN, but at least one part is infinite, the result
1198     * is a one-element list containing {@link #INF}.
1199     *
1200     * @param n Degree of root.
1201     * @return a List of all {@code n}-th roots of {@code this}.
1202     * @throws NotPositiveException if {@code n <= 0}.
1203     * @since 2.0
1204     */
1205    public List<Complex> nthRoot(int n) throws NotPositiveException {
1206
1207        if (n <= 0) {
1208            throw new NotPositiveException(LocalizedFormats.CANNOT_COMPUTE_NTH_ROOT_FOR_NEGATIVE_N,
1209                                           n);
1210        }
1211
1212        final List<Complex> result = new ArrayList<Complex>();
1213
1214        if (isNaN) {
1215            result.add(NaN);
1216            return result;
1217        }
1218        if (isInfinite()) {
1219            result.add(INF);
1220            return result;
1221        }
1222
1223        // nth root of abs -- faster / more accurate to use a solver here?
1224        final double nthRootOfAbs = FastMath.pow(abs(), 1.0 / n);
1225
1226        // Compute nth roots of complex number with k = 0, 1, ... n-1
1227        final double nthPhi = getArgument() / n;
1228        final double slice = 2 * FastMath.PI / n;
1229        double innerPart = nthPhi;
1230        for (int k = 0; k < n ; k++) {
1231            // inner part
1232            final double realPart = nthRootOfAbs *  FastMath.cos(innerPart);
1233            final double imaginaryPart = nthRootOfAbs *  FastMath.sin(innerPart);
1234            result.add(createComplex(realPart, imaginaryPart));
1235            innerPart += slice;
1236        }
1237
1238        return result;
1239    }
