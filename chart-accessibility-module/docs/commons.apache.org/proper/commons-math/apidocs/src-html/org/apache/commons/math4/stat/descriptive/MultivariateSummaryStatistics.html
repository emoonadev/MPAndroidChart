039/**
040 * <p>Computes summary statistics for a stream of n-tuples added using the
041 * {@link #addValue(double[]) addValue} method. The data values are not stored
042 * in memory, so this class can be used to compute statistics for very large
043 * n-tuple streams.</p>
044 *
045 * <p>The {@link StorelessUnivariateStatistic} instances used to maintain
046 * summary state and compute statistics are configurable via setters.
047 * For example, the default implementation for the mean can be overridden by
048 * calling {@link #setMeanImpl(StorelessUnivariateStatistic[])}. Actual
049 * parameters to these methods must implement the
050 * {@link StorelessUnivariateStatistic} interface and configuration must be
051 * completed before <code>addValue</code> is called. No configuration is
052 * necessary to use the default, commons-math provided implementations.</p>
053 *
054 * <p>To compute statistics for a stream of n-tuples, construct a
055 * MultivariateStatistics instance with dimension n and then use
056 * {@link #addValue(double[])} to add n-tuples. The <code>getXxx</code>
057 * methods where Xxx is a statistic return an array of <code>double</code>
058 * values, where for <code>i = 0,...,n-1</code> the i<sup>th</sup> array element is the
059 * value of the given statistic for data range consisting of the i<sup>th</sup> element of
060 * each of the input n-tuples.  For example, if <code>addValue</code> is called
061 * with actual parameters {0, 1, 2}, then {3, 4, 5} and finally {6, 7, 8},
062 * <code>getSum</code> will return a three-element array with values
063 * {0+3+6, 1+4+7, 2+5+8}</p>
064 *
065 * <p>Note: This class is not thread-safe. Use
066 * {@link SynchronizedMultivariateSummaryStatistics} if concurrent access from multiple
067 * threads is required.</p>
068 *
069 * @since 1.2
070 */
071public class MultivariateSummaryStatistics
072    implements StatisticalMultivariateSummary, Serializable {
073

107    /**
108     * Construct a MultivariateSummaryStatistics instance
109     * @param k dimension of the data
110     * @param isCovarianceBiasCorrected if true, the unbiased sample
111     * covariance is computed, otherwise the biased population covariance
112     * is computed
113     */
114    public MultivariateSummaryStatistics(int k, boolean isCovarianceBiasCorrected) {
115        this.k = k;
116
117        sumImpl     = new StorelessUnivariateStatistic[k];
118        sumSqImpl   = new StorelessUnivariateStatistic[k];
119        minImpl     = new StorelessUnivariateStatistic[k];
120        maxImpl     = new StorelessUnivariateStatistic[k];
121        sumLogImpl  = new StorelessUnivariateStatistic[k];
122        geoMeanImpl = new StorelessUnivariateStatistic[k];
123        meanImpl    = new StorelessUnivariateStatistic[k];
124
125        for (int i = 0; i < k; ++i) {
126            sumImpl[i]     = new Sum();
127            sumSqImpl[i]   = new SumOfSquares();
128            minImpl[i]     = new Min();
129            maxImpl[i]     = new Max();
130            sumLogImpl[i]  = new SumOfLogs();
131            geoMeanImpl[i] = new GeometricMean();
132            meanImpl[i]    = new Mean();
133        }
134
135        covarianceImpl =
136            new VectorialCovariance(k, isCovarianceBiasCorrected);
137
138    }
139
140    /**
141     * Add an n-tuple to the data
142     *
143     * @param value  the n-tuple to add
144     * @throws DimensionMismatchException if the length of the array
145     * does not match the one used at construction
146     */
147    public void addValue(double[] value) throws DimensionMismatchException {
148        checkDimension(value.length);
149        for (int i = 0; i < k; ++i) {
150            double v = value[i];
151            sumImpl[i].increment(v);
152            sumSqImpl[i].increment(v);
153            minImpl[i].increment(v);
154            maxImpl[i].increment(v);
155            sumLogImpl[i].increment(v);
156            geoMeanImpl[i].increment(v);
157            meanImpl[i].increment(v);
158        }
159        covarianceImpl.increment(value);
160        n++;
161    }

181    /**
182     * Returns an array of the results of a statistic.
183     * @param stats univariate statistic array
184     * @return results array
185     */
186    private double[] getResults(StorelessUnivariateStatistic[] stats) {
187        double[] results = new double[stats.length];
188        for (int i = 0; i < results.length; ++i) {
189            results[i] = stats[i].getResult();
190        }
191        return results;
192    }

242    /**
243     * Returns an array whose i<sup>th</sup> entry is the standard deviation of the
244     * i<sup>th</sup> entries of the arrays that have been added using
245     * {@link #addValue(double[])}
246     *
247     * @return the array of component standard deviations
248     */
249    @Override
250    public double[] getStandardDeviation() {
251        double[] stdDev = new double[k];
252        if (getN() < 1) {
253            Arrays.fill(stdDev, Double.NaN);
254        } else if (getN() < 2) {
255            Arrays.fill(stdDev, 0.0);
256        } else {
257            RealMatrix matrix = covarianceImpl.getResult();
258            for (int i = 0; i < k; ++i) {
259                stdDev[i] = FastMath.sqrt(matrix.getEntry(i, i));
260            }
261        }
262        return stdDev;
263    }

311    /**
312     * Generates a text report displaying
313     * summary statistics from values that
314     * have been added.
315     * @return String with line feeds displaying statistics
316     */
317    @Override
318    public String toString() {
319        final String separator = ", ";
320        final String suffix = System.getProperty("line.separator");
321        StringBuilder outBuffer = new StringBuilder();
322        outBuffer.append("MultivariateSummaryStatistics:").append(suffix);
323        outBuffer.append("n: ").append(getN()).append(suffix);
324        append(outBuffer, getMin(), "min: ", separator, suffix);
325        append(outBuffer, getMax(), "max: ", separator, suffix);
326        append(outBuffer, getMean(), "mean: ", separator, suffix);
327        append(outBuffer, getGeometricMean(), "geometric mean: ", separator, suffix);
328        append(outBuffer, getSumSq(), "sum of squares: ", separator, suffix);
329        append(outBuffer, getSumLog(), "sum of logarithms: ", separator, suffix);
330        append(outBuffer, getStandardDeviation(), "standard deviation: ", separator, suffix);
331        outBuffer.append("covariance: ").append(getCovariance()).append(suffix);
332        return outBuffer.toString();
333    }

335    /**
336     * Append a text representation of an array to a buffer.
337     * @param buffer buffer to fill
338     * @param data data array
339     * @param prefix text prefix
340     * @param separator elements separator
341     * @param suffix text suffix
342     */
343    private void append(StringBuilder buffer, double[] data,
344                        String prefix, String separator, String suffix) {
345        buffer.append(prefix);
346        for (int i = 0; i < data.length; ++i) {
347            if (i > 0) {
348                buffer.append(separator);
349            }
350            buffer.append(data[i]);
351        }
352        buffer.append(suffix);
353    }

355    /**
356     * Resets all statistics and storage
357     */
358    public void clear() {
359        this.n = 0;
360        for (int i = 0; i < k; ++i) {
361            minImpl[i].clear();
362            maxImpl[i].clear();
363            sumImpl[i].clear();
364            sumLogImpl[i].clear();
365            sumSqImpl[i].clear();
366            geoMeanImpl[i].clear();
367            meanImpl[i].clear();
368        }
369        covarianceImpl.clear();
370    }

372    /**
373     * Returns true iff <code>object</code> is a <code>MultivariateSummaryStatistics</code>
374     * instance and all statistics have the same values as this.
375     * @param object the object to test equality against.
376     * @return true if object equals this
377     */
378    @Override
379    public boolean equals(Object object) {
380        if (object == this ) {
381            return true;
382        }
383        if (object instanceof MultivariateSummaryStatistics == false) {
384            return false;
385        }
386        MultivariateSummaryStatistics stat = (MultivariateSummaryStatistics) object;
387        return MathArrays.equalsIncludingNaN(stat.getGeometricMean(), getGeometricMean()) &&
388               MathArrays.equalsIncludingNaN(stat.getMax(),           getMax())           &&
389               MathArrays.equalsIncludingNaN(stat.getMean(),          getMean())          &&
390               MathArrays.equalsIncludingNaN(stat.getMin(),           getMin())           &&
391               Precision.equalsIncludingNaN(stat.getN(),             getN())             &&
392               MathArrays.equalsIncludingNaN(stat.getSum(),           getSum())           &&
393               MathArrays.equalsIncludingNaN(stat.getSumSq(),         getSumSq())         &&
394               MathArrays.equalsIncludingNaN(stat.getSumLog(),        getSumLog())        &&
395               stat.getCovariance().equals( getCovariance());
396    }

398    /**
399     * Returns hash code based on values of statistics
400     *
401     * @return hash code
402     */
403    @Override
404    public int hashCode() {
405        int result = 31 + MathUtils.hash(getGeometricMean());
406        result = result * 31 + MathUtils.hash(getGeometricMean());
407        result = result * 31 + MathUtils.hash(getMax());
408        result = result * 31 + MathUtils.hash(getMean());
409        result = result * 31 + MathUtils.hash(getMin());
410        result = result * 31 + MathUtils.hash(getN());
411        result = result * 31 + MathUtils.hash(getSum());
412        result = result * 31 + MathUtils.hash(getSumSq());
413        result = result * 31 + MathUtils.hash(getSumLog());
414        result = result * 31 + getCovariance().hashCode();
415        return result;
416    }

420     * Sets statistics implementations.
421     * @param newImpl new implementations for statistics
422     * @param oldImpl old implementations for statistics
423     * @throws DimensionMismatchException if the array dimension
424     * does not match the one used at construction
425     * @throws MathIllegalStateException if data has already been added
426     * (i.e. if n > 0)
427     */
428    private void setImpl(StorelessUnivariateStatistic[] newImpl,
429                         StorelessUnivariateStatistic[] oldImpl) throws MathIllegalStateException,
430                         DimensionMismatchException {
431        checkEmpty();
432        checkDimension(newImpl.length);
433        System.arraycopy(newImpl, 0, oldImpl, 0, newImpl.length);
434    }

625    /**
626     * Throws MathIllegalStateException if the statistic is not empty.
627     * @throws MathIllegalStateException if n > 0.
628     */
629    private void checkEmpty() throws MathIllegalStateException {
630        if (n > 0) {
631            throw new MathIllegalStateException(
632                    LocalizedFormats.VALUES_ADDED_BEFORE_CONFIGURING_STATISTIC, n);
633        }
634    }

636    /**
637     * Throws DimensionMismatchException if dimension != k.
638     * @param dimension dimension to check
639     * @throws DimensionMismatchException if dimension != k
640     */
641    private void checkDimension(int dimension) throws DimensionMismatchException {
642        if (dimension != k) {
643            throw new DimensionMismatchException(dimension, k);
644        }
