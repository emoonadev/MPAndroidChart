1043         double kx       = 2;
1044         double ky       = -3;
1045         double kz       = 5;
1046         double n2       = kx * kx + ky * ky + kz * kz;
1047         double n        = FastMath.sqrt(n2);

1049         double cosTheta = FastMath.cos(theta);
1050         double sinTheta = FastMath.sin(theta);
1051         FieldRotation<DerivativeStructure> r    = new FieldRotation<DerivativeStructure>(createAxis(kx, ky, kz),
1052                                                                                          createAngle(theta),
1053                                                                                          RotationConvention.VECTOR_OPERATOR);
1054         Vector3D a      = new Vector3D(kx / n, ky / n, kz / n);
1055 
1056         // Jacobian of the normalized rotation axis a with respect to the Cartesian vector k
1057         RealMatrix dadk = MatrixUtils.createRealMatrix(new double[][] {
1058             { (ky * ky + kz * kz) / ( n * n2),            -kx * ky / ( n * n2),            -kx * kz / ( n * n2) },
1059             {            -kx * ky / ( n * n2), (kx * kx + kz * kz) / ( n * n2),            -ky * kz / ( n * n2) },
1060             {            -kx * kz / ( n * n2),            -ky * kz / ( n * n2), (kx * kx + ky * ky) / ( n * n2) }
1061         });
1062 
1063         for (double x = -0.9; x < 0.9; x += 0.2) {
1064             for (double y = -0.9; y < 0.9; y += 0.2) {
1065                 for (double z = -0.9; z < 0.9; z += 0.2) {
1066                     Vector3D   u = new Vector3D(x, y, z);
1067                     FieldVector3D<DerivativeStructure> v = r.applyTo(createVector(x, y, z));
1068 
1069                     // explicit formula for rotation of vector u around axis a with angle theta
1070                     double dot     = Vector3D.dotProduct(u, a);
1071                     Vector3D cross = Vector3D.crossProduct(a, u);
1072                     double c1      = 1 - cosTheta;
1073                     double c2      = c1 * dot;
1074                     Vector3D rt    = new Vector3D(cosTheta, u, c2, a, sinTheta, cross);
1075                     Assert.assertEquals(rt.getX(), v.getX().getReal(), eps);
1076                     Assert.assertEquals(rt.getY(), v.getY().getReal(), eps);
1077                     Assert.assertEquals(rt.getZ(), v.getZ().getReal(), eps);

1082                         { c1 * x * a.getX() + c2,           c1 * y * a.getX() + sinTheta * z, c1 * z * a.getX() - sinTheta * y },
