107     @Test
108     public void testLargeDegreesOfFreedom() {
109         FDistribution fd = new FDistribution(100000, 100000);
110         double p = fd.cumulativeProbability(.999);
111         double x = fd.inverseCumulativeProbability(p);
112         Assert.assertEquals(.999, x, 1.0e-5);
113     }

115     @Test
116     public void testSmallDegreesOfFreedom() {
117         FDistribution fd = new FDistribution(1, 1);
118         double p = fd.cumulativeProbability(0.975);
119         double x = fd.inverseCumulativeProbability(p);
120         Assert.assertEquals(0.975, x, 1.0e-5);
121 
122         fd = new FDistribution(1, 2);
123         p = fd.cumulativeProbability(0.975);
124         x = fd.inverseCumulativeProbability(p);
125         Assert.assertEquals(0.975, x, 1.0e-5);
126     }
127 
128     @Test
129     public void testMoments() {
130         final double tol = 1e-9;
131         FDistribution dist;
132 
133         dist = new FDistribution(1, 2);
134         Assert.assertTrue(Double.isNaN(dist.getNumericalMean()));
135         Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
136 
137         dist = new FDistribution(1, 3);
138         Assert.assertEquals(dist.getNumericalMean(), 3d / (3d - 2d), tol);
139         Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
140 
141         dist = new FDistribution(1, 5);
142         Assert.assertEquals(dist.getNumericalMean(), 5d / (5d - 2d), tol);
143         Assert.assertEquals(dist.getNumericalVariance(), (2d * 5d * 5d * 4d) / 9d, tol);
144     }
