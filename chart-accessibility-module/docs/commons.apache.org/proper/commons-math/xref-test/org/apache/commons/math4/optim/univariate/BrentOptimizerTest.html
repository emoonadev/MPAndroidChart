46      @Test
47      public void testSinMin() {
48          UnivariateFunction f = new Sin();
49          UnivariateOptimizer optimizer = new BrentOptimizer(1e-10, 1e-14);
50          Assert.assertEquals(3 * Math.PI / 2, optimizer.optimize(new MaxEval(200),
51                                                                  new UnivariateObjectiveFunction(f),
52                                                                  GoalType.MINIMIZE,
53                                                                  new SearchInterval(4, 5)).getPoint(), 1e-8);
54          Assert.assertTrue(optimizer.getEvaluations() <= 50);
55          Assert.assertEquals(200, optimizer.getMaxEvaluations());
56          Assert.assertEquals(3 * Math.PI / 2, optimizer.optimize(new MaxEval(200),
57                                                                  new UnivariateObjectiveFunction(f),
58                                                                  GoalType.MINIMIZE,
59                                                                  new SearchInterval(1, 5)).getPoint(), 1e-8);
60          Assert.assertTrue(optimizer.getEvaluations() <= 100);
61          Assert.assertTrue(optimizer.getEvaluations() >= 15);
62          try {
63              optimizer.optimize(new MaxEval(10),
64                                 new UnivariateObjectiveFunction(f),
65                                 GoalType.MINIMIZE,
66                                 new SearchInterval(4, 5));
67              Assert.fail("an exception should have been thrown");
68          } catch (TooManyEvaluationsException fee) {
69              // expected
70          }
71      }

73      @Test
74      public void testSinMinWithValueChecker() {
75          final UnivariateFunction f = new Sin();
76          final ConvergenceChecker<UnivariatePointValuePair> checker = new SimpleUnivariateValueChecker(1e-5, 1e-14);
77          // The default stopping criterion of Brent's algorithm should not
78          // pass, but the search will stop at the given relative tolerance
79          // for the function value.
80          final UnivariateOptimizer optimizer = new BrentOptimizer(1e-10, 1e-14, checker);
81          final UnivariatePointValuePair result = optimizer.optimize(new MaxEval(200),
82                                                                     new UnivariateObjectiveFunction(f),
83                                                                     GoalType.MINIMIZE,
84                                                                     new SearchInterval(4, 5));
85          Assert.assertEquals(3 * Math.PI / 2, result.getPoint(), 1e-3);
86      }

88      @Test
89      public void testBoundaries() {
90          final double lower = -1.0;
91          final double upper = +1.0;
92          UnivariateFunction f = new UnivariateFunction() {
93              public double value(double x) {
94                  if (x < lower) {
95                      throw new NumberIsTooSmallException(x, lower, true);
96                  } else if (x > upper) {
97                      throw new NumberIsTooLargeException(x, upper, true);
98                  } else {
99                      return x;
100                 }
101             }
102         };
103         UnivariateOptimizer optimizer = new BrentOptimizer(1e-10, 1e-14);
104         Assert.assertEquals(lower,
105                             optimizer.optimize(new MaxEval(100),
106                                                new UnivariateObjectiveFunction(f),
107                                                GoalType.MINIMIZE,
108                                                new SearchInterval(lower, upper)).getPoint(),
109                             1.0e-8);
110         Assert.assertEquals(upper,
111                             optimizer.optimize(new MaxEval(100),
112                                                new UnivariateObjectiveFunction(f),
113                                                GoalType.MAXIMIZE,
114                                                new SearchInterval(lower, upper)).getPoint(),
115                             1.0e-8);
116     }

141     @Test
142     public void testQuinticMinStatistics() {
143         // The function has local minima at -0.27195613 and 0.82221643.
144         UnivariateFunction f = new QuinticFunction();
145         UnivariateOptimizer optimizer = new BrentOptimizer(1e-11, 1e-14);
146 
147         final DescriptiveStatistics[] stat = new DescriptiveStatistics[2];
148         for (int i = 0; i < stat.length; i++) {
149             stat[i] = new DescriptiveStatistics();
150         }
151 
152         final double min = -0.75;
153         final double max = 0.25;
154         final int nSamples = 200;
155         final double delta = (max - min) / nSamples;
156         for (int i = 0; i < nSamples; i++) {
157             final double start = min + i * delta;
158             stat[0].addValue(optimizer.optimize(new MaxEval(40),
159                                                 new UnivariateObjectiveFunction(f),
160                                                 GoalType.MINIMIZE,
161                                                 new SearchInterval(min, max, start)).getPoint());
162             stat[1].addValue(optimizer.getEvaluations());
163         }
164 
165         final double meanOptValue = stat[0].getMean();
166         final double medianEval = stat[1].getPercentile(50);
167         Assert.assertTrue(meanOptValue > -0.2719561281);
168         Assert.assertTrue(meanOptValue < -0.2719561280);
169         Assert.assertEquals(23, (int) medianEval);
170 
171         // MATH-1121: Ensure that the iteration counter is incremented.
172         Assert.assertTrue(optimizer.getIterations() > 0);
173     }

239      * Contrived example showing that prior to the resolution of MATH-855
240      * (second revision), the algorithm would not return the best point if
241      * it happened to be the initial guess.
242      */
243     @Test
244     public void testKeepInitIfBest() {
245         final double minSin = 3 * Math.PI / 2;
246         final double offset = 1e-8;
247         final double delta = 1e-7;
248         final UnivariateFunction f1 = new Sin();
249         final UnivariateFunction f2 = new StepFunction(new double[] { minSin, minSin + offset, minSin + 2 * offset},
250                                                        new double[] { 0, -1, 0 });
251         final UnivariateFunction f = FunctionUtils.add(f1, f2);
252         // A slightly less stringent tolerance would make the test pass
253         // even with the previous implementation.
254         final double relTol = 1e-8;
255         final UnivariateOptimizer optimizer = new BrentOptimizer(relTol, 1e-100);
256         final double init = minSin + 1.5 * offset;
257         final UnivariatePointValuePair result
258             = optimizer.optimize(new MaxEval(200),
259                                  new UnivariateObjectiveFunction(f),
260                                  GoalType.MINIMIZE,
261                                  new SearchInterval(minSin - 6.789 * delta,
262                                                     minSin + 9.876 * delta,
263                                                     init));
264 
265         final double sol = result.getPoint();
266         final double expected = init;
267 
268 //         System.out.println("numEval=" + numEval);
269 //         System.out.println("min=" + init + " f=" + f.value(init));
270 //         System.out.println("sol=" + sol + " f=" + f.value(sol));
271 //         System.out.println("exp=" + expected + " f=" + f.value(expected));
272 
273         Assert.assertTrue("Best point not reported", f.value(sol) <= f.value(expected));
274     }
