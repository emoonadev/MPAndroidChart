21  import java.util.List;

23  import org.apache.commons.math4.genetics.AbstractListChromosome;
24  import org.apache.commons.math4.genetics.BinaryChromosome;
25  import org.apache.commons.math4.genetics.BinaryMutation;
26  import org.apache.commons.math4.genetics.Chromosome;
27  import org.apache.commons.math4.genetics.ElitisticListPopulation;
28  import org.apache.commons.math4.genetics.FixedGenerationCount;
29  import org.apache.commons.math4.genetics.GeneticAlgorithm;
30  import org.apache.commons.math4.genetics.OnePointCrossover;
31  import org.apache.commons.math4.genetics.Population;
32  import org.apache.commons.math4.genetics.StoppingCondition;
33  import org.apache.commons.math4.genetics.TournamentSelection;
34  import org.junit.Assert;
35  import org.junit.Test;
36  
37  /**
38   * This is also an example of usage.
39   */
40  public class GeneticAlgorithmTestBinary {
41  
42      // parameters for the GA
43      private static final int DIMENSION = 50;
44      private static final int POPULATION_SIZE = 50;
45      private static final int NUM_GENERATIONS = 50;
46      private static final double ELITISM_RATE = 0.2;
47      private static final double CROSSOVER_RATE = 1;
48      private static final double MUTATION_RATE = 0.1;
49      private static final int TOURNAMENT_ARITY = 2;
50  
51      @Test
52      public void test() {
53          // to test a stochastic algorithm is hard, so this will rather be an usage example
54  
55          // initialize a new genetic algorithm
56          GeneticAlgorithm ga = new GeneticAlgorithm(
57                  new OnePointCrossover<Integer>(),
58                  CROSSOVER_RATE, // all selected chromosomes will be recombined (=crosssover)
59                  new BinaryMutation(),
60                  MUTATION_RATE,
61                  new TournamentSelection(TOURNAMENT_ARITY)
62          );
63  
64          Assert.assertEquals(0, ga.getGenerationsEvolved());
65  
66          // initial population
67          Population initial = randomPopulation();
68          // stopping conditions
69          StoppingCondition stopCond = new FixedGenerationCount(NUM_GENERATIONS);
70  
71          // best initial chromosome
72          Chromosome bestInitial = initial.getFittestChromosome();
73  
74          // run the algorithm
75          Population finalPopulation = ga.evolve(initial, stopCond);
76  
77          // best chromosome from the final population
78          Chromosome bestFinal = finalPopulation.getFittestChromosome();
79  
80          // the only thing we can test is whether the final solution is not worse than the initial one
81          // however, for some implementations of GA, this need not be true :)
82  
83          Assert.assertTrue(bestFinal.compareTo(bestInitial) > 0);
84          Assert.assertEquals(NUM_GENERATIONS, ga.getGenerationsEvolved());
85  
86      }
87  
88  
89  
90  
91      /**
92       * Initializes a random population.
93       */
94      private static ElitisticListPopulation randomPopulation() {
95          List<Chromosome> popList = new LinkedList<Chromosome>();
96  
97          for (int i=0; i<POPULATION_SIZE; i++) {
98              BinaryChromosome randChrom = new FindOnes(BinaryChromosome.randomBinaryRepresentation(DIMENSION));
99              popList.add(randChrom);
100         }
101         return new ElitisticListPopulation(popList, popList.size(), ELITISM_RATE);
102     }
103 
104     /**
105      * Chromosomes represented by a binary chromosome.
106      *
107      * The goal is to set all bits (genes) to 1.
108      */
109     private static class FindOnes extends BinaryChromosome {
110 
111         public FindOnes(List<Integer> representation) {
112             super(representation);
113         }
114 
115         /**
116          * Returns number of elements != 0
117          */
118         public double fitness() {
119             int num = 0;
120             for (int val : this.getRepresentation()) {
121                 if (val != 0)
122                     num++;
123             }
124             // number of elements >= 0
125             return num;
126         }
127 
128         @Override
129         public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> chromosomeRepresentation) {
130             return new FindOnes(chromosomeRepresentation);
131         }
132 
133     }
134 }
