231     @Test
232     public void testPowDoubleDS() {
233         for (int maxOrder = 1; maxOrder < 5; ++maxOrder) {
234 
235             DerivativeStructure x = new DerivativeStructure(3, maxOrder, 0, 0.1);
236             DerivativeStructure y = new DerivativeStructure(3, maxOrder, 1, 0.2);
237             DerivativeStructure z = new DerivativeStructure(3, maxOrder, 2, 0.3);
238             List<DerivativeStructure> list = Arrays.asList(x, y, z,
239                                                            x.add(y).add(z),
240                                                            x.multiply(y).multiply(z));
241 
242             for (DerivativeStructure ds : list) {
243                 // the special case a = 0 is included here
244                 for (double a : new double[] { 0.0, 0.1, 1.0, 2.0, 5.0 }) {
245                     DerivativeStructure reference = (a == 0) ?
246                                                     x.getField().getZero() :
247                                                     new DerivativeStructure(3, maxOrder, a).pow(ds);
248                     DerivativeStructure result = DerivativeStructure.pow(a, ds);
249                     checkEquals(reference, result, 1.0e-15);
250                 }
251 
252             }
253 
254             // negative base: -1^x can be evaluated for integers only, so value is sometimes OK, derivatives are always NaN
255             DerivativeStructure negEvenInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 2.0));
256             Assert.assertEquals(4.0, negEvenInteger.getValue(), 1.0e-15);
257             Assert.assertTrue(Double.isNaN(negEvenInteger.getPartialDerivative(1, 0, 0)));
258             DerivativeStructure negOddInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 3.0));
259             Assert.assertEquals(-8.0, negOddInteger.getValue(), 1.0e-15);
260             Assert.assertTrue(Double.isNaN(negOddInteger.getPartialDerivative(1, 0, 0)));
261             DerivativeStructure negNonInteger = DerivativeStructure.pow(-2.0, new DerivativeStructure(3,  maxOrder, 0, 2.001));
262             Assert.assertTrue(Double.isNaN(negNonInteger.getValue()));
263             Assert.assertTrue(Double.isNaN(negNonInteger.getPartialDerivative(1, 0, 0)));
264 
265             DerivativeStructure zeroNeg = DerivativeStructure.pow(0.0, new DerivativeStructure(3,  maxOrder, 0, -1.0));
266             Assert.assertTrue(Double.isNaN(zeroNeg.getValue()));
267             Assert.assertTrue(Double.isNaN(zeroNeg.getPartialDerivative(1, 0, 0)));
268             DerivativeStructure posNeg = DerivativeStructure.pow(2.0, new DerivativeStructure(3,  maxOrder, 0, -2.0));
269             Assert.assertEquals(1.0 / 4.0, posNeg.getValue(), 1.0e-15);
270             Assert.assertEquals(FastMath.log(2.0) / 4.0, posNeg.getPartialDerivative(1, 0, 0), 1.0e-15);
271 
272             // very special case: a = 0 and power = 0
273             DerivativeStructure zeroZero = DerivativeStructure.pow(0.0, new DerivativeStructure(3,  maxOrder, 0, 0.0));
274 
275             // this should be OK for simple first derivative with one variable only ...
276             Assert.assertEquals(1.0, zeroZero.getValue(), 1.0e-15);
277             Assert.assertEquals(Double.NEGATIVE_INFINITY, zeroZero.getPartialDerivative(1, 0, 0), 1.0e-15);
278 
279             // the following checks show a LIMITATION of the current implementation
280             // we have no way to tell x is a pure linear variable x = 0
281             // we only say: "x is a structure with value = 0.0,
282             // first derivative with respect to x = 1.0, and all other derivatives
283             // (first order with respect to y and z and higher derivatives) all 0.0.
284             // We have function f(x) = a^x and x = 0 so we compute:
285             // f(0) = 1, f'(0) = ln(a), f''(0) = ln(a)^2. The limit of these values
286             // when a converges to 0 implies all derivatives keep switching between
287             // +infinity and -infinity.
288             //
289             // Function composition rule for first derivatives is:
290             // d[f(g(x,y,z))]/dy = f'(g(x,y,z)) * dg(x,y,z)/dy
291             // so given that in our case x represents g and does not depend
292             // on y or z, we have dg(x,y,z)/dy = 0
293             // applying the composition rules gives:
294             // d[f(g(x,y,z))]/dy = f'(g(x,y,z)) * dg(x,y,z)/dy
295             //                 = -infinity * 0
296             //                 = NaN
297             // if we knew x is really the x variable and not the identity
298             // function applied to x, we would not have computed f'(g(x,y,z)) * dg(x,y,z)/dy
299             // and we would have found that the result was 0 and not NaN
300             Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 1, 0)));
301             Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 0, 1)));
302 
303             // Function composition rule for second derivatives is:
304             // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
305             // when function f is the a^x root and x = 0 we have:
306             // f(0) = 1, f'(0) = ln(a), f''(0) = ln(a)^2 which for a = 0 implies
307             // all derivatives keep switching between +infinity and -infinity
308             // so given that in our case x represents g, we have g(x) = 0,
309             // g'(x) = 1 and g''(x) = 0
310             // applying the composition rules gives:
311             // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
312             //                 = +infinity * 1^2 + -infinity * 0
313             //                 = +infinity + NaN
314             //                 = NaN
315             // if we knew x is really the x variable and not the identity
316             // function applied to x, we would not have computed f'(g(x)) * g''(x)
317             // and we would have found that the result was +infinity and not NaN
318             if (maxOrder > 1) {
319                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(2, 0, 0)));
320                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 2, 0)));
321                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 0, 2)));
322                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(1, 1, 0)));
323                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(0, 1, 1)));
324                 Assert.assertTrue(Double.isNaN(zeroZero.getPartialDerivative(1, 1, 0)));
325             }
326 
327             // very special case: 0^0 where the power is a primitive
328             DerivativeStructure zeroDsZeroDouble = new DerivativeStructure(3,  maxOrder, 0, 0.0).pow(0.0);
329             boolean first = true;
330             for (final double d : zeroDsZeroDouble.getAllDerivatives()) {
331                 if (first) {
332                     Assert.assertEquals(1.0, d, Precision.EPSILON);
333                     first = false;
334                 } else {
335                     Assert.assertEquals(0.0, d, Precision.SAFE_MIN);
336                 }
337             }
338             DerivativeStructure zeroDsZeroInt = new DerivativeStructure(3,  maxOrder, 0, 0.0).pow(0);
339             first = true;
340             for (final double d : zeroDsZeroInt.getAllDerivatives()) {
341                 if (first) {
342                     Assert.assertEquals(1.0, d, Precision.EPSILON);
343                     first = false;
344                 } else {
345                     Assert.assertEquals(0.0, d, Precision.SAFE_MIN);
346                 }
347             }
348 
349             // 0^p with p smaller than 1.0
350             DerivativeStructure u = new DerivativeStructure(3, maxOrder, 1, -0.0).pow(0.25);
351             for (int i0 = 0; i0 <= maxOrder; ++i0) {
352                 for (int i1 = 0; i1 <= maxOrder; ++i1) {
353                     for (int i2 = 0; i2 <= maxOrder; ++i2) {
354                         if (i0 + i1 + i2 <= maxOrder) {
355                             Assert.assertEquals(0.0, u.getPartialDerivative(i0, i1, i2), 1.0e-10);
356                         }
357                     }
358                 }
359             }
360         }

448     public void testTrigo() {
449         double epsilon = 2.0e-12;
450         for (int maxOrder = 0; maxOrder < 5; ++maxOrder) {
451             for (double x = 0.1; x < 1.2; x += 0.1) {
452                 DerivativeStructure dsX = new DerivativeStructure(3, maxOrder, 0, x);
453                 for (double y = 0.1; y < 1.2; y += 0.1) {
454                     DerivativeStructure dsY = new DerivativeStructure(3, maxOrder, 1, y);
455                     for (double z = 0.1; z < 1.2; z += 0.1) {
456                         DerivativeStructure dsZ = new DerivativeStructure(3, maxOrder, 2, z);
457                         DerivativeStructure f = dsX.divide(dsY.cos().add(dsZ.tan())).sin();
458                         double a = FastMath.cos(y) + FastMath.tan(z);
459                         double f0 = FastMath.sin(x / a);
460                         Assert.assertEquals(f0, f.getValue(), FastMath.abs(epsilon * f0));
461                         if (f.getOrder() > 0) {
462                             double dfdx = FastMath.cos(x / a) / a;
463                             Assert.assertEquals(dfdx, f.getPartialDerivative(1, 0, 0), FastMath.abs(epsilon * dfdx));
464                             double dfdy =  x * FastMath.sin(y) * dfdx / a;
465                             Assert.assertEquals(dfdy, f.getPartialDerivative(0, 1, 0), FastMath.abs(epsilon * dfdy));
466                             double cz = FastMath.cos(z);
467                             double cz2 = cz * cz;
468                             double dfdz = -x * dfdx / (a * cz2);

506     @Test
507     public void testRootNSingularity() {
508         for (int n = 2; n < 10; ++n) {
509             for (int maxOrder = 0; maxOrder < 12; ++maxOrder) {
510                 DerivativeStructure dsZero = new DerivativeStructure(1, maxOrder, 0, 0.0);
511                 DerivativeStructure rootN  = dsZero.rootN(n);
512                 Assert.assertEquals(0.0, rootN.getValue(), 1.0e-20);
513                 if (maxOrder > 0) {
514                     Assert.assertTrue(Double.isInfinite(rootN.getPartialDerivative(1)));
515                     Assert.assertTrue(rootN.getPartialDerivative(1) > 0);
516                     for (int order = 2; order <= maxOrder; ++order) {
517                         // the following checks shows a LIMITATION of the current implementation
518                         // we have no way to tell dsZero is a pure linear variable x = 0
519                         // we only say: "dsZero is a structure with value = 0.0,
520                         // first derivative = 1.0, second and higher derivatives = 0.0".
521                         // Function composition rule for second derivatives is:
522                         // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
523                         // when function f is the nth root and x = 0 we have:
524                         // f(0) = 0, f'(0) = +infinity, f''(0) = -infinity (and higher
525                         // derivatives keep switching between +infinity and -infinity)
526                         // so given that in our case dsZero represents g, we have g(x) = 0,
527                         // g'(x) = 1 and g''(x) = 0
528                         // applying the composition rules gives:
529                         // d2[f(g(x))]/dx2 = f''(g(x)) * [g'(x)]^2 + f'(g(x)) * g''(x)
530                         //                 = -infinity * 1^2 + +infinity * 0
531                         //                 = -infinity + NaN
532                         //                 = NaN
533                         // if we knew dsZero is really the x variable and not the identity
534                         // function applied to x, we would not have computed f'(g(x)) * g''(x)
535                         // and we would have found that the result was -infinity and not NaN
536                         Assert.assertTrue(Double.isNaN(rootN.getPartialDerivative(order)));
537                     }
538                 }
539 
540                 // the following shows that the limitation explained above is NOT a bug...
541                 // if we set up the higher order derivatives for g appropriately, we do
542                 // compute the higher order derivatives of the composition correctly
543                 double[] gDerivatives = new double[ 1 + maxOrder];
544                 gDerivatives[0] = 0.0;
545                 for (int k = 1; k <= maxOrder; ++k) {
546                     gDerivatives[k] = FastMath.pow(-1.0, k + 1);
547                 }
548                 DerivativeStructure correctRoot = new DerivativeStructure(1, maxOrder, gDerivatives).rootN(n);
549                 Assert.assertEquals(0.0, correctRoot.getValue(), 1.0e-20);
550                 if (maxOrder > 0) {
551                     Assert.assertTrue(Double.isInfinite(correctRoot.getPartialDerivative(1)));
552                     Assert.assertTrue(correctRoot.getPartialDerivative(1) > 0);
553                     for (int order = 2; order <= maxOrder; ++order) {
554                         Assert.assertTrue(Double.isInfinite(correctRoot.getPartialDerivative(order)));
555                         if ((order % 2) == 0) {
556                             Assert.assertTrue(correctRoot.getPartialDerivative(order) < 0);
557                         } else {
558                             Assert.assertTrue(correctRoot.getPartialDerivative(order) > 0);
559                         }
560                     }
561                 }

1597 
1598         // check dimension
1599         Assert.assertEquals(ds1.getFreeParameters(), ds2.getFreeParameters());
1600         Assert.assertEquals(ds1.getOrder(), ds2.getOrder());
1601 
1602         int[] derivatives = new int[ds1.getFreeParameters()];
1603         int sum = 0;
1604         while (true) {
1605 
1606             if (sum <= ds1.getOrder()) {
1607                 Assert.assertEquals(ds1.getPartialDerivative(derivatives),
1608                                     ds2.getPartialDerivative(derivatives),
1609                                     epsilon);
1610             }
1611 
1612             boolean increment = true;
1613             sum = 0;
1614             for (int i = derivatives.length - 1; i >= 0; --i) {
1615                 if (increment) {
1616                     if (derivatives[i] == ds1.getOrder()) {
1617                         derivatives[i] = 0;
1618                     } else {
1619                         derivatives[i]++;
1620                         increment = false;
1621                     }
1622                 }
1623                 sum += derivatives[i];
1624             }
1625             if (increment) {
1626                 return;
1627             }
