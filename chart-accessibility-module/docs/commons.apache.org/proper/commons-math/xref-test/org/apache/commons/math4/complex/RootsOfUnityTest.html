17  package org.apache.commons.math4.complex;
18  
19  import org.apache.commons.math4.complex.RootsOfUnity;
20  import org.apache.commons.math4.exception.MathIllegalStateException;
21  import org.apache.commons.math4.exception.ZeroException;
22  import org.apache.commons.math4.util.FastMath;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  
27  /**
28   * Unit tests for the {@link RootsOfUnity} class.
29   *
30   */
31  public class RootsOfUnityTest {
32  
33      @Test(expected = MathIllegalStateException.class)
34      public void testMathIllegalState1() {
35          final RootsOfUnity roots = new RootsOfUnity();
36          roots.getReal(0);
37      }
38  
39      @Test(expected = MathIllegalStateException.class)
40      public void testMathIllegalState2() {
41          final RootsOfUnity roots = new RootsOfUnity();
42          roots.getImaginary(0);
43      }
44  
45      @Test(expected = MathIllegalStateException.class)
46      public void testMathIllegalState3() {
47          final RootsOfUnity roots = new RootsOfUnity();
48          roots.isCounterClockWise();
49      }
50  
51      @Test(expected = ZeroException.class)
52      public void testZeroNumberOfRoots() {
53          final RootsOfUnity roots = new RootsOfUnity();
54          roots.computeRoots(0);
55      }
56  
57      @Test
58      public void testGetNumberOfRoots() {
59          final RootsOfUnity roots = new RootsOfUnity();
60          Assert.assertEquals("", 0, roots.getNumberOfRoots());
61          roots.computeRoots(5);
62          Assert.assertEquals("", 5, roots.getNumberOfRoots());
63          /*
64           * Testing -5 right after 5 is important, as the roots in this case are
65           * not recomputed.
66           */
67          roots.computeRoots(-5);
68          Assert.assertEquals("", 5, roots.getNumberOfRoots());
69          roots.computeRoots(6);
70          Assert.assertEquals("", 6, roots.getNumberOfRoots());
71      }
72  
73      @Test
74      public void testComputeRoots() {
75          final RootsOfUnity roots = new RootsOfUnity();
76          for (int n = -10; n < 11; n++) {
77              /*
78               * Testing -n right after n is important, as the roots in this case
79               * are not recomputed.
80               */
81              if (n != 0) {
82                  roots.computeRoots(n);
83                  doTestComputeRoots(roots);
84                  roots.computeRoots(-n);
85                  doTestComputeRoots(roots);
86              }
87          }
88      }
89  
90      private void doTestComputeRoots(final RootsOfUnity roots) {
91          final int n = roots.isCounterClockWise() ? roots.getNumberOfRoots() :
92              -roots.getNumberOfRoots();
93          final double tol = 10 * Math.ulp(1.0);
94          for (int k = 0; k < n; k++) {
95              final double t = 2.0 * FastMath.PI * k / n;
96              @SuppressWarnings("boxing")
97              final String msg = String.format("n = %d, k = %d", n, k);
98              Assert.assertEquals(msg, FastMath.cos(t), roots.getReal(k), tol);
99              Assert.assertEquals(msg, FastMath.sin(t), roots.getImaginary(k), tol);
100         }
101     }
102 }
