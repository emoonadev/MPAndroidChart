18  package org.apache.commons.math4.ml.neuralnet.sofm;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.ml.neuralnet.sofm.NeighbourhoodSizeFunction;
23  import org.apache.commons.math4.ml.neuralnet.sofm.NeighbourhoodSizeFunctionFactory;
24  import org.junit.Test;
25  import org.junit.Assert;
26  
27  /**
28   * Tests for {@link NeighbourhoodSizeFunctionFactory} class.
29   */
30  public class NeighbourhoodSizeFunctionFactoryTest {
31      @Test(expected=NotStrictlyPositiveException.class)
32      public void testExponentialDecayPrecondition1() {
33          NeighbourhoodSizeFunctionFactory.exponentialDecay(0, 0, 2);
34      }
35      @Test(expected=NotStrictlyPositiveException.class)
36      public void testExponentialDecayPrecondition2() {
37          NeighbourhoodSizeFunctionFactory.exponentialDecay(1, 0, 2);
38      }
39      @Test(expected=NumberIsTooLargeException.class)
40      public void testExponentialDecayPrecondition3() {
41          NeighbourhoodSizeFunctionFactory.exponentialDecay(1, 1, 100);
42      }
43      @Test(expected=NotStrictlyPositiveException.class)
44      public void testExponentialDecayPrecondition4() {
45          NeighbourhoodSizeFunctionFactory.exponentialDecay(2, 1, 0);
46      }
47  
48      @Test
49      public void testExponentialDecayTrivial() {
50          final int n = 65;
51          final int init = 4;
52          final int valueAtN = 3;
53          final NeighbourhoodSizeFunction f
54              = NeighbourhoodSizeFunctionFactory.exponentialDecay(init, valueAtN, n);
55  
56          Assert.assertEquals(init, f.value(0));
57          Assert.assertEquals(valueAtN, f.value(n));
58          Assert.assertEquals(0, f.value(Long.MAX_VALUE));
59      }
60  
61      @Test(expected=NotStrictlyPositiveException.class)
62      public void testQuasiSigmoidDecayPrecondition1() {
63          NeighbourhoodSizeFunctionFactory.quasiSigmoidDecay(0d, -1d, 2);
64      }
65      @Test(expected=NumberIsTooLargeException.class)
66      public void testQuasiSigmoidDecayPrecondition3() {
67          NeighbourhoodSizeFunctionFactory.quasiSigmoidDecay(1d, 0d, 100);
68      }
69      @Test(expected=NotStrictlyPositiveException.class)
70      public void testQuasiSigmoidDecayPrecondition4() {
71          NeighbourhoodSizeFunctionFactory.quasiSigmoidDecay(1d, -1d, 0);
72      }
73  
74      @Test
75      public void testQuasiSigmoidDecayTrivial() {
76          final int n = 65;
77          final double init = 4;
78          final double slope = -1e-1;
79          final NeighbourhoodSizeFunction f
80              = NeighbourhoodSizeFunctionFactory.quasiSigmoidDecay(init, slope, n);
81  
82          Assert.assertEquals(init, f.value(0), 0d);
83          Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
84      }
85  }
