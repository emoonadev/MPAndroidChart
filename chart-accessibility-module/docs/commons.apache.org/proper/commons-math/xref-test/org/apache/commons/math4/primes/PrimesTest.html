63      void assertPrimeFactorsException(int n, Throwable expected) {
64          try {
65              Primes.primeFactors(n);
66              Assert.fail("Exception not thrown");
67          } catch (Throwable e) {
68              Assert.assertEquals(expected.getClass(), e.getClass());
69              if (expected.getMessage() != null) {
70                  Assert.assertEquals(expected.getMessage(), e.getMessage());
71              }
72          }
73      }
74      void assertNextPrimeException(int n, Throwable expected){
75          try {
76              Primes.nextPrime(n);
77              Assert.fail("Exception not thrown");
78          } catch(Throwable e) {
79              Assert.assertEquals(expected.getClass(), e.getClass());
80              if (expected.getMessage() != null) {
81                  Assert.assertEquals(expected.getMessage(), e.getMessage());
82              }
83          }
84      }

86      @Test
87      public void testNextPrime() {
88  
89          Assert.assertEquals(2, Primes.nextPrime(0));
90          Assert.assertEquals(2, Primes.nextPrime(1));
91          Assert.assertEquals(2, Primes.nextPrime(2));
92          Assert.assertEquals(3, Primes.nextPrime(3));
93          Assert.assertEquals(5, Primes.nextPrime(4));
94          Assert.assertEquals(5, Primes.nextPrime(5));
95  
96          for (int i = 0; i < SmallPrimes.PRIMES.length - 1; i++) {
97              for (int j = SmallPrimes.PRIMES[i] + 1; j <= SmallPrimes.PRIMES[i + 1]; j++) {
98                  Assert.assertEquals(SmallPrimes.PRIMES[i+1], Primes.nextPrime(j));
99              }
100         }
101 
102         Assert.assertEquals(25325981, Primes.nextPrime(25325981));
103         for (int i = 25325981 + 1; i <= 25326023; i++) {
104             Assert.assertEquals(25326023, Primes.nextPrime(i));
105         }
106 
107         Assert.assertEquals(Integer.MAX_VALUE, Primes.nextPrime(Integer.MAX_VALUE - 10));
108         Assert.assertEquals(Integer.MAX_VALUE, Primes.nextPrime(Integer.MAX_VALUE - 1));
109         Assert.assertEquals(Integer.MAX_VALUE, Primes.nextPrime(Integer.MAX_VALUE));
110 
111         assertNextPrimeException(Integer.MIN_VALUE, new MathIllegalArgumentException(LocalizedFormats.NUMBER_TOO_SMALL,Integer.MIN_VALUE,0));
112         assertNextPrimeException(-1, new MathIllegalArgumentException(LocalizedFormats.NUMBER_TOO_SMALL,-1,0));
113         assertNextPrimeException(-13, new MathIllegalArgumentException(LocalizedFormats.NUMBER_TOO_SMALL,-13,0));
114     }

116     @Test
117     public void testIsPrime() throws Exception {
118         for (int i : BELOW_2) {
119             Assert.assertFalse(Primes.isPrime(i));
120         }
121         for (int i:NOT_PRIMES) {
122             Assert.assertFalse(Primes.isPrime(i));
123         }
124         for (int i:PRIMES) {
125             Assert.assertTrue(Primes.isPrime(i));
126         }
127     }

129     static int sum(List<Integer> numbers){
130         int out = 0;
131         for (int i:numbers) {
132             out += i;
133         }
134         return out;
135     }
136     static int product(List<Integer> numbers) {
137         int out = 1;
138         for (int i : numbers) {
139             out *= i;
140         }
141         return out;
142     }
143     static final HashSet<Integer> PRIMES_SET = new HashSet<Integer>();
144     static {
145         for (int p : PRIMES) {
146             PRIMES_SET.add(p);
147         }
148     }
149     static void checkPrimeFactors(List<Integer> factors){
150         for (int p : factors) {
151             if (!PRIMES_SET.contains(p)) {
152                 Assert.fail("Not found in primes list: " + p);
153             }
154         }
155     }

157     @Test
158     public void testPrimeFactors() throws Exception {
159         for (int i : BELOW_2) {
160             assertPrimeFactorsException(i, new MathIllegalArgumentException(LocalizedFormats.NUMBER_TOO_SMALL,i,2));
161         }
162         for (int i : NOT_PRIMES) {
163             List<Integer> factors = Primes.primeFactors(i);
164             checkPrimeFactors(factors);
165             int prod = product(factors);
166             Assert.assertEquals(i, prod);
167         }
168         for (int i : PRIMES) {
169             List<Integer> factors = Primes.primeFactors(i);
170             Assert.assertEquals(i, (int)factors.get(0));
171             Assert.assertEquals(1, factors.size());
172         }
173     }
