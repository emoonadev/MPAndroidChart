37      @Test
38      public void testSupport0Point() {
39          List<Vector3D> support = Arrays.asList(new Vector3D[0]);
40          EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
41          Assert.assertTrue(sphere.getRadius() < 0);
42          Assert.assertEquals(0, sphere.getSupportSize());
43          Assert.assertEquals(0, sphere.getSupport().length);
44      }

46      @Test
47      public void testSupport1Point() {
48          List<Vector3D> support = Arrays.asList(new Vector3D(1, 2, 3));
49          EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
50          Assert.assertEquals(0.0, sphere.getRadius(), 1.0e-10);
51          Assert.assertTrue(sphere.contains(support.get(0)));
52          Assert.assertTrue(sphere.contains(support.get(0), 0.5));
53          Assert.assertFalse(sphere.contains(new Vector3D(support.get(0).getX() + 0.1,
54                                                          support.get(0).getY() + 0.1,
55                                                          support.get(0).getZ() + 0.1),
56                                             0.001));
57          Assert.assertTrue(sphere.contains(new Vector3D(support.get(0).getX() + 0.1,
58                                                         support.get(0).getY() + 0.1,
59                                                         support.get(0).getZ() + 0.1),
60                                            0.5));
61          Assert.assertEquals(0, support.get(0).distance(sphere.getCenter()), 1.0e-10);
62          Assert.assertEquals(1, sphere.getSupportSize());
63          Assert.assertTrue(support.get(0) == sphere.getSupport()[0]);
64      }

66      @Test
67      public void testSupport2Points() {
68          List<Vector3D> support = Arrays.asList(new Vector3D(1, 0, 0),
69                                                 new Vector3D(3, 0, 0));
70          EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
71          Assert.assertEquals(1.0, sphere.getRadius(), 1.0e-10);
72          int i = 0;
73          for (Vector3D v : support) {
74              Assert.assertTrue(sphere.contains(v));
75              Assert.assertEquals(1.0, v.distance(sphere.getCenter()), 1.0e-10);
76              Assert.assertTrue(v == sphere.getSupport()[i++]);
77          }
78          Assert.assertTrue(sphere.contains(new Vector3D(2, 0.9, 0)));
79          Assert.assertFalse(sphere.contains(Vector3D.ZERO));
80          Assert.assertEquals(0.0, new Vector3D(2, 0, 0).distance(sphere.getCenter()), 1.0e-10);
81          Assert.assertEquals(2, sphere.getSupportSize());
82      }

84      @Test
85      public void testSupport3Points() {
86          List<Vector3D> support = Arrays.asList(new Vector3D(1, 0, 0),
87                                                 new Vector3D(3, 0, 0),
88                                                 new Vector3D(2, 2, 0));
89          EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
90          Assert.assertEquals(5.0 / 4.0, sphere.getRadius(), 1.0e-10);
91          int i = 0;
92          for (Vector3D v : support) {
93              Assert.assertTrue(sphere.contains(v));
94              Assert.assertEquals(5.0 / 4.0, v.distance(sphere.getCenter()), 1.0e-10);
95              Assert.assertTrue(v == sphere.getSupport()[i++]);
96          }
97          Assert.assertTrue(sphere.contains(new Vector3D(2, 0.9, 0)));
98          Assert.assertFalse(sphere.contains(new Vector3D(0.9,  0, 0)));
99          Assert.assertFalse(sphere.contains(new Vector3D(3.1,  0, 0)));
100         Assert.assertTrue(sphere.contains(new Vector3D(2.0, -0.499, 0)));
101         Assert.assertFalse(sphere.contains(new Vector3D(2.0, -0.501, 0)));
102         Assert.assertTrue(sphere.contains(new Vector3D(2.0, 3.0 / 4.0, -1.249)));
103         Assert.assertFalse(sphere.contains(new Vector3D(2.0, 3.0 / 4.0, -1.251)));
104         Assert.assertEquals(0.0, new Vector3D(2.0, 3.0 / 4.0, 0).distance(sphere.getCenter()), 1.0e-10);
105         Assert.assertEquals(3, sphere.getSupportSize());
106     }

108     @Test
109     public void testSupport4Points() {
110         List<Vector3D> support = Arrays.asList(new Vector3D(17, 14,  18),
111                                                new Vector3D(11, 14,  22),
112                                                new Vector3D( 2, 22,  17),
113                                                new Vector3D(22, 11, -10));
114         EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
115         Assert.assertEquals(25.0, sphere.getRadius(), 1.0e-10);
116         int i = 0;
117         for (Vector3D v : support) {
118             Assert.assertTrue(sphere.contains(v));
119             Assert.assertEquals(25.0, v.distance(sphere.getCenter()), 1.0e-10);
120             Assert.assertTrue(v == sphere.getSupport()[i++]);
121         }
122         Assert.assertTrue(sphere.contains (new Vector3D(-22.999, 2, 2)));
123         Assert.assertFalse(sphere.contains(new Vector3D(-23.001, 2, 2)));
124         Assert.assertTrue(sphere.contains (new Vector3D( 26.999, 2, 2)));
125         Assert.assertFalse(sphere.contains(new Vector3D( 27.001, 2, 2)));
126         Assert.assertTrue(sphere.contains (new Vector3D(2, -22.999, 2)));
127         Assert.assertFalse(sphere.contains(new Vector3D(2, -23.001, 2)));
128         Assert.assertTrue(sphere.contains (new Vector3D(2,  26.999, 2)));
129         Assert.assertFalse(sphere.contains(new Vector3D(2,  27.001, 2)));
130         Assert.assertTrue(sphere.contains (new Vector3D(2, 2, -22.999)));
131         Assert.assertFalse(sphere.contains(new Vector3D(2, 2, -23.001)));
132         Assert.assertTrue(sphere.contains (new Vector3D(2, 2,  26.999)));
133         Assert.assertFalse(sphere.contains(new Vector3D(2, 2,  27.001)));
134         Assert.assertEquals(0.0, new Vector3D(2.0, 2.0, 2.0).distance(sphere.getCenter()), 1.0e-10);
135         Assert.assertEquals(4, sphere.getSupportSize());
136     }

138     @Test
139     public void testRandom() {
140         final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
141                                                                  0xd015982e9f31ee04l);
142         final UnitSphereRandomVectorGenerator sr = new UnitSphereRandomVectorGenerator(3, random);
143         for (int i = 0; i < 100; ++i) {
144             double d = 25 * random.nextDouble();
145             double refRadius = 10 * random.nextDouble();
146             Vector3D refCenter = new Vector3D(d, new Vector3D(sr.nextVector()));
147             List<Vector3D> support = new ArrayList<Vector3D>();
148             for (int j = 0; j < 5; ++j) {
149                 support.add(new Vector3D(1.0, refCenter, refRadius, new Vector3D(sr.nextVector())));
150             }
151             EnclosingBall<Euclidean3D, Vector3D> sphere = new SphereGenerator().ballOnSupport(support);
152             Assert.assertEquals(0.0, refCenter.distance(sphere.getCenter()), 4e-7 * refRadius);
153             Assert.assertEquals(refRadius, sphere.getRadius(), 1e-7 * refRadius);
154         }
155     }
