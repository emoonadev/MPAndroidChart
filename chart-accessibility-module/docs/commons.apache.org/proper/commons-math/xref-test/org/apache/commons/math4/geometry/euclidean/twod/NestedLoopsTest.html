18  package org.apache.commons.math4.geometry.euclidean.twod;
19  
20  import org.junit.Assert;
21  import org.junit.Test;
22  
23  import java.lang.reflect.Field;
24  import java.util.Arrays;
25  import java.util.HashSet;
26  import java.util.List;
27  import java.util.Set;
28  
29  public class NestedLoopsTest {
30  
31      private static final double EPS = Math.ulp(1d);
32  
33      @Test
34      public void testNestedLoops() throws Exception {
35          Vector2D oneOne = new Vector2D(1.0, 1.0);
36          Vector2D oneNegativeOne = new Vector2D(1.0, -1.0);
37          Vector2D negativeOneNegativeOne = new Vector2D(-1.0, -1.0);
38          Vector2D negativeOneOne = new Vector2D(-1.0, 1.0);
39          Vector2D origin = new Vector2D(0, 0);
40  
41          Vector2D [] vertices = new Vector2D[]{
42                  oneOne,
43                  oneNegativeOne,
44                  negativeOneNegativeOne,
45                  negativeOneOne,
46                  origin
47          };
48  
49          NestedLoops nestedLoops = new NestedLoops(0.00000001);
50          nestedLoops.add(vertices);
51          nestedLoops.correctOrientation();
52  
53          Field surroundedField = nestedLoops.getClass().getDeclaredField("surrounded");
54          Field loopField = nestedLoops.getClass().getDeclaredField("loop");
55          surroundedField.setAccessible(Boolean.TRUE);
56          loopField.setAccessible(Boolean.TRUE);
57          List<NestedLoops> surrounded = (List<NestedLoops>) surroundedField.get(nestedLoops);
58          Vector2D[] loop = (Vector2D []) loopField.get(surrounded.get(0));
59          Set<Vector2D> vertexSet = new HashSet<>(Arrays.asList(loop));
60          Assert.assertTrue(vertexSet.contains(oneOne));
61          Assert.assertTrue(vertexSet.contains(oneNegativeOne));
62          Assert.assertTrue(vertexSet.contains(negativeOneNegativeOne));
63          Assert.assertTrue(vertexSet.contains(negativeOneOne));
64          Assert.assertTrue(vertexSet.contains(origin));
65      }
66  
67  }
