38  /**
39   * This is also an example of usage.
40   *
41   * This algorithm does "stochastic sorting" of a sequence 0,...,N.
42   *
43   */
44  public class GeneticAlgorithmTestPermutations {
45  
46      // parameters for the GA
47      private static final int DIMENSION = 20;
48      private static final int POPULATION_SIZE = 80;
49      private static final int NUM_GENERATIONS = 200;
50      private static final double ELITISM_RATE = 0.2;
51      private static final double CROSSOVER_RATE = 1;
52      private static final double MUTATION_RATE = 0.08;
53      private static final int TOURNAMENT_ARITY = 2;
54  
55      // numbers from 0 to N-1
56      private static final List<Integer> sequence = new ArrayList<Integer>();
57      static {
58          for (int i=0; i<DIMENSION; i++) {
59              sequence.add(i);
60          }
61      }
62  
63      @Test
64      public void test() {
65          // to test a stochastic algorithm is hard, so this will rather be an usage example
66  
67          // initialize a new genetic algorithm
68          GeneticAlgorithm ga = new GeneticAlgorithm(
69                  new OnePointCrossover<Integer>(),
70                  CROSSOVER_RATE,
71                  new RandomKeyMutation(),
72                  MUTATION_RATE,
73                  new TournamentSelection(TOURNAMENT_ARITY)
74          );
75  
76          // initial population
77          Population initial = randomPopulation();
78          // stopping conditions
79          StoppingCondition stopCond = new FixedGenerationCount(NUM_GENERATIONS);
80  
81          // best initial chromosome
82          Chromosome bestInitial = initial.getFittestChromosome();
83  
84          // run the algorithm
85          Population finalPopulation = ga.evolve(initial, stopCond);
86  
87          // best chromosome from the final population
88          Chromosome bestFinal = finalPopulation.getFittestChromosome();
89  
90          // the only thing we can test is whether the final solution is not worse than the initial one
91          // however, for some implementations of GA, this need not be true :)
92  
93          Assert.assertTrue(bestFinal.compareTo(bestInitial) > 0);
94  
95          //System.out.println(bestInitial);
96          //System.out.println(bestFinal);
97      }
98  
99  
100     /**
101      * Initializes a random population
102      */
103     private static ElitisticListPopulation randomPopulation() {
104         List<Chromosome> popList = new ArrayList<Chromosome>();
105         for (int i=0; i<POPULATION_SIZE; i++) {
106             Chromosome randChrom = new MinPermutations(RandomKey.randomPermutation(DIMENSION));
107             popList.add(randChrom);
108         }
109         return new ElitisticListPopulation(popList, popList.size(), ELITISM_RATE);
110     }
111 
112     /**
113      * Chromosomes representing a permutation of (0,1,2,...,DIMENSION-1).
114      *
115      * The goal is to sort the sequence.
116      */
117     private static class MinPermutations extends RandomKey<Integer> {
118 
119         public MinPermutations(List<Double> representation) {
120             super(representation);
121         }
122 
123         public double fitness() {
124             int res = 0;
125             List<Integer> decoded = decode(sequence);
126             for (int i=0; i<decoded.size(); i++) {
127                 int value = decoded.get(i);
128                 if (value != i) {
129                     // bad position found
130                     res += FastMath.abs(value - i);
131                 }
132             }
133             // the most fitted chromosome is the one with minimal error
134             // therefore we must return negative value
135             return -res;
136         }
137 
138         @Override
139         public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> chromosomeRepresentation) {
140             return new MinPermutations(chromosomeRepresentation);
141         }
142     }
143 }
