29      @Test
30      public void testRangeMultipleIterations() {
31          // Check that we can iterate several times using the same instance.
32          final int start = 1;
33          final int max = 7;
34          final int step = 2;
35  
36          final List<Integer> seq = new ArrayList<Integer>();
37          final IntegerSequence.Range r = IntegerSequence.range(start, max, step);
38  
39          final int numTimes = 3;
40          for (int n = 0; n < numTimes; n++) {
41              seq.clear();
42              for (Integer i : r) {
43                  seq.add(i);
44              }
45              Assert.assertEquals(4, seq.size());
46              Assert.assertEquals(seq.size(), r.size());
47          }
48      }

50      @Test
51      public void testIncreasingRange() {
52          final int start = 1;
53          final int max = 7;
54          final int step = 2;
55  
56          final List<Integer> seq = new ArrayList<Integer>();
57          final IntegerSequence.Range r = IntegerSequence.range(start, max, step);
58          for (Integer i : r) {
59              seq.add(i);
60          }
61  
62          Assert.assertEquals(4, seq.size());
63          Assert.assertEquals(seq.size(), r.size());
64          for (int i = 0; i < seq.size(); i++) {
65              Assert.assertEquals(start + i * step, seq.get(i).intValue());
66          }
67      }

170     @Test(expected=MaxCountExceededException.class)
171     public void testIncrementorCountExceeded() {
172         final int start = 1;
173         final int max = 7;
174         final int step = 2;
175 
176         final IntegerSequence.Incrementor inc =
177             IntegerSequence.Incrementor.create()
178             .withStart(start)
179             .withMaximalCount(max)
180             .withIncrement(step);
181 
182         Assert.assertTrue(inc.canIncrement(2));
183         Assert.assertFalse(inc.canIncrement(3));
184 
185         while (true) {
186             inc.increment();
187         }
188     }

250     @Test(expected=TooManyEvaluationsException.class)
251     public void testIncrementorAlternateException() {
252         final int start = 1;
253         final int max = 2;
254         final int step = 1;
255 
256         final IntegerSequence.Incrementor.MaxCountExceededCallback cb
257             = new IntegerSequence.Incrementor.MaxCountExceededCallback() {
258                     /** {@inheritDoc} */
259                     public void trigger(int max) {
260                         throw new TooManyEvaluationsException(max);
261                     }
262                 };
263 
264         final IntegerSequence.Incrementor inc
265             = IntegerSequence.Incrementor.create()
266             .withStart(start)
267             .withMaximalCount(max)
268             .withIncrement(step)
269             .withCallback(cb);
270 
271         try {
272             // One call must succeed.
273             inc.increment();
274         } catch (RuntimeException e) {
275             Assert.fail("unexpected exception");
276         }
277 
278         // Second call must fail.
279         inc.increment();
280     }
