171         Fraction first = new Fraction(1, 2);
172         Fraction second = new Fraction(1, 3);
173         Fraction third = new Fraction(1, 2);
174 
175         Assert.assertEquals(0, first.compareTo(first));

335     @Test
336     public void testAdd() {
337         Fraction a = new Fraction(1, 2);
338         Fraction b = new Fraction(2, 3);
339 
340         assertFraction(1, 1, a.add(a));
341         assertFraction(7, 6, a.add(b));
342         assertFraction(7, 6, b.add(a));
343         assertFraction(4, 3, b.add(b));
344 
345         Fraction f1 = new Fraction(Integer.MAX_VALUE - 1, 1);
346         Fraction f2 = Fraction.ONE;
347         Fraction f = f1.add(f2);
348         Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
349         Assert.assertEquals(1, f.getDenominator());
350         f = f1.add(1);
351         Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
352         Assert.assertEquals(1, f.getDenominator());
353 
354         f1 = new Fraction(-1, 13*13*2*2);
355         f2 = new Fraction(-2, 13*17*2);
356         f = f1.add(f2);
357         Assert.assertEquals(13*13*17*2*2, f.getDenominator());
358         Assert.assertEquals(-17 - 2*13*2, f.getNumerator());
359 
360         try {
361             f.add(null);
362             Assert.fail("expecting NullArgumentException");
363         } catch (NullArgumentException ex) {}
364 
365         // if this fraction is added naively, it will overflow.
366         // check that it doesn't.
367         f1 = new Fraction(1,32768*3);
368         f2 = new Fraction(1,59049);
369         f = f1.add(f2);
370         Assert.assertEquals(52451, f.getNumerator());
371         Assert.assertEquals(1934917632, f.getDenominator());
372 
373         f1 = new Fraction(Integer.MIN_VALUE, 3);
374         f2 = new Fraction(1,3);
375         f = f1.add(f2);
376         Assert.assertEquals(Integer.MIN_VALUE+1, f.getNumerator());
377         Assert.assertEquals(3, f.getDenominator());
378 
379         f1 = new Fraction(Integer.MAX_VALUE - 1, 1);
380         f2 = Fraction.ONE;
381         f = f1.add(f2);
382         Assert.assertEquals(Integer.MAX_VALUE, f.getNumerator());
383         Assert.assertEquals(1, f.getDenominator());
384 
385         try {
386             f = f.add(Fraction.ONE); // should overflow
387             Assert.fail("expecting MathArithmeticException but got: " + f.toString());
388         } catch (MathArithmeticException ex) {}
389 
390         // denominator should not be a multiple of 2 or 3 to trigger overflow
391         f1 = new Fraction(Integer.MIN_VALUE, 5);
392         f2 = new Fraction(-1,5);
393         try {
394             f = f1.add(f2); // should overflow
395             Assert.fail("expecting MathArithmeticException but got: " + f.toString());
396         } catch (MathArithmeticException ex) {}
397 
398         try {
399             f= new Fraction(-Integer.MAX_VALUE, 1);

402         } catch (MathArithmeticException ex) {}
403
