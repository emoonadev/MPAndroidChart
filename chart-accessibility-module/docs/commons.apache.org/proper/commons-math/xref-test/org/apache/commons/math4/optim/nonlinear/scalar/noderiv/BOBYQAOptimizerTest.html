297     /**
298      * @param func Function to optimize.
299      * @param startPoint Starting point.
300      * @param boundaries Upper / lower point limit.
301      * @param goal Minimization or maximization.
302      * @param fTol Tolerance relative error on the objective function.
303      * @param pointTol Tolerance for checking that the optimum is correct.
304      * @param maxEvaluations Maximum number of evaluations.
305      * @param additionalInterpolationPoints Number of interpolation to used
306      * in addition to the default (2 * dim + 1).
307      * @param expected Expected point / value.
308      */
309     private void doTest(MultivariateFunction func,
310                         double[] startPoint,
311                         double[][] boundaries,
312                         GoalType goal,
313                         double fTol,
314                         double pointTol,
315                         int maxEvaluations,
316                         int additionalInterpolationPoints,
317                         PointValuePair expected,
318                         String assertMsg) {
319 
320 //         System.out.println(func.getClass().getName() + " BEGIN"); // XXX
321 
322         int dim = startPoint.length;
323         final int numIterpolationPoints = 2 * dim + 1 + additionalInterpolationPoints;
324         BOBYQAOptimizer optim = new BOBYQAOptimizer(numIterpolationPoints);
325         PointValuePair result = boundaries == null ?
326             optim.optimize(new MaxEval(maxEvaluations),
327                            new ObjectiveFunction(func),
328                            goal,
329                            SimpleBounds.unbounded(dim),
330                            new InitialGuess(startPoint)) :
331             optim.optimize(new MaxEval(maxEvaluations),
332                            new ObjectiveFunction(func),
333                            goal,
334                            new InitialGuess(startPoint),
335                            new SimpleBounds(boundaries[0],
336                                             boundaries[1]));
337 //        System.out.println(func.getClass().getName() + " = "
338 //              + optim.getEvaluations() + " f(");
339 //        for (double x: result.getPoint())  System.out.print(x + " ");
340 //        System.out.println(") = " +  result.getValue());
341         Assert.assertEquals(assertMsg, expected.getValue(), result.getValue(), fTol);
342         for (int i = 0; i < dim; i++) {
343             Assert.assertEquals(expected.getPoint()[i],
344                                 result.getPoint()[i], pointTol);
345         }
346 
347 //         System.out.println(func.getClass().getName() + " END"); // XXX
348     }

356     private static double[][] boundaries(int dim,
357             double lower, double upper) {
358         double[][] boundaries = new double[2][dim];
359         for (int i = 0; i < dim; i++)
360             boundaries[0][i] = lower;
361         for (int i = 0; i < dim; i++)
362             boundaries[1][i] = upper;
363         return boundaries;
364     }

525     private static class Rosen implements MultivariateFunction {
526 
527         public double value(double[] x) {
528             double f = 0;
529             for (int i = 0; i < x.length - 1; ++i)
530                 f += 1e2 * (x[i] * x[i] - x[i + 1]) * (x[i] * x[i] - x[i + 1])
531                 + (x[i] - 1.) * (x[i] - 1.);
532             return f;
533         }
534     }

536     private static class Ackley implements MultivariateFunction {
537         private double axisratio;
538 
539         Ackley(double axra) {
540             axisratio = axra;
541         }
542 
543         public Ackley() {
544             this(1);
545         }
546 
547         public double value(double[] x) {
548             double f = 0;
549             double res2 = 0;
550             double fac = 0;
551             for (int i = 0; i < x.length; ++i) {
552                 fac = FastMath.pow(axisratio, (i - 1.) / (x.length - 1.));
553                 f += fac * fac * x[i] * x[i];
554                 res2 += FastMath.cos(2. * FastMath.PI * fac * x[i]);
555             }
556             f = (20. - 20. * FastMath.exp(-0.2 * FastMath.sqrt(f / x.length))
557                     + FastMath.exp(1.) - FastMath.exp(res2 / x.length));
558             return f;
559         }
560     }

562     private static class Rastrigin implements MultivariateFunction {
563 
564         private double axisratio;
565         private double amplitude;
566 
567         Rastrigin() {
568             this(1, 10);
569         }
570 
571         Rastrigin(double axisratio, double amplitude) {
572             this.axisratio = axisratio;
573             this.amplitude = amplitude;
574         }
575 
576         public double value(double[] x) {
577             double f = 0;
578             double fac;
579             for (int i = 0; i < x.length; ++i) {
580                 fac = FastMath.pow(axisratio, (i - 1.) / (x.length - 1.));
581                 if (i == 0 && x[i] < 0)
582                     fac *= 1.;
583                 f += fac * fac * x[i] * x[i] + amplitude
584                 * (1. - FastMath.cos(2. * FastMath.PI * fac * x[i]));
585             }
586             return f;
587         }
588     }

590     private static class Basis {
591         double[][] basis;
592         Random rand = new Random(2); // use not always the same basis
593 
594         double[] Rotate(double[] x) {
595             GenBasis(x.length);
596             double[] y = new double[x.length];
597             for (int i = 0; i < x.length; ++i) {
598                 y[i] = 0;
599                 for (int j = 0; j < x.length; ++j)
600                     y[i] += basis[i][j] * x[j];
601             }
602             return y;
603         }
604 
605         void GenBasis(int DIM) {
606             if (basis != null ? basis.length == DIM : false)
607                 return;
608 
609             double sp;
610             int i, j, k;
611 
612             /* generate orthogonal basis */
613             basis = new double[DIM][DIM];
614             for (i = 0; i < DIM; ++i) {
615                 /* sample components gaussian */
616                 for (j = 0; j < DIM; ++j)
617                     basis[i][j] = rand.nextGaussian();
618                 /* substract projection of previous vectors */
619                 for (j = i - 1; j >= 0; --j) {
620                     for (sp = 0., k = 0; k < DIM; ++k)
621                         sp += basis[i][k] * basis[j][k]; /* scalar product */
622                     for (k = 0; k < DIM; ++k)
623                         basis[i][k] -= sp * basis[j][k]; /* substract */
624                 }
625                 /* normalize */
626                 for (sp = 0., k = 0; k < DIM; ++k)
627                     sp += basis[i][k] * basis[i][k]; /* squared norm */
628                 for (k = 0; k < DIM; ++k)
629                     basis[i][k] /= FastMath.sqrt(sp);
630             }
631         }
