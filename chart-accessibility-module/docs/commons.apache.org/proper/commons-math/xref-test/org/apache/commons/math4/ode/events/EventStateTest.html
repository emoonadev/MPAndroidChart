40      // JIRA: MATH-322
41      @Test
42      public void closeEvents() throws MaxCountExceededException, NoBracketingException {
43  
44          final double r1  = 90.0;
45          final double r2  = 135.0;
46          final double gap = r2 - r1;
47  
48          final double tolerance = 0.1;
49          EventState es = new EventState(new CloseEventsGenerator(r1, r2), 1.5 * gap,
50                                         tolerance, 100,
51                                         new BrentSolver(tolerance));
52          es.setExpandable(new ExpandableStatefulODE(new FirstOrderDifferentialEquations() {
53              public int getDimension() {
54                  return 0;
55              }
56              public void computeDerivatives(double t, double[] y, double[] yDot) {
57              }
58          }));
59  
60          AbstractStepInterpolator interpolator =
61              new DummyStepInterpolator(new double[0], new double[0], true);
62          interpolator.storeTime(r1 - 2.5 * gap);
63          interpolator.shift();
64          interpolator.storeTime(r1 - 1.5 * gap);
65          es.reinitializeBegin(interpolator);
66  
67          interpolator.shift();
68          interpolator.storeTime(r1 - 0.5 * gap);
69          Assert.assertFalse(es.evaluateStep(interpolator));
70  
71          interpolator.shift();
72          interpolator.storeTime(0.5 * (r1 + r2));
73          Assert.assertTrue(es.evaluateStep(interpolator));
74          Assert.assertEquals(r1, es.getEventTime(), tolerance);
75          es.stepAccepted(es.getEventTime(), new double[0]);
76  
77          interpolator.shift();
78          interpolator.storeTime(r2 + 0.4 * gap);
79          Assert.assertTrue(es.evaluateStep(interpolator));
80          Assert.assertEquals(r2, es.getEventTime(), tolerance);
81  
82      }

84      // Jira: MATH-695
85      @Test
86      public void testIssue695()
87          throws DimensionMismatchException, NumberIsTooSmallException,
88                 MaxCountExceededException, NoBracketingException {
89  
90          FirstOrderDifferentialEquations equation = new FirstOrderDifferentialEquations() {
91  
92              public int getDimension() {
93                  return 1;
94              }
95  
96              public void computeDerivatives(double t, double[] y, double[] yDot) {
97                  yDot[0] = 1.0;
98              }
99          };
100 
101         DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.001, 1000, 1.0e-14, 1.0e-14);
102         integrator.addEventHandler(new ResettingEvent(10.99), 0.1, 1.0e-9, 1000);
103         integrator.addEventHandler(new ResettingEvent(11.01), 0.1, 1.0e-9, 1000);
104         integrator.setInitialStepSize(3.0);
105 
106         double target = 30.0;
107         double[] y = new double[1];
108         double tEnd = integrator.integrate(equation, 0.0, y, target, y);
109         Assert.assertEquals(target, tEnd, 1.0e-10);
110         Assert.assertEquals(32.0, y[0], 1.0e-10);
111 
112     }

114     private static class ResettingEvent implements EventHandler {
115 
116         private static double lastTriggerTime = Double.NEGATIVE_INFINITY;
117         private final double tEvent;
118 
119         public ResettingEvent(final double tEvent) {
120             this.tEvent = tEvent;
121         }
122 
123         public void init(double t0, double[] y0, double t) {
124         }
125 
126         public double g(double t, double[] y) {
127             // the bug corresponding to issue 695 causes the g function
128             // to be called at obsolete times t despite an event
129             // occurring later has already been triggered.
130             // When this occurs, the following assertion is violated
131             Assert.assertTrue("going backard in time! (" + t + " < " + lastTriggerTime + ")",
132                               t >= lastTriggerTime);
133             return t - tEvent;
134         }
135 
136         public Action eventOccurred(double t, double[] y, boolean increasing) {
137             // remember in a class variable when the event was triggered
138             lastTriggerTime = t;
139             return Action.RESET_STATE;
140         }
141 
142         public void resetState(double t, double[] y) {
143             y[0] += 1.0;
144         }
145 
146     }

148     // Jira: MATH-965
149     @Test
150     public void testIssue965()
151         throws DimensionMismatchException, NumberIsTooSmallException,
152                MaxCountExceededException, NoBracketingException {
153 
154         ExpandableStatefulODE equation =
155                 new ExpandableStatefulODE(new FirstOrderDifferentialEquations() {
156 
157             public int getDimension() {
158                 return 1;
159             }
160 
161             public void computeDerivatives(double t, double[] y, double[] yDot) {
162                 yDot[0] = 2.0;
163             }
164         });
165         equation.setTime(0.0);
166         equation.setPrimaryState(new double[1]);
167         equation.addSecondaryEquations(new SecondaryEquations() {
168 
169             public int getDimension() {
170                 return 1;
171             }
172 
173             public void computeDerivatives(double t, double[] primary,
174                                            double[] primaryDot, double[] secondary,
175                                            double[] secondaryDot) {
176                 secondaryDot[0] = -3.0;
177             }
178         });
179         int index = equation.getSecondaryMappers()[0].getFirstIndex();
180 
181         DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.001, 1000, 1.0e-14, 1.0e-14);
182         integrator.addEventHandler(new SecondaryStateEvent(index, -3.0), 0.1, 1.0e-9, 1000);
183         integrator.setInitialStepSize(3.0);
184 
185         integrator.integrate(equation, 30.0);
186         Assert.assertEquals( 1.0, equation.getTime(), 1.0e-10);
187         Assert.assertEquals( 2.0, equation.getPrimaryState()[0], 1.0e-10);
188         Assert.assertEquals(-3.0, equation.getSecondaryState(0)[0], 1.0e-10);
189 
190     }

192     private static class SecondaryStateEvent implements EventHandler {
193 
194         private int index;
195         private final double target;
196 
197         public SecondaryStateEvent(final int index, final double target) {
198             this.index  = index;
199             this.target = target;
200         }
201 
202         public void init(double t0, double[] y0, double t) {
203         }
204 
205         public double g(double t, double[] y) {
206             return y[index] - target;
207         }
208 
209         public Action eventOccurred(double t, double[] y, boolean increasing) {
210             return Action.STOP;
211         }
212 
213         public void resetState(double t, double[] y) {
214         }

218     @Test
219     public void testEventsCloserThanThreshold()
220         throws DimensionMismatchException, NumberIsTooSmallException,
221                MaxCountExceededException, NoBracketingException {
222 
223         FirstOrderDifferentialEquations equation = new FirstOrderDifferentialEquations() {
224 
225             public int getDimension() {
226                 return 1;
227             }
228 
229             public void computeDerivatives(double t, double[] y, double[] yDot) {
230                 yDot[0] = 1.0;
231             }
232         };
233 
234         LutherIntegrator integrator = new LutherIntegrator(20.0);
235         CloseEventsGenerator eventsGenerator =
236                         new CloseEventsGenerator(9.0 - 1.0 / 128, 9.0 + 1.0 / 128);
237         integrator.addEventHandler(eventsGenerator, 1.0, 0.02, 1000);
238         double[] y = new double[1];
239         double tEnd = integrator.integrate(equation, 0.0, y, 100.0, y);
240         Assert.assertEquals( 2, eventsGenerator.getCount());
241         Assert.assertEquals( 9.0 + 1.0 / 128, tEnd, 1.0 / 32.0);
242 
243     }

245     private class CloseEventsGenerator implements EventHandler {
246 
247         final double r1;
248         final double r2;
249         int count;
250 
251         public CloseEventsGenerator(final double r1, final double r2) {
252             this.r1    = r1;
253             this.r2    = r2;
254             this.count = 0;
255         }
256 
257         public void init(double t0, double[] y0, double t) {
258         }
259 
260         public void resetState(double t, double[] y) {
261         }
262 
263         public double g(double t, double[] y) {
264             return (t - r1) * (r2 - t);
265         }
266 
267         public Action eventOccurred(double t, double[] y, boolean increasing) {
268             return ++count < 2 ? Action.CONTINUE : Action.STOP;
269         }
270 
271         public int getCount() {
272             return count;
273         }
