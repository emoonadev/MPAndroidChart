18  package org.apache.commons.math4.geometry.euclidean.threed;
19  
20  import java.lang.reflect.Field;
21  
22  import org.apache.commons.math4.geometry.euclidean.threed.RotationOrder;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  
27  public class RotationOrderTest {
28  
29    @Test
30    public void testName() {
31  
32      RotationOrder[] orders = {
33        RotationOrder.XYZ, RotationOrder.XZY, RotationOrder.YXZ,
34        RotationOrder.YZX, RotationOrder.ZXY, RotationOrder.ZYX,
35        RotationOrder.XYX, RotationOrder.XZX, RotationOrder.YXY,
36        RotationOrder.YZY, RotationOrder.ZXZ, RotationOrder.ZYZ
37      };
38  
39      for (int i = 0; i < orders.length; ++i) {
40        Assert.assertEquals(getFieldName(orders[i]), orders[i].toString());
41      }
42  
43    }
44  
45    private String getFieldName(RotationOrder order) {
46      try {
47        Field[] fields = RotationOrder.class.getFields();
48        for (int i = 0; i < fields.length; ++i) {
49          if (fields[i].get(null) == order) {
50            return fields[i].getName();
51          }
52        }
53      } catch (IllegalAccessException iae) {
54        // ignored
55      }
56      return "unknown";
57    }
58  
59  }
