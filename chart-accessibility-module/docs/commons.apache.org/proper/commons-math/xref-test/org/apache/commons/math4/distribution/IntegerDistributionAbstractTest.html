30  /**
31   * Abstract base class for {@link IntegerDistribution} tests.
32   * <p>
33   * To create a concrete test class for an integer distribution implementation,
34   *  implement makeDistribution() to return a distribution instance to use in
35   *  tests and each of the test data generation methods below.  In each case, the
36   *  test points and test values arrays returned represent parallel arrays of
37   *  inputs and expected values for the distribution returned by makeDistribution().
38   *  <p>
39   *  makeDensityTestPoints() -- arguments used to test probability density calculation
40   *  makeDensityTestValues() -- expected probability densities
41   *  makeCumulativeTestPoints() -- arguments used to test cumulative probabilities
42   *  makeCumulativeTestValues() -- expected cumulative probabilites
43   *  makeInverseCumulativeTestPoints() -- arguments used to test inverse cdf evaluation
44   *  makeInverseCumulativeTestValues() -- expected inverse cdf values
45   * <p>
46   *  To implement additional test cases with different distribution instances and test data,
47   *  use the setXxx methods for the instance data in test cases and call the verifyXxx methods
48   *  to verify results.
49   *
50   */
51  public abstract class IntegerDistributionAbstractTest {
52

64      private double[] densityTestValues;

67      private double[] logDensityTestValues;
68

70      private int[] cumulativeTestPoints;

73      private double[] cumulativeTestValues;

76      private double[] inverseCumulativeTestPoints;
77

79      private int[] inverseCumulativeTestValues;

81      //-------------------- Abstract methods -----------------------------------
82  
83      /** Creates the default discrete distribution instance to use in tests. */
84      public abstract IntegerDistribution makeDistribution();
85  
86      /** Creates the default probability density test input values */
87      public abstract int[] makeDensityTestPoints();
88  
89      /** Creates the default probability density test expected values */
90      public abstract double[] makeDensityTestValues();
91  
92      /** Creates the default logarithmic probability density test expected values.
93       *
94       * The default implementation simply computes the logarithm of all the values in
95       * {@link #makeDensityTestValues()}.
96       *
97       * @return double[] the default logarithmic probability density test expected values.
98       */
99      public double[] makeLogDensityTestValues() {
100         final double[] densityTestValues = makeDensityTestValues();
101         final double[] logDensityTestValues = new double[densityTestValues.length];
102         for (int i = 0; i < densityTestValues.length; i++) {
103             logDensityTestValues[i] = FastMath.log(densityTestValues[i]);
104         }
105         return logDensityTestValues;
106     }

154     /**
155      * Verifies that probability density calculations match expected values
156      * using current test instance data
157      */
158     protected void verifyDensities() {
159         for (int i = 0; i < densityTestPoints.length; i++) {
160             Assert.assertEquals("Incorrect density value returned for " + densityTestPoints[i],
161                     densityTestValues[i],
162                     distribution.probability(densityTestPoints[i]), getTolerance());
163         }
164     }

166     /**
167      * Verifies that logarithmic probability density calculations match expected values
168      * using current test instance data.
169      */
170     protected void verifyLogDensities() {
171         for (int i = 0; i < densityTestPoints.length; i++) {
172             // FIXME: when logProbability methods are added to IntegerDistribution in 4.0, remove cast below
173             Assert.assertEquals("Incorrect log density value returned for " + densityTestPoints[i],
174                     logDensityTestValues[i],
175                     ((AbstractIntegerDistribution) distribution).logProbability(densityTestPoints[i]), tolerance);
176         }
177     }

242     @Test
243     public void testConsistencyAtSupportBounds() {
244         final int lower = distribution.getSupportLowerBound();
245         Assert.assertEquals("Cumulative probability mmust be 0 below support lower bound.",
246                 0.0, distribution.cumulativeProbability(lower - 1), 0.0);
247         Assert.assertEquals("Cumulative probability of support lower bound must be equal to probability mass at this point.",
248                 distribution.probability(lower), distribution.cumulativeProbability(lower), getTolerance());
249         Assert.assertEquals("Inverse cumulative probability of 0 must be equal to support lower bound.",
250                 lower, distribution.inverseCumulativeProbability(0.0));
251 
252         final int upper = distribution.getSupportUpperBound();
253         if (upper != Integer.MAX_VALUE)
254             Assert.assertEquals("Cumulative probability of support upper bound must be equal to 1.",
255                     1.0, distribution.cumulativeProbability(upper), 0.0);
256         Assert.assertEquals("Inverse cumulative probability of 1 must be equal to support upper bound.",
257                 upper, distribution.inverseCumulativeProbability(1.0));
258     }

285     /**
286      * Test sampling
287      */
288     @Test
289     public void testSampling() {
290         int[] densityPoints = makeDensityTestPoints();
291         double[] densityValues = makeDensityTestValues();
292         int sampleSize = 1000;
293         int length = TestUtils.eliminateZeroMassPoints(densityPoints, densityValues);
294         AbstractIntegerDistribution distribution = (AbstractIntegerDistribution) makeDistribution();
295         double[] expectedCounts = new double[length];
296         long[] observedCounts = new long[length];
297         for (int i = 0; i < length; i++) {
298             expectedCounts[i] = sampleSize * densityValues[i];
299         }
300         // Use fixed seed.
301         final IntegerDistribution.Sampler sampler =
302             distribution.createSampler(RandomSource.create(RandomSource.WELL_512_A,
303                                                            1000));
304         int[] sample = AbstractIntegerDistribution.sample(sampleSize, sampler);
305         for (int i = 0; i < sampleSize; i++) {
306           for (int j = 0; j < length; j++) {
307               if (sample[i] == densityPoints[j]) {
308                   observedCounts[j]++;
309               }
310           }
311         }
312         TestUtils.assertChiSquareAccept(densityPoints, expectedCounts, observedCounts, .001);
313     }
