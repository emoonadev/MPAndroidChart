43      public void testTransitionMeasurementMatrixMismatch() {
44  
45          // A and H matrix do not match in dimensions

48          RealMatrix A = new Array2DRowRealMatrix(new double[] { 1d });
49          // no control input
50          RealMatrix B = null;
51          // H = [ 1 1 ]
52          RealMatrix H = new Array2DRowRealMatrix(new double[] { 1d, 1d });
53          // Q = [ 0 ]
54          RealMatrix Q = new Array2DRowRealMatrix(new double[] { 0 });
55          // R = [ 0 ]
56          RealMatrix R = new Array2DRowRealMatrix(new double[] { 0 });
57  
58          ProcessModel pm
59              = new DefaultProcessModel(A, B, Q,
60                                        new ArrayRealVector(new double[] { 0 }), null);
61          MeasurementModel mm = new DefaultMeasurementModel(H, R);
62          new KalmanFilter(pm, mm);
63          Assert.fail("transition and measurement matrix should not be compatible");
64      }

90      @Test
91      public void testConstant() {
92          // simulates a simple process with a constant state and no control input
93  
94          double constantValue = 10d;
95          double measurementNoise = 0.1d;
96          double processNoise = 1e-5d;
97  
98          // A = [ 1 ]
99          RealMatrix A = new Array2DRowRealMatrix(new double[] { 1d });
100         // no control input
101         RealMatrix B = null;
102         // H = [ 1 ]
103         RealMatrix H = new Array2DRowRealMatrix(new double[] { 1d });
104         // x = [ 10 ]
105         RealVector x = new ArrayRealVector(new double[] { constantValue });
106         // Q = [ 1e-5 ]
107         RealMatrix Q = new Array2DRowRealMatrix(new double[] { processNoise });
108         // R = [ 0.1 ]
109         RealMatrix R = new Array2DRowRealMatrix(new double[] { measurementNoise });
110 
111         ProcessModel pm
112             = new DefaultProcessModel(A, B, Q,
113                                       new ArrayRealVector(new double[] { constantValue }), null);
114         MeasurementModel mm = new DefaultMeasurementModel(H, R);
115         KalmanFilter filter = new KalmanFilter(pm, mm);
116 
117         Assert.assertEquals(1, filter.getMeasurementDimension());
118         Assert.assertEquals(1, filter.getStateDimension());
119 
120         assertMatrixEquals(Q.getData(), filter.getErrorCovariance());
121 
122         // check the initial state
123         double[] expectedInitialState = new double[] { constantValue };
124         assertVectorEquals(expectedInitialState, filter.getStateEstimation());
125 
126         RealVector pNoise = new ArrayRealVector(1);
127         RealVector mNoise = new ArrayRealVector(1);
128 
129         final RealDistribution.Sampler rand = new NormalDistribution().createSampler(RandomSource.create(RandomSource.WELL_19937_C));
130 
131         // iterate 60 steps
132         for (int i = 0; i < 60; i++) {
133             filter.predict();
134 
135             // Simulate the process
136             pNoise.setEntry(0, processNoise * rand.sample());
137 
138             // x = A * x + p_noise
139             x = A.operate(x).add(pNoise);
140 
141             // Simulate the measurement
142             mNoise.setEntry(0, measurementNoise * rand.sample());
143 
144             // z = H * x + m_noise
145             RealVector z = H.operate(x).add(mNoise);
146 
147             filter.correct(z);
148 
149             // state estimate shouldn't be larger than measurement noise
150             double diff = FastMath.abs(constantValue - filter.getStateEstimation()[0]);
151             // System.out.println(diff);
152             Assert.assertTrue(Precision.compareTo(diff, measurementNoise, 1e-6) < 0);
153         }
154 
155         // error covariance should be already very low (< 0.02)
156         Assert.assertTrue(Precision.compareTo(filter.getErrorCovariance()[0][0],
157                                               0.02d, 1e-6) < 0);
158     }

160     @Test
161     public void testConstantAcceleration() {
162         // simulates a vehicle, accelerating at a constant rate (0.1 m/s)
163 
164         // discrete time interval
165         double dt = 0.1d;
166         // position measurement noise (meter)
167         double measurementNoise = 10d;
168         // acceleration noise (meter/sec^2)
169         double accelNoise = 0.2d;
170 
171         // A = [ 1 dt ]
172         //     [ 0  1 ]
173         RealMatrix A = new Array2DRowRealMatrix(new double[][] { { 1, dt }, { 0, 1 } });
174 
175         // B = [ dt^2/2 ]
176         //     [ dt     ]
177         RealMatrix B = new Array2DRowRealMatrix(
178                 new double[][] { { FastMath.pow(dt, 2d) / 2d }, { dt } });
179 
180         // H = [ 1 0 ]
181         RealMatrix H = new Array2DRowRealMatrix(new double[][] { { 1d, 0d } });
182 
183         // x = [ 0 0 ]
184         RealVector x = new ArrayRealVector(new double[] { 0, 0 });
185 
186         RealMatrix tmp = new Array2DRowRealMatrix(
187                 new double[][] { { FastMath.pow(dt, 4d) / 4d, FastMath.pow(dt, 3d) / 2d },
188                                  { FastMath.pow(dt, 3d) / 2d, FastMath.pow(dt, 2d) } });
189 
190         // Q = [ dt^4/4 dt^3/2 ]
191         //     [ dt^3/2 dt^2   ]
192         RealMatrix Q = tmp.scalarMultiply(FastMath.pow(accelNoise, 2));
193 
194         // P0 = [ 1 1 ]
195         //      [ 1 1 ]
196         RealMatrix P0 = new Array2DRowRealMatrix(new double[][] { { 1, 1 }, { 1, 1 } });
197 
198         // R = [ measurementNoise^2 ]
199         RealMatrix R = new Array2DRowRealMatrix(
200                 new double[] { FastMath.pow(measurementNoise, 2) });
201 
202         // constant control input, increase velocity by 0.1 m/s per cycle
203         RealVector u = new ArrayRealVector(new double[] { 0.1d });
204 
205         ProcessModel pm = new DefaultProcessModel(A, B, Q, x, P0);
206         MeasurementModel mm = new DefaultMeasurementModel(H, R);
207         KalmanFilter filter = new KalmanFilter(pm, mm);
208 
209         Assert.assertEquals(1, filter.getMeasurementDimension());
210         Assert.assertEquals(2, filter.getStateDimension());
211 
212         assertMatrixEquals(P0.getData(), filter.getErrorCovariance());
213 
214         // check the initial state
215         double[] expectedInitialState = new double[] { 0.0, 0.0 };
216         assertVectorEquals(expectedInitialState, filter.getStateEstimation());
217 
218         final RealDistribution.Sampler rand = new NormalDistribution().createSampler(RandomSource.create(RandomSource.WELL_19937_C));
219 
220         RealVector tmpPNoise = new ArrayRealVector(
221                 new double[] { FastMath.pow(dt, 2d) / 2d, dt });
222 
223         // iterate 60 steps
224         for (int i = 0; i < 60; i++) {
225             filter.predict(u);
226 
227             // Simulate the process
228             RealVector pNoise = tmpPNoise.mapMultiply(accelNoise * rand.sample());
229 
230             // x = A * x + B * u + pNoise
231             x = A.operate(x).add(B.operate(u)).add(pNoise);
232 
233             // Simulate the measurement
234             double mNoise = measurementNoise * rand.sample();
235 
236             // z = H * x + m_noise
237             RealVector z = H.operate(x).mapAdd(mNoise);
238 
239             filter.correct(z);
240 
241             // state estimate shouldn't be larger than the measurement noise
242             double diff = FastMath.abs(x.getEntry(0) - filter.getStateEstimation()[0]);
243             Assert.assertTrue(Precision.compareTo(diff, measurementNoise, 1e-6) < 0);
244         }
245 
246         // error covariance of the velocity should be already very low (< 0.1)
247         Assert.assertTrue(Precision.compareTo(filter.getErrorCovariance()[1][1],
248                                               0.1d, 1e-6) < 0);
249     }

251     /**
252      * Represents an idealized Cannonball only taking into account gravity.
253      */
254     public static class Cannonball {
255 
256         private final double[] gravity = { 0, -9.81 };
257 
258         private final double[] velocity;
259         private final double[] location;
260 
261         private double timeslice;
262 
263         public Cannonball(double timeslice, double angle, double initialVelocity) {
264             this.timeslice = timeslice;
265 
266             final double angleInRadians = FastMath.toRadians(angle);
267             this.velocity = new double[] {
268                     initialVelocity * FastMath.cos(angleInRadians),
269                     initialVelocity * FastMath.sin(angleInRadians)
270             };
271 
272             this.location = new double[] { 0, 0 };
273         }
274 
275         public double getX() {
276             return location[0];
277         }
278 
279         public double getY() {
280             return location[1];
281         }
282 
283         public double getXVelocity() {
284             return velocity[0];
285         }
286 
287         public double getYVelocity() {
288             return velocity[1];
289         }
290 
291         public void step() {
292             // break gravitational force into a smaller time slice.
293             double[] slicedGravity = gravity.clone();
294             for ( int i = 0; i < slicedGravity.length; i++ ) {
295                 slicedGravity[i] *= timeslice;
296             }
297 
298             // apply the acceleration to velocity.
299             double[] slicedVelocity = velocity.clone();
300             for ( int i = 0; i < velocity.length; i++ ) {
301                 velocity[i] += slicedGravity[i];
302                 slicedVelocity[i] = velocity[i] * timeslice;
303                 location[i] += slicedVelocity[i];
304             }
305 
306             // cannonballs shouldn't go into the ground.
307             if ( location[1] < 0 ) {
308                 location[1] = 0;
309             }
310         }
311     }

313     @Test
314     public void testCannonball() {
315         // simulates the flight of a cannonball (only taking gravity and initial thrust into account)
316 
317         // number of iterations
318         final int iterations = 144;
319         // discrete time interval
320         final double dt = 0.1d;
321         // position measurement noise (meter)
322         final double measurementNoise = 30d;
323         // the initial velocity of the cannonball
324         final double initialVelocity = 100;
325         // shooting angle
326         final double angle = 45;
327 
328         final Cannonball cannonball = new Cannonball(dt, angle, initialVelocity);
329 
330         final double speedX = cannonball.getXVelocity();
331         final double speedY = cannonball.getYVelocity();
332 
333         // A = [ 1, dt, 0,  0 ]  =>  x(n+1) = x(n) + vx(n)
334         //     [ 0,  1, 0,  0 ]  => vx(n+1) =        vx(n)
335         //     [ 0,  0, 1, dt ]  =>  y(n+1) =              y(n) + vy(n)
336         //     [ 0,  0, 0,  1 ]  => vy(n+1) =                     vy(n)
337         final RealMatrix A = MatrixUtils.createRealMatrix(new double[][] {
338                 { 1, dt, 0,  0 },
339                 { 0,  1, 0,  0 },
340                 { 0,  0, 1, dt },
341                 { 0,  0, 0,  1 }
342         });
343 
344         // The control vector, which adds acceleration to the kinematic equations.
345         // 0          =>  x(n+1) =  x(n+1)
346         // 0          => vx(n+1) = vx(n+1)
347         // -9.81*dt^2 =>  y(n+1) =  y(n+1) - 1/2 * 9.81 * dt^2
348         // -9.81*dt   => vy(n+1) = vy(n+1) - 9.81 * dt
349         final RealVector controlVector =
350                 MatrixUtils.createRealVector(new double[] { 0, 0, 0.5 * -9.81 * dt * dt, -9.81 * dt } );
351 
352         // The control matrix B only expects y and vy, see control vector
353         final RealMatrix B = MatrixUtils.createRealMatrix(new double[][] {
354                 { 0, 0, 0, 0 },
355                 { 0, 0, 0, 0 },
356                 { 0, 0, 1, 0 },
357                 { 0, 0, 0, 1 }
358         });
359 
360         // We only observe the x/y position of the cannonball
361         final RealMatrix H = MatrixUtils.createRealMatrix(new double[][] {
362                 { 1, 0, 0, 0 },
363                 { 0, 0, 0, 0 },
364                 { 0, 0, 1, 0 },
365                 { 0, 0, 0, 0 }
366         });
367 
368         // our guess of the initial state.
369         final RealVector initialState = MatrixUtils.createRealVector(new double[] { 0, speedX, 0, speedY } );
370 
371         // the initial error covariance matrix, the variance = noise^2
372         final double var = measurementNoise * measurementNoise;
373         final RealMatrix initialErrorCovariance = MatrixUtils.createRealMatrix(new double[][] {
374                 { var,    0,   0,    0 },
375                 {   0, 1e-3,   0,    0 },
376                 {   0,    0, var,    0 },
377                 {   0,    0,   0, 1e-3 }
378         });
379 
380         // we assume no process noise -> zero matrix
381         final RealMatrix Q = MatrixUtils.createRealMatrix(4, 4);
382 
383         // the measurement covariance matrix
384         final RealMatrix R = MatrixUtils.createRealMatrix(new double[][] {
385                 { var,    0,   0,    0 },
386                 {   0, 1e-3,   0,    0 },
387                 {   0,    0, var,    0 },
388                 {   0,    0,   0, 1e-3 }
389         });
390 
391         final ProcessModel pm = new DefaultProcessModel(A, B, Q, initialState, initialErrorCovariance);
392         final MeasurementModel mm = new DefaultMeasurementModel(H, R);
393         final KalmanFilter filter = new KalmanFilter(pm, mm);
394 
395         final RealDistribution.Sampler dist = new NormalDistribution(0, measurementNoise).createSampler(RandomSource.create(RandomSource.WELL_19937_C, 1000));
396 
397         for (int i = 0; i < iterations; i++) {
398             // get the "real" cannonball position
399             double x = cannonball.getX();
400             double y = cannonball.getY();
401 
402             // apply measurement noise to current cannonball position
403             double nx = x + dist.sample();
404             double ny = y + dist.sample();
405 
406             cannonball.step();
407 
408             filter.predict(controlVector);
409             // correct the filter with our measurements
410             filter.correct(new double[] { nx, 0, ny, 0 } );
411 
412             // state estimate shouldn't be larger than the measurement noise
413             double diff = FastMath.abs(cannonball.getY() - filter.getStateEstimation()[2]);
414             Assert.assertTrue(Precision.compareTo(diff, measurementNoise, 1e-6) < 0);
415         }
416 
417         // error covariance of the x/y-position should be already very low (< 3m std dev = 9 variance)
418 
419         Assert.assertTrue(Precision.compareTo(filter.getErrorCovariance()[0][0],
420                                               9, 1e-6) < 0);
421 
422         Assert.assertTrue(Precision.compareTo(filter.getErrorCovariance()[2][2],
423                                               9, 1e-6) < 0);
424     }

426     private void assertVectorEquals(double[] expected, double[] result) {
427         Assert.assertEquals("Wrong number of rows.", expected.length,
428                             result.length);
429         for (int i = 0; i < expected.length; i++) {
430             Assert.assertEquals("Wrong value at position [" + i + "]",
431                                 expected[i], result[i], 1.0e-6);
432         }
433     }
434 
435     private void assertMatrixEquals(double[][] expected, double[][] result) {
436         Assert.assertEquals("Wrong number of rows.", expected.length,
437                             result.length);
438         for (int i = 0; i < expected.length; i++) {
439             Assert.assertEquals("Wrong number of columns.", expected[i].length,
440                                 result[i].length);
441             for (int j = 0; j < expected[i].length; j++) {
442                 Assert.assertEquals("Wrong value at position [" + i + "," + j
443                                     + "]", expected[i][j], result[i][j], 1.0e-6);
444             }
445         }
