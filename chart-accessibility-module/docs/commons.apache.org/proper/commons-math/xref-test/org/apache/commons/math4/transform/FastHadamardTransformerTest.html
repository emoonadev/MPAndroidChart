17  package org.apache.commons.math4.transform;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.transform.FastHadamardTransformer;
21  import org.apache.commons.math4.transform.TransformType;
22  import org.apache.commons.math4.util.Precision;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  
27  /**
28   * JUnit Test for HadamardTransformerTest
29   * @see org.apache.commons.math4.transform.FastHadamardTransformer
30   */
31  public final class FastHadamardTransformerTest {
32  
33      /**
34       * Test of transformer for the a 8-point FHT (means n=8)
35       */
36      @Test
37      public void test8Points() {
38          checkAllTransforms(new int[] { 1, 4, -2, 3, 0, 1, 4, -1 },
39                         new int[] { 10, -4, 2, -4, 2, -12, 6, 8 });
40      }
41  
42      /**
43       * Test of transformer for the a 4-points FHT (means n=4)
44       */
45      @Test
46      public void test4Points() {
47          checkAllTransforms(new int[] { 1, 2, 3, 4 },
48                             new int[] { 10, -2, -4, 0 });
49      }
50  
51      /**
52       * Test the inverse transform of an integer vector is not always an integer vector
53       */
54      @Test
55      public void testNoIntInverse() {
56          FastHadamardTransformer transformer = new FastHadamardTransformer();
57          double[] x = transformer.transform(new double[] { 0, 1, 0, 1}, TransformType.INVERSE);
58          Assert.assertEquals( 0.5, x[0], 0);
59          Assert.assertEquals(-0.5, x[1], 0);
60          Assert.assertEquals( 0.0, x[2], 0);
61          Assert.assertEquals( 0.0, x[3], 0);
62      }
63  
64      /**
65       * Test of transformer for wrong number of points
66       */
67      @Test
68      public void test3Points() {
69          try {
70              new FastHadamardTransformer().transform(new double[3], TransformType.FORWARD);
71              Assert.fail("an exception should have been thrown");
72          } catch (MathIllegalArgumentException iae) {
73              // expected
74          }
75      }
76  
77      private void checkAllTransforms(int[]x, int[] y) {
78          checkDoubleTransform(x, y);
79          checkInverseDoubleTransform(x, y);
80          checkIntTransform(x, y);
81      }
82  
83      private void checkDoubleTransform(int[]x, int[] y) {
84          // Initiate the transformer
85          FastHadamardTransformer transformer = new FastHadamardTransformer();
86  
87          // check double transform
88          double[] dX = new double[x.length];
89          for (int i = 0; i < dX.length; ++i) {
90              dX[i] = x[i];
91          }
92          double dResult[] = transformer.transform(dX, TransformType.FORWARD);
93          for (int i = 0; i < dResult.length; i++) {
94              // compare computed results to precomputed results
95              Assert.assertTrue(Precision.equals(y[i], dResult[i], 1));
96          }
97      }
98  
99      private void checkIntTransform(int[]x, int[] y) {
100         // Initiate the transformer
101         FastHadamardTransformer transformer = new FastHadamardTransformer();
102 
103         // check integer transform
104         int iResult[] = transformer.transform(x);
105         for (int i = 0; i < iResult.length; i++) {
106             // compare computed results to precomputed results
107             Assert.assertEquals(y[i], iResult[i]);
108         }
109 
110     }
111 
112     private void checkInverseDoubleTransform(int[]x, int[] y) {
113         // Initiate the transformer
114         FastHadamardTransformer transformer = new FastHadamardTransformer();
115 
116         // check double transform
117         double[] dY = new double[y.length];
118         for (int i = 0; i < dY.length; ++i) {
119             dY[i] = y[i];
120         }
121         double dResult[] = transformer.transform(dY, TransformType.INVERSE);
122         for (int i = 0; i < dResult.length; i++) {
123             // compare computed results to precomputed results
124             Assert.assertTrue(Precision.equals(x[i], dResult[i], 1));
125         }
126 
127     }
