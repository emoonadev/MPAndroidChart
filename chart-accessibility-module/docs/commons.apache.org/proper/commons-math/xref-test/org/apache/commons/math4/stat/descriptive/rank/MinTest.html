17  package org.apache.commons.math4.stat.descriptive.rank;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.rank.Min;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link UnivariateStatistic} class.
27   */
28  public class MinTest extends StorelessUnivariateStatisticAbstractTest{
29  
30      protected Min stat;
31  
32      /**
33       * {@inheritDoc}
34       */
35      @Override
36      public UnivariateStatistic getUnivariateStatistic() {
37          return new Min();
38      }
39  
40      /**
41       * {@inheritDoc}
42       */
43      @Override
44      public double expectedValue() {
45          return this.min;
46      }
47  
48      @Test
49      public void testSpecialValues() {
50          double[] testArray = {0d, Double.NaN, Double.POSITIVE_INFINITY, Double.NEGATIVE_INFINITY};
51          Min min = new Min();
52          Assert.assertTrue(Double.isNaN(min.getResult()));
53          min.increment(testArray[0]);
54          Assert.assertEquals(0d, min.getResult(), 0);
55          min.increment(testArray[1]);
56          Assert.assertEquals(0d, min.getResult(), 0);
57          min.increment(testArray[2]);
58          Assert.assertEquals(0d, min.getResult(), 0);
59          min.increment(testArray[3]);
60          Assert.assertEquals(Double.NEGATIVE_INFINITY, min.getResult(), 0);
61          Assert.assertEquals(Double.NEGATIVE_INFINITY, min.evaluate(testArray), 0);
62      }
63  
64      @Test
65      public void testNaNs() {
66          Min min = new Min();
67          double nan = Double.NaN;
68          Assert.assertEquals(2d, min.evaluate(new double[]{nan, 2d, 3d}), 0);
69          Assert.assertEquals(1d, min.evaluate(new double[]{1d, nan, 3d}), 0);
70          Assert.assertEquals(1d, min.evaluate(new double[]{1d, 2d, nan}), 0);
71          Assert.assertTrue(Double.isNaN(min.evaluate(new double[]{nan, nan, nan})));
72      }
73  
74  }
