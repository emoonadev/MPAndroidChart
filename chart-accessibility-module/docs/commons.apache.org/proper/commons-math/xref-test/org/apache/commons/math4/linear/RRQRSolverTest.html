120     /** test solve */
121     @Test
122     public void testSolve() {
123         RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
124                 { -102, 12250 }, { 544, 24500 }, { 167, -36750 }
125         });
126         RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
127                 { 1, 2515 }, { 2, 422 }, { -3, 898 }
128         });
129 
130 
131         RRQRDecomposition decomposition = new RRQRDecomposition(MatrixUtils.createRealMatrix(testData3x3NonSingular));
132         DecompositionSolver solver = decomposition.getSolver();
133 
134         // using RealMatrix
135         Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), 3.0e-16 * xRef.getNorm());
136 
137         // using ArrayRealVector
138         for (int i = 0; i < b.getColumnDimension(); ++i) {
139             final RealVector x = solver.solve(b.getColumnVector(i));
140             final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
141             Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
142         }
143 
144         // using RealVector with an alternate implementation
145         for (int i = 0; i < b.getColumnDimension(); ++i) {
146             ArrayRealVectorTest.RealVectorTestImpl v =
147                 new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
148             final RealVector x = solver.solve(v);
149             final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
150             Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
151         }
152 
153     }

155     @Test
156     public void testOverdetermined() {
157         final Random r    = new Random(5559252868205245l);
158         int          p    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
159         int          q    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
160         RealMatrix   a    = createTestMatrix(r, p, q);
161         RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
162 
163         // build a perturbed system: A.X + noise = B
164         RealMatrix b = a.multiply(xRef);
165         final double noise = 0.001;
166         b.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
167             @Override
168             public double visit(int row, int column, double value) {
169                 return value * (1.0 + noise * (2 * r.nextDouble() - 1));
170             }
171         });
172 
173         // despite perturbation, the least square solution should be pretty good
174         RealMatrix x = new RRQRDecomposition(a).getSolver().solve(b);
175         Assert.assertEquals(0, x.subtract(xRef).getNorm(), 0.01 * noise * p * q);
176 
177     }

179     @Test
180     public void testUnderdetermined() {
181         final Random r    = new Random(42185006424567123l);
182         int          p    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
183         int          q    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
184         RealMatrix   a    = createTestMatrix(r, p, q);
185         RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
186         RealMatrix   b    = a.multiply(xRef);
187         RRQRDecomposition rrqrd = new RRQRDecomposition(a);
188         RealMatrix   x = rrqrd.getSolver().solve(b);
189 
190         // too many equations, the system cannot be solved at all
191         Assert.assertTrue(x.subtract(xRef).getNorm() / (p * q) > 0.01);
192 
193         // the last permuted unknown should have been set to 0
194         RealMatrix permuted = rrqrd.getP().transpose().multiply(x);
195         Assert.assertEquals(0.0, permuted.getSubMatrix(p, q - 1, 0, permuted.getColumnDimension() - 1).getNorm(), 0);
196 
197     }
198 
199     private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
200         RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
201         m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
202                 @Override
203                     public double visit(int row, int column, double value) {
204                     return 2.0 * r.nextDouble() - 1.0;
205                 }
206             });
207         return m;
208     }
