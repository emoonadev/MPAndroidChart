171     private static Complex[] createComplexData(final int n) {
172         final Random random = new Random(SEED);
173         final Complex[] data = new Complex[n];
174         for (int i = 0; i < n; i++) {
175             final double re = 2.0 * random.nextDouble() - 1.0;
176             final double im = 2.0 * random.nextDouble() - 1.0;
177             data[i] = new Complex(re, im);
178         }
179         return data;
180     }

182     private static double[] createRealData(final int n) {
183         final Random random = new Random(SEED);
184         final double[] data = new double[n];
185         for (int i = 0; i < n; i++) {
186             data[i] = 2.0 * random.nextDouble() - 1.0;
187         }
188         return data;
189     }

191     /** Naive implementation of DFT, for reference. */
192     private static Complex[] dft(final Complex[] x, final int sgn) {
193         final int n = x.length;
194         final double[] cos = new double[n];
195         final double[] sin = new double[n];
196         final Complex[] y = new Complex[n];
197         for (int i = 0; i < n; i++) {
198             final double arg = 2.0 * FastMath.PI * i / n;
199             cos[i] = FastMath.cos(arg);
200             sin[i] = FastMath.sin(arg);
201         }
202         for (int i = 0; i < n; i++) {
203             double yr = 0.0;
204             double yi = 0.0;
205             for (int j = 0; j < n; j++) {
206                 final int index = (i * j) % n;
207                 final double c = cos[index];
208                 final double s = sin[index];
209                 final double xr = x[j].getReal();
210                 final double xi = x[j].getImaginary();
211                 yr += c * xr - sgn * s * xi;
212                 yi += sgn * s * xr + c * xi;
213             }
214             y[i] = new Complex(yr, yi);
215         }
216         return y;
217     }

219     private static void doTestTransformComplex(final int n, final double tol,
220         final DftNormalization normalization,
221         final TransformType type) {
222         final FastFourierTransformer fft;
223         fft = new FastFourierTransformer(normalization);
224         final Complex[] x = createComplexData(n);
225         final Complex[] expected;
226         final double s;
227         if (type==TransformType.FORWARD) {
228             expected = dft(x, -1);
229             if (normalization == DftNormalization.STANDARD){
230                 s = 1.0;
231             } else {
232                 s = 1.0 / FastMath.sqrt(n);
233             }
234         } else {
235             expected = dft(x, 1);
236             if (normalization == DftNormalization.STANDARD) {
237                 s = 1.0 / n;
238             } else {
239                 s = 1.0 / FastMath.sqrt(n);
240             }
241         }
242         final Complex[] actual = fft.transform(x, type);
243         for (int i = 0; i < n; i++) {
244             final String msg;
245             msg = String.format("%s, %s, %d, %d", normalization, type, n, i);
246             final double re = s * expected[i].getReal();
247             Assert.assertEquals(msg, re, actual[i].getReal(),
248                 tol * FastMath.abs(re));
249             final double im = s * expected[i].getImaginary();
250             Assert.assertEquals(msg, im, actual[i].getImaginary(), tol *
251                 FastMath.abs(re));
252         }
253     }

255     private static void doTestTransformReal(final int n, final double tol,
256         final DftNormalization normalization,
257         final TransformType type) {
258         final FastFourierTransformer fft;
259         fft = new FastFourierTransformer(normalization);
260         final double[] x = createRealData(n);
261         final Complex[] xc = new Complex[n];
262         for (int i = 0; i < n; i++) {
263             xc[i] = new Complex(x[i], 0.0);
264         }
265         final Complex[] expected;
266         final double s;
267         if (type == TransformType.FORWARD) {
268             expected = dft(xc, -1);
269             if (normalization == DftNormalization.STANDARD) {
270                 s = 1.0;
271             } else {
272                 s = 1.0 / FastMath.sqrt(n);
273             }
274         } else {
275             expected = dft(xc, 1);
276             if (normalization == DftNormalization.STANDARD) {
277                 s = 1.0 / n;
278             } else {
279                 s = 1.0 / FastMath.sqrt(n);
280             }
281         }
282         final Complex[] actual = fft.transform(x, type);
283         for (int i = 0; i < n; i++) {
284             final String msg;
285             msg = String.format("%s, %s, %d, %d", normalization, type, n, i);

295     private static void doTestTransformFunction(final UnivariateFunction f,
296         final double min, final double max, int n, final double tol,
297         final DftNormalization normalization,
298         final TransformType type) {
299         final FastFourierTransformer fft;
300         fft = new FastFourierTransformer(normalization);
301         final Complex[] x = new Complex[n];
302         for (int i = 0; i < n; i++) {
303             final double t = min + i * (max - min) / n;
304             x[i] = new Complex(f.value(t));
305         }
306         final Complex[] expected;
307         final double s;
308         if (type == TransformType.FORWARD) {
309             expected = dft(x, -1);
310             if (normalization == DftNormalization.STANDARD) {
311                 s = 1.0;
312             } else {
313                 s = 1.0 / FastMath.sqrt(n);
314             }
315         } else {
316             expected = dft(x, 1);
317             if (normalization == DftNormalization.STANDARD) {
318                 s = 1.0 / n;
319             } else {
320                 s = 1.0 / FastMath.sqrt(n);
321             }
322         }
323         final Complex[] actual = fft.transform(f, min, max, n, type);
324         for (int i = 0; i < n; i++) {
325             final String msg = String.format("%d, %d", n, i);
326             final double re = s * expected[i].getReal();
327             Assert.assertEquals(msg, re, actual[i].getReal(),
328                 tol * FastMath.abs(re));
329             final double im = s * expected[i].getImaginary();
330             Assert.assertEquals(msg, im, actual[i].getImaginary(), tol *
331                 FastMath.abs(re));
332         }
333     }

404      * Test of transformer for the ad hoc data taken from Mathematica.

406     @Test
407     public void testAdHocData() {
408         FastFourierTransformer transformer;
409         transformer = new FastFourierTransformer(DftNormalization.STANDARD);
410         Complex result[]; double tolerance = 1E-12;
411 
412         double x[] = {1.3, 2.4, 1.7, 4.1, 2.9, 1.7, 5.1, 2.7};
413         Complex y[] = {
414             new Complex(21.9, 0.0),
415             new Complex(-2.09497474683058, 1.91507575950825),
416             new Complex(-2.6, 2.7),
417             new Complex(-1.10502525316942, -4.88492424049175),
418             new Complex(0.1, 0.0),
419             new Complex(-1.10502525316942, 4.88492424049175),
420             new Complex(-2.6, -2.7),
421             new Complex(-2.09497474683058, -1.91507575950825)};
422 
423         result = transformer.transform(x, TransformType.FORWARD);
424         for (int i = 0; i < result.length; i++) {
425             Assert.assertEquals(y[i].getReal(), result[i].getReal(), tolerance);
426             Assert.assertEquals(y[i].getImaginary(), result[i].getImaginary(), tolerance);
427         }
428 
429         result = transformer.transform(y, TransformType.INVERSE);
430         for (int i = 0; i < result.length; i++) {
431             Assert.assertEquals(x[i], result[i].getReal(), tolerance);
432             Assert.assertEquals(0.0, result[i].getImaginary(), tolerance);
433         }
434 
435         double x2[] = {10.4, 21.6, 40.8, 13.6, 23.2, 32.8, 13.6, 19.2};
436         TransformUtils.scaleArray(x2, 1.0 / FastMath.sqrt(x2.length));
437         Complex y2[] = y;
438 
439         transformer = new FastFourierTransformer(DftNormalization.UNITARY);
440         result = transformer.transform(y2, TransformType.FORWARD);
441         for (int i = 0; i < result.length; i++) {
442             Assert.assertEquals(x2[i], result[i].getReal(), tolerance);
443             Assert.assertEquals(0.0, result[i].getImaginary(), tolerance);
444         }
445 
446         result = transformer.transform(x2, TransformType.INVERSE);
447         for (int i = 0; i < result.length; i++) {
448             Assert.assertEquals(y2[i].getReal(), result[i].getReal(), tolerance);
449             Assert.assertEquals(y2[i].getImaginary(), result[i].getImaginary(), tolerance);
450         }
451     }

453     /**
454      * Test of transformer for the sine function.
455      */
456     @Test
457     public void testSinFunction() {
458         UnivariateFunction f = new Sin();
459         FastFourierTransformer transformer;
460         transformer = new FastFourierTransformer(DftNormalization.STANDARD);
461         Complex result[]; int N = 1 << 8;
462         double min, max, tolerance = 1E-12;
463 
464         min = 0.0; max = 2.0 * FastMath.PI;
465         result = transformer.transform(f, min, max, N, TransformType.FORWARD);
466         Assert.assertEquals(0.0, result[1].getReal(), tolerance);
467         Assert.assertEquals(-(N >> 1), result[1].getImaginary(), tolerance);
468         Assert.assertEquals(0.0, result[N-1].getReal(), tolerance);
469         Assert.assertEquals(N >> 1, result[N-1].getImaginary(), tolerance);
470         for (int i = 0; i < N-1; i += (i == 0 ? 2 : 1)) {
471             Assert.assertEquals(0.0, result[i].getReal(), tolerance);
472             Assert.assertEquals(0.0, result[i].getImaginary(), tolerance);
473         }
474 
475         min = -FastMath.PI; max = FastMath.PI;
476         result = transformer.transform(f, min, max, N, TransformType.INVERSE);
477         Assert.assertEquals(0.0, result[1].getReal(), tolerance);
478         Assert.assertEquals(-0.5, result[1].getImaginary(), tolerance);
479         Assert.assertEquals(0.0, result[N-1].getReal(), tolerance);
480         Assert.assertEquals(0.5, result[N-1].getImaginary(), tolerance);
481         for (int i = 0; i < N-1; i += (i == 0 ? 2 : 1)) {
482             Assert.assertEquals(0.0, result[i].getReal(), tolerance);
483             Assert.assertEquals(0.0, result[i].getImaginary(), tolerance);
484         }
