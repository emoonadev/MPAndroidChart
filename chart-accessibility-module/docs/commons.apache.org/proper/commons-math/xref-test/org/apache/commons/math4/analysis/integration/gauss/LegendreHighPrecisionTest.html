17  package org.apache.commons.math4.analysis.integration.gauss;
18  
19  import org.apache.commons.math4.analysis.UnivariateFunction;
20  import org.apache.commons.math4.analysis.function.Cos;
21  import org.apache.commons.math4.analysis.function.Inverse;
22  import org.apache.commons.math4.analysis.function.Log;
23  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegrator;
24  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegratorFactory;
25  import org.apache.commons.math4.analysis.integration.gauss.LegendreHighPrecisionRuleFactory;
26  import org.junit.Test;
27  import org.junit.Assert;
28  
29  /**
30   * Test of the {@link LegendreHighPrecisionRuleFactory}.
31   *
32   */
33  public class LegendreHighPrecisionTest {
34      private static GaussIntegratorFactory factory = new GaussIntegratorFactory();
35  
36      @Test
37      public void testCos() {
38          final UnivariateFunction cos = new Cos();
39  
40          final GaussIntegrator integrator = factory.legendreHighPrecision(7, 0, Math.PI / 2);
41          final double s = integrator.integrate(cos);
42          // System.out.println("s=" + s + " e=" + 1);
43          Assert.assertEquals(1, s, Math.ulp(1d));
44      }
45  
46  
47      @Test
48      public void testInverse() {
49          final UnivariateFunction inv = new Inverse();
50          final UnivariateFunction log = new Log();
51  
52          final double lo = 12.34;
53          final double hi = 456.78;
54  
55          final GaussIntegrator integrator = factory.legendreHighPrecision(60, lo, hi);
56          final double s = integrator.integrate(inv);
57          final double expected = log.value(hi) - log.value(lo);
58          // System.out.println("s=" + s + " e=" + expected);
59          Assert.assertEquals(expected, s, 1e-15);
60      }
61  }
