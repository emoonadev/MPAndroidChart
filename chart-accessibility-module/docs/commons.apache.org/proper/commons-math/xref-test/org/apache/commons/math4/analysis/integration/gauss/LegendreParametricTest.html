17  package org.apache.commons.math4.analysis.integration.gauss;
18  
19  import java.util.ArrayList;
20  import java.util.Collection;
21  
22  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegratorFactory;
23  import org.apache.commons.math4.analysis.integration.gauss.LegendreRuleFactory;
24  import org.junit.runner.RunWith;
25  import org.junit.runners.Parameterized;
26  import org.junit.runners.Parameterized.Parameters;
27  
28  /**
29   * Test of the {@link LegendreRuleFactory}.
30   * This parameterized test extends the standard test for Gaussian quadrature
31   * rule, where each monomial is tested in turn.
32   * Parametrization allows to test automatically 0, 1, ... , {@link #MAX_NUM_POINTS}
33   * quadrature rules.
34   *
35   */
36  @RunWith(value=Parameterized.class)
37  public class LegendreParametricTest extends GaussianQuadratureAbstractTest {
38      private static final GaussIntegratorFactory factory = new GaussIntegratorFactory();
39  
40      /**
41       * The highest order quadrature rule to be tested.
42       */
43      public static final int MAX_NUM_POINTS = 30;
44  
45      /**
46       * Creates a new instance of this test, with the specified number of nodes
47       * for the Gauss-Legendre quadrature rule.
48       *
49       * @param numberOfPoints Order of integration rule.
50       * @param maxDegree Maximum degree of monomials to be tested.
51       * @param eps Value of &epsilon;.
52       * @param numUlps Value of the maximum relative error (in ulps).
53       */
54      public LegendreParametricTest(int numberOfPoints,
55                                    int maxDegree,
56                                    double eps,
57                                    double numUlps) {
58          super(factory.legendre(numberOfPoints),
59                maxDegree, eps, numUlps);
60      }
61  
62      /**
63       * Returns the collection of parameters to be passed to the constructor of
64       * this class.
65       * Gauss-Legendre quadrature rules of order 1, ..., {@link #MAX_NUM_POINTS}
66       * will be constructed.
67       *
68       * @return the collection of parameters for this parameterized test.
69       */
70      @SuppressWarnings("boxing") // OK here
71      @Parameters
72      public static Collection<Object[]> getParameters() {
73          final ArrayList<Object[]> parameters = new ArrayList<Object[]>();
74          for (int k = 1; k <= MAX_NUM_POINTS; k++) {
75              parameters.add(new Object[] { k, 2 * k - 1, Math.ulp(1d), 91d });
76          }
77          return parameters;
78      }
79  
80      @Override
81      public double getExpectedValue(final int n) {
82          if (n % 2 == 1) {
83              return 0;
84          }
85          return 2d / (n + 1);
86      }
87  }
