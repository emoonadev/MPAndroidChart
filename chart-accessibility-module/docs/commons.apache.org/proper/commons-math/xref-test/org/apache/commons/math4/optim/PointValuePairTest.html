17  package org.apache.commons.math4.optim;
18  
19  import org.apache.commons.math4.TestUtils;
20  import org.apache.commons.math4.optim.PointValuePair;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  public class PointValuePairTest {
25      @Test
26      public void testSerial() {
27          PointValuePair pv1 = new PointValuePair(new double[] { 1.0, 2.0, 3.0 }, 4.0);
28          PointValuePair pv2 = (PointValuePair) TestUtils.serializeAndRecover(pv1);
29          Assert.assertEquals(pv1.getKey().length, pv2.getKey().length);
30          for (int i = 0; i < pv1.getKey().length; ++i) {
31              Assert.assertEquals(pv1.getKey()[i], pv2.getKey()[i], 1.0e-15);
32          }
33          Assert.assertEquals(pv1.getValue(), pv2.getValue(), 1.0e-15);
34      }
35  
36  }
