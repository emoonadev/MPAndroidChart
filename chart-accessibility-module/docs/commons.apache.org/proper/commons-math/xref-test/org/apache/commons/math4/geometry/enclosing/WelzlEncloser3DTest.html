105     @Test
106     public void testLargeSamples() throws IOException {
107         final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
108                                                                  0x35ddecfc78131e1dl);
109         final UnitSphereRandomVectorGenerator sr = new UnitSphereRandomVectorGenerator(3, random);
110         for (int k = 0; k < 50; ++k) {
111 
112             // define the reference sphere we want to compute
113             double d = 25 * random.nextDouble();
114             double refRadius = 10 * random.nextDouble();
115             Vector3D refCenter = new Vector3D(d, new Vector3D(sr.nextVector()));
116             // set up a large sample inside the reference sphere
117             int nbPoints = random.nextInt(1000);
118             List<Vector3D> points = new ArrayList<Vector3D>();
119             for (int i = 0; i < nbPoints; ++i) {
120                 double r = refRadius * random.nextDouble();
121                 points.add(new Vector3D(1.0, refCenter, r, new Vector3D(sr.nextVector())));
122             }
123 
124             // test we find a sphere at most as large as the one used for random drawings
125             checkSphere(points, refRadius);
126 
127         }
128     }

130     private void checkSphere(List<Vector3D> points, double refRadius) {
131 
132         EnclosingBall<Euclidean3D, Vector3D> sphere = checkSphere(points);
133 
134         // compare computed sphere with bounding sphere
135         Assert.assertTrue(sphere.getRadius() <= refRadius);
136 
137         // check removing any point of the support Sphere fails to enclose the point
138         for (int i = 0; i < sphere.getSupportSize(); ++i) {
139             List<Vector3D> reducedSupport = new ArrayList<Vector3D>();
140             int count = 0;
141             for (Vector3D s : sphere.getSupport()) {
142                 if (count++ != i) {
143                     reducedSupport.add(s);
144                 }
145             }
146             EnclosingBall<Euclidean3D, Vector3D> reducedSphere =
147                     new SphereGenerator().ballOnSupport(reducedSupport);
148             boolean foundOutside = false;
149             for (int j = 0; j < points.size() && !foundOutside; ++j) {
150                 if (!reducedSphere.contains(points.get(j), 1.0e-10)) {
151                     foundOutside = true;
152                 }
153             }
154             Assert.assertTrue(foundOutside);
155         }
156 
157     }
158 
159     private EnclosingBall<Euclidean3D, Vector3D> checkSphere(List<Vector3D> points) {
160 
161         WelzlEncloser<Euclidean3D, Vector3D> encloser =
162                 new WelzlEncloser<Euclidean3D, Vector3D>(1.0e-10, new SphereGenerator());
163         EnclosingBall<Euclidean3D, Vector3D> Sphere = encloser.enclose(points);
164 
165         // all points are enclosed
166         for (Vector3D v : points) {
167             Assert.assertTrue(Sphere.contains(v, 1.0e-10));
168         }
169 
170         for (Vector3D v : points) {
171             boolean inSupport = false;
172             for (Vector3D s : Sphere.getSupport()) {
173                 if (v == s) {
174                     inSupport = true;
175                 }
176             }
177             if (inSupport) {
178                 // points on the support should be outside of reduced ball
179                 Assert.assertFalse(Sphere.contains(v, -0.001));
180             }
181         }
182 
183         return Sphere;
184 
185     }
