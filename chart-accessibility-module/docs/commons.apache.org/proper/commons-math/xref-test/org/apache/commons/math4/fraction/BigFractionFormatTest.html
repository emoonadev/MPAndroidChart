108     @Test

111

113             BigFraction c = properFormat.parse(source);
114             Assert.assertNotNull(c);
115             Assert.assertEquals(BigInteger.ONE, c.getNumerator());

117

261     @Test
262     public void testParseBig() {
263         BigFraction f1 =
264             improperFormat.parse("167213075789791382630275400487886041651764456874403" +
265                                  " / " +
266                                  "53225575123090058458126718248444563466137046489291");
267         Assert.assertEquals(FastMath.PI, f1.doubleValue(), 0.0);
268         BigFraction f2 =
269             properFormat.parse("3 " +
270                                "7536350420521207255895245742552351253353317406530" +
271                                " / " +
272                                "53225575123090058458126718248444563466137046489291");
273         Assert.assertEquals(FastMath.PI, f2.doubleValue(), 0.0);
274         Assert.assertEquals(f1, f2);
275         BigDecimal pi =
276             new BigDecimal("3.141592653589793238462643383279502884197169399375105820974944592307816406286208998628034825342117068");
277         Assert.assertEquals(pi, f1.bigDecimalValue(99, BigDecimal.ROUND_HALF_EVEN));
278     }

280     @Test
281     public void testNumeratorFormat() {
282         NumberFormat old = properFormat.getNumeratorFormat();
283         NumberFormat nf = NumberFormat.getInstance();
284         nf.setParseIntegerOnly(true);
285         properFormat.setNumeratorFormat(nf);
286         Assert.assertEquals(nf, properFormat.getNumeratorFormat());
287         properFormat.setNumeratorFormat(old);
288 
289         old = improperFormat.getNumeratorFormat();
290         nf = NumberFormat.getInstance();
291         nf.setParseIntegerOnly(true);
292         improperFormat.setNumeratorFormat(nf);
293         Assert.assertEquals(nf, improperFormat.getNumeratorFormat());
294         improperFormat.setNumeratorFormat(old);
295     }
