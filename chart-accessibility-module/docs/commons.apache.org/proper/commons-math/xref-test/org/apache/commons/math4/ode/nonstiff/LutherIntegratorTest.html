46      @Test
47      public void testMissedEndEvent()
48              throws DimensionMismatchException, NumberIsTooSmallException,
49              MaxCountExceededException, NoBracketingException {
50          final double   t0     = 1878250320.0000029;
51          final double   tEvent = 1878250379.9999986;
52          final double[] k      = { 1.0e-4, 1.0e-5, 1.0e-6 };
53          FirstOrderDifferentialEquations ode = new FirstOrderDifferentialEquations() {
54  
55              public int getDimension() {
56                  return k.length;
57              }
58  
59              public void computeDerivatives(double t, double[] y, double[] yDot) {
60                  for (int i = 0; i < y.length; ++i) {
61                      yDot[i] = k[i] * y[i];
62                  }
63              }
64          };
65  
66          LutherIntegrator integrator = new LutherIntegrator(60.0);
67  
68          double[] y0   = new double[k.length];
69          for (int i = 0; i < y0.length; ++i) {
70              y0[i] = i + 1;
71          }
72          double[] y    = new double[k.length];
73  
74          double finalT = integrator.integrate(ode, t0, y0, tEvent, y);
75          Assert.assertEquals(tEvent, finalT, 1.0e-15);
76          for (int i = 0; i < y.length; ++i) {
77              Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-15);
78          }
79  
80          integrator.addEventHandler(new EventHandler() {
81  
82              public void init(double t0, double[] y0, double t) {
83              }
84  
85              public void resetState(double t, double[] y) {
86              }
87  
88              public double g(double t, double[] y) {
89                  return t - tEvent;
90              }
91  
92              public Action eventOccurred(double t, double[] y, boolean increasing) {
93                  Assert.assertEquals(tEvent, t, 1.0e-15);
94                  return Action.CONTINUE;
95              }
96          }, Double.POSITIVE_INFINITY, 1.0e-20, 100);
97          finalT = integrator.integrate(ode, t0, y0, tEvent + 120, y);
98          Assert.assertEquals(tEvent + 120, finalT, 1.0e-15);
99          for (int i = 0; i < y.length; ++i) {
100             Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-15);
101         }
102 
103     }

135     @Test
136     public void testDecreasingSteps()
137             throws DimensionMismatchException, NumberIsTooSmallException,
138             MaxCountExceededException, NoBracketingException {
139 
140         for (TestProblemAbstract pb : new TestProblemAbstract[] {
141             new TestProblem1(), new TestProblem2(), new TestProblem3(),
142             new TestProblem4(), new TestProblem5(), new TestProblem6()
143         }) {
144 
145             double previousValueError = Double.NaN;
146             double previousTimeError = Double.NaN;
147             for (int i = 4; i < 10; ++i) {
148 
149                 double step = (pb.getFinalTime() - pb.getInitialTime()) * FastMath.pow(2.0, -i);
150 
151                 FirstOrderIntegrator integ = new LutherIntegrator(step);
152                 TestProblemHandler handler = new TestProblemHandler(pb, integ);
153                 integ.addStepHandler(handler);
154                 EventHandler[] functions = pb.getEventsHandlers();
155                 for (int l = 0; l < functions.length; ++l) {
156                     integ.addEventHandler(functions[l],
157                                           Double.POSITIVE_INFINITY, 1.0e-6 * step, 1000);
158                 }
159                 Assert.assertEquals(functions.length, integ.getEventHandlers().size());
160                 double stopTime = integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
161                                                   pb.getFinalTime(), new double[pb.getDimension()]);
162                 if (functions.length == 0) {
163                     Assert.assertEquals(pb.getFinalTime(), stopTime, 1.0e-10);
164                 }
165 
166                 double error = handler.getMaximalValueError();
167                 if (i > 4) {
168                     Assert.assertTrue(error < 1.01 * FastMath.abs(previousValueError));
169                 }
170                 previousValueError = error;
171 
172                 double timeError = handler.getMaximalTimeError();
173                 if (i > 4) {
174                     Assert.assertTrue(timeError <= FastMath.abs(previousTimeError));
175                 }
176                 previousTimeError = timeError;
177 
178                 integ.clearEventHandlers();
179                 Assert.assertEquals(0, integ.getEventHandlers().size());
180             }
181 
182         }
183

186     @Test
187     public void testSmallStep()
188             throws DimensionMismatchException, NumberIsTooSmallException,
189             MaxCountExceededException, NoBracketingException {
190 
191         TestProblem1 pb = new TestProblem1();
192         double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
193 
194         FirstOrderIntegrator integ = new LutherIntegrator(step);
195         TestProblemHandler handler = new TestProblemHandler(pb, integ);
196         integ.addStepHandler(handler);
197         integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
198                         pb.getFinalTime(), new double[pb.getDimension()]);
199 
200         Assert.assertTrue(handler.getLastError() < 9.0e-17);
201         Assert.assertTrue(handler.getMaximalValueError() < 4.0e-15);
202         Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
203         Assert.assertEquals("Luther", integ.getName());
204     }

246     @Test
247     public void testKepler()
248             throws DimensionMismatchException, NumberIsTooSmallException,
249             MaxCountExceededException, NoBracketingException {
250 
251         final TestProblem3 pb  = new TestProblem3(0.9);
252         double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
253 
254         FirstOrderIntegrator integ = new LutherIntegrator(step);
255         integ.addStepHandler(new KeplerHandler(pb));
256         integ.integrate(pb,
257                         pb.getInitialTime(), pb.getInitialState(),
258                         pb.getFinalTime(), new double[pb.getDimension()]);
259     }
260 
261     private static class KeplerHandler implements StepHandler {
262         public KeplerHandler(TestProblem3 pb) {
263             this.pb = pb;
264             maxError = 0;
265         }
266         public void init(double t0, double[] y0, double t) {
267             maxError = 0;
268         }
269         public void handleStep(StepInterpolator interpolator, boolean isLast) {
270 
271             double[] interpolatedY = interpolator.getInterpolatedState ();
272             double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getCurrentTime());
273             double dx = interpolatedY[0] - theoreticalY[0];
274             double dy = interpolatedY[1] - theoreticalY[1];
275             double error = dx * dx + dy * dy;
276             if (error > maxError) {
277                 maxError = error;
278             }
279             if (isLast) {
280                 Assert.assertTrue(maxError < 2.2e-7);
281             }
282         }
283         private double maxError = 0;
284         private TestProblem3 pb;
285     }

288     public void testStepSize()
289             throws DimensionMismatchException, NumberIsTooSmallException,
290             MaxCountExceededException, NoBracketingException {
291         final double step = 1.23456;
292         FirstOrderIntegrator integ = new LutherIntegrator(step);
293         integ.addStepHandler(new StepHandler() {
294             public void handleStep(StepInterpolator interpolator, boolean isLast) {
295                 if (! isLast) {
296                     Assert.assertEquals(step,
297                                         interpolator.getCurrentTime() - interpolator.getPreviousTime(),
298                                         1.0e-12);
299                 }
300             }
301             public void init(double t0, double[] y0, double t) {
302             }
303         });
304         integ.integrate(new FirstOrderDifferentialEquations() {
305             public void computeDerivatives(double t, double[] y, double[] dot) {
306                 dot[0] = 1.0;
307             }
308             public int getDimension() {
309                 return 1;
310             }
311         }, 0.0, new double[] { 0.0 }, 5.0, new double[1]);
312     }

314     @Test
315     public void testSingleStep() {
316 
317         final TestProblem3 pb  = new TestProblem3(0.9);
318         double h = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
319 
320         RungeKuttaIntegrator integ = new LutherIntegrator(Double.NaN);
321         double   t = pb.getInitialTime();
322         double[] y = pb.getInitialState();
323         for (int i = 0; i < 100; ++i) {
324             y = integ.singleStep(pb, t, y, t + h);
325             t += h;
326         }
327         double[] yth = pb.computeTheoreticalState(t);
328         double dx = y[0] - yth[0];
329         double dy = y[1] - yth[1];
330         double error = dx * dx + dy * dy;
331         Assert.assertEquals(0.0, error, 1.0e-11);
332     }
