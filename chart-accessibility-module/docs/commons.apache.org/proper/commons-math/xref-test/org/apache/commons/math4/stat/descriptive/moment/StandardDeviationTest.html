17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.moment.Mean;
22  import org.apache.commons.math4.stat.descriptive.moment.SecondMoment;
23  import org.apache.commons.math4.stat.descriptive.moment.StandardDeviation;
24  import org.apache.commons.math4.util.FastMath;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  /**
29   * Test cases for the {@link UnivariateStatistic} class.
30   *
31   */
32  public class StandardDeviationTest extends StorelessUnivariateStatisticAbstractTest{
33  
34      protected StandardDeviation stat;
35  
36      /**
37       * {@inheritDoc}
38       */
39      @Override
40      public UnivariateStatistic getUnivariateStatistic() {
41          return new StandardDeviation();
42      }
43  
44      /**
45       * {@inheritDoc}
46       */
47      @Override
48      public double expectedValue() {
49          return this.std;
50      }
51  
52      /**
53       * Make sure Double.NaN is returned iff n = 0
54       *
55       */
56      @Test
57      public void testNaN() {
58          StandardDeviation std = new StandardDeviation();
59          Assert.assertTrue(Double.isNaN(std.getResult()));
60          std.increment(1d);
61          Assert.assertEquals(0d, std.getResult(), 0);
62      }
63  
64      /**
65       * Test population version of variance
66       */
67      @Test
68      public void testPopulation() {
69          double[] values = {-1.0d, 3.1d, 4.0d, -2.1d, 22d, 11.7d, 3d, 14d};
70          double sigma = populationStandardDeviation(values);
71          SecondMoment m = new SecondMoment();
72          m.incrementAll(values);  // side effect is to add values
73          StandardDeviation s1 = new StandardDeviation();
74          s1.setBiasCorrected(false);
75          Assert.assertEquals(sigma, s1.evaluate(values), 1E-14);
76          s1.incrementAll(values);
77          Assert.assertEquals(sigma, s1.getResult(), 1E-14);
78          s1 = new StandardDeviation(false, m);
79          Assert.assertEquals(sigma, s1.getResult(), 1E-14);
80          s1 = new StandardDeviation(false);
81          Assert.assertEquals(sigma, s1.evaluate(values), 1E-14);
82          s1.incrementAll(values);
83          Assert.assertEquals(sigma, s1.getResult(), 1E-14);
84      }
85  
86      /**
87       * Definitional formula for population standard deviation
88       */
89      protected double populationStandardDeviation(double[] v) {
90          double mean = new Mean().evaluate(v);
91          double sum = 0;
92          for (int i = 0; i < v.length; i++) {
93              sum += (v[i] - mean) * (v[i] - mean);
94          }
95          return FastMath.sqrt(sum / v.length);
96      }
