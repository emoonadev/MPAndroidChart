153     /**
154      * Tests consistency of weighted statistic computation.
155      * For statistics that support weighted evaluation, this test case compares
156      * the result of direct computation on an array with repeated values with
157      * a weighted computation on the corresponding (shorter) array with each
158      * value appearing only once but with a weight value equal to its multiplicity
159      * in the repeating array.
160      */
161 
162     @Test
163     public void testWeightedConsistency() {
164 
165         // See if this statistic computes weighted statistics
166         // If not, skip this test
167         UnivariateStatistic statistic = getUnivariateStatistic();
168         if (!(statistic instanceof WeightedEvaluation)) {
169             return;
170         }
171 
172         // Create arrays of values and corresponding integral weights
173         // and longer array with values repeated according to the weights
174         final int len = 10;        // length of values array
175         final double mu = 0;       // mean of test data
176         final double sigma = 5;    // std dev of test data
177         double[] values = new double[len];
178         double[] weights = new double[len];
179 
180         // Fill weights array with random int values between 1 and 5
181         int[] intWeights = new int[len];
182         final IntegerDistribution.Sampler weightDist =
183             new UniformIntegerDistribution(1, 5).createSampler(RandomSource.create(RandomSource.WELL_512_A,
184                                                                                    234878544L));
185         for (int i = 0; i < len; i++) {
186             intWeights[i] = weightDist.sample();
187             weights[i] = intWeights[i];
188         }
189 
190         // Fill values array with random data from N(mu, sigma)
191         // and fill valuesList with values from values array with
192         // values[i] repeated weights[i] times, each i
193         final RealDistribution.Sampler valueDist =
194             new NormalDistribution(mu, sigma).createSampler(RandomSource.create(RandomSource.WELL_512_A,
195                                                                                 64925784252L));
196         List<Double> valuesList = new ArrayList<Double>();
197         for (int i = 0; i < len; i++) {
198             double value = valueDist.sample();
199             values[i] = value;
200             for (int j = 0; j < intWeights[i]; j++) {
201                 valuesList.add(new Double(value));
202             }
203         }
204 
205         // Dump valuesList into repeatedValues array
206         int sumWeights = valuesList.size();
207         double[] repeatedValues = new double[sumWeights];
208         for (int i = 0; i < sumWeights; i++) {
209             repeatedValues[i] = valuesList.get(i);
210         }
211 
212         // Compare result of weighted statistic computation with direct computation
213         // on array of repeated values
214         WeightedEvaluation weightedStatistic = (WeightedEvaluation) statistic;
215         TestUtils.assertRelativelyEquals(statistic.evaluate(repeatedValues),
216                 weightedStatistic.evaluate(values, weights, 0, values.length),
217                 10E-12);
218 
219         // Check consistency of weighted evaluation methods
220         Assert.assertEquals(weightedStatistic.evaluate(values, weights, 0, values.length),
221                 weightedStatistic.evaluate(values, weights), Double.MIN_VALUE);
222 
223     }
