18  package org.apache.commons.math4.dfp;
19  
20  import org.apache.commons.math4.dfp.Dfp;
21  import org.apache.commons.math4.dfp.DfpDec;
22  import org.apache.commons.math4.dfp.DfpField;
23  
24  public class Decimal10 extends DfpDec {
25  
26      Decimal10(final DfpField factory) {
27          super(factory);
28      }
29  
30      Decimal10(final DfpField factory, final byte x) {
31          super(factory, x);
32      }
33  
34      Decimal10(final DfpField factory, final int x) {
35          super(factory, x);
36      }
37  
38      Decimal10(final DfpField factory, final long x) {
39          super(factory, x);
40      }
41  
42      Decimal10(final DfpField factory, final double x) {
43          super(factory, x);
44      }
45  
46      public Decimal10(final Dfp d) {
47          super(d);
48      }
49  
50      public Decimal10(final DfpField factory, final String s) {
51          super(factory, s);
52      }
53  
54      protected Decimal10(final DfpField factory, final byte sign, final byte nans) {
55          super(factory, sign, nans);
56      }
57  
58      @Override
59      public Dfp newInstance() {
60          return new Decimal10(getField());
61      }
62  
63      @Override
64      public Dfp newInstance(final byte x) {
65          return new Decimal10(getField(), x);
66      }
67  
68      @Override
69      public Dfp newInstance(final int x) {
70          return new Decimal10(getField(), x);
71      }
72  
73      @Override
74      public Dfp newInstance(final long x) {
75          return new Decimal10(getField(), x);
76      }
77  
78      @Override
79      public Dfp newInstance(final double x) {
80          return new Decimal10(getField(), x);
81      }
82  
83      @Override
84      public Dfp newInstance(final Dfp d) {
85          return new Decimal10(d);
86      }
87  
88      @Override
89      public Dfp newInstance(final String s) {
90          return new Decimal10(getField(), s);
91      }
92  
93      @Override
94      public Dfp newInstance(final byte sign, final byte nans) {
95          return new Decimal10(getField(), sign, nans);
96      }
97  
98      @Override
99      protected int getDecimalDigits() {
100         return 10;
101     }
102 
103 }
