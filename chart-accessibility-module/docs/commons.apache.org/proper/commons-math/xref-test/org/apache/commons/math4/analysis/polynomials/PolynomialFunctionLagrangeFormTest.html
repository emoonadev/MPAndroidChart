24  /**
25   * Test case for Lagrange form of polynomial function.
26   * <p>
27   * We use n+1 points to interpolate a polynomial of degree n. This should
28   * give us the exact same polynomial as result. Thus we can use a very
29   * small tolerance to account only for round-off errors.
30   *
31   */
32  public final class PolynomialFunctionLagrangeFormTest {
33  
34      /**
35       * Test of polynomial for the linear function.
36       */
37      @Test
38      public void testLinearFunction() {
39          PolynomialFunctionLagrangeForm p;
40          double c[], z, expected, result, tolerance = 1E-12;
41  
42          // p(x) = 1.5x - 4
43          double x[] = { 0.0, 3.0 };
44          double y[] = { -4.0, 0.5 };
45          p = new PolynomialFunctionLagrangeForm(x, y);
46  
47          z = 2.0; expected = -1.0; result = p.value(z);
48          Assert.assertEquals(expected, result, tolerance);
49  
50          z = 4.5; expected = 2.75; result = p.value(z);
51          Assert.assertEquals(expected, result, tolerance);
52  
53          z = 6.0; expected = 5.0; result = p.value(z);
54          Assert.assertEquals(expected, result, tolerance);
55  
56          Assert.assertEquals(1, p.degree());
57  
58          c = p.getCoefficients();
59          Assert.assertEquals(2, c.length);
60          Assert.assertEquals(-4.0, c[0], tolerance);
61          Assert.assertEquals(1.5, c[1], tolerance);
62      }
63  
64      /**
65       * Test of polynomial for the quadratic function.
66       */
67      @Test
68      public void testQuadraticFunction() {
69          PolynomialFunctionLagrangeForm p;
70          double c[], z, expected, result, tolerance = 1E-12;
71  
72          // p(x) = 2x^2 + 5x - 3 = (2x - 1)(x + 3)
73          double x[] = { 0.0, -1.0, 0.5 };
74          double y[] = { -3.0, -6.0, 0.0 };
75          p = new PolynomialFunctionLagrangeForm(x, y);
76  
77          z = 1.0; expected = 4.0; result = p.value(z);
78          Assert.assertEquals(expected, result, tolerance);
79  
80          z = 2.5; expected = 22.0; result = p.value(z);
81          Assert.assertEquals(expected, result, tolerance);
82  
83          z = -2.0; expected = -5.0; result = p.value(z);
84          Assert.assertEquals(expected, result, tolerance);
85  
86          Assert.assertEquals(2, p.degree());
87  
88          c = p.getCoefficients();
89          Assert.assertEquals(3, c.length);
90          Assert.assertEquals(-3.0, c[0], tolerance);
91          Assert.assertEquals(5.0, c[1], tolerance);
92          Assert.assertEquals(2.0, c[2], tolerance);
93      }
94  
95      /**
96       * Test of polynomial for the quintic function.
97       */
98      @Test
99      public void testQuinticFunction() {
100         PolynomialFunctionLagrangeForm p;
101         double c[], z, expected, result, tolerance = 1E-12;
102 
103         // p(x) = x^5 - x^4 - 7x^3 + x^2 + 6x = x(x^2 - 1)(x + 2)(x - 3)
104         double x[] = { 1.0, -1.0, 2.0, 3.0, -3.0, 0.5 };
105         double y[] = { 0.0, 0.0, -24.0, 0.0, -144.0, 2.34375 };
106         p = new PolynomialFunctionLagrangeForm(x, y);
107 
108         z = 0.0; expected = 0.0; result = p.value(z);
109         Assert.assertEquals(expected, result, tolerance);
110 
111         z = -2.0; expected = 0.0; result = p.value(z);
112         Assert.assertEquals(expected, result, tolerance);
113 
114         z = 4.0; expected = 360.0; result = p.value(z);
115         Assert.assertEquals(expected, result, tolerance);
116 
117         Assert.assertEquals(5, p.degree());
118 
119         c = p.getCoefficients();
120         Assert.assertEquals(6, c.length);
121         Assert.assertEquals(0.0, c[0], tolerance);
122         Assert.assertEquals(6.0, c[1], tolerance);
123         Assert.assertEquals(1.0, c[2], tolerance);
124         Assert.assertEquals(-7.0, c[3], tolerance);
125         Assert.assertEquals(-1.0, c[4], tolerance);
126         Assert.assertEquals(1.0, c[5], tolerance);
127     }

148             new PolynomialFunctionLagrangeForm(x, y);
149             Assert.fail("Expecting MathIllegalArgumentException - mismatch input arrays");
150         } catch (MathIllegalArgumentException ex) {
151             // expected
152         }
153     }
