102     private static void reportFailedResults(Method mathMethod, Object[] params, Object expected, Object actual, int[] entries){
103         final String methodName = mathMethod.getName();
104         String format = null;
105         long actL=0;
106         long expL=0;
107         if (expected instanceof Double) {
108             Double exp = (Double) expected;
109             Double act = (Double) actual;
110             if (isNumber(exp) && isNumber(act) && exp != 0) { // show difference as hex
111                 actL = Double.doubleToLongBits(act);
112                 expL = Double.doubleToLongBits(exp);
113                 if (Math.abs(actL-expL)==1) {
114                     // Not 100% sure off-by-one errors are allowed everywhere, so only allow for these methods
115                     if (methodName.equals("toRadians") || methodName.equals("atan2")) {
116                         return;
117                     }
118                 }
119                 format = "%016x";
120             }
121         } else if (expected instanceof Float ){
122             Float exp = (Float) expected;
123             Float act = (Float) actual;
124             if (isNumber(exp) && isNumber(act) && exp != 0) { // show difference as hex
125                 actL = Float.floatToIntBits(act);
126                 expL = Float.floatToIntBits(exp);
127                 format = "%08x";
128             }
129         }
130         StringBuilder sb = new StringBuilder();
131         sb.append(mathMethod.getReturnType().getSimpleName());
132         sb.append(" ");
133         sb.append(methodName);
134         sb.append("(");
135         String sep = "";
136         for(Object o : params){
137             sb.append(sep);
138             sb.append(o);
139             sep=", ";
140         }
141         sb.append(") expected ");
142         if (format != null){
143             sb.append(String.format(format, expL));
144         } else {
145             sb.append(expected);
146         }
147         sb.append(" actual ");
148         if (format != null){
149             sb.append(String.format(format, actL));
150         } else {
151             sb.append(actual);
152         }
153         sb.append(" entries ");
154         sb.append(Arrays.toString(entries));
155         String message = sb.toString();
156         final boolean fatal = true;
157         if (fatal) {
158             Assert.fail(message);
159         } else {
160             System.out.println(message);
161         }
162     }

164     private static void callMethods(Method mathMethod, Method fastMethod,
165             Object[] params, int[] entries) throws IllegalAccessException {
166         try {
167             Object expected;
168             try {
169                 expected = mathMethod.invoke(mathMethod, params);
170             } catch (InvocationTargetException ite) {
171                 expected = ite.getCause();
172             }
173             Object actual;
174             try {
175                 actual = fastMethod.invoke(mathMethod, params);
176             } catch (InvocationTargetException ite) {
177                 actual = ite.getCause();
178             }
179             if (expected instanceof ArithmeticException) {
180                 Assert.assertEquals(MathArithmeticException.class, actual.getClass());
181             } else  if (!expected.equals(actual)) {
182                 reportFailedResults(mathMethod, params, expected, actual, entries);
183             }
184         } catch (IllegalArgumentException e) {
185             Assert.fail(mathMethod+" "+e);
186         }
187     }

189     private static void setupMethodCall(Method mathMethod, Method fastMethod,
190             Type[] types, Object[][] valueArrays) throws Exception {
191         Object[] params = new Object[types.length];
192         int entry1 = 0;
193         int[] entries = new int[types.length];
194         for(Object d : valueArrays[0]) {
195             entry1++;
196             params[0] = d;
197             entries[0] = entry1;
198             if (params.length > 1){
199                 int entry2 = 0;
200                 for(Object d1 : valueArrays[1]) {
201                     entry2++;
202                     params[1] = d1;
203                     entries[1] = entry2;
204                     callMethods(mathMethod, fastMethod, params, entries);
205                 }
206             } else {
207                 callMethods(mathMethod, fastMethod, params, entries);
208             }
209         }
210     }
