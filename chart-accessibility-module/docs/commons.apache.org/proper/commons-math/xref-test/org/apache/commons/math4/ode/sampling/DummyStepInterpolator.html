18  package org.apache.commons.math4.ode.sampling;
19  
20  import java.io.IOException;
21  import java.io.ObjectInput;
22  import java.io.ObjectOutput;
23  
24  import org.apache.commons.math4.ode.EquationsMapper;
25  import org.apache.commons.math4.ode.sampling.AbstractStepInterpolator;
26  import org.apache.commons.math4.ode.sampling.StepHandler;
27  import org.apache.commons.math4.ode.sampling.StepInterpolator;
28  
29  /** This class is a step interpolator that does nothing.
30   *
31   * <p>This class is used when the {@link StepHandler "step handler"}
32   * set up by the user does not need step interpolation. It does not
33   * recompute the state when {@link AbstractStepInterpolator#setInterpolatedTime
34   * setInterpolatedTime} is called. This implies the interpolated state
35   * is always the state at the end of the current step.</p>
36   *
37   * @see StepHandler
38   *
39   * @since 1.2
40   */
41  
42  public class DummyStepInterpolator
43    extends AbstractStepInterpolator {
44  
45    /** Serializable version identifier. */
46    private static final long serialVersionUID = 1708010296707839488L;
47  
48    /** Current derivative. */
49    private double[] currentDerivative;
50  
51    /** Simple constructor.
52     * This constructor builds an instance that is not usable yet, the
53     * <code>AbstractStepInterpolator.reinitialize</code> protected method
54     * should be called before using the instance in order to initialize
55     * the internal arrays. This constructor is used only in order to delay
56     * the initialization in some cases. As an example, the {@link
57     * org.apache.commons.math4.ode.nonstiff.EmbeddedRungeKuttaIntegrator} uses
58     * the prototyping design pattern to create the step interpolators by
59     * cloning an uninitialized model and latter initializing the copy.
60     */
61    public DummyStepInterpolator() {
62      super();
63      currentDerivative = null;
64    }
65  
66    /** Simple constructor.
67     * @param y reference to the integrator array holding the state at
68     * the end of the step
69     * @param yDot reference to the integrator array holding the state
70     * derivative at some arbitrary point within the step
71     * @param forward integration direction indicator
72     */
73    public DummyStepInterpolator(final double[] y, final double[] yDot, final boolean forward) {
74      super(y, forward, new EquationsMapper(0, y.length), new EquationsMapper[0]);
75      currentDerivative = yDot;
76    }
77  
78    /** Copy constructor.
79     * @param interpolator interpolator to copy from. The copy is a deep
80     * copy: its arrays are separated from the original arrays of the
81     * instance
82     */
83    public DummyStepInterpolator(final DummyStepInterpolator interpolator) {
84      super(interpolator);
85      if (interpolator.currentDerivative != null) {
86          currentDerivative = interpolator.currentDerivative.clone();
87      }
88    }
89  
90    /** Really copy the finalized instance.
91     * @return a copy of the finalized instance
92     */
93    @Override
94    protected StepInterpolator doCopy() {
95      return new DummyStepInterpolator(this);
96    }
97  
98    /** Compute the state at the interpolated time.
99     * In this class, this method does nothing: the interpolated state
100    * is always the state at the end of the current step.
101    * @param theta normalized interpolation abscissa within the step
102    * (theta is zero at the previous time step and one at the current time step)
103    * @param oneMinusThetaH time gap between the interpolated time and
104    * the current time
105    */
106   @Override
107   protected void computeInterpolatedStateAndDerivatives(final double theta, final double oneMinusThetaH) {
108       System.arraycopy(currentState,      0, interpolatedState,       0, currentState.length);
109       System.arraycopy(currentDerivative, 0, interpolatedDerivatives, 0, currentDerivative.length);
110   }
111 
112   /** Write the instance to an output channel.
113    * @param out output channel
114    * @exception IOException if the instance cannot be written
115    */
116   @Override
117   public void writeExternal(final ObjectOutput out)
118     throws IOException {
119 
120       // save the state of the base class
121     writeBaseExternal(out);
122 
123     if (currentDerivative != null) {
124         for (int i = 0; i < currentDerivative.length; ++i) {
125             out.writeDouble(currentDerivative[i]);
126         }
127     }
128 
129   }
130 
131   /** Read the instance from an input channel.
132    * @param in input channel
133    * @exception IOException if the instance cannot be read
134    */
135   @Override
136   public void readExternal(final ObjectInput in)
137     throws IOException, ClassNotFoundException {
138 
139     // read the base class
140     final double t = readBaseExternal(in);
141 
142     if (currentState == null) {
143         currentDerivative = null;
144     } else {
145         currentDerivative  = new double[currentState.length];
146         for (int i = 0; i < currentDerivative.length; ++i) {
147             currentDerivative[i] = in.readDouble();
148         }
149     }
150 
151     // we can now set the interpolated time and state
152     setInterpolatedTime(t);
153 
154   }
155 
156 }
