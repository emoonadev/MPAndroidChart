18  package org.apache.commons.math4.util;
19  
20  import org.apache.commons.math4.TestUtils;
21  import org.apache.commons.math4.util.DefaultTransformer;
22  import org.apache.commons.math4.util.NumberTransformer;
23  import org.apache.commons.math4.util.TransformerMap;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  
28  /**
29   */
30  public class TransformerMapTest {
31      /**
32       *
33       */
34      @Test
35      public void testPutTransformer(){
36          NumberTransformer expected = new DefaultTransformer();
37  
38          TransformerMap map = new TransformerMap();
39          map.putTransformer(TransformerMapTest.class, expected);
40          Assert.assertEquals(expected, map.getTransformer(TransformerMapTest.class));
41      }
42  
43      /**
44       *
45       */
46      @Test
47      public void testContainsClass(){
48          NumberTransformer expected = new DefaultTransformer();
49          TransformerMap map = new TransformerMap();
50          map.putTransformer(TransformerMapTest.class, expected);
51          Assert.assertTrue(map.containsClass(TransformerMapTest.class));
52      }
53  
54      /**
55       *
56       */
57      @Test
58      public void testContainsTransformer(){
59          NumberTransformer expected = new DefaultTransformer();
60          TransformerMap map = new TransformerMap();
61          map.putTransformer(TransformerMapTest.class, expected);
62          Assert.assertTrue(map.containsTransformer(expected));
63      }
64  
65      /**
66       *
67       */
68      @Test
69      public void testRemoveTransformer(){
70          NumberTransformer expected = new DefaultTransformer();
71  
72          TransformerMap map = new TransformerMap();
73          map.putTransformer(TransformerMapTest.class, expected);
74          Assert.assertTrue(map.containsClass(TransformerMapTest.class));
75          Assert.assertTrue(map.containsTransformer(expected));
76          map.removeTransformer(TransformerMapTest.class);
77          Assert.assertFalse(map.containsClass(TransformerMapTest.class));
78          Assert.assertFalse(map.containsTransformer(expected));
79      }
80  
81      /**
82       *
83       */
84      @Test
85      public void testClear(){
86          NumberTransformer expected = new DefaultTransformer();
87  
88          TransformerMap map = new TransformerMap();
89          map.putTransformer(TransformerMapTest.class, expected);
90          Assert.assertTrue(map.containsClass(TransformerMapTest.class));
91          map.clear();
92          Assert.assertFalse(map.containsClass(TransformerMapTest.class));
93      }
94  
95      /**
96       *
97       */
98      @Test
99      public void testClasses(){
100         NumberTransformer expected = new DefaultTransformer();
101         TransformerMap map = new TransformerMap();
102         map.putTransformer(TransformerMapTest.class, expected);
103         Assert.assertTrue(map.classes().contains(TransformerMapTest.class));
104     }
105 
106     /**
107      *
108      */
109     @Test
110     public void testTransformers(){
111         NumberTransformer expected = new DefaultTransformer();
112         TransformerMap map = new TransformerMap();
113         map.putTransformer(TransformerMapTest.class, expected);
114         Assert.assertTrue(map.transformers().contains(expected));
115     }
116 
117     @Test
118     public void testSerial(){
119         NumberTransformer expected = new DefaultTransformer();
120         TransformerMap map = new TransformerMap();
121         map.putTransformer(TransformerMapTest.class, expected);
122         Assert.assertEquals(map, TestUtils.serializeAndRecover(map));
123     }
124 
125 }
