118     private void testProbability(double x, double a, double b, double expected) {
119         GammaDistribution distribution = new GammaDistribution( a, b );
120         double actual = distribution.cumulativeProbability(x);
121         Assert.assertEquals("probability for " + x, expected, actual, 10e-4);
122     }

124     private void testValue(double expected, double a, double b, double p) {
125         GammaDistribution distribution = new GammaDistribution( a, b );
126         double actual = distribution.inverseCumulativeProbability(p);
127         Assert.assertEquals("critical value for " + p, expected, actual, 10e-4);
128     }

152         GammaDistribution d = new GammaDistribution(alpha, 1 / rate);
153         for (int i = 0; i < x.length; i++) {

165     @Test
166     public void testMoments() {
167         final double tol = 1e-9;
168         GammaDistribution dist;
169 
170         dist = new GammaDistribution(1, 2);
171         Assert.assertEquals(dist.getNumericalMean(), 2, tol);
172         Assert.assertEquals(dist.getNumericalVariance(), 4, tol);

179     private static final double HALF_LOG_2_PI = 0.5 * FastMath.log(2.0 * FastMath.PI);
180 
181     public static double logGamma(double x) {
182         /*
183          * This is a copy of
184          * double Gamma.logGamma(double)
185          * prior to MATH-849
186          */
187         double ret;
188 
189         if (Double.isNaN(x) || (x <= 0.0)) {
190             ret = Double.NaN;
191         } else {
192             double sum = Gamma.lanczos(x);
193             double tmp = x + Gamma.LANCZOS_G + .5;
194             ret = ((x + .5) * FastMath.log(tmp)) - tmp +
195                 HALF_LOG_2_PI + FastMath.log(sum / x);
196         }
197 
198         return ret;
199     }
200 
201     public static double density(final double x, final double shape,
202                                  final double scale) {
203         /*
204          * This is a copy of
205          * double GammaDistribution.density(double)
206          * prior to MATH-753.
207          */
208         if (x < 0) {
209             return 0;
210         }
211         return FastMath.pow(x / scale, shape - 1) / scale *
212                FastMath.exp(-x / scale) / FastMath.exp(logGamma(shape));
213     }

215     /*
216      * MATH-753: large values of x or shape parameter cause density(double) to
217      * overflow. Reference data is generated with the Maxima script
218      * gamma-distribution.mac, which can be found in
219      * src/test/resources/org/apache/commons/math4/distribution.
220      */
221 
222     private void doTestMath753(final double shape,
223         final double meanNoOF, final double sdNoOF,
224         final double meanOF, final double sdOF,
225         final String resourceName) throws IOException {
226         final GammaDistribution distribution = new GammaDistribution(shape, 1.0);
227         final SummaryStatistics statOld = new SummaryStatistics();
228         final SummaryStatistics statNewNoOF = new SummaryStatistics();
229         final SummaryStatistics statNewOF = new SummaryStatistics();
230 
231         final InputStream resourceAsStream;
232         resourceAsStream = this.getClass().getResourceAsStream(resourceName);
233         Assert.assertNotNull("Could not find resource " + resourceName,
234                              resourceAsStream);
235         final BufferedReader in;
236         in = new BufferedReader(new InputStreamReader(resourceAsStream));
237 
238         try {
239             for (String line = in.readLine(); line != null; line = in.readLine()) {
240                 if (line.startsWith("#")) {
241                     continue;
242                 }
243                 final String[] tokens = line.split(", ");
244                 Assert.assertTrue("expected two floating-point values",
245                                   tokens.length == 2);
246                 final double x = Double.parseDouble(tokens[0]);
247                 final String msg = "x = " + x + ", shape = " + shape +
248                                    ", scale = 1.0";
249                 final double expected = Double.parseDouble(tokens[1]);
250                 final double ulp = FastMath.ulp(expected);
251                 final double actualOld = density(x, shape, 1.0);
252                 final double actualNew = distribution.density(x);
253                 final double errOld, errNew;
254                 errOld = FastMath.abs((actualOld - expected) / ulp);
255                 errNew = FastMath.abs((actualNew - expected) / ulp);
256 
257                 if (Double.isNaN(actualOld) || Double.isInfinite(actualOld)) {
258                     Assert.assertFalse(msg, Double.isNaN(actualNew));
259                     Assert.assertFalse(msg, Double.isInfinite(actualNew));
260                     statNewOF.addValue(errNew);
261                 } else {
262                     statOld.addValue(errOld);
263                     statNewNoOF.addValue(errNew);
264                 }
265             }
266             if (statOld.getN() != 0) {
267                 /*
268                  * If no overflow occurs, check that new implementation is
269                  * better than old one.
270                  */
271                 final StringBuilder sb = new StringBuilder("shape = ");
272                 sb.append(shape);
273                 sb.append(", scale = 1.0\n");
274                 sb.append("Old implementation\n");
275                 sb.append("------------------\n");
276                 sb.append(statOld.toString());
277                 sb.append("New implementation\n");
278                 sb.append("------------------\n");
279                 sb.append(statNewNoOF.toString());
280                 final String msg = sb.toString();
281 
282                 final double oldMin = statOld.getMin();
283                 final double newMin = statNewNoOF.getMin();
284                 Assert.assertTrue(msg, newMin <= oldMin);
285 
286                 final double oldMax = statOld.getMax();
287                 final double newMax = statNewNoOF.getMax();
288                 Assert.assertTrue(msg, newMax <= oldMax);
289 
290                 final double oldMean = statOld.getMean();
291                 final double newMean = statNewNoOF.getMean();
292                 Assert.assertTrue(msg, newMean <= oldMean);
293 
294                 final double oldSd = statOld.getStandardDeviation();
295                 final double newSd = statNewNoOF.getStandardDeviation();
296                 Assert.assertTrue(msg, newSd <= oldSd);
297 
298                 Assert.assertTrue(msg, newMean <= meanNoOF);
299                 Assert.assertTrue(msg, newSd <= sdNoOF);
300             }
301             if (statNewOF.getN() != 0) {
302                 final double newMean = statNewOF.getMean();
303                 final double newSd = statNewOF.getStandardDeviation();
304 
305                 final StringBuilder sb = new StringBuilder("shape = ");
306                 sb.append(shape);
307                 sb.append(", scale = 1.0");
308                 sb.append(", max. mean error (ulps) = ");
309                 sb.append(meanOF);
310                 sb.append(", actual mean error (ulps) = ");
311                 sb.append(newMean);
312                 sb.append(", max. sd of error (ulps) = ");
313                 sb.append(sdOF);
314                 sb.append(", actual sd of error (ulps) = ");
315                 sb.append(newSd);
316                 final String msg = sb.toString();
317 
318                 Assert.assertTrue(msg, newMean <= meanOF);
319                 Assert.assertTrue(msg, newSd <= sdOF);
320             }
321         } catch (IOException e) {
322             Assert.fail(e.getMessage());
323         } finally {
324             in.close();
325         }
326     }
