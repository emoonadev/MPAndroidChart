321         final KolmogorovSmirnovTest test = new KolmogorovSmirnovTest(RandomSource.WELL_19937_C, 1000);
322         final int sampleSize = 14;
323         final double tol = .001;
324         final double[] shortUniform = new double[sampleSize];
325         System.arraycopy(uniform, 0, shortUniform, 0, sampleSize);
326         final double[] shortGaussian = new double[sampleSize];
327         final double[] shortGaussian2 = new double[sampleSize];
328         System.arraycopy(gaussian, 0, shortGaussian, 0, sampleSize);
329         System.arraycopy(gaussian, 10, shortGaussian2, 0, sampleSize);
330         final double[] d = {
331             test.kolmogorovSmirnovStatistic(shortGaussian, shortUniform),
332             test.kolmogorovSmirnovStatistic(shortGaussian2, shortGaussian)
333         };
334         for (double dv : d) {
335             double exactPStrict = test.exactP(dv, sampleSize, sampleSize, true);
336             double exactPNonStrict = test.exactP(dv, sampleSize, sampleSize, false);
337             double montePStrict = test.monteCarloP(dv, sampleSize, sampleSize, true,
338                                                    KolmogorovSmirnovTest.MONTE_CARLO_ITERATIONS);
339             double montePNonStrict = test.monteCarloP(dv, sampleSize, sampleSize, false,
340                                                       KolmogorovSmirnovTest.MONTE_CARLO_ITERATIONS);
341             Assert.assertEquals(exactPStrict, montePStrict, tol);
342             Assert.assertEquals(exactPNonStrict, montePNonStrict, tol);
343         }

521     @Test
522     public void testFillBooleanArrayRandomlyWithFixedNumberTrueValues() throws Exception {
523         Method method = KolmogorovSmirnovTest.class.getDeclaredMethod("fillBooleanArrayRandomlyWithFixedNumberTrueValues",
524                                                                       boolean[].class, Integer.TYPE, UniformRandomProvider.class);
525         method.setAccessible(true);
526 
527         final int[][] parameters = {{5, 1}, {5, 2}, {5, 3}, {5, 4}, {8, 1}, {8, 2}, {8, 3}, {8, 4}, {8, 5}, {8, 6}, {8, 7}};
528 
529         final double alpha = 0.001;
530         final int numIterations = 1000000;
531 
532         final UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_19937_C, 0);
533 
534         for (final int[] parameter : parameters) {
535 
536             final int arraySize = parameter[0];
537             final int numberOfTrueValues = parameter[1];
538 
539             final boolean[] b = new boolean[arraySize];
540             final long[] counts = new long[1 << arraySize];
541 
542             for (int i = 0; i < numIterations; ++i) {
543                 method.invoke(KolmogorovSmirnovTest.class, b, numberOfTrueValues, rng);
544                 int x = 0;
545                 for (int j = 0; j < arraySize; ++j) {
546                     x = ((x << 1) | ((b[j])?1:0));
547                 }
548                 counts[x] += 1;
549             }
550 
551             final int numCombinations = (int) CombinatoricsUtils.binomialCoefficient(arraySize, numberOfTrueValues);
552 
553             final long[] observed = new long[numCombinations];
554             final double[] expected = new double[numCombinations];
555             Arrays.fill(expected, numIterations / (double) numCombinations);
556 
557             int observedIdx = 0;
558 
559             for (int i = 0; i < (1 << arraySize); ++i) {
560                 if (Integer.bitCount(i) == numberOfTrueValues) {
561                     observed[observedIdx] = counts[i];
562                     observedIdx += 1;
563                 }
564                 else {
565                     Assert.assertEquals(0, counts[i]);
566                 }
567             }
568 
569             Assert.assertEquals(numCombinations, observedIdx);
570             TestUtils.assertChiSquareAccept(expected, observed, alpha);
571         }
572     }

646     /**
647      * Checks that fixTies eliminates ties in the data but does not otherwise
648      * perturb the ordering.
649      */
650     private void checkFixTies(double[] x, double[] y) throws Exception {
651         final double[] origCombined = MathArrays.concatenate(x, y);
652         fixTies(x, y);
653         Assert.assertFalse(hasTies(x, y));
654         final double[] combined = MathArrays.concatenate(x, y);
655         for (int i = 0; i < combined.length; i++) {
656             for (int j = 0; j < i; j++) {
657                 Assert.assertTrue(combined[i] != combined[j]);
658                 if (combined[i] < combined[j])
659                     Assert.assertTrue(origCombined[i] < origCombined[j]
660                                           || origCombined[i] == origCombined[j]);
661             }
662 
663         }
