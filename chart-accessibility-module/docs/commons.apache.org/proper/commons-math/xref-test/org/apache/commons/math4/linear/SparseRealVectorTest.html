106     /* Check that the operations do not throw an exception (cf. MATH-645). */
107     @Test
108     public void testConcurrentModification() {
109         final RealVector u = new OpenMapRealVector(3, 1e-6);
110         u.setEntry(0, 1);
111         u.setEntry(1, 0);
112         u.setEntry(2, 2);
113 
114         final RealVector v1 = new OpenMapRealVector(3, 1e-6);
115         v1.setEntry(0, 0);
116         v1.setEntry(1, 3);
117         v1.setEntry(2, 0);
118 
119         u.ebeMultiply(v1);
120         u.ebeDivide(v1);
121     }
