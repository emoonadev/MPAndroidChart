18  package org.apache.commons.math4.ml.neuralnet;
19  
20  import org.apache.commons.math4.ml.neuralnet.FeatureInitializer;
21  
22  
23  /**
24   * Wraps a given initializer.
25   */
26  public class OffsetFeatureInitializer
27      implements FeatureInitializer {
28      /** Wrapped initializer. */
29      private final FeatureInitializer orig;
30      /** Offset. */
31      private int inc = 0;
32  
33      /**
34       * Creates a new initializer whose {@link #value()} method
35       * will return {@code orig.value() + offset}, where
36       * {@code offset} is automatically incremented by one at
37       * each call.
38       *
39       * @param orig Original initializer.
40       */
41      public OffsetFeatureInitializer(FeatureInitializer orig) {
42          this.orig = orig;
43      }
44  
45      /** {@inheritDoc} */
46      public double value() {
47          return orig.value() + inc++;
48      }
49  }
