360     @Test
361     public void testAdd() {
362         BigFraction a = new BigFraction(1, 2);
363         BigFraction b = new BigFraction(2, 3);
364 
365         assertFraction(1, 1, a.add(a));
366         assertFraction(7, 6, a.add(b));
367         assertFraction(7, 6, b.add(a));
368         assertFraction(4, 3, b.add(b));
369 
370         BigFraction f1 = new BigFraction(Integer.MAX_VALUE - 1, 1);
371         BigFraction f2 = BigFraction.ONE;
372         BigFraction f = f1.add(f2);
373         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
374         Assert.assertEquals(1, f.getDenominatorAsInt());
375 
376         f1 = new BigFraction(-1, 13 * 13 * 2 * 2);
377         f2 = new BigFraction(-2, 13 * 17 * 2);
378         f = f1.add(f2);
379         Assert.assertEquals(13 * 13 * 17 * 2 * 2, f.getDenominatorAsInt());
380         Assert.assertEquals(-17 - 2 * 13 * 2, f.getNumeratorAsInt());
381 
382         try {
383             f.add((BigFraction) null);
384             Assert.fail("expecting NullArgumentException");
385         } catch (NullArgumentException ex) {
386         }
387 
388         // if this fraction is added naively, it will overflow.
389         // check that it doesn't.
390         f1 = new BigFraction(1, 32768 * 3);
391         f2 = new BigFraction(1, 59049);
392         f = f1.add(f2);
393         Assert.assertEquals(52451, f.getNumeratorAsInt());
394         Assert.assertEquals(1934917632, f.getDenominatorAsInt());
395 
396         f1 = new BigFraction(Integer.MIN_VALUE, 3);
397         f2 = new BigFraction(1, 3);
398         f = f1.add(f2);
399         Assert.assertEquals(Integer.MIN_VALUE + 1, f.getNumeratorAsInt());
400         Assert.assertEquals(3, f.getDenominatorAsInt());
401 
402         f1 = new BigFraction(Integer.MAX_VALUE - 1, 1);
403         f = f1.add(BigInteger.ONE);
404         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
405         Assert.assertEquals(1, f.getDenominatorAsInt());
406 
407         f = f.add(BigInteger.ZERO);
408         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
409         Assert.assertEquals(1, f.getDenominatorAsInt());
410 
411         f1 = new BigFraction(Integer.MAX_VALUE - 1, 1);
412         f = f1.add(1);
413         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
414         Assert.assertEquals(1, f.getDenominatorAsInt());
415 
416         f = f.add(0);
417         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
418         Assert.assertEquals(1, f.getDenominatorAsInt());
419 
420         f1 = new BigFraction(Integer.MAX_VALUE - 1, 1);
421         f = f1.add(1l);
422         Assert.assertEquals(Integer.MAX_VALUE, f.getNumeratorAsInt());
423         Assert.assertEquals(1, f.getDenominatorAsInt());
424 
425         f = f.add(0l);
