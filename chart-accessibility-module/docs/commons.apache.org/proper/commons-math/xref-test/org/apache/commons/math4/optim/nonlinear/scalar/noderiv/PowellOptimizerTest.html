38      public void testBoundsUnsupported() {
39          final MultivariateFunction func = new SumSincFunction(-1);

41                                                            1e-4, 1e-4);
42  
43          optim.optimize(new MaxEval(100),
44                         new ObjectiveFunction(func),
45                         GoalType.MINIMIZE,
46                         new InitialGuess(new double[] { -3, 0 }),
47                         new SimpleBounds(new double[] { -5, -1 },
48                                          new double[] { 5, 1 }));
49      }

51      @Test
52      public void testSumSinc() {
53          final MultivariateFunction func = new SumSincFunction(-1);
54  
55          int dim = 2;
56          final double[] minPoint = new double[dim];
57          for (int i = 0; i < dim; i++) {
58              minPoint[i] = 0;
59          }
60  
61          double[] init = new double[dim];
62  
63          // Initial is minimum.
64          for (int i = 0; i < dim; i++) {
65              init[i] = minPoint[i];
66          }
67          doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-9);
68  
69          // Initial is far from minimum.
70          for (int i = 0; i < dim; i++) {
71              init[i] = minPoint[i] + 3;
72          }
73          doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-5);
74          // More stringent line search tolerance enhances the precision
75          // of the result.
76          doTest(func, minPoint, init, GoalType.MINIMIZE, 1e-9, 1e-9, 1e-7);
77      }

141     /**
142      * Ensure that we do not increase the number of function evaluations when
143      * the function values are scaled up.
144      * Note that the tolerances parameters passed to the constructor must
145      * still hold sensible values because they are used to set the line search
146      * tolerances.
147      */
148     @Test
149     public void testRelativeToleranceOnScaledValues() {
150         final MultivariateFunction func = new MultivariateFunction() {
151                 public double value(double[] x) {
152                     final double a = x[0] - 1;
153                     final double b = x[1] - 1;
154                     return a * a * FastMath.sqrt(FastMath.abs(a)) + b * b + 1;
155                 }
156             };
157 
158         int dim = 2;
159         final double[] minPoint = new double[dim];
160         for (int i = 0; i < dim; i++) {
161             minPoint[i] = 1;
162         }
163 
164         double[] init = new double[dim];
165         // Initial is far from minimum.
166         for (int i = 0; i < dim; i++) {
167             init[i] = minPoint[i] - 20;
168         }
169 
170         final double relTol = 1e-10;
171 
172         final int maxEval = 1000;
173         // Very small absolute tolerance to rely solely on the relative
174         // tolerance as a stopping criterion
175         final PowellOptimizer optim = new PowellOptimizer(relTol, 1e-100);
176 
177         final PointValuePair funcResult = optim.optimize(new MaxEval(maxEval),
178                                                          new ObjectiveFunction(func),
179                                                          GoalType.MINIMIZE,
180                                                          new InitialGuess(init));
181         final double funcValue = func.value(funcResult.getPoint());
182         final int funcEvaluations = optim.getEvaluations();
183 
184         final double scale = 1e10;
185         final MultivariateFunction funcScaled = new MultivariateFunction() {
186                 public double value(double[] x) {
187                     return scale * func.value(x);
188                 }
189             };
190 
191         final PointValuePair funcScaledResult = optim.optimize(new MaxEval(maxEval),
192                                                                new ObjectiveFunction(funcScaled),
193                                                                GoalType.MINIMIZE,
194                                                                new InitialGuess(init));
195         final double funcScaledValue = funcScaled.value(funcScaledResult.getPoint());
196         final int funcScaledEvaluations = optim.getEvaluations();
197 
198         // Check that both minima provide the same objective funciton values,
199         // within the relative function tolerance.
200         Assert.assertEquals(1, funcScaledValue / (scale * funcValue), relTol);
201 
202         // Check that the numbers of evaluations are the same.
203         Assert.assertEquals(funcEvaluations, funcScaledEvaluations);
204     }

206     /**
207      * @param func Function to optimize.
208      * @param optimum Expected optimum.
209      * @param init Starting point.
210      * @param goal Minimization or maximization.
211      * @param fTol Tolerance (relative error on the objective function) for
212      * "Powell" algorithm.
213      * @param pointTol Tolerance for checking that the optimum is correct.
214      */
215     private void doTest(MultivariateFunction func,
216                         double[] optimum,
217                         double[] init,
218                         GoalType goal,
219                         double fTol,
220                         double pointTol) {
221         final PowellOptimizer optim = new PowellOptimizer(fTol, Math.ulp(1d));
222 
223         final PointValuePair result = optim.optimize(new MaxEval(1000),
224                                                      new ObjectiveFunction(func),
225                                                      goal,
226                                                      new InitialGuess(init));
227         final double[] point = result.getPoint();
228 
229         for (int i = 0, dim = optimum.length; i < dim; i++) {
230             Assert.assertEquals("found[" + i + "]=" + point[i] + " value=" + result.getValue(),
231                                 optimum[i], point[i], pointTol);
232         }
233     }

235     /**
236      * @param func Function to optimize.
237      * @param optimum Expected optimum.
238      * @param init Starting point.
239      * @param goal Minimization or maximization.
240      * @param fTol Tolerance (relative error on the objective function) for
241      * "Powell" algorithm.
242      * @param fLineTol Tolerance (relative error on the objective function)
243      * for the internal line search algorithm.
244      * @param pointTol Tolerance for checking that the optimum is correct.
245      */
246     private void doTest(MultivariateFunction func,
247                         double[] optimum,
248                         double[] init,
249                         GoalType goal,
250                         double fTol,
251                         double fLineTol,
252                         double pointTol) {
253         final PowellOptimizer optim = new PowellOptimizer(fTol, Math.ulp(1d),
254                                                           fLineTol, Math.ulp(1d));
255 
256         final PointValuePair result = optim.optimize(new MaxEval(1000),
257                                                      new ObjectiveFunction(func),
258                                                      goal,
259                                                      new InitialGuess(init));
260         final double[] point = result.getPoint();
261 
262         for (int i = 0, dim = optimum.length; i < dim; i++) {
263             Assert.assertEquals("found[" + i + "]=" + point[i] + " value=" + result.getValue(),
264                                 optimum[i], point[i], pointTol);
265         }
266 
267         Assert.assertTrue(optim.getIterations() > 0);
268     }
