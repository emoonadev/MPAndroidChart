17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.exception.NotPositiveException;
20  import org.apache.commons.math4.special.Gamma;
21  
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link CombinatoricsUtils.FactorialLog} class.
27   */
28  public class FactorialLogTest {
29  
30      @Test(expected=NotPositiveException.class)
31      public void testPrecondition1() {
32          CombinatoricsUtils.FactorialLog.create().withCache(-1);
33      }
34  
35      @Test(expected=NotPositiveException.class)
36      public void testNonPositiveArgument() {
37          final CombinatoricsUtils.FactorialLog f = CombinatoricsUtils.FactorialLog.create();
38          f.value(-1);
39      }
40  
41      @Test
42      public void testDelegation() {
43          final CombinatoricsUtils.FactorialLog f = CombinatoricsUtils.FactorialLog.create();
44  
45          // Starting at 21 because for smaller arguments, there is no delegation to the
46          // "Gamma" class.
47          for (int i = 21; i < 10000; i++) {
48              final double expected = Gamma.logGamma(i + 1);
49              Assert.assertEquals(i + "! ",
50                                  expected, f.value(i), 0d);
51          }
52      }
53  
54      @Test
55      public void testCompareDirectWithoutCache() {
56          // This test shows that delegating to the "Gamma" class will also lead to a
57          // less accurate result.
58  
59          final int max = 100;
60          final CombinatoricsUtils.FactorialLog f = CombinatoricsUtils.FactorialLog.create();
61  
62          for (int i = 0; i < max; i++) {
63              final double expected = factorialLog(i);
64              Assert.assertEquals(i + "! ",
65                                  expected, f.value(i), 2 * Math.ulp(expected));
66          }
67      }
68  
69      @Test
70      public void testCompareDirectWithCache() {
71          final int max = 1000;
72          final CombinatoricsUtils.FactorialLog f = CombinatoricsUtils.FactorialLog.create().withCache(max);
73  
74          for (int i = 0; i < max; i++) {
75              final double expected = factorialLog(i);
76              Assert.assertEquals(i + "! ",
77                                  expected, f.value(i), 0d);
78          }
79      }
80  
81      @Test
82      public void testCacheIncrease() {
83          final int max = 100;
84          final CombinatoricsUtils.FactorialLog f1 = CombinatoricsUtils.FactorialLog.create().withCache(max);
85          final CombinatoricsUtils.FactorialLog f2 = f1.withCache(2 * max);
86  
87          final int val = max + max / 2;
88          final double expected = factorialLog(val);
89          Assert.assertEquals(expected, f2.value(val), 0d);
90      }
91  
92      @Test
93      public void testCacheDecrease() {
94          final int max = 100;
95          final CombinatoricsUtils.FactorialLog f1 = CombinatoricsUtils.FactorialLog.create().withCache(max);
96          final CombinatoricsUtils.FactorialLog f2 = f1.withCache(max / 2);
97  
98          final int val = max / 4;
99          final double expected = factorialLog(val);
100         Assert.assertEquals(expected, f2.value(val), 0d);
101     }
102 
103     // Direct implementation.
104     private double factorialLog(final int n) {
105         double logSum = 0;
106         for (int i = 2; i <= n; i++) {
107             logSum += FastMath.log(i);
108         }
109         return logSum;
110     }
111 }
