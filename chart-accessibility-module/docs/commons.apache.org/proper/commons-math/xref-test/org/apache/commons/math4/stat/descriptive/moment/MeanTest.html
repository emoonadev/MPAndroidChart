17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.moment.Mean;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link UnivariateStatistic} class.
27   */
28  public class MeanTest extends StorelessUnivariateStatisticAbstractTest{
29  
30      protected Mean stat;
31  
32      /**
33       * {@inheritDoc}
34       */
35      @Override
36      public UnivariateStatistic getUnivariateStatistic() {
37          return new Mean();
38      }
39  
40      /**
41       * {@inheritDoc}
42       */
43      @Override
44      public double expectedValue() {
45          return this.mean;
46      }
47  
48      /**Expected value for  the testArray defined in UnivariateStatisticAbstractTest */
49      public double expectedWeightedValue() {
50          return this.weightedMean;
51      }
52  
53      @Test
54      public void testSmallSamples() {
55          Mean mean = new Mean();
56          Assert.assertTrue(Double.isNaN(mean.getResult()));
57          mean.increment(1d);
58          Assert.assertEquals(1d, mean.getResult(), 0);
59      }
60  
61      @Test
62      public void testWeightedMean() {
63          Mean mean = new Mean();
64          Assert.assertEquals(expectedWeightedValue(), mean.evaluate(testArray, testWeightsArray, 0, testArray.length), getTolerance());
65          Assert.assertEquals(expectedValue(), mean.evaluate(testArray, identicalWeightsArray, 0, testArray.length), getTolerance());
66      }
67  
68  }
