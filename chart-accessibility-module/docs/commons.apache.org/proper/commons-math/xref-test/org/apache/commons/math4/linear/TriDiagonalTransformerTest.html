56      @Test
57      public void testAEqualQTQt() {
58          checkAEqualQTQt(MatrixUtils.createRealMatrix(testSquare5));
59          checkAEqualQTQt(MatrixUtils.createRealMatrix(testSquare3));
60      }
61  
62      private void checkAEqualQTQt(RealMatrix matrix) {
63          TriDiagonalTransformer transformer = new TriDiagonalTransformer(matrix);
64          RealMatrix q  = transformer.getQ();
65          RealMatrix qT = transformer.getQT();
66          RealMatrix t  = transformer.getT();
67          double norm = q.multiply(t).multiply(qT).subtract(matrix).getNorm();
68          Assert.assertEquals(0, norm, 4.0e-15);
69      }

71      @Test
72      public void testNoAccessBelowDiagonal() {
73          checkNoAccessBelowDiagonal(testSquare5);
74          checkNoAccessBelowDiagonal(testSquare3);
75      }
76  
77      private void checkNoAccessBelowDiagonal(double[][] data) {
78          double[][] modifiedData = new double[data.length][];
79          for (int i = 0; i < data.length; ++i) {
80              modifiedData[i] = data[i].clone();
81              Arrays.fill(modifiedData[i], 0, i, Double.NaN);
82          }
83          RealMatrix matrix = MatrixUtils.createRealMatrix(modifiedData);
84          TriDiagonalTransformer transformer = new TriDiagonalTransformer(matrix);
85          RealMatrix q  = transformer.getQ();
86          RealMatrix qT = transformer.getQT();
87          RealMatrix t  = transformer.getT();
88          double norm = q.multiply(t).multiply(qT).subtract(MatrixUtils.createRealMatrix(data)).getNorm();
89          Assert.assertEquals(0, norm, 4.0e-15);
90      }

98      @Test
99      public void testQTOrthogonal() {
100         checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare5)).getQT());
101         checkOrthogonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare3)).getQT());
102     }
103 
104     private void checkOrthogonal(RealMatrix m) {
105         RealMatrix mTm = m.transpose().multiply(m);
106         RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
107         Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-15);
108     }
109 
110     @Test
111     public void testTTriDiagonal() {
112         checkTriDiagonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare5)).getT());
113         checkTriDiagonal(new TriDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare3)).getT());
114     }
115 
116     private void checkTriDiagonal(RealMatrix m) {
117         final int rows = m.getRowDimension();
118         final int cols = m.getColumnDimension();
119         for (int i = 0; i < rows; ++i) {
120             for (int j = 0; j < cols; ++j) {
121                 if ((i < j - 1) || (i > j + 1)) {
122                     Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
123                 }
124             }
125         }
126     }

145                             new double[][] {

154     private void checkMatricesValues(double[][] matrix, double[][] qRef,
155                                      double[] mainDiagnonal,
156                                      double[] secondaryDiagonal) {
157         TriDiagonalTransformer transformer =
158             new TriDiagonalTransformer(MatrixUtils.createRealMatrix(matrix));
159 
160         // check values against known references
161         RealMatrix q = transformer.getQ();
162         Assert.assertEquals(0, q.subtract(MatrixUtils.createRealMatrix(qRef)).getNorm(), 1.0e-14);
163 
164         RealMatrix t = transformer.getT();
165         double[][] tData = new double[mainDiagnonal.length][mainDiagnonal.length];
166         for (int i = 0; i < mainDiagnonal.length; ++i) {
167             tData[i][i] = mainDiagnonal[i];
168             if (i > 0) {
169                 tData[i][i - 1] = secondaryDiagonal[i - 1];
170             }
171             if (i < secondaryDiagonal.length) {
172                 tData[i][i + 1] = secondaryDiagonal[i];
173             }
174         }
175         Assert.assertEquals(0, t.subtract(MatrixUtils.createRealMatrix(tData)).getNorm(), 1.0e-14);
176 
177         // check the same cached instance is returned the second time
178         Assert.assertTrue(q == transformer.getQ());
179         Assert.assertTrue(t == transformer.getT());
180     }
