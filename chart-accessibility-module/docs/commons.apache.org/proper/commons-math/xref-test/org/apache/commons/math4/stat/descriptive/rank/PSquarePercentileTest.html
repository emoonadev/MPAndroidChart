67       * Verifies that copied statistics remain equal to originals when
68       * incremented the same way by making the copy after a majority of elements
69       * are incremented
70       */
71      @Test
72      public void testCopyConsistencyWithInitialMostElements() {
73  
74          StorelessUnivariateStatistic master =
75                  (StorelessUnivariateStatistic) getUnivariateStatistic();
76  
77          StorelessUnivariateStatistic replica = null;
78  
79          // select a portion of testArray till 75 % of the length to load first
80          long index = FastMath.round(0.75 * testArray.length);
81  
82          // Put first half in master and copy master to replica
83          master.incrementAll(testArray, 0, (int) index);
84          replica = master.copy();
85  
86          // Check same
87          Assert.assertTrue(replica.equals(master));
88          Assert.assertTrue(master.equals(replica));
89  
90          // Now add second part to both and check again
91          master.incrementAll(testArray, (int) index,
92                  (int) (testArray.length - index));
93          replica.incrementAll(testArray, (int) index,
94                  (int) (testArray.length - index));
95          Assert.assertTrue(replica.equals(master));
96          Assert.assertTrue(master.equals(replica));
97      }

543     private void assertValues(Double a, Double b, double delta) {
544         if (Double.isNaN(a)) {
545             Assert.assertTrue("" + b + " is not NaN.", Double.isNaN(a));
546         } else {
547             double max = FastMath.max(a, b);
548             double percentage = FastMath.abs(a - b) / max;
549             double deviation = delta;
550             Assert.assertTrue(String.format(
551                     "Deviated = %f and is beyond %f as a=%f,  b=%f",
552                     percentage, deviation, a, b), percentage < deviation);
553         }
554     }

560     private void doCalculatePercentile(Double percentile, Number[] test,
561             double delta) {
562         PSquarePercentile psquared = new PSquarePercentile(percentile);
563         for (Number value : test) {
564             psquared.increment(value.doubleValue());
565         }
566 
567         Percentile p2 = new Percentile(percentile * 100);
568 
569         double[] dall = new double[test.length];
570         for (int i = 0; i < test.length; i++) {
571             dall[i] = test[i].doubleValue();
572         }
573 
574         Double referenceValue = p2.evaluate(dall);
575         assertValues(psquared.getResult(), referenceValue, delta);
576     }
