17  package org.apache.commons.math4.stat.regression;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.exception.NullArgumentException;
21  import org.apache.commons.math4.linear.RealMatrix;
22  import org.apache.commons.math4.linear.RealVector;
23  import org.apache.commons.math4.stat.regression.AbstractMultipleLinearRegression;
24  import org.junit.Assert;
25  import org.junit.Before;
26  import org.junit.Test;
27  
28  
29  public abstract class MultipleLinearRegressionAbstractTest {
30  
31      protected AbstractMultipleLinearRegression regression;
32  
33      @Before
34      public void setUp(){
35          regression = createRegression();
36      }
37  
38      protected abstract AbstractMultipleLinearRegression createRegression();
39  
40      protected abstract int getNumberOfRegressors();
41  
42      protected abstract int getSampleSize();
43  
44      @Test
45      public void canEstimateRegressionParameters(){
46          double[] beta = regression.estimateRegressionParameters();
47          Assert.assertEquals(getNumberOfRegressors(), beta.length);
48      }
49  
50      @Test
51      public void canEstimateResiduals(){
52          double[] e = regression.estimateResiduals();
53          Assert.assertEquals(getSampleSize(), e.length);
54      }
55  
56      @Test
57      public void canEstimateRegressionParametersVariance(){
58          double[][] variance = regression.estimateRegressionParametersVariance();
59          Assert.assertEquals(getNumberOfRegressors(), variance.length);
60      }
61  
62      @Test
63      public void canEstimateRegressandVariance(){
64          if (getSampleSize() > getNumberOfRegressors()) {
65              double variance = regression.estimateRegressandVariance();
66              Assert.assertTrue(variance > 0.0);
67          }
68      }
69  
70      /**
71       * Verifies that newSampleData methods consistently insert unitary columns
72       * in design matrix.  Confirms the fix for MATH-411.
73       */
74      @Test
75      public void testNewSample() {
76          double[] design = new double[] {
77            1, 19, 22, 33,
78            2, 20, 30, 40,
79            3, 25, 35, 45,
80            4, 27, 37, 47
81          };
82          double[] y = new double[] {1, 2, 3, 4};
83          double[][] x = new double[][] {
84            {19, 22, 33},
85            {20, 30, 40},
86            {25, 35, 45},
87            {27, 37, 47}
88          };
89          AbstractMultipleLinearRegression regression = createRegression();
90          regression.newSampleData(design, 4, 3);
91          RealMatrix flatX = regression.getX().copy();
92          RealVector flatY = regression.getY().copy();
93          regression.newXSampleData(x);
94          regression.newYSampleData(y);
95          Assert.assertEquals(flatX, regression.getX());
96          Assert.assertEquals(flatY, regression.getY());
97  
98          // No intercept
99          regression.setNoIntercept(true);
100         regression.newSampleData(design, 4, 3);
101         flatX = regression.getX().copy();
102         flatY = regression.getY().copy();
103         regression.newXSampleData(x);
104         regression.newYSampleData(y);
105         Assert.assertEquals(flatX, regression.getX());
106         Assert.assertEquals(flatY, regression.getY());
107     }
108

110     public void testNewSampleNullData() {

113     }

116     public void testNewSampleInvalidData() {

118         createRegression().newSampleData(data, 2, 3);
119     }
120 
121     @Test(expected=MathIllegalArgumentException.class)
122     public void testNewSampleInsufficientData() {
123         double[] data = new double[] {1, 2, 3, 4};

125     }

130     }

135     }
