29  public class AdamsMoultonFieldIntegratorTest extends AdamsFieldIntegratorAbstractTest {
30  
31      protected <T extends RealFieldElement<T>> AdamsFieldIntegrator<T>
32      createIntegrator(Field<T> field, final int nSteps, final double minStep, final double maxStep,
33                       final double scalAbsoluteTolerance, final double scalRelativeTolerance) {
34          return new AdamsMoultonFieldIntegrator<T>(field, nSteps, minStep, maxStep,
35                          scalAbsoluteTolerance, scalRelativeTolerance);
36      }
37  
38      protected <T extends RealFieldElement<T>> AdamsFieldIntegrator<T>
39      createIntegrator(Field<T> field, final int nSteps, final double minStep, final double maxStep,
40                       final double[] vecAbsoluteTolerance, final double[] vecRelativeTolerance) {
41          return new AdamsMoultonFieldIntegrator<T>(field, nSteps, minStep, maxStep,
42                          vecAbsoluteTolerance, vecRelativeTolerance);
43      }
44  
45      @Test(expected=NumberIsTooSmallException.class)
46      public void testMinStep() {
47          doDimensionCheck(Decimal64Field.getInstance());
48      }
49  
50      @Test
51      public void testIncreasingTolerance() {
52          // the 0.45 and 8.69 factors are only valid for this test
53          // and has been obtained from trial and error
54          // there are no general relationship between local and global errors
55          doTestIncreasingTolerance(Decimal64Field.getInstance(), 0.45, 8.69);
56      }
57  
58      @Test(expected = MaxCountExceededException.class)
59      public void exceedMaxEvaluations() {
60          doExceedMaxEvaluations(Decimal64Field.getInstance(), 650);
61      }
62  
63      @Test
64      public void backward() {
65          doBackward(Decimal64Field.getInstance(), 3.0e-9, 3.0e-9, 1.0e-16, "Adams-Moulton");
66      }
67  
68      @Test
69      public void polynomial() {
70          doPolynomial(Decimal64Field.getInstance(), 5, 2.2e-05, 1.1e-11);
71      }
72  
73      @Test(expected=MathIllegalStateException.class)
74      public void testStartFailure() {
75          doTestStartFailure(Decimal64Field.getInstance());
76      }
