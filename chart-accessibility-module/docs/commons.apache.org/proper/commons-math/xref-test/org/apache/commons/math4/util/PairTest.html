14  package org.apache.commons.math4.util;
15  
16  import org.apache.commons.math4.util.Pair;
17  import org.junit.Assert;
18  import org.junit.Test;
19  
20  /**
21   * Test for {@link Pair}.
22   */
23  public class PairTest {
24  
25      @Test
26      public void testAccessor() {
27          final Pair<Integer, Double> p
28              = new Pair<Integer, Double>(new Integer(1), new Double(2));
29          Assert.assertEquals(new Integer(1), p.getKey());
30          Assert.assertEquals(2, p.getValue().doubleValue(), Math.ulp(1d));
31      }
32  
33      @Test
34      public void testAccessor2() {
35          final Pair<Integer, Double> p
36              = new Pair<Integer, Double>(new Integer(1), new Double(2));
37  
38          // Check that both APIs refer to the same data.
39  
40          Assert.assertTrue(p.getFirst() == p.getKey());
41          Assert.assertTrue(p.getSecond() == p.getValue());
42      }
43  
44      @Test
45      public void testEquals() {
46          Pair<Integer, Double> p1 = new Pair<Integer, Double>(null, null);
47          Assert.assertFalse(p1.equals(null));
48  
49          Pair<Integer, Double> p2 = new Pair<Integer, Double>(null, null);
50          Assert.assertTrue(p1.equals(p2));
51  
52          p1 = new Pair<Integer, Double>(new Integer(1), new Double(2));
53          Assert.assertFalse(p1.equals(p2));
54  
55          p2 = new Pair<Integer, Double>(new Integer(1), new Double(2));
56          Assert.assertTrue(p1.equals(p2));
57  
58          Pair<Integer, Float> p3 = new Pair<Integer, Float>(new Integer(1), new Float(2));
59          Assert.assertFalse(p1.equals(p3));
60      }
61  
62      @Test
63      public void testHashCode() {
64          final MyInteger m1 = new MyInteger(1);
65          final MyInteger m2 = new MyInteger(1);
66  
67          final Pair<MyInteger, MyInteger> p1 = new Pair<MyInteger, MyInteger>(m1, m1);
68          final Pair<MyInteger, MyInteger> p2 = new Pair<MyInteger, MyInteger>(m2, m2);
69          // Same contents, same hash code.
70          Assert.assertTrue(p1.hashCode() == p2.hashCode());
71  
72          // Different contents, different hash codes.
73          m2.set(2);
74          Assert.assertFalse(p1.hashCode() == p2.hashCode());
75      }
76  
77      @Test
78      public void testToString() {
79          Assert.assertEquals("[null, null]", new Pair<Object, Object>(null, null).toString());
80          Assert.assertEquals("[foo, 3]", new Pair<String, Integer>("foo", 3).toString());
81      }
82  
83      @Test
84      public void testCreate() {
85          final Pair<String, Integer> p1 = Pair.create("foo", 3);
86          Assert.assertNotNull(p1);
87          final Pair<String, Integer> p2 = new Pair<String, Integer>("foo", 3);
88          Assert.assertEquals(p2, p1);
89      }
90  
91      /**
92       * A mutable integer.
93       */
94      private static class MyInteger {
95          private int i;
96  
97          public MyInteger(int i) {
98              this.i = i;
99          }
100 
101         public void set(int i) {
102             this.i = i;
103         }
104 
105         @Override
106         public boolean equals(Object o) {
107             if (!(o instanceof MyInteger)) {
108                 return false;
109             } else {
110                 return i == ((MyInteger) o).i;
111             }
112         }
113 
114         @Override
115         public int hashCode() {
116             return i;
117         }
118     }
119 }
