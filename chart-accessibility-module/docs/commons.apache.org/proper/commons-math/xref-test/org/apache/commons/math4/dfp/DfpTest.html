66      // Generic test function.  Takes params x and y and tests them for
67      // equality.  Then checks the status flags against the flags argument.
68      // If the test fail, it prints the desc string
69      private void test(Dfp x, Dfp y, int flags, String desc)
70      {
71          boolean b = x.equals(y);
72  
73          if (!x.equals(y) && !x.unequal(y))  // NaNs involved
74              b = (x.toString().equals(y.toString()));
75  
76          if (x.equals(field.newDfp("0")))  // distinguish +/- zero
77              b = (b && (x.toString().equals(y.toString())));
78  
79          b = (b && x.getField().getIEEEFlags() == flags);
80  
81          if (!b)
82              Assert.assertTrue("assersion failed "+desc+" x = "+x.toString()+" flags = "+x.getField().getIEEEFlags(), b);
83  
84          x.getField().clearIEEEFlags();
85      }

375     // utility function to help test comparisons
376     private void cmptst(Dfp a, Dfp b, String op, boolean result, double num)
377     {
378         if (op == "equal")
379             if (a.equals(b) != result)
380                 Assert.fail("assersion failed.  "+op+" compare #"+num);
381 
382         if (op == "unequal")
383             if (a.unequal(b) != result)
384                 Assert.fail("assersion failed.  "+op+" compare #"+num);
385 
386         if (op == "lessThan")
387             if (a.lessThan(b) != result)
388                 Assert.fail("assersion failed.  "+op+" compare #"+num);
389 
390         if (op == "greaterThan")
391             if (a.greaterThan(b) != result)
392                 Assert.fail("assersion failed.  "+op+" compare #"+num);
393     }
