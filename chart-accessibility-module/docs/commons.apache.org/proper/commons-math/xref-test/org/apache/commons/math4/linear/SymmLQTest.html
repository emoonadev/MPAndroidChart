44      public void saundersTest(final int n, final boolean goodb,
45                               final boolean precon, final double shift,
46                               final double pertbn) {
47          final RealLinearOperator a = new RealLinearOperator() {
48  
49              @Override
50              public RealVector operate(final RealVector x) {
51                  if (x.getDimension() != n) {
52                      throw new DimensionMismatchException(x.getDimension(), n);
53                  }
54                  final double[] y = new double[n];
55                  for (int i = 0; i < n; i++) {
56                      y[i] = (i + 1) * 1.1 / n * x.getEntry(i);
57                  }
58                  return new ArrayRealVector(y, false);
59              }
60  
61              @Override
62              public int getRowDimension() {
63                  return n;
64              }
65  
66              @Override
67              public int getColumnDimension() {
68                  return n;
69              }
70          };
71          final double shiftm = shift;
72          final double pertm = FastMath.abs(pertbn);
73          final RealLinearOperator minv;
74          if (precon) {
75              minv = new RealLinearOperator() {
76                  @Override
77                  public int getRowDimension() {
78                      return n;
79                  }
80  
81                  @Override
82                  public int getColumnDimension() {
83                      return n;
84                  }
85  
86                  @Override
87                  public RealVector operate(final RealVector x) {
88                      if (x.getDimension() != n) {
89                          throw new DimensionMismatchException(x.getDimension(),
90                                                               n);
91                      }
92                      final double[] y = new double[n];
93                      for (int i = 0; i < n; i++) {
94                          double d = (i + 1) * 1.1 / n;
95                          d = FastMath.abs(d - shiftm);
96                          if (i % 10 == 0) {
97                              d += pertm;
98                          }
99                          y[i] = x.getEntry(i) / d;
100                     }
101                     return new ArrayRealVector(y, false);
102                 }
103             };
104         } else {
105             minv = null;
106         }
107         final RealVector xtrue = new ArrayRealVector(n);
108         for (int i = 0; i < n; i++) {
109             xtrue.setEntry(i, n - i);
110         }
111         final RealVector b = a.operate(xtrue);
112         b.combineToSelf(1.0, -shift, xtrue);
113         final SymmLQ solver = new SymmLQ(2 * n, 1E-12, true);
114         final RealVector x = solver.solve(a, minv, b, goodb, shift);
115         final RealVector y = a.operate(x);
116         final RealVector r1 = new ArrayRealVector(n);
117         for (int i = 0; i < n; i++) {
118             final double bi = b.getEntry(i);
119             final double yi = y.getEntry(i);
120             final double xi = x.getEntry(i);
121             r1.setEntry(i, bi - yi + shift * xi);
122         }
123         final double enorm = x.subtract(xtrue).getNorm() / xtrue.getNorm();
124         final double etol = 1E-5;
125         Assert.assertTrue("enorm=" + enorm + ", " +
126         solver.getIterationManager().getIterations(), enorm <= etol);
127     }

484                 final double delta = 5E-5 * FastMath.abs(expected);
485                 Assert.assertEquals(msg, expected, actual, delta);
486             }

490     @Test
491     public void testEventManagement() {
492         final int n = 5;
493         final int maxIterations = 100;
494         final RealLinearOperator a = new HilbertMatrix(n);
495         final IterativeLinearSolver solver;
496         /*
497          * count[0] = number of calls to initializationPerformed
498          * count[1] = number of calls to iterationStarted
499          * count[2] = number of calls to iterationPerformed
500          * count[3] = number of calls to terminationPerformed
501          */
502         final int[] count = new int[] {0, 0, 0, 0};
503         final RealVector xFromListener = new ArrayRealVector(n);
504         final IterationListener listener = new IterationListener() {
505 
506             public void initializationPerformed(final IterationEvent e) {
507                 ++count[0];
508             }
509 
510             public void iterationPerformed(final IterationEvent e) {
511                 ++count[2];
512                 Assert.assertEquals("iteration performed",
513                                     count[2],
514                                     e.getIterations() - 1);
515             }
516 
517             public void iterationStarted(final IterationEvent e) {
518                 ++count[1];
519                 Assert.assertEquals("iteration started",
520                                     count[1],
521                                     e.getIterations() - 1);
522             }
523 
524             public void terminationPerformed(final IterationEvent e) {
525                 ++count[3];
526                 final IterativeLinearSolverEvent ilse;
527                 ilse = (IterativeLinearSolverEvent) e;
528                 xFromListener.setSubVector(0, ilse.getSolution());
529             }
530         };
531         solver = new SymmLQ(maxIterations, 1E-10, true);
532         solver.getIterationManager().addIterationListener(listener);
533         final RealVector b = new ArrayRealVector(n);
534         for (int j = 0; j < n; j++) {
535             Arrays.fill(count, 0);
536             b.set(0.);
537             b.setEntry(j, 1.);
538             final RealVector xFromSolver = solver.solve(a, b);
539             String msg = String.format("column %d (initialization)", j);
540             Assert.assertEquals(msg, 1, count[0]);
541             msg = String.format("column %d (finalization)", j);
542             Assert.assertEquals(msg, 1, count[3]);
543             /*
544              *  Check that solution is not "over-refined". When the last
545              *  iteration has occurred, no further refinement should be
546              *  performed.
547              */
548             for (int i = 0; i < n; i++){
549                 msg = String.format("row %d, column %d", i, j);
550                 final double expected = xFromSolver.getEntry(i);
551                 final double actual = xFromListener.getEntry(i);
552                 Assert.assertEquals(msg, expected, actual, 0.0);
553             }
554         }
555     }

664             private void doTestNormOfResidual(final IterationEvent e) {
665                 final IterativeLinearSolverEvent evt;
666                 evt = (IterativeLinearSolverEvent) e;
667                 final RealVector x = evt.getSolution();
668                 final RealVector b = evt.getRightHandSideVector();
669                 final RealVector r = b.subtract(a.operate(x));
670                 final double rnorm = p.operate(r).getNorm();
671                 Assert.assertEquals("iteration performed (residual)",
672                     rnorm, evt.getNormOfResidual(),
673                     FastMath.max(1E-5 * rnorm, 1E-10));
674             }
675 
676             public void initializationPerformed(final IterationEvent e) {
677                 doTestNormOfResidual(e);
678             }
679 
680             public void iterationPerformed(final IterationEvent e) {
681                 doTestNormOfResidual(e);
682             }
683 
684             public void iterationStarted(final IterationEvent e) {
685                 doTestNormOfResidual(e);
686             }
687 
688             public void terminationPerformed(final IterationEvent e) {
689                 doTestNormOfResidual(e);
690             }
691         };
692         solver = new SymmLQ(maxIterations, 1E-10, true);
693         solver.getIterationManager().addIterationListener(listener);
694         final RealVector b = new ArrayRealVector(n);
695         for (int j = 0; j < n; j++) {
696             b.set(0.);
