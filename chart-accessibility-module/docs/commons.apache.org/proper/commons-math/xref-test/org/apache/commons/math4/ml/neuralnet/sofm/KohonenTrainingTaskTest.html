44      @Test
45      public void testTravellerSalesmanSquareTourSequentialSolver() {
46          // Cities (in optimal travel order).
47          final City[] squareOfCities = new City[] {
48              new City("o0", 0, 0),
49              new City("o1", 1, 0),
50              new City("o2", 2, 0),
51              new City("o3", 3, 0),
52              new City("o4", 3, 1),
53              new City("o5", 3, 2),
54              new City("o6", 3, 3),
55              new City("o7", 2, 3),
56              new City("o8", 1, 3),
57              new City("o9", 0, 3),
58              new City("i3", 1, 2),
59              new City("i2", 2, 2),
60              new City("i1", 2, 1),
61              new City("i0", 1, 1),
62          };
63  
64          // Seed that allows the unit test to always succeed.
65          final long seed = 1245632379L;
66  
67          final TravellingSalesmanSolver solver = new TravellingSalesmanSolver(squareOfCities, 2, seed);
68          // printSummary("before.travel.seq.dat", solver);
69          final Runnable task = solver.createSequentialTask(15000);
70          task.run();
71  
72          // All update attempts must be successful in the absence of concurrency.
73          Assert.assertEquals(solver.getUpdateRatio(), 1, 0d);
74  
75          // printSummary("after.travel.seq.dat", solver);
76          final City[] result = solver.getCityList();
77          Assert.assertEquals(squareOfCities.length,
78                              uniqueCities(result).size());
79          final double ratio = computeTravelDistance(squareOfCities) / computeTravelDistance(result);
80          Assert.assertEquals(1, ratio, 1e-1); // We do not require the optimal travel.
81      }

84      @Test
85      @Retry
86      public void testTravellerSalesmanSquareTourParallelSolver() throws ExecutionException {
87          // Cities (in optimal travel order).
88          final City[] squareOfCities = new City[] {
89              new City("o0", 0, 0),
90              new City("o1", 1, 0),
91              new City("o2", 2, 0),
92              new City("o3", 3, 0),
93              new City("o4", 3, 1),
94              new City("o5", 3, 2),
95              new City("o6", 3, 3),
96              new City("o7", 2, 3),
97              new City("o8", 1, 3),
98              new City("o9", 0, 3),
99              new City("i3", 1, 2),
100             new City("i2", 2, 2),
101             new City("i1", 2, 1),
102             new City("i0", 1, 1),
103         };
104 
105         // Seed that allows the unit test to always succeed.
106         final long seed = 534712311L;
107 
108         final TravellingSalesmanSolver solver = new TravellingSalesmanSolver(squareOfCities, 2, seed);
109         // printSummary("before.travel.par.dat", solver);
110 
111         // Parallel execution.
112         final ExecutorService service = Executors.newCachedThreadPool();
113         final int numProcs = Runtime.getRuntime().availableProcessors();
114         final Runnable[] tasks = solver.createParallelTasks(numProcs, 5000);
115         final List<Future<?>> execOutput = new ArrayList<Future<?>>();
116         // Run tasks.
117         for (Runnable r : tasks) {
118             execOutput.add(service.submit(r));
119         }
120         // Wait for completion (ignoring return value).
121         try {
122             for (Future<?> f : execOutput) {
123                 f.get();
124             }
125         } catch (InterruptedException ignored) {}
126         // Terminate all threads.
127         service.shutdown();
128 
129         if (numProcs > 1) {
130             // We expect that some update attempts will be concurrent.
131             Assert.assertTrue(solver.getUpdateRatio() < 1);
132         }
133 
134         // printSummary("after.travel.par.dat", solver);
135         final City[] result = solver.getCityList();
136         Assert.assertEquals(squareOfCities.length,
137                             uniqueCities(result).size());
138         final double ratio = computeTravelDistance(squareOfCities) / computeTravelDistance(result);
139         Assert.assertEquals(1, ratio, 1e-1); // We do not require the optimal travel.
140     }

142     /**
143      * Creates a map of the travel suggested by the solver.
144      *
145      * @param solver Solver.
146      * @return a 4-columns table: {@code <x (neuron)> <y (neuron)> <x (city)> <y (city)>}.
147      */
148     private String travelCoordinatesTable(TravellingSalesmanSolver solver) {
149         final StringBuilder s = new StringBuilder();
150         for (double[] c : solver.getCoordinatesList()) {
151             s.append(c[0]).append(" ").append(c[1]).append(" ");
152             final City city = solver.getClosestCity(c[0], c[1]);
153             final double[] cityCoord = city.getCoordinates();
154             s.append(cityCoord[0]).append(" ").append(cityCoord[1]).append(" ");
155             s.append("   # ").append(city.getName()).append("\n");
156         }
157         return s.toString();
158     }

160     /**
161      * Compute the distance covered by the salesman, including
162      * the trip back (from the last to first city).
163      *
164      * @param cityList List of cities visited during the travel.
165      * @return the total distance.
166      */
167     private Collection<City> uniqueCities(City[] cityList) {
168         final Set<City> unique = new HashSet<City>();
169         for (City c : cityList) {
170             unique.add(c);
171         }
172         return unique;
173     }
174 
175     /**
176      * Compute the distance covered by the salesman, including
177      * the trip back (from the last to first city).
178      *
179      * @param cityList List of cities visited during the travel.
180      * @return the total distance.
181      */
182     private double computeTravelDistance(City[] cityList) {
183         double dist = 0;
184         for (int i = 0; i < cityList.length; i++) {
185             final double[] currentCoord = cityList[i].getCoordinates();
186             final double[] nextCoord = cityList[(i + 1) % cityList.length].getCoordinates();
187 
188             final double xDiff = currentCoord[0] - nextCoord[0];
189             final double yDiff = currentCoord[1] - nextCoord[1];
190 
191             dist += FastMath.sqrt(xDiff * xDiff + yDiff * yDiff);
192         }
193 
194         return dist;
195     }

197     /**
198      * Prints a summary of the current state of the solver to the
199      * given filename.
200      *
201      * @param filename File.
202      * @param solver Solver.
203      */
204     @SuppressWarnings("unused")
205     private void printSummary(String filename,
206                               TravellingSalesmanSolver solver) {
207         PrintWriter out = null;
208         try {
209             out = new PrintWriter(filename);
210             out.println(travelCoordinatesTable(solver));
211 
212             final City[] result = solver.getCityList();
213             out.println("# Number of unique cities: " + uniqueCities(result).size());
214             out.println("# Travel distance: " + computeTravelDistance(result));
215         } catch (Exception e) {
216             // Do nothing.
217         } finally {
218             if (out != null) {
219                 out.close();
220             }
221         }
222     }
