17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.exception.DimensionMismatchException;
20  import org.apache.commons.math4.linear.ArrayRealVector;
21  import org.apache.commons.math4.linear.RealLinearOperator;
22  import org.apache.commons.math4.linear.RealVector;
23  
24  /** This class implements Hilbert Matrices as {@link RealLinearOperator}. */
25  public class HilbertMatrix
26      extends RealLinearOperator {
27  
28      /** The size of the matrix. */
29      private final int n;
30  
31      /**
32       * Creates a new instance of this class.
33       *
34       * @param n Size of the matrix to be created..
35       */
36      public HilbertMatrix(final int n) {
37          this.n = n;
38      }
39  
40      /** {@inheritDoc} */
41      @Override
42      public int getColumnDimension() {
43          return n;
44      }
45  
46      /** {@inheritDoc} */
47      @Override
48      public int getRowDimension() {
49          return n;
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      public RealVector operate(final RealVector x) {
55          if (x.getDimension() != n) {
56              throw new DimensionMismatchException(x.getDimension(), n);
57          }
58          final double[] y = new double[n];
59          for (int i = 0; i < n; i++) {
60              double pos = 0.;
61              double neg = 0.;
62              for (int j = 0; j < n; j++) {
63                  final double xj = x.getEntry(j);
64                  final double coeff = 1. / (i + j + 1.);
65                  // Positive and negative values are sorted out in order to limit
66                  // catastrophic cancellations (do not forget that Hilbert
67                  // matrices are *very* ill-conditioned!
68                  if (xj > 0.) {
69                      pos += coeff * xj;
70                  } else {
71                      neg += coeff * xj;
72                  }
73              }
74              y[i] = pos + neg;
75          }
76          return new ArrayRealVector(y, false);
77      }
78  }
