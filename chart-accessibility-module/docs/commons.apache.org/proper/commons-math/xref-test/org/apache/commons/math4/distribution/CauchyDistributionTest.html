18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.distribution.CauchyDistribution;
21  import org.apache.commons.math4.distribution.NormalDistribution;
22  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for CauchyDistribution.
28   * Extends ContinuousDistributionAbstractTest.  See class javadoc for
29   * ContinuousDistributionAbstractTest for details.
30   *
31   */
32  public class CauchyDistributionTest extends RealDistributionAbstractTest {
33  
34      // --------------------- Override tolerance  --------------
35      protected double defaultTolerance = NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY;
36      @Override
37      public void setUp() {
38          super.setUp();
39          setTolerance(defaultTolerance);
40      }
41  
42      //-------------- Implementations for abstract methods -----------------------
43  
44      /** Creates the default continuous distribution instance to use in tests. */
45      @Override
46      public CauchyDistribution makeDistribution() {
47          return new CauchyDistribution(1.2, 2.1);
48      }
49  
50      /** Creates the default cumulative probability distribution test input values */
51      @Override
52      public double[] makeCumulativeTestPoints() {
53          // quantiles computed using R 2.9.2
54          return new double[] {-667.24856187, -65.6230835029, -25.4830299460, -12.0588781808,
55                  -5.26313542807, 669.64856187, 68.0230835029, 27.8830299460, 14.4588781808, 7.66313542807};
56      }
57  
58      /** Creates the default cumulative probability density test expected values */
59      @Override
60      public double[] makeCumulativeTestValues() {
61          return new double[] {0.001, 0.01, 0.025, 0.05, 0.1, 0.999,
62                  0.990, 0.975, 0.950, 0.900};
63      }
64  
65      /** Creates the default probability density test expected values */
66      @Override
67      public double[] makeDensityTestValues() {
68          return new double[] {1.49599158008e-06, 0.000149550440335, 0.000933076881878, 0.00370933207799, 0.0144742330437,
69                  1.49599158008e-06, 0.000149550440335, 0.000933076881878, 0.00370933207799, 0.0144742330437};
70      }
71  
72      //---------------------------- Additional test cases -------------------------
73  
74      @Test
75      public void testInverseCumulativeProbabilityExtremes() {
76          setInverseCumulativeTestPoints(new double[] {0.0, 1.0});
77          setInverseCumulativeTestValues(
78                  new double[] {Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY});
79          verifyInverseCumulativeProbabilities();
80      }
81  
82      @Test
83      public void testMedian() {
84          CauchyDistribution distribution = (CauchyDistribution) getDistribution();
85          Assert.assertEquals(1.2, distribution.getMedian(), 0.0);
86      }
87  
88      @Test
89      public void testScale() {
90          CauchyDistribution distribution = (CauchyDistribution) getDistribution();
91          Assert.assertEquals(2.1, distribution.getScale(), 0.0);
92      }
93  
94      @Test
95      public void testPreconditions() {
96          try {
97              new CauchyDistribution(0, 0);
98              Assert.fail("Cannot have zero scale");
99          } catch (NotStrictlyPositiveException ex) {
100             // Expected.
101         }
102         try {
103             new CauchyDistribution(0, -1);
104             Assert.fail("Cannot have negative scale");
105         } catch (NotStrictlyPositiveException ex) {
106             // Expected.
107         }
108     }
109 
110     @Test
111     public void testMoments() {
112         CauchyDistribution dist;
113 
114         dist = new CauchyDistribution(10.2, 0.15);
115         Assert.assertTrue(Double.isNaN(dist.getNumericalMean()));
116         Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
117 
118         dist = new CauchyDistribution(23.12, 2.12);
119         Assert.assertTrue(Double.isNaN(dist.getNumericalMean()));
120         Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
121     }
122 }
