59    @Test
60    public void testRandomAccess() throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
61  
62      ContinuousOutputModel cm = new ContinuousOutputModel();
63      integ.addStepHandler(cm);
64      integ.integrate(pb,
65                      pb.getInitialTime(), pb.getInitialState(),
66                      pb.getFinalTime(), new double[pb.getDimension()]);
67  
68      Random random = new Random(347588535632l);
69      double maxError    = 0.0;
70      double maxErrorDot = 0.0;
71      for (int i = 0; i < 1000; ++i) {
72        double r = random.nextDouble();
73        double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
74        cm.setInterpolatedTime(time);
75        double[] interpolatedY    = cm.getInterpolatedState();
76        double[] interpolatedYDot = cm.getInterpolatedDerivatives();
77        double[] theoreticalY     = pb.computeTheoreticalState(time);
78        double[] theoreticalYDot  = new double[pb.getDimension()];
79        pb.doComputeDerivatives(time, theoreticalY, theoreticalYDot);
80        double dx = interpolatedY[0] - theoreticalY[0];
81        double dy = interpolatedY[1] - theoreticalY[1];
82        double error = dx * dx + dy * dy;
83        maxError = FastMath.max(maxError, error);
84        double dxDot = interpolatedYDot[0] - theoreticalYDot[0];
85        double dyDot = interpolatedYDot[1] - theoreticalYDot[1];
86        double errorDot = dxDot * dxDot + dyDot * dyDot;
87        maxErrorDot = FastMath.max(maxErrorDot, errorDot);
88      }
89  
90      Assert.assertEquals(0.0, maxError,    1.0e-9);
91      Assert.assertEquals(0.0, maxErrorDot, 4.0e-7);
92  
93    }

95    @Test
96    public void testModelsMerging() throws MaxCountExceededException, MathIllegalArgumentException {
97  
98        // theoretical solution: y[0] = cos(t), y[1] = sin(t)
99        FirstOrderDifferentialEquations problem =
100           new FirstOrderDifferentialEquations() {
101               @Override
102             public void computeDerivatives(double t, double[] y, double[] dot) {
103                   dot[0] = -y[1];
104                   dot[1] =  y[0];
105               }
106               @Override
107             public int getDimension() {
108                   return 2;
109               }
110           };
111 
112       // integrate backward from &pi; to 0;
113       ContinuousOutputModel cm1 = new ContinuousOutputModel();
114       FirstOrderIntegrator integ1 =
115           new DormandPrince853Integrator(0, 1.0, 1.0e-8, 1.0e-8);
116       integ1.addStepHandler(cm1);
117       integ1.integrate(problem, FastMath.PI, new double[] { -1.0, 0.0 },
118                        0, new double[2]);
119 
120       // integrate backward from 2&pi; to &pi;
121       ContinuousOutputModel cm2 = new ContinuousOutputModel();
122       FirstOrderIntegrator integ2 =
123           new DormandPrince853Integrator(0, 0.1, 1.0e-12, 1.0e-12);
124       integ2.addStepHandler(cm2);
125       integ2.integrate(problem, 2.0 * FastMath.PI, new double[] { 1.0, 0.0 },
126                        FastMath.PI, new double[2]);
127 
128       // merge the two half circles
129       ContinuousOutputModel cm = new ContinuousOutputModel();
130       cm.append(cm2);
131       cm.append(new ContinuousOutputModel());
132       cm.append(cm1);
133 
134       // check circle
135       Assert.assertEquals(2.0 * FastMath.PI, cm.getInitialTime(), 1.0e-12);
136       Assert.assertEquals(0, cm.getFinalTime(), 1.0e-12);
137       Assert.assertEquals(cm.getFinalTime(), cm.getInterpolatedTime(), 1.0e-12);
138       for (double t = 0; t < 2.0 * FastMath.PI; t += 0.1) {
139           cm.setInterpolatedTime(t);
140           double[] y = cm.getInterpolatedState();
141           Assert.assertEquals(FastMath.cos(t), y[0], 1.0e-7);
142           Assert.assertEquals(FastMath.sin(t), y[1], 1.0e-7);
143       }
144 
145   }

147   @Test
148   public void testErrorConditions() throws MaxCountExceededException, MathIllegalArgumentException {
149 
150       ContinuousOutputModel cm = new ContinuousOutputModel();
151       cm.handleStep(buildInterpolator(0, new double[] { 0.0, 1.0, -2.0 }, 1), true);
152 
153       // dimension mismatch
154       Assert.assertTrue(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0 }, 2.0));
155 
156       // hole between time ranges
157       Assert.assertTrue(checkAppendError(cm, 10.0, new double[] { 0.0, 1.0, -2.0 }, 20.0));
158 
159       // propagation direction mismatch
160       Assert.assertTrue(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0, -2.0 }, 0.0));
161 
162       // no errors
163       Assert.assertFalse(checkAppendError(cm, 1.0, new double[] { 0.0, 1.0, -2.0 }, 2.0));
164 
165   }
166 
167   private boolean checkAppendError(ContinuousOutputModel cm,
168                                    double t0, double[] y0, double t1)
169       throws MaxCountExceededException, MathIllegalArgumentException {
170       try {
171           ContinuousOutputModel otherCm = new ContinuousOutputModel();
172           otherCm.handleStep(buildInterpolator(t0, y0, t1), true);
173           cm.append(otherCm);
174       } catch(MathIllegalArgumentException iae) {
175           return true; // there was an allowable error
176       }
177       return false; // no allowable error
178   }
179 
180   private StepInterpolator buildInterpolator(double t0, double[] y0, double t1) {
181       DummyStepInterpolator interpolator  = new DummyStepInterpolator(y0, new double[y0.length], t1 >= t0);
182       interpolator.storeTime(t0);
183       interpolator.shift();
184       interpolator.storeTime(t1);
185       return interpolator;
186   }
