17  package org.apache.commons.math4.analysis.solvers;
18  
19  import org.apache.commons.math4.analysis.QuinticFunction;
20  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
21  import org.apache.commons.math4.analysis.function.Sin;
22  import org.apache.commons.math4.analysis.solvers.NewtonRaphsonSolver;
23  import org.apache.commons.math4.util.FastMath;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  
28  /**
29   */
30  public final class NewtonRaphsonSolverTest {
31      /**
32       *
33       */
34      @Test
35      public void testSinZero() {
36          UnivariateDifferentiableFunction f = new Sin();
37          double result;
38  
39          NewtonRaphsonSolver solver = new NewtonRaphsonSolver();
40          result = solver.solve(100, f, 3, 4);
41          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
42  
43          result = solver.solve(100, f, 1, 4);
44          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
45  
46          Assert.assertTrue(solver.getEvaluations() > 0);
47      }
48  
49      /**
50       *
51       */
52      @Test
53      public void testQuinticZero() {
54          final UnivariateDifferentiableFunction f = new QuinticFunction();
55          double result;
56  
57          NewtonRaphsonSolver solver = new NewtonRaphsonSolver();
58          result = solver.solve(100, f, -0.2, 0.2);
59          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
60  
61          result = solver.solve(100, f, -0.1, 0.3);
62          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
63  
64          result = solver.solve(100, f, -0.3, 0.45);
65          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
66  
67          result = solver.solve(100, f, 0.3, 0.7);
68          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
69  
70          result = solver.solve(100, f, 0.2, 0.6);
71          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
72  
73          result = solver.solve(100, f, 0.05, 0.95);
74          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
75  
76          result = solver.solve(100, f, 0.85, 1.25);
77          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
78  
79          result = solver.solve(100, f, 0.8, 1.2);
80          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
81  
82          result = solver.solve(100, f, 0.85, 1.75);
83          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
84  
85          result = solver.solve(100, f, 0.55, 1.45);
86          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
87  
88          result = solver.solve(100, f, 0.85, 5);
89          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
90      }
91  }
