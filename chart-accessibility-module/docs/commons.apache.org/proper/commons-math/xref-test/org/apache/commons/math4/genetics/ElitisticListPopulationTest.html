17  package org.apache.commons.math4.genetics;
18  
19  
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.apache.commons.math4.exception.OutOfRangeException;
24  import org.apache.commons.math4.genetics.Chromosome;
25  import org.apache.commons.math4.genetics.ElitisticListPopulation;
26  import org.apache.commons.math4.genetics.Population;
27  import org.junit.Assert;
28  import org.junit.Test;
29  
30  public class ElitisticListPopulationTest {
31  
32      private static int counter = 0;
33  
34      @Test
35      public void testNextGeneration() {
36          ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
37  
38          for (int i=0; i<pop.getPopulationLimit(); i++) {
39              pop.addChromosome(new DummyChromosome());
40          }
41  
42          Population nextGeneration = pop.nextGeneration();
43  
44          Assert.assertEquals(20, nextGeneration.getPopulationSize());
45      }
46  
47      @Test
48      public void testSetElitismRate() {
49          final double rate = 0.25;
50          final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
51          pop.setElitismRate(rate);
52          Assert.assertEquals(rate, pop.getElitismRate(), 1e-6);
53      }
54  
55      @Test(expected = OutOfRangeException.class)
56      public void testSetElitismRateTooLow() {
57          final double rate = -0.25;
58          final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
59          pop.setElitismRate(rate);
60      }
61  
62      @Test(expected = OutOfRangeException.class)
63      public void testSetElitismRateTooHigh() {
64          final double rate = 1.25;
65          final ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
66          pop.setElitismRate(rate);
67      }
68  
69      @Test(expected = OutOfRangeException.class)
70      public void testConstructorTooLow() {
71          final double rate = -0.25;
72          new ElitisticListPopulation(100, rate);
73      }
74  
75      @Test(expected = OutOfRangeException.class)
76      public void testConstructorTooHigh() {
77          final double rate = 1.25;
78          new ElitisticListPopulation(100, rate);
79      }
80  
81      @Test(expected = OutOfRangeException.class)
82      public void testChromosomeListConstructorTooLow() {
83          final List<Chromosome> chromosomes = Collections.emptyList();
84          final double rate = -0.25;
85          new ElitisticListPopulation(chromosomes, 100, rate);
86      }
87  
88      @Test(expected = OutOfRangeException.class)
89      public void testChromosomeListConstructorTooHigh() {
90          final List<Chromosome> chromosomes = Collections.emptyList();
91          final double rate = 1.25;
92          new ElitisticListPopulation(chromosomes, 100, rate);
93      }
94  
95      private static class DummyChromosome extends Chromosome {
96          private final int fitness;
97  
98          public DummyChromosome() {
99              this.fitness = counter;
100             counter++;
101         }
102 
103         public double fitness() {
104             return this.fitness;
105         }
106     }
107 
108 }
