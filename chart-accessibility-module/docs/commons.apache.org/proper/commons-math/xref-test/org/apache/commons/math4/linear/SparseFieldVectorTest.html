183         final SparseFieldVector<Fraction> u
184             = new SparseFieldVector<Fraction>(FractionField.getInstance(),
185                                               new Fraction[] {new Fraction(1),
186                                                               new Fraction(2),
187                                                               new Fraction(-3)});
188         final SparseFieldVector<Fraction> v
189             = new SparseFieldVector<Fraction>(FractionField.getInstance(),
190                                               new Fraction[] {new Fraction(4),

192

231     /** verifies that two vectors are close (sup norm) */
232     protected void assertEquals(String msg, Fraction[] m, Fraction[] n) {
233         if (m.length != n.length) {
234             Assert.fail("vectors have different lengths");
235         }
236         for (int i = 0; i < m.length; i++) {
237             Assert.assertEquals(msg + " " +  i + " elements differ", m[i],n[i]);
238         }
239     }

241     /** verifies that two vectors are close (sup norm) */
242     protected void assertClose(String msg, Fraction[] m, Fraction[] n, double tolerance) {
243         if (m.length != n.length) {
244             Assert.fail("vectors have different lengths");
245         }
246         for (int i = 0; i < m.length; i++) {
247             Assert.assertEquals(msg + " " +  i + " elements differ", m[i].doubleValue(),n[i].doubleValue(), tolerance);
248         }
