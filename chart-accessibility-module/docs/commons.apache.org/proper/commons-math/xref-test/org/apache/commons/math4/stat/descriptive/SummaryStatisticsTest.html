78      public void testN0andN1Conditions() {
79          SummaryStatistics u = createSummaryStatistics();
80          Assert.assertTrue("Mean of n = 0 set should be NaN",
81                  Double.isNaN( u.getMean() ) );
82          Assert.assertTrue("Standard Deviation of n = 0 set should be NaN",

101                 u.getStandardDeviation() != 0.0);

188     @Test
189     public void testEqualsAndHashCode() {
190         SummaryStatistics u = createSummaryStatistics();
191         SummaryStatistics t = null;
192         int emptyHash = u.hashCode();
193         Assert.assertTrue("reflexive", u.equals(u));
194         Assert.assertFalse("non-null compared to null", u.equals(t));
195         Assert.assertFalse("wrong type", u.equals(Double.valueOf(0)));
196         t = createSummaryStatistics();
197         Assert.assertTrue("empty instances should be equal", t.equals(u));
198         Assert.assertTrue("empty instances should be equal", u.equals(t));
199         Assert.assertEquals("empty hash code", emptyHash, t.hashCode());
200 
201         // Add some data to u
202         u.addValue(2d);
203         u.addValue(1d);
204         u.addValue(3d);
205         u.addValue(4d);
206         Assert.assertFalse("different n's should make instances not equal", t.equals(u));
207         Assert.assertFalse("different n's should make instances not equal", u.equals(t));
208         Assert.assertTrue("different n's should make hashcodes different",
209                 u.hashCode() != t.hashCode());
210 
211         //Add data in same order to t
212         t.addValue(2d);
213         t.addValue(1d);
214         t.addValue(3d);
215         t.addValue(4d);
216         Assert.assertTrue("summaries based on same data should be equal", t.equals(u));
217         Assert.assertTrue("summaries based on same data should be equal", u.equals(t));
218         Assert.assertEquals("summaries based on same data should have same hashcodes",
219                 u.hashCode(), t.hashCode());
220 
221         // Clear and make sure summaries are indistinguishable from empty summary
222         u.clear();
223         t.clear();
224         Assert.assertTrue("empty instances should be equal", t.equals(u));
225         Assert.assertTrue("empty instances should be equal", u.equals(t));
226         Assert.assertEquals("empty hash code", emptyHash, t.hashCode());
227         Assert.assertEquals("empty hash code", emptyHash, u.hashCode());
228     }

232         SummaryStatistics u = createSummaryStatistics();
233         u.addValue(2d);
234         u.addValue(1d);
235         u.addValue(3d);
236         u.addValue(4d);
237         SummaryStatistics v = new SummaryStatistics(u);
238         Assert.assertEquals(u, v);
239         Assert.assertEquals(v, u);
240 
241         // Make sure both behave the same with additional values added
242         u.addValue(7d);
243         u.addValue(9d);
244         u.addValue(11d);
245         u.addValue(23d);
246         v.addValue(7d);
247         v.addValue(9d);
248         v.addValue(11d);
249         v.addValue(23d);
250         Assert.assertEquals(u, v);
251         Assert.assertEquals(v, u);
252 
253         // Check implementation pointers are preserved
254         u.clear();
255         u.setSumImpl(new Sum());
256         SummaryStatistics.copy(u,v);
257         Assert.assertEquals(u.getSumImpl(), v.getSumImpl());
258

273         SummaryStatistics u = createSummaryStatistics();

284         Assert.assertEquals(3, u.getMean(), 1E-14);

286         u.setMeanImpl(new Mean()); // OK after clear

302     @Test
303     public void testQuadraticMean() {
304         final double[] values = { 1.2, 3.4, 5.6, 7.89 };
305         final SummaryStatistics stats = createSummaryStatistics();
306 
307         final int len = values.length;
308         double expected = 0;
309         for (int i = 0; i < len; i++) {
310             final double v = values[i];
311             expected += v * v / len;
312 
313             stats.addValue(v);
314         }
315         expected = Math.sqrt(expected);
316 
317         Assert.assertEquals(expected, stats.getQuadraticMean(), Math.ulp(expected));
318     }

356     @Test
357     public void testToString() {
358         SummaryStatistics u = createSummaryStatistics();
359         for (int i = 0; i < 5; i++) {
360             u.addValue(i);
361         }
362         final String[] labels = {"min", "max", "sum", "geometric mean", "variance",
363                 "population variance", "second moment", "sum of squares", "standard deviation",
364         "sum of logs"};
365         final double[] values = {u.getMin(), u.getMax(), u.getSum(), u.getGeometricMean(),
366                 u.getVariance(), u.getPopulationVariance(), u.getSecondMoment(), u.getSumsq(),
367                 u.getStandardDeviation(), u.getSumOfLogs()};
368         final String toString = u.toString();
369         Assert.assertTrue(toString.indexOf("n: " + u.getN()) > 0); // getN() returns a long
370         for (int i = 0; i < values.length; i++) {
371             Assert.assertTrue(toString.indexOf(labels[i] + ": " + String.valueOf(values[i])) > 0);
372         }
