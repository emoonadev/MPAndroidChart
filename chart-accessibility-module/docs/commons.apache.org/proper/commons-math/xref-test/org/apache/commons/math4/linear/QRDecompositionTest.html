61      /** test dimensions */
62      @Test
63      public void testDimensions() {
64          checkDimension(MatrixUtils.createRealMatrix(testData3x3NonSingular));
65  
66          checkDimension(MatrixUtils.createRealMatrix(testData4x3));
67  
68          checkDimension(MatrixUtils.createRealMatrix(testData3x4));
69  
70          Random r = new Random(643895747384642l);
71          int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
72          int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
73          checkDimension(createTestMatrix(r, p, q));
74          checkDimension(createTestMatrix(r, q, p));
75  
76      }

78      private void checkDimension(RealMatrix m) {
79          int rows = m.getRowDimension();
80          int columns = m.getColumnDimension();
81          QRDecomposition qr = new QRDecomposition(m);
82          Assert.assertEquals(rows,    qr.getQ().getRowDimension());
83          Assert.assertEquals(rows,    qr.getQ().getColumnDimension());
84          Assert.assertEquals(rows,    qr.getR().getRowDimension());
85          Assert.assertEquals(columns, qr.getR().getColumnDimension());
86      }

90      public void testAEqualQR() {
91          checkAEqualQR(MatrixUtils.createRealMatrix(testData3x3NonSingular));

93          checkAEqualQR(MatrixUtils.createRealMatrix(testData3x3Singular));

95          checkAEqualQR(MatrixUtils.createRealMatrix(testData3x4));

101         int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
102         checkAEqualQR(createTestMatrix(r, p, q));
103 
104         checkAEqualQR(createTestMatrix(r, q, p));
105 
106     }

108     private void checkAEqualQR(RealMatrix m) {
109         QRDecomposition qr = new QRDecomposition(m);
110         double norm = qr.getQ().multiply(qr.getR()).subtract(m).getNorm();
111         Assert.assertEquals(0, norm, normTolerance);
112     }

134     private void checkQOrthogonal(RealMatrix m) {
135         QRDecomposition qr = new QRDecomposition(m);
136         RealMatrix eye = MatrixUtils.createRealIdentityMatrix(m.getRowDimension());
137         double norm = qr.getQT().multiply(qr.getQ()).subtract(eye).getNorm();
138         Assert.assertEquals(0, norm, normTolerance);
139     }

142     @Test
143     public void testRUpperTriangular() {
144         RealMatrix matrix = MatrixUtils.createRealMatrix(testData3x3NonSingular);
145         checkUpperTriangular(new QRDecomposition(matrix).getR());
146 
147         matrix = MatrixUtils.createRealMatrix(testData3x3Singular);

149 
150         matrix = MatrixUtils.createRealMatrix(testData3x4);

152 
153         matrix = MatrixUtils.createRealMatrix(testData4x3);
154         checkUpperTriangular(new QRDecomposition(matrix).getR());
155 
156         Random r = new Random(643895747384642l);
157         int    p = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
158         int    q = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
159         matrix = createTestMatrix(r, p, q);
160         checkUpperTriangular(new QRDecomposition(matrix).getR());
161 
162         matrix = createTestMatrix(r, p, q);
163         checkUpperTriangular(new QRDecomposition(matrix).getR());
164 
165     }

167     private void checkUpperTriangular(RealMatrix m) {
168         m.walkInOptimizedOrder(new DefaultRealMatrixPreservingVisitor() {
169             @Override
170             public void visit(int row, int column, double value) {
171                 if (column < row) {
172                     Assert.assertEquals(0.0, value, entryTolerance);
173                 }
174             }
175         });
176     }

214     /** test matrices values */
215     @Test
216     public void testMatricesValues() {
217         QRDecomposition qr =
218             new QRDecomposition(MatrixUtils.createRealMatrix(testData3x3NonSingular));
219         RealMatrix qRef = MatrixUtils.createRealMatrix(new double[][] {
220                 { -12.0 / 14.0,   69.0 / 175.0,  -58.0 / 175.0 },
221                 {  -6.0 / 14.0, -158.0 / 175.0,    6.0 / 175.0 },
222                 {   4.0 / 14.0,  -30.0 / 175.0, -165.0 / 175.0 }
223         });
224         RealMatrix rRef = MatrixUtils.createRealMatrix(new double[][] {
225                 { -14.0,  -21.0, 14.0 },
226                 {   0.0, -175.0, 70.0 },
227                 {   0.0,    0.0, 35.0 }
228         });
229         RealMatrix hRef = MatrixUtils.createRealMatrix(new double[][] {
230                 { 26.0 / 14.0, 0.0, 0.0 },
231                 {  6.0 / 14.0, 648.0 / 325.0, 0.0 },
232                 { -4.0 / 14.0,  36.0 / 325.0, 2.0 }
233         });
234 
235         // check values against known references
236         RealMatrix q = qr.getQ();
237         Assert.assertEquals(0, q.subtract(qRef).getNorm(), 1.0e-13);
238         RealMatrix qT = qr.getQT();
239         Assert.assertEquals(0, qT.subtract(qRef.transpose()).getNorm(), 1.0e-13);
240         RealMatrix r = qr.getR();
241         Assert.assertEquals(0, r.subtract(rRef).getNorm(), 1.0e-13);
242         RealMatrix h = qr.getH();
243         Assert.assertEquals(0, h.subtract(hRef).getNorm(), 1.0e-13);
244 
245         // check the same cached instance is returned the second time
246         Assert.assertTrue(q == qr.getQ());
247         Assert.assertTrue(r == qr.getR());
248         Assert.assertTrue(h == qr.getH());
249 
250     }

259     @Test
260     public void testInvertTallSkinny() {
261         RealMatrix a     = MatrixUtils.createRealMatrix(testData4x3);
262         RealMatrix pinv  = new QRDecomposition(a).getSolver().getInverse();
263         Assert.assertEquals(0, pinv.multiply(a).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
264     }

266     @Test
267     public void testInvertShortWide() {
268         RealMatrix a = MatrixUtils.createRealMatrix(testData3x4);
269         RealMatrix pinv  = new QRDecomposition(a).getSolver().getInverse();
270         Assert.assertEquals(0, a.multiply(pinv).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
271         Assert.assertEquals(0, pinv.multiply(a).getSubMatrix(0, 2, 0, 2).subtract(MatrixUtils.createRealIdentityMatrix(3)).getNorm(), 1.0e-6);
272     }

274     private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
275         RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
276         m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor(){
277             @Override
278             public double visit(int row, int column, double value) {
279                 return 2.0 * r.nextDouble() - 1.0;
280             }
281         });
282         return m;
283     }
