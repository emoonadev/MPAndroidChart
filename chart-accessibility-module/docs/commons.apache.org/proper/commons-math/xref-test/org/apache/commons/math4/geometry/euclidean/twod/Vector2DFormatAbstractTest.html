165     @Test
166     public void testParseIgnoredWhitespace() {
167         Vector2D expected = new Vector2D(1, 1);
168         ParsePosition pos1 = new ParsePosition(0);
169         String source1 = "{1;1}";
170         Assert.assertEquals(expected, vector2DFormat.parse(source1, pos1));
171         Assert.assertEquals(source1.length(), pos1.getIndex());
172         ParsePosition pos2 = new ParsePosition(0);
173         String source2 = " { 1 ; 1 } ";
174         Assert.assertEquals(expected, vector2DFormat.parse(source2, pos2));
175         Assert.assertEquals(source2.length() - 1, pos2.getIndex());
176     }
