17  package org.apache.commons.math4.geometry.spherical.oned;
18  
19  import org.apache.commons.math4.exception.NumberIsTooLargeException;
20  import org.apache.commons.math4.geometry.partitioning.Region;
21  import org.apache.commons.math4.geometry.spherical.oned.Arc;
22  import org.apache.commons.math4.util.FastMath;
23  import org.apache.commons.math4.util.MathUtils;
24  import org.apache.commons.math4.util.Precision;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  public class ArcTest {
29  
30      @Test
31      public void testArc() {
32          Arc arc = new Arc(2.3, 5.7, 1.0e-10);
33          Assert.assertEquals(3.4, arc.getSize(), 1.0e-10);
34          Assert.assertEquals(4.0, arc.getBarycenter(), 1.0e-10);
35          Assert.assertEquals(Region.Location.BOUNDARY, arc.checkPoint(2.3));
36          Assert.assertEquals(Region.Location.BOUNDARY, arc.checkPoint(5.7));
37          Assert.assertEquals(Region.Location.OUTSIDE,  arc.checkPoint(1.2));
38          Assert.assertEquals(Region.Location.OUTSIDE,  arc.checkPoint(8.5));
39          Assert.assertEquals(Region.Location.INSIDE,   arc.checkPoint(8.7));
40          Assert.assertEquals(Region.Location.INSIDE,   arc.checkPoint(3.0));
41          Assert.assertEquals(2.3, arc.getInf(), 1.0e-10);
42          Assert.assertEquals(5.7, arc.getSup(), 1.0e-10);
43          Assert.assertEquals(4.0, arc.getBarycenter(), 1.0e-10);
44          Assert.assertEquals(3.4, arc.getSize(), 1.0e-10);
45      }
46  
47      @Test(expected=NumberIsTooLargeException.class)
48      public void testWrongInterval() {
49          new Arc(1.2, 0.0, 1.0e-10);
50      }
51  
52      @Test
53      public void testTolerance() {
54          Assert.assertEquals(Region.Location.OUTSIDE,  new Arc(2.3, 5.7, 1.0).checkPoint(1.2));
55          Assert.assertEquals(Region.Location.BOUNDARY, new Arc(2.3, 5.7, 1.2).checkPoint(1.2));
56          Assert.assertEquals(Region.Location.OUTSIDE,  new Arc(2.3, 5.7, 0.7).checkPoint(6.5));
57          Assert.assertEquals(Region.Location.BOUNDARY, new Arc(2.3, 5.7, 0.9).checkPoint(6.5));
58          Assert.assertEquals(Region.Location.INSIDE,   new Arc(2.3, 5.7, 0.6).checkPoint(3.0));
59          Assert.assertEquals(Region.Location.BOUNDARY, new Arc(2.3, 5.7, 0.8).checkPoint(3.0));
60      }
61  
62      @Test
63      public void testFullCircle() {
64          Arc arc = new Arc(9.0, 9.0, 1.0e-10);
65          // no boundaries on a full circle
66          Assert.assertEquals(Region.Location.INSIDE, arc.checkPoint(9.0));
67          Assert.assertEquals(.0, arc.getInf(), 1.0e-10);
68          Assert.assertEquals(MathUtils.TWO_PI, arc.getSup(), 1.0e-10);
69          Assert.assertEquals(2.0 * FastMath.PI, arc.getSize(), 1.0e-10);
70          for (double alpha = -20.0; alpha <= 20.0; alpha += 0.1) {
71              Assert.assertEquals(Region.Location.INSIDE, arc.checkPoint(alpha));
72          }
73      }
74  
75      @Test
76      public void testSmall() {
77          Arc arc = new Arc(1.0, FastMath.nextAfter(1.0, Double.POSITIVE_INFINITY), Precision.EPSILON);
78          Assert.assertEquals(2 * Precision.EPSILON, arc.getSize(), Precision.SAFE_MIN);
79          Assert.assertEquals(1.0, arc.getBarycenter(), Precision.EPSILON);
80      }
