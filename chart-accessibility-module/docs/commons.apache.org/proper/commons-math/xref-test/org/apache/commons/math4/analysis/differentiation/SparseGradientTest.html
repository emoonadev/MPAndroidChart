235     public void testPowDoubleDS() {
236         for (int maxOrder = 1; maxOrder < 5; ++maxOrder) {
237 
238             SparseGradient x = SparseGradient.createVariable(0, 0.1);
239             SparseGradient y = SparseGradient.createVariable(1, 0.2);
240             SparseGradient z = SparseGradient.createVariable(2, 0.3);
241             List<SparseGradient> list = Arrays.asList(x, y, z,
242                                                       x.add(y).add(z),
243                                                       x.multiply(y).multiply(z));
244 
245             for (SparseGradient sg : list) {
246                 // the special case a = 0 is included here
247                 for (double a : new double[] { 0.0, 0.1, 1.0, 2.0, 5.0 }) {
248                     SparseGradient reference = (a == 0) ?
249                                                x.getField().getZero() :
250                                                SparseGradient.createConstant(a).pow(sg);
251                     SparseGradient result = SparseGradient.pow(a, sg);
252                     Assert.assertEquals(reference, result);
253                 }
254 
255             }
256 
257             // negative base: -1^x can be evaluated for integers only, so value is sometimes OK, derivatives are always NaN
258             SparseGradient negEvenInteger = SparseGradient.pow(-2.0, SparseGradient.createVariable(0, 2.0));
259             Assert.assertEquals(4.0, negEvenInteger.getValue(), 1.0e-15);
260             Assert.assertTrue(Double.isNaN(negEvenInteger.getDerivative(0)));
261             SparseGradient negOddInteger = SparseGradient.pow(-2.0, SparseGradient.createVariable(0, 3.0));
262             Assert.assertEquals(-8.0, negOddInteger.getValue(), 1.0e-15);
263             Assert.assertTrue(Double.isNaN(negOddInteger.getDerivative(0)));
264             SparseGradient negNonInteger = SparseGradient.pow(-2.0, SparseGradient.createVariable(0, 2.001));
265             Assert.assertTrue(Double.isNaN(negNonInteger.getValue()));
266             Assert.assertTrue(Double.isNaN(negNonInteger.getDerivative(0)));
267 
268             SparseGradient zeroNeg = SparseGradient.pow(0.0, SparseGradient.createVariable(0, -1.0));
269             Assert.assertTrue(Double.isNaN(zeroNeg.getValue()));
270             Assert.assertTrue(Double.isNaN(zeroNeg.getDerivative(0)));
271             SparseGradient posNeg = SparseGradient.pow(2.0, SparseGradient.createVariable(0, -2.0));
272             Assert.assertEquals(1.0 / 4.0, posNeg.getValue(), 1.0e-15);
273             Assert.assertEquals(FastMath.log(2.0) / 4.0, posNeg.getDerivative(0), 1.0e-15);
274 
275             // very special case: a = 0 and power = 0
276             SparseGradient zeroZero = SparseGradient.pow(0.0, SparseGradient.createVariable(0, 0.0));
277 
278             // this should be OK for simple first derivative with one variable only ...
279             Assert.assertEquals(1.0, zeroZero.getValue(), 1.0e-15);
280             Assert.assertEquals(Double.NEGATIVE_INFINITY, zeroZero.getDerivative(0), 1.0e-15);
281             Assert.assertEquals(0.0, zeroZero.getDerivative(1), 1.0e-15);
282             Assert.assertEquals(0.0, zeroZero.getDerivative(2), 1.0e-15);
283 
284         }

289     public void testExpression() {
290         double epsilon = 2.5e-13;
291         for (double x = 0; x < 2; x += 0.2) {
292             SparseGradient sgX = SparseGradient.createVariable(0, x);
293             for (double y = 0; y < 2; y += 0.2) {
294                 SparseGradient sgY = SparseGradient.createVariable(1, y);
295                 for (double z = 0; z >- 2; z -= 0.2) {
296                     SparseGradient sgZ = SparseGradient.createVariable(2, z);
297 
298                     // f(x, y, z) = x + 5 x y - 2 z + (8 z x - y)^3
299                     SparseGradient sg =
300                             sgZ.linearCombination(1, sgX,
301                                                   5, sgX.multiply(sgY),
302                                                  -2, sgZ,
303                                                  1, sgZ.linearCombination(8, sgZ.multiply(sgX), -1, sgY).pow(3));
304                     double f = x + 5 * x * y - 2 * z + FastMath.pow(8 * z * x - y, 3);
305                     Assert.assertEquals(f, sg.getValue(), FastMath.abs(epsilon * f));
306 
307                     // df/dx = 1 + 5 y + 24 (8 z x - y)^2 z
308                     double dfdx = 1 + 5 * y + 24 * z * FastMath.pow(8 * z * x - y, 2);
309                     Assert.assertEquals(dfdx, sg.getDerivative(0), FastMath.abs(epsilon * dfdx));
310 
311                 }
312 
313             }
314         }
315     }

334     public void testTrigo() {
335         double epsilon = 2.0e-12;
336             for (double x = 0.1; x < 1.2; x += 0.1) {
337                 SparseGradient sgX = SparseGradient.createVariable(0, x);
338                 for (double y = 0.1; y < 1.2; y += 0.1) {
339                     SparseGradient sgY = SparseGradient.createVariable(1, y);
340                     for (double z = 0.1; z < 1.2; z += 0.1) {
341                         SparseGradient sgZ = SparseGradient.createVariable(2, z);
342                         SparseGradient f = sgX.divide(sgY.cos().add(sgZ.tan())).sin();
343                         double a = FastMath.cos(y) + FastMath.tan(z);
344                         double f0 = FastMath.sin(x / a);
345                         Assert.assertEquals(f0, f.getValue(), FastMath.abs(epsilon * f0));
346                         double dfdx = FastMath.cos(x / a) / a;
347                         Assert.assertEquals(dfdx, f.getDerivative(0), FastMath.abs(epsilon * dfdx));
348                         double dfdy =  x * FastMath.sin(y) * dfdx / a;
349                         Assert.assertEquals(dfdy, f.getDerivative(1), FastMath.abs(epsilon * dfdy));
350                         double cz = FastMath.cos(z);
351                         double cz2 = cz * cz;
352                         double dfdz = -x * dfdx / (a * cz2);
353                         Assert.assertEquals(dfdz, f.getDerivative(2), FastMath.abs(epsilon * dfdz));
354                     }
355                 }
356             }
