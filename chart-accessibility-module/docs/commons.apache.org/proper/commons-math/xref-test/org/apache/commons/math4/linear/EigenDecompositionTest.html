332         RealMatrix symmetric = MatrixUtils.createRealMatrix(new double[][] {

507     /**
508      * Checks that the eigen decomposition of a general (unsymmetric) matrix is valid by
509      * checking: A*V = V*D
510      */
511     private void checkUnsymmetricMatrix(final RealMatrix m) {
512         try {
513             EigenDecomposition ed = new EigenDecomposition(m);
514 
515             RealMatrix d = ed.getD();
516             RealMatrix v = ed.getV();
517             //RealMatrix vT = ed.getVT();
518 
519             RealMatrix x = m.multiply(v);
520             RealMatrix y = v.multiply(d);
521 
522             double diffNorm = x.subtract(y).getNorm();
523             Assert.assertTrue("The norm of (X-Y) is too large: " + diffNorm + ", matrix=" + m.toString(),
524                     x.subtract(y).getNorm() < 1000 * Precision.EPSILON * FastMath.max(x.getNorm(), y.getNorm()));
525 
526             RealMatrix invV = new LUDecomposition(v).getSolver().getInverse();
527             double norm = v.multiply(d).multiply(invV).subtract(m).getNorm();
528             Assert.assertEquals(0.0, norm, 1.0e-10);
529         } catch (Exception e) {
530             Assert.fail("Failed to create EigenDecomposition for matrix " + m.toString() + ", ex=" + e.toString());
531         }
532     }

657     /**
658      * Verifies that the given EigenDecomposition has eigenvalues equivalent to
659      * the targetValues, ignoring the order of the values and allowing
660      * values to differ by tolerance.
661      */
662     protected void checkEigenValues(double[] targetValues,
663             EigenDecomposition ed, double tolerance) {
664         double[] observed = ed.getRealEigenvalues();
665         for (int i = 0; i < observed.length; i++) {
666             Assert.assertTrue(isIncludedValue(observed[i], targetValues, tolerance));
667             Assert.assertTrue(isIncludedValue(targetValues[i], observed, tolerance));
668         }
669     }

672     /**
673      * Returns true iff there is an entry within tolerance of value in
674      * searchArray.
675      */
676     private boolean isIncludedValue(double value, double[] searchArray,
677             double tolerance) {
678        boolean found = false;
679        int i = 0;
680        while (!found && i < searchArray.length) {
681            if (FastMath.abs(value - searchArray[i]) < tolerance) {
682                found = true;
683            }
684            i++;
685        }
686        return found;
687     }

689     /**
690      * Returns true iff eigenVector is a scalar multiple of one of the columns
691      * of ed.getV().  Does not try linear combinations - i.e., should only be
692      * used to find vectors in one-dimensional eigenspaces.
693      */
694     protected void checkEigenVector(double[] eigenVector,
695             EigenDecomposition ed, double tolerance) {
696         Assert.assertTrue(isIncludedColumn(eigenVector, ed.getV(), tolerance));
697     }

699     /**
700      * Returns true iff there is a column that is a scalar multiple of column
701      * in searchMatrix (modulo tolerance)
702      */
703     private boolean isIncludedColumn(double[] column, RealMatrix searchMatrix,
704             double tolerance) {
705         boolean found = false;
706         int i = 0;
707         while (!found && i < searchMatrix.getColumnDimension()) {
708             double multiplier = 1.0;
709             boolean matching = true;
710             int j = 0;
711             while (matching && j < searchMatrix.getRowDimension()) {
712                 double colEntry = searchMatrix.getEntry(j, i);
713                 // Use the first entry where both are non-zero as scalar
714                 if (FastMath.abs(multiplier - 1.0) <= FastMath.ulp(1.0) && FastMath.abs(colEntry) > 1E-14
715                         && FastMath.abs(column[j]) > 1e-14) {
716                     multiplier = colEntry / column[j];
717                 }
718                 if (FastMath.abs(column[j] * multiplier - colEntry) > tolerance) {
719                     matching = false;
720                 }
721                 j++;
722             }
723             found = matching;
724             i++;
725         }
726         return found;
727     }

743     static RealMatrix createTestMatrix(final Random r, final double[] eigenValues) {
744         final int n = eigenValues.length;
745         final RealMatrix v = createOrthogonalMatrix(r, n);
746         final RealMatrix d = MatrixUtils.createRealDiagonalMatrix(eigenValues);
747         return v.multiply(d).multiply(v.transpose());

750     public static RealMatrix createOrthogonalMatrix(final Random r, final int size) {
751 
752         final double[][] data = new double[size][size];
753 
754         for (int i = 0; i < size; ++i) {
755             final double[] dataI = data[i];
756             double norm2 = 0;
757             do {
758 
759                 // generate randomly row I
760                 for (int j = 0; j < size; ++j) {
761                     dataI[j] = 2 * r.nextDouble() - 1;
762                 }
763 
764                 // project the row in the subspace orthogonal to previous rows
765                 for (int k = 0; k < i; ++k) {
766                     final double[] dataK = data[k];
767                     double dotProduct = 0;
768                     for (int j = 0; j < size; ++j) {
769                         dotProduct += dataI[j] * dataK[j];
770                     }
771                     for (int j = 0; j < size; ++j) {
772                         dataI[j] -= dotProduct * dataK[j];
773                     }
774                 }
775 
776                 // normalize the row
777                 norm2 = 0;
778                 for (final double dataIJ : dataI) {
779                     norm2 += dataIJ * dataIJ;
780                 }
781                 final double inv = 1.0 / FastMath.sqrt(norm2);
782                 for (int j = 0; j < size; ++j) {
783                     dataI[j] *= inv;
784                 }
785 
786             } while (norm2 * size < 0.01);
787         }
788 
789         return MatrixUtils.createRealMatrix(data);
790
