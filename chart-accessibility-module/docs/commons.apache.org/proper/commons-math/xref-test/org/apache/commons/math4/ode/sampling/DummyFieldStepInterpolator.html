18  package org.apache.commons.math4.ode.sampling;
19  
20  import org.apache.commons.math4.RealFieldElement;
21  import org.apache.commons.math4.ode.FieldEquationsMapper;
22  import org.apache.commons.math4.ode.FieldODEStateAndDerivative;
23  
24  public class DummyFieldStepInterpolator<T extends RealFieldElement<T>>
25      extends AbstractFieldStepInterpolator<T> {
26  
27      public DummyFieldStepInterpolator(final boolean forward,
28                                        final FieldODEStateAndDerivative<T> globalPreviousState,
29                                        final FieldODEStateAndDerivative<T> globalCurrentState,
30                                        final FieldODEStateAndDerivative<T> softPreviousState,
31                                        final FieldODEStateAndDerivative<T> softCurrentState,
32                                        final FieldEquationsMapper<T> mapper) {
33          super(forward, globalPreviousState, globalCurrentState, softPreviousState, softCurrentState, mapper);
34      }
35  
36      @Override
37      protected AbstractFieldStepInterpolator<T> create(final boolean newForward,
38                                                        final FieldODEStateAndDerivative<T> newGlobalPreviousState,
39                                                        final FieldODEStateAndDerivative<T> newGlobalCurrentState,
40                                                        final FieldODEStateAndDerivative<T> newSoftPreviousState,
41                                                        final FieldODEStateAndDerivative<T> newSoftCurrentState,
42                                                        final FieldEquationsMapper<T> newMapper) {
43          return new DummyFieldStepInterpolator<T>(newForward,
44                                                   newGlobalPreviousState, newGlobalCurrentState,
45                                                   newSoftPreviousState, newSoftCurrentState,
46                                                   newMapper);
47      }
48  
49      @Override
50      protected FieldODEStateAndDerivative<T> computeInterpolatedStateAndDerivatives(FieldEquationsMapper<T> equationsMapper,
51                                                                                     T time, T theta, T thetaH, T oneMinusThetaH) {
52          return getGlobalCurrentState();
53      }
54  
55  }
