18  package org.apache.commons.math4.random;
19  
20  import org.apache.commons.math4.random.GaussianRandomGenerator;
21  import org.apache.commons.rng.UniformRandomProvider;
22  import org.apache.commons.rng.RandomSource;
23  import org.apache.commons.math4.stat.StatUtils;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  
28  public class GaussianRandomGeneratorTest {
29  
30      @Test
31      public void testMeanAndStandardDeviation() {
32          GaussianRandomGenerator generator = new GaussianRandomGenerator(RandomSource.create(RandomSource.MT, 17399225432l));
33          double[] sample = new double[10000];
34          for (int i = 0; i < sample.length; ++i) {
35              sample[i] = generator.nextNormalizedDouble();
36          }
37          Assert.assertEquals(0.0, StatUtils.mean(sample), 0.012);
38          Assert.assertEquals(1.0, StatUtils.variance(sample), 0.01);
39      }
40  }
