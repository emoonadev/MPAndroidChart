32  public class MultivariateFunctionMappingAdapterTest {
33      @Test
34      public void testStartSimplexInsideRange() {
35          final BiQuadratic biQuadratic = new BiQuadratic(2.0, 2.5, 1.0, 3.0, 2.0, 3.0);
36          final MultivariateFunctionMappingAdapter wrapped
37              = new MultivariateFunctionMappingAdapter(biQuadratic,
38                                                       biQuadratic.getLower(),
39                                                       biQuadratic.getUpper());
40  
41          SimplexOptimizer optimizer = new SimplexOptimizer(1e-10, 1e-30);
42          final AbstractSimplex simplex = new NelderMeadSimplex(new double[][] {
43                  wrapped.boundedToUnbounded(new double[] { 1.5, 2.75 }),
44                  wrapped.boundedToUnbounded(new double[] { 1.5, 2.95 }),
45                  wrapped.boundedToUnbounded(new double[] { 1.7, 2.90 })
46              });
47  
48          final PointValuePair optimum
49              = optimizer.optimize(new MaxEval(300),
50                                   new ObjectiveFunction(wrapped),
51                                   simplex,
52                                   GoalType.MINIMIZE,
53                                   new InitialGuess(wrapped.boundedToUnbounded(new double[] { 1.5, 2.25 })));
54          final double[] bounded = wrapped.unboundedToBounded(optimum.getPoint());
55  
56          Assert.assertEquals(biQuadratic.getBoundedXOptimum(), bounded[0], 2e-7);
57          Assert.assertEquals(biQuadratic.getBoundedYOptimum(), bounded[1], 2e-7);
58      }

145     private static class BiQuadratic implements MultivariateFunction {
146 
147         private final double xOptimum;
148         private final double yOptimum;
149 
150         private final double xMin;
151         private final double xMax;
152         private final double yMin;
153         private final double yMax;
154 
155         public BiQuadratic(final double xOptimum, final double yOptimum,
156                            final double xMin, final double xMax,
157                            final double yMin, final double yMax) {
158             this.xOptimum = xOptimum;
159             this.yOptimum = yOptimum;
160             this.xMin     = xMin;
161             this.xMax     = xMax;
162             this.yMin     = yMin;
163             this.yMax     = yMax;
164         }
165 
166         public double value(double[] point) {
167             // the function should never be called with out of range points
168             Assert.assertTrue(point[0] >= xMin);
169             Assert.assertTrue(point[0] <= xMax);
170             Assert.assertTrue(point[1] >= yMin);
171             Assert.assertTrue(point[1] <= yMax);
172 
173             final double dx = point[0] - xOptimum;
174             final double dy = point[1] - yOptimum;
175             return dx * dx + dy * dy;
176 
177         }
178 
179         public double[] getLower() {
180             return new double[] { xMin, yMin };
181         }
182 
183         public double[] getUpper() {
184             return new double[] { xMax, yMax };
185         }
186 
187         public double getBoundedXOptimum() {
188             return (xOptimum < xMin) ? xMin : ((xOptimum > xMax) ? xMax : xOptimum);
189         }
190 
191         public double getBoundedYOptimum() {
192             return (yOptimum < yMin) ? yMin : ((yOptimum > yMax) ? yMax : yOptimum);
193         }
194     }
