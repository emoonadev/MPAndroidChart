32      /**
33       * tests the value of a constant polynomial.
34       *
35       * <p>value of this is 2.5 everywhere.</p>
36       */
37      @Test
38      public void testConstants() {
39          double[] c = { 2.5 };
40          PolynomialFunction f = new PolynomialFunction(c);
41  
42          // verify that we are equal to c[0] at several (nonsymmetric) places
43          Assert.assertEquals(f.value(0), c[0], tolerance);
44          Assert.assertEquals(f.value(-1), c[0], tolerance);
45          Assert.assertEquals(f.value(-123.5), c[0], tolerance);
46          Assert.assertEquals(f.value(3), c[0], tolerance);
47          Assert.assertEquals(f.value(456.89), c[0], tolerance);
48  
49          Assert.assertEquals(f.degree(), 0);
50          Assert.assertEquals(f.polynomialDerivative().value(0), 0, tolerance);
51  
52          Assert.assertEquals(f.polynomialDerivative().polynomialDerivative().value(0), 0, tolerance);
53      }

55      /**
56       * tests the value of a linear polynomial.
57       *
58       * <p>This will test the function f(x) = 3*x - 1.5</p>
59       * <p>This will have the values
60       *  <tt>f(0) = -1.5, f(-1) = -4.5, f(-2.5) = -9,
61       *      f(0.5) = 0, f(1.5) = 3</tt> and {@code f(3) = 7.5}
62       * </p>
63       */
64      @Test
65      public void testLinear() {
66          double[] c = { -1.5, 3 };
67          PolynomialFunction f = new PolynomialFunction(c);
68  
69          // verify that we are equal to c[0] when x=0
70          Assert.assertEquals(f.value(0), c[0], tolerance);
71  
72          // now check a few other places
73          Assert.assertEquals(-4.5, f.value(-1), tolerance);
74          Assert.assertEquals(-9, f.value(-2.5), tolerance);
75          Assert.assertEquals(0, f.value(0.5), tolerance);
76          Assert.assertEquals(3, f.value(1.5), tolerance);
77          Assert.assertEquals(7.5, f.value(3), tolerance);
78  
79          Assert.assertEquals(f.degree(), 1);
80  
81          Assert.assertEquals(f.polynomialDerivative().polynomialDerivative().value(0), 0, tolerance);
82      }

84      /**
85       * Tests a second order polynomial.
86       * <p> This will test the function f(x) = 2x^2 - 3x -2 = (2x+1)(x-2)</p>
87       */
88      @Test
89      public void testQuadratic() {
90          double[] c = { -2, -3, 2 };
91          PolynomialFunction f = new PolynomialFunction(c);
92  
93          // verify that we are equal to c[0] when x=0
94          Assert.assertEquals(f.value(0), c[0], tolerance);
95  
96          // now check a few other places
97          Assert.assertEquals(0, f.value(-0.5), tolerance);
98          Assert.assertEquals(0, f.value(2), tolerance);
99          Assert.assertEquals(-2, f.value(1.5), tolerance);
100         Assert.assertEquals(7, f.value(-1.5), tolerance);
101         Assert.assertEquals(265.5312, f.value(12.34), tolerance);
102     }

126     /**
127      * tests the firstDerivative function by comparison
128      *
129      * <p>This will test the functions
130      * {@code f(x) = x^3 - 2x^2 + 6x + 3, g(x) = 3x^2 - 4x + 6}
131      * and {@code h(x) = 6x - 4}
132      */
133     @Test
134     public void testfirstDerivativeComparison() {
135         double[] f_coeff = { 3, 6, -2, 1 };
136         double[] g_coeff = { 6, -4, 3 };
137         double[] h_coeff = { -4, 6 };
138 
139         PolynomialFunction f = new PolynomialFunction(f_coeff);
140         PolynomialFunction g = new PolynomialFunction(g_coeff);
141         PolynomialFunction h = new PolynomialFunction(h_coeff);
142 
143         // compare f' = g
144         Assert.assertEquals(f.polynomialDerivative().value(0), g.value(0), tolerance);
145         Assert.assertEquals(f.polynomialDerivative().value(1), g.value(1), tolerance);
146         Assert.assertEquals(f.polynomialDerivative().value(100), g.value(100), tolerance);
147         Assert.assertEquals(f.polynomialDerivative().value(4.1), g.value(4.1), tolerance);
148         Assert.assertEquals(f.polynomialDerivative().value(-3.25), g.value(-3.25), tolerance);
149 
150         // compare g' = h
151         Assert.assertEquals(g.polynomialDerivative().value(FastMath.PI), h.value(FastMath.PI), tolerance);
152         Assert.assertEquals(g.polynomialDerivative().value(FastMath.E),  h.value(FastMath.E),  tolerance);
153     }

155     @Test
156     public void testString() {
157         PolynomialFunction p = new PolynomialFunction(new double[] { -5, 3, 1 });
158         checkPolynomial(p, "-5 + 3 x + x^2");
159         checkPolynomial(new PolynomialFunction(new double[] { 0, -2, 3 }),
160                         "-2 x + 3 x^2");
161         checkPolynomial(new PolynomialFunction(new double[] { 1, -2, 3 }),
162                       "1 - 2 x + 3 x^2");
163         checkPolynomial(new PolynomialFunction(new double[] { 0,  2, 3 }),
164                        "2 x + 3 x^2");
165         checkPolynomial(new PolynomialFunction(new double[] { 1,  2, 3 }),
166                      "1 + 2 x + 3 x^2");
167         checkPolynomial(new PolynomialFunction(new double[] { 1,  0, 3 }),
168                      "1 + 3 x^2");
169         checkPolynomial(new PolynomialFunction(new double[] { 0 }),
170                      "0");
171     }

173     @Test
174     public void testAddition() {
175         PolynomialFunction p1 = new PolynomialFunction(new double[] { -2, 1 });
176         PolynomialFunction p2 = new PolynomialFunction(new double[] { 2, -1, 0 });
177         checkNullPolynomial(p1.add(p2));
178 
179         p2 = p1.add(p1);
180         checkPolynomial(p2, "-4 + 2 x");
181 
182         p1 = new PolynomialFunction(new double[] { 1, -4, 2 });
183         p2 = new PolynomialFunction(new double[] { -1, 3, -2 });
184         p1 = p1.add(p2);
185         Assert.assertEquals(1, p1.degree());
186         checkPolynomial(p1, "-x");
187     }

189     @Test
190     public void testSubtraction() {
191         PolynomialFunction p1 = new PolynomialFunction(new double[] { -2, 1 });
192         checkNullPolynomial(p1.subtract(p1));
193 
194         PolynomialFunction p2 = new PolynomialFunction(new double[] { -2, 6 });
195         p2 = p2.subtract(p1);
196         checkPolynomial(p2, "5 x");
197 
198         p1 = new PolynomialFunction(new double[] { 1, -4, 2 });
199         p2 = new PolynomialFunction(new double[] { -1, 3, 2 });
200         p1 = p1.subtract(p2);
201         Assert.assertEquals(1, p1.degree());
202         checkPolynomial(p1, "2 - 7 x");
203     }

205     @Test
206     public void testMultiplication() {
207         PolynomialFunction p1 = new PolynomialFunction(new double[] { -3, 2 });
208         PolynomialFunction p2 = new PolynomialFunction(new double[] { 3, 2, 1 });
209         checkPolynomial(p1.multiply(p2), "-9 + x^2 + 2 x^3");
210 
211         p1 = new PolynomialFunction(new double[] { 0, 1 });
212         p2 = p1;
213         for (int i = 2; i < 10; ++i) {
214             p2 = p2.multiply(p1);
215             checkPolynomial(p2, "x^" + i);
216         }
217     }

225     /**
226      * tests the firstDerivative function by comparison
227      *
228      * <p>This will test the functions
229      * {@code f(x) = x^3 - 2x^2 + 6x + 3, g(x) = 3x^2 - 4x + 6}
230      * and {@code h(x) = 6x - 4}
231      */
232     @Test
233     public void testMath341() {
234         double[] f_coeff = { 3, 6, -2, 1 };
235         double[] g_coeff = { 6, -4, 3 };
236         double[] h_coeff = { -4, 6 };
237 
238         PolynomialFunction f = new PolynomialFunction(f_coeff);
239         PolynomialFunction g = new PolynomialFunction(g_coeff);
240         PolynomialFunction h = new PolynomialFunction(h_coeff);
241 
242         // compare f' = g
243         Assert.assertEquals(f.polynomialDerivative().value(0), g.value(0), tolerance);
244         Assert.assertEquals(f.polynomialDerivative().value(1), g.value(1), tolerance);
245         Assert.assertEquals(f.polynomialDerivative().value(100), g.value(100), tolerance);
246         Assert.assertEquals(f.polynomialDerivative().value(4.1), g.value(4.1), tolerance);
247         Assert.assertEquals(f.polynomialDerivative().value(-3.25), g.value(-3.25), tolerance);
248 
249         // compare g' = h
250         Assert.assertEquals(g.polynomialDerivative().value(FastMath.PI), h.value(FastMath.PI), tolerance);
251         Assert.assertEquals(g.polynomialDerivative().value(FastMath.E),  h.value(FastMath.E),  tolerance);
252     }

254     public void checkPolynomial(PolynomialFunction p, String reference) {
255         Assert.assertEquals(reference, p.toString());
256     }

258     private void checkNullPolynomial(PolynomialFunction p) {
259         for (double coefficient : p.getCoefficients()) {
260             Assert.assertEquals(0, coefficient, 1e-15);
261         }
