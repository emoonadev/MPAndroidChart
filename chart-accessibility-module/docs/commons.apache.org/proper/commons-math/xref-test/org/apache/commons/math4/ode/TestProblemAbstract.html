18  package org.apache.commons.math4.ode;
19  
20  import org.apache.commons.math4.ode.FirstOrderDifferentialEquations;
21  import org.apache.commons.math4.ode.events.EventHandler;
22  
23  /**
24   * This class is used as the base class of the problems that are
25   * integrated during the junit tests for the ODE integrators.
26   */
27  public abstract class TestProblemAbstract
28    implements FirstOrderDifferentialEquations {
29  
30    /** Dimension of the problem. */
31    private int n;
32  
33    /** Number of functions calls. */
34    private int calls;
35  
36    /** Initial time */
37    private double t0;
38  
39    /** Initial state */
40    private double[] y0;
41  
42    /** Final time */
43    private double t1;
44  
45    /** Error scale */
46    private double[] errorScale;
47  
48    /**
49     * Simple constructor.
50     */
51    protected TestProblemAbstract() {
52      n          = 0;
53      calls      = 0;
54      t0         = 0;
55      y0         = null;
56      t1         = 0;
57      errorScale = null;
58    }
59  
60    /**
61     * Set the initial conditions
62     * @param t0 initial time
63     * @param y0 initial state vector
64     */
65    protected void setInitialConditions(double t0, double[] y0) {
66      calls     = 0;
67      n         = y0.length;
68      this.t0   = t0;
69      this.y0   = y0.clone();
70     }
71  
72    /**
73     * Set the final conditions.
74     * @param t1 final time
75     */
76    protected void setFinalConditions(double t1) {
77      this.t1 = t1;
78    }
79  
80    /**
81     * Set the error scale
82     * @param errorScale error scale
83     */
84    protected void setErrorScale(double[] errorScale) {
85      this.errorScale = errorScale.clone();
86    }
87  
88    public int getDimension() {
89      return n;
90    }
91  
92    /**
93     * Get the initial time.
94     * @return initial time
95     */
96    public double getInitialTime() {
97      return t0;
98    }
99  
100   /**
101    * Get the initial state vector.
102    * @return initial state vector
103    */
104   public double[] getInitialState() {
105     return y0;
106   }
107 
108   /**
109    * Get the final time.
110    * @return final time
111    */
112   public double getFinalTime() {
113     return t1;
114   }
115 
116   /**
117    * Get the error scale.
118    * @return error scale
119    */
120   public double[] getErrorScale() {
121     return errorScale;
122   }
123 
124   /**
125    * Get the events handlers.
126    * @return events handlers   */
127   public EventHandler[] getEventsHandlers() {
128     return new EventHandler[0];
129   }
130 
131   /**
132    * Get the theoretical events times.
133    * @return theoretical events times
134    */
135   public double[] getTheoreticalEventsTimes() {
136       return new double[0];
137   }
138 
139   /**
140    * Get the number of calls.
141    * @return nuber of calls
142    */
143   public int getCalls() {
144     return calls;
145   }
146 
147   public void computeDerivatives(double t, double[] y, double[] yDot) {
148     ++calls;
149     doComputeDerivatives(t, y, yDot);
150   }
151 
152   abstract public void doComputeDerivatives(double t, double[] y, double[] yDot);
153 
154   /**
155    * Compute the theoretical state at the specified time.
156    * @param t time at which the state is required
157    * @return state vector at time t
158    */
159   abstract public double[] computeTheoreticalState(double t);
160 
161 }
