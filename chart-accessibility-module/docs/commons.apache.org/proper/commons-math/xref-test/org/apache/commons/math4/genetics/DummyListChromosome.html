17  package org.apache.commons.math4.genetics;
18  
19  import java.util.Arrays;
20  import java.util.List;
21  
22  import org.apache.commons.math4.genetics.AbstractListChromosome;
23  import org.apache.commons.math4.genetics.InvalidRepresentationException;
24  
25  /**
26   * Implementation of ListChromosome for testing purposes
27   */
28  public class DummyListChromosome extends AbstractListChromosome<Integer> {
29      public DummyListChromosome(final Integer[] representation) {
30          super(representation);
31      }
32  
33      public DummyListChromosome(final List<Integer> representation) {
34          super(representation);
35      }
36  
37      public double fitness() {
38          // Not important.
39          return 0;
40      }
41  
42      @Override
43      protected void checkValidity(final List<Integer> chromosomeRepresentation) throws InvalidRepresentationException {
44          // Not important.
45      }
46  
47      @Override
48      public AbstractListChromosome<Integer> newFixedLengthChromosome(final List<Integer> chromosomeRepresentation) {
49          return new DummyListChromosome(chromosomeRepresentation);
50      }
51  
52      @Override
53      public int hashCode() {
54          final int prime = 31;
55          int result = 1;
56          result = prime * result + (getRepresentation() == null ? 0 : getRepresentation().hashCode());
57          return result;
58      }
59  
60      @Override
61      public boolean equals(final Object obj) {
62          if (this == obj) {
63              return true;
64          }
65          if (obj == null) {
66              return false;
67          }
68          if (!(obj instanceof DummyListChromosome)) {
69              return false;
70          }
71          final DummyListChromosome other = (DummyListChromosome) obj;
72          if (getRepresentation() == null) {
73              if (other.getRepresentation() != null) {
74                  return false;
75              }
76          }
77          final Integer[] rep = getRepresentation().toArray(new Integer[getRepresentation().size()]);
78          final Integer[] otherRep = other.getRepresentation().toArray(new Integer[other.getRepresentation().size()]);
79          return Arrays.equals(rep, otherRep);
80      }
81  }
