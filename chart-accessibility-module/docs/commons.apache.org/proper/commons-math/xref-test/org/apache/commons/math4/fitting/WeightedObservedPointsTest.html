31  public class WeightedObservedPointsTest {
32      @Test
33      public void testAdd1() {
34          final WeightedObservedPoints store = new WeightedObservedPoints();
35  
36          final double x = 1.2;
37          final double y = 34.56;
38          final double w = 0.789;
39  
40          store.add(w, x, y);
41  
42          Assert.assertTrue(lastElementIsSame(store, new WeightedObservedPoint(w, x, y)));
43      }
44

70      @Test
71      public void testClear() {
72          final WeightedObservedPoints store = new WeightedObservedPoints();
73  
74          store.add(new WeightedObservedPoint(1, 2, 3));
75          store.add(new WeightedObservedPoint(2, -1, -2));
76          Assert.assertTrue(store.toList().size() == 2);
77  
78          store.clear();
79          Assert.assertTrue(store.toList().size() == 0);
80      }
81  
82      // Ensure that an instance returned by "toList()" is independent from
83      // the original container.
84      @Test
85      public void testToListCopy() {
86          final WeightedObservedPoints store = new WeightedObservedPoints();
87  
88          store.add(new WeightedObservedPoint(1, 2, 3));
89          store.add(new WeightedObservedPoint(2, -3, -4));
90  
91          final List<WeightedObservedPoint> list = store.toList();
92          Assert.assertTrue(list.size() == 2);
93  
94          // Adding an element to "list" has no impact on "store".
95          list.add(new WeightedObservedPoint(1.2, 3.4, 5.6));
96          Assert.assertFalse(list.size() == store.toList().size());
97  
98          // Clearing "store" has no impact on "list".
99          store.clear();
100         Assert.assertFalse(list.size() == 0);
101     }
102 
103     /**
104      * Checks that the contents of the last element is equal to the
105      * contents of {@code p}.
106      *
107      * @param store Container.
108      * @param point Observation.
109      * @return {@code true} if both elements have the same contents.
110      */
111     private boolean lastElementIsSame(WeightedObservedPoints store,
112                                       WeightedObservedPoint point) {
113         final List<WeightedObservedPoint> list = store.toList();
114         final WeightedObservedPoint lastPoint = list.get(list.size() - 1);
115 
116         if (!Precision.equals(lastPoint.getX(), point.getX())) {
117             return false;
118         }
119         if (!Precision.equals(lastPoint.getY(), point.getY())) {
120             return false;
121         }
122         if (!Precision.equals(lastPoint.getWeight(), point.getWeight())) {
123             return false;
124         }
125 
126         return true;
127     }
