36  public class StepNormalizerTest {
37  
38    public StepNormalizerTest() {
39      pb    = null;
40      integ = null;
41    }
42  
43    @Test
44    public void testBoundaries()
45        throws DimensionMismatchException, NumberIsTooSmallException,
46               MaxCountExceededException, NoBracketingException {
47      double range = pb.getFinalTime() - pb.getInitialTime();
48      setLastSeen(false);
49      integ.addStepHandler(new StepNormalizer(range / 10.0,
50                                         new FixedStepHandler() {
51                                           private boolean firstCall = true;
52                                           public void init(double t0, double[] y0, double t) {
53                                           }
54                                           public void handleStep(double t,
55                                                                  double[] y,
56                                                                  double[] yDot,
57                                                                  boolean isLast) {
58                                             if (firstCall) {
59                                               checkValue(t, pb.getInitialTime());
60                                               firstCall = false;
61                                             }
62                                             if (isLast) {
63                                               setLastSeen(true);
64                                               checkValue(t, pb.getFinalTime());
65                                             }
66                                           }
67                                         }));
68      integ.integrate(pb,
69                      pb.getInitialTime(), pb.getInitialState(),
70                      pb.getFinalTime(), new double[pb.getDimension()]);
71      Assert.assertTrue(lastSeen);
72    }
73  
74    @Test
75    public void testBeforeEnd()
76        throws DimensionMismatchException, NumberIsTooSmallException,
77               MaxCountExceededException, NoBracketingException {
78      final double range = pb.getFinalTime() - pb.getInitialTime();
79      setLastSeen(false);
80      integ.addStepHandler(new StepNormalizer(range / 10.5,
81                                         new FixedStepHandler() {
82                                           public void init(double t0, double[] y0, double t) {
83                                           }
84                                           public void handleStep(double t,
85                                                                  double[] y,
86                                                                  double[] yDot,
87                                                                  boolean isLast) {
88                                             if (isLast) {
89                                               setLastSeen(true);
90                                               checkValue(t,
91                                                          pb.getFinalTime() - range / 21.0);
92                                             }
93                                           }
94                                         }));
95      integ.integrate(pb,
96                      pb.getInitialTime(), pb.getInitialState(),
97                      pb.getFinalTime(), new double[pb.getDimension()]);
98      Assert.assertTrue(lastSeen);
99    }
100 
101   public void checkValue(double value, double reference) {
102     Assert.assertTrue(FastMath.abs(value - reference) < 1.0e-10);
103   }
104 
105   public void setLastSeen(boolean lastSeen) {
106     this.lastSeen = lastSeen;
107   }
108 
109   @Before
110   public void setUp() {
111     pb = new TestProblem3(0.9);
112     double minStep = 0;
113     double maxStep = pb.getFinalTime() - pb.getInitialTime();
114     integ = new DormandPrince54Integrator(minStep, maxStep, 10.e-8, 1.0e-8);
115     lastSeen = false;
116   }
117 
118   @After
119   public void tearDown() {
120     pb    = null;
121     integ = null;
122   }
123 
124   TestProblem3 pb;
125   FirstOrderIntegrator integ;
126   boolean lastSeen;
127 
128 }
