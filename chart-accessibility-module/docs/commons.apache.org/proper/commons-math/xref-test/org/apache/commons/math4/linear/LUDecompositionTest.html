60      /** test dimensions */
61      @Test
62      public void testDimensions() {
63          RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
64          LUDecomposition LU = new LUDecomposition(matrix);
65          Assert.assertEquals(testData.length, LU.getL().getRowDimension());
66          Assert.assertEquals(testData.length, LU.getL().getColumnDimension());
67          Assert.assertEquals(testData.length, LU.getU().getRowDimension());
68          Assert.assertEquals(testData.length, LU.getU().getColumnDimension());
69          Assert.assertEquals(testData.length, LU.getP().getRowDimension());
70          Assert.assertEquals(testData.length, LU.getP().getColumnDimension());

85      /** test PA = LU */
86      @Test
87      public void testPAEqualLU() {
88          RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
89          LUDecomposition lu = new LUDecomposition(matrix);
90          RealMatrix l = lu.getL();
91          RealMatrix u = lu.getU();
92          RealMatrix p = lu.getP();
93          double norm = l.multiply(u).subtract(p.multiply(matrix)).getNorm();
94          Assert.assertEquals(0, norm, normTolerance);
95  
96          matrix = MatrixUtils.createRealMatrix(testDataMinus);
97          lu = new LUDecomposition(matrix);
98          l = lu.getL();
99          u = lu.getU();
100         p = lu.getP();
101         norm = l.multiply(u).subtract(p.multiply(matrix)).getNorm();
102         Assert.assertEquals(0, norm, normTolerance);
103 
104         matrix = MatrixUtils.createRealIdentityMatrix(17);
105         lu = new LUDecomposition(matrix);
106         l = lu.getL();
107         u = lu.getU();
108         p = lu.getP();
109         norm = l.multiply(u).subtract(p.multiply(matrix)).getNorm();
110         Assert.assertEquals(0, norm, normTolerance);
111 
112         matrix = MatrixUtils.createRealMatrix(singular);
113         lu = new LUDecomposition(matrix);
114         Assert.assertFalse(lu.getSolver().isNonSingular());
115         Assert.assertNull(lu.getL());
116         Assert.assertNull(lu.getU());
117         Assert.assertNull(lu.getP());
118 
119         matrix = MatrixUtils.createRealMatrix(bigSingular);
120         lu = new LUDecomposition(matrix);
121         Assert.assertFalse(lu.getSolver().isNonSingular());
122         Assert.assertNull(lu.getL());
123         Assert.assertNull(lu.getU());
124         Assert.assertNull(lu.getP());
125 
126     }

128     /** test that L is lower triangular with unit diagonal */
129     @Test
130     public void testLLowerTriangular() {
131         RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
132         RealMatrix l = new LUDecomposition(matrix).getL();
133         for (int i = 0; i < l.getRowDimension(); i++) {
134             Assert.assertEquals(l.getEntry(i, i), 1, entryTolerance);
135             for (int j = i + 1; j < l.getColumnDimension(); j++) {
136                 Assert.assertEquals(l.getEntry(i, j), 0, entryTolerance);
137             }
138         }
139     }

153     /** test that P is a permutation matrix */
154     @Test
155     public void testPPermutation() {
156         RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
157         RealMatrix p   = new LUDecomposition(matrix).getP();
158 
159         RealMatrix ppT = p.multiply(p.transpose());
160         RealMatrix id  = MatrixUtils.createRealIdentityMatrix(p.getRowDimension());
161         Assert.assertEquals(0, ppT.subtract(id).getNorm(), normTolerance);
162 
163         for (int i = 0; i < p.getRowDimension(); i++) {
164             int zeroCount  = 0;
165             int oneCount   = 0;
166             int otherCount = 0;
167             for (int j = 0; j < p.getColumnDimension(); j++) {
168                 final double e = p.getEntry(i, j);
169                 if (e == 0) {
170                     ++zeroCount;
171                 } else if (e == 1) {
172                     ++oneCount;
173                 } else {
174                     ++otherCount;
175                 }
176             }
177             Assert.assertEquals(p.getColumnDimension() - 1, zeroCount);
178             Assert.assertEquals(1, oneCount);
179             Assert.assertEquals(0, otherCount);
180         }
181 
182         for (int j = 0; j < p.getColumnDimension(); j++) {
183             int zeroCount  = 0;
184             int oneCount   = 0;
185             int otherCount = 0;
186             for (int i = 0; i < p.getRowDimension(); i++) {
187                 final double e = p.getEntry(i, j);
188                 if (e == 0) {
189                     ++zeroCount;
190                 } else if (e == 1) {
191                     ++oneCount;
192                 } else {
193                     ++otherCount;
194                 }
195             }
196             Assert.assertEquals(p.getRowDimension() - 1, zeroCount);
197             Assert.assertEquals(1, oneCount);
198             Assert.assertEquals(0, otherCount);
199         }
200

215     /** test matrices values */
216     @Test
217     public void testMatricesValues1() {
218        LUDecomposition lu =
219             new LUDecomposition(MatrixUtils.createRealMatrix(testData));
220         RealMatrix lRef = MatrixUtils.createRealMatrix(new double[][] {
221                 { 1.0, 0.0, 0.0 },
222                 { 0.5, 1.0, 0.0 },
223                 { 0.5, 0.2, 1.0 }
224         });
225         RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
226                 { 2.0,  5.0, 3.0 },
227                 { 0.0, -2.5, 6.5 },
228                 { 0.0,  0.0, 0.2 }
229         });
230         RealMatrix pRef = MatrixUtils.createRealMatrix(new double[][] {
231                 { 0.0, 1.0, 0.0 },
232                 { 0.0, 0.0, 1.0 },
233                 { 1.0, 0.0, 0.0 }
234         });
235         int[] pivotRef = { 1, 2, 0 };
236 
237         // check values against known references
238         RealMatrix l = lu.getL();
239         Assert.assertEquals(0, l.subtract(lRef).getNorm(), 1.0e-13);
240         RealMatrix u = lu.getU();
241         Assert.assertEquals(0, u.subtract(uRef).getNorm(), 1.0e-13);
242         RealMatrix p = lu.getP();
243         Assert.assertEquals(0, p.subtract(pRef).getNorm(), 1.0e-13);
244         int[] pivot = lu.getPivot();
245         for (int i = 0; i < pivotRef.length; ++i) {
246             Assert.assertEquals(pivotRef[i], pivot[i]);
247         }
248 
249         // check the same cached instance is returned the second time
250         Assert.assertTrue(l == lu.getL());
251         Assert.assertTrue(u == lu.getU());
252         Assert.assertTrue(p == lu.getP());
253 
254     }
