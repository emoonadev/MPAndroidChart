38      public static void checkDerivativesConsistency(final FirstOrderIntegrator integrator,
39                                                     final TestProblemAbstract problem,
40                                                     final double finiteDifferencesRatio,
41                                                     final double threshold)
42          throws DimensionMismatchException, NumberIsTooSmallException,
43                 MaxCountExceededException, NoBracketingException {
44          integrator.addStepHandler(new StepHandler() {
45  
46              public void handleStep(StepInterpolator interpolator, boolean isLast)
47                  throws MaxCountExceededException {
48  
49                  final double dt = interpolator.getCurrentTime() - interpolator.getPreviousTime();
50                  final double h  = finiteDifferencesRatio * dt;
51                  final double t  = interpolator.getCurrentTime() - 0.3 * dt;
52  
53                  if (FastMath.abs(h) < 10 * FastMath.ulp(t)) {
54                      return;
55                  }
56  
57                  interpolator.setInterpolatedTime(t - 4 * h);
58                  final double[] yM4h = interpolator.getInterpolatedState().clone();
59                  interpolator.setInterpolatedTime(t - 3 * h);
60                  final double[] yM3h = interpolator.getInterpolatedState().clone();
61                  interpolator.setInterpolatedTime(t - 2 * h);
62                  final double[] yM2h = interpolator.getInterpolatedState().clone();
63                  interpolator.setInterpolatedTime(t - h);
64                  final double[] yM1h = interpolator.getInterpolatedState().clone();
65                  interpolator.setInterpolatedTime(t + h);
66                  final double[] yP1h = interpolator.getInterpolatedState().clone();
67                  interpolator.setInterpolatedTime(t + 2 * h);
68                  final double[] yP2h = interpolator.getInterpolatedState().clone();
69                  interpolator.setInterpolatedTime(t + 3 * h);
70                  final double[] yP3h = interpolator.getInterpolatedState().clone();
71                  interpolator.setInterpolatedTime(t + 4 * h);
72                  final double[] yP4h = interpolator.getInterpolatedState().clone();
73  
74                  interpolator.setInterpolatedTime(t);
75                  final double[] yDot = interpolator.getInterpolatedDerivatives();
76  
77                  for (int i = 0; i < yDot.length; ++i) {
78                      final double approYDot = ( -3 * (yP4h[i] - yM4h[i]) +
79                                                 32 * (yP3h[i] - yM3h[i]) +
80                                               -168 * (yP2h[i] - yM2h[i]) +
81                                                672 * (yP1h[i] - yM1h[i])) / (840 * h);
82                      Assert.assertEquals("" + (approYDot - yDot[i]), approYDot, yDot[i], threshold);
83                  }
84  
85              }
86  
87              public void init(double t0, double[] y0, double t) {
88              }
89  
90          });
91  
92          integrator.integrate(problem,
93                               problem.getInitialTime(), problem.getInitialState(),
94                               problem.getFinalTime(), new double[problem.getDimension()]);
95  
96      }

98      public static <T extends RealFieldElement<T>> void checkDerivativesConsistency(final FirstOrderFieldIntegrator<T> integrator,
99                                                                                     final TestFieldProblemAbstract<T> problem,
100                                                                                    final double threshold) {
101         integrator.addStepHandler(new FieldStepHandler<T>() {
102 
103             public void handleStep(FieldStepInterpolator<T> interpolator, boolean isLast)
104                 throws MaxCountExceededException {
105 
106                 final T h = interpolator.getCurrentState().getTime().subtract(interpolator.getPreviousState().getTime()).multiply(0.001);
107                 final T t = interpolator.getCurrentState().getTime().subtract(h.multiply(300));
108 
109                 if (h.abs().subtract(FastMath.ulp(t.getReal()) * 10).getReal() < 0) {
110                     return;
111                 }
112 
113                 final T[] yM4h = interpolator.getInterpolatedState(t.add(h.multiply(-4))).getState();
114                 final T[] yM3h = interpolator.getInterpolatedState(t.add(h.multiply(-3))).getState();
115                 final T[] yM2h = interpolator.getInterpolatedState(t.add(h.multiply(-2))).getState();
116                 final T[] yM1h = interpolator.getInterpolatedState(t.add(h.multiply(-1))).getState();
117                 final T[] yP1h = interpolator.getInterpolatedState(t.add(h.multiply( 1))).getState();
118                 final T[] yP2h = interpolator.getInterpolatedState(t.add(h.multiply( 2))).getState();
119                 final T[] yP3h = interpolator.getInterpolatedState(t.add(h.multiply( 3))).getState();
120                 final T[] yP4h = interpolator.getInterpolatedState(t.add(h.multiply( 4))).getState();
121 
122                 final T[] yDot = interpolator.getInterpolatedState(t).getDerivative();
123 
124                 for (int i = 0; i < yDot.length; ++i) {
125                     final T approYDot =     yP4h[i].subtract(yM4h[i]).multiply(  -3).
126                                         add(yP3h[i].subtract(yM3h[i]).multiply(  32)).
127                                         add(yP2h[i].subtract(yM2h[i]).multiply(-168)).
128                                         add(yP1h[i].subtract(yM1h[i]).multiply( 672)).
129                                         divide(h.multiply(840));
130                     Assert.assertEquals(approYDot.getReal(), yDot[i].getReal(), threshold);
131                 }
132 
133             }
134 
135             public void init(FieldODEStateAndDerivative<T> state0, T t) {
136             }
137 
138         });
139 
140         integrator.integrate(new FieldExpandableODE<T>(problem), problem.getInitialState(), problem.getFinalTime());
141 
142     }
