162         referenceRules.put("d(f*g)/dz",      referenceRules.get("d(f*g)/dx").replaceAll("x", "z"));
163         referenceRules.put("d(f*g)/dt",      referenceRules.get("d(f*g)/dx").replaceAll("x", "t"));
164         referenceRules.put("d2(f*g)/dx2",    "f * d2g/dx2 + 2 * df/dx * dg/dx + d2f/dx2 * g");
165         referenceRules.put("d2(f*g)/dy2",    referenceRules.get("d2(f*g)/dx2").replaceAll("x", "y"));
166         referenceRules.put("d2(f*g)/dz2",    referenceRules.get("d2(f*g)/dx2").replaceAll("x", "z"));

174         referenceRules.put("d3(f*g)/dx3",    "f * d3g/dx3 +" +
175                                              " 3 * df/dx * d2g/dx2 +" +
176                                              " 3 * d2f/dx2 * dg/dx +" +
177                                              " d3f/dx3 * g");
178         referenceRules.put("d3(f*g)/dy3",   referenceRules.get("d3(f*g)/dx3").replaceAll("x", "y"));
179         referenceRules.put("d3(f*g)/dz3",   referenceRules.get("d3(f*g)/dx3").replaceAll("x", "z"));

194         referenceRules.put("d3(f*g)/dy2dz",   referenceRules.get("d3(f*g)/dx2dz").replaceAll("x", "y"));
195         referenceRules.put("d3(f*g)/dxdz2",   referenceRules.get("d3(f*g)/dxdy2").replaceAll("y", "z"));
196         referenceRules.put("d3(f*g)/dydz2",   referenceRules.get("d3(f*g)/dxdz2").replaceAll("x", "y"));
197         referenceRules.put("d3(f*g)/dx2dt",   referenceRules.get("d3(f*g)/dx2dz").replaceAll("z", "t"));
198         referenceRules.put("d3(f*g)/dy2dt",   referenceRules.get("d3(f*g)/dx2dt").replaceAll("x", "y"));
199         referenceRules.put("d3(f*g)/dz2dt",   referenceRules.get("d3(f*g)/dx2dt").replaceAll("x", "z"));
200         referenceRules.put("d3(f*g)/dxdt2",   referenceRules.get("d3(f*g)/dxdy2").replaceAll("y", "t"));
201         referenceRules.put("d3(f*g)/dydt2",   referenceRules.get("d3(f*g)/dxdt2").replaceAll("x", "y"));
202         referenceRules.put("d3(f*g)/dzdt2",   referenceRules.get("d3(f*g)/dxdt2").replaceAll("x", "z"));
203         referenceRules.put("d3(f*g)/dxdydz", "f * d3g/dxdydz +" +
204                                              " df/dz * d2g/dxdy +" +
205                                              " df/dy * d2g/dxdz +" +
206                                              " d2f/dydz * dg/dx +" +
207                                              " df/dx * d2g/dydz +" +
208                                              " d2f/dxdz * dg/dy +" +
209                                              " d2f/dxdy * dg/dz +" +
210                                              " d3f/dxdydz * g");
211         referenceRules.put("d3(f*g)/dxdydt", referenceRules.get("d3(f*g)/dxdydz").replaceAll("z", "t"));
212         referenceRules.put("d3(f*g)/dxdzdt", referenceRules.get("d3(f*g)/dxdydt").replaceAll("y", "z"));
213         referenceRules.put("d3(f*g)/dydzdt", referenceRules.get("d3(f*g)/dxdzdt").replaceAll("x", "y"));
214 
215         Field multFieldArrayField = DSCompiler.class.getDeclaredField("multIndirection");
216         multFieldArrayField.setAccessible(true);
217         for (int i = 0; i < 5; ++i) {
218             for (int j = 0; j < 4; ++j) {
219                 DSCompiler compiler = DSCompiler.getCompiler(i, j);
220                 int[][][] multIndirection = (int[][][]) multFieldArrayField.get(compiler);
221                 for (int k = 0; k < multIndirection.length; ++k) {
222                     String product = ordersToString(compiler.getPartialDerivativeOrders(k),
223                                                     "(f*g)", "x", "y", "z", "t");
224                     StringBuilder rule = new StringBuilder();
225                     for (int[] term : multIndirection[k]) {
226                         if (rule.length() > 0) {
227                             rule.append(" + ");
228                         }
229                         if (term[0] > 1) {
230                             rule.append(term[0]).append(" * ");
231                         }
232                         rule.append(ordersToString(compiler.getPartialDerivativeOrders(term[1]),
233                                                    "f", "x", "y", "z", "t"));
234                         rule.append(" * ");
235                         rule.append(ordersToString(compiler.getPartialDerivativeOrders(term[2]),
236                                                    "g", "x", "y", "z", "t"));
237                     }
238                     Assert.assertEquals(product, referenceRules.get(product), rule.toString());
239                 }
240             }
241         }
242     }

439     private String orderToString(int order, String functionName, String parameterName) {
440         if (order == 0) {
441             return functionName;
442         } else if (order == 1) {
443             return "d" + functionName + "/d" + parameterName;
444         } else {
445             return "d" + order + functionName + "/d" + parameterName + order;
446         }
447     }
448 
449     private String ordersToString(int[] orders, String functionName, String ... parametersNames) {
450 
451         int sumOrders = 0;
452         for (int order : orders) {
453             sumOrders += order;
454         }
455 
456         if (sumOrders == 0) {
457             return functionName;
458         }
459 
460         StringBuilder builder = new StringBuilder();
461         builder.append('d');
462         if (sumOrders > 1) {
463             builder.append(sumOrders);
464         }
465         builder.append(functionName).append('/');
466         for (int i = 0; i < orders.length; ++i) {
467             if (orders[i] > 0) {
468                 builder.append('d').append(parametersNames[i]);
469                 if (orders[i] > 1) {
470                     builder.append(orders[i]);
471                 }
472             }
473         }
474         return builder.toString();
