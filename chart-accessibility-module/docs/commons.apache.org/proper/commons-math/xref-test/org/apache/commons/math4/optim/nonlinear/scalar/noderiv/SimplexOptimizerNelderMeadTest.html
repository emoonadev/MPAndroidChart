55      @Test
56      public void testMinimize1() {
57          SimplexOptimizer optimizer = new SimplexOptimizer(1e-10, 1e-30);
58          final FourExtrema fourExtrema = new FourExtrema();
59  
60          final PointValuePair optimum
61              = optimizer.optimize(new MaxEval(100),
62                                   new ObjectiveFunction(fourExtrema),
63                                   GoalType.MINIMIZE,
64                                   new InitialGuess(new double[] { -3, 0 }),
65                                   new NelderMeadSimplex(new double[] { 0.2, 0.2 }));

70          Assert.assertTrue(optimizer.getEvaluations() < 90);
71  
72          // Check that the number of iterations is updated (MATH-949).

176     @Test
177     public void testLeastSquares1() {
178         final RealMatrix factors
179             = new Array2DRowRealMatrix(new double[][] {
180                     { 1, 0 },
181                     { 0, 1 }
182                 }, false);
183         LeastSquaresConverter ls = new LeastSquaresConverter(new MultivariateVectorFunction() {
184                 public double[] value(double[] variables) {
185                     return factors.operate(variables);
186                 }
187             }, new double[] { 2.0, -3.0 });
188         SimplexOptimizer optimizer = new SimplexOptimizer(-1, 1e-6);
189         PointValuePair optimum =
190             optimizer.optimize(new MaxEval(200),
191                                new ObjectiveFunction(ls),
192                                GoalType.MINIMIZE,
193                                new InitialGuess(new double[] { 10, 10 }),
194                                new NelderMeadSimplex(2));
195         Assert.assertEquals( 2, optimum.getPointRef()[0], 3e-5);
196         Assert.assertEquals(-3, optimum.getPointRef()[1], 4e-4);
197         Assert.assertTrue(optimizer.getEvaluations() > 60);
198         Assert.assertTrue(optimizer.getEvaluations() < 80);
199         Assert.assertTrue(optimum.getValue() < 1.0e-6);
200     }

267     private static class FourExtrema implements MultivariateFunction {
268         // The following function has 4 local extrema.
269         final double xM = -3.841947088256863675365;
270         final double yM = -1.391745200270734924416;
271         final double xP =  0.2286682237349059125691;
272         final double yP = -yM;
273         final double valueXmYm = 0.2373295333134216789769; // Local maximum.
274         final double valueXmYp = -valueXmYm; // Local minimum.
275         final double valueXpYm = -0.7290400707055187115322; // Global minimum.
276         final double valueXpYp = -valueXpYm; // Global maximum.
277 
278         public double value(double[] variables) {
279             final double x = variables[0];
280             final double y = variables[1];
281             return (x == 0 || y == 0) ? 0 :
282                 FastMath.atan(x) * FastMath.atan(x + 2) * FastMath.atan(y) * FastMath.atan(y) / (x * y);
283         }
284     }

286     private static class Rosenbrock implements MultivariateFunction {
287         private int count;
288 
289         public Rosenbrock() {
290             count = 0;
291         }
292 
293         public double value(double[] x) {
294             ++count;
295             double a = x[1] - x[0] * x[0];
296             double b = 1.0 - x[0];
297             return 100 * a * a + b * b;
298         }
299 
300         public int getCount() {
301             return count;
302         }
303     }

305     private static class Powell implements MultivariateFunction {
306         private int count;
307 
308         public Powell() {
309             count = 0;
310         }
311 
312         public double value(double[] x) {
313             ++count;
314             double a = x[0] + 10 * x[1];
315             double b = x[2] - x[3];
316             double c = x[1] - 2 * x[2];
317             double d = x[0] - x[3];
318             return a * a + 5 * b * b + c * c * c * c + 10 * d * d * d * d;
319         }
320 
321         public int getCount() {
322             return count;
323         }
324     }
