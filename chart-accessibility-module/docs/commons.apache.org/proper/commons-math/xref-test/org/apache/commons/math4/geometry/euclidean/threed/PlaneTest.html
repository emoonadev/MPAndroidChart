31      public void testContains() throws MathArithmeticException {
32          Plane p = new Plane(new Vector3D(0, 0, 1), new Vector3D(0, 0, 1), 1.0e-10);
33          Assert.assertTrue(p.contains(new Vector3D(0, 0, 1)));
34          Assert.assertTrue(p.contains(new Vector3D(17, -32, 1)));
35          Assert.assertTrue(! p.contains(new Vector3D(17, -32, 1.001)));

38      @Test
39      public void testOffset() throws MathArithmeticException {
40          Vector3D p1 = new Vector3D(1, 1, 1);
41          Plane p = new Plane(p1, new Vector3D(0.2, 0, 0), 1.0e-10);
42          Assert.assertEquals(-5.0, p.getOffset(new Vector3D(-4, 0, 0)), 1.0e-10);
43          Assert.assertEquals(+5.0, p.getOffset(new Vector3D(6, 10, -12)), 1.0e-10);
44          Assert.assertEquals(0.3,
45                              p.getOffset(new Vector3D(1.0, p1, 0.3, p.getNormal())),
46                              1.0e-10);
47          Assert.assertEquals(-0.3,
48                              p.getOffset(new Vector3D(1.0, p1, -0.3, p.getNormal())),
49                              1.0e-10);
50      }

69      @Test
70      public void testRotate() throws MathArithmeticException, MathIllegalArgumentException {
71          Vector3D p1 = new Vector3D(1.2, 3.4, -5.8);
72          Vector3D p2 = new Vector3D(3.4, -5.8, 1.2);
73          Vector3D p3 = new Vector3D(-2.0, 4.3, 0.7);
74          Plane    p  = new Plane(p1, p2, p3, 1.0e-10);
75          Vector3D oldNormal = p.getNormal();
76  
77          p = p.rotate(p2, new Rotation(p2.subtract(p1), 1.7, RotationConvention.VECTOR_OPERATOR));
78          Assert.assertTrue(p.contains(p1));
79          Assert.assertTrue(p.contains(p2));
80          Assert.assertTrue(! p.contains(p3));
81  
82          p = p.rotate(p2, new Rotation(oldNormal, 0.1, RotationConvention.VECTOR_OPERATOR));
83          Assert.assertTrue(! p.contains(p1));
84          Assert.assertTrue(p.contains(p2));
85          Assert.assertTrue(! p.contains(p3));
86  
87          p = p.rotate(p1, new Rotation(oldNormal, 0.1, RotationConvention.VECTOR_OPERATOR));
88          Assert.assertTrue(! p.contains(p1));
89          Assert.assertTrue(! p.contains(p2));
90          Assert.assertTrue(! p.contains(p3));
91  
92      }

94      @Test
95      public void testTranslate() throws MathArithmeticException {
96          Vector3D p1 = new Vector3D(1.2, 3.4, -5.8);
97          Vector3D p2 = new Vector3D(3.4, -5.8, 1.2);
98          Vector3D p3 = new Vector3D(-2.0, 4.3, 0.7);
99          Plane    p  = new Plane(p1, p2, p3, 1.0e-10);
100 
101         p = p.translate(new Vector3D(2.0, p.getU(), -1.5, p.getV()));
102         Assert.assertTrue(p.contains(p1));
103         Assert.assertTrue(p.contains(p2));
104         Assert.assertTrue(p.contains(p3));
105 
106         p = p.translate(new Vector3D(-1.2, p.getNormal()));
107         Assert.assertTrue(! p.contains(p1));
108         Assert.assertTrue(! p.contains(p2));
109         Assert.assertTrue(! p.contains(p3));
110 
111         p = p.translate(new Vector3D(+1.2, p.getNormal()));
112         Assert.assertTrue(p.contains(p1));
113         Assert.assertTrue(p.contains(p2));
114         Assert.assertTrue(p.contains(p3));
115 
116     }

118     @Test
119     public void testIntersection() throws MathArithmeticException, MathIllegalArgumentException {
120         Plane p = new Plane(new Vector3D(1, 2, 3), new Vector3D(-4, 1, -5), 1.0e-10);
121         Line  l = new Line(new Vector3D(0.2, -3.5, 0.7), new Vector3D(1.2, -2.5, -0.3), 1.0e-10);
122         Vector3D point = p.intersection(l);
123         Assert.assertTrue(p.contains(point));
124         Assert.assertTrue(l.contains(point));
125         Assert.assertNull(p.intersection(new Line(new Vector3D(10, 10, 10),
126                                                   new Vector3D(10, 10, 10).add(p.getNormal().orthogonal()),
127                                                   1.0e-10)));
128     }

144         Vector3D reference = new Vector3D (1.2, 3.4, -5.8);

150         Assert.assertEquals(reference.getY(), p.getY(), 1.0e-10);
151         Assert.assertEquals(reference.getZ(), p.getZ(), 1.0e-10);

154     @Test
155     public void testSimilar() throws MathArithmeticException {
156         Vector3D p1  = new Vector3D (1.2, 3.4, -5.8);
157         Vector3D p2  = new Vector3D (3.4, -5.8, 1.2);
158         Vector3D p3  = new Vector3D (-2.0, 4.3, 0.7);
159         Plane    pA  = new Plane(p1, p2, p3, 1.0e-10);
160         Plane    pB  = new Plane(p1, new Vector3D (11.4, -3.8, 5.1), p2, 1.0e-10);
161         Assert.assertTrue(! pA.isSimilarTo(pB));
162         Assert.assertTrue(pA.isSimilarTo(pA));
163         Assert.assertTrue(pA.isSimilarTo(new Plane(p1, p3, p2, 1.0e-10)));
164         Vector3D shift = new Vector3D(0.3, pA.getNormal());
165         Assert.assertTrue(! pA.isSimilarTo(new Plane(p1.add(shift),
166                                                      p3.add(shift),
167                                                      p2.add(shift),
168                                                      1.0e-10)));
169     }
