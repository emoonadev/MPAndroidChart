368     protected final List<Vector2D> createRandomPoints(int size) {
369         // create the cloud container
370         List<Vector2D> points = new ArrayList<Vector2D>(size);
371         // fill the cloud with a random distribution of points
372         for (int i = 0; i < size; i++) {
373             points.add(new Vector2D(random.nextDouble() * 2.0 - 1.0, random.nextDouble() * 2.0 - 1.0));
374         }
375         return points;
376     }

387     protected final void checkConvexHull(final Collection<Vector2D> points, final ConvexHull2D hull,
388                                          final boolean includesCollinearPoints, final double tolerance) {
389         Assert.assertNotNull(hull);
390         Assert.assertTrue(isConvex(hull, includesCollinearPoints, tolerance));
391         checkPointsInsideHullRegion(points, hull, includesCollinearPoints);
392     }
393 
394     // verify that the constructed hull is really convex
395     protected final boolean isConvex(final ConvexHull2D hull, final boolean includesCollinearPoints,
396                                      final double tolerance) {
397 
398         final Vector2D[] points = hull.getVertices();
399         int sign = 0;
400 
401         for (int i = 0; i < points.length; i++) {
402             Vector2D p1 = points[i == 0 ? points.length - 1 : i - 1];
403             Vector2D p2 = points[i];
404             Vector2D p3 = points[i == points.length - 1 ? 0 : i + 1];
405 
406             Vector2D d1 = p2.subtract(p1);
407             Vector2D d2 = p3.subtract(p2);
408 
409             Assert.assertTrue(d1.getNorm() > 1e-10);
410             Assert.assertTrue(d2.getNorm() > 1e-10);
411 
412             final double cross = MathArrays.linearCombination(d1.getX(), d2.getY(), -d1.getY(), d2.getX());
413             final int cmp = Precision.compareTo(cross, 0.0, tolerance);
414 
415             if (sign != 0 && cmp != sign) {
416                 if (includesCollinearPoints && cmp == 0) {
417                     // in case of collinear points the cross product will be zero
418                 } else {
419                     return false;
420                 }
421             }
422 
423             sign = cmp;
424         }
425 
426         return true;
427     }
428 
429     // verify that all points are inside the convex hull region
430     protected final void checkPointsInsideHullRegion(final Collection<Vector2D> points,
431                                                      final ConvexHull2D hull,
432                                                      final boolean includesCollinearPoints) {
433 
434         final Collection<Vector2D> hullVertices = Arrays.asList(hull.getVertices());
435         final Region<Euclidean2D> region = hull.createRegion();
436 
437         for (final Vector2D p : points) {
438             Location location = region.checkPoint(p);
439             Assert.assertTrue(location != Location.OUTSIDE);
440 
441             if (location == Location.BOUNDARY && includesCollinearPoints) {
442                 Assert.assertTrue(hullVertices.contains(p));
443             }
444         }
