33  public class MultivariateFunctionPenaltyAdapterTest {
34      @Test
35      public void testStartSimplexInsideRange() {
36          final BiQuadratic biQuadratic = new BiQuadratic(2.0, 2.5, 1.0, 3.0, 2.0, 3.0);
37          final MultivariateFunctionPenaltyAdapter wrapped
38                = new MultivariateFunctionPenaltyAdapter(biQuadratic,
39                                                         biQuadratic.getLower(),
40                                                         biQuadratic.getUpper(),
41                                                         1000.0, new double[] { 100.0, 100.0 });
42  
43          SimplexOptimizer optimizer = new SimplexOptimizer(1e-10, 1e-30);
44          final AbstractSimplex simplex = new NelderMeadSimplex(new double[] { 1.0, 0.5 });
45  
46          final PointValuePair optimum
47              = optimizer.optimize(new MaxEval(300),
48                                   new ObjectiveFunction(wrapped),
49                                   simplex,
50                                   GoalType.MINIMIZE,
51                                   new InitialGuess(new double[] { 1.5, 2.25 }));
52  
53          Assert.assertEquals(biQuadratic.getBoundedXOptimum(), optimum.getPoint()[0], 2e-7);
54          Assert.assertEquals(biQuadratic.getBoundedYOptimum(), optimum.getPoint()[1], 2e-7);
55      }

153     private static class BiQuadratic implements MultivariateFunction {
154 
155         private final double xOptimum;
156         private final double yOptimum;
157 
158         private final double xMin;
159         private final double xMax;
160         private final double yMin;
161         private final double yMax;
162 
163         public BiQuadratic(final double xOptimum, final double yOptimum,
164                            final double xMin, final double xMax,
165                            final double yMin, final double yMax) {
166             this.xOptimum = xOptimum;
167             this.yOptimum = yOptimum;
168             this.xMin     = xMin;
169             this.xMax     = xMax;
170             this.yMin     = yMin;
171             this.yMax     = yMax;
172         }
173 
174         public double value(double[] point) {
175             // the function should never be called with out of range points
176             Assert.assertTrue(point[0] >= xMin);
177             Assert.assertTrue(point[0] <= xMax);
178             Assert.assertTrue(point[1] >= yMin);
179             Assert.assertTrue(point[1] <= yMax);
180 
181             final double dx = point[0] - xOptimum;
182             final double dy = point[1] - yOptimum;
183             return dx * dx + dy * dy;
184 
185         }
186 
187         public double[] getLower() {
188             return new double[] { xMin, yMin };
189         }
190 
191         public double[] getUpper() {
192             return new double[] { xMax, yMax };
193         }
194 
195         public double getBoundedXOptimum() {
196             return (xOptimum < xMin) ? xMin : ((xOptimum > xMax) ? xMax : xOptimum);
197         }
198 
199         public double getBoundedYOptimum() {
200             return (yOptimum < yMin) ? yMin : ((yOptimum > yMax) ? yMax : yOptimum);
201         }
202 
203     }
