161     private RealMatrix checkAEqualPTPt(RealMatrix matrix) {
162         SchurTransformer transformer = new SchurTransformer(matrix);
163         RealMatrix p  = transformer.getP();
164         RealMatrix t  = transformer.getT();
165         RealMatrix pT = transformer.getPT();
166 
167         RealMatrix result = p.multiply(t).multiply(pT);
168 
169         double norm = result.subtract(matrix).getNorm();
170         Assert.assertEquals(0, norm, 1.0e-9);
171 
172         return t;
173     }
174 
175     private void checkOrthogonal(RealMatrix m) {
176         RealMatrix mTm = m.transpose().multiply(m);
177         RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
178         Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-14);
179     }
180 
181     private void checkSchurForm(final RealMatrix m) {
182         final int rows = m.getRowDimension();
183         final int cols = m.getColumnDimension();
184         for (int i = 0; i < rows; ++i) {
185             for (int j = 0; j < cols; ++j) {
186                 if (i > j + 1) {
187                     Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
188                 }
189             }
190         }
191     }

193     @SuppressWarnings("unused")
194     private void checkMatricesValues(double[][] matrix, double[][] pRef, double[][] hRef) {
195 
196         SchurTransformer transformer =
197             new SchurTransformer(MatrixUtils.createRealMatrix(matrix));
198 
199         // check values against known references
200         RealMatrix p = transformer.getP();
201         Assert.assertEquals(0, p.subtract(MatrixUtils.createRealMatrix(pRef)).getNorm(), 1.0e-14);
202 
203         RealMatrix t = transformer.getT();
204         Assert.assertEquals(0, t.subtract(MatrixUtils.createRealMatrix(hRef)).getNorm(), 1.0e-14);
205 
206         // check the same cached instance is returned the second time
207         Assert.assertTrue(p == transformer.getP());
208         Assert.assertTrue(t == transformer.getT());
209     }
