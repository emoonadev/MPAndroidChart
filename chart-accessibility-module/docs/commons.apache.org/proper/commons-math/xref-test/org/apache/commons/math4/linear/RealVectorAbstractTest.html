73      /**
74       * <p>
75       * This is an attempt at covering most particular cases of combining two
76       * values. Here {@code x} is the value returned by
77       * {@link #getPreferredEntryValue()}, while {@code y} and {@code z} are two
78       * "normal" values.
79       * </p>
80       * <ol>
81       *   <li>
82       *     Addition: the following cases should be covered
83       *     <ul>
84       *       <li>{@code (2 * x) + (-x)}</li>
85       *       <li>{@code (-x) + 2 * x}</li>
86       *       <li>{@code x + y}</li>
87       *       <li>{@code y + x}</li>
88       *       <li>{@code y + z}</li>
89       *       <li>{@code y + (x - y)}</li>
90       *       <li>{@code (y - x) + x}</li>
91       *     </ul>
92       *     The values to be considered are:
93       *     {@code x, y, z, 2 * x, -x, x - y, y - x}.
94       *   </li>
95       *   <li>
96       *     Subtraction: the following cases should be covered
97       *     <ul>
98       *       <li>{@code (2 * x) - x}</li>
99       *       <li>{@code x - y}</li>
100      *       <li>{@code y - x}</li>
101      *       <li>{@code y - z}</li>
102      *       <li>{@code y - (y - x)}</li>
103      *       <li>{@code (y + x) - y}</li>
104      *     </ul>
105      *     The values to be considered are: {@code x, y, z, x + y, y - x}.
106      *   </li>
107      *   <li>
108      *     Multiplication
109      *     <ul>
110      *       <li>{@code (x * x) * (1 / x)}</li>
111      *       <li>{@code (1 / x) * (x * x)}</li>
112      *       <li>{@code x * y}</li>
113      *       <li>{@code y * x}</li>
114      *       <li>{@code y * z}</li>
115      *     </ul>
116      *     The values to be considered are: {@code x, y, z, 1 / x, x * x}.
117      *   </li>
118      *   <li>
119      *     Division
120      *     <ul>
121      *       <li>{@code (x * x) / x}</li>
122      *       <li>{@code x / y}</li>
123      *       <li>{@code y / x}</li>
124      *       <li>{@code y / z}</li>
125      *     </ul>
126      *     The values to be considered are: {@code x, y, z, x * x}.
127      *   </li>
128      * </ol>
129      * Also to be considered {@code NaN}, {@code POSITIVE_INFINITY},
130      * {@code NEGATIVE_INFINITY}, {@code +0.0}, {@code -0.0}.
131      */

466     protected void doTestEbeBinaryOperation(final BinaryOperation op, final boolean mixed, boolean ignoreSpecial) {
467         final double[] data1 = new double[values.length * values.length];
468         final double[] data2 = new double[values.length * values.length];
469         int k = 0;
470         for (int i = 0; i < values.length; i++) {
471             for (int j = 0; j < values.length; j++) {
472                 data1[k] = values[i];
473                 data2[k] = values[j];
474                 ++k;
475             }
476         }
477         final RealVector v1 = create(data1);
478         final RealVector v2 = mixed ? createAlien(data2) : create(data2);
479         final RealVector actual;
480         switch (op) {
481             case ADD:
482                 actual = v1.add(v2);
483                 break;
484             case SUB:
485                 actual = v1.subtract(v2);
486                 break;
487             case MUL:
488                 actual = v1.ebeMultiply(v2);
489                 break;
490             case DIV:
491                 actual = v1.ebeDivide(v2);
492                 break;
493             default:
494                 throw new AssertionError("unexpected value");
495         }
496         final double[] expected = new double[data1.length];
497         for (int i = 0; i < expected.length; i++) {
498             switch (op) {
499                 case ADD:
500                     expected[i] = data1[i] + data2[i];
501                     break;
502                 case SUB:
503                     expected[i] = data1[i] - data2[i];
504                     break;
505                 case MUL:
506                     expected[i] = data1[i] * data2[i];
507                     break;
508                 case DIV:
509                     expected[i] = data1[i] / data2[i];
510                     break;
511                 default:
512                     throw new AssertionError("unexpected value");
513             }
514         }
515         for (int i = 0; i < expected.length; i++) {
516             boolean isSpecial = Double.isNaN(expected[i]) || Double.isInfinite(expected[i]);
517             if (!(isSpecial && ignoreSpecial)) {
518                 final String msg = "entry #"+i+", left = "+data1[i]+", right = " + data2[i];
519                 Assert.assertEquals(msg, expected[i], actual.getEntry(i), 0.0);
520             }
521         }
522     }
