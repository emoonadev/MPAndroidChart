551     /*
552      * Test assumes that the network is
553      *
554      *  0-----1-----2-----3-----4
555      *  |     |     |     |     |
556      *  |     |     |     |     |
557      *  5-----6-----7-----8-----9
558      *  |     |     |     |     |
559      *  |     |     |     |     |
560      *  10----11----12----13---14
561      *  |     |     |     |     |
562      *  |     |     |     |     |
563      *  15----16----17----18---19
564      *  |     |     |     |     |
565      *  |     |     |     |     |
566      *  20----21----22----23---24
567      */
568     @Test
569     public void testConcentricNeighbourhood() {
570         final FeatureInitializer[] initArray = { init };
571         final Network net = new NeuronSquareMesh2D(5, true,
572                                              5, true,
573                                              SquareNeighbourhood.VON_NEUMANN,
574                                              initArray).getNetwork();
575 
576         Collection<Neuron> neighbours;
577         Collection<Neuron> exclude = new HashSet<Neuron>();
578 
579         // Level-1 neighbourhood.
580         neighbours = net.getNeighbours(net.getNeuron(12));
581         for (long nId : new long[] { 7, 11, 13, 17 }) {
582             Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
583         }
584         // Ensures that no other neurons is in the neihbourhood set.
585         Assert.assertEquals(4, neighbours.size());
586 
587         // 1. Add the neuron to the "exclude" list.
588         exclude.add(net.getNeuron(12));
589         // 2. Add all neurons from level-1 neighbourhood.
590         exclude.addAll(neighbours);
591         // 3. Retrieve level-2 neighbourhood.
592         neighbours = net.getNeighbours(neighbours, exclude);
593         for (long nId : new long[] { 6, 8, 16, 18, 2, 10, 14, 22 }) {
594             Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
595         }
596         // Ensures that no other neurons is in the neihbourhood set.
597         Assert.assertEquals(8, neighbours.size());
598     }

618     public void testConcentricNeighbourhood2() {
619         final FeatureInitializer[] initArray = { init };
620         final Network net = new NeuronSquareMesh2D(5, true,
621                                                    5, true,
622                                                    SquareNeighbourhood.MOORE,
623                                                    initArray).getNetwork();
624 
625         Collection<Neuron> neighbours;
626         Collection<Neuron> exclude = new HashSet<Neuron>();
627 
628         // Level-1 neighbourhood.
629         neighbours = net.getNeighbours(net.getNeuron(8));
630         for (long nId : new long[] { 2, 3, 4, 7, 9, 12, 13, 14 }) {
631             Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
632         }
633         // Ensures that no other neurons is in the neihbourhood set.
634         Assert.assertEquals(8, neighbours.size());
635 
636         // 1. Add the neuron to the "exclude" list.
637         exclude.add(net.getNeuron(8));
638         // 2. Add all neurons from level-1 neighbourhood.
639         exclude.addAll(neighbours);
640         // 3. Retrieve level-2 neighbourhood.
641         neighbours = net.getNeighbours(neighbours, exclude);
642         for (long nId : new long[] { 1, 6, 11, 16, 17, 18, 19, 15, 10, 5, 0, 20, 24, 23, 22, 21 }) {
643             Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
644         }
645         // Ensures that no other neurons is in the neihbourhood set.
646         Assert.assertEquals(16, neighbours.size());
