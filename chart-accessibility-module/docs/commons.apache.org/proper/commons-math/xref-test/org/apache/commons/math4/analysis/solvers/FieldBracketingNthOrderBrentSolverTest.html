51      @Test
52      public void testConvergenceOnFunctionAccuracy() {
53          FieldBracketingNthOrderBrentSolver<Dfp> solver =
54                  new FieldBracketingNthOrderBrentSolver<Dfp>(relativeAccuracy, absoluteAccuracy,
55                                                              field.newDfp(1.0e-20), 20);
56          RealFieldUnivariateFunction<Dfp> f = new RealFieldUnivariateFunction<Dfp>() {
57              public Dfp value(Dfp x) {
58                  Dfp one     = field.getOne();
59                  Dfp oneHalf = one.divide(2);
60                  Dfp xMo     = x.subtract(one);
61                  Dfp xMh     = x.subtract(oneHalf);
62                  Dfp xPh     = x.add(oneHalf);
63                  Dfp xPo     = x.add(one);
64                  return xMo.multiply(xMh).multiply(x).multiply(xPh).multiply(xPo);
65              }
66          };
67  
68          Dfp result = solver.solve(20, f, field.newDfp(0.2), field.newDfp(0.9),
69                                    field.newDfp(0.4), AllowedSolution.BELOW_SIDE);
70          Assert.assertTrue(f.value(result).abs().lessThan(solver.getFunctionValueAccuracy()));
71          Assert.assertTrue(f.value(result).negativeOrNull());
72          Assert.assertTrue(result.subtract(field.newDfp(0.5)).subtract(solver.getAbsoluteAccuracy()).positiveOrNull());
73          result = solver.solve(20, f, field.newDfp(-0.9), field.newDfp(-0.2),
74                                field.newDfp(-0.4), AllowedSolution.ABOVE_SIDE);
75          Assert.assertTrue(f.value(result).abs().lessThan(solver.getFunctionValueAccuracy()));
76          Assert.assertTrue(f.value(result).positiveOrNull());
77          Assert.assertTrue(result.add(field.newDfp(0.5)).subtract(solver.getAbsoluteAccuracy()).negativeOrNull());
78      }

80      @Test
81      public void testNeta() {
82  
83          // the following test functions come from Beny Neta's paper:
84          // "Several New Methods for solving Equations"
85          // intern J. Computer Math Vol 23 pp 265-282
86          // available here: http://www.math.nps.navy.mil/~bneta/SeveralNewMethods.PDF
87          for (AllowedSolution allowed : AllowedSolution.values()) {
88              check(new RealFieldUnivariateFunction<Dfp>() {
89                  public Dfp value(Dfp x) {
90                      return DfpMath.sin(x).subtract(x.divide(2));
91                  }
92              }, 200, -2.0, 2.0, allowed);
93  
94              check(new RealFieldUnivariateFunction<Dfp>() {
95                  public Dfp value(Dfp x) {
96                      return DfpMath.pow(x, 5).add(x).subtract(field.newDfp(10000));
97                  }
98              }, 200, -5.0, 10.0, allowed);
99  
100             check(new RealFieldUnivariateFunction<Dfp>() {
101                 public Dfp value(Dfp x) {
102                     return x.sqrt().subtract(field.getOne().divide(x)).subtract(field.newDfp(3));
103                 }
104             }, 200, 0.001, 10.0, allowed);
105 
106             check(new RealFieldUnivariateFunction<Dfp>() {
107                 public Dfp value(Dfp x) {
108                     return DfpMath.exp(x).add(x).subtract(field.newDfp(20));
109                 }
110             }, 200, -5.0, 5.0, allowed);
111 
112             check(new RealFieldUnivariateFunction<Dfp>() {
113                 public Dfp value(Dfp x) {
114                     return DfpMath.log(x).add(x.sqrt()).subtract(field.newDfp(5));
115                 }
116             }, 200, 0.001, 10.0, allowed);
117 
118             check(new RealFieldUnivariateFunction<Dfp>() {
119                 public Dfp value(Dfp x) {
120                     return x.subtract(field.getOne()).multiply(x).multiply(x).subtract(field.getOne());
121                 }
122             }, 200, -0.5, 1.5, allowed);
123         }
124 
125     }

127     private void check(RealFieldUnivariateFunction<Dfp> f, int maxEval, double min, double max,
128                        AllowedSolution allowedSolution) {
129         FieldBracketingNthOrderBrentSolver<Dfp> solver =
130                 new FieldBracketingNthOrderBrentSolver<Dfp>(relativeAccuracy, absoluteAccuracy,
131                                                      functionValueAccuracy, 20);
132         Dfp xResult = solver.solve(maxEval, f, field.newDfp(min), field.newDfp(max),
133                                    allowedSolution);
134         Dfp yResult = f.value(xResult);
135         switch (allowedSolution) {
136         case ANY_SIDE :
137             Assert.assertTrue(yResult.abs().lessThan(functionValueAccuracy.multiply(2)));
138             break;
139         case LEFT_SIDE : {
140             boolean increasing = f.value(xResult).add(absoluteAccuracy).greaterThan(yResult);
141             Assert.assertTrue(increasing ? yResult.negativeOrNull() : yResult.positiveOrNull());
142             break;
143         }
144         case RIGHT_SIDE : {
145             boolean increasing = f.value(xResult).add(absoluteAccuracy).greaterThan(yResult);
146             Assert.assertTrue(increasing ? yResult.positiveOrNull() : yResult.negativeOrNull());
147             break;
148         }
149         case BELOW_SIDE :
150             Assert.assertTrue(yResult.negativeOrNull());
151             break;
152         case ABOVE_SIDE :
153             Assert.assertTrue(yResult.positiveOrNull());
154             break;
155         default :
156             // this should never happen
157             throw new MathInternalError(null);
158         }
159     }
