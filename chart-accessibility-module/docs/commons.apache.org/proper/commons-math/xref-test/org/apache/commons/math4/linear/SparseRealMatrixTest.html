372     /** test examples in user guide */
373     @Test
374     public void testExamples() {
375         // Create a real matrix with two rows and three columns
376         double[][] matrixData = { { 1d, 2d, 3d }, { 2d, 5d, 3d } };
377         RealMatrix m = createSparseMatrix(matrixData);
378         // One more with three rows, two columns
379         double[][] matrixData2 = { { 1d, 2d }, { 2d, 5d }, { 1d, 7d } };
380         RealMatrix n = createSparseMatrix(matrixData2);
381         // Now multiply m by n
382         RealMatrix p = m.multiply(n);
383         Assert.assertEquals(2, p.getRowDimension());
384         Assert.assertEquals(2, p.getColumnDimension());
385         // Invert p
386         RealMatrix pInverse = new LUDecomposition(p).getSolver().getInverse();
387         Assert.assertEquals(2, pInverse.getRowDimension());
388         Assert.assertEquals(2, pInverse.getColumnDimension());
389 
390         // Solve example
391         double[][] coefficientsData = { { 2, 3, -2 }, { -1, 7, 6 },
392                 { 4, -3, -5 } };
393         RealMatrix coefficients = createSparseMatrix(coefficientsData);
394         RealVector constants = new ArrayRealVector(new double[]{ 1, -2, 1 }, false);
395         RealVector solution = new LUDecomposition(coefficients).getSolver().solve(constants);
396         final double cst0 = constants.getEntry(0);
397         final double cst1 = constants.getEntry(1);
398         final double cst2 = constants.getEntry(2);
399         final double sol0 = solution.getEntry(0);
400         final double sol1 = solution.getEntry(1);
401         final double sol2 = solution.getEntry(2);
402         Assert.assertEquals(2 * sol0 + 3 * sol1 - 2 * sol2, cst0, 1E-12);
403         Assert.assertEquals(-1 * sol0 + 7 * sol1 + 6 * sol2, cst1, 1E-12);
404         Assert.assertEquals(4 * sol0 - 3 * sol1 - 5 * sol2, cst2, 1E-12);
405 
406     }

677     /** verifies that two matrices are close (1-norm) */
678     protected void assertClose(String msg, RealMatrix m, RealMatrix n,
679             double tolerance) {
680         Assert.assertTrue(msg, m.subtract(n).getNorm() < tolerance);
681     }
682 
683     /** verifies that two vectors are close (sup norm) */
684     protected void assertClose(String msg, double[] m, double[] n,
685             double tolerance) {
686         if (m.length != n.length) {
687             Assert.fail("vectors not same length");
688         }
689         for (int i = 0; i < m.length; i++) {
690             Assert.assertEquals(msg + " " + i + " elements differ", m[i], n[i],
691                     tolerance);
692         }
693     }

695     private OpenMapRealMatrix createSparseMatrix(double[][] data) {
696         OpenMapRealMatrix matrix = new OpenMapRealMatrix(data.length, data[0].length);
697         for (int row = 0; row < data.length; row++) {
698             for (int col = 0; col < data[row].length; col++) {
699                 matrix.setEntry(row, col, data[row][col]);
700             }
701         }
702         return matrix;
703     }
