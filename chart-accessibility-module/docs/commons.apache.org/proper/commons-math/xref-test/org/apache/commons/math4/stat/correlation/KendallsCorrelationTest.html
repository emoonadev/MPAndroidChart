131     @Test
132     public void testSimpleJumble() {
133         //                                     A    B    C    D
134         final double[] xArray = new double[] {1.0, 2.0, 3.0, 4.0};
135         final double[] yArray = new double[] {1.0, 3.0, 2.0, 4.0};
136 
137         // 6 pairs: (A,B) (A,C) (A,D) (B,C) (B,D) (C,D)
138         // (B,C) is discordant, the other 5 are concordant
139 
140         Assert.assertEquals((5 - 1) / (double) 6,
141                 correlation.correlation(xArray, yArray),
142                 Double.MIN_VALUE);
143     }

159     @Test
160     public void testOrderedTies() {
161         final int length = 10;
162         final double[] xArray = new double[length];
163         final double[] yArray = new double[length];
164         for (int i = 0; i < length; i++) {
165             xArray[i] = i / 2;
166             yArray[i] = i / 2;
167         }
168         // 5 pairs of points that are tied in both values.
169         // 16 + 12 + 8 + 4 = 40 concordant
170         // (40 - 0) / Math.sqrt((45 - 5) * (45 - 5)) = 1
171         Assert.assertEquals(1.0, correlation.correlation(xArray, yArray), Double.MIN_VALUE);
172     }

265     @Test
266     public void testMath1277() {
267         // example that led to a correlation coefficient outside of [-1, 1]
268         // due to a bug reported in MATH-1277
269         UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_1024_A, 0);
270         double[] xArray = new double[120000];
271         double[] yArray = new double[120000];
272         for (int i = 0; i < xArray.length; ++i) {
273             xArray[i] =  rng.nextDouble();
274         }
275         for (int i = 0; i < yArray.length; ++i) {
276             yArray[i] =  rng.nextDouble();
277         }
278         double coefficient = correlation.correlation(xArray, yArray);
279         Assert.assertTrue(1.0 >= coefficient && -1.0 <= coefficient);
280     }
