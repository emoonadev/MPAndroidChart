53      public void testCompose() {
54          UnivariateFunction id = new Identity();
55          Assert.assertEquals(3, FunctionUtils.compose(id, id, id).value(3), EPS);
56  
57          UnivariateFunction c = new Constant(4);
58          Assert.assertEquals(4, FunctionUtils.compose(id, c).value(3), EPS);
59          Assert.assertEquals(4, FunctionUtils.compose(c, id).value(3), EPS);
60  
61          UnivariateFunction m = new Minus();
62          Assert.assertEquals(-3, FunctionUtils.compose(m).value(3), EPS);
63          Assert.assertEquals(3, FunctionUtils.compose(m, m).value(3), EPS);
64  
65          UnivariateFunction inv = new Inverse();
66          Assert.assertEquals(-0.25, FunctionUtils.compose(inv, m, c, id).value(3), EPS);
67  
68          UnivariateFunction pow = new Power(2);
69          Assert.assertEquals(81, FunctionUtils.compose(pow, pow).value(3), EPS);
70      }

72      @Test
73      public void testComposeDifferentiable() {
74          UnivariateDifferentiableFunction id = new Identity();
75          Assert.assertEquals(1, FunctionUtils.compose(id, id, id).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
76  
77          UnivariateDifferentiableFunction c = new Constant(4);
78          Assert.assertEquals(0, FunctionUtils.compose(id, c).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
79          Assert.assertEquals(0, FunctionUtils.compose(c, id).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
80  
81          UnivariateDifferentiableFunction m = new Minus();
82          Assert.assertEquals(-1, FunctionUtils.compose(m).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
83          Assert.assertEquals(1, FunctionUtils.compose(m, m).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
84  
85          UnivariateDifferentiableFunction inv = new Inverse();
86          Assert.assertEquals(0.25, FunctionUtils.compose(inv, m, id).value(new DerivativeStructure(1, 1, 0, 2)).getPartialDerivative(1), EPS);
87  
88          UnivariateDifferentiableFunction pow = new Power(2);
89          Assert.assertEquals(108, FunctionUtils.compose(pow, pow).value(new DerivativeStructure(1, 1, 0, 3)).getPartialDerivative(1), EPS);
90  
91          UnivariateDifferentiableFunction log = new Log();
92          double a = 9876.54321;
93          Assert.assertEquals(pow.value(new DerivativeStructure(1, 1, 0, a)).getPartialDerivative(1) / pow.value(a),
94                              FunctionUtils.compose(log, pow).value(new DerivativeStructure(1, 1, 0, a)).getPartialDerivative(1), EPS);
95      }

137         Assert.assertEquals(8 * a, FunctionUtils.multiply(c, id, id).value(new DerivativeStructure(1, 1, 0, a)).getPartialDerivative(1), EPS);

147                             FunctionUtils.multiply(inv, pow, cosh).value(new DerivativeStructure(1, 1, 0, a)).getPartialDerivative(1), 8 * EPS);

238     @Test
239     public void testToDifferentiableUnivariate() {
240 
241         final UnivariateFunction f0 = new UnivariateFunction() {
242             @Override
243             public double value(final double x) {
244                 return x * x;
245             }
246         };
247         final UnivariateFunction f1 = new UnivariateFunction() {
248             @Override
249             public double value(final double x) {
250                 return 2 * x;
251             }
252         };
253         final UnivariateFunction f2 = new UnivariateFunction() {
254             @Override
255             public double value(final double x) {
256                 return 2;
257             }
258         };
259         final UnivariateDifferentiableFunction f = FunctionUtils.toDifferentiable(f0, f1, f2);
260 
261         for (double t = -1.0; t < 1; t += 0.01) {
262             // x = sin(t)
263             DerivativeStructure dsT = new DerivativeStructure(1, 2, 0, t);
264             DerivativeStructure y = f.value(dsT.sin());
265             Assert.assertEquals(FastMath.sin(t) * FastMath.sin(t),               f.value(FastMath.sin(t)),  1.0e-15);
266             Assert.assertEquals(FastMath.sin(t) * FastMath.sin(t),               y.getValue(),              1.0e-15);
267             Assert.assertEquals(2 * FastMath.cos(t) * FastMath.sin(t),           y.getPartialDerivative(1), 1.0e-15);
268             Assert.assertEquals(2 * (1 - 2 * FastMath.sin(t) * FastMath.sin(t)), y.getPartialDerivative(2), 1.0e-15);
269         }
270 
271         try {
272             f.value(new DerivativeStructure(1, 3, 0.0));
273             Assert.fail("an exception should have been thrown");
274         } catch (NumberIsTooLargeException e) {
275             Assert.assertEquals(2, e.getMax());
276             Assert.assertEquals(3, e.getArgument());
277         }
278     }

280     @Test
281     public void testToDifferentiableMultivariate() {
282 
283         final double a = 1.5;
284         final double b = 0.5;
285         final MultivariateFunction f = new MultivariateFunction() {
286             @Override
287             public double value(final double[] point) {
288                 return a * point[0] + b * point[1];
289             }
290         };
291         final MultivariateVectorFunction gradient = new MultivariateVectorFunction() {
292             @Override
293             public double[] value(final double[] point) {
294                 return new double[] { a, b };
295             }
296         };
297         final MultivariateDifferentiableFunction mdf = FunctionUtils.toDifferentiable(f, gradient);
298 
299         for (double t = -1.0; t < 1; t += 0.01) {
300             // x = sin(t), y = cos(t), hence the method really becomes univariate
301             DerivativeStructure dsT = new DerivativeStructure(1, 1, 0, t);
302             DerivativeStructure y = mdf.value(new DerivativeStructure[] { dsT.sin(), dsT.cos() });
303             Assert.assertEquals(a * FastMath.sin(t) + b * FastMath.cos(t), y.getValue(),              1.0e-15);
304             Assert.assertEquals(a * FastMath.cos(t) - b * FastMath.sin(t), y.getPartialDerivative(1), 1.0e-15);
305         }
306 
307         for (double u = -1.0; u < 1; u += 0.01) {
308             DerivativeStructure dsU = new DerivativeStructure(2, 1, 0, u);
309             for (double v = -1.0; v < 1; v += 0.01) {
310                 DerivativeStructure dsV = new DerivativeStructure(2, 1, 1, v);
311                 DerivativeStructure y = mdf.value(new DerivativeStructure[] { dsU, dsV });
312                 Assert.assertEquals(a * u + b * v, mdf.value(new double[] { u, v }), 1.0e-15);
313                 Assert.assertEquals(a * u + b * v, y.getValue(),                     1.0e-15);
314                 Assert.assertEquals(a,             y.getPartialDerivative(1, 0),     1.0e-15);
315                 Assert.assertEquals(b,             y.getPartialDerivative(0, 1),     1.0e-15);
316             }
317         }
318 
319         try {
320             mdf.value(new DerivativeStructure[] { new DerivativeStructure(1, 3, 0.0), new DerivativeStructure(1, 3, 0.0) });
321             Assert.fail("an exception should have been thrown");
322         } catch (NumberIsTooLargeException e) {
323             Assert.assertEquals(1, e.getMax());
324             Assert.assertEquals(3, e.getArgument());
325         }
326     }

386     @Test
387     public void testDerivativeMultivariate() {
388 
389         final double a = 1.5;
390         final double b = 0.5;
391         final double c = 0.25;
392         final MultivariateDifferentiableFunction mdf = new MultivariateDifferentiableFunction() {
393 
394             @Override
395             public double value(double[] point) {
396                 return a * point[0] * point[0] + b * point[1] * point[1] + c * point[0] * point[1];
397             }
398 
399             @Override
400             public DerivativeStructure value(DerivativeStructure[] point) {
401                 DerivativeStructure x  = point[0];
402                 DerivativeStructure y  = point[1];
403                 DerivativeStructure x2 = x.multiply(x);
404                 DerivativeStructure y2 = y.multiply(y);
405                 DerivativeStructure xy = x.multiply(y);
406                 return x2.multiply(a).add(y2.multiply(b)).add(xy.multiply(c));
407             }
408 
409         };
410 
411         final MultivariateFunction f       = FunctionUtils.derivative(mdf, new int[] { 0, 0 });
412         final MultivariateFunction dfdx    = FunctionUtils.derivative(mdf, new int[] { 1, 0 });
413         final MultivariateFunction dfdy    = FunctionUtils.derivative(mdf, new int[] { 0, 1 });
414         final MultivariateFunction d2fdx2  = FunctionUtils.derivative(mdf, new int[] { 2, 0 });
415         final MultivariateFunction d2fdy2  = FunctionUtils.derivative(mdf, new int[] { 0, 2 });
416         final MultivariateFunction d2fdxdy = FunctionUtils.derivative(mdf, new int[] { 1, 1 });
417 
418         for (double x = -1.0; x < 1; x += 0.01) {
419             for (double y = -1.0; y < 1; y += 0.01) {
420                 Assert.assertEquals(a * x * x + b * y * y + c * x * y, f.value(new double[]       { x, y }), 1.0e-15);
421                 Assert.assertEquals(2 * a * x + c * y,                 dfdx.value(new double[]    { x, y }), 1.0e-15);
422                 Assert.assertEquals(2 * b * y + c * x,                 dfdy.value(new double[]    { x, y }), 1.0e-15);
423                 Assert.assertEquals(2 * a,                             d2fdx2.value(new double[]  { x, y }), 1.0e-15);
424                 Assert.assertEquals(2 * b,                             d2fdy2.value(new double[]  { x, y }), 1.0e-15);
425                 Assert.assertEquals(c,                                 d2fdxdy.value(new double[] { x, y }), 1.0e-15);
426             }
427         }
