17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.distribution.AbstractIntegerDistribution;
20  import org.junit.Assert;
21  import org.junit.Test;
22  
23  /**
24   * Test cases for AbstractIntegerDistribution default implementations.
25   *
26   */
27  public class AbstractIntegerDistributionTest {
28      protected final DiceDistribution diceDistribution = new DiceDistribution();
29      protected final double p = diceDistribution.probability(1);
30  
31      @Test
32      public void testInverseCumulativeProbabilityMethod()
33      {
34          double precision = 0.000000000000001;
35          Assert.assertEquals(1, diceDistribution.inverseCumulativeProbability(0));
36          Assert.assertEquals(1, diceDistribution.inverseCumulativeProbability((1d-Double.MIN_VALUE)/6d));
37          Assert.assertEquals(2, diceDistribution.inverseCumulativeProbability((1d+precision)/6d));
38          Assert.assertEquals(2, diceDistribution.inverseCumulativeProbability((2d-Double.MIN_VALUE)/6d));
39          Assert.assertEquals(3, diceDistribution.inverseCumulativeProbability((2d+precision)/6d));
40          Assert.assertEquals(3, diceDistribution.inverseCumulativeProbability((3d-Double.MIN_VALUE)/6d));
41          Assert.assertEquals(4, diceDistribution.inverseCumulativeProbability((3d+precision)/6d));
42          Assert.assertEquals(4, diceDistribution.inverseCumulativeProbability((4d-Double.MIN_VALUE)/6d));
43          Assert.assertEquals(5, diceDistribution.inverseCumulativeProbability((4d+precision)/6d));
44          Assert.assertEquals(5, diceDistribution.inverseCumulativeProbability((5d-precision)/6d));//Can't use Double.MIN
45          Assert.assertEquals(6, diceDistribution.inverseCumulativeProbability((5d+precision)/6d));
46          Assert.assertEquals(6, diceDistribution.inverseCumulativeProbability((6d-precision)/6d));//Can't use Double.MIN
47          Assert.assertEquals(6, diceDistribution.inverseCumulativeProbability((6d)/6d));
48      }
49  
50      @Test
51      public void testCumulativeProbabilitiesSingleArguments() {
52          for (int i = 1; i < 7; i++) {
53              Assert.assertEquals(p * i,
54                      diceDistribution.cumulativeProbability(i), Double.MIN_VALUE);
55          }
56          Assert.assertEquals(0.0,
57                  diceDistribution.cumulativeProbability(0), Double.MIN_VALUE);
58          Assert.assertEquals(1.0,
59                  diceDistribution.cumulativeProbability(7), Double.MIN_VALUE);
60      }
61  
62      @Test
63      public void testProbabilitiesRangeArguments() {
64          int lower = 0;
65          int upper = 6;
66          for (int i = 0; i < 2; i++) {
67              // cum(0,6) = p(0 < X <= 6) = 1, cum(1,5) = 4/6, cum(2,4) = 2/6
68              Assert.assertEquals(1 - p * 2 * i,
69                      diceDistribution.probability(lower, upper), 1E-12);
70              lower++;
71              upper--;
72          }
73          for (int i = 0; i < 6; i++) {
74              Assert.assertEquals(p, diceDistribution.probability(i, i+1), 1E-12);
75          }
76      }
77  
78      /**
79       * Simple distribution modeling a 6-sided die
80       */
81      class DiceDistribution extends AbstractIntegerDistribution {
82          public static final long serialVersionUID = 23734213;
83  
84          private final double p = 1d/6d;
85  
86          public double probability(int x) {
87              if (x < 1 || x > 6) {
88                  return 0;
89              } else {
90                  return p;
91              }
92          }
93  
94          public double cumulativeProbability(int x) {
95              if (x < 1) {
96                  return 0;
97              } else if (x >= 6) {
98                  return 1;
99              } else {
100                 return p * x;
101             }
102         }
103 
104         public double getNumericalMean() {
105             return 3.5;
106         }
107 
108         public double getNumericalVariance() {
109             return 70/24;  // E(X^2) - E(X)^2
110         }
111 
112         public int getSupportLowerBound() {
113             return 1;
114         }
115 
116         public int getSupportUpperBound() {
117             return 6;
118         }
119 
120         public final boolean isSupportConnected() {
121             return true;
122         }
123     }
124 }
