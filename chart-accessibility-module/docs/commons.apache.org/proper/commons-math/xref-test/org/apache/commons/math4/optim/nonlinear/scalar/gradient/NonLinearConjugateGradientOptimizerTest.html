210     @Test
211     public void testTwoSets() {
212         final double epsilon = 1.0e-7;
213         LinearProblem problem = new LinearProblem(new double[][] {
214                 {  2,  1,   0,  4,       0, 0 },
215                 { -4, -2,   3, -7,       0, 0 },
216                 {  4,  1,  -2,  8,       0, 0 },
217                 {  0, -3, -12, -1,       0, 0 },
218                 {  0,  0,   0,  0, epsilon, 1 },
219                 {  0,  0,   0,  0,       1, 1 }
220         }, new double[] { 2, -9, 2, 2, 1 + epsilon * epsilon, 2});
221 
222         final Preconditioner preconditioner
223             = new Preconditioner() {
224                     public double[] precondition(double[] point, double[] r) {
225                         double[] d = r.clone();
226                         d[0] /=  72.0;
227                         d[1] /=  30.0;
228                         d[2] /= 314.0;
229                         d[3] /= 260.0;
230                         d[4] /= 2 * (1 + epsilon * epsilon);
231                         d[5] /= 4.0;
232                         return d;
233                     }
234                 };
235 
236         NonLinearConjugateGradientOptimizer optimizer
237            = new NonLinearConjugateGradientOptimizer(NonLinearConjugateGradientOptimizer.Formula.POLAK_RIBIERE,
238                                                      new SimpleValueChecker(1e-13, 1e-13),
239                                                      1e-7, 1e-7, 1,
240                                                      preconditioner);
241 
242         PointValuePair optimum
243             = optimizer.optimize(new MaxEval(100),
244                                  problem.getObjectiveFunction(),
245                                  problem.getObjectiveFunctionGradient(),
246                                  GoalType.MINIMIZE,
247                                  new InitialGuess(new double[] { 0, 0, 0, 0, 0, 0 }));
248 
249         final double[] result = optimum.getPoint();
250         final double[] expected = {3, 4, -1, -2, 1 + epsilon, 1 - epsilon};
251 
252         Assert.assertEquals(expected[0], result[0], 1.0e-7);
253         Assert.assertEquals(expected[1], result[1], 1.0e-7);
254         Assert.assertEquals(expected[2], result[2], 1.0e-9);
255         Assert.assertEquals(expected[3], result[3], 1.0e-8);
256         Assert.assertEquals(expected[4] + epsilon, result[4], 1.0e-6);
257         Assert.assertEquals(expected[5] - epsilon, result[5], 1.0e-6);
258 
259     }

439     private static class LinearProblem {
440         final RealMatrix factors;
441         final double[] target;
442 
443         public LinearProblem(double[][] factors,
444                              double[] target) {
445             this.factors = new BlockRealMatrix(factors);
446             this.target  = target;
447         }
448 
449         public ObjectiveFunction getObjectiveFunction() {
450             return new ObjectiveFunction(new MultivariateFunction() {
451                     public double value(double[] point) {
452                         double[] y = factors.operate(point);
453                         double sum = 0;
454                         for (int i = 0; i < y.length; ++i) {
455                             double ri = y[i] - target[i];
456                             sum += ri * ri;
457                         }
458                         return sum;
459                     }
460                 });
461         }
462 
463         public ObjectiveFunctionGradient getObjectiveFunctionGradient() {
464             return new ObjectiveFunctionGradient(new MultivariateVectorFunction() {
465                     public double[] value(double[] point) {
466                         double[] r = factors.operate(point);
467                         for (int i = 0; i < r.length; ++i) {
468                             r[i] -= target[i];
469                         }
470                         double[] p = factors.transpose().operate(r);
471                         for (int i = 0; i < p.length; ++i) {
472                             p[i] *= 2;
473                         }
474                         return p;
475                     }
476                 });
477         }
