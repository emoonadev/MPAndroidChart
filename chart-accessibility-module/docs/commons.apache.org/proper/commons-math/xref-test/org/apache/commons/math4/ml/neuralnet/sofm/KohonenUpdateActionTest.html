38  /**
39   * Tests for {@link KohonenUpdateAction} class.
40   */
41  public class KohonenUpdateActionTest {
42      /*
43       * Test assumes that the network is
44       *
45       *  0-----1-----2
46       */
47      @Test
48      public void testUpdate() {
49          final FeatureInitializer init
50              = new OffsetFeatureInitializer(FeatureInitializerFactory.uniform(0, 0.1));
51          final FeatureInitializer[] initArray = { init };
52  
53          final int netSize = 3;
54          final Network net = new NeuronString(netSize, false, initArray).getNetwork();
55          final DistanceMeasure dist = new EuclideanDistance();
56          final LearningFactorFunction learning
57              = LearningFactorFunctionFactory.exponentialDecay(1, 0.1, 100);
58          final NeighbourhoodSizeFunction neighbourhood
59              = NeighbourhoodSizeFunctionFactory.exponentialDecay(3, 1, 100);
60          final UpdateAction update = new KohonenUpdateAction(dist, learning, neighbourhood);
61  
62          // The following test ensures that, after one "update",
63          // 1. when the initial learning rate equal to 1, the best matching
64          //    neuron's features are mapped to the input's features,
65          // 2. when the initial neighbourhood is larger than the network's size,
66          //    all neuron's features get closer to the input's features.
67  
68          final double[] features = new double[] { 0.3 };
69          final double[] distancesBefore = new double[netSize];
70          int count = 0;
71          for (Neuron n : net) {
72              distancesBefore[count++] = dist.compute(n.getFeatures(), features);
73          }
74          final Neuron bestBefore = MapUtils.findBest(features, net, dist);
75  
76          // Initial distance from the best match is larger than zero.
77          Assert.assertTrue(dist.compute(bestBefore.getFeatures(), features) >= 0.2);
78  
79          update.update(net, features);
80  
81          final double[] distancesAfter = new double[netSize];
82          count = 0;
83          for (Neuron n : net) {
84              distancesAfter[count++] = dist.compute(n.getFeatures(), features);
85          }
86          final Neuron bestAfter = MapUtils.findBest(features, net, dist);
87  
88          Assert.assertEquals(bestBefore, bestAfter);
89          // Distance is now zero.
90          Assert.assertEquals(0, dist.compute(bestAfter.getFeatures(), features), 1e-16);
91  
92          for (int i = 0; i < netSize; i++) {
93              // All distances have decreased.
94              Assert.assertTrue(distancesAfter[i] < distancesBefore[i]);
95          }
96      }
97  }
