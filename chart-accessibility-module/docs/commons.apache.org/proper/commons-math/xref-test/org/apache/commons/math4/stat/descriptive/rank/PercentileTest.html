195         final Percentile percentile = new Percentile(50);
196         double[] specialValues = new double[] {0d, 1d, 2d, 3d, 4d,  Double.NaN};
197         Assert.assertEquals(/*2.5d*/2d, percentile.evaluate(specialValues), 0);
198         specialValues =  new double[] {Double.NEGATIVE_INFINITY, 1d, 2d, 3d,
199                 Double.NaN, Double.POSITIVE_INFINITY};
200         Assert.assertEquals(/*2.5d*/2d, percentile.evaluate(specialValues), 0);

204         specialValues = new double[] {1d, 1d, Double.NaN,
205                 Double.NaN};

207         Assert.assertTrue(1d==percentile.evaluate(specialValues));
208         specialValues = new double[] {1d, 1d, Double.NEGATIVE_INFINITY,
209                 Double.NEGATIVE_INFINITY};
210         // Interpolation results in NEGATIVE_INFINITY + POSITIVE_INFINITY
211         Assert.assertTrue(Double.isNaN(percentile.evaluate(specialValues)));
