17  package org.apache.commons.math4.optim.linear;
18  
19  import java.util.ArrayList;
20  import java.util.Collection;
21  
22  import org.apache.commons.math4.TestUtils;
23  import org.apache.commons.math4.optim.linear.LinearConstraint;
24  import org.apache.commons.math4.optim.linear.LinearObjectiveFunction;
25  import org.apache.commons.math4.optim.linear.Relationship;
26  import org.apache.commons.math4.optim.linear.SimplexTableau;
27  import org.apache.commons.math4.optim.nonlinear.scalar.GoalType;
28  import org.junit.Assert;
29  import org.junit.Test;
30  
31  public class SimplexTableauTest {
32  
33      @Test
34      public void testInitialization() {
35          LinearObjectiveFunction f = createFunction();
36          Collection<LinearConstraint> constraints = createConstraints();
37          SimplexTableau tableau =
38              new SimplexTableau(f, constraints, GoalType.MAXIMIZE, false, 1.0e-6);
39          double[][] expectedInitialTableau = {
40                                               {-1, 0,  -1,  -1,  2, 0, 0, 0, -4},
41                                               { 0, 1, -15, -10, 25, 0, 0, 0,  0},
42                                               { 0, 0,   1,   0, -1, 1, 0, 0,  2},
43                                               { 0, 0,   0,   1, -1, 0, 1, 0,  3},
44                                               { 0, 0,   1,   1, -2, 0, 0, 1,  4}
45          };
46          assertMatrixEquals(expectedInitialTableau, tableau.getData());
47      }
48  
49      @Test
50      public void testDropPhase1Objective() {
51          LinearObjectiveFunction f = createFunction();
52          Collection<LinearConstraint> constraints = createConstraints();
53          SimplexTableau tableau =
54              new SimplexTableau(f, constraints, GoalType.MAXIMIZE, false, 1.0e-6);
55          double[][] expectedTableau = {
56                                        { 1, -15, -10, 0, 0, 0, 0},
57                                        { 0,   1,   0, 1, 0, 0, 2},
58                                        { 0,   0,   1, 0, 1, 0, 3},
59                                        { 0,   1,   1, 0, 0, 1, 4}
60          };
61          tableau.dropPhase1Objective();
62          assertMatrixEquals(expectedTableau, tableau.getData());
63      }
64  
65      @Test
66      public void testTableauWithNoArtificialVars() {
67          LinearObjectiveFunction f = new LinearObjectiveFunction(new double[] {15, 10}, 0);
68          Collection<LinearConstraint> constraints = new ArrayList<LinearConstraint>();
69          constraints.add(new LinearConstraint(new double[] {1, 0}, Relationship.LEQ, 2));
70          constraints.add(new LinearConstraint(new double[] {0, 1}, Relationship.LEQ, 3));
71          constraints.add(new LinearConstraint(new double[] {1, 1}, Relationship.LEQ, 4));
72          SimplexTableau tableau =
73              new SimplexTableau(f, constraints, GoalType.MAXIMIZE, false, 1.0e-6);
74          double[][] initialTableau = {
75                                       {1, -15, -10, 25, 0, 0, 0, 0},
76                                       {0,   1,   0, -1, 1, 0, 0, 2},
77                                       {0,   0,   1, -1, 0, 1, 0, 3},
78                                       {0,   1,   1, -2, 0, 0, 1, 4}
79          };
80          assertMatrixEquals(initialTableau, tableau.getData());
81      }
82  
83      @Test
84      public void testSerial() {
85          LinearObjectiveFunction f = createFunction();
86          Collection<LinearConstraint> constraints = createConstraints();
87          SimplexTableau tableau =
88              new SimplexTableau(f, constraints, GoalType.MAXIMIZE, false, 1.0e-6);
89          Assert.assertEquals(tableau, TestUtils.serializeAndRecover(tableau));
90      }
91  
92      private LinearObjectiveFunction createFunction() {
93          return new LinearObjectiveFunction(new double[] {15, 10}, 0);
94      }
95  
96      private Collection<LinearConstraint> createConstraints() {
97          Collection<LinearConstraint> constraints = new ArrayList<LinearConstraint>();
98          constraints.add(new LinearConstraint(new double[] {1, 0}, Relationship.LEQ, 2));
99          constraints.add(new LinearConstraint(new double[] {0, 1}, Relationship.LEQ, 3));
100         constraints.add(new LinearConstraint(new double[] {1, 1}, Relationship.EQ, 4));
101         return constraints;
102     }
103 
104     private void assertMatrixEquals(double[][] expected, double[][] result) {
105         Assert.assertEquals("Wrong number of rows.", expected.length, result.length);
106         for (int i = 0; i < expected.length; i++) {
107             Assert.assertEquals("Wrong number of columns.", expected[i].length, result[i].length);
108             for (int j = 0; j < expected[i].length; j++) {
109                 Assert.assertEquals("Wrong value at position [" + i + "," + j + "]", expected[i][j], result[i][j], 1.0e-15);
110             }
111         }
112     }
113 }
