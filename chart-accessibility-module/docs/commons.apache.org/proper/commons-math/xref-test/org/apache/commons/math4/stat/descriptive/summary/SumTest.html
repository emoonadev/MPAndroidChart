17  package org.apache.commons.math4.stat.descriptive.summary;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatistic;
20  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
21  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
22  import org.apache.commons.math4.stat.descriptive.summary.Sum;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for the {@link Sum} class.
28   */
29  public class SumTest extends StorelessUnivariateStatisticAbstractTest{
30  
31      protected Sum stat;
32  
33      /**
34       * {@inheritDoc}
35       */
36      @Override
37      public UnivariateStatistic getUnivariateStatistic() {
38          return new Sum();
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public double expectedValue() {
46          return this.sum;
47      }
48  
49      /**Expected value for  the testArray defined in UnivariateStatisticAbstractTest */
50      public double expectedWeightedValue() {
51          return this.weightedSum;
52      }
53  
54      @Test
55      public void testSpecialValues() {
56          Sum sum = new Sum();
57          Assert.assertEquals(0, sum.getResult(), 0);
58          sum.increment(1);
59          Assert.assertEquals(1, sum.getResult(), 0);
60          sum.increment(Double.POSITIVE_INFINITY);
61          Assert.assertEquals(Double.POSITIVE_INFINITY, sum.getResult(), 0);
62          sum.increment(Double.NEGATIVE_INFINITY);
63          Assert.assertTrue(Double.isNaN(sum.getResult()));
64          sum.increment(1);
65          Assert.assertTrue(Double.isNaN(sum.getResult()));
66      }
67  
68      @Test
69      public void testWeightedSum() {
70          Sum sum = new Sum();
71          Assert.assertEquals(expectedWeightedValue(),
72                              sum.evaluate(testArray, testWeightsArray, 0, testArray.length), getTolerance());
73          Assert.assertEquals(expectedValue(),
74                              sum.evaluate(testArray, unitWeightsArray, 0, testArray.length), getTolerance());
75      }
76  
77      @Override
78      protected void checkClearValue(StorelessUnivariateStatistic statistic) {
79          Assert.assertEquals(0, statistic.getResult(), 0);
80      }
81  
82  }
