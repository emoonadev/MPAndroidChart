48      /**
49       * Create a {@link LeastSquaresBuilder} from a {@link StatisticalReferenceDataset}.
50       *
51       * @param dataset the source data
52       * @return a builder for further customization.
53       */
54      public LeastSquaresBuilder builder(StatisticalReferenceDataset dataset) {
55          StatisticalReferenceDataset.LeastSquaresProblem problem
56                  = dataset.getLeastSquaresProblem();
57          final double[] start = dataset.getParameters();
58          final double[] observed = dataset.getData()[1];
59          final double[] weights = new double[observed.length];
60          Arrays.fill(weights, 1d);
61  
62          return new LeastSquaresBuilder()
63                  .model(problem.getModelFunction(), problem.getModelFunctionJacobian())
64                  .target(observed)
65                  .weight(new DiagonalMatrix(weights))
66                  .start(start);
67      }

69      @Test
70      public void testComputeResiduals() {
71          //setup
72          RealVector point = new ArrayRealVector(2);
73          Evaluation evaluation = new LeastSquaresBuilder()
74                  .target(new ArrayRealVector(new double[]{3,-1}))
75                  .model(new MultivariateJacobianFunction() {
76                      public Pair<RealVector, RealMatrix> value(RealVector point) {
77                          return new Pair<RealVector, RealMatrix>(
78                                  new ArrayRealVector(new double[]{1, 2}),
79                                  MatrixUtils.createRealIdentityMatrix(2)
80                          );
81                      }
82                  })
83                  .weight(MatrixUtils.createRealIdentityMatrix(2))
84                  .build()
85                  .evaluate(point);
86  
87          //action + verify
88          Assert.assertArrayEquals(
89                  evaluation.getResiduals().toArray(),
90                  new double[]{2, -3},
91                  Precision.EPSILON);
92      }

94      @Test
95      public void testComputeCovariance() throws IOException {
96          //setup
97          RealVector point = new ArrayRealVector(2);
98          Evaluation evaluation = new LeastSquaresBuilder()
99                  .model(new MultivariateJacobianFunction() {
100                     public Pair<RealVector, RealMatrix> value(RealVector point) {
101                         return new Pair<RealVector, RealMatrix>(
102                                 new ArrayRealVector(2),
103                                 MatrixUtils.createRealDiagonalMatrix(new double[]{1, 1e-2})
104                         );
105                     }
106                 })
107                 .weight(MatrixUtils.createRealDiagonalMatrix(new double[]{1, 1}))
108                 .target(new ArrayRealVector(2))
109                 .build()
110                 .evaluate(point);
111 
112         //action
113         TestUtils.assertEquals(
114                 "covariance",
115                 evaluation.getCovariances(FastMath.nextAfter(1e-4, 0.0)),
116                 MatrixUtils.createRealMatrix(new double[][]{{1, 0}, {0, 1e4}}),
117                 Precision.EPSILON
118         );
119 
120         //singularity fail
121         try {
122             evaluation.getCovariances(FastMath.nextAfter(1e-4, 1.0));
123             Assert.fail("Expected Exception");
124         } catch (SingularMatrixException e) {
125             //expected
126         }
127     }

129     @Test
130     public void testComputeValueAndJacobian() {
131         //setup
132         final RealVector point = new ArrayRealVector(new double[]{1, 2});
133         Evaluation evaluation = new LeastSquaresBuilder()
134                 .weight(new DiagonalMatrix(new double[]{16, 4}))
135                 .model(new MultivariateJacobianFunction() {
136                     public Pair<RealVector, RealMatrix> value(RealVector actualPoint) {
137                         //verify correct values passed in
138                         Assert.assertArrayEquals(
139                                 point.toArray(), actualPoint.toArray(), Precision.EPSILON);
140                         //return values
141                         return new Pair<RealVector, RealMatrix>(
142                                 new ArrayRealVector(new double[]{3, 4}),
143                                 MatrixUtils.createRealMatrix(new double[][]{{5, 6}, {7, 8}})
144                         );
145                     }
146                 })
147                 .target(new double[2])
148                 .build()
149                 .evaluate(point);
150 
151         //action
152         RealVector residuals = evaluation.getResiduals();
153         RealMatrix jacobian = evaluation.getJacobian();
154 
155         //verify
156         Assert.assertArrayEquals(evaluation.getPoint().toArray(), point.toArray(), 0);
157         Assert.assertArrayEquals(new double[]{-12, -8}, residuals.toArray(), Precision.EPSILON);
158         TestUtils.assertEquals(
159                 "jacobian",
160                 jacobian,
161                 MatrixUtils.createRealMatrix(new double[][]{{20, 24},{14, 16}}),
162                 Precision.EPSILON);
163     }

165     @Test
166     public void testComputeCost() throws IOException {
167         final StatisticalReferenceDataset dataset
168             = StatisticalReferenceDatasetFactory.createKirby2();
169 
170         final LeastSquaresProblem lsp = builder(dataset).build();
171 
172         final double expected = dataset.getResidualSumOfSquares();
173         final double cost = lsp.evaluate(lsp.getStart()).getCost();
174         final double actual = cost * cost;
175         Assert.assertEquals(dataset.getName(), expected, actual, 1e-11 * expected);
176     }

191     @Test
192     public void testComputeSigma() throws IOException {
193         final StatisticalReferenceDataset dataset
194             = StatisticalReferenceDatasetFactory.createKirby2();
195 
196         final LeastSquaresProblem lsp = builder(dataset).build();
197 
198         final double[] expected = dataset.getParametersStandardDeviations();
199 
200         final Evaluation evaluation = lsp.evaluate(lsp.getStart());
201         final double cost = evaluation.getCost();
202         final RealVector sig = evaluation.getSigma(1e-14);
203         final int dof = lsp.getObservationSize() - lsp.getParameterSize();
204         for (int i = 0; i < sig.getDimension(); i++) {
205             final double actual = FastMath.sqrt(cost * cost / dof) * sig.getEntry(i);
206             Assert.assertEquals(dataset.getName() + ", parameter #" + i,
207                                 expected[i], actual, 1e-6 * expected[i]);
208         }
209     }

213         //setup

215                 = StatisticalReferenceDatasetFactory.createKirby2();
216         LeastSquaresProblem lsp = builder(dataset).build();

222         //verify
223         Assert.assertNotSame(point, evaluation.getPoint());

228     @Test
229     public void testLazyEvaluation() {
230         final RealVector dummy = new ArrayRealVector(new double[] { 0 });
231 
232         final LeastSquaresProblem p
233             = LeastSquaresFactory.create(LeastSquaresFactory.model(dummyModel(), dummyJacobian()),
234                                          dummy, dummy, null, null, 0, 0, true, null);
235 
236         // Should not throw because actual evaluation is deferred.
237         final Evaluation eval = p.evaluate(dummy);
238 
239         try {
240             eval.getResiduals();
241             Assert.fail("Exception expected");
242         } catch (RuntimeException e) {
243             // Expecting exception.
244             Assert.assertEquals("dummyModel", e.getMessage());
245         }
246 
247         try {
248             eval.getJacobian();
249             Assert.fail("Exception expected");
250         } catch (RuntimeException e) {
251             // Expecting exception.
252             Assert.assertEquals("dummyJacobian", e.getMessage());
253         }
254     }

257     @Test
258     public void testLazyEvaluationPrecondition() {
259         final RealVector dummy = new ArrayRealVector(new double[] { 0 });
260 
261         // "ValueAndJacobianFunction" is required but we implement only
262         // "MultivariateJacobianFunction".
263         final MultivariateJacobianFunction m1 = new MultivariateJacobianFunction() {
264                 public Pair<RealVector, RealMatrix> value(RealVector notUsed) {
265                     return new Pair<RealVector, RealMatrix>(null, null);
266                 }
267             };
268 
269         try {
270             // Should throw.
271             LeastSquaresFactory.create(m1, dummy, dummy, null, null, 0, 0, true, null);
272             Assert.fail("Expecting MathIllegalStateException");
273         } catch (MathIllegalStateException e) {
274             // Expected.
275         }
276 
277         final MultivariateJacobianFunction m2 = new ValueAndJacobianFunction() {
278                 public Pair<RealVector, RealMatrix> value(RealVector notUsed) {
279                     return new Pair<RealVector, RealMatrix>(null, null);
280                 }
281                 public RealVector computeValue(final double[] params) {
282                     return null;
283                 }
284                 public RealMatrix computeJacobian(final double[] params) {
285                     return null;
286                 }
287             };
288 
289         // Should pass.
290         LeastSquaresFactory.create(m2, dummy, dummy, null, null, 0, 0, true, null);
291     }

293     @Test
294     public void testDirectEvaluation() {
295         final RealVector dummy = new ArrayRealVector(new double[] { 0 });
296 
297         final LeastSquaresProblem p
298             = LeastSquaresFactory.create(LeastSquaresFactory.model(dummyModel(), dummyJacobian()),
299                                          dummy, dummy, null, null, 0, 0, false, null);
300 
301         try {
302             // Should throw.
303             p.evaluate(dummy);
304             Assert.fail("Exception expected");
305         } catch (RuntimeException e) {
306             // Expecting exception.
307             // Whether it is model or Jacobian that caused it is not significant.
308             final String msg = e.getMessage();
309             Assert.assertTrue(msg.equals("dummyModel") ||
310                               msg.equals("dummyJacobian"));
311         }
312     }
