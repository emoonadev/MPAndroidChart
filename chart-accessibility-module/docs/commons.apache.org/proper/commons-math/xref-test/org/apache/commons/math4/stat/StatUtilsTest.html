50      public void testStats() {
51          double[] values = new double[] { ONE, TWO, TWO, THREE };
52          Assert.assertEquals("sum", SUM, StatUtils.sum(values), TOLERANCE);
53          Assert.assertEquals("sumsq", SUMSQ, StatUtils.sumSq(values), TOLERANCE);
54          Assert.assertEquals("var", VAR, StatUtils.variance(values), TOLERANCE);
55          Assert.assertEquals("var with mean", VAR, StatUtils.variance(values, MEAN), TOLERANCE);
56          Assert.assertEquals("mean", MEAN, StatUtils.mean(values), TOLERANCE);
57          Assert.assertEquals("min", MIN, StatUtils.min(values), TOLERANCE);
58          Assert.assertEquals("max", MAX, StatUtils.max(values), TOLERANCE);

61      @Test
62      public void testN0andN1Conditions() {
63          double[] values = new double[0];
64  
65          Assert.assertTrue(
66              "Mean of n = 0 set should be NaN",
67              Double.isNaN(StatUtils.mean(values)));
68          Assert.assertTrue(
69              "Variance of n = 0 set should be NaN",
70              Double.isNaN(StatUtils.variance(values)));
71  
72          values = new double[] { ONE };
73  
74          Assert.assertTrue(
75              "Mean of n = 1 set should be value of single item n1",
76              StatUtils.mean(values) == ONE);
77          Assert.assertTrue(
78              "Variance of n = 1 set should be zero",
79              StatUtils.variance(values) == 0);

419     @Test
420     public void testDifferenceStats() {
421         double sample1[] = {1d, 2d, 3d, 4d};
422         double sample2[] = {1d, 3d, 4d, 2d};
423         double diff[] = {0d, -1d, -1d, 2d};
424         double small[] = {1d, 4d};
425         double meanDifference = StatUtils.meanDifference(sample1, sample2);
426         Assert.assertEquals(StatUtils.sumDifference(sample1, sample2), StatUtils.sum(diff), TOLERANCE);
427         Assert.assertEquals(meanDifference, StatUtils.mean(diff), TOLERANCE);
428         Assert.assertEquals(StatUtils.varianceDifference(sample1, sample2, meanDifference),
429                 StatUtils.variance(diff), TOLERANCE);
430         try {
431             StatUtils.meanDifference(sample1, small);
432             Assert.fail("Expecting MathIllegalArgumentException");
433         } catch (MathIllegalArgumentException ex) {
434             // expected
435         }
436         try {
437             StatUtils.varianceDifference(sample1, small, meanDifference);
438             Assert.fail("Expecting MathIllegalArgumentException");
439         } catch (MathIllegalArgumentException ex) {
440             // expected
441         }
442         try {
443             double[] single = {1.0};
444             StatUtils.varianceDifference(single, single, meanDifference);
445             Assert.fail("Expecting MathIllegalArgumentException");
446         } catch (MathIllegalArgumentException ex) {
447             // expected
448         }
449     }

483     /**
484      * Run with 77 random values, assuming that the outcome has a mean of 0 and a standard deviation of 1 with a
485      * precision of 1E-10.
486      */
487 
488     @Test
489     public void testNormalize2() {
490         // create an sample with 77 values
491         int length = 77;
492         double sample[] = new double[length];
493         for (int i = 0; i < length; i++) {
494             sample[i] = FastMath.random();
495         }
496         // normalize this sample
497         double standardizedSample[] = StatUtils.normalize(sample);
498 
499         DescriptiveStatistics stats = new DescriptiveStatistics();
500         // Add the data from the array
501         for (int i = 0; i < length; i++) {
502             stats.addValue(standardizedSample[i]);
503         }
504         // the calculations do have a limited precision
505         double distance = 1E-10;
506         // check the mean an standard deviation
507         Assert.assertEquals(0.0, stats.getMean(), distance);
508         Assert.assertEquals(1.0, stats.getStandardDeviation(), distance);
509 
510     }

512     @Test
513     public void testMode() {
514         final double[] singleMode = {0, 1, 0, 2, 7, 11, 12};
515         final double[] modeSingle = StatUtils.mode(singleMode);
516         Assert.assertEquals(0, modeSingle[0], Double.MIN_VALUE);
517         Assert.assertEquals(1, modeSingle.length);
518 
519         final double[] twoMode = {0, 1, 2, 0, 2, 3, 7, 11};
520         final double[] modeDouble = StatUtils.mode(twoMode);
521         Assert.assertEquals(0, modeDouble[0], Double.MIN_VALUE);
522         Assert.assertEquals(2, modeDouble[1], Double.MIN_VALUE);
523         Assert.assertEquals(2, modeDouble.length);
524 
525         final double[] nanInfested = {0, 0, 0, Double.NaN, Double.NaN, Double.NaN, Double.NaN, 2, 2, 2, 3, 5};
526         final double[] modeNan = StatUtils.mode(nanInfested);
527         Assert.assertEquals(0, modeNan[0], Double.MIN_VALUE);
528         Assert.assertEquals(2, modeNan[1], Double.MIN_VALUE);
529         Assert.assertEquals(2, modeNan.length);
530 
531         final double[] infInfested = {0, 0, Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY,
532             Double.NEGATIVE_INFINITY, Double.NEGATIVE_INFINITY, 2, 2, 3, 5};
533         final double[] modeInf = StatUtils.mode(infInfested);
534         Assert.assertEquals(Double.NEGATIVE_INFINITY, modeInf[0], Double.MIN_VALUE);
535         Assert.assertEquals(0, modeInf[1], Double.MIN_VALUE);
536         Assert.assertEquals(2, modeInf[2], Double.MIN_VALUE);
537         Assert.assertEquals(Double.POSITIVE_INFINITY, modeInf[3], Double.MIN_VALUE);
538         Assert.assertEquals(4, modeInf.length);
539 
540         final double[] noData = {};
541         final double[] modeNodata = StatUtils.mode(noData);
542         Assert.assertEquals(0, modeNodata.length);
543 
544         final double[] nansOnly = {Double.NaN, Double.NaN};
545         final double[] modeNansOnly = StatUtils.mode(nansOnly);
546         Assert.assertEquals(0, modeNansOnly.length);
547 
548         final double[] nullArray = null;
549         try {
550             StatUtils.mode(nullArray);

552         } catch (NullArgumentException ex) {
553             // Expected
