18  package org.apache.commons.math4.stat.descriptive.moment;
19  
20  import org.apache.commons.math4.TestUtils;
21  import org.apache.commons.math4.exception.DimensionMismatchException;
22  import org.apache.commons.math4.stat.descriptive.moment.VectorialMean;
23  import org.junit.Test;
24  import org.junit.Assert;
25  
26  public class VectorialMeanTest {
27      private double[][] points;
28  
29      public VectorialMeanTest() {
30          points = new double[][] {
31              { 1.2, 2.3,  4.5},
32              {-0.7, 2.3,  5.0},
33              { 3.1, 0.0, -3.1},
34              { 6.0, 1.2,  4.2},
35              {-0.7, 2.3,  5.0}
36          };
37      }
38  
39      @Test
40      public void testMismatch() {
41          try {
42              new VectorialMean(8).increment(new double[5]);
43              Assert.fail("an exception should have been thrown");
44          } catch (DimensionMismatchException dme) {
45              Assert.assertEquals(5, dme.getArgument());
46              Assert.assertEquals(8, dme.getDimension());
47          }
48      }
49  
50      @Test
51      public void testSimplistic() {
52          VectorialMean stat = new VectorialMean(2);
53          stat.increment(new double[] {-1.0,  1.0});
54          stat.increment(new double[] { 1.0, -1.0});
55          double[] mean = stat.getResult();
56          Assert.assertEquals(0.0, mean[0], 1.0e-12);
57          Assert.assertEquals(0.0, mean[1], 1.0e-12);
58      }
59  
60      @Test
61      public void testBasicStats() {
62  
63          VectorialMean stat = new VectorialMean(points[0].length);
64          for (int i = 0; i < points.length; ++i) {
65              stat.increment(points[i]);
66          }
67  
68          Assert.assertEquals(points.length, stat.getN());
69  
70          double[] mean = stat.getResult();
71          double[]   refMean = new double[] { 1.78, 1.62,  3.12};
72  
73          for (int i = 0; i < mean.length; ++i) {
74              Assert.assertEquals(refMean[i], mean[i], 1.0e-12);
75          }
76  
77      }
78  
79      @Test
80      public void testSerial() {
81          VectorialMean stat = new VectorialMean(points[0].length);
82          for (int i = 0; i < points.length; ++i) {
83              stat.increment(points[i]);
84          }
85          Assert.assertEquals(stat, TestUtils.serializeAndRecover(stat));
86      }
87  }
