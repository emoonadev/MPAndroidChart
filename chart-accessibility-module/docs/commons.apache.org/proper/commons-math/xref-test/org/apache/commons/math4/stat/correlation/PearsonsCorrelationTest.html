170     /**
171      * Test p-value near 0. JIRA: MATH-371
172      */
173     @Test
174     public void testPValueNearZero() {
175         /*
176          * Create a dataset that has r -> 1, p -> 0 as dimension increases.
177          * Prior to the fix for MATH-371, p vanished for dimension >= 14.
178          * Post fix, p-values diminish smoothly, vanishing at dimension = 127.
179          * Tested value is ~1E-303.
180          */
181         int dimension = 120;
182         double[][] data = new double[dimension][2];
183         for (int i = 0; i < dimension; i++) {
184             data[i][0] = i;
185             data[i][1] = i + 1/((double)i + 1);
186         }
187         PearsonsCorrelation corrInstance = new PearsonsCorrelation(data);
188         Assert.assertTrue(corrInstance.getCorrelationPValues().getEntry(0, 1) > 0);
189     }

227     /**
228      * Verify that direct t-tests using standard error estimates are consistent
229      * with reported p-values
230      */
231     @Test
232     public void testStdErrorConsistency() {
233         TDistribution tDistribution = new TDistribution(45);
234         RealMatrix matrix = createRealMatrix(swissData, 47, 5);
235         PearsonsCorrelation corrInstance = new PearsonsCorrelation(matrix);
236         RealMatrix rValues = corrInstance.getCorrelationMatrix();
237         RealMatrix pValues = corrInstance.getCorrelationPValues();
238         RealMatrix stdErrors = corrInstance.getCorrelationStandardErrors();
239         for (int i = 0; i < 5; i++) {
240             for (int j = 0; j < i; j++) {
241                 double t = FastMath.abs(rValues.getEntry(i, j)) / stdErrors.getEntry(i, j);
242                 double p = 2 * (1 - tDistribution.cumulativeProbability(t));
243                 Assert.assertEquals(p, pValues.getEntry(i, j), 10E-15);
244             }
245         }
246     }

248     /**
249      * Verify that creating correlation from covariance gives same results as
250      * direct computation from the original matrix
251      */
252     @Test
253     public void testCovarianceConsistency() {
254         RealMatrix matrix = createRealMatrix(longleyData, 16, 7);
255         PearsonsCorrelation corrInstance = new PearsonsCorrelation(matrix);
256         Covariance covInstance = new Covariance(matrix);
257         PearsonsCorrelation corrFromCovInstance = new PearsonsCorrelation(covInstance);
258         TestUtils.assertEquals("correlation values", corrInstance.getCorrelationMatrix(),
259                 corrFromCovInstance.getCorrelationMatrix(), 10E-15);
260         TestUtils.assertEquals("p values", corrInstance.getCorrelationPValues(),
261                 corrFromCovInstance.getCorrelationPValues(), 10E-15);
262         TestUtils.assertEquals("standard errors", corrInstance.getCorrelationStandardErrors(),
263                 corrFromCovInstance.getCorrelationStandardErrors(), 10E-15);
264 
265         PearsonsCorrelation corrFromCovInstance2 =
266             new PearsonsCorrelation(covInstance.getCovarianceMatrix(), 16);
267         TestUtils.assertEquals("correlation values", corrInstance.getCorrelationMatrix(),
268                 corrFromCovInstance2.getCorrelationMatrix(), 10E-15);
269         TestUtils.assertEquals("p values", corrInstance.getCorrelationPValues(),
270                 corrFromCovInstance2.getCorrelationPValues(), 10E-15);
271         TestUtils.assertEquals("standard errors", corrInstance.getCorrelationStandardErrors(),
272                 corrFromCovInstance2.getCorrelationStandardErrors(), 10E-15);
273     }

277     public void testConsistency() {
278         RealMatrix matrix = createRealMatrix(longleyData, 16, 7);
279         PearsonsCorrelation corrInstance = new PearsonsCorrelation(matrix);
280         double[][] data = matrix.getData();
281         double[] x = matrix.getColumn(0);
282         double[] y = matrix.getColumn(1);
283         Assert.assertEquals(new PearsonsCorrelation().correlation(x, y),
284                 corrInstance.getCorrelationMatrix().getEntry(0, 1), Double.MIN_VALUE);
285         TestUtils.assertEquals("Correlation matrix", corrInstance.getCorrelationMatrix(),
286                 new PearsonsCorrelation().computeCorrelationMatrix(data), Double.MIN_VALUE);
287     }

289     protected RealMatrix createRealMatrix(double[] data, int nRows, int nCols) {
290         double[][] matrixData = new double[nRows][nCols];
291         int ptr = 0;
292         for (int i = 0; i < nRows; i++) {
293             System.arraycopy(data, ptr, matrixData[i], 0, nCols);
294             ptr += nCols;
295         }
296         return new BlockRealMatrix(matrixData);
297     }
298 
299     protected RealMatrix createLowerTriangularRealMatrix(double[] data, int dimension) {
300         int ptr = 0;
301         RealMatrix result = new BlockRealMatrix(dimension, dimension);
302         for (int i = 1; i < dimension; i++) {
303             for (int j = 0; j < i; j++) {
304                 result.setEntry(i, j, data[ptr]);
305                 ptr++;
306             }
307         }
308         return result;
309     }
310 
311     protected void fillUpper(RealMatrix matrix, double diagonalValue) {
312         int dimension = matrix.getColumnDimension();
313         for (int i = 0; i < dimension; i++) {
314             matrix.setEntry(i, i, diagonalValue);
315             for (int j = i+1; j < dimension; j++) {
316                 matrix.setEntry(i, j, matrix.getEntry(j, i));
317             }
318         }
