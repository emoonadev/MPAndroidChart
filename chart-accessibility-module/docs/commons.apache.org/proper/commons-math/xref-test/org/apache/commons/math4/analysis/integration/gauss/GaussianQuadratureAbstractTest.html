17  package org.apache.commons.math4.analysis.integration.gauss;
18  
19  import org.apache.commons.math4.analysis.function.Power;
20  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegrator;
21  import org.apache.commons.math4.util.FastMath;
22  import org.junit.Test;
23  import org.junit.Assert;
24  
25  /**
26   * Base class for standard testing of Gaussian quadrature rules,
27   * which are exact for polynomials up to a certain degree. In this test, each
28   * monomial in turn is tested against the specified quadrature rule.
29   *
30   */
31  public abstract class GaussianQuadratureAbstractTest {
32      /**
33       * The maximum absolute error (for zero testing).
34       */
35      private final double eps;
36      /**
37       * The maximum relative error (in ulps).
38       */
39      private final double numUlps;
40      /**
41       * The quadrature rule under test.
42       */
43      private final GaussIntegrator integrator;
44      /**
45       * Maximum degree of monomials to be tested.
46       */
47      private final int maxDegree;
48  
49      /**
50       * Creates a new instance of this abstract test with the specified
51       * quadrature rule.
52       * If the expected value is non-zero, equality of actual and expected values
53       * is checked in the relative sense <center>
54       * |x<sub>act</sub>&nbsp;-&nbsp;x<sub>exp</sub>|&nbsp;&le;&nbsp; n&nbsp;
55       * <code>Math.ulp(</code>x<sub>exp</sub><code>)</code>, </center> where n is
56       * the maximum relative error (in ulps). If the expected value is zero, the
57       * test checks that <center> |x<sub>act</sub>|&nbsp;&le;&nbsp;&epsilon;,
58       * </center> where &epsilon; is the maximum absolute error.
59       *
60       * @param integrator Quadrature rule under test.
61       * @param maxDegree Maximum degree of monomials to be tested.
62       * @param eps &epsilon;.
63       * @param numUlps Value of the maximum relative error (in ulps).
64       */
65      public GaussianQuadratureAbstractTest(GaussIntegrator integrator,
66                                            int maxDegree,
67                                            double eps,
68                                            double numUlps) {
69          this.integrator = integrator;
70          this.maxDegree = maxDegree;
71          this.eps = eps;
72          this.numUlps = numUlps;
73      }
74  
75      /**
76       * Returns the expected value of the integral of the specified monomial.
77       * The integration is carried out on the natural interval of the quadrature
78       * rule under test.
79       *
80       * @param n Degree of the monomial.
81       * @return the expected value of the integral of x<sup>n</sup>.
82       */
83      public abstract double getExpectedValue(final int n);
84  
85      /**
86       * Checks that the value of the integral of each monomial
87       *   <code>x<sup>0</sup>, ... , x<sup>p</sup></code>
88       * returned by the quadrature rule under test conforms with the expected
89       * value.
90       * Here {@code p} denotes the degree of the highest polynomial for which
91       * exactness is to be expected.
92       */
93      @Test
94      public void testAllMonomials() {
95          for (int n = 0; n <= maxDegree; n++) {
96              final double expected = getExpectedValue(n);
97  
98              final Power monomial = new Power(n);
99              final double actual = integrator.integrate(monomial);
100 
101             // System.out.println(n + "/" + maxDegree + " " + integrator.getNumberOfPoints()
102             //                    + " " + expected + " " + actual + " " + Math.ulp(expected));
103             if (expected == 0) {
104                 Assert.assertEquals("while integrating monomial x**" + n +
105                                     " with a " +
106                                     integrator.getNumberOfPoints() + "-point quadrature rule",
107                                     expected, actual, eps);
108             } else {
109                 double err = FastMath.abs(actual - expected) / Math.ulp(expected);
110                 Assert.assertEquals("while integrating monomial x**" + n + " with a " +
111                                     + integrator.getNumberOfPoints() + "-point quadrature rule, " +
112                                     " error was " + err + " ulps",
113                                     expected, actual, Math.ulp(expected) * numUlps);
114             }
115         }
116     }
117 }
