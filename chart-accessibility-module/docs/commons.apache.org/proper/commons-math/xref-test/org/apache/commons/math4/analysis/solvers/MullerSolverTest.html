31  /**
32   * Test case for {@link MullerSolver Muller} solver.
33   * <p>
34   * Muller's method converges almost quadratically near roots, but it can
35   * be very slow in regions far away from zeros. Test runs show that for
36   * reasonably good initial values, for a default absolute accuracy of 1E-6,
37   * it generally takes 5 to 10 iterations for the solver to converge.
38   * <p>
39   * Tests for the exponential function illustrate the situations where
40   * Muller solver performs poorly.
41   *
42   */
43  public final class MullerSolverTest {
44      /**
45       * Test of solver for the sine function.
46       */
47      @Test
48      public void testSinFunction() {
49          UnivariateFunction f = new Sin();
50          UnivariateSolver solver = new MullerSolver();
51          double min, max, expected, result, tolerance;
52  
53          min = 3.0; max = 4.0; expected = FastMath.PI;
54          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
55                      FastMath.abs(expected * solver.getRelativeAccuracy()));
56          result = solver.solve(100, f, min, max);
57          Assert.assertEquals(expected, result, tolerance);
58  
59          min = -1.0; max = 1.5; expected = 0.0;
60          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
61                      FastMath.abs(expected * solver.getRelativeAccuracy()));
62          result = solver.solve(100, f, min, max);
63          Assert.assertEquals(expected, result, tolerance);
64      }

66      /**
67       * Test of solver for the quintic function.
68       */
69      @Test
70      public void testQuinticFunction() {
71          UnivariateFunction f = new QuinticFunction();
72          UnivariateSolver solver = new MullerSolver();
73          double min, max, expected, result, tolerance;
74  
75          min = -0.4; max = 0.2; expected = 0.0;
76          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
77                      FastMath.abs(expected * solver.getRelativeAccuracy()));
78          result = solver.solve(100, f, min, max);
79          Assert.assertEquals(expected, result, tolerance);
80  
81          min = 0.75; max = 1.5; expected = 1.0;
82          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
83                      FastMath.abs(expected * solver.getRelativeAccuracy()));
84          result = solver.solve(100, f, min, max);
85          Assert.assertEquals(expected, result, tolerance);
86  
87          min = -0.9; max = -0.2; expected = -0.5;
88          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
89                      FastMath.abs(expected * solver.getRelativeAccuracy()));
90          result = solver.solve(100, f, min, max);
91          Assert.assertEquals(expected, result, tolerance);
92      }

94      /**
95       * Test of solver for the exponential function.
96       * <p>
97       * It takes 10 to 15 iterations for the last two tests to converge.
98       * In fact, if not for the bisection alternative, the solver would
99       * exceed the default maximal iteration of 100.
100      */
101     @Test
102     public void testExpm1Function() {
103         UnivariateFunction f = new Expm1();
104         UnivariateSolver solver = new MullerSolver();
105         double min, max, expected, result, tolerance;
106 
107         min = -1.0; max = 2.0; expected = 0.0;
108         tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
109                     FastMath.abs(expected * solver.getRelativeAccuracy()));
110         result = solver.solve(100, f, min, max);
111         Assert.assertEquals(expected, result, tolerance);
112 
113         min = -20.0; max = 10.0; expected = 0.0;
114         tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
115                     FastMath.abs(expected * solver.getRelativeAccuracy()));
116         result = solver.solve(100, f, min, max);
117         Assert.assertEquals(expected, result, tolerance);
118 
119         min = -50.0; max = 100.0; expected = 0.0;
120         tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
121                     FastMath.abs(expected * solver.getRelativeAccuracy()));
122         result = solver.solve(100, f, min, max);
123         Assert.assertEquals(expected, result, tolerance);
124     }

126     /**
127      * Test of parameters for the solver.
128      */
129     @Test
130     public void testParameters() {
131         UnivariateFunction f = new Sin();
132         UnivariateSolver solver = new MullerSolver();
133 
134         try {
135             // bad interval
136             double root = solver.solve(100, f, 1, -1);
137             System.out.println("root=" + root);
138             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
139         } catch (NumberIsTooLargeException ex) {
140             // expected
141         }
142         try {
143             // no bracketing
144             solver.solve(100, f, 2, 3);
145             Assert.fail("Expecting NoBracketingException - no bracketing");
146         } catch (NoBracketingException ex) {
147             // expected
148         }
149     }
