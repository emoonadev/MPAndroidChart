17  package org.apache.commons.math4.stat.descriptive.summary;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatistic;
20  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
21  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
22  import org.apache.commons.math4.stat.descriptive.summary.Product;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for the {@link UnivariateStatistic} class.
28   */
29  public class ProductTest extends StorelessUnivariateStatisticAbstractTest{
30  
31      protected Product stat;
32  
33      /**
34       * {@inheritDoc}
35       */
36      @Override
37      public UnivariateStatistic getUnivariateStatistic() {
38          return new Product();
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public double getTolerance() {
46          return 10E8;    //sic -- big absolute error due to only 15 digits of accuracy in double
47      }
48  
49      /**
50       * {@inheritDoc}
51       */
52      @Override
53      public double expectedValue() {
54          return this.product;
55      }
56  
57      /** Expected value for  the testArray defined in UnivariateStatisticAbstractTest */
58      public double expectedWeightedValue() {
59          return this.weightedProduct;
60      }
61  
62      @Test
63      public void testSpecialValues() {
64          Product product = new Product();
65          Assert.assertEquals(1, product.getResult(), 0);
66          product.increment(1);
67          Assert.assertEquals(1, product.getResult(), 0);
68          product.increment(Double.POSITIVE_INFINITY);
69          Assert.assertEquals(Double.POSITIVE_INFINITY, product.getResult(), 0);
70          product.increment(Double.NEGATIVE_INFINITY);
71          Assert.assertEquals(Double.NEGATIVE_INFINITY, product.getResult(), 0);
72          product.increment(Double.NaN);
73          Assert.assertTrue(Double.isNaN(product.getResult()));
74          product.increment(1);
75          Assert.assertTrue(Double.isNaN(product.getResult()));
76      }
77  
78      @Test
79      public void testWeightedProduct() {
80          Product product = new Product();
81          Assert.assertEquals(expectedWeightedValue(),
82                              product.evaluate(testArray, testWeightsArray, 0, testArray.length),getTolerance());
83          Assert.assertEquals(expectedValue(),
84                              product.evaluate(testArray, unitWeightsArray, 0, testArray.length), getTolerance());
85      }
86  
87      @Override
88      protected void checkClearValue(StorelessUnivariateStatistic statistic){
89          Assert.assertEquals(1, statistic.getResult(), 0);
90      }
91  
92  }
