41      @Test
42      public void testChiSquare() {
43  
44          // Target values computed using R version 1.8.1
45          // Some assembly required ;-)
46          //      Use sum((obs - exp)^2/exp) for the chi-square statistic and
47          //      1 - pchisq(sum((obs - exp)^2/exp), length(obs) - 1) for the p-value
48  
49          long[] observed = {10, 9, 11};
50          double[] expected = {10, 10, 10};
51          Assert.assertEquals("chi-square statistic", 0.2,  InferenceTestUtils.chiSquare(expected, observed), 10E-12);
52          Assert.assertEquals("chi-square p-value", 0.904837418036, InferenceTestUtils.chiSquareTest(expected, observed), 1E-10);
53  
54          long[] observed1 = { 500, 623, 72, 70, 31 };
55          double[] expected1 = { 485, 541, 82, 61, 37 };
56          Assert.assertEquals( "chi-square test statistic", 9.023307936427388, InferenceTestUtils.chiSquare(expected1, observed1), 1E-10);
57          Assert.assertEquals("chi-square p-value", 0.06051952647453607, InferenceTestUtils.chiSquareTest(expected1, observed1), 1E-9);
58          Assert.assertTrue("chi-square test reject", InferenceTestUtils.chiSquareTest(expected1, observed1, 0.07));
59          Assert.assertTrue("chi-square test accept", !InferenceTestUtils.chiSquareTest(expected1, observed1, 0.05));
60  
61          try {
62              InferenceTestUtils.chiSquareTest(expected1, observed1, 95);
63              Assert.fail("alpha out of range, OutOfRangeException expected");
64          } catch (OutOfRangeException ex) {
65              // expected
66          }
67  
68          long[] tooShortObs = { 0 };
69          double[] tooShortEx = { 1 };
70          try {
71              InferenceTestUtils.chiSquare(tooShortEx, tooShortObs);
72              Assert.fail("arguments too short, DimensionMismatchException expected");
73          } catch (DimensionMismatchException ex) {
74              // expected
75          }
76  
77          // unmatched arrays
78          long[] unMatchedObs = { 0, 1, 2, 3 };
79          double[] unMatchedEx = { 1, 1, 2 };
80          try {
81              InferenceTestUtils.chiSquare(unMatchedEx, unMatchedObs);
82              Assert.fail("arrays have different lengths, DimensionMismatchException expected");
83          } catch (DimensionMismatchException ex) {
84              // expected
85          }
86  
87          // 0 expected count
88          expected[0] = 0;
89          try {
90              InferenceTestUtils.chiSquareTest(expected, observed, .01);
91              Assert.fail("bad expected count, NotStrictlyPositiveException expected");
92          } catch (NotStrictlyPositiveException ex) {
93              // expected
94          }
95  
96          // negative observed count
97          expected[0] = 1;
98          observed[0] = -1;
99          try {
100             InferenceTestUtils.chiSquareTest(expected, observed, .01);
101             Assert.fail("bad expected count, NotPositiveException expected");
102         } catch (NotPositiveException ex) {
103             // expected
104         }
