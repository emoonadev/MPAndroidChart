157     public void testHole() {
158         Vector2D[][] vertices = new Vector2D[][] {
159             new Vector2D[] {
160                 new Vector2D(0.0, 0.0),
161                 new Vector2D(3.0, 0.0),
162                 new Vector2D(3.0, 3.0),
163                 new Vector2D(0.0, 3.0)
164             }, new Vector2D[] {
165                 new Vector2D(1.0, 2.0),
166                 new Vector2D(2.0, 2.0),
167                 new Vector2D(2.0, 1.0),
168                 new Vector2D(1.0, 1.0)
169             }
170         };
171         PolygonsSet set = buildSet(vertices);
172         checkPoints(Region.Location.INSIDE, set, new Vector2D[] {
173             new Vector2D(0.5, 0.5),
174             new Vector2D(1.5, 0.5),
175             new Vector2D(2.5, 0.5),
176             new Vector2D(0.5, 1.5),
177             new Vector2D(2.5, 1.5),
178             new Vector2D(0.5, 2.5),
179             new Vector2D(1.5, 2.5),
180             new Vector2D(2.5, 2.5),
181             new Vector2D(0.5, 1.0)
182         });
183         checkPoints(Region.Location.OUTSIDE, set, new Vector2D[] {
184             new Vector2D(1.5, 1.5),
185             new Vector2D(3.5, 1.0),
186             new Vector2D(4.0, 1.5),
187             new Vector2D(6.0, 6.0)
188         });
189         checkPoints(Region.Location.BOUNDARY, set, new Vector2D[] {
190             new Vector2D(1.0, 1.0),
191             new Vector2D(1.5, 0.0),
192             new Vector2D(1.5, 1.0),
193             new Vector2D(1.5, 2.0),
194             new Vector2D(1.5, 3.0),
195             new Vector2D(3.0, 3.0)
196         });
197         checkVertices(set.getVertices(), vertices);
198 
199         for (double x = -0.999; x < 3.999; x += 0.11) {
200             Vector2D v = new Vector2D(x, x + 0.5);
201             BoundaryProjection<Euclidean2D> projection = set.projectToBoundary(v);
202             Assert.assertTrue(projection.getOriginal() == v);
203             Vector2D p = (Vector2D) projection.getProjected();
204             if (x < -0.5) {
205                 Assert.assertEquals(0.0,      p.getX(), 1.0e-10);
206                 Assert.assertEquals(0.0,      p.getY(), 1.0e-10);
207                 Assert.assertEquals(+v.distance(Vector2D.ZERO), projection.getOffset(), 1.0e-10);
208             } else if (x < 0.5) {
209                 Assert.assertEquals(0.0,      p.getX(), 1.0e-10);
210                 Assert.assertEquals(v.getY(), p.getY(), 1.0e-10);
211                 Assert.assertEquals(-v.getX(), projection.getOffset(), 1.0e-10);
212             } else if (x < 1.25) {
213                 Assert.assertEquals(1.0,      p.getX(), 1.0e-10);
214                 Assert.assertEquals(v.getY(), p.getY(), 1.0e-10);
215                 Assert.assertEquals(v.getX() - 1.0, projection.getOffset(), 1.0e-10);
216             } else if (x < 2.0) {
217                 Assert.assertEquals(v.getX(), p.getX(), 1.0e-10);
218                 Assert.assertEquals(2.0,      p.getY(), 1.0e-10);
219                 Assert.assertEquals(2.0 - v.getY(), projection.getOffset(), 1.0e-10);
220             } else if (x < 3.0) {
221                 Assert.assertEquals(v.getX(), p.getX(), 1.0e-10);
222                 Assert.assertEquals(3.0,      p.getY(), 1.0e-10);
223                 Assert.assertEquals(v.getY() - 3.0, projection.getOffset(), 1.0e-10);
224             } else {
225                 Assert.assertEquals(3.0,      p.getX(), 1.0e-10);
226                 Assert.assertEquals(3.0,      p.getY(), 1.0e-10);
227                 Assert.assertEquals(+v.distance(new Vector2D(3, 3)), projection.getOffset(), 1.0e-10);
228             }
229 
230         }

301     @Test
302     public void testLineIntersection() {
303         Vector2D[][] vertices = new Vector2D[][] {
304             new Vector2D[] {
305                 new Vector2D( 0.0,  0.0),
306                 new Vector2D( 2.0,  0.0),
307                 new Vector2D( 2.0,  1.0),
308                 new Vector2D( 3.0,  1.0),
309                 new Vector2D( 3.0,  3.0),
310                 new Vector2D( 1.0,  3.0),
311                 new Vector2D( 1.0,  2.0),
312                 new Vector2D( 0.0,  2.0)
313             }
314         };
315         PolygonsSet set = buildSet(vertices);
316 
317         Line l1 = new Line(new Vector2D(-1.5, 0.0), FastMath.PI / 4, 1.0e-10);
318         SubLine s1 = (SubLine) set.intersection(l1.wholeHyperplane());
319         List<Interval> i1 = ((IntervalsSet) s1.getRemainingRegion()).asList();
320         Assert.assertEquals(2, i1.size());
321         Interval v10 = i1.get(0);
322         Vector2D p10Lower = l1.toSpace(new Vector1D(v10.getInf()));
323         Assert.assertEquals(0.0, p10Lower.getX(), 1.0e-10);
324         Assert.assertEquals(1.5, p10Lower.getY(), 1.0e-10);
325         Vector2D p10Upper = l1.toSpace(new Vector1D(v10.getSup()));
326         Assert.assertEquals(0.5, p10Upper.getX(), 1.0e-10);
327         Assert.assertEquals(2.0, p10Upper.getY(), 1.0e-10);
328         Interval v11 = i1.get(1);
329         Vector2D p11Lower = l1.toSpace(new Vector1D(v11.getInf()));
330         Assert.assertEquals(1.0, p11Lower.getX(), 1.0e-10);
331         Assert.assertEquals(2.5, p11Lower.getY(), 1.0e-10);
332         Vector2D p11Upper = l1.toSpace(new Vector1D(v11.getSup()));
333         Assert.assertEquals(1.5, p11Upper.getX(), 1.0e-10);
334         Assert.assertEquals(3.0, p11Upper.getY(), 1.0e-10);
335 
336         Line l2 = new Line(new Vector2D(-1.0, 2.0), 0, 1.0e-10);
337         SubLine s2 = (SubLine) set.intersection(l2.wholeHyperplane());
338         List<Interval> i2 = ((IntervalsSet) s2.getRemainingRegion()).asList();
339         Assert.assertEquals(1, i2.size());
340         Interval v20 = i2.get(0);
341         Vector2D p20Lower = l2.toSpace(new Vector1D(v20.getInf()));
342         Assert.assertEquals(1.0, p20Lower.getX(), 1.0e-10);
343         Assert.assertEquals(2.0, p20Lower.getY(), 1.0e-10);
344         Vector2D p20Upper = l2.toSpace(new Vector1D(v20.getSup()));
345         Assert.assertEquals(3.0, p20Upper.getX(), 1.0e-10);
346         Assert.assertEquals(2.0, p20Upper.getY(), 1.0e-10);

1280     private boolean checkInSegment(Vector2D p,
1281                                    Vector2D p1, Vector2D p2,
1282                                    double tolerance) {
1283         Line line = new Line(p1, p2, 1.0e-10);
1284         if (line.getOffset(p) < tolerance) {
1285             double x  = (line.toSubSpace(p)).getX();
1286             double x1 = (line.toSubSpace(p1)).getX();
1287             double x2 = (line.toSubSpace(p2)).getX();
1288             return (((x - x1) * (x - x2) <= 0.0)
1289                     || (p1.distance(p) < tolerance)
1290                     || (p2.distance(p) < tolerance));
1291         } else {
1292             return false;
1293         }

1296     private void checkVertices(Vector2D[][] rebuiltVertices,
1297                                Vector2D[][] vertices) {
1298 
1299         // each rebuilt vertex should be in a segment joining two original vertices
1300         for (int i = 0; i < rebuiltVertices.length; ++i) {
1301             for (int j = 0; j < rebuiltVertices[i].length; ++j) {
1302                 boolean inSegment = false;
1303                 Vector2D p = rebuiltVertices[i][j];
1304                 for (int k = 0; k < vertices.length; ++k) {
1305                     Vector2D[] loop = vertices[k];
1306                     int length = loop.length;
1307                     for (int l = 0; (! inSegment) && (l < length); ++l) {
1308                         inSegment = checkInSegment(p, loop[l], loop[(l + 1) % length], 1.0e-10);
1309                     }
1310                 }
1311                 Assert.assertTrue(inSegment);
1312             }
1313         }
1314 
1315         // each original vertex should have a corresponding rebuilt vertex
1316         for (int k = 0; k < vertices.length; ++k) {
1317             for (int l = 0; l < vertices[k].length; ++l) {
1318                 double min = Double.POSITIVE_INFINITY;
1319                 for (int i = 0; i < rebuiltVertices.length; ++i) {
1320                     for (int j = 0; j < rebuiltVertices[i].length; ++j) {
1321                         min = FastMath.min(vertices[k][l].distance(rebuiltVertices[i][j]),
1322                                        min);
1323                     }
1324                 }
1325                 Assert.assertEquals(0.0, min, 1.0e-10);
1326             }
1327         }
