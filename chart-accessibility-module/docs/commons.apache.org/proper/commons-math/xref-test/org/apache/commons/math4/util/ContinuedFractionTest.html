17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.util.ContinuedFraction;
20  import org.junit.Assert;
21  import org.junit.Test;
22  
23  
24  /**
25   */
26  public class ContinuedFractionTest {
27  
28      @Test
29      public void testGoldenRatio() throws Exception {
30          ContinuedFraction cf = new ContinuedFraction() {
31  
32              @Override
33              public double getA(int n, double x) {
34                  return 1.0;
35              }
36  
37              @Override
38              public double getB(int n, double x) {
39                  return 1.0;
40              }
41          };
42  
43          double gr = cf.evaluate(0.0, 10e-9);
44          Assert.assertEquals(1.61803399, gr, 10e-9);
45      }
46  }
