34  public class UniformCrossoverTest {
35      private static final int LEN = 10000;
36      private static final List<Integer> p1 = new ArrayList<Integer>(LEN);
37      private static final List<Integer> p2 = new ArrayList<Integer>(LEN);
38  
39      @SuppressWarnings("boxing")
40      @BeforeClass
41      public static void setUpBeforeClass() {
42          for (int i = 0; i < LEN; i++) {
43              p1.add(0);
44              p2.add(1);
45          }
46      }
47  
48      @Test(expected = OutOfRangeException.class)
49      public void testRatioTooLow() {
50          new UniformCrossover<Integer>(-0.5d);
51      }
52  
53      @Test(expected = OutOfRangeException.class)
54      public void testRatioTooHigh() {
55          new UniformCrossover<Integer>(1.5d);
56      }
57  
58      @Test
59      public void testCrossover() {
60          // test crossover with different ratios
61          performCrossover(0.5);
62          performCrossover(0.7);
63          performCrossover(0.2);
64      }
65  
66      private void performCrossover(double ratio) {
67          final DummyBinaryChromosome p1c = new DummyBinaryChromosome(p1);
68          final DummyBinaryChromosome p2c = new DummyBinaryChromosome(p2);
69  
70          final CrossoverPolicy cp = new UniformCrossover<Integer>(ratio);
71  
72          // make a number of rounds
73          for (int i = 0; i < 20; i++) {
74              final ChromosomePair pair = cp.crossover(p1c, p2c);
75  
76              final List<Integer> c1 = ((DummyBinaryChromosome) pair.getFirst()).getRepresentation();
77              final List<Integer> c2 = ((DummyBinaryChromosome) pair.getSecond()).getRepresentation();
78  
79              int from1 = 0;
80              int from2 = 0;
81  
82              // check first child
83              for (int val : c1) {
84                  if (val == 0) {
85                      from1++;
86                  } else {
87                      from2++;
88                  }
89              }
90  
91              Assert.assertEquals(1.0 - ratio, (double) from1 / LEN, 0.1);
92              Assert.assertEquals(ratio, (double) from2 / LEN, 0.1);
93  
94              from1 = 0;
95              from2 = 0;
96  
97              // check second child
98              for (int val : c2) {
99                  if (val == 0) {
100                     from1++;
101                 } else {
102                     from2++;
103                 }
104             }
105 
106             Assert.assertEquals(ratio, (double) from1 / LEN, 0.1);
107             Assert.assertEquals(1.0 - ratio, (double) from2 / LEN, 0.1);
108         }
109     }
