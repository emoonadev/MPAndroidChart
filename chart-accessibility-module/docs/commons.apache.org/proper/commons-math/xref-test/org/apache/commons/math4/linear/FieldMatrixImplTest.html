385     @Test
386     public void testExamples() {
387         // Create a real matrix with two rows and three columns
388         Fraction[][] matrixData = {
389                 {new Fraction(1),new Fraction(2),new Fraction(3)},
390                 {new Fraction(2),new Fraction(5),new Fraction(3)}
391         };
392         FieldMatrix<Fraction> m = new Array2DRowFieldMatrix<Fraction>(matrixData);
393         // One more with three rows, two columns
394         Fraction[][] matrixData2 = {
395                 {new Fraction(1),new Fraction(2)},
396                 {new Fraction(2),new Fraction(5)},
397                 {new Fraction(1), new Fraction(7)}
398         };
399         FieldMatrix<Fraction> n = new Array2DRowFieldMatrix<Fraction>(matrixData2);
400         // Now multiply m by n
401         FieldMatrix<Fraction> p = m.multiply(n);
402         Assert.assertEquals(2, p.getRowDimension());
403         Assert.assertEquals(2, p.getColumnDimension());
404         // Invert p
405         FieldMatrix<Fraction> pInverse = new FieldLUDecomposition<Fraction>(p).getSolver().getInverse();
406         Assert.assertEquals(2, pInverse.getRowDimension());
407         Assert.assertEquals(2, pInverse.getColumnDimension());
408 
409         // Solve example
410         Fraction[][] coefficientsData = {
411                 {new Fraction(2), new Fraction(3), new Fraction(-2)},
412                 {new Fraction(-1), new Fraction(7), new Fraction(6)},
413                 {new Fraction(4), new Fraction(-3), new Fraction(-5)}
414         };
415         FieldMatrix<Fraction> coefficients = new Array2DRowFieldMatrix<Fraction>(coefficientsData);
416         Fraction[] constants = {
417             new Fraction(1), new Fraction(-2), new Fraction(1)
418         };
419         Fraction[] solution;
420         solution = new FieldLUDecomposition<Fraction>(coefficients)
421             .getSolver()
422             .solve(new ArrayFieldVector<Fraction>(constants, false)).toArray();
423         Assert.assertEquals(new Fraction(2).multiply(solution[0]).
424                      add(new Fraction(3).multiply(solution[1])).
425                      subtract(new Fraction(2).multiply(solution[2])), constants[0]);
426         Assert.assertEquals(new Fraction(-1).multiply(solution[0]).
427                      add(new Fraction(7).multiply(solution[1])).
428                      add(new Fraction(6).multiply(solution[2])), constants[1]);
429         Assert.assertEquals(new Fraction(4).multiply(solution[0]).
430                      subtract(new Fraction(3).multiply(solution[1])).
431                      subtract(new Fraction(5).multiply(solution[2])), constants[2]);
432 
433     }

456     private void checkGetSubMatrix(FieldMatrix<Fraction> m, Fraction[][] reference,
457                                    int startRow, int endRow, int startColumn, int endColumn) {

1098     /** extracts the l  and u matrices from compact lu representation */
1099     protected void splitLU(FieldMatrix<Fraction> lu,
1100                            Fraction[][] lowerData,
1101                            Fraction[][] upperData) {
1102         if (!lu.isSquare()) {
1103             throw new NonSquareMatrixException(lu.getRowDimension(), lu.getColumnDimension());
1104         }
1105         if (lowerData.length != lowerData[0].length) {
1106             throw new DimensionMismatchException(lowerData.length, lowerData[0].length);
1107         }
1108         if (upperData.length != upperData[0].length) {
1109             throw new DimensionMismatchException(upperData.length, upperData[0].length);
1110         }
1111         if (lowerData.length != upperData.length) {
1112             throw new DimensionMismatchException(lowerData.length, upperData.length);
1113         }
1114         if (lowerData.length != lu.getRowDimension()) {
1115             throw new DimensionMismatchException(lowerData.length, lu.getRowDimension());
1116         }
1117         int n = lu.getRowDimension();
1118         for (int i = 0; i < n; i++) {
1119             for (int j = 0; j < n; j++) {
1120                 if (j < i) {
1121                     lowerData[i][j] = lu.getEntry(i, j);
1122                     upperData[i][j] = Fraction.ZERO;
1123                 } else if (i == j) {
1124                     lowerData[i][j] = Fraction.ONE;
1125                     upperData[i][j] = lu.getEntry(i, j);
1126                 } else {
1127                     lowerData[i][j] = Fraction.ZERO;
1128                     upperData[i][j] = lu.getEntry(i, j);
1129                 }
1130             }
1131         }
1132     }

1134     /** Returns the result of applying the given row permutation to the matrix */
1135     protected FieldMatrix<Fraction> permuteRows(FieldMatrix<Fraction> matrix, int[] permutation) {
1136         if (!matrix.isSquare()) {
1137             throw new NonSquareMatrixException(matrix.getRowDimension(),
1138                                                matrix.getColumnDimension());
1139         }
1140         if (matrix.getRowDimension() != permutation.length) {
1141             throw new DimensionMismatchException(matrix.getRowDimension(), permutation.length);
1142         }
1143         int n = matrix.getRowDimension();
1144         int m = matrix.getColumnDimension();
1145         Fraction out[][] = new Fraction[m][n];
1146         for (int i = 0; i < n; i++) {
1147             for (int j = 0; j < m; j++) {
1148                 out[i][j] = matrix.getEntry(permutation[i], j);
1149             }
1150         }
1151         return new Array2DRowFieldMatrix<Fraction>(out);
1152     }
