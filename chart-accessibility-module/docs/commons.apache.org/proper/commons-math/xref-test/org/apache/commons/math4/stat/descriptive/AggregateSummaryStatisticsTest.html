40      /**
41       * Tests the standard aggregation behavior
42       */
43      @Test
44      public void testAggregation() {
45          AggregateSummaryStatistics aggregate = new AggregateSummaryStatistics();
46          SummaryStatistics setOneStats = aggregate.createContributingStatistics();
47          SummaryStatistics setTwoStats = aggregate.createContributingStatistics();
48  
49          Assert.assertNotNull("The set one contributing stats are null", setOneStats);
50          Assert.assertNotNull("The set two contributing stats are null", setTwoStats);
51          Assert.assertNotSame("Contributing stats objects are the same", setOneStats, setTwoStats);
52  
53          setOneStats.addValue(2);
54          setOneStats.addValue(3);
55          setOneStats.addValue(5);
56          setOneStats.addValue(7);
57          setOneStats.addValue(11);
58          Assert.assertEquals("Wrong number of set one values", 5, setOneStats.getN());
59          Assert.assertTrue("Wrong sum of set one values", Precision.equals(28.0, setOneStats.getSum(), 1));
60  
61          setTwoStats.addValue(2);
62          setTwoStats.addValue(4);
63          setTwoStats.addValue(8);
64          Assert.assertEquals("Wrong number of set two values", 3, setTwoStats.getN());
65          Assert.assertTrue("Wrong sum of set two values", Precision.equals(14.0, setTwoStats.getSum(), 1));
66  
67          Assert.assertEquals("Wrong number of aggregate values", 8, aggregate.getN());
68          Assert.assertTrue("Wrong aggregate sum", Precision.equals(42.0, aggregate.getSum(), 1));
69      }

71      /**
72       * Verify that aggregating over a partition gives the same results
73       * as direct computation.
74       *
75       *  1) Randomly generate a dataset of 10-100 values
76       *     from [-100, 100]
77       *  2) Divide the dataset it into 2-5 partitions
78       *  3) Create an AggregateSummaryStatistic and ContributingStatistics
79       *     for each partition
80       *  4) Compare results from the AggregateSummaryStatistic with values
81       *     returned by a single SummaryStatistics instance that is provided
82       *     the full dataset
83       */
84      @Test
85      public void testAggregationConsistency() {
86  
87          // Generate a random sample and random partition
88          double[] totalSample = generateSample();
89          double[][] subSamples = generatePartition(totalSample);
90          int nSamples = subSamples.length;
91  
92          // Create aggregator and total stats for comparison
93          AggregateSummaryStatistics aggregate = new AggregateSummaryStatistics();
94          SummaryStatistics totalStats = new SummaryStatistics();
95  
96          // Create array of component stats
97          SummaryStatistics componentStats[] = new SummaryStatistics[nSamples];
98  
99          for (int i = 0; i < nSamples; i++) {
100 
101             // Make componentStats[i] a contributing statistic to aggregate
102             componentStats[i] = aggregate.createContributingStatistics();
103 
104             // Add values from subsample
105             for (int j = 0; j < subSamples[i].length; j++) {
106                 componentStats[i].addValue(subSamples[i][j]);
107             }
108         }
109 
110         // Compute totalStats directly
111         for (int i = 0; i < totalSample.length; i++) {
112             totalStats.addValue(totalSample[i]);
113         }
114 
115         /*
116          * Compare statistics in totalStats with aggregate.
117          * Note that guaranteed success of this comparison depends on the
118          * fact that <aggregate> gets values in exactly the same order
119          * as <totalStats>.
120          *
121          */
122         Assert.assertEquals(totalStats.getSummary(), aggregate.getSummary());
123 
124     }

126     /**
127      * Test aggregate function by randomly generating a dataset of 10-100 values
128      * from [-100, 100], dividing it into 2-5 partitions, computing stats for each
129      * partition and comparing the result of aggregate(...) applied to the collection
130      * of per-partition SummaryStatistics with a single SummaryStatistics computed
131      * over the full sample.
132      */
133     @Test
134     public void testAggregate() {
135 
136         // Generate a random sample and random partition
137         double[] totalSample = generateSample();
138         double[][] subSamples = generatePartition(totalSample);
139         int nSamples = subSamples.length;
140 
141         // Compute combined stats directly
142         SummaryStatistics totalStats = new SummaryStatistics();
143         for (int i = 0; i < totalSample.length; i++) {
144             totalStats.addValue(totalSample[i]);
145         }
146 
147         // Now compute subsample stats individually and aggregate
148         SummaryStatistics[] subSampleStats = new SummaryStatistics[nSamples];
149         for (int i = 0; i < nSamples; i++) {
150             subSampleStats[i] = new SummaryStatistics();
151         }
152         Collection<SummaryStatistics> aggregate = new ArrayList<SummaryStatistics>();
153         for (int i = 0; i < nSamples; i++) {
154             for (int j = 0; j < subSamples[i].length; j++) {
155                 subSampleStats[i].addValue(subSamples[i][j]);
156             }
157             aggregate.add(subSampleStats[i]);
158         }
159 
160         // Compare values
161         StatisticalSummary aggregatedStats = AggregateSummaryStatistics.aggregate(aggregate);
162         assertEquals(totalStats.getSummary(), aggregatedStats, 10E-12);
163     }

261     /**
262      * Verifies that a StatisticalSummary and a StatisticalSummaryValues are equal up
263      * to delta, with NaNs, infinities returned in the same spots. For max, min, n, values
264      * have to agree exactly, delta is used only for sum, mean, variance, std dev.
265      */
266     protected static void assertEquals(StatisticalSummary expected, StatisticalSummary observed, double delta) {
267         TestUtils.assertEquals(expected.getMax(), observed.getMax(), 0);
268         TestUtils.assertEquals(expected.getMin(), observed.getMin(), 0);
269         Assert.assertEquals(expected.getN(), observed.getN());
270         TestUtils.assertEquals(expected.getSum(), observed.getSum(), delta);
271         TestUtils.assertEquals(expected.getMean(), observed.getMean(), delta);
272         TestUtils.assertEquals(expected.getStandardDeviation(), observed.getStandardDeviation(), delta);
273         TestUtils.assertEquals(expected.getVariance(), observed.getVariance(), delta);
274     }

277     /**
278      * Generates a random sample of double values.
279      * Sample size is random, between 10 and 100 and values are
280      * uniformly distributed over [-100, 100].
281      *
282      * @return array of random double values
283      */
284     private double[] generateSample() {
285         final IntegerDistribution.Sampler size =
286             new UniformIntegerDistribution(10, 100).createSampler(RandomSource.create(RandomSource.WELL_512_A,
287                                                                                       327652));
288         final RealDistribution.Sampler randomData
289             = new UniformRealDistribution(-100, 100).createSampler(RandomSource.create(RandomSource.WELL_512_A,
290                                                                                        64925784252L));;
291         final int sampleSize = size.sample();
292         final double[] out = AbstractRealDistribution.sample(sampleSize, randomData);
293         return out;
294     }

296     /**
297      * Generates a partition of <sample> into up to 5 sequentially selected
298      * subsamples with randomly selected partition points.
299      *
300      * @param sample array to partition
301      * @return rectangular array with rows = subsamples
302      */
303     private double[][] generatePartition(double[] sample) {
304         final int length = sample.length;
305         final double[][] out = new double[5][];
306         int cur = 0;          // beginning of current partition segment
307         int offset = 0;       // end of current partition segment
308         int sampleCount = 0;  // number of segments defined
309         for (int i = 0; i < 5; i++) {
310             if (cur == length || offset == length) {
311                 break;
312             }
313             final int next;
314             if (i == 4 || cur == length - 1) {
315                 next = length - 1;
316             } else {
317                 final IntegerDistribution.Sampler sampler =
318                     new UniformIntegerDistribution(cur, length - 1).createSampler(RandomSource.create(RandomSource.WELL_512_A));
319                 next = sampler.sample();
320             }
321             final int subLength = next - cur + 1;
322             out[i] = new double[subLength];
323             System.arraycopy(sample, offset, out[i], 0, subLength);
324             cur = next + 1;
325             sampleCount++;
326             offset += subLength;
327         }
328         if (sampleCount < 5) {
329             double[][] out2 = new double[sampleCount][];
330             for (int j = 0; j < sampleCount; j++) {
331                 final int curSize = out[j].length;
332                 out2[j] = new double[curSize];
333                 System.arraycopy(out[j], 0, out2[j], 0, curSize);
334             }
335             return out2;
336         } else {
337             return out;
338         }
339     }
