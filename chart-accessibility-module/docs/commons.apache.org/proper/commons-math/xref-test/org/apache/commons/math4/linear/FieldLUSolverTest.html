33  public class FieldLUSolverTest {
34      private int[][] testData = {
35              { 1, 2, 3},
36              { 2, 5, 3},
37              { 1, 0, 8}
38      };
39      private int[][] luData = {
40              { 2, 3, 3 },
41              { 0, 5, 7 },
42              { 6, 9, 8 }
43      };
44  
45      // singular matrices
46      private int[][] singular = {
47              { 2, 3 },
48              { 2, 3 }
49      };
50      private int[][] bigSingular = {
51              { 1, 2,   3,    4 },
52              { 2, 5,   3,    4 },
53              { 7, 3, 256, 1930 },
54              { 3, 7,   6,    8 }
55      }; // 4th row = 1st + 2nd
56  
57      public static FieldMatrix<Fraction> createFractionMatrix(final int[][] data) {
58          final int numRows = data.length;
59          final int numCols = data[0].length;
60          final Array2DRowFieldMatrix<Fraction> m;
61          m = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(),
62                                                  numRows, numCols);
63          for (int i = 0; i < numRows; i++) {
64              for (int j = 0; j < numCols; j++) {
65                  m.setEntry(i, j, new Fraction(data[i][j], 1));
66              }
67          }
68          return m;
69      }
70  
71      /** test singular */
72      @Test
73      public void testSingular() {
74          FieldDecompositionSolver<Fraction> solver;
75          solver = new FieldLUDecomposition<Fraction>(createFractionMatrix(testData))
76              .getSolver();
77          Assert.assertTrue(solver.isNonSingular());
78          solver = new FieldLUDecomposition<Fraction>(createFractionMatrix(singular))
79              .getSolver();
80          Assert.assertFalse(solver.isNonSingular());
81          solver = new FieldLUDecomposition<Fraction>(createFractionMatrix(bigSingular))
82              .getSolver();
83          Assert.assertFalse(solver.isNonSingular());
84      }
85  
86      /** test solve dimension errors */
87      @Test
88      public void testSolveDimensionErrors() {
89          FieldDecompositionSolver<Fraction> solver;
90          solver = new FieldLUDecomposition<Fraction>(createFractionMatrix(testData))
91              .getSolver();
92          FieldMatrix<Fraction> b = createFractionMatrix(new int[2][2]);
93          try {
94              solver.solve(b);
95              Assert.fail("an exception should have been thrown");
96          } catch (MathIllegalArgumentException iae) {
97              // expected behavior
98          }
99          try {
100             solver.solve(b.getColumnVector(0));
101             Assert.fail("an exception should have been thrown");
102         } catch (MathIllegalArgumentException iae) {
103             // expected behavior
104         }
105     }

128     /** test solve */
129     @Test
130     public void testSolve() {
131         FieldDecompositionSolver<Fraction> solver;
132         solver = new FieldLUDecomposition<Fraction>(createFractionMatrix(testData))
133             .getSolver();
134         FieldMatrix<Fraction> b = createFractionMatrix(new int[][] {
135                 { 1, 0 }, { 2, -5 }, { 3, 1 }
136         });
137         FieldMatrix<Fraction> xRef = createFractionMatrix(new int[][] {
138                 { 19, -71 }, { -6, 22 }, { -2, 9 }
139         });
140 
141         // using FieldMatrix
142         FieldMatrix<Fraction> x = solver.solve(b);
143         for (int i = 0; i < x.getRowDimension(); i++){
144             for (int j = 0; j < x.getColumnDimension(); j++){
145                 Assert.assertEquals("(" + i + ", " + j + ")",
146                                     xRef.getEntry(i, j), x.getEntry(i, j));
147             }
148         }
149 
150         // using ArrayFieldVector
151         for (int j = 0; j < b.getColumnDimension(); j++) {
152             final FieldVector<Fraction> xj = solver.solve(b.getColumnVector(j));
153             for (int i = 0; i < xj.getDimension(); i++){
154                 Assert.assertEquals("(" + i + ", " + j + ")",
155                                     xRef.getEntry(i, j), xj.getEntry(i));
156             }
157         }
158 
159         // using SparseFieldVector
160         for (int j = 0; j < b.getColumnDimension(); j++) {
161             final SparseFieldVector<Fraction> bj;
162             bj = new SparseFieldVector<Fraction>(FractionField.getInstance(),
163                                                  b.getColumn(j));
164             final FieldVector<Fraction> xj = solver.solve(bj);
165             for (int i = 0; i < xj.getDimension(); i++) {
166                 Assert.assertEquals("(" + i + ", " + j + ")",
167                                     xRef.getEntry(i, j), xj.getEntry(i));
168             }
169         }
170     }

172     /** test determinant */
173     @Test
174     public void testDeterminant() {
175         Assert.assertEquals( -1, getDeterminant(createFractionMatrix(testData)), 1E-15);
176         Assert.assertEquals(-10, getDeterminant(createFractionMatrix(luData)), 1E-14);
177         Assert.assertEquals(  0, getDeterminant(createFractionMatrix(singular)), 1E-15);
178         Assert.assertEquals(  0, getDeterminant(createFractionMatrix(bigSingular)), 1E-15);
179     }
180 
181     private double getDeterminant(final FieldMatrix<Fraction> m) {
182         return new FieldLUDecomposition<Fraction>(m).getDeterminant().doubleValue();
183     }
