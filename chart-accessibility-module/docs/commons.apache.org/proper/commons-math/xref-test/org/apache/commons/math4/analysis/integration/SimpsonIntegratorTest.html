31  /**
32   * Test case for Simpson integrator.
33   * <p>
34   * Test runs show that for a default relative accuracy of 1E-6, it
35   * generally takes 5 to 10 iterations for the integral to converge.
36   *
37   */
38  public final class SimpsonIntegratorTest {
39  
40      /**
41       * Test of integrator for the sine function.
42       */
43      @Test
44      public void testSinFunction() {
45          UnivariateFunction f = new Sin();
46          UnivariateIntegrator integrator = new SimpsonIntegrator();
47          double min, max, expected, result, tolerance;
48  
49          min = 0; max = FastMath.PI; expected = 2;
50          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
51          result = integrator.integrate(1000, f, min, max);
52          Assert.assertTrue(integrator.getEvaluations() < 100);
53          Assert.assertTrue(integrator.getIterations()  < 10);
54          Assert.assertEquals(expected, result, tolerance);
55  
56          min = -FastMath.PI/3; max = 0; expected = -0.5;
57          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
58          result = integrator.integrate(1000, f, min, max);
59          Assert.assertTrue(integrator.getEvaluations() < 50);
60          Assert.assertTrue(integrator.getIterations()  < 10);
61          Assert.assertEquals(expected, result, tolerance);
62      }
63  
64      /**
65       * Test of integrator for the quintic function.
66       */
67      @Test
68      public void testQuinticFunction() {
69          UnivariateFunction f = new QuinticFunction();
70          UnivariateIntegrator integrator = new SimpsonIntegrator();
71          double min, max, expected, result, tolerance;
72  
73          min = 0; max = 1; expected = -1.0/48;
74          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
75          result = integrator.integrate(1000, f, min, max);
76          Assert.assertTrue(integrator.getEvaluations() < 150);
77          Assert.assertTrue(integrator.getIterations()  < 10);
78          Assert.assertEquals(expected, result, tolerance);
79  
80          min = 0; max = 0.5; expected = 11.0/768;
81          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
82          result = integrator.integrate(1000, f, min, max);
83          Assert.assertTrue(integrator.getEvaluations() < 100);
84          Assert.assertTrue(integrator.getIterations()  < 10);
85          Assert.assertEquals(expected, result, tolerance);
86  
87          min = -1; max = 4; expected = 2048/3.0 - 78 + 1.0/48;
88          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
89          result = integrator.integrate(1000, f, min, max);
90          Assert.assertTrue(integrator.getEvaluations() < 150);
91          Assert.assertTrue(integrator.getIterations()  < 10);
92          Assert.assertEquals(expected, result, tolerance);
93      }
94  
95      /**
96       * Test of parameters for the integrator.
97       */
98      @Test
99      public void testParameters() {
100         UnivariateFunction f = new Sin();
101         try {
102             // bad interval
103             new SimpsonIntegrator().integrate(1000, f, 1, -1);
104             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
105         } catch (NumberIsTooLargeException ex) {
106             // expected
107         }
108         try {
109             // bad iteration limits
110             new SimpsonIntegrator(5, 4);
111             Assert.fail("Expecting NumberIsTooSmallException - bad iteration limits");
112         } catch (NumberIsTooSmallException ex) {
113             // expected
114         }
115         try {
116             // bad iteration limits
117             new SimpsonIntegrator(10, 99);
118             Assert.fail("Expecting NumberIsTooLargeException - bad iteration limits");
119         } catch (NumberIsTooLargeException ex) {
120             // expected
121         }
122     }
