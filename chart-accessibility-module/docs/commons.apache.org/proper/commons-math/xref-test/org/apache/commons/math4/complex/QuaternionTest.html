106     @Test
107     public final void testProductQuaternionQuaternion() {
108 
109         // Case : analytic test case
110 
111         final Quaternion qA = new Quaternion(1, 0.5, -3, 4);
112         final Quaternion qB = new Quaternion(6, 2, 1, -9);
113         final Quaternion qResult = Quaternion.multiply(qA, qB);
114 
115         Assert.assertEquals(44, qResult.getQ0(), EPS);
116         Assert.assertEquals(28, qResult.getQ1(), EPS);
117         Assert.assertEquals(-4.5, qResult.getQ2(), EPS);
118         Assert.assertEquals(21.5, qResult.getQ3(), EPS);
119 
120         // comparison with the result given by the formula :
121         // qResult = (scalarA * scalarB - vectorA . vectorB) + (scalarA * vectorB + scalarB * vectorA + vectorA ^
122         // vectorB)
123 
124         final Vector3D vectorA = new Vector3D(qA.getVectorPart());
125         final Vector3D vectorB = new Vector3D(qB.getVectorPart());
126         final Vector3D vectorResult = new Vector3D(qResult.getVectorPart());
127 
128         final double scalarPartRef = qA.getScalarPart() * qB.getScalarPart() - Vector3D.dotProduct(vectorA, vectorB);
129 
130         Assert.assertEquals(scalarPartRef, qResult.getScalarPart(), EPS);
131 
132         final Vector3D vectorPartRef = ((vectorA.scalarMultiply(qB.getScalarPart())).add(vectorB.scalarMultiply(qA
133                 .getScalarPart()))).add(Vector3D.crossProduct(vectorA, vectorB));
134         final double norm = (vectorResult.subtract(vectorPartRef)).getNorm();
135 
136         Assert.assertEquals(0, norm, EPS);
137 
138         // Conjugate of the product of two quaternions and product of their conjugates :
139         // Conj(qA * qB) = Conj(qB) * Conj(qA)
140 
141         final Quaternion conjugateOfProduct = qB.getConjugate().multiply(qA.getConjugate());
142         final Quaternion productOfConjugate = (qA.multiply(qB)).getConjugate();
143 
144         Assert.assertEquals(conjugateOfProduct.getQ0(), productOfConjugate.getQ0(), EPS);
145         Assert.assertEquals(conjugateOfProduct.getQ1(), productOfConjugate.getQ1(), EPS);
146         Assert.assertEquals(conjugateOfProduct.getQ2(), productOfConjugate.getQ2(), EPS);
147         Assert.assertEquals(conjugateOfProduct.getQ3(), productOfConjugate.getQ3(), EPS);
148     }

150     @Test
151     public final void testProductQuaternionVector() {
152 
153         // Case : Product between a vector and a quaternion : QxV
154 
155         final Quaternion quaternion = new Quaternion(4, 7, -1, 2);
156         final double[] vector = {2.0, 1.0, 3.0};
157         final Quaternion qResultQxV = Quaternion.multiply(quaternion, new Quaternion(vector));
158 
159         Assert.assertEquals(-19, qResultQxV.getQ0(), EPS);
160         Assert.assertEquals(3, qResultQxV.getQ1(), EPS);
161         Assert.assertEquals(-13, qResultQxV.getQ2(), EPS);
162         Assert.assertEquals(21, qResultQxV.getQ3(), EPS);
163 
164         // comparison with the result given by the formula :
165         // qResult = (- vectorQ . vector) + (scalarQ * vector + vectorQ ^ vector)
166 
167         final double[] vectorQ = quaternion.getVectorPart();
168         final double[] vectorResultQxV = qResultQxV.getVectorPart();
169 
170         final double scalarPartRefQxV = -Vector3D.dotProduct(new Vector3D(vectorQ), new Vector3D(vector));
171         Assert.assertEquals(scalarPartRefQxV, qResultQxV.getScalarPart(), EPS);
172 
173         final Vector3D vectorPartRefQxV = (new Vector3D(vector).scalarMultiply(quaternion.getScalarPart())).add(Vector3D
174                 .crossProduct(new Vector3D(vectorQ), new Vector3D(vector)));
175         final double normQxV = (new Vector3D(vectorResultQxV).subtract(vectorPartRefQxV)).getNorm();
176         Assert.assertEquals(0, normQxV, EPS);
177 
178         // Case : Product between a vector and a quaternion : VxQ
179 
180         final Quaternion qResultVxQ = Quaternion.multiply(new Quaternion(vector), quaternion);
181 
182         Assert.assertEquals(-19, qResultVxQ.getQ0(), EPS);
183         Assert.assertEquals(13, qResultVxQ.getQ1(), EPS);
184         Assert.assertEquals(21, qResultVxQ.getQ2(), EPS);
185         Assert.assertEquals(3, qResultVxQ.getQ3(), EPS);
186 
187         final double[] vectorResultVxQ = qResultVxQ.getVectorPart();
188 
189         // comparison with the result given by the formula :
190         // qResult = (- vector . vectorQ) + (scalarQ * vector + vector ^ vectorQ)
191 
192         final double scalarPartRefVxQ = -Vector3D.dotProduct(new Vector3D(vectorQ), new Vector3D(vector));
193         Assert.assertEquals(scalarPartRefVxQ, qResultVxQ.getScalarPart(), EPS);
194 
195         final Vector3D vectorPartRefVxQ = (new Vector3D(vector).scalarMultiply(quaternion.getScalarPart())).add(Vector3D
196                 .crossProduct(new Vector3D(vector), new Vector3D(vectorQ)));
197         final double normVxQ = (new Vector3D(vectorResultVxQ).subtract(vectorPartRefVxQ)).getNorm();
198         Assert.assertEquals(0, normVxQ, EPS);
199     }

352         Assert.assertTrue(q1.equals(q4, 1.1 * inc));

401     @Test
402     public final void testPolarForm() {
403         final Random r = new Random(48);
404         final int numberOfTrials = 1000;
405         for (int i = 0; i < numberOfTrials; i++) {
406             final Quaternion q = new Quaternion(2 * (r.nextDouble() - 0.5), 2 * (r.nextDouble() - 0.5),
407                                                 2 * (r.nextDouble() - 0.5), 2 * (r.nextDouble() - 0.5));
408             final Quaternion qP = q.getPositivePolarForm();
409 
410             Assert.assertTrue(qP.isUnitQuaternion(COMPARISON_EPS));
411             Assert.assertTrue(qP.getQ0() >= 0);
412 
413             final Rotation rot = new Rotation(q.getQ0(), q.getQ1(), q.getQ2(), q.getQ3(), true);
414             final Rotation rotP = new Rotation(qP.getQ0(), qP.getQ1(), qP.getQ2(), qP.getQ3(), true);
415 
416             Assert.assertEquals(rot.getAngle(), rotP.getAngle(), COMPARISON_EPS);
417             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getX(),
418                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getX(),
419                                 COMPARISON_EPS);
420             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getY(),
421                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getY(),
422                                 COMPARISON_EPS);
423             Assert.assertEquals(rot.getAxis(RotationConvention.VECTOR_OPERATOR).getZ(),
424                                 rot.getAxis(RotationConvention.VECTOR_OPERATOR).getZ(),
425                                 COMPARISON_EPS);
426         }

429     @Test
430     public final void testGetInverse() {
431         final Quaternion q = new Quaternion(1.5, 4, 2, -2.5);
432 
433         final Quaternion inverseQ = q.getInverse();
434         Assert.assertEquals(1.5 / 28.5, inverseQ.getQ0(), 0);
435         Assert.assertEquals(-4.0 / 28.5, inverseQ.getQ1(), 0);
436         Assert.assertEquals(-2.0 / 28.5, inverseQ.getQ2(), 0);
437         Assert.assertEquals(2.5 / 28.5, inverseQ.getQ3(), 0);
438 
439         final Quaternion product = Quaternion.multiply(inverseQ, q);
440         Assert.assertEquals(1, product.getQ0(), EPS);
441         Assert.assertEquals(0, product.getQ1(), EPS);
442         Assert.assertEquals(0, product.getQ2(), EPS);
443         Assert.assertEquals(0, product.getQ3(), EPS);
444 
445         final Quaternion qNul = new Quaternion(0, 0, 0, 0);
446         try {
447             final Quaternion inverseQNul = qNul.getInverse();
448             Assert.fail("expecting ZeroException but got : " + inverseQNul);

450             // expected
