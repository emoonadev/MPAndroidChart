18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.distribution.ConstantRealDistribution;
21  import org.apache.commons.math4.distribution.RealDistribution;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for ConstantRealDistribution.
27   */
28  public class ConstantRealDistributionTest extends RealDistributionAbstractTest {
29  
30      // --- Override tolerance -------------------------------------------------
31  
32      @Override
33      public void setUp() {
34          super.setUp();
35          setTolerance(0);
36      }
37  
38      //--- Implementations for abstract methods --------------------------------
39  
40      /** Creates the default uniform real distribution instance to use in tests. */
41      @Override
42      public ConstantRealDistribution makeDistribution() {
43          return new ConstantRealDistribution(1);
44      }
45  
46      /** Creates the default cumulative probability distribution test input values */
47      @Override
48      public double[] makeCumulativeTestPoints() {
49          return new double[] {0, 0.5, 1};
50      }
51  
52      /** Creates the default cumulative probability distribution test expected values */
53      @Override
54      public double[] makeCumulativeTestValues() {
55          return new double[] {0, 0, 1};
56      }
57  
58      /** Creates the default probability density test expected values */
59      @Override
60      public double[] makeDensityTestValues() {
61          return new double[] {0, 0, 1};
62      }
63  
64      /** Override default test, verifying that inverse cum is constant */
65      @Override
66      @Test
67      public void testInverseCumulativeProbabilities() {
68          RealDistribution dist = getDistribution();
69          for (double x : getCumulativeTestValues()) {
70              Assert.assertEquals(1,dist.inverseCumulativeProbability(x), 0);
71          }
72      }
73  
74      //--- Additional test cases -----------------------------------------------
75  
76      @Test
77      public void testMeanVariance() {
78          ConstantRealDistribution dist;
79  
80          dist = new ConstantRealDistribution(-1);
81          Assert.assertEquals(dist.getNumericalMean(), -1, 0d);
82          Assert.assertEquals(dist.getNumericalVariance(), 0, 0d);
83      }
84  
85      @Test
86      @Override
87      public void testSampler() {
88          final double value = 12.345;
89          final RealDistribution.Sampler sampler = new ConstantRealDistribution(value).createSampler(null);
90          for (int i = 0; i < 10; i++) {
91              Assert.assertEquals(value, sampler.sample(), 0);
92          }
93      }
94  }
