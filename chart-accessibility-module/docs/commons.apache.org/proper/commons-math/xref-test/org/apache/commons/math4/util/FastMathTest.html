251     @Test
252     public void testLogAccuracy() {
253         double maxerrulp = 0.0;
254 
255         for (int i = 0; i < NUMBER_OF_TRIALS; i++) {
256             double x = Math.exp(generator.nextDouble() * 1416.0 - 708.0) * generator.nextDouble();
257             // double x = generator.nextDouble()*2.0;
258             double tst = FastMath.log(x);
259             double ref = DfpMath.log(field.newDfp(x)).toDouble();
260             double err = (tst - ref) / ref;
261 
262             if (err != 0.0) {
263                 double ulp = Math.abs(ref -
264                                       Double.longBitsToDouble((Double.doubleToLongBits(ref) ^ 1)));
265                 double errulp = field.newDfp(tst).subtract(DfpMath.log(field.newDfp(x))).divide(field.newDfp(ulp)).toDouble();
266 //                System.out.println(x + "\t" + tst + "\t" + ref + "\t" + err + "\t" + errulp);
267 
268                 maxerrulp = Math.max(maxerrulp, Math.abs(errulp));
269             }
270         }
271 
272         assertTrue("log() had errors in excess of " + MAX_ERROR_ULP + " ULP", maxerrulp < MAX_ERROR_ULP);
273     }

1917     private long poorManFloorDiv(long a, long b) {
1918 
1919         // find q0, r0 such that a = q0 b + r0
1920         BigInteger q0 = BigInteger.valueOf(a / b);
1921         BigInteger r0 = BigInteger.valueOf(a % b);
1922         BigInteger fd = BigInteger.valueOf(Integer.MIN_VALUE);
1923         BigInteger bigB = BigInteger.valueOf(b);
1924 
1925         for (int k = -2; k < 2; ++k) {
1926             // find another pair q, r such that a = q b + r
1927             BigInteger bigK = BigInteger.valueOf(k);
1928             BigInteger q = q0.subtract(bigK);
1929             BigInteger r = r0.add(bigK.multiply(bigB));
1930             if (r.abs().compareTo(bigB.abs()) < 0 &&
1931                 (r.longValue() == 0l || ((r.longValue() ^ b) & 0x8000000000000000l) == 0)) {
1932                 if (fd.compareTo(q) < 0) {
1933                     fd = q;
1934                 }
1935             }
1936         }
1937 
1938         return fd.longValue();
1939 
1940     }

1942     private long poorManFloorMod(long a, long b) {
1943         return a - b * poorManFloorDiv(a, b);
1944     }
