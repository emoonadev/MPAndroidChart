32  public class ExponentialDistributionTest extends RealDistributionAbstractTest {

34      // --------------------- Override tolerance  --------------
35      @Override
36      public void setUp() {
37          super.setUp();
38          setTolerance(1E-9);
39      }

41      //-------------- Implementations for abstract methods -----------------------
42  
43      /** Creates the default continuous distribution instance to use in tests. */
44      @Override
45      public ExponentialDistribution makeDistribution() {
46          return new ExponentialDistribution(5.0);
47      }
48  
49      /** Creates the default cumulative probability distribution test input values */
50      @Override
51      public double[] makeCumulativeTestPoints() {
52          // quantiles computed using R version 2.9.2
53          return new double[] {0.00500250166792, 0.0502516792675, 0.126589039921, 0.256466471938,
54                  0.526802578289, 34.5387763949, 23.0258509299, 18.4443972706, 14.9786613678, 11.5129254650};
55      }
56  
57      /** Creates the default cumulative probability density test expected values */
58      @Override
59      public double[] makeCumulativeTestValues() {
60          return new double[] {0.001, 0.01, 0.025, 0.05, 0.1, 0.999,
61                  0.990, 0.975, 0.950, 0.900};
62      }
63  
64      /** Creates the default probability density test expected values */
65      @Override
66      public double[] makeDensityTestValues() {
67          return new double[] {0.1998, 0.198, 0.195, 0.19, 0.18, 0.000200000000000,
68                  0.00200000000002, 0.00499999999997, 0.00999999999994, 0.0199999999999};
69      }

73      @Test
74      public void testCumulativeProbabilityExtremes() {
75          setCumulativeTestPoints(new double[] {-2, 0});
76          setCumulativeTestValues(new double[] {0, 0});
77          verifyCumulativeProbabilities();
78      }
79  
80      @Test
81      public void testInverseCumulativeProbabilityExtremes() {
82           setInverseCumulativeTestPoints(new double[] {0, 1});
83           setInverseCumulativeTestValues(new double[] {0, Double.POSITIVE_INFINITY});
84           verifyInverseCumulativeProbabilities();
85      }
86  
87      @Test
88      public void testCumulativeProbability2() {
89          double actual = getDistribution().probability(0.25, 0.75);
90          Assert.assertEquals(0.0905214, actual, 10e-4);
91      }
92  
93      @Test
94      public void testDensity() {
95          ExponentialDistribution d1 = new ExponentialDistribution(1);
96          Assert.assertTrue(Precision.equals(0.0, d1.density(-1e-9), 1));
97          Assert.assertTrue(Precision.equals(1.0, d1.density(0.0), 1));
98          Assert.assertTrue(Precision.equals(0.0, d1.density(1000.0), 1));
99          Assert.assertTrue(Precision.equals(FastMath.exp(-1), d1.density(1.0), 1));
100         Assert.assertTrue(Precision.equals(FastMath.exp(-2), d1.density(2.0), 1));
101 
102         ExponentialDistribution d2 = new ExponentialDistribution(3);
103         Assert.assertTrue(Precision.equals(1/3.0, d2.density(0.0), 1));
104         // computed using  print(dexp(1, rate=1/3), digits=10) in R 2.5
105         Assert.assertEquals(0.2388437702, d2.density(1.0), 1e-8);
106 
107         // computed using  print(dexp(2, rate=1/3), digits=10) in R 2.5
108         Assert.assertEquals(0.1711390397, d2.density(2.0), 1e-8);
109     }
110 
111     @Test
112     public void testMeanAccessors() {
113         ExponentialDistribution distribution = (ExponentialDistribution) getDistribution();
114         Assert.assertEquals(5d, distribution.getMean(), Double.MIN_VALUE);
115     }
116 
117     @Test(expected=NotStrictlyPositiveException.class)
118     public void testPreconditions() {
119         new ExponentialDistribution(0);
120     }
121 
122     @Test
123     public void testMoments() {
124         final double tol = 1e-9;
125         ExponentialDistribution dist;
126 
127         dist = new ExponentialDistribution(11d);
128         Assert.assertEquals(dist.getNumericalMean(), 11d, tol);
129         Assert.assertEquals(dist.getNumericalVariance(), 11d * 11d, tol);
130 
131         dist = new ExponentialDistribution(10.5d);
132         Assert.assertEquals(dist.getNumericalMean(), 10.5d, tol);
133         Assert.assertEquals(dist.getNumericalVariance(), 10.5d * 10.5d, tol);
134     }
