17  package org.apache.commons.math4.random;
18  
19  import org.apache.commons.rng.RandomSource;
20  import org.apache.commons.rng.UniformRandomProvider;
21  import org.apache.commons.math4.random.UnitSphereRandomVectorGenerator;
22  import org.apache.commons.math4.util.FastMath;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class UnitSphereRandomVectorGeneratorTest {
27      /**
28       * Test the distribution of points from {@link UnitSphereRandomVectorGenerator#nextVector()}
29       * in two dimensions.
30       */
31      @Test
32      public void test2DDistribution() {
33  
34          UniformRandomProvider rg = RandomSource.create(RandomSource.XOR_SHIFT_1024_S, 17399225432L);
35          UnitSphereRandomVectorGenerator generator = new UnitSphereRandomVectorGenerator(2, rg);
36  
37          // In 2D, angles with a given vector should be uniformly distributed
38          int[] angleBuckets = new int[100];
39          int steps = 1000000;
40          for (int i = 0; i < steps; ++i) {
41              final double[] v = generator.nextVector();
42              Assert.assertEquals(2, v.length);
43              Assert.assertEquals(1, length(v), 1e-10);
44              // Compute angle formed with vector (1,0)
45              // Cosine of angle is their dot product, because both are unit length
46              // Dot product here is just the first element of the vector by construction
47              final double angle = FastMath.acos(v[0]);
48              final int bucket = (int) (angleBuckets.length * (angle / FastMath.PI));
49              ++angleBuckets[bucket];
50          }
51  
52          // Simplistic test for roughly even distribution
53          final int expectedBucketSize = steps / angleBuckets.length;
54          for (int bucket : angleBuckets) {
55              Assert.assertTrue("Bucket count " + bucket + " vs expected " + expectedBucketSize,
56                                FastMath.abs(expectedBucketSize - bucket) < 350);
57          }
58      }
59  
60      /**
61       * @return length (L2 norm) of given vector
62       */
63      private static double length(double[] vector) {
64          double total = 0;
65          for (double d : vector) {
66              total += d * d;
67          }
68          return FastMath.sqrt(total);
69      }
70  }
