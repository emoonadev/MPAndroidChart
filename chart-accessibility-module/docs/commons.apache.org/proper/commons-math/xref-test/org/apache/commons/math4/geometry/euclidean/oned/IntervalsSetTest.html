32  public class IntervalsSetTest {
33  
34      @Test
35      public void testInterval() {
36          IntervalsSet set = new IntervalsSet(2.3, 5.7, 1.0e-10);
37          Assert.assertEquals(3.4, set.getSize(), 1.0e-10);
38          Assert.assertEquals(4.0, ((Vector1D) set.getBarycenter()).getX(), 1.0e-10);
39          Assert.assertEquals(Region.Location.BOUNDARY, set.checkPoint(new Vector1D(2.3)));
40          Assert.assertEquals(Region.Location.BOUNDARY, set.checkPoint(new Vector1D(5.7)));
41          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(1.2)));
42          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(8.7)));
43          Assert.assertEquals(Region.Location.INSIDE,   set.checkPoint(new Vector1D(3.0)));
44          Assert.assertEquals(2.3, set.getInf(), 1.0e-10);
45          Assert.assertEquals(5.7, set.getSup(), 1.0e-10);
46      }
47  
48      @Test
49      public void testInfinite() {
50          IntervalsSet set = new IntervalsSet(9.0, Double.POSITIVE_INFINITY, 1.0e-10);
51          Assert.assertEquals(Region.Location.BOUNDARY, set.checkPoint(new Vector1D(9.0)));
52          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(8.4)));
53          for (double e = 1.0; e <= 6.0; e += 1.0) {
54              Assert.assertEquals(Region.Location.INSIDE,
55                                  set.checkPoint(new Vector1D(FastMath.pow(10.0, e))));
56          }
57          Assert.assertTrue(Double.isInfinite(set.getSize()));
58          Assert.assertEquals(9.0, set.getInf(), 1.0e-10);
59          Assert.assertTrue(Double.isInfinite(set.getSup()));
60  
61          set = (IntervalsSet) new RegionFactory<Euclidean1D>().getComplement(set);
62          Assert.assertEquals(9.0, set.getSup(), 1.0e-10);
63          Assert.assertTrue(Double.isInfinite(set.getInf()));
64  
65      }
66  
67      @Test
68      public void testMultiple() {
69          RegionFactory<Euclidean1D> factory = new RegionFactory<Euclidean1D>();
70          IntervalsSet set = (IntervalsSet)
71          factory.intersection(factory.union(factory.difference(new IntervalsSet(1.0, 6.0, 1.0e-10),
72                                                                new IntervalsSet(3.0, 5.0, 1.0e-10)),
73                                                                new IntervalsSet(9.0, Double.POSITIVE_INFINITY, 1.0e-10)),
74                                                                new IntervalsSet(Double.NEGATIVE_INFINITY, 11.0, 1.0e-10));
75          Assert.assertEquals(5.0, set.getSize(), 1.0e-10);
76          Assert.assertEquals(5.9, ((Vector1D) set.getBarycenter()).getX(), 1.0e-10);
77          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(0.0)));
78          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(4.0)));
79          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(8.0)));
80          Assert.assertEquals(Region.Location.OUTSIDE,  set.checkPoint(new Vector1D(12.0)));
81          Assert.assertEquals(Region.Location.INSIDE,   set.checkPoint(new Vector1D(1.2)));
82          Assert.assertEquals(Region.Location.INSIDE,   set.checkPoint(new Vector1D(5.9)));
83          Assert.assertEquals(Region.Location.INSIDE,   set.checkPoint(new Vector1D(9.01)));
84          Assert.assertEquals(Region.Location.BOUNDARY, set.checkPoint(new Vector1D(5.0)));
85          Assert.assertEquals(Region.Location.BOUNDARY, set.checkPoint(new Vector1D(11.0)));
86          Assert.assertEquals( 1.0, set.getInf(), 1.0e-10);
87          Assert.assertEquals(11.0, set.getSup(), 1.0e-10);
88  
89          List<Interval> list = set.asList();
90          Assert.assertEquals(3, list.size());
91          Assert.assertEquals( 1.0, list.get(0).getInf(), 1.0e-10);
92          Assert.assertEquals( 3.0, list.get(0).getSup(), 1.0e-10);
93          Assert.assertEquals( 5.0, list.get(1).getInf(), 1.0e-10);
94          Assert.assertEquals( 6.0, list.get(1).getSup(), 1.0e-10);
95          Assert.assertEquals( 9.0, list.get(2).getInf(), 1.0e-10);
96          Assert.assertEquals(11.0, list.get(2).getSup(), 1.0e-10);
97  
98      }

100     @Test
101     public void testSinglePoint() {
102         IntervalsSet set = new IntervalsSet(1.0, 1.0, 1.0e-10);
103         Assert.assertEquals(0.0, set.getSize(), Precision.SAFE_MIN);
104         Assert.assertEquals(1.0, ((Vector1D) set.getBarycenter()).getX(), Precision.EPSILON);
105     }
