41  public class MultiStartMultivariateOptimizerTest {
42      @Test
43      public void testCircleFitting() {
44          CircleScalar circle = new CircleScalar();
45          circle.addPoint( 30.0,  68.0);
46          circle.addPoint( 50.0,  -6.0);
47          circle.addPoint(110.0, -20.0);
48          circle.addPoint( 35.0,  15.0);
49          circle.addPoint( 45.0,  97.0);
50          // TODO: the wrapper around NonLinearConjugateGradientOptimizer is a temporary hack for
51          // version 3.1 of the library. It should be removed when NonLinearConjugateGradientOptimizer
52          // will officially be declared as implementing MultivariateDifferentiableOptimizer
53          GradientMultivariateOptimizer underlying
54              = new NonLinearConjugateGradientOptimizer(NonLinearConjugateGradientOptimizer.Formula.POLAK_RIBIERE,
55                                                        new SimpleValueChecker(1e-10, 1e-10));
56          UniformRandomProvider g = RandomSource.create(RandomSource.MT_64, 753289573253l);
57          RandomVectorGenerator generator
58              = new UncorrelatedRandomVectorGenerator(new double[] { 50, 50 },
59                                                      new double[] { 10, 10 },
60                                                      new GaussianRandomGenerator(g));
61          int nbStarts = 10;
62          MultiStartMultivariateOptimizer optimizer
63              = new MultiStartMultivariateOptimizer(underlying, nbStarts, generator);
64          PointValuePair optimum
65              = optimizer.optimize(new MaxEval(1000),
66                                   circle.getObjectiveFunction(),
67                                   circle.getObjectiveFunctionGradient(),
68                                   GoalType.MINIMIZE,
69                                   new InitialGuess(new double[] { 98.680, 47.345 }));
70          Assert.assertEquals(1000, optimizer.getMaxEvaluations());
71          PointValuePair[] optima = optimizer.getOptima();
72          Assert.assertEquals(nbStarts, optima.length);
73          for (PointValuePair o : optima) {
74              // we check the results of all intermediate restarts here (there are 10 such results)
75              Vector2D center = new Vector2D(o.getPointRef()[0], o.getPointRef()[1]);
76              Assert.assertEquals(69.9597, circle.getRadius(center), 1e-3);
77              Assert.assertEquals(96.07535, center.getX(), 1.4e-3);
78              Assert.assertEquals(48.1349, center.getY(), 5e-3);
79          }
80  
81          Assert.assertTrue(optimizer.getEvaluations() > 800);
82          Assert.assertTrue(optimizer.getEvaluations() < 900);
83  
84          Assert.assertEquals(3.1267527, optimum.getValue(), 1e-8);
85      }

87      @Test
88      public void testRosenbrock() {
89          Rosenbrock rosenbrock = new Rosenbrock();
90          SimplexOptimizer underlying
91              = new SimplexOptimizer(new SimpleValueChecker(-1, 1e-3));
92          NelderMeadSimplex simplex = new NelderMeadSimplex(new double[][] {
93                  { -1.2,  1.0 },
94                  { 0.9, 1.2 } ,
95                  {  3.5, -2.3 }
96              });
97          UniformRandomProvider g = RandomSource.create(RandomSource.MT_64, 16069223052l);
98          RandomVectorGenerator generator
99              = new UncorrelatedRandomVectorGenerator(2, new GaussianRandomGenerator(g));
100         int nbStarts = 10;
101         MultiStartMultivariateOptimizer optimizer
102             = new MultiStartMultivariateOptimizer(underlying, nbStarts, generator);
103         PointValuePair optimum
104             = optimizer.optimize(new MaxEval(1100),
105                                  new ObjectiveFunction(rosenbrock),
106                                  GoalType.MINIMIZE,
107                                  simplex,
108                                  new InitialGuess(new double[] { -1.2, 1.0 }));
109         Assert.assertEquals(nbStarts, optimizer.getOptima().length);
110 
111         Assert.assertEquals(rosenbrock.getCount(), optimizer.getEvaluations());
112         Assert.assertTrue(optimizer.getEvaluations() > 900);
113         Assert.assertTrue(optimizer.getEvaluations() < 1200);
114         Assert.assertTrue(optimum.getValue() < 5e-5);
115     }
116 
117     private static class Rosenbrock implements MultivariateFunction {
118         private int count;
119 
120         public Rosenbrock() {
121             count = 0;
122         }
123 
124         public double value(double[] x) {
125             ++count;
126             double a = x[1] - x[0] * x[0];
127             double b = 1 - x[0];
128             return 100 * a * a + b * b;
129         }
130 
131         public int getCount() {
132             return count;
133         }
134     }
