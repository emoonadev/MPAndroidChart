122     /**
123      * Test the normal approximation of the Poisson distribution by
124      * calculating P(90 &le; X &le; 110) for X = Po(100) and
125      * P(9900 &le; X &le; 10200) for X  = Po(10000)
126      */
127     @Test
128     public void testNormalApproximateProbability() {
129         PoissonDistribution dist = new PoissonDistribution(100);
130         double result = dist.normalApproximateProbability(110)
131                 - dist.normalApproximateProbability(89);
132         Assert.assertEquals(0.706281887248, result, 1E-10);
133 
134         dist = new PoissonDistribution(10000);
135         result = dist.normalApproximateProbability(10200)
136         - dist.normalApproximateProbability(9899);
137         Assert.assertEquals(0.820070051552, result, 1E-10);
138     }

161     @Test
162     public void testLargeMeanCumulativeProbability() {
163         double mean = 1.0;
164         while (mean <= 10000000.0) {
165             PoissonDistribution dist = new PoissonDistribution(mean);
166 
167             double x = mean * 2.0;
168             double dx = x / 10.0;
169             double p = Double.NaN;
170             double sigma = FastMath.sqrt(mean);
171             while (x >= 0) {
172                 try {
173                     p = dist.cumulativeProbability((int) x);
174                     Assert.assertFalse("NaN cumulative probability returned for mean = " +
175                             mean + " x = " + x,Double.isNaN(p));
176                     if (x > mean - 2 * sigma) {
177                         Assert.assertTrue("Zero cum probaility returned for mean = " +
178                                 mean + " x = " + x, p > 0);
179                     }
180                 } catch (Exception ex) {
181                     Assert.fail("mean of " + mean + " and x of " + x + " caused " + ex.getMessage());
182                 }
183                 x -= dx;
184             }
185 
186             mean *= 10.0;
187         }
188     }

205     private void checkProbability(PoissonDistribution dist, int x) {
206         double p = dist.cumulativeProbability(x);
207         Assert.assertFalse("NaN cumulative probability returned for mean = " +
208                 dist.getMean() + " x = " + x, Double.isNaN(p));
209         Assert.assertTrue("Zero cum probability returned for mean = " +
210                 dist.getMean() + " x = " + x, p > 0);
211     }

213     @Test
214     public void testLargeMeanInverseCumulativeProbability() {
215         double mean = 1.0;
216         while (mean <= 100000.0) { // Extended test value: 1E7.  Reduced to limit run time.
217             PoissonDistribution dist = new PoissonDistribution(mean);
218             double p = 0.1;
219             double dp = p;
220             while (p < .99) {
221                 try {
222                     int ret = dist.inverseCumulativeProbability(p);
223                     // Verify that returned value satisties definition
224                     Assert.assertTrue(p <= dist.cumulativeProbability(ret));
225                     Assert.assertTrue(p > dist.cumulativeProbability(ret - 1));
226                 } catch (Exception ex) {
227                     Assert.fail("mean of " + mean + " and p of " + p + " caused " + ex.getMessage());
228                 }
229                 p += dp;
230             }
231             mean *= 10.0;
232         }
233     }
