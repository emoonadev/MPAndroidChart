33      @Test
34      public void testScaledVectorTripleConstructor() {
35          Vector2D oneOne = new Vector2D(1.0,1.0);
36          Vector2D oneTwo = new Vector2D(1.0,2.0);
37          Vector2D oneThree = new Vector2D(1.0,3.0);
38  
39          Vector2D tripleCombo = new Vector2D(3.0, oneOne, 1.0, oneTwo, 2.5, oneThree);
40  
41          Assert.assertEquals(3.0 * 1 + 1.0 * 1 + 2.5 * 1,tripleCombo.getX(), EPS);
42          Assert.assertEquals(3.0 * 1 + 1.0 * 2 + 2.5 * 3,tripleCombo.getY(), EPS);
43      }

45      @Test
46      public void testScaledVectorQuadrupleConstructor() {
47          Vector2D oneOne = new Vector2D(1.0, 1.0);
48          Vector2D oneTwo = new Vector2D(1.0, 2.0);
49          Vector2D oneThree = new Vector2D(1.0, 3.0);
50          Vector2D oneFour = new Vector2D(1.0, 4.0);
51  
52          Vector2D tripleCombo = new Vector2D(3.0, oneOne, 1.0, oneTwo, 2.5, oneThree, 2.0, oneFour);
53  
54          Assert.assertEquals(3.0 * 1.0 + 1.0 * 1.0 + 2.5 * 1.0 + 2.0 * 1.0,tripleCombo.getX(), EPS);
55          Assert.assertEquals(3.0 * 1.0 + 1.0 * 2.0 + 2.5 * 3.0 + 2.0 * 4.0,tripleCombo.getY(), EPS);
56      }

138     @Test
139     public void testNormalize() {
140         Vector2D minusOneTwo = new Vector2D(-1.0,2.0);
141         Vector2D normalizedMinusOneTwo = minusOneTwo.normalize();
142         Assert.assertEquals(-1.0/FastMath.sqrt(5), normalizedMinusOneTwo.getX(), EPS);
143         Assert.assertEquals(2.0/FastMath.sqrt(5), normalizedMinusOneTwo.getY(), EPS);
144         Vector2D zero = minusOneTwo.getZero();
145         try {
146             zero.normalize();
147         }
148         catch (Exception e) {
149             Assert.assertTrue(e instanceof MathArithmeticException);
150         }
151     }

153     @Test
154     public void testAngle() {
155         Vector2D oneOne = new Vector2D(1.0, 1.0);
156         try {
157             Vector2D.angle(oneOne.getZero(), oneOne.getZero());
158         }
159         catch (Exception e) {
160             Assert.assertTrue(e instanceof MathArithmeticException);
161         }
162         Vector2D oneZero = new Vector2D(1.0,0.0);
163         double angle = Vector2D.angle(oneOne, oneZero);
164         Assert.assertEquals(FastMath.PI/4, angle, EPS);
165         Assert.assertEquals(0.004999958333958323, Vector2D.angle(new Vector2D(20.0,0.0), new Vector2D(20.0,0.1)), EPS);
166     }
