35      public void testPreconditions() {
36          MultidimensionalCounter c;
37  
38          try {
39              c = new MultidimensionalCounter(0, 1);
40              Assert.fail("NotStrictlyPositiveException expected");
41          } catch (NotStrictlyPositiveException e) {
42              // Expected.
43          }
44          try {
45              c = new MultidimensionalCounter(2, 0);
46              Assert.fail("NotStrictlyPositiveException expected");
47          } catch (NotStrictlyPositiveException e) {
48              // Expected.
49          }
50          try {
51              c = new MultidimensionalCounter(-1, 1);
52              Assert.fail("NotStrictlyPositiveException expected");
53          } catch (NotStrictlyPositiveException e) {
54              // Expected.
55          }
56  
57          c = new MultidimensionalCounter(2, 3);
58          try {
59              c.getCount(1, 1, 1);
60              Assert.fail("DimensionMismatchException expected");
61          } catch (DimensionMismatchException e) {
62              // Expected.
63          }
64          try {
65              c.getCount(3, 1);
66              Assert.fail("OutOfRangeException expected");
67          } catch (OutOfRangeException e) {
68              // Expected.
69          }
70          try {
71              c.getCount(0, -1);
72              Assert.fail("OutOfRangeException expected");
73          } catch (OutOfRangeException e) {
74              // Expected.
75          }
76          try {
77              c.getCounts(-1);
78              Assert.fail("OutOfRangeException expected");
79          } catch (OutOfRangeException e) {
80              // Expected.
81          }
82          try {
83              c.getCounts(6);
84              Assert.fail("OutOfRangeException expected");
85          } catch (OutOfRangeException e) {
86              // Expected.
87          }
88      }

107     @Test
108     public void testIterator() {
109         final int dim1 = 3;
110         final int dim2 = 4;
111 
112         final MultidimensionalCounter.Iterator iter
113             = new MultidimensionalCounter(dim1, dim2).iterator();
114 
115         final int max = dim1 * dim2;
116         for (int i = 0; i < max; i++) {
117             Assert.assertTrue(iter.hasNext());
118 
119             // Should not throw.
120             iter.next();
121         }
122 
123         Assert.assertFalse(iter.hasNext());
124     }

158     @Test
159     public void testIterationConsistency() {
160         final MultidimensionalCounter c = new MultidimensionalCounter(2, 3, 4);
161         final int[][] expected = new int[][] {
162             { 0, 0, 0 },
163             { 0, 0, 1 },
164             { 0, 0, 2 },
165             { 0, 0, 3 },
166             { 0, 1, 0 },
167             { 0, 1, 1 },
168             { 0, 1, 2 },
169             { 0, 1, 3 },
170             { 0, 2, 0 },
171             { 0, 2, 1 },
172             { 0, 2, 2 },
173             { 0, 2, 3 },
174             { 1, 0, 0 },
175             { 1, 0, 1 },
176             { 1, 0, 2 },
177             { 1, 0, 3 },
178             { 1, 1, 0 },
179             { 1, 1, 1 },
180             { 1, 1, 2 },
181             { 1, 1, 3 },
182             { 1, 2, 0 },
183             { 1, 2, 1 },
184             { 1, 2, 2 },
185             { 1, 2, 3 }
186         };
187 
188         final int totalSize = c.getSize();
189         Assert.assertEquals(expected.length, totalSize);
190 
191         final int nDim = c.getDimension();
192         final MultidimensionalCounter.Iterator iter = c.iterator();
193         for (int i = 0; i < totalSize; i++) {
194             if (!iter.hasNext()) {
195                 Assert.fail("Too short");
196             }
197             final int uniDimIndex = iter.next().intValue();
198             Assert.assertEquals("Wrong iteration at " + i, i, uniDimIndex);
199 
200             for (int dimIndex = 0; dimIndex < nDim; dimIndex++) {
201                 Assert.assertEquals("Wrong multidimensional index for [" + i + "][" + dimIndex + "]",
202                                     expected[i][dimIndex], iter.getCount(dimIndex));
203             }
204 
205             Assert.assertEquals("Wrong unidimensional index for [" + i + "]",
206                                 c.getCount(expected[i]), uniDimIndex);
207 
208             final int[] indices = c.getCounts(uniDimIndex);
209             for (int dimIndex = 0; dimIndex < nDim; dimIndex++) {
210                 Assert.assertEquals("Wrong multidimensional index for [" + i + "][" + dimIndex + "]",
211                                     expected[i][dimIndex], indices[dimIndex]);
212             }
213         }
214 
215         if (iter.hasNext()) {
216             Assert.fail("Too long");
217         }
218     }
