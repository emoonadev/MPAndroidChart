17  package org.apache.commons.math4.stat.descriptive;
18  
19  import java.io.Serializable;
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.apache.commons.math4.exception.MathIllegalArgumentException;
24  import org.apache.commons.math4.stat.descriptive.DescriptiveStatistics;
25  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
26  import org.apache.commons.math4.util.DefaultTransformer;
27  import org.apache.commons.math4.util.FastMath;
28  import org.apache.commons.math4.util.NumberTransformer;
29  
30  /**
31   */
32  public class ListUnivariateImpl extends DescriptiveStatistics implements Serializable {
33  
34      /** Serializable version identifier */
35      private static final long serialVersionUID = -8837442489133392138L;
36  
37      /**
38       * Holds a reference to a list - GENERICs are going to make
39       * our lives easier here as we could only accept List<Number>
40       */
41      protected List<Object> list;
42  
43      /** Number Transformer maps Objects to Number for us. */
44      protected NumberTransformer transformer;
45  
46      /**
47       * No argument Constructor
48       */
49      public ListUnivariateImpl(){
50          this(new ArrayList<Object>());
51      }
52  
53      /**
54       * Construct a ListUnivariate with a specific List.
55       * @param list The list that will back this DescriptiveStatistics
56       */
57      public ListUnivariateImpl(List<Object> list) {
58          this(list, new DefaultTransformer());
59      }
60  
61      /**
62       * Construct a ListUnivariate with a specific List.
63       * @param list The list that will back this DescriptiveStatistics
64       * @param transformer the number transformer used to convert the list items.
65       */
66      public ListUnivariateImpl(List<Object> list, NumberTransformer transformer) {
67          super();
68          this.list = list;
69          this.transformer = transformer;
70      }
71  
72      /** {@inheritDoc} */
73      @Override
74      public double[] getValues() {
75  
76          int length = list.size();
77  
78          // If the window size is not INFINITE_WINDOW AND
79          // the current list is larger that the window size, we need to
80          // take into account only the last n elements of the list
81          // as definied by windowSize
82  
83          final int wSize = getWindowSize();
84          if (wSize != DescriptiveStatistics.INFINITE_WINDOW && wSize < list.size()) {
85              length = list.size() - FastMath.max(0, list.size() - wSize);
86          }
87  
88          // Create an array to hold all values
89          double[] copiedArray = new double[length];
90  
91          for (int i = 0; i < copiedArray.length; i++) {
92              copiedArray[i] = getElement(i);
93          }
94          return copiedArray;
95      }
96  
97      /** {@inheritDoc} */
98      @Override
99      public double getElement(int index) {
100 
101         double value = Double.NaN;
102 
103         int calcIndex = index;
104 
105         final int wSize = getWindowSize();
106         if (wSize != DescriptiveStatistics.INFINITE_WINDOW && wSize < list.size()) {
107             calcIndex = (list.size() - wSize) + index;
108         }
109 
110 
111         try {
112             value = transformer.transform(list.get(calcIndex));
113         } catch (MathIllegalArgumentException e) {
114             e.printStackTrace();
115         }
116 
117         return value;
118     }
119 
120     /** {@inheritDoc} */
121     @Override
122     public long getN() {
123         int n = 0;
124 
125         final int wSize = getWindowSize();
126         if (wSize != DescriptiveStatistics.INFINITE_WINDOW) {
127             if (list.size() > wSize) {
128                 n = wSize;
129             } else {
130                 n = list.size();
131             }
132         } else {
133             n = list.size();
134         }
135         return n;
136     }
137 
138     /** {@inheritDoc} */
139     @Override
140     public void addValue(double v) {
141         list.add(Double.valueOf(v));
142     }
143 
144     /**
145      * Adds an object to this list.
146      * @param o Object to add to the list
147      */
148     public void addObject(Object o) {
149         list.add(o);
150     }
151 
152     /**
153      * Clears all statistics.
154      * <p>
155      * <strong>N.B.: </strong> This method has the side effect of clearing the underlying list.
156      */
157     @Override
158     public void clear() {
159         list.clear();
160     }
161 
162     /**
163      * Apply the given statistic to this univariate collection.
164      * @param stat the statistic to apply
165      * @return the computed value of the statistic.
166      */
167     @Override
168     public double apply(UnivariateStatistic stat) {
169         double[] v = this.getValues();
170 
171         if (v != null) {
172             return stat.evaluate(v, 0, v.length);
173         }
174         return Double.NaN;
175     }
176 
177     /**
178      * Access the number transformer.
179      * @return the number transformer.
180      */
181     public NumberTransformer getTransformer() {
182         return transformer;
183     }
184 
185     /**
186      * Modify the number transformer.
187      * @param transformer the new number transformer.
188      */
189     public void setTransformer(NumberTransformer transformer) {
190         this.transformer = transformer;
191     }
192 
193     /** {@inheritDoc} */
194     @Override
195     public void setWindowSize(int windowSize) {
196         super.setWindowSize(windowSize);
197         //Discard elements from the front of the list if the windowSize is less than
198         // the size of the list.
199         int extra = list.size() - windowSize;
200         for (int i = 0; i < extra; i++) {
201             list.remove(0);
202         }
203     }
204 
205 }
