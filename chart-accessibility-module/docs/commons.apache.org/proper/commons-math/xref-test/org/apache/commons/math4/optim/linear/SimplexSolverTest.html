46      public void testMath842Cycle() {
47          // from http://www.math.toronto.edu/mpugh/Teaching/APM236_04/bland
48          //      maximize 10 x1 - 57 x2 - 9 x3 - 24 x4
49          //      subject to
50          //          1/2 x1 - 11/2 x2 - 5/2 x3 + 9 x4  <= 0
51          //          1/2 x1 -  3/2 x2 - 1/2 x3 +   x4  <= 0
52          //              x1                  <= 1
53          //      x1,x2,x3,x4 >= 0
54  
55          LinearObjectiveFunction f = new LinearObjectiveFunction(new double[] { 10, -57, -9, -24}, 0);
56  
57          ArrayList<LinearConstraint> constraints = new ArrayList<LinearConstraint>();
58  
59          constraints.add(new LinearConstraint(new double[] {0.5, -5.5, -2.5, 9}, Relationship.LEQ, 0));
60          constraints.add(new LinearConstraint(new double[] {0.5, -1.5, -0.5, 1}, Relationship.LEQ, 0));
61          constraints.add(new LinearConstraint(new double[] {  1,    0,    0, 0}, Relationship.LEQ, 1));
62  
63          double epsilon = 1e-6;
64          SimplexSolver solver = new SimplexSolver();
65          PointValuePair solution = solver.optimize(f, new LinearConstraintSet(constraints),
66                                                    GoalType.MAXIMIZE,
67                                                    new NonNegativeConstraint(true),
68                                                    PivotSelectionRule.BLAND);
69          Assert.assertEquals(1.0d, solution.getValue(), epsilon);
70          Assert.assertTrue(validSolution(solution, constraints, epsilon));
71      }

822     /**
823      * Converts a test string to a {@link LinearConstraint}.
824      * Ex: x0 + x1 + x2 + x3 - x12 = 0
825      */
826     private LinearConstraint equationFromString(int numCoefficients, String s) {
827         Relationship relationship;
828         if (s.contains(">=")) {
829             relationship = Relationship.GEQ;
830         } else if (s.contains("<=")) {
831             relationship = Relationship.LEQ;
832         } else if (s.contains("=")) {
833             relationship = Relationship.EQ;
834         } else {
835             throw new IllegalArgumentException();
836         }
837 
838         String[] equationParts = s.split("[>|<]?=");
839         double rhs = Double.parseDouble(equationParts[1].trim());
840 
841         double[] lhs = new double[numCoefficients];
842         String left = equationParts[0].replaceAll(" ?x", "");
843         String[] coefficients = left.split(" ");
844         for (String coefficient : coefficients) {
845             double value = coefficient.charAt(0) == '-' ? -1 : 1;
846             int index = Integer.parseInt(coefficient.replaceFirst("[+|-]", "").trim());
847             lhs[index] = value;
848         }
849         return new LinearConstraint(lhs, relationship, rhs);
850     }

852     private static boolean validSolution(PointValuePair solution, List<LinearConstraint> constraints, double epsilon) {
853         double[] vals = solution.getPoint();
854         for (LinearConstraint c : constraints) {
855             double[] coeffs = c.getCoefficients().toArray();
856             double result = 0.0d;
857             for (int i = 0; i < vals.length; i++) {
858                 result += vals[i] * coeffs[i];
859             }
860 
861             switch (c.getRelationship()) {
862             case EQ:
863                 if (!Precision.equals(result, c.getValue(), epsilon)) {
864                     return false;
865                 }
866                 break;
867 
868             case GEQ:
869                 if (Precision.compareTo(result, c.getValue(), epsilon) < 0) {
870                     return false;
871                 }
872                 break;
873 
874             case LEQ:
875                 if (Precision.compareTo(result, c.getValue(), epsilon) > 0) {
876                     return false;
877                 }
878                 break;
879             }
880         }
881 
882         return true;
883     }
