17  package org.apache.commons.math4.stat.interval;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.stat.interval.BinomialConfidenceInterval;
21  import org.apache.commons.math4.stat.interval.ConfidenceInterval;
22  import org.junit.Before;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the BinomialConfidenceInterval implementations.
27   *
28   */
29  public abstract class BinomialConfidenceIntervalAbstractTest {
30  
31      protected BinomialConfidenceInterval testStatistic;
32  
33      private final int successes = 50;
34      private final int trials = 500;
35      private final double confidenceLevel = 0.9;
36  
37      protected abstract BinomialConfidenceInterval createBinomialConfidenceInterval();
38  
39      /**
40       * Returns the confidence interval for the given statistic with the following values:
41       *
42       * <ul>
43       *  <li>trials: 500</li>
44       *  <li>successes: 50</li>
45       *  <li>confidenceLevel: 0.9</li>
46       * </ul>
47       * @return the Confidence Interval for the given values
48       */
49      protected ConfidenceInterval createStandardTestInterval() {
50          return testStatistic.createInterval(trials, successes, confidenceLevel);
51      }
52  
53      @Before
54      public void setUp() {
55          testStatistic = createBinomialConfidenceInterval();
56      }
57  
58      @Test(expected = MathIllegalArgumentException.class)
59      public void testZeroConfidencelevel() {
60          testStatistic.createInterval(trials, successes, 0d);
61      }
62  
63      @Test(expected = MathIllegalArgumentException.class)
64      public void testOneConfidencelevel() {
65          testStatistic.createInterval(trials, successes, 1d);
66      }
67  
68      @Test(expected = MathIllegalArgumentException.class)
69      public void testZeroTrials() {
70          testStatistic.createInterval(0, 0, confidenceLevel);
71      }
72  
73      @Test(expected = MathIllegalArgumentException.class)
74      public void testNegativeSuccesses() {
75          testStatistic.createInterval(trials, -1, confidenceLevel);
76      }
77  
78      @Test(expected = MathIllegalArgumentException.class)
79      public void testSuccessesExceedingTrials() {
80          testStatistic.createInterval(trials, trials + 1, confidenceLevel);
81      }
82  }
