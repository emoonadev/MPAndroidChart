66      @Test
67      public void testInverseCumulativeProbabilitySmallPAccuracy() {
68          WeibullDistribution dist = new WeibullDistribution(2, 3);
69          double t = dist.inverseCumulativeProbability(1e-17);
70          // Analytically, answer is solution to 1e-17 = 1-exp(-(x/3)^2)
71          // x = sqrt(-9*log(1-1e-17))
72          // If we're not careful, answer will be 0. Answer below is computed with care in Octave:
73          Assert.assertEquals(9.48683298050514e-9, t, 1e-17);
74      }

108     @Test
109     public void testMoments() {
110         final double tol = 1e-9;
111         WeibullDistribution dist;
112 
113         dist = new WeibullDistribution(2.5, 3.5);
114         // In R: 3.5*gamma(1+(1/2.5)) (or emperically: mean(rweibull(10000, 2.5, 3.5)))
115         Assert.assertEquals(dist.getNumericalMean(), 3.5 * FastMath.exp(Gamma.logGamma(1 + (1 / 2.5))), tol);
116         Assert.assertEquals(dist.getNumericalVariance(), (3.5 * 3.5) *
117                 FastMath.exp(Gamma.logGamma(1 + (2 / 2.5))) -
118                 (dist.getNumericalMean() * dist.getNumericalMean()), tol);
119 
120         dist = new WeibullDistribution(10.4, 2.222);
121         Assert.assertEquals(dist.getNumericalMean(), 2.222 * FastMath.exp(Gamma.logGamma(1 + (1 / 10.4))), tol);
122         Assert.assertEquals(dist.getNumericalVariance(), (2.222 * 2.222) *
123                 FastMath.exp(Gamma.logGamma(1 + (2 / 10.4))) -
124                 (dist.getNumericalMean() * dist.getNumericalMean()), tol);
125     }
