35      private final double EPS = Math.ulp(1d);

42      @Test
43      public void testSomeValues() {
44          final UnivariateFunction f = new Gaussian();
45  
46          Assert.assertEquals(1 / FastMath.sqrt(2 * Math.PI), f.value(0), EPS);
47      }

49      @Test
50      public void testLargeArguments() {
51          final UnivariateFunction f = new Gaussian();
52  
53          Assert.assertEquals(0, f.value(Double.NEGATIVE_INFINITY), 0);
54          Assert.assertEquals(0, f.value(-Double.MAX_VALUE), 0);
55          Assert.assertEquals(0, f.value(-1e2), 0);
56          Assert.assertEquals(0, f.value(1e2), 0);
57          Assert.assertEquals(0, f.value(Double.MAX_VALUE), 0);
58          Assert.assertEquals(0, f.value(Double.POSITIVE_INFINITY), 0);
59      }
60  
61      @Test
62      public void testDerivatives() {
63          final UnivariateDifferentiableFunction gaussian = new Gaussian(2.0, 0.9, 3.0);
64          final DerivativeStructure dsX = new DerivativeStructure(1, 4, 0, 1.1);
65          final DerivativeStructure dsY = gaussian.value(dsX);
66          Assert.assertEquals( 1.9955604901712128349,   dsY.getValue(),              EPS);
67          Assert.assertEquals(-0.044345788670471396332, dsY.getPartialDerivative(1), EPS);
68          Assert.assertEquals(-0.22074348138190206174,  dsY.getPartialDerivative(2), EPS);
69          Assert.assertEquals( 0.014760030401924800557, dsY.getPartialDerivative(3), EPS);
70          Assert.assertEquals( 0.073253159785035691678, dsY.getPartialDerivative(4), EPS);
71      }
72  
73      @Test
74      public void testDerivativeLargeArguments() {
75          final Gaussian f = new Gaussian(0, 1e-50);
76  
77          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, Double.NEGATIVE_INFINITY)).getPartialDerivative(1), 0);
78          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, -Double.MAX_VALUE)).getPartialDerivative(1), 0);
79          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, -1e50)).getPartialDerivative(1), 0);
80          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, -1e2)).getPartialDerivative(1), 0);
81          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, 1e2)).getPartialDerivative(1), 0);
82          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, 1e50)).getPartialDerivative(1), 0);
83          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, Double.MAX_VALUE)).getPartialDerivative(1), 0);
84          Assert.assertEquals(0, f.value(new DerivativeStructure(1, 1, 0, Double.POSITIVE_INFINITY)).getPartialDerivative(1), 0);
85      }

87      @Test
88      public void testDerivativesNaN() {
89          final Gaussian f = new Gaussian(0, 1e-50);
90          final DerivativeStructure fx = f.value(new DerivativeStructure(1, 5, 0, Double.NaN));
91          for (int i = 0; i <= fx.getOrder(); ++i) {
92              Assert.assertTrue(Double.isNaN(fx.getPartialDerivative(i)));
93          }
94      }

132     @Test
133     public void testParametricValue() {
134         final double norm = 2;
135         final double mean = 3;
136         final double sigma = 4;
137         final Gaussian f = new Gaussian(norm, mean, sigma);
138 
139         final Gaussian.Parametric g = new Gaussian.Parametric();
140         Assert.assertEquals(f.value(-1), g.value(-1, new double[] {norm, mean, sigma}), 0);
141         Assert.assertEquals(f.value(0), g.value(0, new double[] {norm, mean, sigma}), 0);
142         Assert.assertEquals(f.value(2), g.value(2, new double[] {norm, mean, sigma}), 0);
143     }
144 
145     @Test
146     public void testParametricGradient() {
147         final double norm = 2;
148         final double mean = 3;
149         final double sigma = 4;
150         final Gaussian.Parametric f = new Gaussian.Parametric();
151 
152         final double x = 1;
153         final double[] grad = f.gradient(1, new double[] {norm, mean, sigma});
154         final double diff = x - mean;
155         final double n = FastMath.exp(-diff * diff / (2 * sigma * sigma));
156         Assert.assertEquals(n, grad[0], EPS);
157         final double m = norm * n * diff / (sigma * sigma);
158         Assert.assertEquals(m, grad[1], EPS);
159         final double s = m * diff / sigma;
160         Assert.assertEquals(s, grad[2], EPS);
161     }
