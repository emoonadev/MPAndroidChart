106     @Test
107     public void testParse() {
108         String source = "1 / 2";
109 
110         try {
111             Fraction c = properFormat.parse(source);
112             Assert.assertNotNull(c);
113             Assert.assertEquals(1, c.getNumerator());
114             Assert.assertEquals(2, c.getDenominator());
115 
116             c = improperFormat.parse(source);
117             Assert.assertNotNull(c);
118             Assert.assertEquals(1, c.getNumerator());
119             Assert.assertEquals(2, c.getDenominator());
120         } catch (MathParseException ex) {
121             Assert.fail(ex.getMessage());
122         }
123     }

299     @Test
300     public void testNumeratorFormat() {
301         NumberFormat old = properFormat.getNumeratorFormat();
302         NumberFormat nf = NumberFormat.getInstance();
303         nf.setParseIntegerOnly(true);
304         properFormat.setNumeratorFormat(nf);
305         Assert.assertEquals(nf, properFormat.getNumeratorFormat());
306         properFormat.setNumeratorFormat(old);
307 
308         old = improperFormat.getNumeratorFormat();
309         nf = NumberFormat.getInstance();
310         nf.setParseIntegerOnly(true);
311         improperFormat.setNumeratorFormat(nf);
312         Assert.assertEquals(nf, improperFormat.getNumeratorFormat());
313         improperFormat.setNumeratorFormat(old);
314     }
