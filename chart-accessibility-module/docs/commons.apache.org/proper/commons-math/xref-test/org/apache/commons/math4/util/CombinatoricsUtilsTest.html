51      @Test
52      public void testBinomialCoefficient() {
53          long[] bcoef5 = {
54              1,
55              5,
56              10,
57              10,
58              5,
59              1 };
60          long[] bcoef6 = {
61              1,
62              6,
63              15,
64              20,
65              15,
66              6,
67              1 };
68          for (int i = 0; i < 6; i++) {
69              Assert.assertEquals("5 choose " + i, bcoef5[i], CombinatoricsUtils.binomialCoefficient(5, i));
70          }
71          for (int i = 0; i < 7; i++) {
72              Assert.assertEquals("6 choose " + i, bcoef6[i], CombinatoricsUtils.binomialCoefficient(6, i));
73          }
74  
75          for (int n = 1; n < 10; n++) {
76              for (int k = 0; k <= n; k++) {
77                  Assert.assertEquals(n + " choose " + k, binomialCoefficient(n, k), CombinatoricsUtils.binomialCoefficient(n, k));
78                  Assert.assertEquals(n + " choose " + k, binomialCoefficient(n, k), CombinatoricsUtils.binomialCoefficientDouble(n, k), Double.MIN_VALUE);
79                  Assert.assertEquals(n + " choose " + k, FastMath.log(binomialCoefficient(n, k)), CombinatoricsUtils.binomialCoefficientLog(n, k), 10E-12);
80              }
81          }
82  
83          int[] n = { 34, 66, 100, 1500, 1500 };
84          int[] k = { 17, 33, 10, 1500 - 4, 4 };
85          for (int i = 0; i < n.length; i++) {
86              long expected = binomialCoefficient(n[i], k[i]);
87              Assert.assertEquals(n[i] + " choose " + k[i], expected,
88                  CombinatoricsUtils.binomialCoefficient(n[i], k[i]));
89              Assert.assertEquals(n[i] + " choose " + k[i], expected,
90                  CombinatoricsUtils.binomialCoefficientDouble(n[i], k[i]), 0.0);
91              Assert.assertEquals("log(" + n[i] + " choose " + k[i] + ")", FastMath.log(expected),
92                  CombinatoricsUtils.binomialCoefficientLog(n[i], k[i]), 0.0);
93          }
94      }

155     /**
156      * Tests correctness for large n and sharpness of upper bound in API doc
157      * JIRA: MATH-241
158      */
159     @Test
160     public void testBinomialCoefficientLarge() throws Exception {
161         // This tests all legal and illegal values for n <= 200.
162         for (int n = 0; n <= 200; n++) {
163             for (int k = 0; k <= n; k++) {
164                 long ourResult = -1;
165                 long exactResult = -1;
166                 boolean shouldThrow = false;
167                 boolean didThrow = false;
168                 try {
169                     ourResult = CombinatoricsUtils.binomialCoefficient(n, k);
170                 } catch (MathArithmeticException ex) {
171                     didThrow = true;
172                 }
173                 try {
174                     exactResult = binomialCoefficient(n, k);
175                 } catch (MathArithmeticException ex) {
176                     shouldThrow = true;
177                 }
178                 Assert.assertEquals(n + " choose " + k, exactResult, ourResult);
179                 Assert.assertEquals(n + " choose " + k, shouldThrow, didThrow);
180                 Assert.assertTrue(n + " choose " + k, (n > 66 || !didThrow));
181 
182                 if (!shouldThrow && exactResult > 1) {
183                     Assert.assertEquals(n + " choose " + k, 1.,
184                         CombinatoricsUtils.binomialCoefficientDouble(n, k) / exactResult, 1e-10);
185                     Assert.assertEquals(n + " choose " + k, 1,
186                         CombinatoricsUtils.binomialCoefficientLog(n, k) / FastMath.log(exactResult), 1e-10);
187                 }
188             }
189         }
190 
191         long ourResult = CombinatoricsUtils.binomialCoefficient(300, 3);
192         long exactResult = binomialCoefficient(300, 3);
193         Assert.assertEquals(exactResult, ourResult);
194 
195         ourResult = CombinatoricsUtils.binomialCoefficient(700, 697);
196         exactResult = binomialCoefficient(700, 697);
197         Assert.assertEquals(exactResult, ourResult);
198 
199         // This one should throw
200         try {
201             CombinatoricsUtils.binomialCoefficient(700, 300);
202             Assert.fail("Expecting MathArithmeticException");
203         } catch (MathArithmeticException ex) {
204             // Expected
205         }
206 
207         int n = 10000;
208         ourResult = CombinatoricsUtils.binomialCoefficient(n, 3);
209         exactResult = binomialCoefficient(n, 3);
210         Assert.assertEquals(exactResult, ourResult);
211         Assert.assertEquals(1, CombinatoricsUtils.binomialCoefficientDouble(n, 3) / exactResult, 1e-10);
212         Assert.assertEquals(1, CombinatoricsUtils.binomialCoefficientLog(n, 3) / FastMath.log(exactResult), 1e-10);
213 
214     }

333     /**
334      * Exact (caching) recursive implementation to test against
335      */
336     private long binomialCoefficient(int n, int k) throws MathArithmeticException {
337         if (binomialCache.size() > n) {
338             Long cachedResult = binomialCache.get(n).get(Integer.valueOf(k));
339             if (cachedResult != null) {
340                 return cachedResult.longValue();
341             }
342         }
343         long result = -1;
344         if ((n == k) || (k == 0)) {
345             result = 1;
346         } else if ((k == 1) || (k == n - 1)) {
347             result = n;
348         } else {
349             // Reduce stack depth for larger values of n
350             if (k < n - 100) {
351                 binomialCoefficient(n - 100, k);
352             }
353             if (k > 100) {
354                 binomialCoefficient(n - 100, k - 100);
355             }
356             result = ArithmeticUtils.addAndCheck(binomialCoefficient(n - 1, k - 1),
357                 binomialCoefficient(n - 1, k));
358         }
359         if (result == -1) {
360             throw new MathArithmeticException();
361         }
362         for (int i = binomialCache.size(); i < n + 1; i++) {
363             binomialCache.add(new HashMap<Integer, Long>());
364         }
365         binomialCache.get(n).put(Integer.valueOf(k), Long.valueOf(result));
366         return result;
367     }

369     /**
370      * Exact direct multiplication implementation to test against
371      */
372     private long factorial(int n) {
373         long result = 1;
374         for (int i = 2; i <= n; i++) {
375             result *= i;
376         }
377         return result;
378     }
