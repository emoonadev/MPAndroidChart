85      public void testSmallDf() {
86          setDistribution(new ChiSquaredDistribution(0.1d));
87          setTolerance(1E-4);
88          // quantiles computed using R version 1.8.1 (linux version)
89          setCumulativeTestPoints(new double[] {1.168926E-60, 1.168926E-40, 1.063132E-32,

91                  0.5318646, 0.1526342});
92          setInverseCumulativeTestValues(getCumulativeTestPoints());
93          setInverseCumulativeTestPoints(getCumulativeTestValues());

95          verifyInverseCumulativeProbabilities();

104     @Test
105     public void testDensity() {
106         double[] x = new double[]{-0.1, 1e-6, 0.5, 1, 2, 5};
107         //R 2.5: print(dchisq(x, df=1), digits=10)
108         checkDensity(1, x, new double[]{0.00000000000, 398.94208093034, 0.43939128947, 0.24197072452, 0.10377687436, 0.01464498256});
109         //R 2.5: print(dchisq(x, df=0.1), digits=10)
110         checkDensity(0.1, x, new double[]{0.000000000e+00, 2.486453997e+04, 7.464238732e-02, 3.009077718e-02, 9.447299159e-03, 8.827199396e-04});
111         //R 2.5: print(dchisq(x, df=2), digits=10)
112         checkDensity(2, x, new double[]{0.00000000000, 0.49999975000, 0.38940039154, 0.30326532986, 0.18393972059, 0.04104249931});
113         //R 2.5: print(dchisq(x, df=10), digits=10)
114         checkDensity(10, x, new double[]{0.000000000e+00, 1.302082682e-27, 6.337896998e-05, 7.897534632e-04, 7.664155024e-03, 6.680094289e-02});
115     }
116 
117     private void checkDensity(double df, double[] x, double[] expected) {
118         ChiSquaredDistribution d = new ChiSquaredDistribution(df);
119         for (int i = 0; i < x.length; i++) {
120             Assert.assertEquals(expected[i], d.density(x[i]), 1e-5);
121         }
122     }

124     @Test
125     public void testMoments() {
126         final double tol = 1e-9;
127         ChiSquaredDistribution dist;
128 
129         dist = new ChiSquaredDistribution(1500);
130         Assert.assertEquals(dist.getNumericalMean(), 1500, tol);
131         Assert.assertEquals(dist.getNumericalVariance(), 3000, tol);
132 
133         dist = new ChiSquaredDistribution(1.12);
134         Assert.assertEquals(dist.getNumericalMean(), 1.12, tol);
135         Assert.assertEquals(dist.getNumericalVariance(), 2.24, tol);
136     }
