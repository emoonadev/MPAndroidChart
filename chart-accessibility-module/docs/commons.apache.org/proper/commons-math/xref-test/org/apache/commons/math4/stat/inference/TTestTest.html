48      @Test
49      public void testOneSampleT() {
50          double[] observed =
51              {93.0, 103.0, 95.0, 101.0, 91.0, 105.0, 96.0, 94.0, 101.0,  88.0, 98.0, 94.0, 101.0, 92.0, 95.0 };
52          double mu = 100.0;
53          SummaryStatistics sampleStats = null;
54          sampleStats = new SummaryStatistics();
55          for (int i = 0; i < observed.length; i++) {
56              sampleStats.addValue(observed[i]);
57          }
58  
59          // Target comparison values computed using R version 1.8.1 (Linux version)
60          Assert.assertEquals("t statistic",  -2.81976445346,
61                  testStatistic.t(mu, observed), 10E-10);
62          Assert.assertEquals("t statistic",  -2.81976445346,
63                  testStatistic.t(mu, sampleStats), 10E-10);
64          Assert.assertEquals("p value", 0.0136390585873,
65                  testStatistic.tTest(mu, observed), 10E-10);
66          Assert.assertEquals("p value", 0.0136390585873,
67                  testStatistic.tTest(mu, sampleStats), 10E-10);
68  
69          try {
70              testStatistic.t(mu, (double[]) null);
71              Assert.fail("arguments too short, NullArgumentException expected");
72          } catch (NullArgumentException ex) {
73              // expected
74          }
75  
76          try {
77              testStatistic.t(mu, (SummaryStatistics) null);
78              Assert.fail("arguments too short, NullArgumentException expected");
79          } catch (NullArgumentException ex) {
80              // expected

124     @Test
125     public void testOneSampleTTest() {
126         double[] oneSidedP =
127             {2d, 0d, 6d, 6d, 3d, 3d, 2d, 3d, -6d, 6d, 6d, 6d, 3d, 0d, 1d, 1d, 0d, 2d, 3d, 3d };
128         SummaryStatistics oneSidedPStats = new SummaryStatistics();
129         for (int i = 0; i < oneSidedP.length; i++) {
130             oneSidedPStats.addValue(oneSidedP[i]);
131         }
132         // Target comparison values computed using R version 1.8.1 (Linux version)
133         Assert.assertEquals("one sample t stat", 3.86485535541,
134                 testStatistic.t(0d, oneSidedP), 10E-10);
135         Assert.assertEquals("one sample t stat", 3.86485535541,
136                 testStatistic.t(0d, oneSidedPStats),1E-10);
137         Assert.assertEquals("one sample p value", 0.000521637019637,
138                 testStatistic.tTest(0d, oneSidedP) / 2d, 10E-10);
139         Assert.assertEquals("one sample p value", 0.000521637019637,
140                 testStatistic.tTest(0d, oneSidedPStats) / 2d, 10E-5);
141         Assert.assertTrue("one sample t-test reject", testStatistic.tTest(0d, oneSidedP, 0.01));
142         Assert.assertTrue("one sample t-test reject", testStatistic.tTest(0d, oneSidedPStats, 0.01));
143         Assert.assertTrue("one sample t-test accept", !testStatistic.tTest(0d, oneSidedP, 0.0001));
144         Assert.assertTrue("one sample t-test accept", !testStatistic.tTest(0d, oneSidedPStats, 0.0001));
145 
146         try {
147             testStatistic.tTest(0d, oneSidedP, 95);
148             Assert.fail("alpha out of range, OutOfRangeException expected");
149         } catch (OutOfRangeException ex) {
150             // expected
151         }
152 
153         try {
154             testStatistic.tTest(0d, oneSidedPStats, 95);
155             Assert.fail("alpha out of range, OutOfRangeException expected");
156         } catch (OutOfRangeException ex) {
157             // expected
158         }
159 
160     }

162     @Test
163     public void testTwoSampleTHeterscedastic() {
164         double[] sample1 = { 7d, -4d, 18d, 17d, -3d, -5d, 1d, 10d, 11d, -2d };
165         double[] sample2 = { -1d, 12d, -1d, -3d, 3d, -5d, 5d, 2d, -11d, -1d, -3d };
166         SummaryStatistics sampleStats1 = new SummaryStatistics();
167         for (int i = 0; i < sample1.length; i++) {
168             sampleStats1.addValue(sample1[i]);
169         }
170         SummaryStatistics sampleStats2 = new SummaryStatistics();
171         for (int i = 0; i < sample2.length; i++) {
172             sampleStats2.addValue(sample2[i]);
173         }
174 
175         // Target comparison values computed using R version 1.8.1 (Linux version)
176         Assert.assertEquals("two sample heteroscedastic t stat", 1.60371728768,
177                 testStatistic.t(sample1, sample2), 1E-10);
178         Assert.assertEquals("two sample heteroscedastic t stat", 1.60371728768,
179                 testStatistic.t(sampleStats1, sampleStats2), 1E-10);
180         Assert.assertEquals("two sample heteroscedastic p value", 0.128839369622,
181                 testStatistic.tTest(sample1, sample2), 1E-10);
182         Assert.assertEquals("two sample heteroscedastic p value", 0.128839369622,
183                 testStatistic.tTest(sampleStats1, sampleStats2), 1E-10);
184         Assert.assertTrue("two sample heteroscedastic t-test reject",
185                 testStatistic.tTest(sample1, sample2, 0.2));

249     @Test
250     public void testTwoSampleTHomoscedastic() {
251         double[] sample1 ={2, 4, 6, 8, 10, 97};
252         double[] sample2 = {4, 6, 8, 10, 16};
253         SummaryStatistics sampleStats1 = new SummaryStatistics();
254         for (int i = 0; i < sample1.length; i++) {
255             sampleStats1.addValue(sample1[i]);
256         }
257         SummaryStatistics sampleStats2 = new SummaryStatistics();
258         for (int i = 0; i < sample2.length; i++) {
259             sampleStats2.addValue(sample2[i]);
260         }
261 
262         // Target comparison values computed using R version 1.8.1 (Linux version)
263         Assert.assertEquals("two sample homoscedastic t stat", 0.73096310086,
264               testStatistic.homoscedasticT(sample1, sample2), 10E-11);
265         Assert.assertEquals("two sample homoscedastic p value", 0.4833963785,
266                 testStatistic.homoscedasticTTest(sampleStats1, sampleStats2), 1E-10);
267         Assert.assertTrue("two sample homoscedastic t-test reject",
268                 testStatistic.homoscedasticTTest(sample1, sample2, 0.49));
269         Assert.assertTrue("two sample homoscedastic t-test accept",
270                 !testStatistic.homoscedasticTTest(sample1, sample2, 0.48));
271     }

285     @Test
286     public void testPaired() {
287         double[] sample1 = {1d, 3d, 5d, 7d};
288         double[] sample2 = {0d, 6d, 11d, 2d};
289         double[] sample3 = {5d, 7d, 8d, 10d};
290 
291         // Target values computed using R, version 1.8.1 (linux version)
292         Assert.assertEquals(-0.3133, testStatistic.pairedT(sample1, sample2), 1E-4);
293         Assert.assertEquals(0.774544295819, testStatistic.pairedTTest(sample1, sample2), 1E-10);
294         Assert.assertEquals(0.001208, testStatistic.pairedTTest(sample1, sample3), 1E-6);
295         Assert.assertFalse(testStatistic.pairedTTest(sample1, sample3, .001));
296         Assert.assertTrue(testStatistic.pairedTTest(sample1, sample3, .002));
