18  package org.apache.commons.math4.analysis.solvers;
19  
20  import org.apache.commons.math4.analysis.UnivariateFunction;
21  import org.apache.commons.math4.analysis.solvers.RegulaFalsiSolver;
22  import org.apache.commons.math4.analysis.solvers.UnivariateSolver;
23  import org.apache.commons.math4.exception.ConvergenceException;
24  import org.apache.commons.math4.util.FastMath;
25  import org.junit.Test;
26  import org.junit.Assert;
27  
28  /**
29   * Test case for {@link RegulaFalsiSolver Regula Falsi} solver.
30   *
31   */
32  public final class RegulaFalsiSolverTest extends BaseSecantSolverAbstractTest {
33      /** {@inheritDoc} */
34      @Override
35      protected UnivariateSolver getSolver() {
36          return new RegulaFalsiSolver();
37      }
38  
39      /** {@inheritDoc} */
40      @Override
41      protected int[] getQuinticEvalCounts() {
42          // While the Regula Falsi method guarantees convergence, convergence
43          // may be extremely slow. The last test case does not converge within
44          // even a million iterations. As such, it was disabled.
45          return new int[] {3, 7, 8, 19, 18, 11, 67, 55, 288, 151, -1};
46      }
47  
48      @Test(expected=ConvergenceException.class)
49      public void testIssue631() {
50          final UnivariateFunction f = new UnivariateFunction() {
51                  /** {@inheritDoc} */
52                  public double value(double x) {
53                      return FastMath.exp(x) - FastMath.pow(Math.PI, 3.0);
54                  }
55              };
56  
57          final UnivariateSolver solver = new RegulaFalsiSolver();
58          final double root = solver.solve(3624, f, 1, 10);
59          Assert.assertEquals(3.4341896575482003, root, 1e-15);
60      }
61  }
