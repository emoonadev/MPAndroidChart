18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.linear.CholeskyDecomposition;
21  import org.apache.commons.math4.linear.MatrixUtils;
22  import org.apache.commons.math4.linear.NonPositiveDefiniteMatrixException;
23  import org.apache.commons.math4.linear.NonSquareMatrixException;
24  import org.apache.commons.math4.linear.NonSymmetricMatrixException;
25  import org.apache.commons.math4.linear.RealMatrix;
26  import org.junit.Test;
27  import org.junit.Assert;
28  
29  public class CholeskyDecompositionTest {
30  
31      private double[][] testData = new double[][] {
32              {  1,  2,   4,   7,  11 },
33              {  2, 13,  23,  38,  58 },
34              {  4, 23,  77, 122, 182 },
35              {  7, 38, 122, 294, 430 },
36              { 11, 58, 182, 430, 855 }
37      };
38  
39      /** test dimensions */
40      @Test
41      public void testDimensions() {
42          CholeskyDecomposition llt =
43              new CholeskyDecomposition(MatrixUtils.createRealMatrix(testData));
44          Assert.assertEquals(testData.length, llt.getL().getRowDimension());
45          Assert.assertEquals(testData.length, llt.getL().getColumnDimension());
46          Assert.assertEquals(testData.length, llt.getLT().getRowDimension());
47          Assert.assertEquals(testData.length, llt.getLT().getColumnDimension());
48      }
49  
50      /** test non-square matrix */
51      @Test(expected = NonSquareMatrixException.class)
52      public void testNonSquare() {
53          new CholeskyDecomposition(MatrixUtils.createRealMatrix(new double[3][2]));
54      }
55  
56      /** test non-symmetric matrix */
57      @Test(expected = NonSymmetricMatrixException.class)
58      public void testNotSymmetricMatrixException() {
59          double[][] changed = testData.clone();
60          changed[0][changed[0].length - 1] += 1.0e-5;
61          new CholeskyDecomposition(MatrixUtils.createRealMatrix(changed));
62      }
63  
64      /** test non positive definite matrix */
65      @Test(expected = NonPositiveDefiniteMatrixException.class)
66      public void testNotPositiveDefinite() {
67          new CholeskyDecomposition(MatrixUtils.createRealMatrix(new double[][] {
68                  { 14, 11, 13, 15, 24 },
69                  { 11, 34, 13, 8,  25 },
70                  { 13, 13, 14, 15, 21 },
71                  { 15, 8,  15, 18, 23 },
72                  { 24, 25, 21, 23, 45 }
73          }));
74      }
75  
76      @Test(expected = NonPositiveDefiniteMatrixException.class)
77      public void testMath274() {
78          new CholeskyDecomposition(MatrixUtils.createRealMatrix(new double[][] {
79                  { 0.40434286, -0.09376327, 0.30328980, 0.04909388 },
80                  {-0.09376327,  0.10400408, 0.07137959, 0.04762857 },
81                  { 0.30328980,  0.07137959, 0.30458776, 0.04882449 },
82                  { 0.04909388,  0.04762857, 0.04882449, 0.07543265 }
83  
84          }));
85      }
86  
87      /** test A = LLT */
88      @Test
89      public void testAEqualLLT() {
90          RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
91          CholeskyDecomposition llt = new CholeskyDecomposition(matrix);
92          RealMatrix l  = llt.getL();
93          RealMatrix lt = llt.getLT();
94          double norm = l.multiply(lt).subtract(matrix).getNorm();
95          Assert.assertEquals(0, norm, 1.0e-15);
96      }
97  
98      /** test that L is lower triangular */
99      @Test
100     public void testLLowerTriangular() {
101         RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
102         RealMatrix l = new CholeskyDecomposition(matrix).getL();
103         for (int i = 0; i < l.getRowDimension(); i++) {
104             for (int j = i + 1; j < l.getColumnDimension(); j++) {
105                 Assert.assertEquals(0.0, l.getEntry(i, j), 0.0);
106             }
107         }
108     }
109 
110     /** test that LT is transpose of L */
111     @Test
112     public void testLTTransposed() {
113         RealMatrix matrix = MatrixUtils.createRealMatrix(testData);
114         CholeskyDecomposition llt = new CholeskyDecomposition(matrix);
115         RealMatrix l  = llt.getL();
116         RealMatrix lt = llt.getLT();
117         double norm = l.subtract(lt.transpose()).getNorm();
118         Assert.assertEquals(0, norm, 1.0e-15);
119     }
120 
121     /** test matrices values */
122     @Test
123     public void testMatricesValues() {
124         RealMatrix lRef = MatrixUtils.createRealMatrix(new double[][] {
125                 {  1,  0,  0,  0,  0 },
126                 {  2,  3,  0,  0,  0 },
127                 {  4,  5,  6,  0,  0 },
128                 {  7,  8,  9, 10,  0 },
129                 { 11, 12, 13, 14, 15 }
130         });
131        CholeskyDecomposition llt =
132             new CholeskyDecomposition(MatrixUtils.createRealMatrix(testData));
133 
134         // check values against known references
135         RealMatrix l = llt.getL();
136         Assert.assertEquals(0, l.subtract(lRef).getNorm(), 1.0e-13);
137         RealMatrix lt = llt.getLT();
138         Assert.assertEquals(0, lt.subtract(lRef.transpose()).getNorm(), 1.0e-13);
139 
140         // check the same cached instance is returned the second time
141         Assert.assertTrue(l  == llt.getL());
142         Assert.assertTrue(lt == llt.getLT());
143     }
144 }
