18  package org.apache.commons.math4.distribution;
19  
20  import org.apache.commons.math4.distribution.UniformRealDistribution;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for UniformRealDistribution. See class javadoc for
27   * {@link RealDistributionAbstractTest} for further details.
28   */
29  public class UniformRealDistributionTest extends RealDistributionAbstractTest {
30  
31      // --- Override tolerance -------------------------------------------------
32  
33      @Override
34      public void setUp() {
35          super.setUp();
36          setTolerance(1e-4);
37      }
38  
39      //--- Implementations for abstract methods --------------------------------
40  
41      /** Creates the default uniform real distribution instance to use in tests. */
42      @Override
43      public UniformRealDistribution makeDistribution() {
44          return new UniformRealDistribution(-0.5, 1.25);
45      }
46  
47      /** Creates the default cumulative probability distribution test input values */
48      @Override
49      public double[] makeCumulativeTestPoints() {
50          return new double[] {-0.5001, -0.5, -0.4999, -0.25, -0.0001, 0.0,
51                               0.0001, 0.25, 1.0, 1.2499, 1.25, 1.2501};
52      }
53  
54      /** Creates the default cumulative probability density test expected values */
55      @Override
56      public double[] makeCumulativeTestValues() {
57          return new double[] {0.0, 0.0, 0.0001, 0.25/1.75, 0.4999/1.75,
58                               0.5/1.75, 0.5001/1.75, 0.75/1.75, 1.5/1.75,
59                               1.7499/1.75, 1.0, 1.0};
60      }
61  
62      /** Creates the default probability density test expected values */
63      @Override
64      public double[] makeDensityTestValues() {
65          double d = 1 / 1.75;
66          return new double[] {0, d, d, d, d, d, d, d, d, d, d, 0};
67      }
68  
69      //--- Additional test cases -----------------------------------------------
70  
71      /** Test lower bound getter. */
72      @Test
73      public void testGetLowerBound() {
74          UniformRealDistribution distribution = makeDistribution();
75          Assert.assertEquals(-0.5, distribution.getSupportLowerBound(), 0);
76      }
77  
78      /** Test upper bound getter. */
79      @Test
80      public void testGetUpperBound() {
81          UniformRealDistribution distribution = makeDistribution();
82          Assert.assertEquals(1.25, distribution.getSupportUpperBound(), 0);
83      }
84  
85      /** Test pre-condition for equal lower/upper bound. */
86      @Test(expected=NumberIsTooLargeException.class)
87      public void testPreconditions1() {
88          new UniformRealDistribution(0, 0);
89      }
90  
91      /** Test pre-condition for lower bound larger than upper bound. */
92      @Test(expected=NumberIsTooLargeException.class)
93      public void testPreconditions2() {
94          new UniformRealDistribution(1, 0);
95      }
96  
97      /** Test mean/variance. */
98      @Test
99      public void testMeanVariance() {
100         UniformRealDistribution dist;
101 
102         dist = new UniformRealDistribution(0, 1);
103         Assert.assertEquals(dist.getNumericalMean(), 0.5, 0);
104         Assert.assertEquals(dist.getNumericalVariance(), 1/12.0, 0);
105 
106         dist = new UniformRealDistribution(-1.5, 0.6);
107         Assert.assertEquals(dist.getNumericalMean(), -0.45, 0);
108         Assert.assertEquals(dist.getNumericalVariance(), 0.3675, 0);
109 
110         dist = new UniformRealDistribution(-0.5, 1.25);
111         Assert.assertEquals(dist.getNumericalMean(), 0.375, 0);
112         Assert.assertEquals(dist.getNumericalVariance(), 0.2552083333333333, 0);
113     }
114 
115     /**
116      * Check accuracy of analytical inverse CDF. Fails if a solver is used
117      * with the default accuracy.
118      */
119     @Test
120     public void testInverseCumulativeDistribution() {
121         UniformRealDistribution dist = new UniformRealDistribution(0, 1e-9);
122 
123         Assert.assertEquals(2.5e-10, dist.inverseCumulativeProbability(0.25), 0);
124     }
125 }
