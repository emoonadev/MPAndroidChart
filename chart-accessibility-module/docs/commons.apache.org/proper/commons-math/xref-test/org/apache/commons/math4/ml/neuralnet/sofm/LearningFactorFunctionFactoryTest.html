18  package org.apache.commons.math4.ml.neuralnet.sofm;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.exception.OutOfRangeException;
23  import org.apache.commons.math4.ml.neuralnet.sofm.LearningFactorFunction;
24  import org.apache.commons.math4.ml.neuralnet.sofm.LearningFactorFunctionFactory;
25  import org.junit.Test;
26  import org.junit.Assert;
27  
28  /**
29   * Tests for {@link LearningFactorFunctionFactory} class.
30   */
31  public class LearningFactorFunctionFactoryTest {
32      @Test(expected=OutOfRangeException.class)
33      public void testExponentialDecayPrecondition0() {
34          LearningFactorFunctionFactory.exponentialDecay(0d, 0d, 2);
35      }
36      @Test(expected=OutOfRangeException.class)
37      public void testExponentialDecayPrecondition1() {
38          LearningFactorFunctionFactory.exponentialDecay(1 + 1e-10, 0d, 2);
39      }
40      @Test(expected=NotStrictlyPositiveException.class)
41      public void testExponentialDecayPrecondition2() {
42          LearningFactorFunctionFactory.exponentialDecay(1d, 0d, 2);
43      }
44      @Test(expected=NumberIsTooLargeException.class)
45      public void testExponentialDecayPrecondition3() {
46          LearningFactorFunctionFactory.exponentialDecay(1d, 1d, 100);
47      }
48      @Test(expected=NotStrictlyPositiveException.class)
49      public void testExponentialDecayPrecondition4() {
50          LearningFactorFunctionFactory.exponentialDecay(1d, 0.2, 0);
51      }
52  
53      @Test
54      public void testExponentialDecayTrivial() {
55          final int n = 65;
56          final double init = 0.5;
57          final double valueAtN = 0.1;
58          final LearningFactorFunction f
59              = LearningFactorFunctionFactory.exponentialDecay(init, valueAtN, n);
60  
61          Assert.assertEquals(init, f.value(0), 0d);
62          Assert.assertEquals(valueAtN, f.value(n), 0d);
63          Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
64      }
65  
66      @Test(expected=OutOfRangeException.class)
67      public void testQuasiSigmoidDecayPrecondition0() {
68          LearningFactorFunctionFactory.quasiSigmoidDecay(0d, -1d, 2);
69      }
70      @Test(expected=OutOfRangeException.class)
71      public void testQuasiSigmoidDecayPrecondition1() {
72          LearningFactorFunctionFactory.quasiSigmoidDecay(1 + 1e-10, -1d, 2);
73      }
74      @Test(expected=NumberIsTooLargeException.class)
75      public void testQuasiSigmoidDecayPrecondition3() {
76          LearningFactorFunctionFactory.quasiSigmoidDecay(1d, 0d, 100);
77      }
78      @Test(expected=NotStrictlyPositiveException.class)
79      public void testQuasiSigmoidDecayPrecondition4() {
80          LearningFactorFunctionFactory.quasiSigmoidDecay(1d, -1d, 0);
81      }
82  
83      @Test
84      public void testQuasiSigmoidDecayTrivial() {
85          final int n = 65;
86          final double init = 0.5;
87          final double slope = -1e-1;
88          final LearningFactorFunction f
89              = LearningFactorFunctionFactory.quasiSigmoidDecay(init, slope, n);
90  
91          Assert.assertEquals(init, f.value(0), 0d);
92          // Very approximate derivative.
93          Assert.assertEquals(slope, f.value(n) - f.value(n - 1), 1e-2);
94          Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
95      }
96  }
