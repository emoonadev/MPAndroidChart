18  package org.apache.commons.math4.ode.nonstiff;

20  
21  import org.apache.commons.math4.Field;
22  import org.apache.commons.math4.RealFieldElement;
23  import org.apache.commons.math4.util.Decimal64Field;
24  
25  public class HighamHall54FieldIntegratorTest extends EmbeddedRungeKuttaFieldIntegratorAbstractTest {
26  
27      protected <T extends RealFieldElement<T>> EmbeddedRungeKuttaFieldIntegrator<T>
28      createIntegrator(Field<T> field, final double minStep, final double maxStep,
29                       final double scalAbsoluteTolerance, final double scalRelativeTolerance) {
30          return new HighamHall54FieldIntegrator<T>(field, minStep, maxStep, scalAbsoluteTolerance, scalRelativeTolerance);
31      }
32  
33      protected <T extends RealFieldElement<T>> EmbeddedRungeKuttaFieldIntegrator<T>
34      createIntegrator(Field<T> field, final double minStep, final double maxStep,
35                       final double[] vecAbsoluteTolerance, final double[] vecRelativeTolerance) {
36          return new HighamHall54FieldIntegrator<T>(field, minStep, maxStep, vecAbsoluteTolerance, vecRelativeTolerance);
37      }
38  
39      @Override
40      public void testNonFieldIntegratorConsistency() {
41          doTestNonFieldIntegratorConsistency(Decimal64Field.getInstance());
42      }
43  
44      @Override
45      public void testSanityChecks() {
46          doTestSanityChecks(Decimal64Field.getInstance());
47      }
48  
49      @Override
50      public void testBackward() {
51          doTestBackward(Decimal64Field.getInstance(), 5.0e-7, 5.0e-7, 1.0e-12, "Higham-Hall 5(4)");
52      }
53  
54      @Override
55      public void testKepler() {
56          doTestKepler(Decimal64Field.getInstance(), 1.5e-4);
57      }
58  
59      @Override
60      public void testForwardBackwardExceptions() {
61          doTestForwardBackwardExceptions(Decimal64Field.getInstance());
62      }
63  
64      @Override
65      public void testMinStep() {
66          doTestMinStep(Decimal64Field.getInstance());
67      }
68  
69      @Override
70      public void testIncreasingTolerance() {
71          // the 1.3 factor is only valid for this test
72          // and has been obtained from trial and error
73          // there is no general relation between local and global errors
74          doTestIncreasingTolerance(Decimal64Field.getInstance(), 1.3, 1.0e-12);
75      }
76  
77      @Override
78      public void testEvents() {
79          doTestEvents(Decimal64Field.getInstance(), 1.0e-7, "Higham-Hall 5(4)");
80      }
81  
82      @Override
83      public void testEventsErrors() {
84          doTestEventsErrors(Decimal64Field.getInstance());
85      }
86  
87      @Override
88      public void testEventsNoConvergence() {
89          doTestEventsNoConvergence(Decimal64Field.getInstance());
90      }
91  
92      @Override
93      public void testPartialDerivatives() {
94          doTestPartialDerivatives(1.2e-11, new double[] { 6.4e-11, 1.8e-11, 2.4e-12, 2.2e-12, 1.8e-11 });
95      }
