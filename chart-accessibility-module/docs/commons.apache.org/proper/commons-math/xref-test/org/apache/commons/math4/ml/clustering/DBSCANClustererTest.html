139     @Test
140     public void testSingleLink() {
141         final DoublePoint[] points = {
142                 new DoublePoint(new int[] {10, 10}), // A
143                 new DoublePoint(new int[] {12, 9}),
144                 new DoublePoint(new int[] {10, 8}),
145                 new DoublePoint(new int[] {8, 8}),
146                 new DoublePoint(new int[] {8, 6}),
147                 new DoublePoint(new int[] {7, 7}),
148                 new DoublePoint(new int[] {5, 6}),  // B
149                 new DoublePoint(new int[] {14, 8}), // C
150                 new DoublePoint(new int[] {7, 15}), // N - Noise, should not be present
151                 new DoublePoint(new int[] {17, 8}), // D - single-link connected to C should not be present
152 
153         };
154 
155         final DBSCANClusterer<DoublePoint> clusterer = new DBSCANClusterer<DoublePoint>(3, 3);
156         List<Cluster<DoublePoint>> clusters = clusterer.cluster(Arrays.asList(points));
157 
158         Assert.assertEquals(1, clusters.size());
159 
160         final List<DoublePoint> clusterOne =
161                 Arrays.asList(points[0], points[1], points[2], points[3], points[4], points[5], points[6], points[7]);
162         Assert.assertTrue(clusters.get(0).getPoints().containsAll(clusterOne));
163     }
