35    @Test
36    public void testNoReset() throws MaxCountExceededException {
37  
38      double[]   y    =   { 0.0, 1.0, -2.0 };
39      DummyStepInterpolator interpolator = new DummyStepInterpolator(y, new double[y.length], true);
40      interpolator.storeTime(0);
41      interpolator.shift();
42      interpolator.storeTime(1);
43  
44      double[] result = interpolator.getInterpolatedState();
45      for (int i = 0; i < result.length; ++i) {
46        Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
47      }
48  
49    }

51    @Test
52    public void testFixedState() throws MaxCountExceededException {
53  
54      double[]   y    =   { 1.0, 3.0, -4.0 };
55      DummyStepInterpolator interpolator = new DummyStepInterpolator(y, new double[y.length], true);
56      interpolator.storeTime(0);
57      interpolator.shift();
58      interpolator.storeTime(1);
59  
60      interpolator.setInterpolatedTime(0.1);
61      double[] result = interpolator.getInterpolatedState();
62      for (int i = 0; i < result.length; ++i) {
63          Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
64      }
65  
66      interpolator.setInterpolatedTime(0.5);
67      result = interpolator.getInterpolatedState();
68      for (int i = 0; i < result.length; ++i) {
69          Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
70      }
71  
72    }

74    @Test
75    public void testSerialization()
76    throws IOException, ClassNotFoundException, MaxCountExceededException {
77  
78      double[]   y    =   { 0.0, 1.0, -2.0 };
79      DummyStepInterpolator interpolator = new DummyStepInterpolator(y, new double[y.length], true);
80      interpolator.storeTime(0);
81      interpolator.shift();
82      interpolator.storeTime(1);
83  
84      ByteArrayOutputStream bos = new ByteArrayOutputStream();
85      ObjectOutputStream    oos = new ObjectOutputStream(bos);
86      oos.writeObject(interpolator);
87  
88      Assert.assertTrue(bos.size () > 300);
89      Assert.assertTrue(bos.size () < 500);
90  
91      ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
92      ObjectInputStream     ois = new ObjectInputStream(bis);
93      DummyStepInterpolator dsi = (DummyStepInterpolator) ois.readObject();
94  
95      dsi.setInterpolatedTime(0.5);
96      double[] result = dsi.getInterpolatedState();
97      for (int i = 0; i < result.length; ++i) {
98          Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
99      }
100 
101   }

103   @Test
104   public void testImpossibleSerialization()
105   throws IOException {
106 
107     double[] y = { 0.0, 1.0, -2.0 };
108     AbstractStepInterpolator interpolator = new BadStepInterpolator(y, true);
109     interpolator.storeTime(0);
110     interpolator.shift();
111     interpolator.storeTime(1);
112 
113     ByteArrayOutputStream bos = new ByteArrayOutputStream();
114     ObjectOutputStream    oos = new ObjectOutputStream(bos);
115     try {
116         oos.writeObject(interpolator);
117         Assert.fail("an exception should have been thrown");
118     } catch (LocalException le) {
119         // expected behavior
120     }
121   }
122 
123   private static class BadStepInterpolator extends DummyStepInterpolator {
124       public BadStepInterpolator(double[] y, boolean forward) {
125           super(y, new double[y.length], forward);
126       }
127       @Override
128       protected void doFinalize() {
129           throw new LocalException();
130       }
131   }
132 
133   private static class LocalException extends RuntimeException {
134     private static final long serialVersionUID = 1L;
135   }
