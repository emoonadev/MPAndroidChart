188     @Test
189     public void testPolar2Complex() {
190         TestUtils.assertEquals(Complex.ONE, ComplexUtils.polar2Complex(1, 0), 10e-12);
191         TestUtils.assertEquals(Complex.ZERO, ComplexUtils.polar2Complex(0, 1), 10e-12);
192         TestUtils.assertEquals(Complex.ZERO, ComplexUtils.polar2Complex(0, -1), 10e-12);
193         TestUtils.assertEquals(Complex.I, ComplexUtils.polar2Complex(1, pi / 2), 10e-12);
194         TestUtils.assertEquals(Complex.I.negate(), ComplexUtils.polar2Complex(1, -pi / 2), 10e-12);
195         double r = 0;
196         for (int i = 0; i < 5; i++) {
197             r += i;
198             double theta = 0;
199             for (int j = 0; j < 20; j++) {
200                 theta += pi / 6;
201                 TestUtils.assertEquals(altPolar(r, theta), ComplexUtils.polar2Complex(r, theta), 10e-12);
202             }
203             theta = -2 * pi;
204             for (int j = 0; j < 20; j++) {
205                 theta -= pi / 6;
206                 TestUtils.assertEquals(altPolar(r, theta), ComplexUtils.polar2Complex(r, theta), 10e-12);
207             }
208         }
209     }
210 
211     protected Complex altPolar(double r, double theta) {
212         return Complex.I.multiply(new Complex(theta, 0)).exp().multiply(new Complex(r, 0));
213     }
