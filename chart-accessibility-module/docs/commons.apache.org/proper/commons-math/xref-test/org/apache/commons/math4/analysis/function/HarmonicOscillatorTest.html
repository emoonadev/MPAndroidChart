33  public class HarmonicOscillatorTest {
34      private final double EPS = Math.ulp(1d);
35  
36      @Test
37      public void testSomeValues() {
38          final double a = -1.2;
39          final double w = 0.34;
40          final double p = 5.6;
41          final UnivariateFunction f = new HarmonicOscillator(a, w, p);
42  
43          final double d = 0.12345;
44          for (int i = 0; i < 10; i++) {
45              final double v = i * d;
46              Assert.assertEquals(a * FastMath.cos(w * v + p), f.value(v), 0);
47          }
48      }
49  
50      @Test
51      public void testDerivative() {
52          final double a = -1.2;
53          final double w = 0.34;
54          final double p = 5.6;
55          final HarmonicOscillator f = new HarmonicOscillator(a, w, p);
56  
57          for (int maxOrder = 0; maxOrder < 6; ++maxOrder) {
58              final double d = 0.12345;
59              for (int i = 0; i < 10; i++) {
60                  final double v = i * d;
61                  final DerivativeStructure h = f.value(new DerivativeStructure(1, maxOrder, 0, v));
62                  for (int k = 0; k <= maxOrder; ++k) {
63                      final double trigo;
64                      switch (k % 4) {
65                          case 0:
66                              trigo = +FastMath.cos(w * v + p);
67                              break;
68                          case 1:
69                              trigo = -FastMath.sin(w * v + p);
70                              break;
71                          case 2:
72                              trigo = -FastMath.cos(w * v + p);
73                              break;
74                          default:
75                              trigo = +FastMath.sin(w * v + p);
76                              break;
77                      }
78                      Assert.assertEquals(a * FastMath.pow(w, k) * trigo,
79                                          h.getPartialDerivative(k),
80                                          Precision.EPSILON);
81                  }
82              }
83          }
84      }

110     @Test
111     public void testParametricValue() {
112         final double amplitude = 2;
113         final double omega = 3;
114         final double phase = 4;
115         final HarmonicOscillator f = new HarmonicOscillator(amplitude, omega, phase);
116 
117         final HarmonicOscillator.Parametric g = new HarmonicOscillator.Parametric();
118         Assert.assertEquals(f.value(-1), g.value(-1, new double[] {amplitude, omega, phase}), 0);
119         Assert.assertEquals(f.value(0), g.value(0, new double[] {amplitude, omega, phase}), 0);
120         Assert.assertEquals(f.value(2), g.value(2, new double[] {amplitude, omega, phase}), 0);
121     }
122 
123     @Test
124     public void testParametricGradient() {
125         final double amplitude = 2;
126         final double omega = 3;
127         final double phase = 4;
128         final HarmonicOscillator.Parametric f = new HarmonicOscillator.Parametric();
129 
130         final double x = 1;
131         final double[] grad = f.gradient(1, new double[] {amplitude, omega, phase});
132         final double xTimesOmegaPlusPhase = omega * x + phase;
133         final double a = FastMath.cos(xTimesOmegaPlusPhase);
134         Assert.assertEquals(a, grad[0], EPS);
135         final double w = -amplitude * x * FastMath.sin(xTimesOmegaPlusPhase);
136         Assert.assertEquals(w, grad[1], EPS);
137         final double p = -amplitude * FastMath.sin(xTimesOmegaPlusPhase);
138         Assert.assertEquals(p, grad[2], EPS);
139     }
