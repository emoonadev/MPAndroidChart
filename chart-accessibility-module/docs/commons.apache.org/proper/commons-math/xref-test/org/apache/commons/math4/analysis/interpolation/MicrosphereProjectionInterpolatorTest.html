17  package org.apache.commons.math4.analysis.interpolation;
18  
19  import org.apache.commons.math4.analysis.MultivariateFunction;
20  import org.apache.commons.math4.util.FastMath;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test case for the {@link MicrosphereProjectionInterpolator
26   * "microsphere projection"} interpolator.
27   */
28  public final class MicrosphereProjectionInterpolatorTest {
29      /**
30       * Test of interpolator for a plane.
31       * <p>
32       * y = 2 x<sub>1</sub> - 3 x<sub>2</sub> + 5
33       */
34      @Test
35      public void testLinearFunction2D() {
36          MultivariateFunction f = new MultivariateFunction() {
37              @Override
38              public double value(double[] x) {
39                  if (x.length != 2) {
40                      throw new IllegalArgumentException();
41                  }
42                  return 2 * x[0] - 3 * x[1] + 5;
43              }
44          };
45  
46          final double darkFraction = 0.5;
47          final double darkThreshold = 1e-2;
48          final double background = Double.NaN;
49          final double exponent = 1.1;
50          final boolean shareSphere = true;
51          final double noInterpolationTolerance = Math.ulp(1d);
52  
53          // N-dimensional interpolator.
54          final MultivariateInterpolator interpolator
55              = new MicrosphereProjectionInterpolator(2, 500,
56                                                      darkFraction,
57                                                      darkThreshold,
58                                                      background,
59                                                      exponent,
60                                                      shareSphere,
61                                                      noInterpolationTolerance);
62  
63          // 2D interpolator.
64          final MultivariateInterpolator interpolator2D
65              = new MicrosphereProjectionInterpolator(new InterpolatingMicrosphere2D(16,
66                                                                                     darkFraction,
67                                                                                     darkThreshold,
68                                                                                     background),
69                                                      exponent,
70                                                      shareSphere,
71                                                      noInterpolationTolerance);
72  
73          final double min = -1;
74          final double max = 1;
75          final double range = max - min;
76          final int res = 5;
77          final int n = res * res; // Number of sample points.
78          final int dim = 2;
79          double[][] x = new double[n][dim];
80          double[] y = new double[n];
81          int index = 0;
82          for (int i = 0; i < res; i++) {
83              final double x1Val = toCoordinate(min, range, res, i);
84              for (int j = 0; j < res; j++) {
85                  final double x2Val = toCoordinate(min, range, res, j);
86                  x[index][0] = x1Val;
87                  x[index][1] = x2Val;
88                  y[index] = f.value(x[index]);
89                  ++index;
90              }
91          }
92  
93          final MultivariateFunction p = interpolator.interpolate(x, y);
94          final MultivariateFunction p2D = interpolator2D.interpolate(x, y);
95  
96          double[] c = new double[dim];
97          double expected, result, result2D;
98  
99          final int sampleIndex = 2;
100         c[0] = x[sampleIndex][0];
101         c[1] = x[sampleIndex][1];
102         expected = f.value(c);
103         result = p.value(c);
104         result2D = p2D.value(c);
105         Assert.assertEquals("on sample point (exact)", expected, result2D, FastMath.ulp(1d));
106         Assert.assertEquals("on sample point (ND vs 2D)", result2D, result, FastMath.ulp(1d));
107 
108         // Interpolation.
109         c[0] = 0.654321;
110         c[1] = -0.345678;
111         expected = f.value(c);
112         result = p.value(c);
113         result2D = p2D.value(c);
114         Assert.assertEquals("interpolation (exact)", expected, result2D, 1e-1);
115         Assert.assertEquals("interpolation (ND vs 2D)", result2D, result, 1e-1);
116 
117         // Extrapolation.
118         c[0] = 0 - 1e-2;
119         c[1] = 1 + 1e-2;
120         expected = f.value(c);
121         result = p.value(c);
122         result2D = p2D.value(c);
123         Assert.assertFalse(Double.isNaN(result));
124         Assert.assertFalse(Double.isNaN(result2D));
125         Assert.assertEquals("extrapolation (exact)", expected, result2D, 1e-1);
126         Assert.assertEquals("extrapolation (ND vs 2D)", result2D, result, 1e-2);
127 
128         // Far away.
129         c[0] = 20;
130         c[1] = -30;
131         result = p.value(c);
132         Assert.assertTrue(result + " should be NaN", Double.isNaN(result));
133         result2D = p2D.value(c);
134         Assert.assertTrue(result2D + " should be NaN", Double.isNaN(result2D));
135     }
136 
137     /**
138      * @param min Minimum of the coordinate range.
139      * @param range Extent of the coordinate interval.
140      * @param res Number of pixels.
141      * @param pixel Pixel index.
142      */
143     private static double toCoordinate(double min,
144                                        double range,
145                                        int res,
146                                        int pixel) {
147         return pixel * range / (res - 1) + min;
148     }
149 }
