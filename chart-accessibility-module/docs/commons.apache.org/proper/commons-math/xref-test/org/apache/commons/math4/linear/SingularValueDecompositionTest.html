97      @Test
98      public void testHadamard() {
99          RealMatrix matrix = new Array2DRowRealMatrix(new double[][] {
100                 {15.0 / 2.0,  5.0 / 2.0,  9.0 / 2.0,  3.0 / 2.0 },
101                 { 5.0 / 2.0, 15.0 / 2.0,  3.0 / 2.0,  9.0 / 2.0 },
102                 { 9.0 / 2.0,  3.0 / 2.0, 15.0 / 2.0,  5.0 / 2.0 },
103                 { 3.0 / 2.0,  9.0 / 2.0,  5.0 / 2.0, 15.0 / 2.0 }
104         }, false);
105         SingularValueDecomposition svd = new SingularValueDecomposition(matrix);
106         Assert.assertEquals(16.0, svd.getSingularValues()[0], 1.0e-14);
107         Assert.assertEquals( 8.0, svd.getSingularValues()[1], 1.0e-14);
108         Assert.assertEquals( 4.0, svd.getSingularValues()[2], 1.0e-14);
109         Assert.assertEquals( 2.0, svd.getSingularValues()[3], 1.0e-14);
110 
111         RealMatrix fullCovariance = new Array2DRowRealMatrix(new double[][] {
112                 {  85.0 / 1024, -51.0 / 1024, -75.0 / 1024,  45.0 / 1024 },
113                 { -51.0 / 1024,  85.0 / 1024,  45.0 / 1024, -75.0 / 1024 },
114                 { -75.0 / 1024,  45.0 / 1024,  85.0 / 1024, -51.0 / 1024 },
115                 {  45.0 / 1024, -75.0 / 1024, -51.0 / 1024,  85.0 / 1024 }
116         }, false);
117         Assert.assertEquals(0.0,
118                      fullCovariance.subtract(svd.getCovariance(0.0)).getNorm(),
119                      1.0e-14);
120 
121         RealMatrix halfCovariance = new Array2DRowRealMatrix(new double[][] {
122                 {   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024 },
123                 {  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024 },
124                 {   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024 },
125                 {  -3.0 / 1024,   5.0 / 1024,  -3.0 / 1024,   5.0 / 1024 }
126         }, false);
127         Assert.assertEquals(0.0,
128                      halfCovariance.subtract(svd.getCovariance(6.0)).getNorm(),
129                      1.0e-14);

141     public void checkAEqualUSVt(final RealMatrix matrix) {
142         SingularValueDecomposition svd = new SingularValueDecomposition(matrix);
143         RealMatrix u = svd.getU();
144         RealMatrix s = svd.getS();
145         RealMatrix v = svd.getV();
146         double norm = u.multiply(s).multiply(v.transpose()).subtract(matrix).getNorm();
147         Assert.assertEquals(0, norm, normTolerance);
148 
149     }

167     public void checkOrthogonal(final RealMatrix m) {
168         RealMatrix mTm = m.transpose().multiply(m);
169         RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
170         Assert.assertEquals(0, mTm.subtract(id).getNorm(), normTolerance);

179         RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
180                 { 3.0 / 5.0, -4.0 / 5.0 },

198         Assert.assertEquals(0, v.subtract(vRef).getNorm(), normTolerance);
199 
200         // check the same cached instance is returned the second time
201         Assert.assertTrue(u == svd.getU());
202         Assert.assertTrue(s == svd.getS());
203         Assert.assertTrue(v == svd.getV());

279     private void loadRealMatrix(RealMatrix m, String resourceName) {
280         try {
281             DataInputStream in = new DataInputStream(getClass().getResourceAsStream(resourceName));
282             BufferedReader br = new BufferedReader(new InputStreamReader(in));
283             String strLine;
284             int row = 0;
285             while ((strLine = br.readLine()) != null) {
286                 if (!strLine.startsWith("#")) {
287                     int col = 0;
288                     for (String entry : strLine.split(",")) {
289                         m.setEntry(row, col++, Double.parseDouble(entry));
290                     }
291                     row++;
292                 }
293             }
294             in.close();
295         } catch (IOException e) {}
296     }

314     private RealMatrix createTestMatrix(final Random r, final int rows, final int columns,
315                                         final double[] singularValues) {
316         final RealMatrix u = EigenDecompositionTest.createOrthogonalMatrix(r, rows);
317         final RealMatrix d = new Array2DRowRealMatrix(rows, columns);
318         d.setSubMatrix(MatrixUtils.createRealDiagonalMatrix(singularValues).getData(), 0, 0);
319         final RealMatrix v = EigenDecompositionTest.createOrthogonalMatrix(r, columns);
320         return u.multiply(d).multiply(v);
321     }
