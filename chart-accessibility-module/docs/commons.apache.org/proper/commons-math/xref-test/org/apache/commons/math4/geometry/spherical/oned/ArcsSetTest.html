77      @Test
78      public void testSplitOver2Pi() {
79          ArcsSet set = new ArcsSet(1.0e-10);
80          Arc     arc = new Arc(1.5 * FastMath.PI, 2.5 * FastMath.PI, 1.0e-10);
81          ArcsSet.Split split = set.split(arc);
82          for (double alpha = 0.0; alpha <= MathUtils.TWO_PI; alpha += 0.01) {
83              S1Point p = new S1Point(alpha);
84              if (alpha < 0.5 * FastMath.PI || alpha > 1.5 * FastMath.PI) {
85                  Assert.assertEquals(Location.OUTSIDE, split.getPlus().checkPoint(p));
86                  Assert.assertEquals(Location.INSIDE,  split.getMinus().checkPoint(p));
87              } else {
88                  Assert.assertEquals(Location.INSIDE,  split.getPlus().checkPoint(p));
89                  Assert.assertEquals(Location.OUTSIDE, split.getMinus().checkPoint(p));
90              }
91          }
92      }
