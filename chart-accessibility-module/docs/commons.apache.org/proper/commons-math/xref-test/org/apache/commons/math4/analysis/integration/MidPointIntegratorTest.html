30  /**
31   * Test case for midpoint integrator.
32   * <p>
33   * Test runs show that for a default relative accuracy of 1E-6, it generally
34   * takes 10 to 15 iterations for the integral to converge.
35   *
36   */
37  public final class MidPointIntegratorTest {
38  
39      /**
40       * Test of integrator for the sine function.
41       */
42      @Test
43      public void testLowAccuracy() {
44          UnivariateFunction f = new QuinticFunction();
45          UnivariateIntegrator integrator = new MidPointIntegrator(0.01, 1.0e-10, 2, 4);
46  
47          double min = -10;
48          double max =  -9;
49          double expected = -3697001.0 / 48.0;
50          double tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
51          double result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
52          Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
53          Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
54          Assert.assertEquals(expected, result, tolerance);
55  
56      }
57  
58      /**
59       * Test of integrator for the sine function.
60       */
61      @Test
62      public void testSinFunction() {
63          UnivariateFunction f = new Sin();
64          UnivariateIntegrator integrator = new MidPointIntegrator();
65  
66          double min = 0;
67          double max = FastMath.PI;
68          double expected = 2;
69          double tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
70          double result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
71          Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
72          Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
73          Assert.assertEquals(expected, result, tolerance);
74  
75          min = -FastMath.PI/3;
76          max = 0;
77          expected = -0.5;
78          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
79          result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
80          Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
81          Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
82          Assert.assertEquals(expected, result, tolerance);
83  
84      }
85  
86      /**
87       * Test of integrator for the quintic function.
88       */
89      @Test
90      public void testQuinticFunction() {
91          UnivariateFunction f = new QuinticFunction();
92          UnivariateIntegrator integrator = new MidPointIntegrator();
93  
94          double min = 0;
95          double max = 1;
96          double expected = -1.0 / 48;
97          double tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
98          double result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
99          Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
100         Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
101         Assert.assertEquals(expected, result, tolerance);
102 
103         min = 0;
104         max = 0.5;
105         expected = 11.0 / 768;
106         tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
107         result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
108         Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
109         Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
110         Assert.assertEquals(expected, result, tolerance);
111 
112         min = -1;
113         max = 4;
114         expected = 2048 / 3.0 - 78 + 1.0 / 48;
115         tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
116         result = integrator.integrate(Integer.MAX_VALUE, f, min, max);
117         Assert.assertTrue(integrator.getEvaluations() < Integer.MAX_VALUE / 2);
118         Assert.assertTrue(integrator.getIterations() < MidPointIntegrator.MIDPOINT_MAX_ITERATIONS_COUNT / 2);
119         Assert.assertEquals(expected, result, tolerance);
120 
121     }

123     /**
124      * Test of parameters for the integrator.
125      */
126     @Test
127     public void testParameters() {
128         UnivariateFunction f = new Sin();
129 
130         try {
131             // bad interval
132             new MidPointIntegrator().integrate(1000, f, 1, -1);
133             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
134         } catch (NumberIsTooLargeException ex) {
135             // expected
136         }
137         try {
138             // bad iteration limits
139             new MidPointIntegrator(5, 4);
140             Assert.fail("Expecting NumberIsTooSmallException - bad iteration limits");
141         } catch (NumberIsTooSmallException ex) {
142             // expected
143         }
144         try {
145             // bad iteration limits
146             new MidPointIntegrator(10, 99);
147             Assert.fail("Expecting NumberIsTooLargeException - bad iteration limits");
148         } catch (NumberIsTooLargeException ex) {
149             // expected
150         }
151     }
