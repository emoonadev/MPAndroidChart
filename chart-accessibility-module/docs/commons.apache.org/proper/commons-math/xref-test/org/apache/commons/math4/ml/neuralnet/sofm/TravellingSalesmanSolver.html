51  /**
52   * Solves the "Travelling Salesman's Problem" (i.e. trying to find the
53   * sequence of cities that minimizes the travel distance) using a 1D
54   * SOFM.
55   */
56  public class TravellingSalesmanSolver {
57      private static final long FIRST_NEURON_ID = 0;
58      /** RNG. */
59      private final UniformRandomProvider random;
60      /** Set of cities. */
61      private final Set<City> cities = new HashSet<City>();
62      /** SOFM. */
63      private final Network net;
64      /** Distance function. */
65      private final DistanceMeasure distance = new EuclideanDistance();
66      /** Total number of neurons. */
67      private final int numberOfNeurons;
68  
69      /**
70       * @param cityList List of cities to visit in a single travel.
71       * @param numNeuronsPerCity Number of neurons per city.
72       */
73      public TravellingSalesmanSolver(City[] cityList,
74                                      double numNeuronsPerCity) {
75          this(cityList, numNeuronsPerCity, RandomSource.createLong());
76      }
77  
78      /**
79       * @param cityList List of cities to visit in a single travel.
80       * @param numNeuronsPerCity Number of neurons per city.
81       * @param seed Seed for the RNG that is used to present the samples
82       * to the trainer.
83       */
84      public TravellingSalesmanSolver(City[] cityList,
85                                      double numNeuronsPerCity,
86                                      long seed) {
87          random = RandomSource.create(RandomSource.WELL_1024_A, seed);
88  
89          // Make sure that each city will appear only once in the list.
90          for (City city : cityList) {
91              cities.add(city);
92          }
93  
94          // Total number of neurons.
95          numberOfNeurons = (int) numNeuronsPerCity * cities.size();
96  
97          // Create a network with circle topology.
98          net = new NeuronString(numberOfNeurons, true, makeInitializers()).getNetwork();
99      }
100 
101     /**
102      * Creates training tasks.
103      *
104      * @param numTasks Number of tasks to create.
105      * @param numSamplesPerTask Number of training samples per task.
106      * @return the created tasks.
107      */
108     public Runnable[] createParallelTasks(int numTasks,
109                                           long numSamplesPerTask) {
110         final Runnable[] tasks = new Runnable[numTasks];
111         final LearningFactorFunction learning
112             = LearningFactorFunctionFactory.exponentialDecay(2e-1,
113                                                              5e-2,
114                                                              numSamplesPerTask / 2);
115         final NeighbourhoodSizeFunction neighbourhood
116             = NeighbourhoodSizeFunctionFactory.exponentialDecay(0.5 * numberOfNeurons,
117                                                                 0.1 * numberOfNeurons,
118                                                                 numSamplesPerTask / 2);
119 
120         for (int i = 0; i < numTasks; i++) {
121             final KohonenUpdateAction action = new KohonenUpdateAction(distance,
122                                                                        learning,
123                                                                        neighbourhood);
124             tasks[i] = new KohonenTrainingTask(net,
125                                                createRandomIterator(numSamplesPerTask),
126                                                action);
127         }
128 
129         return tasks;
130     }
131 
132     /**
133      * Creates a training task.
134      *
135      * @param numSamples Number of training samples.
136      * @return the created task.
137      */
138     public Runnable createSequentialTask(long numSamples) {
139         return createParallelTasks(1, numSamples)[0];
140     }

142     /**
143      * Measures the network's concurrent update performance.
144      *
145      * @return the ratio between the number of succesful network updates
146      * and the number of update attempts.
147      */
148     public double getUpdateRatio() {
149         return computeUpdateRatio(net);
150     }
151 
152     /**
153      * Measures the network's concurrent update performance.
154      *
155      * @param net Network to be trained with the SOFM algorithm.
156      * @return the ratio between the number of successful network updates
157      * and the number of update attempts.
158      */
159     private static double computeUpdateRatio(Network net) {
160         long numAttempts = 0;
161         long numSuccesses = 0;
162 
163         for (Neuron n : net) {
164             numAttempts += n.getNumberOfAttemptedUpdates();
165             numSuccesses += n.getNumberOfSuccessfulUpdates();
166         }
167 
168         return (double) numSuccesses / (double) numAttempts;
169     }

171     /**
172      * Creates an iterator that will present a series of city's coordinates in
173      * a random order.
174      *
175      * @param numSamples Number of samples.
176      * @return the iterator.
177      */
178     private Iterator<double[]> createRandomIterator(final long numSamples) {
179         final List<City> cityList = new ArrayList<City>();
180         cityList.addAll(cities);
181 
182         return new Iterator<double[]>() {
183             /** Number of samples. */
184             private long n = 0;
185             /** {@inheritDoc} */
186             public boolean hasNext() {
187                 return n < numSamples;
188             }
189             /** {@inheritDoc} */
190             public double[] next() {
191                 ++n;
192                 return cityList.get(random.nextInt(cityList.size())).getCoordinates();
193             }
194             /** {@inheritDoc} */
195             public void remove() {
196                 throw new MathUnsupportedOperationException();
197             }
198         };
199     }
200 
201     /**
202      * @return the list of linked neurons (i.e. the one-dimensional
203      * SOFM).
204      */
205     private List<Neuron> getNeuronList() {
206         // Sequence of coordinates.
207         final List<Neuron> list = new ArrayList<Neuron>();
208 
209         // First neuron.
210         Neuron current = net.getNeuron(FIRST_NEURON_ID);
211         while (true) {
212             list.add(current);
213             final Collection<Neuron> neighbours
214                 = net.getNeighbours(current, list);
215 
216             final Iterator<Neuron> iter = neighbours.iterator();
217             if (!iter.hasNext()) {
218                 // All neurons have been visited.
219                 break;
220             }
221 
222             current = iter.next();
223         }
224 
225         return list;
226     }
227 
228     /**
229      * @return the list of features (coordinates) of linked neurons.
230      */
231     public List<double[]> getCoordinatesList() {
232         // Sequence of coordinates.
233         final List<double[]> coordinatesList = new ArrayList<double[]>();
234 
235         for (Neuron n : getNeuronList()) {
236             coordinatesList.add(n.getFeatures());
237         }
238 
239         return coordinatesList;
240     }

242     /**
243      * Returns the travel proposed by the solver.
244      * Note: cities can be missing or duplicated.
245      *
246      * @return the list of cities in travel order.
247      */
248     public City[] getCityList() {
249         final List<double[]> coord = getCoordinatesList();
250         final City[] cityList = new City[coord.size()];
251         for (int i = 0; i < cityList.length; i++) {
252             final double[] c = coord.get(i);
253             cityList[i] = getClosestCity(c[0], c[1]);
254         }
255         return cityList;
256     }
257 
258     /**
259      * @param x x-coordinate.
260      * @param y y-coordinate.
261      * @return the city whose coordinates are closest to {@code (x, y)}.
262      */
263     public City getClosestCity(double x,
264                                double y) {
265         City closest = null;
266         double min = Double.POSITIVE_INFINITY;
267         for (City c : cities) {
268             final double d = c.distance(x, y);
269             if (d < min) {
270                 min = d;
271                 closest = c;
272             }
273         }
274         return closest;
275     }

277     /**
278      * Computes the barycentre of all city locations.
279      *
280      * @param cities City list.
281      * @return the barycentre.
282      */
283     private static double[] barycentre(Set<City> cities) {
284         double xB = 0;
285         double yB = 0;
286 
287         int count = 0;
288         for (City c : cities) {
289             final double[] coord = c.getCoordinates();
290             xB += coord[0];
291             yB += coord[1];
292 
293             ++count;
294         }
295 
296         return new double[] { xB / count, yB / count };
297     }
298 
299     /**
300      * Computes the largest distance between the point at coordinates
301      * {@code (x, y)} and any of the cities.
302      *
303      * @param x x-coodinate.
304      * @param y y-coodinate.
305      * @param cities City list.
306      * @return the largest distance.
307      */
308     private static double largestDistance(double x,
309                                           double y,
310                                           Set<City> cities) {
311         double maxDist = 0;
312         for (City c : cities) {
313             final double dist = c.distance(x, y);
314             if (dist > maxDist) {
315                 maxDist = dist;
316             }
317         }
318 
319         return maxDist;
320     }
321 
322     /**
323      * Creates the features' initializers: an approximate circle around the
324      * barycentre of the cities.
325      *
326      * @return an array containing the two initializers.
327      */
328     private FeatureInitializer[] makeInitializers() {
329         // Barycentre.
330         final double[] centre = barycentre(cities);
331         // Largest distance from centre.
332         final double radius = 0.5 * largestDistance(centre[0], centre[1], cities);
333 
334         final double omega = 2 * Math.PI / numberOfNeurons;
335         final UnivariateFunction h1 = new HarmonicOscillator(radius, omega, 0);
336         final UnivariateFunction h2 = new HarmonicOscillator(radius, omega, 0.5 * Math.PI);
337 
338         final UnivariateFunction f1 = FunctionUtils.add(h1, new Constant(centre[0]));
339         final UnivariateFunction f2 = FunctionUtils.add(h2, new Constant(centre[1]));
340 
341         final RealDistribution u = new UniformRealDistribution(-0.05 * radius, 0.05 * radius);
342 
343         return new FeatureInitializer[] {
344             FeatureInitializerFactory.randomize(u.createSampler(random),
345                                                 FeatureInitializerFactory.function(f1, 0, 1)),
346             FeatureInitializerFactory.randomize(u.createSampler(random),
347                                                 FeatureInitializerFactory.function(f2, 0, 1))
348         };
349     }

352 /**
353  * A city, represented by a name and two-dimensional coordinates.
354  */
355 class City {
356     /** Identifier. */
357     final String name;
358     /** x-coordinate. */
359     final double x;
360     /** y-coordinate. */
361     final double y;
362 
363     /**
364      * @param name Name.
365      * @param x Cartesian x-coordinate.
366      * @param y Cartesian y-coordinate.
367      */
368     public City(String name,
369                 double x,
370                 double y) {
371         this.name = name;
372         this.x = x;
373         this.y = y;
374     }
375 
376     /**
377      * @retun the name.
378      */
379     public String getName() {
380         return name;
381     }
382 
383     /**
384      * @return the (x, y) coordinates.
385      */
386     public double[] getCoordinates() {
387         return new double[] { x, y };
388     }
389 
390     /**
391      * Computes the distance between this city and
392      * the given point.
393      *
394      * @param x x-coodinate.
395      * @param y y-coodinate.
396      * @return the distance between {@code (x, y)} and this
397      * city.
398      */
399     public double distance(double x,
400                            double y) {
401         final double xDiff = this.x - x;
402         final double yDiff = this.y - y;
403 
404         return FastMath.sqrt(xDiff * xDiff + yDiff * yDiff);
405     }
406 
407     /** {@inheritDoc} */
408     public boolean equals(Object o) {
409         if (o instanceof City) {
410             final City other = (City) o;
411             return x == other.x &&
412                 y == other.y;
413         }
414         return false;
415     }
416 
417     /** {@inheritDoc} */
418     public int hashCode() {
419         int result = 17;
420 
421         final long c1 = Double.doubleToLongBits(x);
422         result = 31 * result + (int) (c1 ^ (c1 >>> 32));
423 
424         final long c2 = Double.doubleToLongBits(y);
425         result = 31 * result + (int) (c2 ^ (c2 >>> 32));
426 
427         return result;
428     }
