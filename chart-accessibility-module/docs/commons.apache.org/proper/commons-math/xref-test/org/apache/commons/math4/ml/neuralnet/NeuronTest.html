34      @Test
35      public void testGetIdentifier() {
36          final long id = 1234567;
37          final Neuron n = new Neuron(id, new double[] { 0 });
38  
39          Assert.assertEquals(id, n.getIdentifier());
40      }

42      @Test
43      public void testGetSize() {
44          final double[] features = { -1, -1e-97, 0, 23.456, 9.01e203 } ;
45          final Neuron n = new Neuron(1, features);
46          Assert.assertEquals(features.length, n.getSize());
47      }

49      @Test
50      public void testGetFeatures() {
51          final double[] features = { -1, -1e-97, 0, 23.456, 9.01e203 } ;
52          final Neuron n = new Neuron(1, features);
53  
54          final double[] f = n.getFeatures();
55          // Accessor returns a copy.
56          Assert.assertFalse(f == features);
57  
58          // Values are the same.
59          Assert.assertEquals(features.length, f.length);
60          for (int i = 0; i < features.length; i++) {
61              Assert.assertEquals(features[i], f[i], 0d);
62          }
63      }
64  
65      @Test
66      public void testCompareAndSetFeatures() {
67          final Neuron n = new Neuron(1, new double[] { 0 });
68          double[] expect = n.getFeatures();
69          double[] update = new double[] { expect[0] + 1.23 };
70  
71          // Test "success".
72          boolean ok = n.compareAndSetFeatures(expect, update);
73          // Check that the update is reported as successful.
74          Assert.assertTrue(ok);
75          // Check that the new value is correct.
76          Assert.assertEquals(update[0],  n.getFeatures()[0], 0d);
77  
78          // Test "failure".
79          double[] update1 = new double[] { update[0] + 4.56 };
80          // Must return "false" because the neuron has been
81          // updated: a new update can only succeed if "expect"
82          // is set to the new features.
83          ok = n.compareAndSetFeatures(expect, update1);
84          // Check that the update is reported as failed.
85          Assert.assertFalse(ok);
86          // Check that the value was not changed.
87          Assert.assertEquals(update[0],  n.getFeatures()[0], 0d);
88      }
89  
90      @Test
91      public void testCopy() {
92          final Neuron n = new Neuron(1, new double[] { 9.87 });
93  
94          // Update original.
95          double[] update = new double[] { n.getFeatures()[0] + 2.34 };
96          n.compareAndSetFeatures(n.getFeatures(), update);
97  
98          // Create a copy.
99          final Neuron copy = n.copy();
100 
101         // Check that original and copy have the same value.
102         Assert.assertTrue(n.getFeatures()[0] == copy.getFeatures()[0]);
103         Assert.assertEquals(n.getNumberOfAttemptedUpdates(),
104                             copy.getNumberOfAttemptedUpdates());
105 
106         // Update original.
107         update = new double[] { 1.23 * n.getFeatures()[0] };
108         n.compareAndSetFeatures(n.getFeatures(), update);
109 
110         // Check that original and copy differ.
111         Assert.assertFalse(n.getFeatures()[0] == copy.getFeatures()[0]);
112         Assert.assertNotEquals(n.getNumberOfSuccessfulUpdates(),
113                                copy.getNumberOfSuccessfulUpdates());
114     }

116     @Test
117     public void testSerialize()
118         throws IOException,
119                ClassNotFoundException {
120         final Neuron out = new Neuron(123, new double[] { -98.76, -1, 0, 1e-23, 543.21, 1e234 });
121         final ByteArrayOutputStream bos = new ByteArrayOutputStream();
122         final ObjectOutputStream oos = new ObjectOutputStream(bos);
123         oos.writeObject(out);
124 
125         final ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
126         final ObjectInputStream ois = new ObjectInputStream(bis);
127         final Neuron in = (Neuron) ois.readObject();
128 
129         // Same identifier.
130         Assert.assertEquals(out.getIdentifier(),
131                             in.getIdentifier());
132         // Same values.
133         final double[] outF = out.getFeatures();
134         final double[] inF = in.getFeatures();
135         Assert.assertEquals(outF.length, inF.length);
136         for (int i = 0; i < outF.length; i++) {
137             Assert.assertEquals(outF[i], inF[i], 0d);
138         }
139     }
