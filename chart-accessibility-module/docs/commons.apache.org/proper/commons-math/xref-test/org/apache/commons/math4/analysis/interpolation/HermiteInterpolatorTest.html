32      @Test
33      public void testZero() {
34          HermiteInterpolator interpolator = new HermiteInterpolator();
35          interpolator.addSamplePoint(0.0, new double[] { 0.0 });
36          for (double x = -10; x < 10; x += 1.0) {
37              DerivativeStructure y = interpolator.value(new DerivativeStructure(1, 1, 0, x))[0];
38              Assert.assertEquals(0.0, y.getValue(), 1.0e-15);
39              Assert.assertEquals(0.0, y.getPartialDerivative(1), 1.0e-15);
40          }
41          checkPolynomial(new PolynomialFunction(new double[] { 0.0 }),
42                          interpolator.getPolynomials()[0]);
43      }

45      @Test
46      public void testQuadratic() {
47          HermiteInterpolator interpolator = new HermiteInterpolator();
48          interpolator.addSamplePoint(0.0, new double[] { 2.0 });
49          interpolator.addSamplePoint(1.0, new double[] { 0.0 });
50          interpolator.addSamplePoint(2.0, new double[] { 0.0 });
51          for (double x = -10; x < 10; x += 1.0) {
52              DerivativeStructure y = interpolator.value(new DerivativeStructure(1, 1, 0, x))[0];
53              Assert.assertEquals((x - 1.0) * (x - 2.0), y.getValue(), 1.0e-15);
54              Assert.assertEquals(2 * x - 3.0, y.getPartialDerivative(1), 1.0e-15);
55          }
56          checkPolynomial(new PolynomialFunction(new double[] { 2.0, -3.0, 1.0 }),
57                          interpolator.getPolynomials()[0]);
58      }

60      @Test
61      public void testMixedDerivatives() {
62          HermiteInterpolator interpolator = new HermiteInterpolator();
63          interpolator.addSamplePoint(0.0, new double[] { 1.0 }, new double[] { 2.0 });
64          interpolator.addSamplePoint(1.0, new double[] { 4.0 });
65          interpolator.addSamplePoint(2.0, new double[] { 5.0 }, new double[] { 2.0 });
66          Assert.assertEquals(4, interpolator.getPolynomials()[0].degree());
67          DerivativeStructure y0 = interpolator.value(new DerivativeStructure(1, 1, 0, 0.0))[0];
68          Assert.assertEquals(1.0, y0.getValue(), 1.0e-15);
69          Assert.assertEquals(2.0, y0.getPartialDerivative(1), 1.0e-15);
70          Assert.assertEquals(4.0, interpolator.value(1.0)[0], 1.0e-15);
71          DerivativeStructure y2 = interpolator.value(new DerivativeStructure(1, 1, 0, 2.0))[0];
72          Assert.assertEquals(5.0, y2.getValue(), 1.0e-15);
73          Assert.assertEquals(2.0, y2.getPartialDerivative(1), 1.0e-15);
74          checkPolynomial(new PolynomialFunction(new double[] { 1.0, 2.0, 4.0, -4.0, 1.0 }),
75                          interpolator.getPolynomials()[0]);
76      }

78      @Test
79      public void testRandomPolynomialsValuesOnly() {
80  
81          Random random = new Random(0x42b1e7dbd361a932l);
82  
83          for (int i = 0; i < 100; ++i) {
84  
85              int maxDegree = 0;
86              PolynomialFunction[] p = new PolynomialFunction[5];
87              for (int k = 0; k < p.length; ++k) {
88                  int degree = random.nextInt(7);
89                  p[k] = randomPolynomial(degree, random);
90                  maxDegree = FastMath.max(maxDegree, degree);
91              }
92  
93              HermiteInterpolator interpolator = new HermiteInterpolator();
94              for (int j = 0; j < 1 + maxDegree; ++j) {
95                  double x = 0.1 * j;
96                  double[] values = new double[p.length];
97                  for (int k = 0; k < p.length; ++k) {
98                      values[k] = p[k].value(x);
99                  }
100                 interpolator.addSamplePoint(x, values);
101             }
102 
103             for (double x = 0; x < 2; x += 0.1) {
104                 double[] values = interpolator.value(x);
105                 Assert.assertEquals(p.length, values.length);
106                 for (int k = 0; k < p.length; ++k) {
107                     Assert.assertEquals(p[k].value(x), values[k], 1.0e-8 * FastMath.abs(p[k].value(x)));
108                 }
109             }
110 
111             PolynomialFunction[] result = interpolator.getPolynomials();
112             for (int k = 0; k < p.length; ++k) {
113                 checkPolynomial(p[k], result[k]);
114             }
115 
116         }
117     }

119     @Test
120     public void testRandomPolynomialsFirstDerivative() {
121 
122         Random random = new Random(0x570803c982ca5d3bl);
123 
124         for (int i = 0; i < 100; ++i) {
125 
126             int maxDegree = 0;
127             PolynomialFunction[] p      = new PolynomialFunction[5];
128             PolynomialFunction[] pPrime = new PolynomialFunction[5];
129             for (int k = 0; k < p.length; ++k) {
130                 int degree = random.nextInt(7);
131                 p[k]      = randomPolynomial(degree, random);
132                 pPrime[k] = p[k].polynomialDerivative();
133                 maxDegree = FastMath.max(maxDegree, degree);
134             }
135 
136             HermiteInterpolator interpolator = new HermiteInterpolator();
137             for (int j = 0; j < 1 + maxDegree / 2; ++j) {
138                 double x = 0.1 * j;
139                 double[] values      = new double[p.length];
140                 double[] derivatives = new double[p.length];
141                 for (int k = 0; k < p.length; ++k) {
142                     values[k]      = p[k].value(x);
143                     derivatives[k] = pPrime[k].value(x);
144                 }
145                 interpolator.addSamplePoint(x, values, derivatives);
146             }
147 
148             for (double x = 0; x < 2; x += 0.1) {
149                 DerivativeStructure[] y = interpolator.value(new DerivativeStructure(1, 1, 0, x));
150                 Assert.assertEquals(p.length, y.length);
151                 for (int k = 0; k < p.length; ++k) {
152                     Assert.assertEquals(p[k].value(x), y[k].getValue(), 1.0e-8 * FastMath.abs(p[k].value(x)));
153                     Assert.assertEquals(pPrime[k].value(x), y[k].getPartialDerivative(1), 4.0e-8 * FastMath.abs(p[k].value(x)));
154                 }
155             }
156 
157             PolynomialFunction[] result = interpolator.getPolynomials();
158             for (int k = 0; k < p.length; ++k) {
159                 checkPolynomial(p[k], result[k]);
160             }
161 
162         }
163     }

165     @Test
166     public void testSine() {
167         HermiteInterpolator interpolator = new HermiteInterpolator();
168         for (double x = 0; x < FastMath.PI; x += 0.5) {
169             interpolator.addSamplePoint(x, new double[] { FastMath.sin(x) });
170         }
171         for (double x = 0.1; x <= 2.9; x += 0.01) {
172             DerivativeStructure y = interpolator.value(new DerivativeStructure(1, 2, 0, x))[0];
173             Assert.assertEquals( FastMath.sin(x), y.getValue(), 3.5e-5);
174             Assert.assertEquals( FastMath.cos(x), y.getPartialDerivative(1), 1.3e-4);
175             Assert.assertEquals(-FastMath.sin(x), y.getPartialDerivative(2), 2.9e-3);
176         }
177     }

192     @Test
193     public void testWikipedia() {
194         // this test corresponds to the example from Wikipedia page:
195         // http://en.wikipedia.org/wiki/Hermite_interpolation
196         HermiteInterpolator interpolator = new HermiteInterpolator();
197         interpolator.addSamplePoint(-1, new double[] { 2 }, new double[] { -8 }, new double[] { 56 });
198         interpolator.addSamplePoint( 0, new double[] { 1 }, new double[] {  0 }, new double[] {  0 });
199         interpolator.addSamplePoint( 1, new double[] { 2 }, new double[] {  8 }, new double[] { 56 });
200         for (double x = -1.0; x <= 1.0; x += 0.125) {
201             DerivativeStructure y = interpolator.value(new DerivativeStructure(1, 1, 0, x))[0];
202             double x2 = x * x;
203             double x4 = x2 * x2;
204             double x8 = x4 * x4;
205             Assert.assertEquals(x8 + 1, y.getValue(), 1.0e-15);
206             Assert.assertEquals(8 * x4 * x2 * x, y.getPartialDerivative(1), 1.0e-15);
207         }
208         checkPolynomial(new PolynomialFunction(new double[] { 1, 0, 0, 0, 0, 0, 0, 0, 1 }),
209                         interpolator.getPolynomials()[0]);
210     }

212     @Test
213     public void testOnePointParabola() {
214         HermiteInterpolator interpolator = new HermiteInterpolator();
215         interpolator.addSamplePoint(0, new double[] { 1 }, new double[] { 1 }, new double[] { 2 });
216         for (double x = -1.0; x <= 1.0; x += 0.125) {
217             DerivativeStructure y = interpolator.value(new DerivativeStructure(1, 1, 0, x))[0];
218             Assert.assertEquals(1 + x * (1 + x), y.getValue(), 1.0e-15);
219             Assert.assertEquals(1 + 2 * x, y.getPartialDerivative(1), 1.0e-15);
220         }
221         checkPolynomial(new PolynomialFunction(new double[] { 1, 1, 1 }),
222                         interpolator.getPolynomials()[0]);
223     }

225     private PolynomialFunction randomPolynomial(int degree, Random random) {
226         double[] coeff = new double[ 1 + degree];
227         for (int j = 0; j < degree; ++j) {
228             coeff[j] = random.nextDouble();
229         }
230         return new PolynomialFunction(coeff);
231     }

245     private void checkPolynomial(PolynomialFunction expected, PolynomialFunction result) {
246         Assert.assertTrue(result.degree() >= expected.degree());
247         double[] cE = expected.getCoefficients();
248         double[] cR = result.getCoefficients();
249         for (int i = 0; i < cE.length; ++i) {
250             Assert.assertEquals(cE[i], cR[i], 1.0e-8 * FastMath.abs(cE[i]));
251         }
252         for (int i = cE.length; i < cR.length; ++i) {
253             Assert.assertEquals(0.0, cR[i], 1.0e-9);
254         }
255     }
