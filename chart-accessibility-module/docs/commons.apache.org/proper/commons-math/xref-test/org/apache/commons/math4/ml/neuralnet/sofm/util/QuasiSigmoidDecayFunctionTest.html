18  package org.apache.commons.math4.ml.neuralnet.sofm.util;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.ml.neuralnet.sofm.util.QuasiSigmoidDecayFunction;
23  import org.junit.Test;
24  import org.junit.Assert;
25  
26  /**
27   * Tests for {@link QuasiSigmoidDecayFunction} class
28   */
29  public class QuasiSigmoidDecayFunctionTest {
30      @Test(expected=NotStrictlyPositiveException.class)
31      public void testPrecondition1() {
32          new QuasiSigmoidDecayFunction(0d, -1d, 2);
33      }
34      @Test(expected=NumberIsTooLargeException.class)
35      public void testPrecondition3() {
36          new QuasiSigmoidDecayFunction(1d, 0d, 100);
37      }
38      @Test(expected=NotStrictlyPositiveException.class)
39      public void testPrecondition4() {
40          new QuasiSigmoidDecayFunction(1d, -1d, 0);
41      }
42  
43      @Test
44      public void testTrivial() {
45          final int n = 65;
46          final double init = 4;
47          final double slope = -1e-1;
48          final QuasiSigmoidDecayFunction f = new QuasiSigmoidDecayFunction(init, slope, n);
49  
50          Assert.assertEquals(init, f.value(0), 0d);
51          // Very approximate derivative.
52          Assert.assertEquals(slope, f.value(n + 1) - f.value(n), 1e-4);
53          Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
54      }
55  }
