40      extends AbstractLeastSquaresOptimizerAbstractTest {
41  
42      @Override
43      public int getMaxIterations() {
44          return 1000;
45      }
46  
47      @Override
48      public LeastSquaresOptimizer getOptimizer() {
49          return new GaussNewtonOptimizer(Decomposition.CHOLESKY);
50      }
51  
52      @Override
53      @Test
54      public void testMoreEstimatedParametersSimple() {
55          /*
56           * Exception is expected with this optimizer
57           */
58          try {
59              super.testMoreEstimatedParametersSimple();
60              fail(optimizer);
61          } catch (ConvergenceException e) {
62              //expected
63          }
64      }
65

74              fail(optimizer);
75          }catch (ConvergenceException e){
76              //expected
77          }
78      }
79  
80      @Test
81      public void testMaxEvaluations() throws Exception {
82          try{
83          CircleVectorial circle = new CircleVectorial();
84          circle.addPoint( 30.0,  68.0);
85          circle.addPoint( 50.0,  -6.0);
86          circle.addPoint(110.0, -20.0);
87          circle.addPoint( 35.0,  15.0);
88          circle.addPoint( 45.0,  97.0);
89  
90          LeastSquaresProblem lsp = builder(circle)
91                  .checkerPair(new SimpleVectorValueChecker(1e-30, 1e-30))
92                  .maxIterations(Integer.MAX_VALUE)
93                  .start(new double[]{98.680, 47.345})
94                  .build();
95  
96          optimizer.optimize(lsp);
97  
98              fail(optimizer);
99          }catch (TooManyEvaluationsException e){
100             //expected
101         }
102     }
