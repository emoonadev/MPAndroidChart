85      /**
86       * Returns the offsets for the unit tests below, in the order they are
87       * given below. For each test, the left and right offsets are returned.
88       * @return the offsets for the unit tests below, in the order they are
89       * given below
90       */
91      protected abstract int[][] getO();
92  
93      /**
94       * Get the array, given left and right offsets.
95       * @param a the input array
96       * @param offsetL the left side offset
97       * @param offsetR the right side offset
98       * @return the modified array
99       */
100     private double[] getArray(double[] a, int offsetL, int offsetR) {
101         double[] copy = new double[a.length - offsetR - offsetL];
102         System.arraycopy(a, offsetL, copy, 0, copy.length);
103         return copy;
104     }

234     /**
235      * The actual step normalizer output test code, shared by all the unit
236      * tests.
237      *
238      * @param mode the step normalizer mode to use
239      * @param bounds the step normalizer bounds setting to use
240      * @param expected the expected output (normalized time points)
241      * @param reverse whether to reverse the integration direction
242      * @throws NoBracketingException
243      * @throws MaxCountExceededException
244      * @throws NumberIsTooSmallException
245      * @throws DimensionMismatchException
246      */
247     private void doTest(StepNormalizerMode mode, StepNormalizerBounds bounds,
248                         double[] expected, boolean reverse)
249         throws DimensionMismatchException, NumberIsTooSmallException,
250                MaxCountExceededException, NoBracketingException {
251         // Forward test.
252         FirstOrderIntegrator integ = new GraggBulirschStoerIntegrator(
253                                                         1e-8, 1.0, 1e-5, 1e-5);
254         integ.addStepHandler(new StepNormalizer(0.5, this, mode, bounds));
255         double[] y   = {0.0};
256         double start = reverse ? getEnd()   : getStart();
257         double end   = reverse ? getStart() : getEnd();
258         output       = new ArrayList<Double>();
259         integ.integrate(this, start, y, end, y);
260         double[] actual = new double[output.size()];
261         for(int i = 0; i < actual.length; i++) {
262             actual[i] = output.get(i);
263         }
264         Assert.assertArrayEquals(expected, actual, 1e-5);
265     }
