18  package org.apache.commons.math4.ml.neuralnet.twod.util;
19  
20  import org.apache.commons.math4.ml.neuralnet.Neuron;
21  import org.apache.commons.math4.ml.neuralnet.Network;
22  import org.apache.commons.math4.ml.neuralnet.FeatureInitializer;
23  import org.apache.commons.math4.ml.neuralnet.FeatureInitializerFactory;
24  import org.apache.commons.math4.ml.neuralnet.SquareNeighbourhood;
25  import org.apache.commons.math4.ml.neuralnet.twod.NeuronSquareMesh2D;
26  import org.junit.Assert;
27  import org.junit.Test;
28  
29  /**
30   * Test for {@link LocationFinder}.
31   */
32  public class LocationFinderTest {
33      final FeatureInitializer init = FeatureInitializerFactory.uniform(0, 2);
34  
35      /*
36       * Test assumes that the network is
37       *
38       *  0-----1
39       *  |     |
40       *  |     |
41       *  2-----3
42       */
43      @Test
44      public void test2x2Network() {
45          final FeatureInitializer[] initArray = { init };
46          final NeuronSquareMesh2D map = new NeuronSquareMesh2D(2, false,
47                                                                2, false,
48                                                                SquareNeighbourhood.VON_NEUMANN,
49                                                                initArray);
50          final LocationFinder finder = new LocationFinder(map);
51          final Network net = map.getNetwork();
52          LocationFinder.Location loc;
53  
54          loc = finder.getLocation(net.getNeuron(0));
55          Assert.assertEquals(0, loc.getRow());
56          Assert.assertEquals(0, loc.getColumn());
57  
58          loc = finder.getLocation(net.getNeuron(1));
59          Assert.assertEquals(0, loc.getRow());
60          Assert.assertEquals(1, loc.getColumn());
61  
62          loc = finder.getLocation(net.getNeuron(2));
63          Assert.assertEquals(1, loc.getRow());
64          Assert.assertEquals(0, loc.getColumn());
65  
66          loc = finder.getLocation(net.getNeuron(3));
67          Assert.assertEquals(1, loc.getRow());
68          Assert.assertEquals(1, loc.getColumn());
69      }
70  }
