18  package org.apache.commons.math4.util;
19  
20  import java.math.BigDecimal;
21  
22  import org.apache.commons.math4.TestUtils;
23  import org.apache.commons.math4.exception.MathIllegalArgumentException;
24  import org.apache.commons.math4.exception.NullArgumentException;
25  import org.apache.commons.math4.util.DefaultTransformer;
26  import org.junit.Assert;
27  import org.junit.Test;
28  
29  /**
30   */
31  public class DefaultTransformerTest {
32      /**
33       *
34       */
35      @Test
36      public void testTransformDouble() throws Exception {
37          double expected = 1.0;
38          Double input = Double.valueOf(expected);
39          DefaultTransformer t = new DefaultTransformer();
40          Assert.assertEquals(expected, t.transform(input), 1.0e-4);
41      }
42  
43      /**
44       *
45       */
46      @Test
47      public void testTransformNull() throws Exception {
48          DefaultTransformer t = new DefaultTransformer();
49          try {
50              t.transform(null);
51              Assert.fail("Expecting NullArgumentException");
52          } catch (NullArgumentException e) {
53              // expected
54          }
55      }
56  
57      /**
58       *
59       */
60      @Test
61      public void testTransformInteger() throws Exception {
62          double expected = 1.0;
63          Integer input = Integer.valueOf(1);
64          DefaultTransformer t = new DefaultTransformer();
65          Assert.assertEquals(expected, t.transform(input), 1.0e-4);
66      }
67  
68      /**
69       *
70       */
71      @Test
72      public void testTransformBigDecimal() throws Exception {
73          double expected = 1.0;

77      }
78  
79      /**
80       *
81       */

83      public void testTransformString() throws Exception {
84          double expected = 1.0;

88      }
89  
90      /**
91       *
92       */
93      @Test(expected=MathIllegalArgumentException.class)
94      public void testTransformObject(){
95          Boolean input = Boolean.TRUE;

98      }
99

103     }
