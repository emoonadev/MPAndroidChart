62      public void testBox() {
63          PolyhedronsSet tree = new PolyhedronsSet(0, 1, 0, 1, 0, 1, 1.0e-10);
64          Assert.assertEquals(1.0, tree.getSize(), 1.0e-10);
65          Assert.assertEquals(6.0, tree.getBoundarySize(), 1.0e-10);
66          Vector3D barycenter = (Vector3D) tree.getBarycenter();
67          Assert.assertEquals(0.5, barycenter.getX(), 1.0e-10);
68          Assert.assertEquals(0.5, barycenter.getY(), 1.0e-10);
69          Assert.assertEquals(0.5, barycenter.getZ(), 1.0e-10);
70          for (double x = -0.25; x < 1.25; x += 0.1) {
71              boolean xOK = (x >= 0.0) && (x <= 1.0);
72              for (double y = -0.25; y < 1.25; y += 0.1) {
73                  boolean yOK = (y >= 0.0) && (y <= 1.0);
74                  for (double z = -0.25; z < 1.25; z += 0.1) {
75                      boolean zOK = (z >= 0.0) && (z <= 1.0);
76                      Region.Location expected =
77                          (xOK && yOK && zOK) ? Region.Location.INSIDE : Region.Location.OUTSIDE;
78                      Assert.assertEquals(expected, tree.checkPoint(new Vector3D(x, y, z)));
79                  }
80              }
81          }
82          checkPoints(Region.Location.BOUNDARY, tree, new Vector3D[] {
83              new Vector3D(0.0, 0.5, 0.5),
84              new Vector3D(1.0, 0.5, 0.5),
85              new Vector3D(0.5, 0.0, 0.5),
86              new Vector3D(0.5, 1.0, 0.5),
87              new Vector3D(0.5, 0.5, 0.0),
88              new Vector3D(0.5, 0.5, 1.0)
89          });
90          checkPoints(Region.Location.OUTSIDE, tree, new Vector3D[] {
91              new Vector3D(0.0, 1.2, 1.2),
92              new Vector3D(1.0, 1.2, 1.2),
93              new Vector3D(1.2, 0.0, 1.2),
94              new Vector3D(1.2, 1.0, 1.2),
95              new Vector3D(1.2, 1.2, 0.0),
96              new Vector3D(1.2, 1.2, 1.0)

101     public void testTetrahedron() throws MathArithmeticException {
102         Vector3D vertex1 = new Vector3D(1, 2, 3);
103         Vector3D vertex2 = new Vector3D(2, 2, 4);
104         Vector3D vertex3 = new Vector3D(2, 3, 3);
105         Vector3D vertex4 = new Vector3D(1, 3, 4);
106         PolyhedronsSet tree =
107             (PolyhedronsSet) new RegionFactory<Euclidean3D>().buildConvex(
108                 new Plane(vertex3, vertex2, vertex1, 1.0e-10),
109                 new Plane(vertex2, vertex3, vertex4, 1.0e-10),
110                 new Plane(vertex4, vertex3, vertex1, 1.0e-10),
111                 new Plane(vertex1, vertex2, vertex4, 1.0e-10));
112         Assert.assertEquals(1.0 / 3.0, tree.getSize(), 1.0e-10);
113         Assert.assertEquals(2.0 * FastMath.sqrt(3.0), tree.getBoundarySize(), 1.0e-10);
114         Vector3D barycenter = (Vector3D) tree.getBarycenter();
115         Assert.assertEquals(1.5, barycenter.getX(), 1.0e-10);
116         Assert.assertEquals(2.5, barycenter.getY(), 1.0e-10);
117         Assert.assertEquals(3.5, barycenter.getZ(), 1.0e-10);
118         double third = 1.0 / 3.0;
119         checkPoints(Region.Location.BOUNDARY, tree, new Vector3D[] {
120             vertex1, vertex2, vertex3, vertex4,
121             new Vector3D(third, vertex1, third, vertex2, third, vertex3),
122             new Vector3D(third, vertex2, third, vertex3, third, vertex4),
123             new Vector3D(third, vertex3, third, vertex4, third, vertex1),
124             new Vector3D(third, vertex4, third, vertex1, third, vertex2)
125         });
126         checkPoints(Region.Location.OUTSIDE, tree, new Vector3D[] {
127             new Vector3D(1, 2, 4),
128             new Vector3D(2, 2, 3),
129             new Vector3D(2, 3, 4),
130             new Vector3D(1, 3, 3)
131         });

134     @Test
135     public void testIsometry() throws MathArithmeticException, MathIllegalArgumentException {
136         Vector3D vertex1 = new Vector3D(1.1, 2.2, 3.3);
137         Vector3D vertex2 = new Vector3D(2.0, 2.4, 4.2);
138         Vector3D vertex3 = new Vector3D(2.8, 3.3, 3.7);
139         Vector3D vertex4 = new Vector3D(1.0, 3.6, 4.5);
140         PolyhedronsSet tree =
141             (PolyhedronsSet) new RegionFactory<Euclidean3D>().buildConvex(
142                 new Plane(vertex3, vertex2, vertex1, 1.0e-10),
143                 new Plane(vertex2, vertex3, vertex4, 1.0e-10),
144                 new Plane(vertex4, vertex3, vertex1, 1.0e-10),
145                 new Plane(vertex1, vertex2, vertex4, 1.0e-10));
146         Vector3D barycenter = (Vector3D) tree.getBarycenter();
147         Vector3D s = new Vector3D(10.2, 4.3, -6.7);
148         Vector3D c = new Vector3D(-0.2, 2.1, -3.2);
149         Rotation r = new Rotation(new Vector3D(6.2, -4.4, 2.1), 0.12, RotationConvention.VECTOR_OPERATOR);
150 
151         tree = tree.rotate(c, r).translate(s);
152 
153         Vector3D newB =
154             new Vector3D(1.0, s,
155                          1.0, c,
156                          1.0, r.applyTo(barycenter.subtract(c)));
157         Assert.assertEquals(0.0,
158                             newB.subtract((Vector<Euclidean3D>) tree.getBarycenter()).getNorm(),
159                             1.0e-10);
160 
161         final Vector3D[] expectedV = new Vector3D[] {
162             new Vector3D(1.0, s,
163                          1.0, c,
164                          1.0, r.applyTo(vertex1.subtract(c))),
165                          new Vector3D(1.0, s,
166                                       1.0, c,
167                                       1.0, r.applyTo(vertex2.subtract(c))),
168                                       new Vector3D(1.0, s,
169                                                    1.0, c,
170                                                    1.0, r.applyTo(vertex3.subtract(c))),
171                                                    new Vector3D(1.0, s,
172                                                                 1.0, c,
173                                                                 1.0, r.applyTo(vertex4.subtract(c)))
174         };
175         tree.getTree(true).visit(new BSPTreeVisitor<Euclidean3D>() {
176 
177             public Order visitOrder(BSPTree<Euclidean3D> node) {
178                 return Order.MINUS_SUB_PLUS;
179             }
180 
181             public void visitInternalNode(BSPTree<Euclidean3D> node) {
182                 @SuppressWarnings("unchecked")
183                 BoundaryAttribute<Euclidean3D> attribute =
184                     (BoundaryAttribute<Euclidean3D>) node.getAttribute();
185                 if (attribute.getPlusOutside() != null) {
186                     checkFacet((SubPlane) attribute.getPlusOutside());
187                 }
188                 if (attribute.getPlusInside() != null) {
189                     checkFacet((SubPlane) attribute.getPlusInside());
190                 }
191             }
192 
193             public void visitLeafNode(BSPTree<Euclidean3D> node) {
194             }
195 
196             private void checkFacet(SubPlane facet) {
197                 Plane plane = (Plane) facet.getHyperplane();
198                 Vector2D[][] vertices =
199                     ((PolygonsSet) facet.getRemainingRegion()).getVertices();
200                 Assert.assertEquals(1, vertices.length);
201                 for (int i = 0; i < vertices[0].length; ++i) {
202                     Vector3D v = plane.toSpace(vertices[0][i]);
203                     double d = Double.POSITIVE_INFINITY;
204                     for (int k = 0; k < expectedV.length; ++k) {
205                         d = FastMath.min(d, v.subtract(expectedV[k]).getNorm());
206                     }
207                     Assert.assertEquals(0, d, 1.0e-10);
208                 }
209             }
210 
211         });
212 
213     }

236         double y = 2.0;
237         double z = 3.0;
238         double w = 0.1;
239         double l = 1.0;
240         PolyhedronsSet xBeam =
241             new PolyhedronsSet(x - l, x + l, y - w, y + w, z - w, z + w, 1.0e-10);
242         PolyhedronsSet yBeam =
243             new PolyhedronsSet(x - w, x + w, y - l, y + l, z - w, z + w, 1.0e-10);
244         PolyhedronsSet zBeam =
245             new PolyhedronsSet(x - w, x + w, y - w, y + w, z - l, z + l, 1.0e-10);
246         RegionFactory<Euclidean3D> factory = new RegionFactory<Euclidean3D>();
247         PolyhedronsSet tree = (PolyhedronsSet) factory.union(xBeam, factory.union(yBeam, zBeam));
248         Vector3D barycenter = (Vector3D) tree.getBarycenter();
249 
250         Assert.assertEquals(x, barycenter.getX(), 1.0e-10);
251         Assert.assertEquals(y, barycenter.getY(), 1.0e-10);
252         Assert.assertEquals(z, barycenter.getZ(), 1.0e-10);
253         Assert.assertEquals(8 * w * w * (3 * l - 2 * w), tree.getSize(), 1.0e-10);
254         Assert.assertEquals(24 * w * (2 * l - w), tree.getBoundarySize(), 1.0e-10);

258     @Test
259     public void testIssue780() throws MathArithmeticException {
260         float[] coords = {
261             1.000000f, -1.000000f, -1.000000f,
262             1.000000f, -1.000000f, 1.000000f,
263             -1.000000f, -1.000000f, 1.000000f,
264             -1.000000f, -1.000000f, -1.000000f,
265             1.000000f, 1.000000f, -1f,
266             0.999999f, 1.000000f, 1.000000f,   // 1.000000f, 1.000000f, 1.000000f,
267             -1.000000f, 1.000000f, 1.000000f,
268             -1.000000f, 1.000000f, -1.000000f};
269         int[] indices = {
270             0, 1, 2, 0, 2, 3,
271             4, 7, 6, 4, 6, 5,
272             0, 4, 5, 0, 5, 1,
273             1, 5, 6, 1, 6, 2,
274             2, 6, 7, 2, 7, 3,
275             4, 0, 3, 4, 3, 7};
276         ArrayList<SubHyperplane<Euclidean3D>> subHyperplaneList = new ArrayList<SubHyperplane<Euclidean3D>>();
277         for (int idx = 0; idx < indices.length; idx += 3) {
278             int idxA = indices[idx] * 3;
279             int idxB = indices[idx + 1] * 3;
280             int idxC = indices[idx + 2] * 3;
281             Vector3D v_1 = new Vector3D(coords[idxA], coords[idxA + 1], coords[idxA + 2]);
282             Vector3D v_2 = new Vector3D(coords[idxB], coords[idxB + 1], coords[idxB + 2]);
283             Vector3D v_3 = new Vector3D(coords[idxC], coords[idxC + 1], coords[idxC + 2]);
284             Vector3D[] vertices = {v_1, v_2, v_3};
285             Plane polyPlane = new Plane(v_1, v_2, v_3, 1.0e-10);
286             ArrayList<SubHyperplane<Euclidean2D>> lines = new ArrayList<SubHyperplane<Euclidean2D>>();
287 
288             Vector2D[] projPts = new Vector2D[vertices.length];
289             for (int ptIdx = 0; ptIdx < projPts.length; ptIdx++) {
290                 projPts[ptIdx] = polyPlane.toSubSpace(vertices[ptIdx]);
291             }
292 
293             SubLine lineInPlane = null;
294             for (int ptIdx = 0; ptIdx < projPts.length; ptIdx++) {
295                 lineInPlane = new SubLine(projPts[ptIdx], projPts[(ptIdx + 1) % projPts.length], 1.0e-10);
296                 lines.add(lineInPlane);
297             }
298             Region<Euclidean2D> polyRegion = new PolygonsSet(lines, 1.0e-10);
299             SubPlane polygon = new SubPlane(polyPlane, polyRegion);
300             subHyperplaneList.add(polygon);
301         }
302         PolyhedronsSet polyhedronsSet = new PolyhedronsSet(subHyperplaneList, 1.0e-10);
303         Assert.assertEquals( 8.0, polyhedronsSet.getSize(), 3.0e-6);
304         Assert.assertEquals(24.0, polyhedronsSet.getBoundarySize(), 5.0e-6);
305     }

432                 Assert.fail(nsfe.getLocalizedMessage());
433             } catch (IllegalAccessException iae) {
434                 Assert.fail(iae.getLocalizedMessage());
435             }
436         }

439     @Test
440     public void testIssue1211() throws IOException, ParseException {
441 
442         PolyhedronsSet polyset = RegionParser.parsePolyhedronsSet(loadTestData("issue-1211.bsp"));
443         UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A, 0xb97c9d1ade21e40al);
444         int nrays = 1000;
445         for (int i = 0; i < nrays; i++) {
446             Vector3D origin    = Vector3D.ZERO;
447             Vector3D direction = new Vector3D(2 * random.nextDouble() - 1,
448                                               2 * random.nextDouble() - 1,
449                                               2 * random.nextDouble() - 1).normalize();
450             Line line = new Line(origin, origin.add(direction), polyset.getTolerance());
451             SubHyperplane<Euclidean3D> plane = polyset.firstIntersection(origin, line);
452             if (plane != null) {
453                 Vector3D intersectionPoint = ((Plane)plane.getHyperplane()).intersection(line);
454                 double dotProduct = direction.dotProduct(intersectionPoint.subtract(origin));
455                 Assert.assertTrue(dotProduct > 0);
456             }
457         }
458     }

460     private String loadTestData(final String resourceName)
461             throws IOException {
462             InputStream stream = getClass().getResourceAsStream(resourceName);
463             Reader reader = new InputStreamReader(stream, "UTF-8");
464             StringBuilder builder = new StringBuilder();

466                 builder.append((char) c);
467             }
468             return builder.toString();
