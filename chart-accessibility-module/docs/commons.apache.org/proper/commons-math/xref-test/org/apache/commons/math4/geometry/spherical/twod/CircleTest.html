41      @Test
42      public void testEquator() {
43          Circle circle = new Circle(new Vector3D(0, 0, 1000), 1.0e-10).copySelf();
44          Assert.assertEquals(Vector3D.PLUS_K, circle.getPole());
45          Assert.assertEquals(1.0e-10, circle.getTolerance(), 1.0e-20);
46          circle.revertSelf();
47          Assert.assertEquals(Vector3D.MINUS_K, circle.getPole());
48          Assert.assertEquals(Vector3D.PLUS_K, circle.getReverse().getPole());
49          Assert.assertEquals(Vector3D.MINUS_K, circle.getPole());
50      }

52      @Test
53      public void testXY() {
54          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
55          Assert.assertEquals(0.0, circle.getPointAt(0).distance(circle.getXAxis()), 1.0e-10);
56          Assert.assertEquals(0.0, circle.getPointAt(0.5 * FastMath.PI).distance(circle.getYAxis()), 1.0e-10);
57          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getXAxis(), circle.getYAxis()), 1.0e-10);
58          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getXAxis(), circle.getPole()), 1.0e-10);
59          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), circle.getYAxis()), 1.0e-10);
60          Assert.assertEquals(0.0,
61                              circle.getPole().distance(Vector3D.crossProduct(circle.getXAxis(), circle.getYAxis())),
62                              1.0e-10);
63      }

65      @Test
66      public void testReverse() {
67          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
68          Circle reversed = circle.getReverse();
69          Assert.assertEquals(0.0, reversed.getPointAt(0).distance(reversed.getXAxis()), 1.0e-10);
70          Assert.assertEquals(0.0, reversed.getPointAt(0.5 * FastMath.PI).distance(reversed.getYAxis()), 1.0e-10);
71          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getXAxis(), reversed.getYAxis()), 1.0e-10);
72          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getXAxis(), reversed.getPole()), 1.0e-10);
73          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(reversed.getPole(), reversed.getYAxis()), 1.0e-10);
74          Assert.assertEquals(0.0,
75                              reversed.getPole().distance(Vector3D.crossProduct(reversed.getXAxis(), reversed.getYAxis())),
76                              1.0e-10);
77  
78          Assert.assertEquals(0, Vector3D.angle(circle.getXAxis(), reversed.getXAxis()), 1.0e-10);
79          Assert.assertEquals(FastMath.PI, Vector3D.angle(circle.getYAxis(), reversed.getYAxis()), 1.0e-10);
80          Assert.assertEquals(FastMath.PI, Vector3D.angle(circle.getPole(), reversed.getPole()), 1.0e-10);
81  
82          Assert.assertTrue(circle.sameOrientationAs(circle));
83          Assert.assertFalse(circle.sameOrientationAs(reversed));
84  
85      }

87      @Test
88      public void testPhase() {
89          Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
90          Vector3D p = new Vector3D(1, 2, -4);
91          Vector3D samePhase = circle.getPointAt(circle.getPhase(p));
92          Assert.assertEquals(0.0,
93                              Vector3D.angle(Vector3D.crossProduct(circle.getPole(), p),
94                                             Vector3D.crossProduct(circle.getPole(), samePhase)),
95                              1.0e-10);
96          Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), samePhase), 1.0e-10);
97          Assert.assertEquals(circle.getPhase(p), circle.getPhase(samePhase), 1.0e-10);
98          Assert.assertEquals(0.0, circle.getPhase(circle.getXAxis()), 1.0e-10);
99          Assert.assertEquals(0.5 * FastMath.PI, circle.getPhase(circle.getYAxis()), 1.0e-10);
100 
101     }

112     @Test
113     public void testSpace() {
114         Circle circle = new Circle(new S2Point(1.2, 2.5), new S2Point(-4.3, 0), 1.0e-10);
115         for (double alpha = 0; alpha < MathUtils.TWO_PI; alpha += 0.1) {
116             Vector3D p = new Vector3D(FastMath.cos(alpha), circle.getXAxis(),
117                                       FastMath.sin(alpha), circle.getYAxis());
118             Vector3D q = circle.toSpace(new S1Point(alpha)).getVector();
119             Assert.assertEquals(0.0, p.distance(q), 1.0e-10);
120             Assert.assertEquals(0.5 * FastMath.PI, Vector3D.angle(circle.getPole(), q), 1.0e-10);
121         }
122     }

136     @Test
137     public void testInsideArc() {
138         UnitSphereRandomVectorGenerator sphRandom = new UnitSphereRandomVectorGenerator(3, RandomSource.create(RandomSource.WELL_1024_A,
139                                                                                                                0xbfd34e92231bbcfel));
140         for (int i = 0; i < 100; ++i) {
141             Circle c1 = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
142             Circle c2 = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
143             checkArcIsInside(c1, c2);
144             checkArcIsInside(c2, c1);
145         }
146     }
147 
148     private void checkArcIsInside(final Circle arcCircle, final Circle otherCircle) {
149         Arc arc = arcCircle.getInsideArc(otherCircle);
150         Assert.assertEquals(FastMath.PI, arc.getSize(), 1.0e-10);
151         for (double alpha = arc.getInf(); alpha < arc.getSup(); alpha += 0.1) {
152             Assert.assertTrue(otherCircle.getOffset(arcCircle.getPointAt(alpha)) <= 2.0e-15);
153         }
154         for (double alpha = arc.getSup(); alpha < arc.getInf() + MathUtils.TWO_PI; alpha += 0.1) {
155             Assert.assertTrue(otherCircle.getOffset(arcCircle.getPointAt(alpha)) >= -2.0e-15);
156         }
157     }

159     @Test
160     public void testTransform() {
161         UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
162                                                            0x16992fc4294bf2f1l);
163         UnitSphereRandomVectorGenerator sphRandom = new UnitSphereRandomVectorGenerator(3, random);
164         for (int i = 0; i < 100; ++i) {
165 
166             Rotation r = new Rotation(new Vector3D(sphRandom.nextVector()),
167                                       FastMath.PI * random.nextDouble(),
168                                       RotationConvention.VECTOR_OPERATOR);
169             Transform<Sphere2D, Sphere1D> t = Circle.getTransform(r);
170 
171             S2Point  p = new S2Point(new Vector3D(sphRandom.nextVector()));
172             S2Point tp = (S2Point) t.apply(p);
173             Assert.assertEquals(0.0, r.applyTo(p.getVector()).distance(tp.getVector()), 1.0e-10);
174 
175             Circle  c = new Circle(new Vector3D(sphRandom.nextVector()), 1.0e-10);
176             Circle tc = (Circle) t.apply(c);
177             Assert.assertEquals(0.0, r.applyTo(c.getPole()).distance(tc.getPole()),   1.0e-10);
178             Assert.assertEquals(0.0, r.applyTo(c.getXAxis()).distance(tc.getXAxis()), 1.0e-10);
179             Assert.assertEquals(0.0, r.applyTo(c.getYAxis()).distance(tc.getYAxis()), 1.0e-10);
180             Assert.assertEquals(c.getTolerance(), ((Circle) t.apply(c)).getTolerance(), 1.0e-10);
181 
182             SubLimitAngle  sub = new LimitAngle(new S1Point(MathUtils.TWO_PI * random.nextDouble()),
183                                                 random.nextBoolean(), 1.0e-10).wholeHyperplane();
184             Vector3D psub = c.getPointAt(((LimitAngle) sub.getHyperplane()).getLocation().getAlpha());
185             SubLimitAngle tsub = (SubLimitAngle) t.apply(sub, c, tc);
186             Vector3D ptsub = tc.getPointAt(((LimitAngle) tsub.getHyperplane()).getLocation().getAlpha());
187             Assert.assertEquals(0.0, r.applyTo(psub).distance(ptsub), 1.0e-10);
188 
189         }
190     }
