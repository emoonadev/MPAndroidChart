43      @Test
44      public void testLowAccuracyExternalDifferentiation()
45          throws NumberIsTooSmallException, DimensionMismatchException,
46                 MaxCountExceededException, NoBracketingException {
47          // this test does not really test JacobianMatrices,
48          // it only shows that WITHOUT this class, attempting to recover
49          // the jacobians from external differentiation on simple integration
50          // results with low accuracy gives very poor results. In fact,
51          // the curves dy/dp = g(b) when b varies from 2.88 to 3.08 are
52          // essentially noise.
53          // This test is taken from Hairer, Norsett and Wanner book
54          // Solving Ordinary Differential Equations I (Nonstiff problems),
55          // the curves dy/dp = g(b) are in figure 6.5
56          FirstOrderIntegrator integ =
57              new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-4, 1.0e-4 }, new double[] { 1.0e-4, 1.0e-4 });
58          double hP = 1.0e-12;
59          SummaryStatistics residualsP0 = new SummaryStatistics();
60          SummaryStatistics residualsP1 = new SummaryStatistics();
61          for (double b = 2.88; b < 3.08; b += 0.001) {
62              Brusselator brusselator = new Brusselator(b);
63              double[] y = { 1.3, b };
64              integ.integrate(brusselator, 0, y, 20.0, y);
65              double[] yP = { 1.3, b + hP };
66              integ.integrate(brusselator, 0, yP, 20.0, yP);
67              residualsP0.addValue((yP[0] - y[0]) / hP - brusselator.dYdP0());
68              residualsP1.addValue((yP[1] - y[1]) / hP - brusselator.dYdP1());
69          }
70          Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) > 500);
71          Assert.assertTrue(residualsP0.getStandardDeviation() > 30);
72          Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) > 700);
73          Assert.assertTrue(residualsP1.getStandardDeviation() > 40);
74      }

76      @Test
77      public void testHighAccuracyExternalDifferentiation()
78          throws NumberIsTooSmallException, DimensionMismatchException,
79                 MaxCountExceededException, NoBracketingException, UnknownParameterException {
80          FirstOrderIntegrator integ =
81              new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-10, 1.0e-10 }, new double[] { 1.0e-10, 1.0e-10 });
82          double hP = 1.0e-12;
83          SummaryStatistics residualsP0 = new SummaryStatistics();
84          SummaryStatistics residualsP1 = new SummaryStatistics();
85          for (double b = 2.88; b < 3.08; b += 0.001) {
86              ParamBrusselator brusselator = new ParamBrusselator(b);
87              double[] y = { 1.3, b };
88              integ.integrate(brusselator, 0, y, 20.0, y);
89              double[] yP = { 1.3, b + hP };
90              brusselator.setParameter("b", b + hP);
91              integ.integrate(brusselator, 0, yP, 20.0, yP);
92              residualsP0.addValue((yP[0] - y[0]) / hP - brusselator.dYdP0());
93              residualsP1.addValue((yP[1] - y[1]) / hP - brusselator.dYdP1());
94          }
95          Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) > 0.02);
96          Assert.assertTrue((residualsP0.getMax() - residualsP0.getMin()) < 0.03);
97          Assert.assertTrue(residualsP0.getStandardDeviation() > 0.003);
98          Assert.assertTrue(residualsP0.getStandardDeviation() < 0.004);
99          Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) > 0.04);
100         Assert.assertTrue((residualsP1.getMax() - residualsP1.getMin()) < 0.05);
101         Assert.assertTrue(residualsP1.getStandardDeviation() > 0.007);
102         Assert.assertTrue(residualsP1.getStandardDeviation() < 0.008);
103     }

203     @Test
204     public void testFinalResult()
205         throws MaxCountExceededException, DimensionMismatchException,
206                NumberIsTooSmallException, NoBracketingException,
207                UnknownParameterException, MismatchedEquations {
208 
209         AbstractIntegrator integ =
210             new DormandPrince54Integrator(1.0e-8, 100.0, new double[] { 1.0e-10, 1.0e-10 }, new double[] { 1.0e-10, 1.0e-10 });
211         double[] y = new double[] { 0.0, 1.0 };
212         Circle circle = new Circle(y, 1.0, 1.0, 0.1);
213 
214         JacobianMatrices jacob = new JacobianMatrices(circle, Circle.CX, Circle.CY, Circle.OMEGA);
215         jacob.addParameterJacobianProvider(circle);
216         jacob.setInitialMainStateJacobian(circle.exactDyDy0(0));
217         jacob.setInitialParameterJacobian(Circle.CX, circle.exactDyDcx(0));
218         jacob.setInitialParameterJacobian(Circle.CY, circle.exactDyDcy(0));
219         jacob.setInitialParameterJacobian(Circle.OMEGA, circle.exactDyDom(0));
220 
221         ExpandableStatefulODE efode = new ExpandableStatefulODE(circle);
222         efode.setTime(0);
223         efode.setPrimaryState(y);
224         jacob.registerVariationalEquations(efode);
225 
226         integ.setMaxEvaluations(5000);
227 
228         double t = 18 * FastMath.PI;
229         integ.integrate(efode, t);
230         y = efode.getPrimaryState();
231         for (int i = 0; i < y.length; ++i) {
232             Assert.assertEquals(circle.exactY(t)[i], y[i], 1.0e-9);
233         }
234 
235         double[][] dydy0 = new double[2][2];
236         jacob.getCurrentMainSetJacobian(dydy0);
237         for (int i = 0; i < dydy0.length; ++i) {
238             for (int j = 0; j < dydy0[i].length; ++j) {
239                 Assert.assertEquals(circle.exactDyDy0(t)[i][j], dydy0[i][j], 1.0e-9);
240             }
241         }
242         double[] dydcx = new double[2];
243         jacob.getCurrentParameterJacobian(Circle.CX, dydcx);
244         for (int i = 0; i < dydcx.length; ++i) {
245             Assert.assertEquals(circle.exactDyDcx(t)[i], dydcx[i], 1.0e-7);
246         }
247         double[] dydcy = new double[2];
248         jacob.getCurrentParameterJacobian(Circle.CY, dydcy);
249         for (int i = 0; i < dydcy.length; ++i) {
250             Assert.assertEquals(circle.exactDyDcy(t)[i], dydcy[i], 1.0e-7);
251         }
252         double[] dydom = new double[2];
253         jacob.getCurrentParameterJacobian(Circle.OMEGA, dydom);
254         for (int i = 0; i < dydom.length; ++i) {
255             Assert.assertEquals(circle.exactDyDom(t)[i], dydom[i], 1.0e-7);
256         }
257     }

320     private static class Brusselator extends AbstractParameterizable
321         implements MainStateJacobianProvider, ParameterJacobianProvider {
322 
323         public static final String B = "b";
324 
325         private double b;
326 
327         public Brusselator(double b) {
328             super(B);
329             this.b = b;
330         }
331 
332         public int getDimension() {
333             return 2;
334         }
335 
336         public void computeDerivatives(double t, double[] y, double[] yDot) {
337             double prod = y[0] * y[0] * y[1];
338             yDot[0] = 1 + prod - (b + 1) * y[0];
339             yDot[1] = b * y[0] - prod;
340         }
341 
342         public void computeMainStateJacobian(double t, double[] y, double[] yDot,
343                                              double[][] dFdY) {
344             double p = 2 * y[0] * y[1];
345             double y02 = y[0] * y[0];
346             dFdY[0][0] = p - (1 + b);
347             dFdY[0][1] = y02;
348             dFdY[1][0] = b - p;
349             dFdY[1][1] = -y02;
350         }
351 
352         public void computeParameterJacobian(double t, double[] y, double[] yDot,
353                                              String paramName, double[] dFdP) {
354             if (isSupported(paramName)) {
355                 dFdP[0] = -y[0];
356                 dFdP[1] = y[0];
357             } else {
358                 dFdP[0] = 0;
359                 dFdP[1] = 0;
360             }
361         }
362 
363         public double dYdP0() {
364             return -1088.232716447743 + (1050.775747149553 + (-339.012934631828 + 36.52917025056327 * b) * b) * b;
365         }
366 
367         public double dYdP1() {
368             return 1502.824469929139 + (-1438.6974831849952 + (460.959476642384 - 49.43847385647082 * b) * b) * b;
369         }
370 
371     }

419     /** ODE representing a point moving on a circle with provided center and angular rate. */
420     private static class Circle extends AbstractParameterizable
421         implements MainStateJacobianProvider, ParameterJacobianProvider {
422 
423         public static final String CX = "cx";
424         public static final String CY = "cy";
425         public static final String OMEGA = "omega";
426 
427         private final double[] y0;
428         private double cx;
429         private double cy;
430         private double omega;
431 
432         public Circle(double[] y0, double cx, double cy, double omega) {
433             super(CX,CY,OMEGA);
434             this.y0    = y0.clone();
435             this.cx    = cx;
436             this.cy    = cy;
437             this.omega = omega;
438         }
439 
440         public int getDimension() {
441             return 2;
442         }
443 
444         public void computeDerivatives(double t, double[] y, double[] yDot) {
445             yDot[0] = omega * (cy - y[1]);
446             yDot[1] = omega * (y[0] - cx);
447         }
448 
449         public void computeMainStateJacobian(double t, double[] y,
450                                              double[] yDot, double[][] dFdY) {
451             dFdY[0][0] = 0;
452             dFdY[0][1] = -omega;
453             dFdY[1][0] = omega;
454             dFdY[1][1] = 0;
455         }
456 
457         public void computeParameterJacobian(double t, double[] y, double[] yDot,
458                                              String paramName, double[] dFdP)
459             throws UnknownParameterException {
460             complainIfNotSupported(paramName);
461             if (paramName.equals(CX)) {
462                 dFdP[0] = 0;
463                 dFdP[1] = -omega;
464             } else if (paramName.equals(CY)) {
465                 dFdP[0] = omega;
466                 dFdP[1] = 0;
467             }  else {
468                 dFdP[0] = cy - y[1];
469                 dFdP[1] = y[0] - cx;
470             }
471         }
472 
473         public double[] exactY(double t) {
474             double cos = FastMath.cos(omega * t);
475             double sin = FastMath.sin(omega * t);
476             double dx0 = y0[0] - cx;
477             double dy0 = y0[1] - cy;
478             return new double[] {
479                 cx + cos * dx0 - sin * dy0,
480                 cy + sin * dx0 + cos * dy0
481             };
482         }
483 
484         public double[][] exactDyDy0(double t) {
485             double cos = FastMath.cos(omega * t);
486             double sin = FastMath.sin(omega * t);
487             return new double[][] {
488                 { cos, -sin },
489                 { sin,  cos }
490             };
491         }
492 
493         public double[] exactDyDcx(double t) {
494             double cos = FastMath.cos(omega * t);
495             double sin = FastMath.sin(omega * t);
496             return new double[] {1 - cos, -sin};
497         }
498 
499         public double[] exactDyDcy(double t) {
500             double cos = FastMath.cos(omega * t);
501             double sin = FastMath.sin(omega * t);
502             return new double[] {sin, 1 - cos};
503         }
504 
505         public double[] exactDyDom(double t) {
506             double cos = FastMath.cos(omega * t);
507             double sin = FastMath.sin(omega * t);
508             double dx0 = y0[0] - cx;
509             double dy0 = y0[1] - cy;
510             return new double[] { -t * (sin * dx0 + cos * dy0) , t * (cos * dx0 - sin * dy0) };
511         }
512 
513     }

519         public static final String CX = "cx";
520         public static final String CY = "cy";
521         public static final String OMEGA = "omega";
522 
523         private final double[] y0;
524         private double cx;
525         private double cy;
526         private double omega;
527 
528         public ParameterizedCircle(double[] y0, double cx, double cy, double omega) {
529             super(CX,CY,OMEGA);
530             this.y0    = y0.clone();
531             this.cx    = cx;
532             this.cy    = cy;
533             this.omega = omega;
534         }
535 
536         public int getDimension() {
537             return 2;
538         }
539 
540         public void computeDerivatives(double t, double[] y, double[] yDot) {
541             yDot[0] = omega * (cy - y[1]);
542             yDot[1] = omega * (y[0] - cx);
543         }
544 
545         public double getParameter(final String name)
546             throws UnknownParameterException {
547             if (name.equals(CX)) {
548                 return cx;
549             } else if (name.equals(CY)) {
550                     return cy;
551             } else if (name.equals(OMEGA)) {
552                 return omega;
553             } else {
554                 throw new UnknownParameterException(name);
555             }
556         }
557 
558         public void setParameter(final String name, final double value)
559             throws UnknownParameterException {
560             if (name.equals(CX)) {
561                 cx = value;
562             } else if (name.equals(CY)) {
563                 cy = value;
564             } else if (name.equals(OMEGA)) {
565                 omega = value;
566             } else {
567                 throw new UnknownParameterException(name);
568             }
569         }

571         public double[] exactY(double t) {
572             double cos = FastMath.cos(omega * t);
573             double sin = FastMath.sin(omega * t);
574             double dx0 = y0[0] - cx;
575             double dy0 = y0[1] - cy;
576             return new double[] {
577                 cx + cos * dx0 - sin * dy0,
578                 cy + sin * dx0 + cos * dy0
579             };
580         }
581 
582         public double[][] exactDyDy0(double t) {
583             double cos = FastMath.cos(omega * t);
584             double sin = FastMath.sin(omega * t);
585             return new double[][] {
586                 { cos, -sin },
587                 { sin,  cos }
588             };
589         }
590 
591         public double[] exactDyDcx(double t) {
592             double cos = FastMath.cos(omega * t);
593             double sin = FastMath.sin(omega * t);
594             return new double[] {1 - cos, -sin};
595         }
596 
597         public double[] exactDyDcy(double t) {
598             double cos = FastMath.cos(omega * t);
599             double sin = FastMath.sin(omega * t);
600             return new double[] {sin, 1 - cos};
601         }
602 
603         public double[] exactDyDom(double t) {
604             double cos = FastMath.cos(omega * t);
605             double sin = FastMath.sin(omega * t);
606             double dx0 = y0[0] - cx;
607             double dy0 = y0[1] - cy;
608             return new double[] { -t * (sin * dx0 + cos * dy0) , t * (cos * dx0 - sin * dy0) };
609         }
