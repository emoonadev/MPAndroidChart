188     @Test
189     public void testMisc() {
190         UnivariateFunction f = new QuinticFunction();
191         double result;
192         // Static solve method
193         result = UnivariateSolverUtils.solve(f, -0.2, 0.2);
194         Assert.assertEquals(result, 0, 1E-8);
195         result = UnivariateSolverUtils.solve(f, -0.1, 0.3);
196         Assert.assertEquals(result, 0, 1E-8);
197         result = UnivariateSolverUtils.solve(f, -0.3, 0.45);
198         Assert.assertEquals(result, 0, 1E-6);
199         result = UnivariateSolverUtils.solve(f, 0.3, 0.7);
200         Assert.assertEquals(result, 0.5, 1E-6);
201         result = UnivariateSolverUtils.solve(f, 0.2, 0.6);
202         Assert.assertEquals(result, 0.5, 1E-6);
203         result = UnivariateSolverUtils.solve(f, 0.05, 0.95);
204         Assert.assertEquals(result, 0.5, 1E-6);
205         result = UnivariateSolverUtils.solve(f, 0.85, 1.25);
206         Assert.assertEquals(result, 1.0, 1E-6);
207         result = UnivariateSolverUtils.solve(f, 0.8, 1.2);
208         Assert.assertEquals(result, 1.0, 1E-6);
209         result = UnivariateSolverUtils.solve(f, 0.85, 1.75);
210         Assert.assertEquals(result, 1.0, 1E-6);
211         result = UnivariateSolverUtils.solve(f, 0.55, 1.45);
212         Assert.assertEquals(result, 1.0, 1E-6);
213         result = UnivariateSolverUtils.solve(f, 0.85, 5);
214         Assert.assertEquals(result, 1.0, 1E-6);
215     }
