171     @Test
172     public void testParseIgnoredWhitespace() {
173         Vector3D expected = new Vector3D(1, 1, 1);
174         ParsePosition pos1 = new ParsePosition(0);
175         String source1 = "{1;1;1}";
176         Assert.assertEquals(expected, vector3DFormat.parse(source1, pos1));
177         Assert.assertEquals(source1.length(), pos1.getIndex());
178         ParsePosition pos2 = new ParsePosition(0);
179         String source2 = " { 1 ; 1 ; 1 } ";
180         Assert.assertEquals(expected, vector3DFormat.parse(source2, pos2));
181         Assert.assertEquals(source2.length() - 1, pos2.getIndex());
182     }
