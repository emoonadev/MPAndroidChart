17  package org.apache.commons.math4.ml.distance;
18  
19  import org.junit.Assert;
20  import org.apache.commons.math4.ml.distance.DistanceMeasure;
21  import org.apache.commons.math4.ml.distance.EarthMoversDistance;
22  import org.junit.Test;
23  
24  /**
25   * Tests for {@link EarthMoversDistance} class.
26   */
27  public class EarthMoversDistanceTest {
28  
29      final DistanceMeasure distance = new EarthMoversDistance();
30      final double[] a = { 0.1, 0.3, 0.2, 0.0, 0.25, 0.15 };
31      final double[] b = { 0.3, 0.0, 0.4, 0.1, 0.0, 0.2 };
32  
33      @Test
34      public void testZero() {
35          Assert.assertEquals(0, distance.compute(a, a), 0d);
36      }
37  
38      @Test
39      public void testZero2() {
40          final double[] zero = new double[] { 0, 0 };
41          Assert.assertEquals(0, distance.compute(zero, zero), 0d);
42      }
43  
44      @Test
45      public void test() {
46          double expected = 0.65; //0.2 + 0.1 + 0.1 + 0.2 + 0.05 + 0.0;
47          Assert.assertEquals(expected, distance.compute(a, b), 1e-10);
48          Assert.assertEquals(expected, distance.compute(b, a), 1e-10);
49      }
50  }
