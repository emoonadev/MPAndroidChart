64      private Map<Integer, Fraction> generate() {
65          Map<Integer, Fraction> map = new HashMap<Integer, Fraction>();
66          Random r = new Random();
67          double dd=0;
68          for (int i = 0; i < 2000; ++i)
69              dd = r.nextDouble();
70              try {
71                  map.put(r.nextInt(), new Fraction(dd));
72              } catch (FractionConversionException e) {
73                  throw new IllegalStateException("Invalid :"+dd, e);
74              }
75          return map;
76      }

78      private OpenIntToFieldHashMap<Fraction> createFromJavaMap(Field<Fraction> field) {
79          OpenIntToFieldHashMap<Fraction> map = new OpenIntToFieldHashMap<Fraction>(field);
80          for (Map.Entry<Integer, Fraction> mapEntry : javaMap.entrySet()) {
81              map.put(mapEntry.getKey(), mapEntry.getValue());
82          }
83          return map;
84      }

104     private void assertPutAndGet(OpenIntToFieldHashMap<Fraction> map) {

108     private void assertPutAndGet(OpenIntToFieldHashMap<Fraction> map, int mapSize,
109             Set<Integer> keysInMap) {
110         Assert.assertEquals(mapSize, map.size());
111         for (Map.Entry<Integer, Fraction> mapEntry : javaMap.entrySet()) {
112             map.put(mapEntry.getKey(), mapEntry.getValue());
113             if (!keysInMap.contains(mapEntry.getKey()))
114                 ++mapSize;
115             Assert.assertEquals(mapSize, map.size());
116             Assert.assertEquals(mapEntry.getValue(), map.get(mapEntry.getKey()));
117         }
118     }

213     /**
214      * Returns a map with at least 100 elements where each element is absent from javaMap.
215      */
216     private Map<Integer, Fraction> generateAbsent() {
217         Map<Integer, Fraction> generated = new HashMap<Integer, Fraction>();
218         do {
219             generated.putAll(generate());
220             for (Integer key : javaMap.keySet())
221                 generated.remove(key);
222         } while (generated.size() < 100);
223         return generated;
224     }

236     @Test
237     public void testContainsKey() {
238         OpenIntToFieldHashMap<Fraction> map = createFromJavaMap(field);
239         for (Entry<Integer, Fraction> mapEntry : javaMap.entrySet()) {
240             Assert.assertTrue(map.containsKey(mapEntry.getKey()));
241         }
242         for (Map.Entry<Integer, Fraction> mapEntry : generateAbsent().entrySet()) {
243             Assert.assertFalse(map.containsKey(mapEntry.getKey()));
244         }
245         for (Entry<Integer, Fraction> mapEntry : javaMap.entrySet()) {
246             int key = mapEntry.getKey();
247             Assert.assertTrue(map.containsKey(key));
248             map.remove(key);
249             Assert.assertFalse(map.containsKey(key));
250         }
251     }

253     @Test
254     public void testIterator() {
255         OpenIntToFieldHashMap<Fraction> map = createFromJavaMap(field);
256         OpenIntToFieldHashMap<Fraction>.Iterator iterator = map.iterator();
257         for (int i = 0; i < map.size(); ++i) {
258             Assert.assertTrue(iterator.hasNext());
259             iterator.advance();
260             int key = iterator.key();
261             Assert.assertTrue(map.containsKey(key));
262             Assert.assertEquals(javaMap.get(key), map.get(key));
263             Assert.assertEquals(javaMap.get(key), iterator.value());
264             Assert.assertTrue(javaMap.containsKey(key));
265         }
266         Assert.assertFalse(iterator.hasNext());
267         try {
268             iterator.advance();
269             Assert.fail("an exception should have been thrown");
270         } catch (NoSuchElementException nsee) {
271             // expected
272         }
273     }

288     /**
289      * Regression test for a bug in findInsertionIndex where the hashing in the second probing
290      * loop was inconsistent with the first causing duplicate keys after the right sequence
291      * of puts and removes.
292      */
293     @Test
294     public void testPutKeysWithCollisions() {
295         OpenIntToFieldHashMap<Fraction> map = new OpenIntToFieldHashMap<Fraction>(field);
296         int key1 = -1996012590;
297         Fraction value1 = new Fraction(1);
298         map.put(key1, value1);
299         int key2 = 835099822;
300         map.put(key2, value1);
301         int key3 = 1008859686;
302         map.put(key3, value1);
303         Assert.assertEquals(value1, map.get(key3));
304         Assert.assertEquals(3, map.size());
305 
306         map.remove(key2);
307         Fraction value2 = new Fraction(2);
308         map.put(key3, value2);
309         Assert.assertEquals(value2, map.get(key3));
310         Assert.assertEquals(2, map.size());
311     }
