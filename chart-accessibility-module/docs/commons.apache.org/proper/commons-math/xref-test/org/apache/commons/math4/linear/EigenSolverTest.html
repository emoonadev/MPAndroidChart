57      /** test invertible matrix */
58      @Test
59      public void testInvertible() {
60          Random r = new Random(9994100315209l);
61          RealMatrix m =
62              EigenDecompositionTest.createTestMatrix(r, new double[] { 1.0, 0.5, -1.0, -2.0, -3.0 });
63          DecompositionSolver es = new EigenDecomposition(m).getSolver();
64          Assert.assertTrue(es.isNonSingular());
65          RealMatrix inverse = es.getInverse();
66          RealMatrix error =
67              m.multiply(inverse).subtract(MatrixUtils.createRealIdentityMatrix(m.getRowDimension()));
68          Assert.assertEquals(0, error.getNorm(), 4.0e-15);
69      }

71      /**
72       * Verifies operation on very small values.
73       * Matrix with eigenvalues {8e-100, -1e-100, -1e-100}
74       */
75      @Test
76      public void testInvertibleTinyValues() {
77          final double tiny = 1e-100;
78          RealMatrix m = MatrixUtils.createRealMatrix(new double[][] {
79                  {3,  2,  4},
80                  {2,  0,  2},
81                  {4,  2,  3}
82          });
83          m = m.scalarMultiply(tiny);
84  
85          final EigenDecomposition ed = new EigenDecomposition(m);
86          RealMatrix inv = ed.getSolver().getInverse();
87  
88          final RealMatrix id = m.multiply(inv);
89          for (int i = 0; i < m.getRowDimension(); i++) {
90              for (int j = 0; j < m.getColumnDimension(); j++) {
91                  if (i == j) {
92                      Assert.assertTrue(Precision.equals(1, id.getEntry(i, j), 1e-15));
93                  } else {
94                      Assert.assertTrue(Precision.equals(0, id.getEntry(i, j), 1e-15));
95                  }
96              }
97          }
98      }

142             es.solve(b.getColumnVector(0));
143             Assert.fail("an exception should have been thrown");
144         } catch (MathIllegalArgumentException iae) {
145             // expected behavior
146         }
147         try {
148             es.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
149             Assert.fail("an exception should have been thrown");
150         } catch (MathIllegalArgumentException iae) {
151             // expected behavior
152         }

155     /** test solve */
156     @Test
157     public void testSolve() {
158         RealMatrix m = MatrixUtils.createRealMatrix(new double[][] {
159                 { 91,  5, 29, 32, 40, 14 },
160                 {  5, 34, -1,  0,  2, -1 },
161                 { 29, -1, 12,  9, 21,  8 },
162                 { 32,  0,  9, 14,  9,  0 },
163                 { 40,  2, 21,  9, 51, 19 },
164                 { 14, -1,  8,  0, 19, 14 }
165         });
166         DecompositionSolver es = new EigenDecomposition(m).getSolver();
167         RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
168                 { 1561, 269, 188 },
169                 {   69, -21,  70 },
170                 {  739, 108,  63 },
171                 {  324,  86,  59 },
172                 { 1624, 194, 107 },
173                 {  796,  69,  36 }
174         });
175         RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
176                 { 1,   2, 1 },
177                 { 2,  -1, 2 },
178                 { 4,   2, 3 },
179                 { 8,  -1, 0 },
180                 { 16,  2, 0 },
181                 { 32, -1, 0 }
182         });
183 
184         // using RealMatrix
185         RealMatrix solution=es.solve(b);
186         Assert.assertEquals(0, solution.subtract(xRef).getNorm(), 2.5e-12);
187 
188         // using RealVector
189         for (int i = 0; i < b.getColumnDimension(); ++i) {
190             Assert.assertEquals(0,
191                          es.solve(b.getColumnVector(i)).subtract(xRef.getColumnVector(i)).getNorm(),
192                          2.0e-11);
193         }
194 
195         // using RealVector with an alternate implementation
196         for (int i = 0; i < b.getColumnDimension(); ++i) {
197             ArrayRealVectorTest.RealVectorTestImpl v =
198                 new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
199             Assert.assertEquals(0,
200                          es.solve(v).subtract(xRef.getColumnVector(i)).getNorm(),
201                          2.0e-11);
202         }
203     }
