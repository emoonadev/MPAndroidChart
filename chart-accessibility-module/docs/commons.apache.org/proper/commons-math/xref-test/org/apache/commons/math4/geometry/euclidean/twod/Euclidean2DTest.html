17  package org.apache.commons.math4.geometry.euclidean.twod;
18  
19  import org.apache.commons.math4.TestUtils;
20  import org.apache.commons.math4.geometry.Space;
21  import org.apache.commons.math4.geometry.euclidean.oned.Euclidean1D;
22  import org.apache.commons.math4.geometry.euclidean.twod.Euclidean2D;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class Euclidean2DTest {
27  
28      @Test
29      public void testDimension() {
30          Assert.assertEquals(2, Euclidean2D.getInstance().getDimension());
31      }
32  
33      @Test
34      public void testSubSpace() {
35          Assert.assertTrue(Euclidean1D.getInstance() == Euclidean2D.getInstance().getSubSpace());
36      }
37  
38      @Test
39      public void testSerialization() {
40          Space e2 = Euclidean2D.getInstance();
41          Space deserialized = (Space) TestUtils.serializeAndRecover(e2);
42          Assert.assertTrue(e2 == deserialized);
43      }
44  
45  }
