17  package org.apache.commons.math4.exception;
18  
19  import org.apache.commons.math4.exception.NonMonotonicSequenceException;
20  import org.apache.commons.math4.util.MathArrays;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test for {@link NonMonotonicSequenceException}.
26   *
27   */
28  public class NonMonotonicSequenceExceptionTest {
29      @Test
30      public void testAccessors() {
31          NonMonotonicSequenceException e
32              = new NonMonotonicSequenceException(0, -1, 1, MathArrays.OrderDirection.DECREASING, false);
33          Assert.assertEquals(0, e.getArgument());
34          Assert.assertEquals(-1, e.getPrevious());
35          Assert.assertEquals(1, e.getIndex());
36          Assert.assertTrue(e.getDirection() == MathArrays.OrderDirection.DECREASING);
37          Assert.assertFalse(e.getStrict());
38  
39          e = new NonMonotonicSequenceException(-1, 0, 1);
40          Assert.assertEquals(-1, e.getArgument());
41          Assert.assertEquals(0, e.getPrevious());
42          Assert.assertEquals(1, e.getIndex());
43          Assert.assertTrue(e.getDirection() == MathArrays.OrderDirection.INCREASING);
44          Assert.assertTrue(e.getStrict());
45      }
46  }
