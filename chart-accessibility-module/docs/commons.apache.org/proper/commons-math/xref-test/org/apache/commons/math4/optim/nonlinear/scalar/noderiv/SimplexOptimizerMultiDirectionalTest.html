54          final FourExtrema fourExtrema = new FourExtrema();

135     @Test
136     public void testRosenbrock() {
137         MultivariateFunction rosenbrock
138             = new MultivariateFunction() {
139                     public double value(double[] x) {
140                         ++count;
141                         double a = x[1] - x[0] * x[0];
142                         double b = 1.0 - x[0];
143                         return 100 * a * a + b * b;
144                     }
145                 };
146 
147         count = 0;
148         SimplexOptimizer optimizer = new SimplexOptimizer(-1, 1e-3);
149         PointValuePair optimum
150            = optimizer.optimize(new MaxEval(100),
151                                 new ObjectiveFunction(rosenbrock),
152                                 GoalType.MINIMIZE,
153                                 new InitialGuess(new double[] { -1.2, 1 }),
154                                 new MultiDirectionalSimplex(new double[][] {
155                                         { -1.2,  1.0 },
156                                         { 0.9, 1.2 },
157                                         {  3.5, -2.3 } }));
158 
159         Assert.assertEquals(count, optimizer.getEvaluations());
160         Assert.assertTrue(optimizer.getEvaluations() > 50);
161         Assert.assertTrue(optimizer.getEvaluations() < 100);
162         Assert.assertTrue(optimum.getValue() > 1e-2);
163     }

165     @Test
166     public void testPowell() {
167         MultivariateFunction powell
168             = new MultivariateFunction() {
169                     public double value(double[] x) {
170                         ++count;
171                         double a = x[0] + 10 * x[1];
172                         double b = x[2] - x[3];
173                         double c = x[1] - 2 * x[2];
174                         double d = x[0] - x[3];
175                         return a * a + 5 * b * b + c * c * c * c + 10 * d * d * d * d;
176                     }
177                 };
178 
179         count = 0;
180         SimplexOptimizer optimizer = new SimplexOptimizer(-1, 1e-3);
181         PointValuePair optimum
182             = optimizer.optimize(new MaxEval(1000),
183                                  new ObjectiveFunction(powell),
184                                  GoalType.MINIMIZE,
185                                  new InitialGuess(new double[] { 3, -1, 0, 1 }),
186                                  new MultiDirectionalSimplex(4));
187         Assert.assertEquals(count, optimizer.getEvaluations());
188         Assert.assertTrue(optimizer.getEvaluations() > 800);
189         Assert.assertTrue(optimizer.getEvaluations() < 900);
190         Assert.assertTrue(optimum.getValue() > 1e-2);
191     }

193     @Test
194     public void testMath283() {
195         // fails because MultiDirectional.iterateSimplex is looping forever
196         // the while(true) should be replaced with a convergence check
197         SimplexOptimizer optimizer = new SimplexOptimizer(1e-14, 1e-14);
198         final Gaussian2D function = new Gaussian2D(0, 0, 1);
199         PointValuePair estimate = optimizer.optimize(new MaxEval(1000),
200                                                      new ObjectiveFunction(function),
201                                                      GoalType.MAXIMIZE,
202                                                      new InitialGuess(function.getMaximumPosition()),
203                                                      new MultiDirectionalSimplex(2));
204         final double EPSILON = 1e-5;
205         final double expectedMaximum = function.getMaximum();
206         final double actualMaximum = estimate.getValue();
207         Assert.assertEquals(expectedMaximum, actualMaximum, EPSILON);
208 
209         final double[] expectedPosition = function.getMaximumPosition();
210         final double[] actualPosition = estimate.getPoint();
211         Assert.assertEquals(expectedPosition[0], actualPosition[0], EPSILON );
212         Assert.assertEquals(expectedPosition[1], actualPosition[1], EPSILON );
213     }

215     private static class FourExtrema implements MultivariateFunction {
216         // The following function has 4 local extrema.
217         final double xM = -3.841947088256863675365;
218         final double yM = -1.391745200270734924416;
219         final double xP =  0.2286682237349059125691;
220         final double yP = -yM;
221         final double valueXmYm = 0.2373295333134216789769; // Local maximum.
222         final double valueXmYp = -valueXmYm; // Local minimum.
223         final double valueXpYm = -0.7290400707055187115322; // Global minimum.
224         final double valueXpYp = -valueXpYm; // Global maximum.
225 
226         public double value(double[] variables) {
227             final double x = variables[0];
228             final double y = variables[1];
229             return (x == 0 || y == 0) ? 0 :
230                 FastMath.atan(x) * FastMath.atan(x + 2) * FastMath.atan(y) * FastMath.atan(y) / (x * y);
231         }
232     }
233 
234     private static class Gaussian2D implements MultivariateFunction {
235         private final double[] maximumPosition;
236         private final double std;
237 
238         public Gaussian2D(double xOpt, double yOpt, double std) {
239             maximumPosition = new double[] { xOpt, yOpt };
240             this.std = std;
241         }
242 
243         public double getMaximum() {
244             return value(maximumPosition);
245         }
246 
247         public double[] getMaximumPosition() {
248             return maximumPosition.clone();
249         }
250 
251         public double value(double[] point) {
252             final double x = point[0], y = point[1];
253             final double twoS2 = 2.0 * std * std;
254             return 1.0 / (twoS2 * FastMath.PI) * FastMath.exp(-(x * x + y * y) / twoS2);
255         }
256     }
