26  /**
27   * Test cases for {@link TriangularDistribution}. See class javadoc for

30  public class TriangularDistributionTest extends RealDistributionAbstractTest {
31  
32      // --- Override tolerance -------------------------------------------------
33  
34      @Override
35      public void setUp() {
36          super.setUp();
37          setTolerance(1e-4);
38      }

40      //--- Implementations for abstract methods --------------------------------
41  
42      /**
43       * Creates the default triangular distribution instance to use in tests.
44       */
45      @Override
46      public TriangularDistribution makeDistribution() {
47          // Left side 5 wide, right side 10 wide.
48          return new TriangularDistribution(-3, 2, 12);
49      }
50  
51      /**
52       * Creates the default cumulative probability distribution test input
53       * values.
54       */
55      @Override
56      public double[] makeCumulativeTestPoints() {
57          return new double[] { -3.0001,                 // below lower limit
58                                -3.0,                    // at lower limit
59                                -2.0, -1.0, 0.0, 1.0,    // on lower side
60                                2.0,                     // at mode
61                                3.0, 4.0, 10.0, 11.0,    // on upper side
62                                12.0,                    // at upper limit
63                                12.0001                  // above upper limit
64                              };
65      }
66  
67      /**
68       * Creates the default cumulative probability density test expected values.
69       */
70      @Override
71      public double[] makeCumulativeTestValues() {
72          // Top at 2 / (b - a) = 2 / (12 - -3) = 2 / 15 = 7.5
73          // Area left  = 7.5 * 5  * 0.5 = 18.75 (1/3 of the total area)
74          // Area right = 7.5 * 10 * 0.5 = 37.5  (2/3 of the total area)
75          // Area total = 18.75 + 37.5 = 56.25
76          // Derivative left side = 7.5 / 5 = 1.5
77          // Derivative right side = -7.5 / 10 = -0.75
78          double third = 1 / 3.0;
79          double left = 18.75;
80          double area = 56.25;
81          return new double[] { 0.0,
82                                0.0,
83                                0.75 / area, 3 / area, 6.75 / area, 12 / area,
84                                third,
85                                (left + 7.125) / area, (left + 13.5) / area,
86                                (left + 36) / area, (left + 37.125) / area,
87                                1.0,
88                                1.0
89                              };
90      }
91  
92      /**
93       * Creates the default inverse cumulative probability distribution test
94       * input values.
95       */
96      @Override
97      public double[] makeInverseCumulativeTestPoints() {
98          // Exclude the points outside the limits, as they have cumulative
99          // probability of zero and one, meaning the inverse returns the
100         // limits and not the points outside the limits.
101         double[] points = makeCumulativeTestValues();
102         double[] points2 = new double[points.length-2];
103         System.arraycopy(points, 1, points2, 0, points2.length);
104         return points2;
105         //return Arrays.copyOfRange(points, 1, points.length - 1);
106     }
107

139     /** Test lower bound getter. */
140     @Test
141     public void testGetLowerBound() {
142         TriangularDistribution distribution = makeDistribution();
143         Assert.assertEquals(-3.0, distribution.getSupportLowerBound(), 0);
144     }
145 
146     /** Test upper bound getter. */
147     @Test
148     public void testGetUpperBound() {
149         TriangularDistribution distribution = makeDistribution();
150         Assert.assertEquals(12.0, distribution.getSupportUpperBound(), 0);
151     }

153     /** Test pre-condition for equal lower/upper limit. */
154     @Test(expected=NumberIsTooLargeException.class)
155     public void testPreconditions1() {
156         new TriangularDistribution(0, 0, 0);
157     }
158 
159     /** Test pre-condition for lower limit larger than upper limit. */
160     @Test(expected=NumberIsTooLargeException.class)
161     public void testPreconditions2() {
162         new TriangularDistribution(1, 1, 0);
163     }
164 
165     /** Test pre-condition for mode larger than upper limit. */
166     @Test(expected=NumberIsTooLargeException.class)
167     public void testPreconditions3() {
168         new TriangularDistribution(0, 2, 1);
169     }
170 
171     /** Test pre-condition for mode smaller than lower limit. */
172     @Test(expected=NumberIsTooSmallException.class)
173     public void testPreconditions4() {
174         new TriangularDistribution(2, 1, 3);
175     }
176 
177     /** Test mean/variance. */
178     @Test
179     public void testMeanVariance() {
180         TriangularDistribution dist;
181 
182         dist = new TriangularDistribution(0, 0.5, 1.0);
183         Assert.assertEquals(dist.getNumericalMean(), 0.5, 0);
184         Assert.assertEquals(dist.getNumericalVariance(), 1 / 24.0, 0);
185 
186         dist = new TriangularDistribution(0, 1, 1);
187         Assert.assertEquals(dist.getNumericalMean(), 2 / 3.0, 0);
188         Assert.assertEquals(dist.getNumericalVariance(), 1 / 18.0, 0);
189 
190         dist = new TriangularDistribution(-3, 2, 12);
191         Assert.assertEquals(dist.getNumericalMean(), 3 + (2 / 3.0), 0);
192         Assert.assertEquals(dist.getNumericalVariance(), 175 / 18.0, 0);
193     }
