175     private RealMatrix checkAEqualPHPt(RealMatrix matrix) {
176         HessenbergTransformer transformer = new HessenbergTransformer(matrix);
177         RealMatrix p  = transformer.getP();
178         RealMatrix pT = transformer.getPT();
179         RealMatrix h  = transformer.getH();
180 
181         RealMatrix result = p.multiply(h).multiply(pT);
182         double norm = result.subtract(matrix).getNorm();
183         Assert.assertEquals(0, norm, 1.0e-10);
184 
185         for (int i = 0; i < matrix.getRowDimension(); ++i) {
186             for (int j = 0; j < matrix.getColumnDimension(); ++j) {
187                 if (i > j + 1) {
188                     Assert.assertEquals(matrix.getEntry(i, j), result.getEntry(i, j), 1.0e-12);
189                 }
190             }
191         }
192 
193         return transformer.getH();
194     }
195 
196     private void checkOrthogonal(RealMatrix m) {
197         RealMatrix mTm = m.transpose().multiply(m);
198         RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
199         Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-14);
200     }
201 
202     private void checkHessenbergForm(RealMatrix m) {
203         final int rows = m.getRowDimension();
204         final int cols = m.getColumnDimension();
205         for (int i = 0; i < rows; ++i) {
206             for (int j = 0; j < cols; ++j) {
207                 if (i > j + 1) {
208                     Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
209                 }
210             }
211         }
212     }
213 
214     private void checkMatricesValues(double[][] matrix, double[][] pRef, double[][] hRef) {
215         HessenbergTransformer transformer =
216             new HessenbergTransformer(MatrixUtils.createRealMatrix(matrix));
217 
218         // check values against known references
219         RealMatrix p = transformer.getP();
220         Assert.assertEquals(0, p.subtract(MatrixUtils.createRealMatrix(pRef)).getNorm(), 1.0e-14);
221 
222         RealMatrix h = transformer.getH();
223         Assert.assertEquals(0, h.subtract(MatrixUtils.createRealMatrix(hRef)).getNorm(), 1.0e-14);
224 
225         // check the same cached instance is returned the second time
226         Assert.assertTrue(p == transformer.getP());
227         Assert.assertTrue(h == transformer.getH());
228     }
