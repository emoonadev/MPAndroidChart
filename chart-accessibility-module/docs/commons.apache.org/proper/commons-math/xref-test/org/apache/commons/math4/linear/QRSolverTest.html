118     /** test solve */
119     @Test
120     public void testSolve() {
121         QRDecomposition decomposition =
122             new QRDecomposition(MatrixUtils.createRealMatrix(testData3x3NonSingular));
123         DecompositionSolver solver = decomposition.getSolver();
124         RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
125                 { -102, 12250 }, { 544, 24500 }, { 167, -36750 }
126         });
127         RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
128                 { 1, 2515 }, { 2, 422 }, { -3, 898 }
129         });
130 
131         // using RealMatrix
132         Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), 2.0e-16 * xRef.getNorm());
133 
134         // using ArrayRealVector
135         for (int i = 0; i < b.getColumnDimension(); ++i) {
136             final RealVector x = solver.solve(b.getColumnVector(i));
137             final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
138             Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
139         }
140 
141         // using RealVector with an alternate implementation
142         for (int i = 0; i < b.getColumnDimension(); ++i) {
143             ArrayRealVectorTest.RealVectorTestImpl v =
144                 new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
145             final RealVector x = solver.solve(v);
146             final double error = x.subtract(xRef.getColumnVector(i)).getNorm();
147             Assert.assertEquals(0, error, 3.0e-16 * xRef.getColumnVector(i).getNorm());
148         }
149 
150     }

152     @Test
153     public void testOverdetermined() {
154         final Random r    = new Random(5559252868205245l);
155         int          p    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
156         int          q    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
157         RealMatrix   a    = createTestMatrix(r, p, q);
158         RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
159 
160         // build a perturbed system: A.X + noise = B
161         RealMatrix b = a.multiply(xRef);
162         final double noise = 0.001;
163         b.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
164             @Override
165             public double visit(int row, int column, double value) {
166                 return value * (1.0 + noise * (2 * r.nextDouble() - 1));
167             }
168         });
169 
170         // despite perturbation, the least square solution should be pretty good
171         RealMatrix x = new QRDecomposition(a).getSolver().solve(b);
172         Assert.assertEquals(0, x.subtract(xRef).getNorm(), 0.01 * noise * p * q);
173 
174     }

176     @Test
177     public void testUnderdetermined() {
178         final Random r    = new Random(42185006424567123l);
179         int          p    = (5 * BlockRealMatrix.BLOCK_SIZE) / 4;
180         int          q    = (7 * BlockRealMatrix.BLOCK_SIZE) / 4;
181         RealMatrix   a    = createTestMatrix(r, p, q);
182         RealMatrix   xRef = createTestMatrix(r, q, BlockRealMatrix.BLOCK_SIZE + 3);
183         RealMatrix   b    = a.multiply(xRef);
184         RealMatrix   x = new QRDecomposition(a).getSolver().solve(b);
185 
186         // too many equations, the system cannot be solved at all
187         Assert.assertTrue(x.subtract(xRef).getNorm() / (p * q) > 0.01);
188 
189         // the last unknown should have been set to 0
190         Assert.assertEquals(0.0, x.getSubMatrix(p, q - 1, 0, x.getColumnDimension() - 1).getNorm(), 0);
191     }
192 
193     private RealMatrix createTestMatrix(final Random r, final int rows, final int columns) {
194         RealMatrix m = MatrixUtils.createRealMatrix(rows, columns);
195         m.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
196                 @Override
197                     public double visit(int row, int column, double value) {
198                     return 2.0 * r.nextDouble() - 1.0;
199                 }
200             });
201         return m;
202     }
