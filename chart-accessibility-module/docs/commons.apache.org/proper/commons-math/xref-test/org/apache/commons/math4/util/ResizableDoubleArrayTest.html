48      @Test
49      public void testConstructors() {
50          float defaultExpansionFactor = 2.0f;
51          double defaultContractionCriteria = 2.5;
52          ExpansionMode defaultMode = ResizableDoubleArray.ExpansionMode.MULTIPLICATIVE;
53  
54          ResizableDoubleArray testDa = new ResizableDoubleArray(2);
55          Assert.assertEquals(0, testDa.getNumElements());
56          Assert.assertEquals(2, testDa.getCapacity());
57          Assert.assertEquals(defaultExpansionFactor, testDa.getExpansionFactor(), 0);
58          Assert.assertEquals(defaultContractionCriteria, testDa.getContractionCriterion(), 0);
59          Assert.assertEquals(defaultMode, testDa.getExpansionMode());
60          try {
61              da = new ResizableDoubleArray(-1);
62              Assert.fail("Expecting MathIllegalArgumentException");
63          } catch (MathIllegalArgumentException ex) {
64              // expected
65          }
66  
67          testDa = new ResizableDoubleArray((double[]) null);
68          Assert.assertEquals(0, testDa.getNumElements());
69  
70          double[] initialArray = new double[] { 0, 1, 2 };
71          testDa = new ResizableDoubleArray(initialArray);
72          Assert.assertEquals(3, testDa.getNumElements());
73  
74          testDa = new ResizableDoubleArray(2, 2.0);
75          Assert.assertEquals(0, testDa.getNumElements());
76          Assert.assertEquals(2, testDa.getCapacity());
77          Assert.assertEquals(defaultExpansionFactor, testDa.getExpansionFactor(), 0);
78          Assert.assertEquals(defaultContractionCriteria, testDa.getContractionCriterion(), 0);
79          Assert.assertEquals(defaultMode, testDa.getExpansionMode());
80  
81          try {
82              da = new ResizableDoubleArray(2, 0.5);
83              Assert.fail("Expecting MathIllegalArgumentException");
84          } catch (MathIllegalArgumentException ex) {
85              // expected
86          }
87  
88          testDa = new ResizableDoubleArray(2, 3.0);
89          Assert.assertEquals(3.0f, testDa.getExpansionFactor(), 0);
90          Assert.assertEquals(3.5f, testDa.getContractionCriterion(), 0);
91  
92          testDa = new ResizableDoubleArray(2, 2.0, 3.0);
93          Assert.assertEquals(0, testDa.getNumElements());
94          Assert.assertEquals(2, testDa.getCapacity());
95          Assert.assertEquals(defaultExpansionFactor, testDa.getExpansionFactor(), 0);
96          Assert.assertEquals(3.0f, testDa.getContractionCriterion(), 0);
97          Assert.assertEquals(defaultMode, testDa.getExpansionMode());
98  
99          try {
100             da = new ResizableDoubleArray(2, 2.0, 1.5);
101             Assert.fail("Expecting MathIllegalArgumentException");
102         } catch (MathIllegalArgumentException ex) {
103             // expected
104         }
105 
106         testDa = new ResizableDoubleArray(2, 2.0, 3.0, ResizableDoubleArray.ExpansionMode.ADDITIVE);
107         Assert.assertEquals(0, testDa.getNumElements());
108         Assert.assertEquals(2, testDa.getCapacity());
109         Assert.assertEquals(defaultExpansionFactor, testDa.getExpansionFactor(), 0);
110         Assert.assertEquals(3.0f, testDa.getContractionCriterion(), 0);
111         Assert.assertEquals(ResizableDoubleArray.ExpansionMode.ADDITIVE, testDa.getExpansionMode());
112 
113         try {
114             da = new ResizableDoubleArray(2, 2.0d, 2.5d, null);
115             Assert.fail("Expecting NullArgumentException");
116         } catch (NullArgumentException ex) {
117             // expected
118         }
119 
120         // Copy constructor
121         testDa = new ResizableDoubleArray(2, 2.0, 3.0,
122                                           ResizableDoubleArray.ExpansionMode.ADDITIVE);
123         testDa.addElement(2.0);
124         testDa.addElement(3.2);
125         ResizableDoubleArray copyDa = new ResizableDoubleArray(testDa);
126         Assert.assertEquals(copyDa, testDa);
127         Assert.assertEquals(testDa, copyDa);
128 
129         // JIRA: MATH-1252
130         final double[] values = {1};
131         testDa = new ResizableDoubleArray(values);
132         Assert.assertArrayEquals(values, testDa.getElements(), 0);
133         Assert.assertEquals(1, testDa.getNumElements());
134         Assert.assertEquals(1, testDa.getElement(0), 0);
135     }

207     @Test
208     public void testAddElements() {
209         ResizableDoubleArray testDa = new ResizableDoubleArray();
210 
211         // MULTIPLICATIVE_MODE
212         testDa.addElements(new double[] {4, 5, 6});
213         Assert.assertEquals(3, testDa.getNumElements(), 0);
214         Assert.assertEquals(4, testDa.getElement(0), 0);
215         Assert.assertEquals(5, testDa.getElement(1), 0);
216         Assert.assertEquals(6, testDa.getElement(2), 0);
217 
218         testDa.addElements(new double[] {4, 5, 6});
219         Assert.assertEquals(6, testDa.getNumElements());
220 
221         // ADDITIVE_MODE  (x's are occupied storage locations, 0's are open)
222         testDa = new ResizableDoubleArray(2, 2.0, 2.5,
223                                           ResizableDoubleArray.ExpansionMode.ADDITIVE);
224         Assert.assertEquals(2, testDa.getCapacity());
225         testDa.addElements(new double[] { 1d }); // x,0
226         testDa.addElements(new double[] { 2d }); // x,x
227         testDa.addElements(new double[] { 3d }); // x,x,x,0 -- expanded
228         Assert.assertEquals(1d, testDa.getElement(0), 0);
229         Assert.assertEquals(2d, testDa.getElement(1), 0);
230         Assert.assertEquals(3d, testDa.getElement(2), 0);
231         Assert.assertEquals(4, testDa.getCapacity());  // x,x,x,0
232         Assert.assertEquals(3, testDa.getNumElements());
233     }

238         super.testAddElementRolling();
239 
240         // MULTIPLICATIVE_MODE
241         da.clear();
242         da.addElement(1);
243         da.addElement(2);
244         da.addElementRolling(3);
245         Assert.assertEquals(3, da.getElement(1), 0);
246         da.addElementRolling(4);
247         Assert.assertEquals(3, da.getElement(0), 0);
248         Assert.assertEquals(4, da.getElement(1), 0);
249         da.addElement(5);
250         Assert.assertEquals(5, da.getElement(2), 0);
251         da.addElementRolling(6);
252         Assert.assertEquals(4, da.getElement(0), 0);
253         Assert.assertEquals(5, da.getElement(1), 0);
254         Assert.assertEquals(6, da.getElement(2), 0);
255 
256         // ADDITIVE_MODE  (x's are occupied storage locations, 0's are open)
257         ResizableDoubleArray testDa = new ResizableDoubleArray(2, 2.0, 2.5,
258                                                                ResizableDoubleArray.ExpansionMode.ADDITIVE);
259         Assert.assertEquals(2, testDa.getCapacity());
260         testDa.addElement(1d); // x,0
261         testDa.addElement(2d); // x,x
262         testDa.addElement(3d); // x,x,x,0 -- expanded
263         Assert.assertEquals(1d, testDa.getElement(0), 0);
264         Assert.assertEquals(2d, testDa.getElement(1), 0);
265         Assert.assertEquals(3d, testDa.getElement(2), 0);
266         Assert.assertEquals(4, testDa.getCapacity());  // x,x,x,0
267         Assert.assertEquals(3, testDa.getNumElements());
268         testDa.addElementRolling(4d);
269         Assert.assertEquals(2d, testDa.getElement(0), 0);
270         Assert.assertEquals(3d, testDa.getElement(1), 0);
271         Assert.assertEquals(4d, testDa.getElement(2), 0);
272         Assert.assertEquals(4, testDa.getCapacity());  // 0,x,x,x
273         Assert.assertEquals(3, testDa.getNumElements());
274         testDa.addElementRolling(5d);   // 0,0,x,x,x,0 -- time to contract
275         Assert.assertEquals(3d, testDa.getElement(0), 0);
276         Assert.assertEquals(4d, testDa.getElement(1), 0);
277         Assert.assertEquals(5d, testDa.getElement(2), 0);
278         Assert.assertEquals(4, testDa.getCapacity());  // contracted -- x,x,x,0
279         Assert.assertEquals(3, testDa.getNumElements());
280         try {
281             testDa.getElement(4);
282             Assert.fail("Expecting ArrayIndexOutOfBoundsException");
283         } catch (ArrayIndexOutOfBoundsException ex) {
284             // expected
285         }
286         try {
287             testDa.getElement(-1);
288             Assert.fail("Expecting ArrayIndexOutOfBoundsException");
289         } catch (ArrayIndexOutOfBoundsException ex) {
290             // expected
291         }

451     @Test
452     public void testEqualsAndHashCode() throws Exception {
453 
454         // Wrong type
455         ResizableDoubleArray first = new ResizableDoubleArray();
456         Double other = new Double(2);
457         Assert.assertFalse(first.equals(other));
458 
459         // Null
460         other = null;
461         Assert.assertFalse(first.equals(other));
462 
463         // Reflexive
464         Assert.assertTrue(first.equals(first));
465 
466         // Argumentless constructor
467         ResizableDoubleArray second = new ResizableDoubleArray();
468         verifyEquality(first, second);
469 
470         // Equals iff same data, same properties
471         ResizableDoubleArray third = new ResizableDoubleArray(3, 2.0, 2.0);
472         verifyInequality(third, first);
473         ResizableDoubleArray fourth = new ResizableDoubleArray(3, 2.0, 2.0);
474         ResizableDoubleArray fifth = new ResizableDoubleArray(2, 2.0, 2.0);
475         verifyEquality(third, fourth);
476         verifyInequality(third, fifth);
477         third.addElement(4.1);
478         third.addElement(4.2);
479         third.addElement(4.3);
480         fourth.addElement(4.1);
481         fourth.addElement(4.2);
482         fourth.addElement(4.3);
483         verifyEquality(third, fourth);
484 
485         // expand
486         fourth.addElement(4.4);
487         verifyInequality(third, fourth);
488         third.addElement(4.4);
489         verifyEquality(third, fourth);
490         fourth.addElement(4.4);
491         verifyInequality(third, fourth);
492         third.addElement(4.4);
493         verifyEquality(third, fourth);
494         fourth.addElementRolling(4.5);
495         third.addElementRolling(4.5);
496         verifyEquality(third, fourth);
497 
498         // discard
499         third.discardFrontElements(1);
500         verifyInequality(third, fourth);
501         fourth.discardFrontElements(1);
502         verifyEquality(third, fourth);
503 
504         // discard recent
505         third.discardMostRecentElements(2);
506         fourth.discardMostRecentElements(2);
507         verifyEquality(third, fourth);
508 
509         // wrong order
510         third.addElement(18);
511         fourth.addElement(17);
512         third.addElement(17);
513         fourth.addElement(18);
514         verifyInequality(third, fourth);
515 
516         // Copy constructor
517         verifyEquality(fourth, new ResizableDoubleArray(fourth));
518 
519         // Instance copy
520         verifyEquality(fourth, fourth.copy());
521 
522     }

541     @Test
542     public void testCompute() {
543         final ResizableDoubleArray a = new ResizableDoubleArray();
544         final int max = 20;
545         for (int i = 1; i <= max; i++) {
546             a.setElement(i, i);
547         }
548 
549         final MathArrays.Function add = new MathArrays.Function() {
550                 @Override
551                 public double evaluate(double[] a, int index, int num) {
552                     double sum = 0;
553                     final int max = index + num;
554                     for (int i = index; i < max; i++) {
555                         sum += a[i];
556                     }
557                     return sum;
558                 }
559                 @Override
560                 public double evaluate(double[] a) {
561                     return evaluate(a, 0, a.length);
562                 }
563             };
564 
565         final double sum = a.compute(add);
566         Assert.assertEquals(0.5 * max * (max + 1), sum, 0);
567     }

569     private void verifyEquality(ResizableDoubleArray a, ResizableDoubleArray b) {
570         Assert.assertTrue(b.equals(a));
571         Assert.assertTrue(a.equals(b));
572         Assert.assertEquals(a.hashCode(), b.hashCode());
573     }

575     private void verifyInequality(ResizableDoubleArray a, ResizableDoubleArray b) {
576         Assert.assertFalse(b.equals(a));
577         Assert.assertFalse(a.equals(b));
578         Assert.assertFalse(a.hashCode() == b.hashCode());
579     }
