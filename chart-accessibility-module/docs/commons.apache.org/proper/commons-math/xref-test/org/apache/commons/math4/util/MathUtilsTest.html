33      public void testEqualsDouble() {
34          final double x = 1234.5678;
35          Assert.assertTrue(MathUtils.equals(x, x));
36          Assert.assertFalse(MathUtils.equals(x, -x));
37  
38          // Special cases (cf. semantics of JDK's "Double").
39          // 1. NaN
40          Assert.assertTrue(MathUtils.equals(Double.NaN, Double.NaN));
41          // 2. Negative zero
42          final double mZero = -0d;
43          final double zero = 0d;
44          Assert.assertTrue(MathUtils.equals(zero, zero));
45          Assert.assertTrue(MathUtils.equals(mZero, mZero));
46          Assert.assertFalse(MathUtils.equals(mZero, zero));
47      }

49      @Test
50      public void testHash() {
51          double[] testArray = {
52              Double.NaN,
53              Double.POSITIVE_INFINITY,
54              Double.NEGATIVE_INFINITY,
55              1d,
56              0d,
57              1E-14,
58              (1 + 1E-14),
59              Double.MIN_VALUE,
60              Double.MAX_VALUE };
61          for (int i = 0; i < testArray.length; i++) {
62              for (int j = 0; j < testArray.length; j++) {
63                  if (i == j) {
64                      Assert.assertEquals(MathUtils.hash(testArray[i]), MathUtils.hash(testArray[j]));
65                      Assert.assertEquals(MathUtils.hash(testArray[j]), MathUtils.hash(testArray[i]));
66                  } else {
67                      Assert.assertTrue(MathUtils.hash(testArray[i]) != MathUtils.hash(testArray[j]));
68                      Assert.assertTrue(MathUtils.hash(testArray[j]) != MathUtils.hash(testArray[i]));
69                  }
70              }
71          }
72      }

74      @Test
75      public void testArrayHash() {

77          Assert.assertEquals(MathUtils.hash(new double[] {

79                                        Double.NEGATIVE_INFINITY, 1d, 0d
80                                      }),
81                       MathUtils.hash(new double[] {
82                                        Double.NaN, Double.POSITIVE_INFINITY,
83                                        Double.NEGATIVE_INFINITY, 1d, 0d
84                                      }));
85          Assert.assertFalse(MathUtils.hash(new double[] { 1d }) ==
86                      MathUtils.hash(new double[] { FastMath.nextAfter(1d, 2d) }));
87          Assert.assertFalse(MathUtils.hash(new double[] { 1d }) ==
88                      MathUtils.hash(new double[] { 1d, 1d }));

91      /**
92       * Make sure that permuted arrays do not hash to the same value.
93       */
94      @Test
95      public void testPermutedArrayHash() {
96          double[] original = new double[10];
97          double[] permuted = new double[10];
98  
99          final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_512_A,
100                                                                  64925784252L);
101 
102         // Generate 10 distinct random values
103         for (int i = 0; i < 10; i++) {
104             final RealDistribution.Sampler u
105                 = new UniformRealDistribution(i + 0.5, i + 0.75).createSampler(random);
106             original[i] = u.sample();
107         }
108 
109         // Generate a random permutation, making sure it is not the identity
110         boolean isIdentity = true;
111         do {
112             int[] permutation = MathArrays.natural(10);
113             MathArrays.shuffle(permutation, random);
114             for (int i = 0; i < 10; i++) {
115                 if (i != permutation[i]) {
116                     isIdentity = false;
117                     break;
118                 }
119                 permuted[i] = original[permutation[i]];
120             }
121         } while (isIdentity);
122 
123         // Verify that permuted array has different hash
124         Assert.assertFalse(MathUtils.hash(original) == MathUtils.hash(permuted));
125     }

155     @Test
156     public void testNormalizeAngle() {
157         for (double a = -15.0; a <= 15.0; a += 0.1) {
158             for (double b = -15.0; b <= 15.0; b += 0.2) {
159                 double c = MathUtils.normalizeAngle(a, b);
160                 Assert.assertTrue((b - FastMath.PI) <= c);
161                 Assert.assertTrue(c <= (b + FastMath.PI));
162                 double twoK = FastMath.rint((a - c) / FastMath.PI);
163                 Assert.assertEquals(c, a - twoK * FastMath.PI, 1.0e-14);
164             }
165         }
166     }

168     @Test
169     public void testReduce() {
170         final double period = -12.222;
171         final double offset = 13;
172 
173         final double delta = 1.5;
174 
175         double orig = offset + 122456789 * period + delta;
176         double expected = delta;
177         Assert.assertEquals(expected,
178                             MathUtils.reduce(orig, period, offset),
179                             1e-7);
180         Assert.assertEquals(expected,
181                             MathUtils.reduce(orig, -period, offset),
182                             1e-7);
183 
184         orig = offset - 123356789 * period - delta;
185         expected = FastMath.abs(period) - delta;
186         Assert.assertEquals(expected,
187                             MathUtils.reduce(orig, period, offset),
188                             1e-6);
189         Assert.assertEquals(expected,
190                             MathUtils.reduce(orig, -period, offset),
191                             1e-6);
192 
193         orig = offset - 123446789 * period + delta;
194         expected = delta;
195         Assert.assertEquals(expected,
196                             MathUtils.reduce(orig, period, offset),
197                             1e-6);
198         Assert.assertEquals(expected,
199                             MathUtils.reduce(orig, -period, offset),
200                             1e-6);
201 
202         Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, Double.NaN, offset)));
203         Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.NaN, period, offset)));
204         Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, period, Double.NaN)));
205         Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig, period,
206                 Double.POSITIVE_INFINITY)));
207         Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
208                 period, offset)));
209         Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig,
210                 Double.POSITIVE_INFINITY, offset)));
211         Assert.assertTrue(Double.isNaN(MathUtils.reduce(orig,
212                 Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY)));
213         Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
214                 period, Double.POSITIVE_INFINITY)));
215         Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
216                 Double.POSITIVE_INFINITY, offset)));
217         Assert.assertTrue(Double.isNaN(MathUtils.reduce(Double.POSITIVE_INFINITY,
218                 Double.POSITIVE_INFINITY,  Double.POSITIVE_INFINITY)));
219     }

221     @Test
222     public void testReduceComparedWithNormalizeAngle() {
223         final double tol = Math.ulp(1d);
224         final double period = 2 * Math.PI;
225         for (double a = -15; a <= 15; a += 0.5) {
226             for (double center = -15; center <= 15; center += 1) {
227                 final double nA = MathUtils.normalizeAngle(a, center);
228                 final double offset = center - Math.PI;
229                 final double r = MathUtils.reduce(a, period, offset);
230                 Assert.assertEquals(nA, r + offset, tol);
231             }
232         }
233     }

324     @Test
325     public void testCopySignByte() {
326         byte a = MathUtils.copySign(Byte.MIN_VALUE, (byte) -1);
327         Assert.assertEquals(Byte.MIN_VALUE, a);
328 
329         final byte minValuePlusOne = Byte.MIN_VALUE + (byte) 1;
330         a = MathUtils.copySign(minValuePlusOne, (byte) 1);
331         Assert.assertEquals(Byte.MAX_VALUE, a);
332 
333         a = MathUtils.copySign(Byte.MAX_VALUE, (byte) -1);
334         Assert.assertEquals(minValuePlusOne, a);
335 
336         final byte one = 1;
337         byte val = -2;
338         a = MathUtils.copySign(val, one);
339         Assert.assertEquals(-val, a);
340 
341         final byte minusOne = -one;
342         val = 2;
343         a = MathUtils.copySign(val, minusOne);
344         Assert.assertEquals(-val, a);
345 
346         val = 0;
347         a = MathUtils.copySign(val, minusOne);
348         Assert.assertEquals(val, a);
349 
350         val = 0;
351         a = MathUtils.copySign(val, one);
352         Assert.assertEquals(val, a);
353     }
