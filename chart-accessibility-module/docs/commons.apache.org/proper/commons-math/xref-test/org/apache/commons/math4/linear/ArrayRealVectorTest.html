19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.linear.ArrayRealVector;
21  import org.apache.commons.math4.linear.RealVector;
22  import org.junit.Assert;
23  import org.junit.Test;

25  /**
26   * Test cases for the {@link ArrayRealVector} class.
27   *
28   */
29  public class ArrayRealVectorTest extends RealVectorAbstractTest {
30  
31  
32      @Override
33      public RealVector create(final double[] data) {
34          return new ArrayRealVector(data, true);
35      }
36  
37      @Test
38      public void testConstructors() {
39          final double[] vec1 = {1d, 2d, 3d};
40          final double[] vec3 = {7d, 8d, 9d};
41          final double[] vec4 = {1d, 2d, 3d, 4d, 5d, 6d, 7d, 8d, 9d};
42          final Double[] dvec1 = {1d, 2d, 3d, 4d, 5d, 6d, 7d, 8d, 9d};
43  
44          ArrayRealVector v0 = new ArrayRealVector();
45          Assert.assertEquals("testData len", 0, v0.getDimension());
46  
47          ArrayRealVector v1 = new ArrayRealVector(7);
48          Assert.assertEquals("testData len", 7, v1.getDimension());
49          Assert.assertEquals("testData is 0.0 ", 0.0, v1.getEntry(6), 0);
50  
51          ArrayRealVector v2 = new ArrayRealVector(5, 1.23);
52          Assert.assertEquals("testData len", 5, v2.getDimension());
53          Assert.assertEquals("testData is 1.23 ", 1.23, v2.getEntry(4), 0);
54  
55          ArrayRealVector v3 = new ArrayRealVector(vec1);
56          Assert.assertEquals("testData len", 3, v3.getDimension());
57          Assert.assertEquals("testData is 2.0 ", 2.0, v3.getEntry(1), 0);
58  
59          ArrayRealVector v3_bis = new ArrayRealVector(vec1, true);
60          Assert.assertEquals("testData len", 3, v3_bis.getDimension());
61          Assert.assertEquals("testData is 2.0 ", 2.0, v3_bis.getEntry(1), 0);
62          Assert.assertNotSame(v3_bis.getDataRef(), vec1);
63          Assert.assertNotSame(v3_bis.toArray(), vec1);
64  
65          ArrayRealVector v3_ter = new ArrayRealVector(vec1, false);
66          Assert.assertEquals("testData len", 3, v3_ter.getDimension());
67          Assert.assertEquals("testData is 2.0 ", 2.0, v3_ter.getEntry(1), 0);
68          Assert.assertSame(v3_ter.getDataRef(), vec1);
69          Assert.assertNotSame(v3_ter.toArray(), vec1);
70  
71          ArrayRealVector v4 = new ArrayRealVector(vec4, 3, 2);
72          Assert.assertEquals("testData len", 2, v4.getDimension());
73          Assert.assertEquals("testData is 4.0 ", 4.0, v4.getEntry(0), 0);
74          try {
75              new ArrayRealVector(vec4, 8, 3);
76              Assert.fail("MathIllegalArgumentException expected");
77          } catch (MathIllegalArgumentException ex) {
78              // expected behavior
79          }
80  
81          RealVector v5_i = new ArrayRealVector(dvec1);
82          Assert.assertEquals("testData len", 9, v5_i.getDimension());
83          Assert.assertEquals("testData is 9.0 ", 9.0, v5_i.getEntry(8), 0);
84  
85          ArrayRealVector v5 = new ArrayRealVector(dvec1);
86          Assert.assertEquals("testData len", 9, v5.getDimension());
87          Assert.assertEquals("testData is 9.0 ", 9.0, v5.getEntry(8), 0);
88  
89          ArrayRealVector v6 = new ArrayRealVector(dvec1, 3, 2);
90          Assert.assertEquals("testData len", 2, v6.getDimension());
91          Assert.assertEquals("testData is 4.0 ", 4.0, v6.getEntry(0), 0);
92          try {
93              new ArrayRealVector(dvec1, 8, 3);
94              Assert.fail("MathIllegalArgumentException expected");
95          } catch (MathIllegalArgumentException ex) {
96              // expected behavior
97          }
98  
99          ArrayRealVector v7 = new ArrayRealVector(v1);
100         Assert.assertEquals("testData len", 7, v7.getDimension());
101         Assert.assertEquals("testData is 0.0 ", 0.0, v7.getEntry(6), 0);
102 
103         RealVectorTestImpl v7_i = new RealVectorTestImpl(vec1);
104 
105         ArrayRealVector v7_2 = new ArrayRealVector(v7_i);
106         Assert.assertEquals("testData len", 3, v7_2.getDimension());
107         Assert.assertEquals("testData is 0.0 ", 2.0d, v7_2.getEntry(1), 0);
108 
109         ArrayRealVector v8 = new ArrayRealVector(v1, true);
110         Assert.assertEquals("testData len", 7, v8.getDimension());
111         Assert.assertEquals("testData is 0.0 ", 0.0, v8.getEntry(6), 0);
112         Assert.assertNotSame("testData not same object ", v1.getDataRef(), v8.getDataRef());
113 
114         ArrayRealVector v8_2 = new ArrayRealVector(v1, false);
115         Assert.assertEquals("testData len", 7, v8_2.getDimension());
116         Assert.assertEquals("testData is 0.0 ", 0.0, v8_2.getEntry(6), 0);
117         Assert.assertEquals("testData same object ", v1.getDataRef(), v8_2.getDataRef());
118 
119         ArrayRealVector v9 = new ArrayRealVector(v1, v3);
120         Assert.assertEquals("testData len", 10, v9.getDimension());
121         Assert.assertEquals("testData is 1.0 ", 1.0, v9.getEntry(7), 0);
122 
123         ArrayRealVector v10 = new ArrayRealVector(v2, new RealVectorTestImpl(vec3));
124         Assert.assertEquals("testData len", 8, v10.getDimension());
125         Assert.assertEquals("testData is 1.23 ", 1.23, v10.getEntry(4), 0);
126         Assert.assertEquals("testData is 7.0 ", 7.0, v10.getEntry(5), 0);
127 
128         ArrayRealVector v11 = new ArrayRealVector(new RealVectorTestImpl(vec3), v2);
129         Assert.assertEquals("testData len", 8, v11.getDimension());
130         Assert.assertEquals("testData is 9.0 ", 9.0, v11.getEntry(2), 0);
131         Assert.assertEquals("testData is 1.23 ", 1.23, v11.getEntry(3), 0);
132 
133         ArrayRealVector v12 = new ArrayRealVector(v2, vec3);
134         Assert.assertEquals("testData len", 8, v12.getDimension());
135         Assert.assertEquals("testData is 1.23 ", 1.23, v12.getEntry(4), 0);
136         Assert.assertEquals("testData is 7.0 ", 7.0, v12.getEntry(5), 0);
137 
138         ArrayRealVector v13 = new ArrayRealVector(vec3, v2);
139         Assert.assertEquals("testData len", 8, v13.getDimension());
140         Assert.assertEquals("testData is 9.0 ", 9.0, v13.getEntry(2), 0);
141         Assert.assertEquals("testData is 1.23 ", 1.23, v13.getEntry(3), 0);
142 
143         ArrayRealVector v14 = new ArrayRealVector(vec3, vec4);
144         Assert.assertEquals("testData len", 12, v14.getDimension());
145         Assert.assertEquals("testData is 9.0 ", 9.0, v14.getEntry(2), 0);
146         Assert.assertEquals("testData is 1.0 ", 1.0, v14.getEntry(3), 0);
147 
148     }

150     @Test
151     public void testGetDataRef() {
152         final double[] data = {1d, 2d, 3d, 4d};
153         final ArrayRealVector v = new ArrayRealVector(data);
154         v.getDataRef()[0] = 0d;
155         Assert.assertEquals("", 0d, v.getEntry(0), 0);
156     }

158     @Test
159     public void testPredicates() {
160 
161         Assert.assertEquals(create(new double[] { Double.NaN, 1, 2 }).hashCode(),
162                      create(new double[] { 0, Double.NaN, 2 }).hashCode());
163 
164         Assert.assertTrue(create(new double[] { Double.NaN, 1, 2 }).hashCode() !=
165                    create(new double[] { 0, 1, 2 }).hashCode());
166     }
167 
168     @Test
169     public void testZeroVectors() {
170         Assert.assertEquals(0, new ArrayRealVector(new double[0]).getDimension());
171         Assert.assertEquals(0, new ArrayRealVector(new double[0], true).getDimension());
172         Assert.assertEquals(0, new ArrayRealVector(new double[0], false).getDimension());
173     }
