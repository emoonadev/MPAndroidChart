17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.moment.Skewness;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link UnivariateStatistic} class.
27   *
28   */
29  public class SkewnessTest extends StorelessUnivariateStatisticAbstractTest{
30  
31      protected Skewness stat;
32  
33      /**
34       * {@inheritDoc}
35       */
36      @Override
37      public UnivariateStatistic getUnivariateStatistic() {
38          return new Skewness();
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public double expectedValue() {
46          return this.skew;
47      }
48  
49      /**
50       * Make sure Double.NaN is returned iff n < 3
51       */
52      @Test
53      public void testNaN() {
54          Skewness skew = new Skewness();
55          Assert.assertTrue(Double.isNaN(skew.getResult()));
56          skew.increment(1d);
57          Assert.assertTrue(Double.isNaN(skew.getResult()));
58          skew.increment(1d);
59          Assert.assertTrue(Double.isNaN(skew.getResult()));
60          skew.increment(1d);
61          Assert.assertFalse(Double.isNaN(skew.getResult()));
62      }
63  
64  }
