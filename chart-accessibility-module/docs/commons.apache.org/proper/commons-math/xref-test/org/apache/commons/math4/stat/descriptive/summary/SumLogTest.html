17  package org.apache.commons.math4.stat.descriptive.summary;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatistic;
20  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
21  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
22  import org.apache.commons.math4.stat.descriptive.summary.SumOfLogs;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for the {@link UnivariateStatistic} class.
28   */
29  public class SumLogTest extends StorelessUnivariateStatisticAbstractTest{
30  
31      protected SumOfLogs stat;
32  
33      /**
34       * {@inheritDoc}
35       */
36      @Override
37      public UnivariateStatistic getUnivariateStatistic() {
38          return new SumOfLogs();
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public double expectedValue() {
46          return this.sumLog;
47      }
48  
49      @Test
50      public void testSpecialValues() {
51          SumOfLogs sum = new SumOfLogs();
52          // empty
53          Assert.assertEquals(0, sum.getResult(), 0);
54  
55          // finite data
56          sum.increment(1d);
57          Assert.assertFalse(Double.isNaN(sum.getResult()));
58  
59          // add negative infinity
60          sum.increment(0d);
61          Assert.assertEquals(Double.NEGATIVE_INFINITY, sum.getResult(), 0);
62  
63          // add positive infinity -- should make NaN
64          sum.increment(Double.POSITIVE_INFINITY);
65          Assert.assertTrue(Double.isNaN(sum.getResult()));
66  
67          // clear
68          sum.clear();
69          Assert.assertEquals(0, sum.getResult(), 0);
70  
71          // positive infinity by itself
72          sum.increment(Double.POSITIVE_INFINITY);
73          Assert.assertEquals(Double.POSITIVE_INFINITY, sum.getResult(), 0);
74  
75          // negative value -- should make NaN
76          sum.increment(-2d);
77          Assert.assertTrue(Double.isNaN(sum.getResult()));
78      }
79  
80      @Override
81      protected void checkClearValue(StorelessUnivariateStatistic statistic) {
82          Assert.assertEquals(0, statistic.getResult(), 0);
83      }
84  
85  }
