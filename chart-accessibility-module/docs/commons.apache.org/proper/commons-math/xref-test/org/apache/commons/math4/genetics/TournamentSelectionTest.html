17  package org.apache.commons.math4.genetics;
18  
19  import org.apache.commons.math4.genetics.Chromosome;
20  import org.apache.commons.math4.genetics.ChromosomePair;
21  import org.apache.commons.math4.genetics.ElitisticListPopulation;
22  import org.apache.commons.math4.genetics.TournamentSelection;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class TournamentSelectionTest {
27  
28      private static int counter = 0;
29  
30      @Test
31      public void testSelect() {
32          TournamentSelection ts = new TournamentSelection(2);
33          ElitisticListPopulation pop = new ElitisticListPopulation(100, 0.203);
34  
35          for (int i=0; i<pop.getPopulationLimit(); i++) {
36              pop.addChromosome(new DummyChromosome());
37          }
38          // how to write a test for stochastic method?
39          for (int i=0; i<20; i++) {
40              ChromosomePair pair = ts.select(pop);
41              // the worst chromosome should NEVER be selected
42              Assert.assertTrue(pair.getFirst().getFitness() > 0);
43              Assert.assertTrue(pair.getSecond().getFitness() > 0);
44          }
45      }
46  
47      private static class DummyChromosome extends Chromosome {
48          private final int fitness;
49  
50          public DummyChromosome() {
51              this.fitness = counter;
52              counter++;
53          }
54  
55          public double fitness() {
56              return this.fitness;
57          }
58      }
59  
60  }
