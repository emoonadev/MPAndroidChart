106     public void testHistory(FilterType type, double t0, double t1, double refSwitch, double signEven) {
107         Event onlyIncreasing = new Event(false, true);
108         EventFilter eventFilter =
109                 new EventFilter(onlyIncreasing, type);
110         eventFilter.init(t0, new double[] {1.0,  0.0}, t1);
111 
112         // first pass to set up switches history for a long period
113         double h = FastMath.copySign(0.05, t1 - t0);
114         double n = (int) FastMath.floor((t1 - t0) / h);
115         for (int i = 0; i < n; ++i) {
116             double t = t0 + i * h;
117             eventFilter.g(t, new double[] { FastMath.sin(t), FastMath.cos(t) });
118         }
119 
120         // verify old events are preserved, even if randomly accessed
121         UniformRandomProvider rng = RandomSource.create(RandomSource.TWO_CMRES, 0xb0e7401265af8cd3l);
122         for (int i = 0; i < 5000; i++) {
123             double t = t0 + (t1 - t0) * rng.nextDouble();
124             double g = eventFilter.g(t, new double[] { FastMath.sin(t), FastMath.cos(t) });
125             int turn = (int) FastMath.floor((t - refSwitch) / (2 * FastMath.PI));
126             if (turn % 2 == 0) {
127                 Assert.assertEquals( signEven * FastMath.sin(t), g, 1.0e-10);
128             } else {
129                 Assert.assertEquals(-signEven * FastMath.sin(t), g, 1.0e-10);
130             }
131         }
132

135     @Test
136     public void testIncreasingOnly()
137         throws DimensionMismatchException, NumberIsTooSmallException,
138                MaxCountExceededException, NoBracketingException {
139         double e = 1e-15;
140         FirstOrderIntegrator integrator;
141         integrator = new DormandPrince853Integrator(1.0e-3, 100.0, 1e-7, 1e-7);
142         Event allEvents = new Event(true, true);
143         integrator.addEventHandler(allEvents, 0.1, e, 1000,
144                                    new BracketingNthOrderBrentSolver(1.0e-7, 5));
145         Event onlyIncreasing = new Event(false, true);
146         integrator.addEventHandler(new EventFilter(onlyIncreasing,
147                                                    FilterType.TRIGGER_ONLY_INCREASING_EVENTS),
148                                    0.1, e, 100,
149                                    new BracketingNthOrderBrentSolver(1.0e-7, 5));
150         double t0 = 0.5 * FastMath.PI;
151         double tEnd = 5.5 * FastMath.PI;
152         double[] y = { 0.0, 1.0 };
153         Assert.assertEquals(tEnd,
154                             integrator.integrate(new SineCosine(), t0, y, tEnd, y),
155                             1.0e-7);
156 
157         Assert.assertEquals(5, allEvents.getEventCount());
158         Assert.assertEquals(2, onlyIncreasing.getEventCount());
159 
160     }

222     private static class SineCosine implements FirstOrderDifferentialEquations {
223         public int getDimension() {
224             return 2;
225         }
226 
227         public void computeDerivatives(double t, double[] y, double[] yDot) {
228             yDot[0] =  y[1];
229             yDot[1] = -y[0];
230         }
231     }

233     /** State events for this unit test. */
234     protected static class Event implements EventHandler {
235 
236         private final boolean expectDecreasing;
237         private final boolean expectIncreasing;
238         private int eventCount;
239 
240         public Event(boolean expectDecreasing, boolean expectIncreasing) {
241             this.expectDecreasing = expectDecreasing;
242             this.expectIncreasing = expectIncreasing;
243         }
244 
245         public int getEventCount() {
246             return eventCount;
247         }
248 
249         public void init(double t0, double[] y0, double t) {
250             eventCount = 0;
251         }
252 
253         public double g(double t, double[] y) {
254             return y[0];
255         }
256 
257         public Action eventOccurred(double t, double[] y, boolean increasing) {
258             if (increasing) {
259                 Assert.assertTrue(expectIncreasing);
260             } else {
261                 Assert.assertTrue(expectDecreasing);
262             }
263             eventCount++;
264             return Action.RESET_STATE;
265         }
266 
267         public void resetState(double t, double[] y) {
268             // in fact, we don't really reset anything for this test
269         }
270 
271     }
