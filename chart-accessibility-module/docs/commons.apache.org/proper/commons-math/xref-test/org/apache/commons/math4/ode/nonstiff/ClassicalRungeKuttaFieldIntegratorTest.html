21  import org.apache.commons.math4.Field;
22  import org.apache.commons.math4.RealFieldElement;
23  import org.apache.commons.math4.util.Decimal64Field;
24  
25  public class ClassicalRungeKuttaFieldIntegratorTest extends RungeKuttaFieldIntegratorAbstractTest {
26  
27      protected <T extends RealFieldElement<T>> RungeKuttaFieldIntegrator<T>
28      createIntegrator(Field<T> field, T step) {
29          return new ClassicalRungeKuttaFieldIntegrator<T>(field, step);
30      }
31  
32      @Override
33      public void testNonFieldIntegratorConsistency() {
34          doTestNonFieldIntegratorConsistency(Decimal64Field.getInstance());
35      }
36  
37      @Override
38      public void testMissedEndEvent() {
39          doTestMissedEndEvent(Decimal64Field.getInstance(), 5.0e-6, 1.0e-9);
40      }
41  
42      @Override
43      public void testSanityChecks() {
44          doTestSanityChecks(Decimal64Field.getInstance());
45      }
46  
47      @Override
48      public void testDecreasingSteps() {
49          doTestDecreasingSteps(Decimal64Field.getInstance(), 1.0, 1.0, 1.0e-10);
50      }
51  
52      @Override
53      public void testSmallStep() {
54          doTestSmallStep(Decimal64Field.getInstance(), 2.0e-13, 4.0e-12, 1.0e-12, "classical Runge-Kutta");
55      }
56  
57      @Override
58      public void testBigStep() {
59          doTestBigStep(Decimal64Field.getInstance(), 0.0004, 0.005, 1.0e-12, "classical Runge-Kutta");
60  
61      }
62  
63      @Override
64      public void testBackward() {
65          doTestBackward(Decimal64Field.getInstance(), 5.0e-10, 7.0e-10, 1.0e-12, "classical Runge-Kutta");
66      }
67  
68      @Override
69      public void testKepler() {
70          doTestKepler(Decimal64Field.getInstance(), 5.82e-3, 1.0e-5);
71      }
72  
73      @Override
74      public void testStepSize() {
75          doTestStepSize(Decimal64Field.getInstance(), 1.0e-12);
76      }
77  
78      @Override
79      public void testSingleStep() {
80          doTestSingleStep(Decimal64Field.getInstance(), 9.3e-9);
81      }
82  
83      @Override
84      public void testTooLargeFirstStep() {
85          doTestTooLargeFirstStep(Decimal64Field.getInstance());
86      }
87  
88      @Override
89      public void testUnstableDerivative() {
90          doTestUnstableDerivative(Decimal64Field.getInstance(), 1.0e-12);
91      }
92  
93      @Override
94      public void testDerivativesConsistency() {
95          doTestDerivativesConsistency(Decimal64Field.getInstance(), 1.0e-10);
96      }
97  
98      @Override
99      public void testPartialDerivatives() {
100         doTestPartialDerivatives(3.2e-10, new double[] { 2.1e-9, 5.9e-10, 7.0e-11, 7.0e-11, 5.9e-10 });
101     }
