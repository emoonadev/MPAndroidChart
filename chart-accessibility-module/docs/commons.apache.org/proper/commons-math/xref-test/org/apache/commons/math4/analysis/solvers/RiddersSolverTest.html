31  /**
32   * Test case for {@link RiddersSolver Ridders} solver.
33   * <p>
34   * Ridders' method converges superlinearly, more specific, its rate of
35   * convergence is sqrt(2). Test runs show that for a default absolute
36   * accuracy of 1E-6, it generally takes less than 5 iterations for close
37   * initial bracket and 5 to 10 iterations for distant initial bracket
38   * to converge.
39   *
40   */
41  public final class RiddersSolverTest {
42      /**
43       * Test of solver for the sine function.
44       */
45      @Test
46      public void testSinFunction() {
47          UnivariateFunction f = new Sin();
48          UnivariateSolver solver = new RiddersSolver();
49          double min, max, expected, result, tolerance;
50  
51          min = 3.0; max = 4.0; expected = FastMath.PI;
52          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
53                      FastMath.abs(expected * solver.getRelativeAccuracy()));
54          result = solver.solve(100, f, min, max);
55          Assert.assertEquals(expected, result, tolerance);
56  
57          min = -1.0; max = 1.5; expected = 0.0;
58          tolerance = FastMath.max(solver.getAbsoluteAccuracy(),
59                      FastMath.abs(expected * solver.getRelativeAccuracy()));
60          result = solver.solve(100, f, min, max);
61          Assert.assertEquals(expected, result, tolerance);
62      }

120     /**
121      * Test of parameters for the solver.
122      */
123     @Test
124     public void testParameters() {
125         UnivariateFunction f = new Sin();
126         UnivariateSolver solver = new RiddersSolver();
127 
128         try {
129             // bad interval
130             solver.solve(100, f, 1, -1);
131             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
132         } catch (NumberIsTooLargeException ex) {
133             // expected
134         }
135         try {
136             // no bracketing
137             solver.solve(100, f, 2, 3);
138             Assert.fail("Expecting NoBracketingException - no bracketing");
139         } catch (NoBracketingException ex) {
140             // expected
141         }
142     }
