68      @Test
69      public void testWindowSize() {
70          DescriptiveStatistics stats = createDescriptiveStatistics();
71          stats.setWindowSize(300);
72          for (int i = 0; i < 100; ++i) {
73              stats.addValue(i + 1);
74          }
75          int refSum = (100 * 101) / 2;
76          Assert.assertEquals(refSum / 100.0, stats.getMean(), 1E-10);
77          Assert.assertEquals(300, stats.getWindowSize());
78          try {
79              stats.setWindowSize(-3);
80              Assert.fail("an exception should have been thrown");
81          } catch (MathIllegalArgumentException iae) {
82              // expected
83          }
84          Assert.assertEquals(300, stats.getWindowSize());
85          stats.setWindowSize(50);
86          Assert.assertEquals(50, stats.getWindowSize());
87          int refSum2 = refSum - (50 * 51) / 2;
88          Assert.assertEquals(refSum2 / 50.0, stats.getMean(), 1E-10);
89      }

110     @Test
111     public void testQuadraticMean() {
112         final double[] values = { 1.2, 3.4, 5.6, 7.89 };
113         final DescriptiveStatistics stats = new DescriptiveStatistics(values);
114 
115         final int len = values.length;
116         double expected = 0;
117         for (int i = 0; i < len; i++) {
118             final double v = values[i];
119             expected += v * v / len;
120         }
121         expected = Math.sqrt(expected);
122 
123         Assert.assertEquals(expected, stats.getQuadraticMean(), Math.ulp(expected));
124     }

146     @Test
147     public void testShuffledStatistics() {
148         // the purpose of this test is only to check the get/set methods
149         // we are aware shuffling statistics like this is really not
150         // something sensible to do in production ...
151         DescriptiveStatistics reference = createDescriptiveStatistics();
152         DescriptiveStatistics shuffled  = createDescriptiveStatistics();
153 
154         UnivariateStatistic tmp = shuffled.getGeometricMeanImpl();
155         shuffled.setGeometricMeanImpl(shuffled.getMeanImpl());
156         shuffled.setMeanImpl(shuffled.getKurtosisImpl());
157         shuffled.setKurtosisImpl(shuffled.getSkewnessImpl());
158         shuffled.setSkewnessImpl(shuffled.getVarianceImpl());
159         shuffled.setVarianceImpl(shuffled.getMaxImpl());
160         shuffled.setMaxImpl(shuffled.getMinImpl());
161         shuffled.setMinImpl(shuffled.getSumImpl());
162         shuffled.setSumImpl(shuffled.getSumsqImpl());
163         shuffled.setSumsqImpl(tmp);
164 
165         for (int i = 100; i > 0; --i) {
166             reference.addValue(i);
167             shuffled.addValue(i);
168         }
169 
170         Assert.assertEquals(reference.getMean(),          shuffled.getGeometricMean(), 1.0e-10);
171         Assert.assertEquals(reference.getKurtosis(),      shuffled.getMean(),          1.0e-10);
172         Assert.assertEquals(reference.getSkewness(),      shuffled.getKurtosis(), 1.0e-10);
173         Assert.assertEquals(reference.getVariance(),      shuffled.getSkewness(), 1.0e-10);
174         Assert.assertEquals(reference.getMax(),           shuffled.getVariance(), 1.0e-10);
175         Assert.assertEquals(reference.getMin(),           shuffled.getMax(), 1.0e-10);
176         Assert.assertEquals(reference.getSum(),           shuffled.getMin(), 1.0e-10);
177         Assert.assertEquals(reference.getSumsq(),         shuffled.getSum(), 1.0e-10);
178         Assert.assertEquals(reference.getGeometricMean(), shuffled.getSumsq(), 1.0e-10);
179 
180     }

231     @Test
232     public void testSummaryConsistency() {
233         final DescriptiveStatistics dstats = new DescriptiveStatistics();
234         final SummaryStatistics sstats = new SummaryStatistics();
235         final int windowSize = 5;
236         dstats.setWindowSize(windowSize);
237         final double tol = 1E-12;
238         for (int i = 0; i < 20; i++) {
239             dstats.addValue(i);
240             sstats.clear();
241             double[] values = dstats.getValues();
242             for (int j = 0; j < values.length; j++) {
243                 sstats.addValue(values[j]);
244             }
245             TestUtils.assertEquals(dstats.getMean(), sstats.getMean(), tol);
246             TestUtils.assertEquals(new Mean().evaluate(values), dstats.getMean(), tol);
247             TestUtils.assertEquals(dstats.getMax(), sstats.getMax(), tol);
248             TestUtils.assertEquals(new Max().evaluate(values), dstats.getMax(), tol);
249             TestUtils.assertEquals(dstats.getGeometricMean(), sstats.getGeometricMean(), tol);
250             TestUtils.assertEquals(new GeometricMean().evaluate(values), dstats.getGeometricMean(), tol);
251             TestUtils.assertEquals(dstats.getMin(), sstats.getMin(), tol);
252             TestUtils.assertEquals(new Min().evaluate(values), dstats.getMin(), tol);
253             TestUtils.assertEquals(dstats.getStandardDeviation(), sstats.getStandardDeviation(), tol);
254             TestUtils.assertEquals(dstats.getVariance(), sstats.getVariance(), tol);
255             TestUtils.assertEquals(new Variance().evaluate(values), dstats.getVariance(), tol);
256             TestUtils.assertEquals(dstats.getSum(), sstats.getSum(), tol);
257             TestUtils.assertEquals(new Sum().evaluate(values), dstats.getSum(), tol);
258             TestUtils.assertEquals(dstats.getSumsq(), sstats.getSumsq(), tol);
259             TestUtils.assertEquals(new SumOfSquares().evaluate(values), dstats.getSumsq(), tol);
260             TestUtils.assertEquals(dstats.getPopulationVariance(), sstats.getPopulationVariance(), tol);
261             TestUtils.assertEquals(new Variance(false).evaluate(values), dstats.getPopulationVariance(), tol);
262         }

295     public void checkremoval(DescriptiveStatistics dstat, int wsize,
296                              double mean1, double mean2, double mean3) {
297 
298         dstat.setWindowSize(wsize);
299         dstat.clear();
300 
301         for (int i = 1 ; i <= 6 ; ++i) {
302             dstat.addValue(i);
303         }
304 
305         Assert.assertTrue(Precision.equalsIncludingNaN(mean1, dstat.getMean()));
306         dstat.replaceMostRecentValue(0);
307         Assert.assertTrue(Precision.equalsIncludingNaN(mean2, dstat.getMean()));
308         dstat.removeMostRecentValue();
309         Assert.assertTrue(Precision.equalsIncludingNaN(mean3, dstat.getMean()));
310 
311     }

335     /**
336      * Test percentile implementation - wraps a Percentile
337      */
338     static class goodPercentile implements UnivariateStatistic {
339         private final Percentile percentile = new Percentile();
340         public void setQuantile(double quantile) {
341             percentile.setQuantile(quantile);
342         }
343         @Override
344         public double evaluate(double[] values, int begin, int length) {
345             return percentile.evaluate(values, begin, length);
346         }
347         @Override
348         public double evaluate(double[] values) {
349             return percentile.evaluate(values);
350         }
351         @Override
352         public UnivariateStatistic copy() {
353             goodPercentile result = new goodPercentile();
354             result.setQuantile(percentile.getQuantile());
355             return result;
