17  package org.apache.commons.math4.analysis.solvers;
18  
19  import org.apache.commons.math4.analysis.QuinticFunction;
20  import org.apache.commons.math4.analysis.UnivariateFunction;
21  import org.apache.commons.math4.analysis.function.Sin;
22  import org.apache.commons.math4.analysis.solvers.BisectionSolver;
23  import org.apache.commons.math4.util.FastMath;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  /**
28   */
29  public final class BisectionSolverTest {
30      @Test
31      public void testSinZero() {
32          UnivariateFunction f = new Sin();
33          double result;
34  
35          BisectionSolver solver = new BisectionSolver();
36          result = solver.solve(100, f, 3, 4);
37          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
38  
39          result = solver.solve(100, f, 1, 4);
40          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
41      }
42  
43      @Test
44      public void testQuinticZero() {
45          UnivariateFunction f = new QuinticFunction();
46          double result;
47  
48          BisectionSolver solver = new BisectionSolver();
49          result = solver.solve(100, f, -0.2, 0.2);
50          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
51  
52          result = solver.solve(100, f, -0.1, 0.3);
53          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
54  
55          result = solver.solve(100, f, -0.3, 0.45);
56          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
57  
58          result = solver.solve(100, f, 0.3, 0.7);
59          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
60  
61          result = solver.solve(100, f, 0.2, 0.6);
62          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
63  
64          result = solver.solve(100, f, 0.05, 0.95);
65          Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
66  
67          result = solver.solve(100, f, 0.85, 1.25);
68          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
69  
70          result = solver.solve(100, f, 0.8, 1.2);
71          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
72  
73          result = solver.solve(100, f, 0.85, 1.75);
74          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
75  
76          result = solver.solve(100, f, 0.55, 1.45);
77          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
78  
79          result = solver.solve(100, f, 0.85, 5);
80          Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
81  
82          Assert.assertTrue(solver.getEvaluations() > 0);
83      }
84  
85      @Test
86      public void testMath369() {
87          UnivariateFunction f = new Sin();
88          BisectionSolver solver = new BisectionSolver();
89          Assert.assertEquals(FastMath.PI, solver.solve(100, f, 3.0, 3.2, 3.1), solver.getAbsoluteAccuracy());
90      }
91  }
