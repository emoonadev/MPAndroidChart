35      protected WilcoxonSignedRankTest testStatistic = new WilcoxonSignedRankTest();

37      @Test
38      public void testWilcoxonSignedRankSimple() {
39          /* Target values computed using R version 2.11.1
40           * x <- c(1.83, 0.50, 1.62, 2.48, 1.68, 1.88, 1.55, 3.06, 1.30)
41           * y <- c(0.878, 0.647, 0.598, 2.05, 1.06, 1.29, 1.06, 3.14, 1.29)
42           */
43          final double x[] = {1.83, 0.50, 1.62, 2.48, 1.68, 1.88, 1.55, 3.06, 1.30};
44          final double y[] = {0.878, 0.647, 0.598, 2.05, 1.06, 1.29, 1.06, 3.14, 1.29};
45  
46          /* EXACT:
47           * wilcox.test(x, y, alternative = "two.sided", mu = 0, paired = TRUE, exact = TRUE, correct = FALSE)
48           * V = 40, p-value = 0.03906
49           *
50           * Corresponds to the value obtained in R.
51           */
52          Assert.assertEquals(40, testStatistic.wilcoxonSignedRank(x, y), 1e-10);
53          Assert.assertEquals(0.03906, testStatistic.wilcoxonSignedRankTest(x, y, true), 1e-5);
54  
55          /* ASYMPTOTIC:
56           * wilcox.test(x, y, alternative = "two.sided", mu = 0, paired = TRUE, exact = FALSE, correct = FALSE)
57           * V = 40, p-value = 0.03815
58           *
59           * This is not entirely the same due to different corrects,
60           * e.g. http://mlsc.lboro.ac.uk/resources/statistics/wsrt.pdf
61           * and src/library/stats/R/wilcox.test.R in the R source
62           */
63          Assert.assertEquals(40, testStatistic.wilcoxonSignedRank(x, y), 1e-10);
64          Assert.assertEquals(0.0329693812, testStatistic.wilcoxonSignedRankTest(x, y, false), 1e-10);
65      }

67      @Test
68      public void testWilcoxonSignedRankInputValidation() {
69          /*
70           * Exact only for sample size <= 30
71           */
72          final double[] x1 = new double[30];
73          final double[] x2 = new double[31];
74          final double[] y1 = new double[30];
75          final double[] y2 = new double[31];
76          for (int i = 0; i < 30; ++i) {
77              x1[i] = x2[i] = y1[i] = y2[i] = i;
78          }
79  
80          // Exactly 30 is okay
81          //testStatistic.wilcoxonSignedRankTest(x1, y1, true);
82  
83          try {
84              testStatistic.wilcoxonSignedRankTest(x2, y2, true);
85              Assert.fail("More than 30 samples and exact chosen, NumberIsTooLargeException expected");
86          } catch (NumberIsTooLargeException ex) {
87              // expected
88          }
89  
90          /* Samples must be present, i.e. length > 0
91           */
92          try {
93              testStatistic.wilcoxonSignedRankTest(new double[] { }, new double[] { 1.0 }, true);
94              Assert.fail("x does not contain samples (exact), NoDataException expected");
95          } catch (NoDataException ex) {
96              // expected
97          }
98  
99          try {
100             testStatistic.wilcoxonSignedRankTest(new double[] { }, new double[] { 1.0 }, false);
101             Assert.fail("x does not contain samples (asymptotic), NoDataException expected");
102         } catch (NoDataException ex) {
103             // expected
104         }
105 
106         try {
107             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, new double[] { }, true);
108             Assert.fail("y does not contain samples (exact), NoDataException expected");
109         } catch (NoDataException ex) {
110             // expected
111         }
112 
113         try {
114             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, new double[] { }, false);
115             Assert.fail("y does not contain samples (asymptotic), NoDataException expected");
116         } catch (NoDataException ex) {
117             // expected
118         }
119 
120         /* Samples not same size, i.e. cannot be pairred
121          */
122         try {
123             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0, 2.0 }, new double[] { 3.0 }, true);
124             Assert.fail("x and y not same size (exact), DimensionMismatchException expected");
125         } catch (DimensionMismatchException ex) {
126             // expected
127         }
128 
129         try {
130             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0, 2.0 }, new double[] { 3.0 }, false);
131             Assert.fail("x and y not same size (asymptotic), DimensionMismatchException expected");
132         } catch (DimensionMismatchException ex) {
133             // expected
134         }
135 
136         /*
137          * x and y is null
138          */
139         try {
140             testStatistic.wilcoxonSignedRankTest(null, null, true);
141             Assert.fail("x and y is null (exact), NullArgumentException expected");
142         } catch (NullArgumentException ex) {
143             // expected
144         }
145 
146         try {
147             testStatistic.wilcoxonSignedRankTest(null, null, false);
148             Assert.fail("x and y is null (asymptotic), NullArgumentException expected");
149         } catch (NullArgumentException ex) {
150             // expected
151         }
152 
153         /*
154          * x or y is null
155          */
156         try {
157             testStatistic.wilcoxonSignedRankTest(null, new double[] { 1.0 }, true);
158             Assert.fail("x is null (exact), NullArgumentException expected");
159         } catch (NullArgumentException ex) {
160             // expected
161         }
162 
163         try {
164             testStatistic.wilcoxonSignedRankTest(null, new double[] { 1.0 }, false);
165             Assert.fail("x is null (asymptotic), NullArgumentException expected");
166         } catch (NullArgumentException ex) {
167             // expected
168         }
169 
170         try {
171             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, null, true);
172             Assert.fail("y is null (exact), NullArgumentException expected");
173         } catch (NullArgumentException ex) {
174             // expected
175         }
176 
177         try {
178             testStatistic.wilcoxonSignedRankTest(new double[] { 1.0 }, null, false);
179             Assert.fail("y is null (asymptotic), NullArgumentException expected");
180         } catch (NullArgumentException ex) {
181             // expected
182         }
183     }
