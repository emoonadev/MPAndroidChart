253     @Test
254     public void testSeveralBlocks() {
255         FieldMatrix<Fraction> m =
256             new BlockFieldMatrix<Fraction>(FractionField.getInstance(), 37, 41);
257         for (int i = 0; i < m.getRowDimension(); ++i) {
258             for (int j = 0; j < m.getColumnDimension(); ++j) {
259                 m.setEntry(i, j, new Fraction(i * 11 + j, 11));
260             }
261         }
262 
263         FieldMatrix<Fraction> mT = m.transpose();
264         Assert.assertEquals(m.getRowDimension(), mT.getColumnDimension());
265         Assert.assertEquals(m.getColumnDimension(), mT.getRowDimension());
266         for (int i = 0; i < mT.getRowDimension(); ++i) {
267             for (int j = 0; j < mT.getColumnDimension(); ++j) {
268                 Assert.assertEquals(m.getEntry(j, i), mT.getEntry(i, j));
269             }
270         }
271 
272         FieldMatrix<Fraction> mPm = m.add(m);
273         for (int i = 0; i < mPm.getRowDimension(); ++i) {
274             for (int j = 0; j < mPm.getColumnDimension(); ++j) {
275                 Assert.assertEquals(m.getEntry(i, j).multiply(new Fraction(2)), mPm.getEntry(i, j));
276             }
277         }
278 
279         FieldMatrix<Fraction> mPmMm = mPm.subtract(m);
280         for (int i = 0; i < mPmMm.getRowDimension(); ++i) {
281             for (int j = 0; j < mPmMm.getColumnDimension(); ++j) {
282                 Assert.assertEquals(m.getEntry(i, j), mPmMm.getEntry(i, j));
283             }
284         }
285 
286         FieldMatrix<Fraction> mTm = mT.multiply(m);
287         for (int i = 0; i < mTm.getRowDimension(); ++i) {
288             for (int j = 0; j < mTm.getColumnDimension(); ++j) {
289                 Fraction sum = Fraction.ZERO;
290                 for (int k = 0; k < mT.getColumnDimension(); ++k) {
291                     sum = sum.add(new Fraction(k * 11 + i, 11).multiply(new Fraction(k * 11 + j, 11)));
292                 }
293                 Assert.assertEquals(sum, mTm.getEntry(i, j));
294             }
295         }
296 
297         FieldMatrix<Fraction> mmT = m.multiply(mT);
298         for (int i = 0; i < mmT.getRowDimension(); ++i) {
299             for (int j = 0; j < mmT.getColumnDimension(); ++j) {
300                 Fraction sum = Fraction.ZERO;
301                 for (int k = 0; k < m.getColumnDimension(); ++k) {
302                     sum = sum.add(new Fraction(i * 11 + k, 11).multiply(new Fraction(j * 11 + k, 11)));
303                 }
304                 Assert.assertEquals(sum, mmT.getEntry(i, j));
305             }
306         }
307 
308         FieldMatrix<Fraction> sub1 = m.getSubMatrix(2, 9, 5, 20);
309         for (int i = 0; i < sub1.getRowDimension(); ++i) {
310             for (int j = 0; j < sub1.getColumnDimension(); ++j) {
311                 Assert.assertEquals(new Fraction((i + 2) * 11 + (j + 5), 11), sub1.getEntry(i, j));
312             }
313         }
314 
315         FieldMatrix<Fraction> sub2 = m.getSubMatrix(10, 12, 3, 40);
316         for (int i = 0; i < sub2.getRowDimension(); ++i) {
317             for (int j = 0; j < sub2.getColumnDimension(); ++j) {
318                 Assert.assertEquals(new Fraction((i + 10) * 11 + (j + 3), 11), sub2.getEntry(i, j));
319             }
320         }
321 
322         FieldMatrix<Fraction> sub3 = m.getSubMatrix(30, 34, 0, 5);
323         for (int i = 0; i < sub3.getRowDimension(); ++i) {
324             for (int j = 0; j < sub3.getColumnDimension(); ++j) {
325                 Assert.assertEquals(new Fraction((i + 30) * 11 + (j + 0), 11), sub3.getEntry(i, j));
326             }
327         }
328 
329         FieldMatrix<Fraction> sub4 = m.getSubMatrix(30, 32, 32, 35);
330         for (int i = 0; i < sub4.getRowDimension(); ++i) {
331             for (int j = 0; j < sub4.getColumnDimension(); ++j) {
332                 Assert.assertEquals(new Fraction((i + 30) * 11 + (j + 32), 11), sub4.getEntry(i, j));
333             }
334         }

525         Fraction[][] matrixData = {
526                 {new Fraction(1),new Fraction(2),new Fraction(3)},
527                 {new Fraction(2),new Fraction(5),new Fraction(3)}
528         };
529         FieldMatrix<Fraction> m = new BlockFieldMatrix<Fraction>(matrixData);
530         // One more with three rows, two columns
531         Fraction[][] matrixData2 = {
532                 {new Fraction(1),new Fraction(2)},
533                 {new Fraction(2),new Fraction(5)},
534                 {new Fraction(1), new Fraction(7)}
535         };
536         FieldMatrix<Fraction> n = new BlockFieldMatrix<Fraction>(matrixData2);
537         // Now multiply m by n
538         FieldMatrix<Fraction> p = m.multiply(n);
539         Assert.assertEquals(2, p.getRowDimension());
540         Assert.assertEquals(2, p.getColumnDimension());
541         // Invert p
542         FieldMatrix<Fraction> pInverse = new FieldLUDecomposition<Fraction>(p).getSolver().getInverse();
543         Assert.assertEquals(2, pInverse.getRowDimension());
544         Assert.assertEquals(2, pInverse.getColumnDimension());
545 
546         // Solve example
547         Fraction[][] coefficientsData = {
548                 {new Fraction(2), new Fraction(3), new Fraction(-2)},
549                 {new Fraction(-1), new Fraction(7), new Fraction(6)},
550                 {new Fraction(4), new Fraction(-3), new Fraction(-5)}
551         };
552         FieldMatrix<Fraction> coefficients = new BlockFieldMatrix<Fraction>(coefficientsData);
553         Fraction[] constants = {
554             new Fraction(1), new Fraction(-2), new Fraction(1)
555         };
556         Fraction[] solution;
557         solution = new FieldLUDecomposition<Fraction>(coefficients)
558             .getSolver()
559             .solve(new ArrayFieldVector<Fraction>(constants, false)).toArray();
560         Assert.assertEquals(new Fraction(2).multiply(solution[0]).
561                      add(new Fraction(3).multiply(solution[1])).
562                      subtract(new Fraction(2).multiply(solution[2])),
563                      constants[0]);
564         Assert.assertEquals(new Fraction(-1).multiply(solution[0]).
565                      add(new Fraction(7).multiply(solution[1])).
566                      add(new Fraction(6).multiply(solution[2])),
567                      constants[1]);
568         Assert.assertEquals(new Fraction(4).multiply(solution[0]).
569                      subtract(new Fraction(3).multiply(solution[1])).
570                      subtract(new Fraction(5).multiply(solution[2])),
571                      constants[2]);
572
