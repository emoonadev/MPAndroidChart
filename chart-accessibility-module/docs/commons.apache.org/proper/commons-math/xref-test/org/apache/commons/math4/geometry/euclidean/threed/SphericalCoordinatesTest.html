31      @Test
32      public void testCoordinatesStoC() throws DimensionMismatchException {
33          double piO2 = 0.5 * FastMath.PI;
34          SphericalCoordinates sc1 = new SphericalCoordinates(2.0, 0, piO2);
35          Assert.assertEquals(0, sc1.getCartesian().distance(new Vector3D(2, 0, 0)), 1.0e-10);
36          SphericalCoordinates sc2 = new SphericalCoordinates(2.0, piO2, piO2);
37          Assert.assertEquals(0, sc2.getCartesian().distance(new Vector3D(0, 2, 0)), 1.0e-10);
38          SphericalCoordinates sc3 = new SphericalCoordinates(2.0, FastMath.PI, piO2);
39          Assert.assertEquals(0, sc3.getCartesian().distance(new Vector3D(-2, 0, 0)), 1.0e-10);
40          SphericalCoordinates sc4 = new SphericalCoordinates(2.0, -piO2, piO2);
41          Assert.assertEquals(0, sc4.getCartesian().distance(new Vector3D(0, -2, 0)), 1.0e-10);
42          SphericalCoordinates sc5 = new SphericalCoordinates(2.0, 1.23456, 0);
43          Assert.assertEquals(0, sc5.getCartesian().distance(new Vector3D(0, 0, 2)), 1.0e-10);
44          SphericalCoordinates sc6 = new SphericalCoordinates(2.0, 6.54321, FastMath.PI);
45          Assert.assertEquals(0, sc6.getCartesian().distance(new Vector3D(0, 0, -2)), 1.0e-10);
46      }

48      @Test
49      public void testCoordinatesCtoS() throws DimensionMismatchException {
50          double piO2 = 0.5 * FastMath.PI;
51          SphericalCoordinates sc1 = new SphericalCoordinates(new Vector3D(2, 0, 0));
52          Assert.assertEquals(2,           sc1.getR(),     1.0e-10);
53          Assert.assertEquals(0,           sc1.getTheta(), 1.0e-10);
54          Assert.assertEquals(piO2,        sc1.getPhi(),   1.0e-10);
55          SphericalCoordinates sc2 = new SphericalCoordinates(new Vector3D(0, 2, 0));
56          Assert.assertEquals(2,           sc2.getR(),     1.0e-10);
57          Assert.assertEquals(piO2,        sc2.getTheta(), 1.0e-10);
58          Assert.assertEquals(piO2,        sc2.getPhi(),   1.0e-10);
59          SphericalCoordinates sc3 = new SphericalCoordinates(new Vector3D(-2, 0, 0));
60          Assert.assertEquals(2,           sc3.getR(),     1.0e-10);
61          Assert.assertEquals(FastMath.PI, sc3.getTheta(), 1.0e-10);
62          Assert.assertEquals(piO2,        sc3.getPhi(),   1.0e-10);
63          SphericalCoordinates sc4 = new SphericalCoordinates(new Vector3D(0, -2, 0));
64          Assert.assertEquals(2,           sc4.getR(),     1.0e-10);
65          Assert.assertEquals(-piO2,       sc4.getTheta(), 1.0e-10);
66          Assert.assertEquals(piO2,        sc4.getPhi(),   1.0e-10);
67          SphericalCoordinates sc5 = new SphericalCoordinates(new Vector3D(0, 0, 2));
68          Assert.assertEquals(2,           sc5.getR(),     1.0e-10);
69          //  don't check theta on poles, as it is singular
70          Assert.assertEquals(0,           sc5.getPhi(),   1.0e-10);
71          SphericalCoordinates sc6 = new SphericalCoordinates(new Vector3D(0, 0, -2));
72          Assert.assertEquals(2,           sc6.getR(),     1.0e-10);
73          //  don't check theta on poles, as it is singular
74          Assert.assertEquals(FastMath.PI, sc6.getPhi(),   1.0e-10);
75      }

77      @Test
78      public void testGradient() {
79          for (double r = 0.2; r < 10; r += 0.5) {
80              for (double theta = 0; theta < 2 * FastMath.PI; theta += 0.1) {
81                  for (double phi = 0.1; phi < FastMath.PI; phi += 0.1) {
82                      SphericalCoordinates sc = new SphericalCoordinates(r, theta, phi);
83  
84                      DerivativeStructure svalue = valueSpherical(new DerivativeStructure(3, 1, 0, r),
85                                                                  new DerivativeStructure(3, 1, 1, theta),
86                                                                  new DerivativeStructure(3, 1, 2, phi));
87                      double[] sGradient = new double[] {
88                          svalue.getPartialDerivative(1, 0, 0),
89                          svalue.getPartialDerivative(0, 1, 0),
90                          svalue.getPartialDerivative(0, 0, 1),
91                      };
92  
93                      DerivativeStructure cvalue = valueCartesian(new DerivativeStructure(3, 1, 0, sc.getCartesian().getX()),
94                                                                  new DerivativeStructure(3, 1, 1, sc.getCartesian().getY()),
95                                                                  new DerivativeStructure(3, 1, 2, sc.getCartesian().getZ()));
96                      Vector3D refCGradient = new Vector3D(cvalue.getPartialDerivative(1, 0, 0),
97                                                           cvalue.getPartialDerivative(0, 1, 0),
98                                                           cvalue.getPartialDerivative(0, 0, 1));
99  
100                     Vector3D testCGradient = new Vector3D(sc.toCartesianGradient(sGradient));
101 
102                     Assert.assertEquals(0, testCGradient.distance(refCGradient) / refCGradient.getNorm(), 5.0e-14);
103 
104                 }
105             }
106         }
107     }

109     @Test
110     public void testHessian() {
111         for (double r = 0.2; r < 10; r += 0.5) {
112             for (double theta = 0; theta < 2 * FastMath.PI; theta += 0.2) {
113                 for (double phi = 0.1; phi < FastMath.PI; phi += 0.2) {
114                     SphericalCoordinates sc = new SphericalCoordinates(r, theta, phi);
115 
116                     DerivativeStructure svalue = valueSpherical(new DerivativeStructure(3, 2, 0, r),
117                                                                 new DerivativeStructure(3, 2, 1, theta),
118                                                                 new DerivativeStructure(3, 2, 2, phi));
119                     double[] sGradient = new double[] {
120                         svalue.getPartialDerivative(1, 0, 0),
121                         svalue.getPartialDerivative(0, 1, 0),
122                         svalue.getPartialDerivative(0, 0, 1),
123                     };
124                     double[][] sHessian = new double[3][3];
125                     sHessian[0][0] = svalue.getPartialDerivative(2, 0, 0); // d2F/dR2
126                     sHessian[1][0] = svalue.getPartialDerivative(1, 1, 0); // d2F/dRdTheta
127                     sHessian[2][0] = svalue.getPartialDerivative(1, 0, 1); // d2F/dRdPhi
128                     sHessian[0][1] = Double.NaN; // just to check upper-right part is not used
129                     sHessian[1][1] = svalue.getPartialDerivative(0, 2, 0); // d2F/dTheta2
130                     sHessian[2][1] = svalue.getPartialDerivative(0, 1, 1); // d2F/dThetadPhi
131                     sHessian[0][2] = Double.NaN; // just to check upper-right part is not used
132                     sHessian[1][2] = Double.NaN; // just to check upper-right part is not used
133                     sHessian[2][2] = svalue.getPartialDerivative(0, 0, 2); // d2F/dPhi2
134 
135                     DerivativeStructure cvalue = valueCartesian(new DerivativeStructure(3, 2, 0, sc.getCartesian().getX()),
136                                                                 new DerivativeStructure(3, 2, 1, sc.getCartesian().getY()),
137                                                                 new DerivativeStructure(3, 2, 2, sc.getCartesian().getZ()));
138                     double[][] refCHessian = new double[3][3];
139                     refCHessian[0][0] = cvalue.getPartialDerivative(2, 0, 0); // d2F/dX2
140                     refCHessian[1][0] = cvalue.getPartialDerivative(1, 1, 0); // d2F/dXdY
141                     refCHessian[2][0] = cvalue.getPartialDerivative(1, 0, 1); // d2F/dXdZ
142                     refCHessian[0][1] = refCHessian[1][0];
143                     refCHessian[1][1] = cvalue.getPartialDerivative(0, 2, 0); // d2F/dY2
144                     refCHessian[2][1] = cvalue.getPartialDerivative(0, 1, 1); // d2F/dYdZ
145                     refCHessian[0][2] = refCHessian[2][0];
146                     refCHessian[1][2] = refCHessian[2][1];
147                     refCHessian[2][2] = cvalue.getPartialDerivative(0, 0, 2); // d2F/dZ2
148                     double norm =  0;
149                     for (int i = 0; i < 3; ++i) {
150                         for (int j = 0; j < 3; ++j) {
151                             norm = FastMath.max(norm, FastMath.abs(refCHessian[i][j]));
152                         }
153                     }
154 
155                     double[][] testCHessian = sc.toCartesianHessian(sHessian, sGradient);
156                     for (int i = 0; i < 3; ++i) {
157                         for (int j = 0; j < 3; ++j) {
158                             Assert.assertEquals("" + FastMath.abs((refCHessian[i][j] - testCHessian[i][j]) / norm),
159                                                 refCHessian[i][j], testCHessian[i][j], 1.0e-14 * norm);
160                         }
161                     }
162 
163                 }
164             }
165         }
166     }

168     public DerivativeStructure valueCartesian(DerivativeStructure x, DerivativeStructure y, DerivativeStructure z) {
169         return x.divide(y.multiply(5).add(10)).multiply(z.pow(3));
170     }
171 
172     public DerivativeStructure valueSpherical(DerivativeStructure r, DerivativeStructure theta, DerivativeStructure phi) {
173         return valueCartesian(r.multiply(theta.cos()).multiply(phi.sin()),
174                               r.multiply(theta.sin()).multiply(phi.sin()),
175                               r.multiply(phi.cos()));
176     }

178     @Test
179     public void testSerialization() {
180         SphericalCoordinates a = new SphericalCoordinates(3, 2, 1);
181         SphericalCoordinates b = (SphericalCoordinates) TestUtils.serializeAndRecover(a);
182         Assert.assertEquals(0, a.getCartesian().distance(b.getCartesian()), 1.0e-10);
183         Assert.assertEquals(a.getR(),     b.getR(),     1.0e-10);
184         Assert.assertEquals(a.getTheta(), b.getTheta(), 1.0e-10);
185         Assert.assertEquals(a.getPhi(),   b.getPhi(),   1.0e-10);
186     }
