17  package org.apache.commons.math4.optim;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.optim.PointValuePair;
21  import org.apache.commons.math4.optim.SimpleValueChecker;
22  import org.junit.Test;
23  import org.junit.Assert;
24  
25  public class SimpleValueCheckerTest {
26      @Test(expected=NotStrictlyPositiveException.class)
27      public void testIterationCheckPrecondition() {
28          new SimpleValueChecker(1e-1, 1e-2, 0);
29      }
30  
31      @Test
32      public void testIterationCheck() {
33          final int max = 10;
34          final SimpleValueChecker checker = new SimpleValueChecker(1e-1, 1e-2, max);
35          Assert.assertTrue(checker.converged(max, null, null));
36          Assert.assertTrue(checker.converged(max + 1, null, null));
37      }
38  
39      @Test
40      public void testIterationCheckDisabled() {
41          final SimpleValueChecker checker = new SimpleValueChecker(1e-8, 1e-8);
42  
43          final PointValuePair a = new PointValuePair(new double[] { 1d }, 1d);
44          final PointValuePair b = new PointValuePair(new double[] { 10d }, 10d);
45  
46          Assert.assertFalse(checker.converged(-1, a, b));
47          Assert.assertFalse(checker.converged(0, a, b));
48          Assert.assertFalse(checker.converged(1000000, a, b));
49  
50          Assert.assertTrue(checker.converged(-1, a, a));
51          Assert.assertTrue(checker.converged(-1, b, b));
52      }
53  
54  }
