47       * {@link #testQuinticZero} unit test. A value of {@code -1} indicates that

49       * @return the expected number of evaluations for the
50       * {@link #testQuinticZero} unit test
51       */
52      protected abstract int[] getQuinticEvalCounts();

54      @Test
55      public void testSinZero() {
56          // The sinus function is behaved well around the root at pi. The second
57          // order derivative is zero, which means linear approximating methods
58          // still converge quadratically.
59          UnivariateFunction f = new Sin();
60          double result;
61          UnivariateSolver solver = getSolver();
62  
63          result = solver.solve(100, f, 3, 4);
64          //System.out.println(
65          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
66          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
67          Assert.assertTrue(solver.getEvaluations() <= 6);
68          result = solver.solve(100, f, 1, 4);
69          //System.out.println(
70          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
71          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
72          Assert.assertTrue(solver.getEvaluations() <= 7);
73      }

75      @Test
76      public void testQuinticZero() {
77          // The quintic function has zeros at 0, +-0.5 and +-1.
78          // Around the root of 0 the function is well behaved, with a second
79          // derivative of zero a 0.
80          // The other roots are less well to find, in particular the root at 1,
81          // because the function grows fast for x>1.
82          // The function has extrema (first derivative is zero) at 0.27195613
83          // and 0.82221643, intervals containing these values are harder for
84          // the solvers.
85          UnivariateFunction f = new QuinticFunction();
86          double result;
87          UnivariateSolver solver = getSolver();
88          double atol = solver.getAbsoluteAccuracy();
89          int[] counts = getQuinticEvalCounts();
90  
91          // Tests data: initial bounds, and expected solution, per test case.
92          double[][] testsData = {{-0.2,  0.2,  0.0},
93                                  {-0.1,  0.3,  0.0},
94                                  {-0.3,  0.45, 0.0},
95                                  { 0.3,  0.7,  0.5},
96                                  { 0.2,  0.6,  0.5},
97                                  { 0.05, 0.95, 0.5},
98                                  { 0.85, 1.25, 1.0},
99                                  { 0.8,  1.2,  1.0},
100                                 { 0.85, 1.75, 1.0},
101                                 { 0.55, 1.45, 1.0},
102                                 { 0.85, 5.0,  1.0},
103                                };
104         int maxIter = 500;
105 
106         for(int i = 0; i < testsData.length; i++) {
107             // Skip test, if needed.
108             if (counts[i] == -1) continue;
109 
110             // Compute solution.
111             double[] testData = testsData[i];
112             result = solver.solve(maxIter, f, testData[0], testData[1]);
113             //System.out.println(
114             //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
115 
116             // Check solution.
117             Assert.assertEquals(result, testData[2], atol);
118             Assert.assertTrue(solver.getEvaluations() <= counts[i] + 1);
119         }
120     }

122     @Test
123     public void testRootEndpoints() {
124         UnivariateFunction f = new XMinus5Function();
125         UnivariateSolver solver = getSolver();
126 
127         // End-point is root. This should be a special case in the solver, and
128         // the initial end-point should be returned exactly.
129         double result = solver.solve(100, f, 5.0, 6.0);
130         Assert.assertEquals(5.0, result, 0.0);
131 
132         result = solver.solve(100, f, 4.0, 5.0);
133         Assert.assertEquals(5.0, result, 0.0);
134 
135         result = solver.solve(100, f, 5.0, 6.0, 5.5);
136         Assert.assertEquals(5.0, result, 0.0);
137 
138         result = solver.solve(100, f, 4.0, 5.0, 4.5);
139         Assert.assertEquals(5.0, result, 0.0);
140     }

144         UnivariateFunction f = new Sin();

162             // expected
163         }

166     @Test
167     public void testSolutionLeftSide() {
168         UnivariateFunction f = new Sin();
169         UnivariateSolver solver = getSolver();
170         double left = -1.5;
171         double right = 0.05;
172         for(int i = 0; i < 10; i++) {
173             // Test whether the allowed solutions are taken into account.
174             double solution = getSolution(solver, 100, f, left, right, AllowedSolution.LEFT_SIDE);
175             if (!Double.isNaN(solution)) {
176                 Assert.assertTrue(solution <= 0.0);
177             }
178 
179             // Prepare for next test.
180             left -= 0.1;
181             right += 0.3;
182         }
183     }

241     private double getSolution(UnivariateSolver solver, int maxEval, UnivariateFunction f,
242                                double left, double right, AllowedSolution allowedSolution) {
243         try {
244             @SuppressWarnings("unchecked")
245             BracketedUnivariateSolver<UnivariateFunction> bracketing =
246             (BracketedUnivariateSolver<UnivariateFunction>) solver;
247             return bracketing.solve(100, f, left, right, allowedSolution);
248         } catch (ClassCastException cce) {
249             double baseRoot = solver.solve(maxEval, f, left, right);
250             if ((baseRoot <= left) || (baseRoot >= right)) {
251                 // the solution slipped out of interval
252                 return Double.NaN;
253             }
254             PegasusSolver bracketing =
255                     new PegasusSolver(solver.getRelativeAccuracy(), solver.getAbsoluteAccuracy(),
256                                       solver.getFunctionValueAccuracy());
257             return UnivariateSolverUtils.forceSide(maxEval - solver.getEvaluations(),
258                                                        f, bracketing, baseRoot, left, right,
259                                                        allowedSolution);
260         }
261     }
