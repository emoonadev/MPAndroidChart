30  /**
31   * Test case for Neville interpolator.
32   * <p>
33   * The error of polynomial interpolation is
34   *     f(z) - p(z) = f^(n)(zeta) * (z-x[0])(z-x[1])...(z-x[n-1]) / n!
35   * where f^(n) is the n-th derivative of the approximated function and
36   * zeta is some point in the interval determined by x[] and z.
37   * <p>
38   * Since zeta is unknown, f^(n)(zeta) cannot be calculated. But we can bound
39   * it and use the absolute value upper bound for estimates. For reference,
40   * see <b>Introduction to Numerical Analysis</b>, ISBN 038795452X, chapter 2.
41   *
42   */
43  public final class NevilleInterpolatorTest {
44  
45      /**
46       * Test of interpolator for the sine function.
47       * <p>
48       * |sin^(n)(zeta)| <= 1.0, zeta in [0, 2*PI]
49       */
50      @Test
51      public void testSinFunction() {
52          UnivariateFunction f = new Sin();
53          UnivariateInterpolator interpolator = new NevilleInterpolator();
54          double x[], y[], z, expected, result, tolerance;
55  
56          // 6 interpolating points on interval [0, 2*PI]
57          int n = 6;
58          double min = 0.0, max = 2 * FastMath.PI;
59          x = new double[n];
60          y = new double[n];
61          for (int i = 0; i < n; i++) {
62              x[i] = min + i * (max - min) / n;
63              y[i] = f.value(x[i]);
64          }
65          double derivativebound = 1.0;
66          UnivariateFunction p = interpolator.interpolate(x, y);
67  
68          z = FastMath.PI / 4; expected = f.value(z); result = p.value(z);
69          tolerance = FastMath.abs(derivativebound * partialerror(x, z));
70          Assert.assertEquals(expected, result, tolerance);
71  
72          z = FastMath.PI * 1.5; expected = f.value(z); result = p.value(z);
73          tolerance = FastMath.abs(derivativebound * partialerror(x, z));
74          Assert.assertEquals(expected, result, tolerance);
75      }

77      /**
78       * Test of interpolator for the exponential function.
79       * <p>
80       * |expm1^(n)(zeta)| <= e, zeta in [-1, 1]
81       */
82      @Test
83      public void testExpm1Function() {
84          UnivariateFunction f = new Expm1();
85          UnivariateInterpolator interpolator = new NevilleInterpolator();
86          double x[], y[], z, expected, result, tolerance;
87  
88          // 5 interpolating points on interval [-1, 1]
89          int n = 5;
90          double min = -1.0, max = 1.0;
91          x = new double[n];
92          y = new double[n];
93          for (int i = 0; i < n; i++) {
94              x[i] = min + i * (max - min) / n;
95              y[i] = f.value(x[i]);
96          }
97          double derivativebound = FastMath.E;
98          UnivariateFunction p = interpolator.interpolate(x, y);
99  
100         z = 0.0; expected = f.value(z); result = p.value(z);
101         tolerance = FastMath.abs(derivativebound * partialerror(x, z));
102         Assert.assertEquals(expected, result, tolerance);
103 
104         z = 0.5; expected = f.value(z); result = p.value(z);
105         tolerance = FastMath.abs(derivativebound * partialerror(x, z));
106         Assert.assertEquals(expected, result, tolerance);
107 
108         z = -0.5; expected = f.value(z); result = p.value(z);
109         tolerance = FastMath.abs(derivativebound * partialerror(x, z));
110         Assert.assertEquals(expected, result, tolerance);
111     }

114      * Test of parameters for the interpolator.
115      */
116     @Test
117     public void testParameters() {
118         UnivariateInterpolator interpolator = new NevilleInterpolator();
119 
120         try {
121             // bad abscissas array
122             double x[] = { 1.0, 2.0, 2.0, 4.0 };
123             double y[] = { 0.0, 4.0, 4.0, 2.5 };
124             UnivariateFunction p = interpolator.interpolate(x, y);
125             p.value(0.0);
126             Assert.fail("Expecting NonMonotonicSequenceException - bad abscissas array");
127         } catch (NonMonotonicSequenceException ex) {
128             // expected
129         }
130     }

132     /**
133      * Returns the partial error term (z-x[0])(z-x[1])...(z-x[n-1])/n!
134      */
135     protected double partialerror(double x[], double z) throws
136         IllegalArgumentException {
137 
138         if (x.length < 1) {
139             throw new IllegalArgumentException
140                 ("Interpolation array cannot be empty.");
141         }
142         double out = 1;
143         for (int i = 0; i < x.length; i++) {
144             out *= (z - x[i]) / (i + 1);
145         }
146         return out;
147     }
