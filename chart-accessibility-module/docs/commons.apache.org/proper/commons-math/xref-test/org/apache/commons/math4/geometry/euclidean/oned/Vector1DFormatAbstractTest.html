40          vector1DFormat = Vector1DFormat.getInstance(getLocale());

139     @Test
140     public void testParseIgnoredWhitespace() {
141         Vector1D expected = new Vector1D(1);
142         ParsePosition pos1 = new ParsePosition(0);
143         String source1 = "{1}";
144         Assert.assertEquals(expected, vector1DFormat.parse(source1, pos1));
145         Assert.assertEquals(source1.length(), pos1.getIndex());
146         ParsePosition pos2 = new ParsePosition(0);
147         String source2 = " { 1 } ";
148         Assert.assertEquals(expected, vector1DFormat.parse(source2, pos2));
149         Assert.assertEquals(source2.length() - 1, pos2.getIndex());
150     }
