49          }

52              tcf = new TricubicInterpolator().interpolate(xval, wyval, zval, fval);

87      public void testIsValid() {
88          double[] xval = new double[] {3, 4, 5, 6.5};
89          double[] yval = new double[] {-4, -3, -1, 2.5};
90          double[] zval = new double[] {-12, -8, -5.5, -3, 0, 2.5};
91          double[][][] fval = new double[xval.length][yval.length][zval.length];
92  
93          TricubicInterpolatingFunction tcf = new TricubicInterpolator().interpolate(xval, yval, zval, fval);
94  
95          // Valid.
96          Assert.assertTrue(tcf.isValidPoint(4, -3, -8));
97          Assert.assertTrue(tcf.isValidPoint(5, -3, -8));
98          Assert.assertTrue(tcf.isValidPoint(4, -1, -8));
99          Assert.assertTrue(tcf.isValidPoint(5, -1, -8));
100         Assert.assertTrue(tcf.isValidPoint(4, -3, 0));
101         Assert.assertTrue(tcf.isValidPoint(5, -3, 0));
102         Assert.assertTrue(tcf.isValidPoint(4, -1, 0));
103         Assert.assertTrue(tcf.isValidPoint(5, -1, 0));
104 
105         // Invalid.
106         Assert.assertFalse(tcf.isValidPoint(3.5, -3, -8));
107         Assert.assertFalse(tcf.isValidPoint(4.5, -3.1, -8));
108         Assert.assertFalse(tcf.isValidPoint(4.5, -2, 0));
109         Assert.assertFalse(tcf.isValidPoint(4.5, 0, -3.5));
110         Assert.assertFalse(tcf.isValidPoint(-10, 4.1, -1));
111     }

113     /**
114      * Test for a plane.
115      * <p>
116      *  f(x, y, z) = 2 x - 3 y - 4 z + 5
117      * </p>
118      */
119     @Test
120     public void testPlane() {
121         double[] xval = new double[] {3, 4, 5, 6.5};
122         double[] yval = new double[] {-4, -3, -1, 2, 2.5};
123         double[] zval = new double[] {-12, -8, -5.5, -3, 0, 2.5};
124 
125         // Function values
126         TrivariateFunction f = new TrivariateFunction() {
127                 @Override
128                 public double value(double x, double y, double z) {
129                     return 2 * x - 3 * y - 4 * z + 5;
130                 }
131             };
132 
133         double[][][] fval = new double[xval.length][yval.length][zval.length];
134 
135         for (int i = 0; i < xval.length; i++) {
136             for (int j = 0; j < yval.length; j++) {
137                 for (int k = 0; k < zval.length; k++) {
138                     fval[i][j][k] = f.value(xval[i], yval[j], zval[k]);
139                 }
140             }
141         }
142 
143         TrivariateFunction tcf = new TricubicInterpolator().interpolate(xval,
144                                                                         yval,
145                                                                         zval,
146                                                                         fval);
147         double x, y, z;
148         double expected, result;
149 
150         x = 4;
151         y = -3;
152         z = 0;
153         expected = f.value(x, y, z);
154         result = tcf.value(x, y, z);
155         Assert.assertEquals("On sample point",
156                             expected, result, 1e-15);
157 
158         x = 4.5;
159         y = -1.5;
160         z = -4.25;
161         expected = f.value(x, y, z);
162         result = tcf.value(x, y, z);
163         Assert.assertEquals("Half-way between sample points (middle of the patch)",
164                             expected, result, 1e-14);
165     }

167     /**
168      * Sine wave.
169      * <p>
170      *  f(x, y, z) = a cos [&omega; z - k<sub>y</sub> x - k<sub>y</sub> y]
171      * </p>
172      * with A = 0.2, &omega; = 0.5, k<sub>x</sub> = 2, k<sub>y</sub> = 1.
173      */
174     @Test
175     public void testWave() {
176         double[] xval = new double[] {3, 4, 5, 6.5};
177         double[] yval = new double[] {-4, -3, -1, 2, 2.5};
178         double[] zval = new double[] {-12, -8, -5.5, -3, 0, 4};
179 
180         final double a = 0.2;
181         final double omega = 0.5;
182         final double kx = 2;
183         final double ky = 1;
184 
185         // Function values
186         TrivariateFunction f = new TrivariateFunction() {
187                 @Override
188                 public double value(double x, double y, double z) {
189                     return a * FastMath.cos(omega * z - kx * x - ky * y);
190                 }
191             };
192 
193         double[][][] fval = new double[xval.length][yval.length][zval.length];
194         for (int i = 0; i < xval.length; i++) {
195             for (int j = 0; j < yval.length; j++) {
196                 for (int k = 0; k < zval.length; k++) {
197                     fval[i][j][k] = f.value(xval[i], yval[j], zval[k]);
198                 }
199             }
200         }
201 
202         TrivariateFunction tcf = new TricubicInterpolator().interpolate(xval,
203                                                                         yval,
204                                                                         zval,
205                                                                         fval);
206 
207         double x, y, z;
208         double expected, result;
209 
210         x = 4;
211         y = -3;
212         z = 0;
213         expected = f.value(x, y, z);
214         result = tcf.value(x, y, z);
215         Assert.assertEquals("On sample point",
216                             expected, result, 1e-14);
217 
218         x = 4.5;
219         y = -1.5;
220         z = -4.25;
221         expected = f.value(x, y, z);
222         result = tcf.value(x, y, z);
223         Assert.assertEquals("Half-way between sample points (middle of the patch)",
224                             expected, result, 1e-1); // XXX Too high tolerance!
225     }
