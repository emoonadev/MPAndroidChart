18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.linear.ArrayRealVector;
21  import org.apache.commons.math4.linear.OpenMapRealVector;
22  import org.apache.commons.math4.linear.RealVector;
23  
24  /**
25   * This is an implementation of {@link UnmodifiableRealVectorAbstractTest} for
26   * unmodifiable views of {@link OpenMapRealVector}.
27   *
28   */
29  public class UnmodifiableOpenMapRealVectorTest
30      extends UnmodifiableRealVectorAbstractTest {
31      /** To ensure sufficient sparsity. */
32      public static final double PROBABILITY_OF_ZERO = 0.5;
33  
34      /**
35       * Returns a random vector of type {@link ArrayRealVector}.
36       *
37       * @return a new random {@link ArrayRealVector}.
38       */
39      @Override
40      public RealVector createVector() {
41          OpenMapRealVector v = new OpenMapRealVector(DIM, EPS);
42          for (int i = 0; i < DIM; i++) {
43              if (RANDOM.nextDouble() > PROBABILITY_OF_ZERO) {
44                  v.setEntry(i, RANDOM.nextDouble());
45              }
46          }
47          return v;
48      }
49  }
