318     @Override
319     public double[] makeCumulativeTestValues() {
320         /*
321          * Bins should be [0, 10], (10, 20], ..., (9990, 10000]
322          * Kernels should be N(4.5, 3.02765), N(14.5, 3.02765)...
323          * Each bin should have mass 10/10000 = .001
324          */
325         final double[] testPoints = getCumulativeTestPoints();
326         final double[] cumValues = new double[testPoints.length];
327         final EmpiricalDistribution empiricalDistribution = (EmpiricalDistribution) makeDistribution();
328         final double[] binBounds = empiricalDistribution.getUpperBounds();
329         for (int i = 0; i < testPoints.length; i++) {
330             final int bin = findBin(testPoints[i]);
331             final double lower = bin == 0 ? empiricalDistribution.getSupportLowerBound() :
332                 binBounds[bin - 1];
333             final double upper = binBounds[bin];
334             // Compute bMinus = sum or mass of bins below the bin containing the point
335             // First bin has mass 11 / 10000, the rest have mass 10 / 10000.
336             final double bMinus = bin == 0 ? 0 : (bin - 1) * binMass + firstBinMass;
337             final RealDistribution kernel = findKernel(lower, upper);
338             final double withinBinKernelMass = kernel.probability(lower, upper);
339             final double kernelCum = kernel.probability(lower, testPoints[i]);
340             cumValues[i] = bMinus + (bin == 0 ? firstBinMass : binMass) * kernelCum/withinBinKernelMass;
341         }
342         return cumValues;
343     }

345     @Override
346     public double[] makeDensityTestValues() {
347         final double[] testPoints = getCumulativeTestPoints();
348         final double[] densityValues = new double[testPoints.length];
349         final EmpiricalDistribution empiricalDistribution = (EmpiricalDistribution) makeDistribution();
350         final double[] binBounds = empiricalDistribution.getUpperBounds();
351         for (int i = 0; i < testPoints.length; i++) {
352             final int bin = findBin(testPoints[i]);
353             final double lower = bin == 0 ? empiricalDistribution.getSupportLowerBound() :
354                 binBounds[bin - 1];
355             final double upper = binBounds[bin];
356             final RealDistribution kernel = findKernel(lower, upper);
357             final double withinBinKernelMass = kernel.probability(lower, upper);
358             final double density = kernel.density(testPoints[i]);
359             densityValues[i] = density * (bin == 0 ? firstBinMass : binMass) / withinBinKernelMass;
360         }
361         return densityValues;
362     }

364     /**
365      * Modify test integration bounds from the default. Because the distribution
366      * has discontinuities at bin boundaries, integrals spanning multiple bins
367      * will face convergence problems.  Only test within-bin integrals and spans
368      * across no more than 3 bin boundaries.
369      */
370     @Override
371     @Test
372     public void testDensityIntegrals() {
373         final RealDistribution distribution = makeDistribution();
374         final double tol = 1.0e-9;
375         final BaseAbstractUnivariateIntegrator integrator =
376             new IterativeLegendreGaussIntegrator(5, 1.0e-12, 1.0e-10);
377         final UnivariateFunction d = new UnivariateFunction() {
378             @Override
379             public double value(double x) {
380                 return distribution.density(x);
381             }
382         };
383         final double[] lower = {0, 5, 1000, 5001, 9995};
384         final double[] upper = {5, 12, 1030, 5010, 10000};
385         for (int i = 1; i < 5; i++) {
386             Assert.assertEquals(
387                     distribution.probability(
388                             lower[i], upper[i]),
389                             integrator.integrate(
390                                     1000000, // Triangle integrals are very slow to converge
391                                     d, lower[i], upper[i]), tol);
392         }
393     }

414         for (int i = 0; i < 1000; i++) {
415             final double dev = sampler.sample();
416             Assert.assertTrue(dev < 1);
417             Assert.assertTrue(dev > 0);
418         }

441     /**
442      * Find the bin that x belongs (relative to {@link #makeDistribution()}).
443      */
444     private int findBin(double x) {
445         // Number of bins below x should be trunc(x/10)
446         final double nMinus = FastMath.floor(x / 10);
447         final int bin =  (int) FastMath.round(nMinus);
448         // If x falls on a bin boundary, it is in the lower bin
449         return FastMath.floor(x / 10) == x / 10 ? bin - 1 : bin;
450     }

452     /**
453      * Find the within-bin kernel for the bin with lower bound lower
454      * and upper bound upper. All bins other than the first contain 10 points
455      * exclusive of the lower bound and are centered at (lower + upper + 1) / 2.
456      * The first bin includes its lower bound, 0, so has different mean and
457      * standard deviation.
458      */
459     private RealDistribution findKernel(double lower, double upper) {
460         if (lower < 1) {
461             return new NormalDistribution(5d, 3.3166247903554);
462         } else {
463             return new NormalDistribution((upper + lower + 1) / 2d, 3.0276503540974917);
464         }
465     }

498         final double[] data = {1d,2d,3d, 4d,5d,6d, 7d,8d,9d, 10d,11d,12d, 13d,14d,15d};
499         dist.load(data);
500         RealDistribution.Sampler sampler
501             = dist.createSampler(RandomSource.create(RandomSource.WELL_19937_C, 1000));
502         // Kernels are uniform distributions on [1,3], [4,6], [7,9], [10,12], [13,15]
503         final double bounds[] = {3d, 6d, 9d, 12d};
504         final double tol = 10E-12;
505         for (int i = 0; i < 20; i++) {
506             final double v = sampler.sample();
507             // Make sure v is not in the excluded range between bins - that is (bounds[i], bounds[i] + 1)
508             for (int j = 0; j < bounds.length; j++) {
509                 Assert.assertFalse(v > bounds[j] + tol && v < bounds[j] + 1 - tol);
510             }
511         }
512         Assert.assertEquals(0.0, dist.cumulativeProbability(1), tol);
513         Assert.assertEquals(0.1, dist.cumulativeProbability(2), tol);
514         Assert.assertEquals(0.6, dist.cumulativeProbability(10), tol);
515         Assert.assertEquals(0.8, dist.cumulativeProbability(12), tol);
516         Assert.assertEquals(0.8, dist.cumulativeProbability(13), tol);
517         Assert.assertEquals(1.0, dist.cumulativeProbability(15), tol);
518 
519         Assert.assertEquals(2.0, dist.inverseCumulativeProbability(0.1), tol);
520         Assert.assertEquals(3.0, dist.inverseCumulativeProbability(0.2), tol);
521         Assert.assertEquals(5.0, dist.inverseCumulativeProbability(0.3), tol);
522         Assert.assertEquals(6.0, dist.inverseCumulativeProbability(0.4), tol);
