31  /**
32   * <p>Some of the unit tests are re-implementations of the MINPACK <a
33   * href="http://www.netlib.org/minpack/ex/file17">file17</a> and <a
34   * href="http://www.netlib.org/minpack/ex/file22">file22</a> test files.

37   *
38   */
39  public class GaussNewtonOptimizerWithQRTest
40      extends AbstractLeastSquaresOptimizerAbstractTest {
41  
42      @Override
43      public int getMaxIterations() {
44          return 1000;
45      }
46  
47      @Override
48      public LeastSquaresOptimizer getOptimizer() {
49          return new GaussNewtonOptimizer(Decomposition.QR);
50      }
51  
52      @Override
53      @Test
54      public void testMoreEstimatedParametersUnsorted() {
55          /*
56           * Exception is expected with this optimizer
57           */
58          try{
59              super.testMoreEstimatedParametersUnsorted();
60              fail(optimizer);
61          }catch (ConvergenceException e){
62              //expected
63          }
64      }
65  
66      @Test
67      public void testMaxEvaluations() throws Exception {
68          try{
69          CircleVectorial circle = new CircleVectorial();
70          circle.addPoint( 30.0,  68.0);
71          circle.addPoint( 50.0,  -6.0);
72          circle.addPoint(110.0, -20.0);
73          circle.addPoint( 35.0,  15.0);
74          circle.addPoint( 45.0,  97.0);
75  
76          LeastSquaresProblem lsp = builder(circle)
77                  .checkerPair(new SimpleVectorValueChecker(1e-30, 1e-30))
78                  .maxIterations(Integer.MAX_VALUE)
79                  .start(new double[]{98.680, 47.345})
80                  .build();
81  
82          optimizer.optimize(lsp);
83  
84              fail(optimizer);
85          }catch (TooManyEvaluationsException e){
86              //expected
87          }
88      }
89  
90      @Override
91      @Test
92      public void testCircleFittingBadInit() {
93          /*
94           * This test does not converge with this optimizer.
95           */
96          try{
97              super.testCircleFittingBadInit();
98              fail(optimizer);
99          }catch (ConvergenceException e){
100             //expected
101         }
102     }
103 
104     @Override
105     @Test
106     public void testHahn1()
107         throws IOException {
108         /*
109          * TODO This test leads to a singular problem with the Gauss-Newton

120     }
