41    @Test
42    public void testWrongDerivative()
43        throws DimensionMismatchException, NumberIsTooSmallException,
44               MaxCountExceededException, NoBracketingException {
45        HighamHall54Integrator integrator =
46            new HighamHall54Integrator(0.0, 1.0, 1.0e-10, 1.0e-10);
47        FirstOrderDifferentialEquations equations =
48            new FirstOrderDifferentialEquations() {
49              public void computeDerivatives(double t, double[] y, double[] dot) {
50              if (t < -0.5) {
51                  throw new LocalException();
52              } else {
53                  throw new RuntimeException("oops");
54             }
55            }
56            public int getDimension() {
57                return 1;
58            }
59        };
60  
61        try  {
62          integrator.integrate(equations, -1.0, new double[1], 0.0, new double[1]);
63          Assert.fail("an exception should have been thrown");
64        } catch(LocalException de) {
65          // expected behavior
66        }
67  
68        try  {
69          integrator.integrate(equations, 0.0, new double[1], 1.0, new double[1]);
70          Assert.fail("an exception should have been thrown");
71        } catch(RuntimeException de) {
72          // expected behavior
73        }
74  
75    }

100   @Test
101   public void testIncreasingTolerance()
102       throws DimensionMismatchException, NumberIsTooSmallException,
103              MaxCountExceededException, NoBracketingException {
104 
105     int previousCalls = Integer.MAX_VALUE;
106     for (int i = -12; i < -2; ++i) {
107       TestProblem1 pb = new TestProblem1();
108       double minStep = 0;
109       double maxStep = pb.getFinalTime() - pb.getInitialTime();
110       double scalAbsoluteTolerance = FastMath.pow(10.0, i);
111       double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
112 
113       FirstOrderIntegrator integ = new HighamHall54Integrator(minStep, maxStep,
114                                                               scalAbsoluteTolerance,
115                                                               scalRelativeTolerance);
116       TestProblemHandler handler = new TestProblemHandler(pb, integ);
117       integ.addStepHandler(handler);
118       integ.integrate(pb,
119                       pb.getInitialTime(), pb.getInitialState(),
120                       pb.getFinalTime(), new double[pb.getDimension()]);
121 
122       // the 1.3 factor is only valid for this test
123       // and has been obtained from trial and error
124       // there is no general relation between local and global errors
125       Assert.assertTrue(handler.getMaximalValueError() < (1.3 * scalAbsoluteTolerance));
126       Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
127 
128       int calls = pb.getCalls();
129       Assert.assertEquals(integ.getEvaluations(), calls);
130       Assert.assertTrue(calls <= previousCalls);
131       previousCalls = calls;
132 
133     }
134

162   @Test
163   public void testEvents()
164       throws DimensionMismatchException, NumberIsTooSmallException,
165              MaxCountExceededException, NoBracketingException {
166 
167     TestProblem4 pb = new TestProblem4();
168     double minStep = 0;
169     double maxStep = pb.getFinalTime() - pb.getInitialTime();
170     double scalAbsoluteTolerance = 1.0e-8;
171     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
172 
173     FirstOrderIntegrator integ = new HighamHall54Integrator(minStep, maxStep,
174                                                             scalAbsoluteTolerance,
175                                                             scalRelativeTolerance);
176     TestProblemHandler handler = new TestProblemHandler(pb, integ);
177     integ.addStepHandler(handler);
178     EventHandler[] functions = pb.getEventsHandlers();
179     double convergence = 1.0e-8 * maxStep;
180     for (int l = 0; l < functions.length; ++l) {
181       integ.addEventHandler(functions[l],
182                                  Double.POSITIVE_INFINITY, convergence, 1000);
183     }
184     Assert.assertEquals(functions.length, integ.getEventHandlers().size());
185     integ.integrate(pb,
186                     pb.getInitialTime(), pb.getInitialState(),
187                     pb.getFinalTime(), new double[pb.getDimension()]);
188 
189     Assert.assertTrue(handler.getMaximalValueError() < 1.0e-7);
190     Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
191     Assert.assertEquals(12.0, handler.getLastTime(), convergence);
192     integ.clearEventHandlers();
193     Assert.assertEquals(0, integ.getEventHandlers().size());
194 
195   }

199       throws DimensionMismatchException, NumberIsTooSmallException,
200              MaxCountExceededException, NoBracketingException {
201 
202       final TestProblem1 pb = new TestProblem1();
203       double minStep = 0;
204       double maxStep = pb.getFinalTime() - pb.getInitialTime();
205       double scalAbsoluteTolerance = 1.0e-8;
206       double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
207 
208       FirstOrderIntegrator integ =
209           new HighamHall54Integrator(minStep, maxStep,
210                                      scalAbsoluteTolerance, scalRelativeTolerance);
211       TestProblemHandler handler = new TestProblemHandler(pb, integ);
212       integ.addStepHandler(handler);
213 
214       integ.addEventHandler(new EventHandler() {
215         public void init(double t0, double[] y0, double t) {
216         }
217         public Action eventOccurred(double t, double[] y, boolean increasing) {
218           return Action.CONTINUE;
219         }
220         public double g(double t, double[] y) {
221           double middle = (pb.getInitialTime() + pb.getFinalTime()) / 2;
222           double offset = t - middle;
223           if (offset > 0) {
224             throw new LocalException();
225           }
226           return offset;
227         }
228         public void resetState(double t, double[] y) {
229         }
230       }, Double.POSITIVE_INFINITY, 1.0e-8 * maxStep, 1000);
231 
232       integ.integrate(pb,
233                       pb.getInitialTime(), pb.getInitialState(),
234                       pb.getFinalTime(), new double[pb.getDimension()]);
235 
236   }

250     double scalAbsoluteTolerance = 1.0e-8;
251     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
252

254         new HighamHall54Integrator(minStep, maxStep,

258

264       }
265       public double g(double t, double[] y) {
266         double middle = (pb.getInitialTime() + pb.getFinalTime()) / 2;
267         double offset = t - middle;
268         return (offset > 0) ? (offset + 0.5) : (offset - 0.5);
269       }
270       public void resetState(double t, double[] y) {
271       }
272     }, Double.POSITIVE_INFINITY, 1.0e-8 * maxStep, 3);
