198     public void testLcm() {
199         int a = 30;
200         int b = 50;
201         int c = 77;
202 
203         Assert.assertEquals(0, ArithmeticUtils.lcm(0, b));
204         Assert.assertEquals(0, ArithmeticUtils.lcm(a, 0));
205         Assert.assertEquals(b, ArithmeticUtils.lcm(1, b));
206         Assert.assertEquals(a, ArithmeticUtils.lcm(a, 1));
207         Assert.assertEquals(150, ArithmeticUtils.lcm(a, b));
208         Assert.assertEquals(150, ArithmeticUtils.lcm(-a, b));
209         Assert.assertEquals(150, ArithmeticUtils.lcm(a, -b));
210         Assert.assertEquals(150, ArithmeticUtils.lcm(-a, -b));
211         Assert.assertEquals(2310, ArithmeticUtils.lcm(a, c));
212 
213         // Assert that no intermediate value overflows:
214         // The naive implementation of lcm(a,b) would be (a*b)/gcd(a,b)
215         Assert.assertEquals((1<<20)*15, ArithmeticUtils.lcm((1<<20)*3, (1<<20)*5));
216 
217         // Special case
218         Assert.assertEquals(0, ArithmeticUtils.lcm(0, 0));
219 
220         try {
221             // lcm == abs(MIN_VALUE) cannot be represented as a nonnegative int
222             ArithmeticUtils.lcm(Integer.MIN_VALUE, 1);
223             Assert.fail("Expecting MathArithmeticException");
224         } catch (MathArithmeticException expected) {
225             // expected
226         }
227 
228         try {
229             // lcm == abs(MIN_VALUE) cannot be represented as a nonnegative int
230             ArithmeticUtils.lcm(Integer.MIN_VALUE, 1<<20);
231             Assert.fail("Expecting MathArithmeticException");
232         } catch (MathArithmeticException expected) {
233             // expected
234         }
235 
236         try {
237             ArithmeticUtils.lcm(Integer.MAX_VALUE, Integer.MAX_VALUE - 1);

585     /**
586      * Testing helper method.
587      * @return an array of int numbers containing corner cases:<ul>
588      * <li>values near the beginning of int range,</li>
589      * <li>values near the end of int range,</li>
590      * <li>values near zero</li>
591      * <li>and some randomly distributed values.</li>
592      * </ul>
593      */
594     private static int[] getIntSpecialCases() {
595         int ints[] = new int[100];
596         int i = 0;
597         ints[i++] = Integer.MAX_VALUE;
598         ints[i++] = Integer.MAX_VALUE - 1;
599         ints[i++] = 100;
600         ints[i++] = 101;
601         ints[i++] = 102;
602         ints[i++] = 300;
603         ints[i++] = 567;
604         for (int j = 0; j < 20; j++) {
605             ints[i++] = j;
606         }
607         for (int j = i - 1; j >= 0; j--) {
608             ints[i++] = ints[j] > 0 ? -ints[j] : Integer.MIN_VALUE;
609         }
610         java.util.Random r = new java.util.Random(System.nanoTime());
611         for (; i < ints.length;) {
612             ints[i++] = r.nextInt();
613         }
614         return ints;
615     }

622      * <li>values near the beginning of int range,</li>
623      * <li>values near the end of int range,</li>
624      * <li>values near zero</li>
625      * <li>and some randomly distributed values.</li>
626      * </ul>
627      */
628     private static long[] getLongSpecialCases() {
629         long longs[] = new long[100];
630         int i = 0;
631         longs[i++] = Long.MAX_VALUE;
632         longs[i++] = Long.MAX_VALUE - 1L;
633         longs[i++] = (long) Integer.MAX_VALUE + 1L;
634         longs[i++] = Integer.MAX_VALUE;
635         longs[i++] = Integer.MAX_VALUE - 1;
636         longs[i++] = 100L;
637         longs[i++] = 101L;
638         longs[i++] = 102L;
639         longs[i++] = 300L;
640         longs[i++] = 567L;
641         for (int j = 0; j < 20; j++) {
642             longs[i++] = j;
643         }
644         for (int j = i - 1; j >= 0; j--) {
645             longs[i++] = longs[j] > 0L ? -longs[j] : Long.MIN_VALUE;
646         }
647         java.util.Random r = new java.util.Random(System.nanoTime());
648         for (; i < longs.length;) {
649             longs[i++] = r.nextLong();
650         }
651         return longs;
652     }

654     private static long toUnsignedLong(int number) {
655         return number < 0 ? 0x100000000L + (long)number : (long)number;
656     }
657 
658     private static int remainderUnsignedExpected(int dividend, int divisor) {
659         return (int)remainderUnsignedExpected(toUnsignedLong(dividend), toUnsignedLong(divisor));
660     }
661 
662     private static int divideUnsignedExpected(int dividend, int divisor) {
663         return (int)divideUnsignedExpected(toUnsignedLong(dividend), toUnsignedLong(divisor));
664     }

666     private static BigInteger toUnsignedBigInteger(long number) {
667         return number < 0L ? BigInteger.ONE.shiftLeft(64).add(BigInteger.valueOf(number)) : BigInteger.valueOf(number);
668     }
669 
670     private static long remainderUnsignedExpected(long dividend, long divisor) {
671         return toUnsignedBigInteger(dividend).remainder(toUnsignedBigInteger(divisor)).longValue();
672     }

674     private static long divideUnsignedExpected(long dividend, long divisor) {
675         return toUnsignedBigInteger(dividend).divide(toUnsignedBigInteger(divisor)).longValue();
676     }
