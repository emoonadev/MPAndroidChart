18  package org.apache.commons.math4.stat.descriptive.moment;
19  
20  import org.apache.commons.math4.TestUtils;
21  import org.apache.commons.math4.exception.DimensionMismatchException;
22  import org.apache.commons.math4.linear.RealMatrix;
23  import org.apache.commons.math4.stat.descriptive.moment.VectorialCovariance;
24  import org.junit.Test;
25  import org.junit.Assert;
26  
27  public class VectorialCovarianceTest {
28      private double[][] points;
29  
30      public VectorialCovarianceTest() {
31          points = new double[][] {
32              { 1.2, 2.3,  4.5},
33              {-0.7, 2.3,  5.0},
34              { 3.1, 0.0, -3.1},
35              { 6.0, 1.2,  4.2},
36              {-0.7, 2.3,  5.0}
37          };
38      }
39  
40      @Test
41      public void testMismatch() {
42          try {
43              new VectorialCovariance(8, true).increment(new double[5]);
44              Assert.fail("an exception should have been thrown");
45          } catch (DimensionMismatchException dme) {
46              Assert.assertEquals(5, dme.getArgument());
47              Assert.assertEquals(8, dme.getDimension());
48          }
49      }
50  
51      @Test
52      public void testSimplistic() {
53          VectorialCovariance stat = new VectorialCovariance(2, true);
54          stat.increment(new double[] {-1.0,  1.0});
55          stat.increment(new double[] { 1.0, -1.0});
56          RealMatrix c = stat.getResult();
57          Assert.assertEquals( 2.0, c.getEntry(0, 0), 1.0e-12);
58          Assert.assertEquals(-2.0, c.getEntry(1, 0), 1.0e-12);
59          Assert.assertEquals( 2.0, c.getEntry(1, 1), 1.0e-12);
60      }
61  
62      @Test
63      public void testBasicStats() {
64  
65          VectorialCovariance stat = new VectorialCovariance(points[0].length, true);
66          for (int i = 0; i < points.length; ++i) {
67              stat.increment(points[i]);
68          }
69  
70          Assert.assertEquals(points.length, stat.getN());
71  
72          RealMatrix c = stat.getResult();
73          double[][] refC    = new double[][] {
74                  { 8.0470, -1.9195, -3.4445},
75                  {-1.9195,  1.0470,  3.2795},
76                  {-3.4445,  3.2795, 12.2070}
77          };
78  
79          for (int i = 0; i < c.getRowDimension(); ++i) {
80              for (int j = 0; j <= i; ++j) {
81                  Assert.assertEquals(refC[i][j], c.getEntry(i, j), 1.0e-12);
82              }
83          }
84  
85      }
86  
87      @Test
88      public void testSerial(){
89          VectorialCovariance stat = new VectorialCovariance(points[0].length, true);
90          Assert.assertEquals(stat, TestUtils.serializeAndRecover(stat));
91      }
92  }
