14  package org.apache.commons.math4.util;
15  
16  import org.apache.commons.math4.exception.MaxCountExceededException;
17  import org.apache.commons.math4.exception.TooManyEvaluationsException;
18  import org.apache.commons.math4.util.Incrementor;
19  import org.junit.Assert;
20  import org.junit.Test;
21  
22  /**
23   * Test for {@link Incrementor}.
24   */
25  public class IncrementorTest {
26      @Test
27      public void testConstructor1() {
28          final Incrementor i = new Incrementor();
29          Assert.assertEquals(0, i.getMaximalCount());
30          Assert.assertEquals(0, i.getCount());
31      }
32  
33      @Test
34      public void testConstructor2() {
35          final Incrementor i = new Incrementor(10);
36          Assert.assertEquals(10, i.getMaximalCount());
37          Assert.assertEquals(0, i.getCount());
38      }
39  
40      @Test
41      public void testCanIncrement1() {
42          final Incrementor i = new Incrementor(3);
43          Assert.assertTrue(i.canIncrement());
44          i.incrementCount();
45          Assert.assertTrue(i.canIncrement());
46          i.incrementCount();
47          Assert.assertTrue(i.canIncrement());
48          i.incrementCount();
49          Assert.assertFalse(i.canIncrement());
50      }
51  
52      @Test
53      public void testCanIncrement2() {
54          final Incrementor i = new Incrementor(3);
55          while (i.canIncrement()) {
56              i.incrementCount();
57          }
58  
59          // Must keep try/catch because the exception must be generated here,
60          // and not in the previous loop.
61          try {
62              i.incrementCount();
63              Assert.fail("MaxCountExceededException expected");
64          } catch (MaxCountExceededException e) {
65              // Expected.
66          }
67      }
68  
69      @Test
70      public void testAccessor() {
71          final Incrementor i = new Incrementor();
72  
73          i.setMaximalCount(10);
74          Assert.assertEquals(10, i.getMaximalCount());
75          Assert.assertEquals(0, i.getCount());
76      }
77  
78      @Test
79      public void testBelowMaxCount() {
80          final Incrementor i = new Incrementor();
81  
82          i.setMaximalCount(3);
83          i.incrementCount();
84          i.incrementCount();
85          i.incrementCount();
86  
87          Assert.assertEquals(3, i.getCount());
88      }
89  
90      @Test(expected=MaxCountExceededException.class)
91      public void testAboveMaxCount() {
92          final Incrementor i = new Incrementor();
93  
94          i.setMaximalCount(3);
95          i.incrementCount();
96          i.incrementCount();
97          i.incrementCount();
98          i.incrementCount();
99      }
100 
101     @Test(expected=TooManyEvaluationsException.class)
102     public void testAlternateException() {
103         final Incrementor.MaxCountExceededCallback cb
104             = new Incrementor.MaxCountExceededCallback() {
105                     /** {@inheritDoc} */
106                     public void trigger(int max) {
107                         throw new TooManyEvaluationsException(max);
108                     }
109                 };
110 
111         final Incrementor i = new Incrementor(0, cb);
112         i.incrementCount();
113     }
114 
115     @Test
116     public void testReset() {
117         final Incrementor i = new Incrementor();
118 
119         i.setMaximalCount(3);
120         i.incrementCount();
121         i.incrementCount();
122         i.incrementCount();
123         Assert.assertEquals(3, i.getCount());
124         i.resetCount();
125         Assert.assertEquals(0, i.getCount());
126     }
127 
128     @Test
129     public void testBulkIncrement() {
130         final Incrementor i = new Incrementor();
131 
132         i.setMaximalCount(3);
133         i.incrementCount(2);
134         Assert.assertEquals(2, i.getCount());
135         i.incrementCount(1);
136         Assert.assertEquals(3, i.getCount());
137     }
138 }
