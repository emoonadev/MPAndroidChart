59      /** test dimensions */
60      @Test
61      public void testDimensions() {
62          FieldMatrix<Fraction> matrix =
63              new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testData);
64          FieldLUDecomposition<Fraction> LU = new FieldLUDecomposition<Fraction>(matrix);
65          Assert.assertEquals(testData.length, LU.getL().getRowDimension());
66          Assert.assertEquals(testData.length, LU.getL().getColumnDimension());
67          Assert.assertEquals(testData.length, LU.getU().getRowDimension());
68          Assert.assertEquals(testData.length, LU.getU().getColumnDimension());
69          Assert.assertEquals(testData.length, LU.getP().getRowDimension());
70          Assert.assertEquals(testData.length, LU.getP().getColumnDimension());

72      }

90      /** test PA = LU */
91      @Test
92      public void testPAEqualLU() {
93          FieldMatrix<Fraction> matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testData);
94          FieldLUDecomposition<Fraction> lu = new FieldLUDecomposition<Fraction>(matrix);
95          FieldMatrix<Fraction> l = lu.getL();
96          FieldMatrix<Fraction> u = lu.getU();
97          FieldMatrix<Fraction> p = lu.getP();
98          TestUtils.assertEquals(p.multiply(matrix), l.multiply(u));
99  
100         matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testDataMinus);
101         lu = new FieldLUDecomposition<Fraction>(matrix);
102         l = lu.getL();
103         u = lu.getU();
104         p = lu.getP();
105         TestUtils.assertEquals(p.multiply(matrix), l.multiply(u));
106 
107         matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), 17, 17);
108         for (int i = 0; i < matrix.getRowDimension(); ++i) {
109             matrix.setEntry(i, i, Fraction.ONE);
110         }
111         lu = new FieldLUDecomposition<Fraction>(matrix);
112         l = lu.getL();
113         u = lu.getU();
114         p = lu.getP();
115         TestUtils.assertEquals(p.multiply(matrix), l.multiply(u));
116 
117         matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), singular);
118         lu = new FieldLUDecomposition<Fraction>(matrix);
119         Assert.assertFalse(lu.getSolver().isNonSingular());
120         Assert.assertNull(lu.getL());
121         Assert.assertNull(lu.getU());
122         Assert.assertNull(lu.getP());
123 
124         matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), bigSingular);
125         lu = new FieldLUDecomposition<Fraction>(matrix);
126         Assert.assertFalse(lu.getSolver().isNonSingular());
127         Assert.assertNull(lu.getL());
128         Assert.assertNull(lu.getU());
129         Assert.assertNull(lu.getP());
130 
131     }

133     /** test that L is lower triangular with unit diagonal */
134     @Test
135     public void testLLowerTriangular() {
136         FieldMatrix<Fraction> matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testData);
137         FieldMatrix<Fraction> l = new FieldLUDecomposition<Fraction>(matrix).getL();
138         for (int i = 0; i < l.getRowDimension(); i++) {
139             Assert.assertEquals(Fraction.ONE, l.getEntry(i, i));
140             for (int j = i + 1; j < l.getColumnDimension(); j++) {
141                 Assert.assertEquals(Fraction.ZERO, l.getEntry(i, j));
142             }
143         }

158     /** test that P is a permutation matrix */
159     @Test
160     public void testPPermutation() {
161         FieldMatrix<Fraction> matrix = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testData);
162         FieldMatrix<Fraction> p   = new FieldLUDecomposition<Fraction>(matrix).getP();
163 
164         FieldMatrix<Fraction> ppT = p.multiply(p.transpose());
165         FieldMatrix<Fraction> id  =
166             new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(),
167                                           p.getRowDimension(), p.getRowDimension());
168         for (int i = 0; i < id.getRowDimension(); ++i) {
169             id.setEntry(i, i, Fraction.ONE);
170         }
171         TestUtils.assertEquals(id, ppT);
172 
173         for (int i = 0; i < p.getRowDimension(); i++) {
174             int zeroCount  = 0;
175             int oneCount   = 0;
176             int otherCount = 0;
177             for (int j = 0; j < p.getColumnDimension(); j++) {
178                 final Fraction e = p.getEntry(i, j);
179                 if (e.equals(Fraction.ZERO)) {
180                     ++zeroCount;
181                 } else if (e.equals(Fraction.ONE)) {
182                     ++oneCount;
183                 } else {
184                     ++otherCount;
185                 }
186             }
187             Assert.assertEquals(p.getColumnDimension() - 1, zeroCount);
188             Assert.assertEquals(1, oneCount);
189             Assert.assertEquals(0, otherCount);
190         }
191 
192         for (int j = 0; j < p.getColumnDimension(); j++) {
193             int zeroCount  = 0;
194             int oneCount   = 0;
195             int otherCount = 0;
196             for (int i = 0; i < p.getRowDimension(); i++) {
197                 final Fraction e = p.getEntry(i, j);
198                 if (e.equals(Fraction.ZERO)) {
199                     ++zeroCount;
200                 } else if (e.equals(Fraction.ONE)) {
201                     ++oneCount;
202                 } else {
203                     ++otherCount;
204                 }
205             }
206             Assert.assertEquals(p.getRowDimension() - 1, zeroCount);
207             Assert.assertEquals(1, oneCount);
208             Assert.assertEquals(0, otherCount);
209         }
210 
211     }

226     /** test matrices values */
227     @Test
228     public void testMatricesValues1() {
229        FieldLUDecomposition<Fraction> lu =
230             new FieldLUDecomposition<Fraction>(new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), testData));
231         FieldMatrix<Fraction> lRef = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), new Fraction[][] {
232                 { new Fraction(1), new Fraction(0), new Fraction(0) },
233                 { new Fraction(2), new Fraction(1), new Fraction(0) },
234                 { new Fraction(1), new Fraction(-2), new Fraction(1) }
235         });
236         FieldMatrix<Fraction> uRef = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), new Fraction[][] {
237                 { new Fraction(1),  new Fraction(2), new Fraction(3) },
238                 { new Fraction(0), new Fraction(1), new Fraction(-3) },
239                 { new Fraction(0),  new Fraction(0), new Fraction(-1) }
240         });
241         FieldMatrix<Fraction> pRef = new Array2DRowFieldMatrix<Fraction>(FractionField.getInstance(), new Fraction[][] {
242                 { new Fraction(1), new Fraction(0), new Fraction(0) },
243                 { new Fraction(0), new Fraction(1), new Fraction(0) },
244                 { new Fraction(0), new Fraction(0), new Fraction(1) }
245         });
246         int[] pivotRef = { 0, 1, 2 };
247 
248         // check values against known references
249         FieldMatrix<Fraction> l = lu.getL();
250         TestUtils.assertEquals(lRef, l);
251         FieldMatrix<Fraction> u = lu.getU();
252         TestUtils.assertEquals(uRef, u);
253         FieldMatrix<Fraction> p = lu.getP();
254         TestUtils.assertEquals(pRef, p);
255         int[] pivot = lu.getPivot();
256         for (int i = 0; i < pivotRef.length; ++i) {
257             Assert.assertEquals(pivotRef[i], pivot[i]);
258         }
259 
260         // check the same cached instance is returned the second time
261         Assert.assertTrue(l == lu.getL());
262         Assert.assertTrue(u == lu.getU());
263         Assert.assertTrue(p == lu.getP());
264 
265     }
