17  package org.apache.commons.math4.fitting.leastsquares;

19  import org.apache.commons.math4.fitting.leastsquares.EvaluationRmsChecker;
20  import org.apache.commons.math4.fitting.leastsquares.LeastSquaresProblem.Evaluation;

27  /** Unit tests for {@link EvaluationRmsChecker}. */
28  public class EvaluationRmsCheckerTest {
29  
30      /** check {@link ConvergenceChecker#converged(int, Object, Object)}. */
31      @Test
32      public void testConverged() {
33          //setup
34          ConvergenceChecker<Evaluation> checker = new EvaluationRmsChecker(0.1, 1);
35          Evaluation e200 = mockEvaluation(200);
36          Evaluation e1 = mockEvaluation(1);
37  
38          //action + verify
39          //just matches rel tol
40          Assert.assertEquals(true, checker.converged(0, e200, mockEvaluation(210)));
41          //just matches abs tol
42          Assert.assertEquals(true, checker.converged(0, e1, mockEvaluation(1.9)));
43          //matches both
44          Assert.assertEquals(true, checker.converged(0, e1, mockEvaluation(1.01)));
45          //matches neither
46          Assert.assertEquals(false, checker.converged(0, e200, mockEvaluation(300)));
47      }
48  
49      /**
50       * Create a mock {@link Evaluation}.
51       *
52       * @param rms the evaluation's rms.
53       * @return a new mock evaluation.
54       */
55      private static Evaluation mockEvaluation(final double rms) {
56          return new Evaluation() {
57              public RealMatrix getCovariances(double threshold) {
58                  return null;
59              }
60  
61              public RealVector getSigma(double covarianceSingularityThreshold) {
62                  return null;
63              }
64  
65              public double getRMS() {
66                  return rms;
67              }
68  
69              public RealMatrix getJacobian() {
70                  return null;
71              }
72  
73              public double getCost() {
74                  return 0;
75              }
76  
77              public double getChiSquare() {
78                  return 0;
79              }
80  
81              public double getReducedChiSquare(int n) {
82                  return 0;
83              }
84  
85              public RealVector getResiduals() {
86                  return null;
87              }
88  
89              public RealVector getPoint() {
90                  return null;
91              }
92          };
93      }
