73      @Test
74      public abstract void testIncreasingTolerance();
75  
76      protected <T extends RealFieldElement<T>> void doTestIncreasingTolerance(final Field<T> field,
77                                                                               double ratioMin, double ratioMax) {
78  
79          int previousCalls = Integer.MAX_VALUE;
80          for (int i = -12; i < -2; ++i) {
81              TestFieldProblem1<T> pb = new TestFieldProblem1<T>(field);
82              double minStep = 0;
83              double maxStep = pb.getFinalTime().subtract(pb.getInitialState().getTime()).getReal();
84              double scalAbsoluteTolerance = FastMath.pow(10.0, i);
85              double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
86  
87              FirstOrderFieldIntegrator<T> integ = createIntegrator(field, 4, minStep, maxStep,
88                                                                    scalAbsoluteTolerance,
89                                                                    scalRelativeTolerance);
90              TestFieldProblemHandler<T> handler = new TestFieldProblemHandler<T>(pb, integ);
91              integ.addStepHandler(handler);
92              integ.integrate(new FieldExpandableODE<T>(pb), pb.getInitialState(), pb.getFinalTime());
93  
94              Assert.assertTrue(handler.getMaximalValueError().getReal() > ratioMin * scalAbsoluteTolerance);
95              Assert.assertTrue(handler.getMaximalValueError().getReal() < ratioMax * scalAbsoluteTolerance);
96  
97              int calls = pb.getCalls();
98              Assert.assertEquals(integ.getEvaluations(), calls);
99              Assert.assertTrue(calls <= previousCalls);
100             previousCalls = calls;
101 
102         }

145     @Test
146     public abstract void polynomial();
147 
148     protected <T extends RealFieldElement<T>> void doPolynomial(final Field<T> field,
149                                                                 final int nLimit,
150                                                                 final double epsilonBad,
151                                                                 final double epsilonGood) {
152         TestFieldProblem6<T> pb = new TestFieldProblem6<T>(field);
153         double range = pb.getFinalTime().subtract(pb.getInitialState().getTime()).abs().getReal();
154 
155         for (int nSteps = 2; nSteps < 8; ++nSteps) {
156             AdamsFieldIntegrator<T> integ = createIntegrator(field, nSteps, 1.0e-6 * range, 0.1 * range, 1.0e-4, 1.0e-4);
157             integ.setStarterIntegrator(new PerfectStarter<T>(pb, nSteps));
158             TestFieldProblemHandler<T> handler = new TestFieldProblemHandler<T>(pb, integ);
159             integ.addStepHandler(handler);
160             integ.integrate(new FieldExpandableODE<T>(pb), pb.getInitialState(), pb.getFinalTime());
161             if (nSteps < nLimit) {
162                 Assert.assertTrue(handler.getMaximalValueError().getReal() > epsilonBad);
163             } else {
164                 Assert.assertTrue(handler.getMaximalValueError().getReal() < epsilonGood);
165             }
166         }

190     private static class PerfectStarter<T extends RealFieldElement<T>> extends AbstractFieldIntegrator<T> {
191 
192         private final PerfectInterpolator<T> interpolator;
193         private final int nbSteps;
194 
195         public PerfectStarter(final TestFieldProblemAbstract<T> problem, final int nbSteps) {
196             super(problem.getField(), "perfect-starter");
197             this.interpolator = new PerfectInterpolator<T>(problem);
198             this.nbSteps      = nbSteps;
199         }
200 
201         public FieldODEStateAndDerivative<T> integrate(FieldExpandableODE<T> equations,
202                                                        FieldODEState<T> initialState, T finalTime) {
203             T tStart = initialState.getTime().add(finalTime.subtract(initialState.getTime()).multiply(0.01));
204             getEvaluationsCounter().increment(nbSteps);
205             interpolator.setCurrentTime(initialState.getTime());
206             for (int i = 0; i < nbSteps; ++i) {
207                 T tK = initialState.getTime().multiply(nbSteps - 1 - (i + 1)).add(tStart.multiply(i + 1)).divide(nbSteps - 1);
208                 interpolator.setPreviousTime(interpolator.getCurrentTime());
209                 interpolator.setCurrentTime(tK);
210                 for (FieldStepHandler<T> handler : getStepHandlers()) {
211                     handler.handleStep(interpolator, i == nbSteps - 1);
212                 }
213             }
214             return interpolator.getInterpolatedState(tStart);
215         }
216 
217     }

219     private static class PerfectInterpolator<T extends RealFieldElement<T>> implements FieldStepInterpolator<T> {
220         private final TestFieldProblemAbstract<T> problem;
221         private T previousTime;
222         private T currentTime;
223 
224         public PerfectInterpolator(final TestFieldProblemAbstract<T> problem) {
225             this.problem = problem;
226         }
227 
228         public void setPreviousTime(T previousTime) {
229             this.previousTime = previousTime;
230         }
231 
232         public void setCurrentTime(T currentTime) {
233             this.currentTime = currentTime;
234         }
235 
236         public T getCurrentTime() {
237             return currentTime;
238         }
239 
240         public boolean isForward() {
241             return problem.getFinalTime().subtract(problem.getInitialState().getTime()).getReal() >= 0;
242         }
243 
244         public FieldODEStateAndDerivative<T> getPreviousState() {
245             return getInterpolatedState(previousTime);
246         }
247 
248         public FieldODEStateAndDerivative<T> getCurrentState() {
249             return getInterpolatedState(currentTime);
250         }
251 
252         public FieldODEStateAndDerivative<T> getInterpolatedState(T time) {
253             T[] y    = problem.computeTheoreticalState(time);
254             T[] yDot = problem.computeDerivatives(time, y);
255             return new FieldODEStateAndDerivative<T>(time, y, yDot);
256         }
