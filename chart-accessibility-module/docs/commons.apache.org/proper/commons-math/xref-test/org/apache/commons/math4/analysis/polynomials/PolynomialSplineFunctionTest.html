17  package org.apache.commons.math4.analysis.polynomials;
18  
19  import java.util.Arrays;
20  
21  import org.apache.commons.math4.analysis.UnivariateFunction;
22  import org.apache.commons.math4.exception.MathIllegalArgumentException;
23  import org.apache.commons.math4.exception.MathIllegalStateException;
24  import org.apache.commons.math4.exception.OutOfRangeException;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  /**
29   * Tests the PolynomialSplineFunction implementation.
30   *
31   */
32  public class PolynomialSplineFunctionTest {
33  
34      /** Error tolerance for tests */
35      protected double tolerance = 1.0e-12;
36  
37      /**
38       * Quadratic polynomials used in tests:
39       *
40       * x^2 + x            [-1, 0)
41       * x^2 + x + 2        [0, 1)
42       * x^2 + x + 4        [1, 2)
43       *
44       * Defined so that evaluation using PolynomialSplineFunction evaluation
45       * algorithm agrees at knot point boundaries.
46       */
47      protected PolynomialFunction[] polynomials = {
48          new PolynomialFunction(new double[] {0d, 1d, 1d}),
49          new PolynomialFunction(new double[] {2d, 1d, 1d}),
50          new PolynomialFunction(new double[] {4d, 1d, 1d})
51      };
52  
53      /** Knot points  */
54      protected double[] knots = {-1, 0, 1, 2};
55  
56      /** Derivative of test polynomials -- 2x + 1  */
57      protected PolynomialFunction dp =
58          new PolynomialFunction(new double[] {1d, 2d});
59  
60  
61      @Test
62      public void testConstructor() {
63          PolynomialSplineFunction spline =
64              new PolynomialSplineFunction(knots, polynomials);
65          Assert.assertTrue(Arrays.equals(knots, spline.getKnots()));
66          Assert.assertEquals(1d, spline.getPolynomials()[0].getCoefficients()[2], 0);
67          Assert.assertEquals(3, spline.getN());
68  
69          try { // too few knots
70              new PolynomialSplineFunction(new double[] {0}, polynomials);
71              Assert.fail("Expecting MathIllegalArgumentException");
72          } catch (MathIllegalArgumentException ex) {
73              // expected
74          }
75  
76          try { // too many knots
77              new PolynomialSplineFunction(new double[] {0,1,2,3,4}, polynomials);
78              Assert.fail("Expecting MathIllegalArgumentException");
79          } catch (MathIllegalArgumentException ex) {
80              // expected
81          }
82  
83          try { // knots not increasing
84              new PolynomialSplineFunction(new double[] {0,1, 3, 2}, polynomials);
85              Assert.fail("Expecting MathIllegalArgumentException");
86          } catch (MathIllegalArgumentException ex) {
87              // expected
88          }
89      }
90  
91      @Test
92      public void testValues() {
93          PolynomialSplineFunction spline =
94              new PolynomialSplineFunction(knots, polynomials);
95          UnivariateFunction dSpline = spline.polynomialSplineDerivative();
96  
97          /**
98           * interior points -- spline value at x should equal p(x - knot)
99           * where knot is the largest knot point less than or equal to x and p
100          * is the polynomial defined over the knot segment to which x belongs.
101          */
102         double x = -1;
103         int index = 0;
104         for (int i = 0; i < 10; i++) {
105            x+=0.25;
106            index = findKnot(knots, x);
107            Assert.assertEquals("spline function evaluation failed for x=" + x,
108                    polynomials[index].value(x - knots[index]), spline.value(x), tolerance);
109            Assert.assertEquals("spline derivative evaluation failed for x=" + x,
110                    dp.value(x - knots[index]), dSpline.value(x), tolerance);
111         }
112 
113         // knot points -- centering should zero arguments
114         for (int i = 0; i < 3; i++) {
115             Assert.assertEquals("spline function evaluation failed for knot=" + knots[i],
116                     polynomials[i].value(0), spline.value(knots[i]), tolerance);
117             Assert.assertEquals("spline function evaluation failed for knot=" + knots[i],
118                     dp.value(0), dSpline.value(knots[i]), tolerance);
119         }
120 
121         try { //outside of domain -- under min
122             x = spline.value(-1.5);
123             Assert.fail("Expecting OutOfRangeException");
124         } catch (OutOfRangeException ex) {
125             // expected
126         }
127 
128         try { //outside of domain -- over max
129             x = spline.value(2.5);
130             Assert.fail("Expecting OutOfRangeException");
131         } catch (OutOfRangeException ex) {
132             // expected
133         }
134     }
135 
136     @Test
137     public void testIsValidPoint() {
138         final PolynomialSplineFunction spline =
139             new PolynomialSplineFunction(knots, polynomials);
140         final double xMin = knots[0];
141         final double xMax = knots[knots.length - 1];
142 
143         double x;
144 
145         x = xMin;
146         Assert.assertTrue(spline.isValidPoint(x));
147         // Ensure that no exception is thrown.
148         spline.value(x);
149 
150         x = xMax;
151         Assert.assertTrue(spline.isValidPoint(x));
152         // Ensure that no exception is thrown.
153         spline.value(x);
154 
155         final double xRange = xMax - xMin;
156         x = xMin + xRange / 3.4;
157         Assert.assertTrue(spline.isValidPoint(x));
158         // Ensure that no exception is thrown.
159         spline.value(x);
160 
161         final double small = 1e-8;
162         x = xMin - small;
163         Assert.assertFalse(spline.isValidPoint(x));
164         // Ensure that an exception would have been thrown.
165         try {
166             spline.value(x);
167             Assert.fail("OutOfRangeException expected");
168         } catch (OutOfRangeException expected) {}
169     }
170 
171     /**
172      *  Do linear search to find largest knot point less than or equal to x.
173      *  Implementation does binary search.
174      */
175      protected int findKnot(double[] knots, double x) {
176          if (x < knots[0] || x >= knots[knots.length -1]) {
177              throw new OutOfRangeException(x, knots[0], knots[knots.length -1]);
178          }
179          for (int i = 0; i < knots.length; i++) {
180              if (knots[i] > x) {
181                  return i - 1;
182              }
183          }
184          throw new MathIllegalStateException();
185      }
186 }
