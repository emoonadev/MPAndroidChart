47      @Test
48      public void testCompareSigmoid() {
49          final UnivariateFunction sig = new Sigmoid();
50          final UnivariateFunction sigL = new Logistic(1, 0, 1, 1, 0, 1);
51  
52          final double min = -2;
53          final double max = 2;
54          final int n = 100;
55          final double delta = (max - min) / n;
56          for (int i = 0; i < n; i++) {
57              final double x = min + i * delta;
58              Assert.assertEquals("x=" + x, sig.value(x), sigL.value(x), EPS);
59          }
60      }

62      @Test
63      public void testSomeValues() {
64          final double k = 4;
65          final double m = 5;
66          final double b = 2;
67          final double q = 3;
68          final double a = -1;
69          final double n = 2;
70  
71          final UnivariateFunction f = new Logistic(k, m, b, q, a, n);
72  
73          double x;
74          x = m;
75          Assert.assertEquals("x=" + x, a + (k - a) / FastMath.sqrt(1 + q), f.value(x), EPS);
76  
77          x = Double.NEGATIVE_INFINITY;
78          Assert.assertEquals("x=" + x, a, f.value(x), EPS);
79  
80          x = Double.POSITIVE_INFINITY;
81          Assert.assertEquals("x=" + x, k, f.value(x), EPS);
82      }

84      @Test
85      public void testCompareDerivativeSigmoid() {
86          final double k = 3;
87          final double a = 2;
88  
89          final Logistic f = new Logistic(k, 0, 1, 1, a, 1);
90          final Sigmoid g = new Sigmoid(a, k);
91  
92          final double min = -10;
93          final double max = 10;
94          final double n = 20;
95          final double delta = (max - min) / n;
96          for (int i = 0; i < n; i++) {
97              final DerivativeStructure x = new DerivativeStructure(1, 5, 0, min + i * delta);
98              for (int order = 0; order <= x.getOrder(); ++order) {
99                  Assert.assertEquals("x=" + x.getValue(),
100                                     g.value(x).getPartialDerivative(order),
101                                     f.value(x).getPartialDerivative(order),
102                                     3.0e-15);
103             }
104         }
105     }

143     @Test
144     public void testGradientComponent0Component4() {
145         final double k = 3;
146         final double a = 2;
147 
148         final Logistic.Parametric f = new Logistic.Parametric();
149         // Compare using the "Sigmoid" function.
150         final Sigmoid.Parametric g = new Sigmoid.Parametric();
151 
152         final double x = 0.12345;
153         final double[] gf = f.gradient(x, new double[] {k, 0, 1, 1, a, 1});
154         final double[] gg = g.gradient(x, new double[] {a, k});
155 
156         Assert.assertEquals(gg[0], gf[4], EPS);
157         Assert.assertEquals(gg[1], gf[0], EPS);
158     }

160     @Test
161     public void testGradientComponent5() {
162         final double m = 1.2;
163         final double k = 3.4;
164         final double a = 2.3;
165         final double q = 0.567;
166         final double b = -FastMath.log(q);
167         final double n = 3.4;
168 
169         final Logistic.Parametric f = new Logistic.Parametric();
170 
171         final double x = m - 1;
172         final double qExp1 = 2;
173 
174         final double[] gf = f.gradient(x, new double[] {k, m, b, q, a, n});
175 
176         Assert.assertEquals((k - a) * FastMath.log(qExp1) / (n * n * FastMath.pow(qExp1, 1 / n)),
177                             gf[5], EPS);
178     }
179 
180     @Test
181     public void testGradientComponent1Component2Component3() {
182         final double m = 1.2;
183         final double k = 3.4;
184         final double a = 2.3;
185         final double b = 0.567;
186         final double q = 1 / FastMath.exp(b * m);
187         final double n = 3.4;
188 
189         final Logistic.Parametric f = new Logistic.Parametric();
190 
191         final double x = 0;
192         final double qExp1 = 2;
193 
194         final double[] gf = f.gradient(x, new double[] {k, m, b, q, a, n});
195 
196         final double factor = (a - k) / (n * FastMath.pow(qExp1, 1 / n + 1));
197         Assert.assertEquals(factor * b, gf[1], EPS);
198         Assert.assertEquals(factor * m, gf[2], EPS);
199         Assert.assertEquals(factor / q, gf[3], EPS);
200     }
