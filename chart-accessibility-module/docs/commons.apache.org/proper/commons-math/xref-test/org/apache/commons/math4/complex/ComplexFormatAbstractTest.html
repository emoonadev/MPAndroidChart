57      @Test
58      public void testTrimOneImaginary() {
59          final ComplexFormat fmt = ComplexFormat.getInstance(getLocale());
60          fmt.getImaginaryFormat().setMaximumFractionDigits(1);
61  
62          Complex c = new Complex(1, 1.04);
63          String expected = "1 + i";
64          String actual = fmt.format(c);
65          Assert.assertEquals(expected, actual);
66  
67          c = new Complex(1, 1.09);
68          expected = "1 + 1" + getDecimalCharacter() + "1i";
69          actual = fmt.format(c);
70          Assert.assertEquals(expected, actual);
71  
72          c = new Complex(1, -1.09);
73          expected = "1 - 1" + getDecimalCharacter() + "1i";
74          actual = fmt.format(c);
75          Assert.assertEquals(expected, actual);
76  
77          c = new Complex(1, -1.04);
78          expected = "1 - i";
79          actual = fmt.format(c);
80          Assert.assertEquals(expected, actual);
81      }

288     @Test
289     public void testConstructorExceptions() {
290         NumberFormat nullFormat = null;
291         NumberFormat format = NumberFormat.getInstance();
292         try {
293             ComplexFormat cf = new ComplexFormat(nullFormat);
294         }
295         catch (Exception e) {
296             Assert.assertTrue(e instanceof NullArgumentException);
297         }
298         try {
299             ComplexFormat cf = new ComplexFormat(nullFormat, format);
300         }
301         catch (Exception e) {
302             Assert.assertTrue(e instanceof NullArgumentException);
303         }
304         try {
305             ComplexFormat cf = new ComplexFormat(format, nullFormat);
306         }
307         catch (Exception e) {
308             Assert.assertTrue(e instanceof NullArgumentException);
309         }
