17  package org.apache.commons.math4.analysis.integration.gauss;
18  
19  import org.apache.commons.math4.analysis.UnivariateFunction;
20  import org.apache.commons.math4.analysis.function.Constant;
21  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegrator;
22  import org.apache.commons.math4.util.Pair;
23  import org.junit.Test;
24  import org.junit.Assert;
25  
26  /**
27   * Test for {@link GaussIntegrator} class.
28   *
29   */
30  public class GaussIntegratorTest {
31      @Test
32      public void testGetWeights() {
33          final double[] points = { 0, 1.2, 3.4 };
34          final double[] weights = { 9.8, 7.6, 5.4 };
35  
36          final GaussIntegrator integrator
37              = new GaussIntegrator(new Pair<double[], double[]>(points, weights));
38  
39          Assert.assertEquals(weights.length, integrator.getNumberOfPoints());
40  
41          for (int i = 0; i < integrator.getNumberOfPoints(); i++) {
42              Assert.assertEquals(weights[i], integrator.getWeight(i), 0d);
43          }
44      }
45  
46      @Test
47      public void testGetPoints() {
48          final double[] points = { 0, 1.2, 3.4 };
49          final double[] weights = { 9.8, 7.6, 5.4 };
50  
51          final GaussIntegrator integrator
52              = new GaussIntegrator(new Pair<double[], double[]>(points, weights));
53  
54          Assert.assertEquals(points.length, integrator.getNumberOfPoints());
55  
56          for (int i = 0; i < integrator.getNumberOfPoints(); i++) {
57              Assert.assertEquals(points[i], integrator.getPoint(i), 0d);
58          }
59      }
60  
61      @Test
62      public void testIntegrate() {
63          final double[] points = { 0, 1, 2, 3, 4, 5 };
64          final double[] weights = { 1, 1, 1, 1, 1, 1 };
65  
66          final GaussIntegrator integrator
67              = new GaussIntegrator(new Pair<double[], double[]>(points, weights));
68  
69          final double val = 123.456;
70          final UnivariateFunction c = new Constant(val);
71  
72          final double s = integrator.integrate(c);
73          Assert.assertEquals(points.length * val, s, 0d);
74      }
75  }
