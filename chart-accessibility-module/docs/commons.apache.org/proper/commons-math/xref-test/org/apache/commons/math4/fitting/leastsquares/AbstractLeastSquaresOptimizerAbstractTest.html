61      /** default absolute tolerance of comparisons */
62      public static final double TOl = 1e-10;
63  
64      public LeastSquaresBuilder base() {
65          return new LeastSquaresBuilder()
66                  .checkerPair(new SimpleVectorValueChecker(1e-6, 1e-6))
67                  .maxEvaluations(100)
68                  .maxIterations(getMaxIterations());
69      }
70  
71      public LeastSquaresBuilder builder(CircleVectorial c) {
72          final double[] weights = new double[c.getN()];
73          Arrays.fill(weights, 1.0);
74          return base()
75                  .model(c.getModelFunction(), c.getModelFunctionJacobian())
76                  .target(new double[c.getN()])
77                  .weight(new DiagonalMatrix(weights));
78      }
79  
80      public LeastSquaresBuilder builder(StatisticalReferenceDataset dataset) {
81          StatisticalReferenceDataset.LeastSquaresProblem problem
82                  = dataset.getLeastSquaresProblem();
83          final double[] weights = new double[dataset.getNumObservations()];
84          Arrays.fill(weights, 1.0);
85          return base()
86                  .model(problem.getModelFunction(), problem.getModelFunctionJacobian())
87                  .target(dataset.getData()[1])
88                  .weight(new DiagonalMatrix(weights))
89                  .start(dataset.getStartingPoint(0));
90      }
91  
92      public void fail(LeastSquaresOptimizer optimizer) {
93          Assert.fail("Expected Exception from: " + optimizer.toString());
94      }

96      /**
97       * Check the value of a vector.
98       * @param tolerance the absolute tolerance of comparisons
99       * @param actual the vector to test
100      * @param expected the expected values
101      */
102     public void assertEquals(double tolerance, RealVector actual, double... expected){
103         for (int i = 0; i < expected.length; i++) {
104             Assert.assertEquals(expected[i], actual.getEntry(i), tolerance);
105         }
106         Assert.assertEquals(expected.length, actual.getDimension());
107     }

397     @Test
398     public void testCircleFitting() {
399         CircleVectorial circle = new CircleVectorial();
400         circle.addPoint(30, 68);
401         circle.addPoint(50, -6);
402         circle.addPoint(110, -20);
403         circle.addPoint(35, 15);
404         circle.addPoint(45, 97);
405         final double[] start = {98.680, 47.345};
406 
407         Optimum optimum = optimizer.optimize(builder(circle).start(start).build());
408 
409         Assert.assertTrue(optimum.getEvaluations() < 10);
410 
411         double rms = optimum.getRMS();
412         Assert.assertEquals(1.768262623567235, FastMath.sqrt(circle.getN()) * rms, TOl);
413 
414         Vector2D center = new Vector2D(optimum.getPoint().getEntry(0), optimum.getPoint().getEntry(1));
415         Assert.assertEquals(69.96016176931406, circle.getRadius(center), 1e-6);
416         Assert.assertEquals(96.07590211815305, center.getX(), 1e-6);
417         Assert.assertEquals(48.13516790438953, center.getY(), 1e-6);
418 
419         double[][] cov = optimum.getCovariances(1e-14).getData();
420         Assert.assertEquals(1.839, cov[0][0], 0.001);
421         Assert.assertEquals(0.731, cov[0][1], 0.001);
422         Assert.assertEquals(cov[0][1], cov[1][0], 1e-14);
423         Assert.assertEquals(0.786, cov[1][1], 0.001);
424 
425         // add perfect measurements and check formal errors are reduced
426         double r = circle.getRadius(center);
427         for (double d = 0; d < 2 * FastMath.PI; d += 0.01) {
428             circle.addPoint(center.getX() + r * FastMath.cos(d), center.getY() + r * FastMath.sin(d));
429         }
430 
431         double[] weights = new double[circle.getN()];
432         Arrays.fill(weights, 2);
433 
434         optimum = optimizer.optimize(
435                 builder(circle).weight(new DiagonalMatrix(weights)).start(start).build());
436 
437         cov = optimum.getCovariances(1e-14).getData();
438         Assert.assertEquals(0.0016, cov[0][0], 0.001);
439         Assert.assertEquals(3.2e-7, cov[0][1], 1e-9);
440         Assert.assertEquals(cov[0][1], cov[1][0], 1e-14);
441         Assert.assertEquals(0.0016, cov[1][1], 0.001);
442     }

516                            final LeastSquaresOptimizer optimizer,
517                            final double errParams,
518                            final double errParamsSd) {
519 
520         final Optimum optimum = optimizer.optimize(builder(dataset).build());
521 
522         final RealVector actual = optimum.getPoint();
523         for (int i = 0; i < actual.getDimension(); i++) {
524             double expected = dataset.getParameter(i);
525             double delta = FastMath.abs(errParams * expected);
526             Assert.assertEquals(dataset.getName() + ", param #" + i,
527                     expected, actual.getEntry(i), delta);
528         }
529     }

541     @Test
542     public void testPointCopy() {
543         LinearProblem problem = new LinearProblem(new double[][]{
544                 {1, 0, 0},
545                 {-1, 1, 0},
546                 {0, -1, 1}
547         }, new double[]{1, 1, 1});
548         //mutable boolean
549         final boolean[] checked = {false};
550 
551         final LeastSquaresBuilder builder = problem.getBuilder()
552                 .checker(new ConvergenceChecker<Evaluation>() {
553                     public boolean converged(int iteration, Evaluation previous, Evaluation current) {
554                         Assert.assertThat(
555                                 previous.getPoint(),
556                                 not(sameInstance(current.getPoint())));
557                         Assert.assertArrayEquals(new double[3], previous.getPoint().toArray(), 0);
558                         Assert.assertArrayEquals(new double[] {1, 2, 3}, current.getPoint().toArray(), TOl);
559                         checked[0] = true;
560                         return true;
561                     }
562                 });
563         optimizer.optimize(builder.build());
564 
565         Assert.assertThat(checked[0], is(true));
566     }

568     class LinearProblem {
569         private final RealMatrix factors;
570         private final double[] target;
571 
572         public LinearProblem(double[][] factors, double[] target) {
573             this.factors = new BlockRealMatrix(factors);
574             this.target = target;
575         }
576 
577         public double[] getTarget() {
578             return target;
579         }
580 
581         public MultivariateVectorFunction getModelFunction() {
582             return new MultivariateVectorFunction() {
583                 public double[] value(double[] params) {
584                     return factors.operate(params);
585                 }
586             };
587         }
588 
589         public MultivariateMatrixFunction getModelFunctionJacobian() {
590             return new MultivariateMatrixFunction() {
591                 public double[][] value(double[] params) {
592                     return factors.getData();
593                 }
594             };
595         }
596 
597         public LeastSquaresBuilder getBuilder() {
598             final double[] weights = new double[target.length];
599             Arrays.fill(weights, 1.0);
600             return base()
601                     .model(getModelFunction(), getModelFunctionJacobian())
602                     .target(target)
603                     .weight(new DiagonalMatrix(weights))
604                     .start(new double[factors.getColumnDimension()]);
605         }
606     }
