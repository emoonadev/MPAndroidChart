17  package org.apache.commons.math4.geometry.spherical.oned;
18  
19  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
20  import org.apache.commons.math4.geometry.spherical.oned.S1Point;
21  import org.apache.commons.math4.geometry.spherical.oned.Sphere1D;
22  import org.apache.commons.math4.util.FastMath;
23  import org.apache.commons.math4.util.MathUtils;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  public class S1PointTest {
28  
29      @Test
30      public void testS1Point() {
31          for (int k = -2; k < 3; ++k) {
32              S1Point p = new S1Point(1.0 + k * MathUtils.TWO_PI);
33              Assert.assertEquals(FastMath.cos(1.0), p.getVector().getX(), 1.0e-10);
34              Assert.assertEquals(FastMath.sin(1.0), p.getVector().getY(), 1.0e-10);
35              Assert.assertFalse(p.isNaN());
36          }
37      }
38  
39      @Test
40      public void testNaN() {
41          Assert.assertTrue(S1Point.NaN.isNaN());
42          Assert.assertTrue(S1Point.NaN.equals(new S1Point(Double.NaN)));
43          Assert.assertFalse(new S1Point(1.0).equals(S1Point.NaN));
44      }
45  
46      @Test
47      public void testEquals() {
48          S1Point a = new S1Point(1.0);
49          S1Point b = new S1Point(1.0);
50          Assert.assertEquals(a.hashCode(), b.hashCode());
51          Assert.assertFalse(a == b);
52          Assert.assertTrue(a.equals(b));
53          Assert.assertTrue(a.equals(a));
54          Assert.assertFalse(a.equals('a'));
55      }
56  
57      @Test
58      public void testDistance() {
59          S1Point a = new S1Point(1.0);
60          S1Point b = new S1Point(a.getAlpha() + 0.5 * FastMath.PI);
61          Assert.assertEquals(0.5 * FastMath.PI, a.distance(b), 1.0e-10);
62      }
63  
64      @Test
65      public void testSpace() {
66          S1Point a = new S1Point(1.0);
67          Assert.assertTrue(a.getSpace() instanceof Sphere1D);
68          Assert.assertEquals(1, a.getSpace().getDimension());
69          try {
70              a.getSpace().getSubSpace();
71              Assert.fail("an exception should have been thrown");
72          } catch (MathUnsupportedOperationException muoe) {
73              // expected
74          }
75      }
76  
77  }
