97      public void testDegenerate0() {
98          BinomialDistribution dist = new BinomialDistribution(5, 0.0d);
99          setDistribution(dist);

101         setCumulativeTestValues(new double[] { 0d, 1d, 1d, 1d, 1d });
102         setDensityTestPoints(new int[] { -1, 0, 1, 10, 11 });
103         setDensityTestValues(new double[] { 0d, 1d, 0d, 0d, 0d });
104         setInverseCumulativeTestPoints(new double[] { 0.1d, 0.5d });
105         setInverseCumulativeTestValues(new int[] { 0, 0 });
106         verifyDensities();
107         verifyCumulativeProbabilities();
108         verifyInverseCumulativeProbabilities();
109         Assert.assertEquals(dist.getSupportLowerBound(), 0);
110         Assert.assertEquals(dist.getSupportUpperBound(), 0);

149     @Test
150     public void testMoments() {
151         final double tol = 1e-9;
152         BinomialDistribution dist;
153 
154         dist = new BinomialDistribution(10, 0.5);
155         Assert.assertEquals(dist.getNumericalMean(), 10d * 0.5d, tol);
156         Assert.assertEquals(dist.getNumericalVariance(), 10d * 0.5d * 0.5d, tol);
157 
158         dist = new BinomialDistribution(30, 0.3);
159         Assert.assertEquals(dist.getNumericalMean(), 30d * 0.3d, tol);
160         Assert.assertEquals(dist.getNumericalVariance(), 30d * 0.3d * (1d - 0.3d), tol);
161     }

163     @Test
164     public void testMath718() {
165         // for large trials the evaluation of ContinuedFraction was inaccurate
166         // do a sweep over several large trials to test if the current implementation is
167         // numerically stable.
168 
169         for (int trials = 500000; trials < 20000000; trials += 100000) {
170             BinomialDistribution dist = new BinomialDistribution(trials, 0.5);
171             int p = dist.inverseCumulativeProbability(0.5);
172             Assert.assertEquals(trials / 2, p);
173         }
174     }
