29  /**
30   * Test class for MonotoneChain.
31   */
32  public class MonotoneChainTest extends ConvexHullGenerator2DAbstractTest {
33  
34      @Override
35      protected ConvexHullGenerator2D createConvexHullGenerator(boolean includeCollinearPoints) {
36          return new MonotoneChain(includeCollinearPoints);
37      }
38  
39      // ------------------------------------------------------------------------------
40  
41      @Test(expected=ConvergenceException.class)
42      public void testConvergenceException() {
43          final Collection<Vector2D> points = new ArrayList<Vector2D>();
44  
45          points.add(new Vector2D(1, 1));
46          points.add(new Vector2D(1, 5));
47          points.add(new Vector2D(0, 7));
48          points.add(new Vector2D(1, 10));
49          points.add(new Vector2D(1, 20));
50          points.add(new Vector2D(20, 20));
51          points.add(new Vector2D(20, 40));
52          points.add(new Vector2D(40, 1));
53  
54          @SuppressWarnings("unused")
55          final ConvexHull2D hull = new MonotoneChain(true, 2).generate(points);
56      }

58  }
