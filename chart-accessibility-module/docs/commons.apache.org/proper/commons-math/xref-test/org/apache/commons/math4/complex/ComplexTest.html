523     @Test
524     public void testFloatingPointEquals() {
525         double re = -3.21;
526         double im = 456789e10;
527 
528         final Complex x = new Complex(re, im);
529         Complex y = new Complex(re, im);
530 
531         Assert.assertTrue(x.equals(y));
532         Assert.assertTrue(Complex.equals(x, y));
533 
534         final int maxUlps = 5;
535         for (int i = 0; i < maxUlps; i++) {
536             re = FastMath.nextUp(re);
537             im = FastMath.nextUp(im);
538         }
539         y = new Complex(re, im);
540         Assert.assertTrue(Complex.equals(x, y, maxUlps));
541 
542         re = FastMath.nextUp(re);
543         im = FastMath.nextUp(im);
544         y = new Complex(re, im);
545         Assert.assertFalse(Complex.equals(x, y, maxUlps));
546     }

631     @Test
632     public void testHashCode() {
633         Complex x = new Complex(0.0, 0.0);
634         Complex y = new Complex(0.0, 0.0 + Double.MIN_VALUE);
635         Assert.assertFalse(x.hashCode()==y.hashCode());
636         y = new Complex(0.0 + Double.MIN_VALUE, 0.0);
637         Assert.assertFalse(x.hashCode()==y.hashCode());
638         Complex realNaN = new Complex(Double.NaN, 0.0);
639         Complex imaginaryNaN = new Complex(0.0, Double.NaN);
640         Assert.assertEquals(realNaN.hashCode(), imaginaryNaN.hashCode());
641         Assert.assertEquals(imaginaryNaN.hashCode(), Complex.NaN.hashCode());
642 
643         // MATH-1118
644         // "equals" and "hashCode" must be compatible: if two objects have
645         // different hash codes, "equals" must return false.
646         final String msg = "'equals' not compatible with 'hashCode'";
647 
648         x = new Complex(0.0, 0.0);
649         y = new Complex(0.0, -0.0);
650         Assert.assertTrue(x.hashCode() != y.hashCode());
651         Assert.assertFalse(msg, x.equals(y));
652 
653         x = new Complex(0.0, 0.0);
654         y = new Complex(-0.0, 0.0);
