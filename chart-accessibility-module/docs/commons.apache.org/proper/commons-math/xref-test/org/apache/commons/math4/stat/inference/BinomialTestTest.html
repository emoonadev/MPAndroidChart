17  package org.apache.commons.math4.stat.inference;
18  
19  import org.apache.commons.math4.exception.MathIllegalArgumentException;
20  import org.apache.commons.math4.exception.NotPositiveException;
21  import org.apache.commons.math4.stat.inference.AlternativeHypothesis;
22  import org.apache.commons.math4.stat.inference.BinomialTest;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for the BinomialTest class.
28   */
29  public class BinomialTestTest {
30  
31      protected BinomialTest testStatistic = new BinomialTest();
32  
33      private static int successes = 51;
34      private static int trials = 235;
35      private static double probability = 1.0 / 6.0;
36  
37      @Test
38      public void testBinomialTestPValues() {
39          Assert.assertEquals(0.04375, testStatistic.binomialTest(
40              trials, successes, probability, AlternativeHypothesis.TWO_SIDED), 1E-4);
41          Assert.assertEquals(0.02654, testStatistic.binomialTest(
42              trials, successes, probability, AlternativeHypothesis.GREATER_THAN), 1E-4);
43          Assert.assertEquals(0.982, testStatistic.binomialTest(
44              trials, successes, probability, AlternativeHypothesis.LESS_THAN), 1E-4);
45      }
46  
47      @Test
48      public void testBinomialTestExceptions() {
49          try {
50              testStatistic.binomialTest(10, -1, 0.5, AlternativeHypothesis.TWO_SIDED);
51              Assert.fail("Expected not positive exception");
52          } catch (NotPositiveException e) {
53              // expected exception;
54          }
55  
56          try {
57              testStatistic.binomialTest(10, 11, 0.5, AlternativeHypothesis.TWO_SIDED);
58              Assert.fail("Expected illegal argument exception");
59          } catch (MathIllegalArgumentException e) {
60              // expected exception;
61          }
62          try {
63              testStatistic.binomialTest(10, 11, 0.5, null);
64              Assert.fail("Expected illegal argument exception");
65          } catch (MathIllegalArgumentException e) {
66              // expected exception;
67          }
68      }
69  
70      @Test
71      public void testBinomialTestAcceptReject() {
72          double alpha05 = 0.05;
73          double alpha01 = 0.01;
74  
75          Assert.assertTrue(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.TWO_SIDED, alpha05));
76          Assert.assertTrue(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.GREATER_THAN, alpha05));
77          Assert.assertFalse(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.LESS_THAN, alpha05));
78  
79          Assert.assertFalse(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.TWO_SIDED, alpha01));
80          Assert.assertFalse(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.GREATER_THAN, alpha01));
81          Assert.assertFalse(testStatistic.binomialTest(trials, successes, probability, AlternativeHypothesis.LESS_THAN, alpha05));
82      }
83  }
