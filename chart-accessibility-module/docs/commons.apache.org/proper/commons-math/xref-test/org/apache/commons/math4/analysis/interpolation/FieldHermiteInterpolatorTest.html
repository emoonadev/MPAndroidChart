34      @Test
35      public void testZero() {
36          FieldHermiteInterpolator<BigFraction> interpolator = new FieldHermiteInterpolator<BigFraction>();
37          interpolator.addSamplePoint(new BigFraction(0), new BigFraction[] { new BigFraction(0) });
38          for (int x = -10; x < 10; x++) {
39              BigFraction y = interpolator.value(new BigFraction(x))[0];
40              Assert.assertEquals(BigFraction.ZERO, y);
41              BigFraction[][] derivatives = interpolator.derivatives(new BigFraction(x), 1);
42              Assert.assertEquals(BigFraction.ZERO, derivatives[0][0]);
43              Assert.assertEquals(BigFraction.ZERO, derivatives[1][0]);
44          }
45      }

47      @Test
48      public void testQuadratic() {
49          FieldHermiteInterpolator<BigFraction> interpolator = new FieldHermiteInterpolator<BigFraction>();
50          interpolator.addSamplePoint(new BigFraction(0), new BigFraction[] { new BigFraction(2) });
51          interpolator.addSamplePoint(new BigFraction(1), new BigFraction[] { new BigFraction(0) });
52          interpolator.addSamplePoint(new BigFraction(2), new BigFraction[] { new BigFraction(0) });
53          for (double x = -10; x < 10; x += 1.0) {
54              BigFraction y = interpolator.value(new BigFraction(x))[0];
55              Assert.assertEquals((x - 1) * (x - 2), y.doubleValue(), 1.0e-15);
56              BigFraction[][] derivatives = interpolator.derivatives(new BigFraction(x), 3);
57              Assert.assertEquals((x - 1) * (x - 2), derivatives[0][0].doubleValue(), 1.0e-15);
58              Assert.assertEquals(2 * x - 3, derivatives[1][0].doubleValue(), 1.0e-15);
59              Assert.assertEquals(2, derivatives[2][0].doubleValue(), 1.0e-15);
60              Assert.assertEquals(0, derivatives[3][0].doubleValue(), 1.0e-15);
61          }
62      }

64      @Test
65      public void testMixedDerivatives() {
66          FieldHermiteInterpolator<BigFraction> interpolator = new FieldHermiteInterpolator<BigFraction>();
67          interpolator.addSamplePoint(new BigFraction(0), new BigFraction[] { new BigFraction(1) }, new BigFraction[] { new BigFraction(2) });
68          interpolator.addSamplePoint(new BigFraction(1), new BigFraction[] { new BigFraction(4) });
69          interpolator.addSamplePoint(new BigFraction(2), new BigFraction[] { new BigFraction(5) }, new BigFraction[] { new BigFraction(2) });
70          BigFraction[][] derivatives = interpolator.derivatives(new BigFraction(0), 5);
71          Assert.assertEquals(new BigFraction(  1), derivatives[0][0]);
72          Assert.assertEquals(new BigFraction(  2), derivatives[1][0]);
73          Assert.assertEquals(new BigFraction(  8), derivatives[2][0]);
74          Assert.assertEquals(new BigFraction(-24), derivatives[3][0]);
75          Assert.assertEquals(new BigFraction( 24), derivatives[4][0]);
76          Assert.assertEquals(new BigFraction(  0), derivatives[5][0]);
77          derivatives = interpolator.derivatives(new BigFraction(1), 5);
78          Assert.assertEquals(new BigFraction(  4), derivatives[0][0]);
79          Assert.assertEquals(new BigFraction(  2), derivatives[1][0]);
80          Assert.assertEquals(new BigFraction( -4), derivatives[2][0]);
81          Assert.assertEquals(new BigFraction(  0), derivatives[3][0]);
82          Assert.assertEquals(new BigFraction( 24), derivatives[4][0]);
83          Assert.assertEquals(new BigFraction(  0), derivatives[5][0]);
84          derivatives = interpolator.derivatives(new BigFraction(2), 5);
85          Assert.assertEquals(new BigFraction(  5), derivatives[0][0]);
86          Assert.assertEquals(new BigFraction(  2), derivatives[1][0]);
87          Assert.assertEquals(new BigFraction(  8), derivatives[2][0]);
88          Assert.assertEquals(new BigFraction( 24), derivatives[3][0]);
89          Assert.assertEquals(new BigFraction( 24), derivatives[4][0]);
90          Assert.assertEquals(new BigFraction(  0), derivatives[5][0]);
91      }

93      @Test
94      public void testRandomPolynomialsValuesOnly() {
95  
96          Random random = new Random(0x42b1e7dbd361a932l);
97  
98          for (int i = 0; i < 100; ++i) {
99  
100             int maxDegree = 0;
101             PolynomialFunction[] p = new PolynomialFunction[5];
102             for (int k = 0; k < p.length; ++k) {
103                 int degree = random.nextInt(7);
104                 p[k] = randomPolynomial(degree, random);
105                 maxDegree = FastMath.max(maxDegree, degree);
106             }
107 
108             DfpField field = new DfpField(30);
109             Dfp step = field.getOne().divide(field.newDfp(10));
110             FieldHermiteInterpolator<Dfp> interpolator = new FieldHermiteInterpolator<Dfp>();
111             for (int j = 0; j < 1 + maxDegree; ++j) {
112                 Dfp x = field.newDfp(j).multiply(step);
113                 Dfp[] values = new Dfp[p.length];
114                 for (int k = 0; k < p.length; ++k) {
115                     values[k] = field.newDfp(p[k].value(x.getReal()));
116                 }
117                 interpolator.addSamplePoint(x, values);
118             }
119 
120             for (int j = 0; j < 20; ++j) {
121                 Dfp x = field.newDfp(j).multiply(step);
122                 Dfp[] values = interpolator.value(x);
123                 Assert.assertEquals(p.length, values.length);
124                 for (int k = 0; k < p.length; ++k) {
125                     Assert.assertEquals(p[k].value(x.getReal()),
126                                         values[k].getReal(),
127                                         1.0e-8 * FastMath.abs(p[k].value(x.getReal())));
128                 }
129             }
130 
131         }
132

135     @Test
136     public void testRandomPolynomialsFirstDerivative() {
137 
138         Random random = new Random(0x570803c982ca5d3bl);
139 
140         for (int i = 0; i < 100; ++i) {
141 
142             int maxDegree = 0;
143             PolynomialFunction[] p      = new PolynomialFunction[5];
144             PolynomialFunction[] pPrime = new PolynomialFunction[5];
145             for (int k = 0; k < p.length; ++k) {
146                 int degree = random.nextInt(7);
147                 p[k]      = randomPolynomial(degree, random);
148                 pPrime[k] = p[k].polynomialDerivative();
149                 maxDegree = FastMath.max(maxDegree, degree);
150             }
151 
152             DfpField field = new DfpField(30);
153             Dfp step = field.getOne().divide(field.newDfp(10));
154             FieldHermiteInterpolator<Dfp> interpolator = new FieldHermiteInterpolator<Dfp>();
155             for (int j = 0; j < 1 + maxDegree / 2; ++j) {
156                 Dfp x = field.newDfp(j).multiply(step);
157                 Dfp[] values      = new Dfp[p.length];
158                 Dfp[] derivatives = new Dfp[p.length];
159                 for (int k = 0; k < p.length; ++k) {
160                     values[k]      = field.newDfp(p[k].value(x.getReal()));
161                     derivatives[k] = field.newDfp(pPrime[k].value(x.getReal()));
162                 }
163                 interpolator.addSamplePoint(x, values, derivatives);
164             }
165 
166             Dfp h = step.divide(field.newDfp(100000));
167             for (int j = 0; j < 20; ++j) {
168                 Dfp x = field.newDfp(j).multiply(step);
169                 Dfp[] y  = interpolator.value(x);
170                 Dfp[] yP = interpolator.value(x.add(h));
171                 Dfp[] yM = interpolator.value(x.subtract(h));
172                 Assert.assertEquals(p.length, y.length);
173                 for (int k = 0; k < p.length; ++k) {
174                     Assert.assertEquals(p[k].value(x.getReal()),
175                                         y[k].getReal(),
176                                         1.0e-8 * FastMath.abs(p[k].value(x.getReal())));
177                     Assert.assertEquals(pPrime[k].value(x.getReal()),
178                                         yP[k].subtract(yM[k]).divide(h.multiply(2)).getReal(),
179                                         4.0e-8 * FastMath.abs(p[k].value(x.getReal())));
180                 }
181             }
182 
183         }
184     }

186     @Test
187     public void testSine() {
188         DfpField field = new DfpField(30);
189         FieldHermiteInterpolator<Dfp> interpolator = new FieldHermiteInterpolator<Dfp>();
190         for (Dfp x = field.getZero(); x.getReal() < FastMath.PI; x = x.add(0.5)) {
191             interpolator.addSamplePoint(x, new Dfp[] { x.sin() });
192         }
193         for (Dfp x = field.newDfp(0.1); x.getReal() < 2.9; x = x.add(0.01)) {
194             Dfp y = interpolator.value(x)[0];
195             Assert.assertEquals( x.sin().getReal(), y.getReal(), 3.5e-5);
196         }
197     }

212     @Test
213     public void testWikipedia() {
214         // this test corresponds to the example from Wikipedia page:
215         // http://en.wikipedia.org/wiki/Hermite_interpolation
216         FieldHermiteInterpolator<BigFraction> interpolator = new FieldHermiteInterpolator<BigFraction>();
217         interpolator.addSamplePoint(new BigFraction(-1),
218                                     new BigFraction[] { new BigFraction( 2) },
219                                     new BigFraction[] { new BigFraction(-8) },
220                                     new BigFraction[] { new BigFraction(56) });
221         interpolator.addSamplePoint(new BigFraction( 0),
222                                     new BigFraction[] { new BigFraction( 1) },
223                                     new BigFraction[] { new BigFraction( 0) },
224                                     new BigFraction[] { new BigFraction( 0) });
225         interpolator.addSamplePoint(new BigFraction( 1),
226                                     new BigFraction[] { new BigFraction( 2) },
227                                     new BigFraction[] { new BigFraction( 8) },
228                                     new BigFraction[] { new BigFraction(56) });
229         for (BigFraction x = new BigFraction(-1); x.doubleValue() <= 1.0; x = x.add(new BigFraction(1, 8))) {
230             BigFraction y = interpolator.value(x)[0];
231             BigFraction x2 = x.multiply(x);
232             BigFraction x4 = x2.multiply(x2);
233             BigFraction x8 = x4.multiply(x4);
234             Assert.assertEquals(x8.add(new BigFraction(1)), y);
235         }
236     }

238     @Test
239     public void testOnePointParabola() {
240         FieldHermiteInterpolator<BigFraction> interpolator = new FieldHermiteInterpolator<BigFraction>();
241         interpolator.addSamplePoint(new BigFraction(0),
242                                     new BigFraction[] { new BigFraction(1) },
243                                     new BigFraction[] { new BigFraction(1) },
244                                     new BigFraction[] { new BigFraction(2) });
245         for (BigFraction x = new BigFraction(-1); x.doubleValue() <= 1.0; x = x.add(new BigFraction(1, 8))) {
246             BigFraction y = interpolator.value(x)[0];
247             Assert.assertEquals(BigFraction.ONE.add(x.multiply(BigFraction.ONE.add(x))), y);
248         }
249     }

251     private PolynomialFunction randomPolynomial(int degree, Random random) {
252         double[] coeff = new double[ 1 + degree];
253         for (int j = 0; j < degree; ++j) {
254             coeff[j] = random.nextDouble();
255         }
256         return new PolynomialFunction(coeff);
257     }
