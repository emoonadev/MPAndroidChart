17  package org.apache.commons.math4.analysis;
18  
19  import org.apache.commons.math4.analysis.differentiation.DerivativeStructure;
20  import org.apache.commons.math4.analysis.differentiation.UnivariateDifferentiableFunction;
21  
22  /**
23   * Auxiliary class for testing solvers.
24   *
25   */
26  public class XMinus5Function implements UnivariateDifferentiableFunction {
27  
28      /* Evaluate x - 5 function.
29       * @see org.apache.commons.math4.UnivariateFunction#value(double)
30       */
31      @Override
32      public double value(double x) {
33          return x - 5;
34      }
35  
36      @Override
37      public DerivativeStructure value(DerivativeStructure t) {
38          return t.subtract(5);
39      }
40  
41  }
