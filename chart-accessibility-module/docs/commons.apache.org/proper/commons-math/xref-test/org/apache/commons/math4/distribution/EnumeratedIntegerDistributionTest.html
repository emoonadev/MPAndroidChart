38      /**
39       * The distribution object used for testing.
40       */
41      private final EnumeratedIntegerDistribution testDistribution;

43      /**
44       * Creates the default distribution object used for testing.
45       */
46      public EnumeratedIntegerDistributionTest() {
47          // Non-sorted singleton array with duplicates should be allowed.
48          // Values with zero-probability do not extend the support.
49          testDistribution = new EnumeratedIntegerDistribution(
50                  new int[]{3, -1, 3, 7, -2, 8},
51                  new double[]{0.2, 0.2, 0.3, 0.3, 0.0, 0.0});
52      }

54      /**
55       * Tests if the EnumeratedIntegerDistribution constructor throws
56       * exceptions for invalid data.
57       */
58      @Test
59      public void testExceptions() {
60          EnumeratedIntegerDistribution invalid = null;
61          try {
62              new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0});
63              Assert.fail("Expected DimensionMismatchException");
64          } catch (DimensionMismatchException e) {
65          }
66          try {
67              new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, -1.0});
68              Assert.fail("Expected NotPositiveException");
69          } catch (NotPositiveException e) {
70          }
71          try {
72              new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, 0.0});
73              Assert.fail("Expected MathArithmeticException");
74          } catch (MathArithmeticException e) {
75          }
76          try {
77            new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, Double.NaN});
78              Assert.fail("Expected NotANumberException");
79          } catch (NotANumberException e) {
80          }
81          try {
82          new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, Double.POSITIVE_INFINITY});
83              Assert.fail("Expected NotFiniteNumberException");
84          } catch (NotFiniteNumberException e) {
85          }
86          Assert.assertNull("Expected non-initialized DiscreteRealDistribution", invalid);
87      }
88  
89      /**
90       * Tests if the distribution returns proper probability values.
91       */
92      @Test
93      public void testProbability() {
94          int[] points = new int[]{-2, -1, 0, 1, 2, 3, 4, 5, 6, 7, 8};
95          double[] results = new double[]{0, 0.2, 0, 0, 0, 0.5, 0, 0, 0, 0.3, 0};
96          for (int p = 0; p < points.length; p++) {
97              double probability = testDistribution.probability(points[p]);
98              Assert.assertEquals(results[p], probability, 0.0);
99          }
100     }
101 
102     /**
103      * Tests if the distribution returns proper cumulative probability values.
104      */
105     @Test
106     public void testCumulativeProbability() {
107         int[] points = new int[]{-2, -1, 0, 1, 2, 3, 4, 5, 6, 7, 8};
108         double[] results = new double[]{0, 0.2, 0.2, 0.2, 0.2, 0.7, 0.7, 0.7, 0.7, 1.0, 1.0};
109         for (int p = 0; p < points.length; p++) {
110             double probability = testDistribution.cumulativeProbability(points[p]);
111             Assert.assertEquals(results[p], probability, 1e-10);
112         }
113     }

155     /**
156      * Tests sampling.
157      */
158     @Test
159     public void testSample() {
160         final int n = 1000000;
161         final IntegerDistribution.Sampler sampler
162             = testDistribution.createSampler(RandomSource.create(RandomSource.WELL_19937_C,
163                                                                  -334759360)); // fixed seed
164         final int[] samples = AbstractIntegerDistribution.sample(n, sampler);
165         Assert.assertEquals(n, samples.length);
166         double sum = 0;
167         double sumOfSquares = 0;
168         for (int i = 0; i < samples.length; i++) {
169             sum += samples[i];
170             sumOfSquares += samples[i] * samples[i];
171         }
172         Assert.assertEquals(testDistribution.getNumericalMean(),
173                 sum / n, 1e-2);
174         Assert.assertEquals(testDistribution.getNumericalVariance(),
175                 sumOfSquares / n - FastMath.pow(sum / n, 2), 1e-2);
176     }

178     @Test
179     public void testCreateFromIntegers() {
180         final int[] data = new int[] {0, 1, 1, 2, 2, 2};
181         EnumeratedIntegerDistribution distribution = new EnumeratedIntegerDistribution(data);
182         assertEquals(0.5, distribution.probability(2), 0);
183         assertEquals(0.5, distribution.cumulativeProbability(1), 0);
184     }
