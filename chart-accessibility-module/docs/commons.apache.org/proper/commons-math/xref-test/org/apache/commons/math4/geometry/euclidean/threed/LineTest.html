29      @Test
30      public void testContains() throws MathIllegalArgumentException, MathArithmeticException {
31          Vector3D p1 = new Vector3D(0, 0, 1);
32          Line l = new Line(p1, new Vector3D(0, 0, 2), 1.0e-10);
33          Assert.assertTrue(l.contains(p1));
34          Assert.assertTrue(l.contains(new Vector3D(1.0, p1, 0.3, l.getDirection())));
35          Vector3D u = l.getDirection().orthogonal();
36          Vector3D v = Vector3D.crossProduct(l.getDirection(), u);
37          for (double alpha = 0; alpha < 2 * FastMath.PI; alpha += 0.3) {
38              Assert.assertTrue(! l.contains(p1.add(new Vector3D(FastMath.cos(alpha), u,
39                                                                 FastMath.sin(alpha), v))));
40          }
41      }

43      @Test
44      public void testSimilar() throws MathIllegalArgumentException, MathArithmeticException {
45          Vector3D p1  = new Vector3D (1.2, 3.4, -5.8);
46          Vector3D p2  = new Vector3D (3.4, -5.8, 1.2);
47          Line     lA  = new Line(p1, p2, 1.0e-10);
48          Line     lB  = new Line(p2, p1, 1.0e-10);
49          Assert.assertTrue(lA.isSimilarTo(lB));
50          Assert.assertTrue(! lA.isSimilarTo(new Line(p1, p1.add(lA.getDirection().orthogonal()), 1.0e-10)));
51      }

53      @Test
54      public void testPointDistance() throws MathIllegalArgumentException {
55          Line l = new Line(new Vector3D(0, 1, 1), new Vector3D(0, 2, 2), 1.0e-10);
56          Assert.assertEquals(FastMath.sqrt(3.0 / 2.0), l.distance(new Vector3D(1, 0, 1)), 1.0e-10);
57          Assert.assertEquals(0, l.distance(new Vector3D(0, -4, -4)), 1.0e-10);

60      @Test
61      public void testLineDistance() throws MathIllegalArgumentException {
62          Line l = new Line(new Vector3D(0, 1, 1), new Vector3D(0, 2, 2), 1.0e-10);
63          Assert.assertEquals(1.0,
64                              l.distance(new Line(new Vector3D(1, 0, 1), new Vector3D(1, 0, 2), 1.0e-10)),
65                              1.0e-10);
66          Assert.assertEquals(0.5,
67                              l.distance(new Line(new Vector3D(-0.5, 0, 0), new Vector3D(-0.5, -1, -1), 1.0e-10)),
68                              1.0e-10);
69          Assert.assertEquals(0.0,
70                              l.distance(l),
71                              1.0e-10);
72          Assert.assertEquals(0.0,
73                              l.distance(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -5, -5), 1.0e-10)),
74                              1.0e-10);
75          Assert.assertEquals(0.0,
76                              l.distance(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -3, -4), 1.0e-10)),
77                              1.0e-10);
78          Assert.assertEquals(0.0,
79                              l.distance(new Line(new Vector3D(0, -4, -4), new Vector3D(1, -4, -4), 1.0e-10)),
80                              1.0e-10);
81          Assert.assertEquals(FastMath.sqrt(8),
82                              l.distance(new Line(new Vector3D(0, -4, 0), new Vector3D(1, -4, 0), 1.0e-10)),
83                              1.0e-10);
84      }

86      @Test
87      public void testClosest() throws MathIllegalArgumentException {
88          Line l = new Line(new Vector3D(0, 1, 1), new Vector3D(0, 2, 2), 1.0e-10);
89          Assert.assertEquals(0.0,
90                              l.closestPoint(new Line(new Vector3D(1, 0, 1), new Vector3D(1, 0, 2), 1.0e-10)).distance(new Vector3D(0, 0, 0)),
91                              1.0e-10);
92          Assert.assertEquals(0.5,
93                              l.closestPoint(new Line(new Vector3D(-0.5, 0, 0), new Vector3D(-0.5, -1, -1), 1.0e-10)).distance(new Vector3D(-0.5, 0, 0)),
94                              1.0e-10);
95          Assert.assertEquals(0.0,
96                              l.closestPoint(l).distance(new Vector3D(0, 0, 0)),
97                              1.0e-10);
98          Assert.assertEquals(0.0,
99                              l.closestPoint(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -5, -5), 1.0e-10)).distance(new Vector3D(0, 0, 0)),
100                             1.0e-10);
101         Assert.assertEquals(0.0,
102                             l.closestPoint(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -3, -4), 1.0e-10)).distance(new Vector3D(0, -4, -4)),
103                             1.0e-10);
104         Assert.assertEquals(0.0,
105                             l.closestPoint(new Line(new Vector3D(0, -4, -4), new Vector3D(1, -4, -4), 1.0e-10)).distance(new Vector3D(0, -4, -4)),
106                             1.0e-10);
107         Assert.assertEquals(0.0,
108                             l.closestPoint(new Line(new Vector3D(0, -4, 0), new Vector3D(1, -4, 0), 1.0e-10)).distance(new Vector3D(0, -2, -2)),
109                             1.0e-10);

112     @Test
113     public void testIntersection() throws MathIllegalArgumentException {
114         Line l = new Line(new Vector3D(0, 1, 1), new Vector3D(0, 2, 2), 1.0e-10);
115         Assert.assertNull(l.intersection(new Line(new Vector3D(1, 0, 1), new Vector3D(1, 0, 2), 1.0e-10)));
116         Assert.assertNull(l.intersection(new Line(new Vector3D(-0.5, 0, 0), new Vector3D(-0.5, -1, -1), 1.0e-10)));
117         Assert.assertEquals(0.0,
118                             l.intersection(l).distance(new Vector3D(0, 0, 0)),
119                             1.0e-10);
120         Assert.assertEquals(0.0,
121                             l.intersection(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -5, -5), 1.0e-10)).distance(new Vector3D(0, 0, 0)),
122                             1.0e-10);
123         Assert.assertEquals(0.0,
124                             l.intersection(new Line(new Vector3D(0, -4, -4), new Vector3D(0, -3, -4), 1.0e-10)).distance(new Vector3D(0, -4, -4)),
125                             1.0e-10);
126         Assert.assertEquals(0.0,
127                             l.intersection(new Line(new Vector3D(0, -4, -4), new Vector3D(1, -4, -4), 1.0e-10)).distance(new Vector3D(0, -4, -4)),
128                             1.0e-10);
129         Assert.assertNull(l.intersection(new Line(new Vector3D(0, -4, 0), new Vector3D(1, -4, 0), 1.0e-10)));
130     }

134

137                              new Vector3D(1650757.5050732433, 6160710.879908984, 0.9),

140 
141         // action
142         Line reverted = line.revert();

144         // verify
