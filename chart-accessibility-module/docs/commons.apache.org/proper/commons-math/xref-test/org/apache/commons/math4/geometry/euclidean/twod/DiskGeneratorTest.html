36      @Test
37      public void testSupport0Point() {
38          List<Vector2D> support = Arrays.asList(new Vector2D[0]);
39          EnclosingBall<Euclidean2D, Vector2D> disk = new DiskGenerator().ballOnSupport(support);
40          Assert.assertTrue(disk.getRadius() < 0);
41          Assert.assertEquals(0, disk.getSupportSize());
42          Assert.assertEquals(0, disk.getSupport().length);
43      }
44  
45      @Test
46      public void testSupport1Point() {
47          List<Vector2D> support = Arrays.asList(new Vector2D(1, 2));
48          EnclosingBall<Euclidean2D, Vector2D> disk = new DiskGenerator().ballOnSupport(support);
49          Assert.assertEquals(0.0, disk.getRadius(), 1.0e-10);
50          Assert.assertTrue(disk.contains(support.get(0)));
51          Assert.assertTrue(disk.contains(support.get(0), 0.5));
52          Assert.assertFalse(disk.contains(new Vector2D(support.get(0).getX() + 0.1,
53                                                        support.get(0).getY() - 0.1),
54                                           0.001));
55          Assert.assertTrue(disk.contains(new Vector2D(support.get(0).getX() + 0.1,
56                                                       support.get(0).getY() - 0.1),
57                                          0.5));
58          Assert.assertEquals(0, support.get(0).distance(disk.getCenter()), 1.0e-10);
59          Assert.assertEquals(1, disk.getSupportSize());
60          Assert.assertTrue(support.get(0) == disk.getSupport()[0]);
61      }
62  
63      @Test
64      public void testSupport2Points() {
65          List<Vector2D> support = Arrays.asList(new Vector2D(1, 0),
66                                                 new Vector2D(3, 0));
67          EnclosingBall<Euclidean2D, Vector2D> disk = new DiskGenerator().ballOnSupport(support);
68          Assert.assertEquals(1.0, disk.getRadius(), 1.0e-10);
69          int i = 0;
70          for (Vector2D v : support) {
71              Assert.assertTrue(disk.contains(v));
72              Assert.assertEquals(1.0, v.distance(disk.getCenter()), 1.0e-10);
73              Assert.assertTrue(v == disk.getSupport()[i++]);
74          }
75          Assert.assertTrue(disk.contains(new Vector2D(2, 0.9)));
76          Assert.assertFalse(disk.contains(Vector2D.ZERO));
77          Assert.assertEquals(0.0, new Vector2D(2, 0).distance(disk.getCenter()), 1.0e-10);
78          Assert.assertEquals(2, disk.getSupportSize());
79      }
80  
81      @Test
82      public void testSupport3Points() {
83          List<Vector2D> support = Arrays.asList(new Vector2D(1, 0),
84                                                 new Vector2D(3, 0),
85                                                 new Vector2D(2, 2));
86          EnclosingBall<Euclidean2D, Vector2D> disk = new DiskGenerator().ballOnSupport(support);
87          Assert.assertEquals(5.0 / 4.0, disk.getRadius(), 1.0e-10);
88          int i = 0;
89          for (Vector2D v : support) {
90              Assert.assertTrue(disk.contains(v));
91              Assert.assertEquals(5.0 / 4.0, v.distance(disk.getCenter()), 1.0e-10);
92              Assert.assertTrue(v == disk.getSupport()[i++]);
93          }
94          Assert.assertTrue(disk.contains(new Vector2D(2, 0.9)));
95          Assert.assertFalse(disk.contains(new Vector2D(0.9,  0)));
96          Assert.assertFalse(disk.contains(new Vector2D(3.1,  0)));
97          Assert.assertTrue(disk.contains(new Vector2D(2.0, -0.499)));
98          Assert.assertFalse(disk.contains(new Vector2D(2.0, -0.501)));
99          Assert.assertEquals(0.0, new Vector2D(2.0, 3.0 / 4.0).distance(disk.getCenter()), 1.0e-10);
100         Assert.assertEquals(3, disk.getSupportSize());
101     }

103     @Test
104     public void testRandom() {
105         final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
106                                                                  0x12faa818373ffe90l);
107         final UnitSphereRandomVectorGenerator sr = new UnitSphereRandomVectorGenerator(2, random);
108         for (int i = 0; i < 500; ++i) {
109             double d = 25 * random.nextDouble();
110             double refRadius = 10 * random.nextDouble();
111             Vector2D refCenter = new Vector2D(d, new Vector2D(sr.nextVector()));
112             List<Vector2D> support = new ArrayList<Vector2D>();
113             for (int j = 0; j < 3; ++j) {
114                 support.add(new Vector2D(1.0, refCenter, refRadius, new Vector2D(sr.nextVector())));
115             }
116             EnclosingBall<Euclidean2D, Vector2D> disk = new DiskGenerator().ballOnSupport(support);
117             Assert.assertEquals(0.0, refCenter.distance(disk.getCenter()), 3e-9 * refRadius);
118             Assert.assertEquals(refRadius, disk.getRadius(), 7e-10 * refRadius);
119         }
