17  package org.apache.commons.math4.util;
18  
19  import org.apache.commons.math4.stat.StatUtils;
20  import org.apache.commons.math4.util.DoubleArray;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  
25  /**
26   * This class contains test cases for the ExpandableDoubleArray.
27   *
28   */
29  public abstract class DoubleArrayAbstractTest {
30  
31      protected DoubleArray da = null;
32  
33      // Array used to test rolling
34      protected DoubleArray ra = null;
35  
36      @Test
37      public void testAdd1000() {
38  
39          for (int i = 0; i < 1000; i++) {
40              da.addElement(i);
41          }
42  
43          Assert.assertEquals(
44              "Number of elements should be equal to 1000 after adding 1000 values",
45              1000,
46              da.getNumElements());
47  
48          Assert.assertEquals(
49              "The element at the 56th index should be 56",
50              56.0,
51              da.getElement(56),
52              Double.MIN_VALUE);
53  
54      }
55  
56      @Test
57      public void testGetValues() {
58          double[] controlArray = { 2.0, 4.0, 6.0 };
59  
60          da.addElement(2.0);
61          da.addElement(4.0);
62          da.addElement(6.0);
63          double[] testArray = da.getElements();
64  
65          for (int i = 0; i < da.getNumElements(); i++) {
66              Assert.assertEquals(
67                  "The testArray values should equal the controlArray values, index i: "
68                      + i
69                      + " does not match",
70                  testArray[i],
71                  controlArray[i],
72                  Double.MIN_VALUE);
73          }
74  
75      }
76  
77      @Test
78      public void testAddElementRolling() {
79          ra.addElement(0.5);
80          ra.addElement(1.0);
81          ra.addElement(1.0);
82          ra.addElement(1.0);
83          ra.addElement(1.0);
84          ra.addElement(1.0);
85          ra.addElementRolling(2.0);
86  
87          Assert.assertEquals(
88              "There should be 6 elements in the eda",
89              6,
90              ra.getNumElements());
91          Assert.assertEquals(
92              "The max element should be 2.0",
93              2.0,
94              StatUtils.max(ra.getElements()),
95              Double.MIN_VALUE);
96          Assert.assertEquals(
97              "The min element should be 1.0",
98              1.0,
99              StatUtils.min(ra.getElements()),
100             Double.MIN_VALUE);
101 
102         for (int i = 0; i < 1024; i++) {
103             ra.addElementRolling(i);
104         }
105 
106         Assert.assertEquals(
107             "We just inserted 1024 rolling elements, num elements should still be 6",
108             6,
109             ra.getNumElements());
110     }
111 
112     @Test
113     public void testMinMax() {
114         da.addElement(2.0);
115         da.addElement(22.0);
116         da.addElement(-2.0);
117         da.addElement(21.0);
118         da.addElement(22.0);
119         da.addElement(42.0);
120         da.addElement(62.0);
121         da.addElement(22.0);
122         da.addElement(122.0);
123         da.addElement(1212.0);
124 
125         Assert.assertEquals("Min should be -2.0", -2.0, StatUtils.min(da.getElements()), Double.MIN_VALUE);
126         Assert.assertEquals(
127             "Max should be 1212.0",
128             1212.0,
129             StatUtils.max(da.getElements()),
130             Double.MIN_VALUE);
131     }
132 
133 }
