111     @Override
112     @Test
113     public void testConsistency() {
114         RealMatrix matrix = createRealMatrix(longleyData, 16, 7);
115         SpearmansCorrelation corrInstance = new SpearmansCorrelation(matrix);
116         double[][] data = matrix.getData();
117         double[] x = matrix.getColumn(0);
118         double[] y = matrix.getColumn(1);
119         Assert.assertEquals(new SpearmansCorrelation().correlation(x, y),
120                 corrInstance.getCorrelationMatrix().getEntry(0, 1), Double.MIN_VALUE);
121         TestUtils.assertEquals("Correlation matrix", corrInstance.getCorrelationMatrix(),
122                 new SpearmansCorrelation().computeCorrelationMatrix(data), Double.MIN_VALUE);
123     }

125     @Test(expected = MathIllegalArgumentException.class)
126     public void testMath891Array() {
127         // NaNStrategy.REMOVED is not supported since 4.0
128         final double[] xArray = new double[] { Double.NaN, 1.9, 2, 100, 3 };
129         final double[] yArray = new double[] { 10, 2, 10, Double.NaN, 4 };
130 
131         NaturalRanking ranking = new NaturalRanking(NaNStrategy.REMOVED);
132         SpearmansCorrelation spearman = new SpearmansCorrelation(ranking);
133 
134         Assert.assertEquals(0.5, spearman.correlation(xArray, yArray), Double.MIN_VALUE);
135     }
136 
137     @Test(expected = MathIllegalArgumentException.class)
138     public void testMath891Matrix() {
139         // NaNStrategy.REMOVED is not supported since 4.0
140         final double[] xArray = new double[] { Double.NaN, 1.9, 2, 100, 3 };
141         final double[] yArray = new double[] { 10, 2, 10, Double.NaN, 4 };
142 
143         RealMatrix matrix = MatrixUtils.createRealMatrix(xArray.length, 2);
144         for (int i = 0; i < xArray.length; i++) {
145             matrix.addToEntry(i, 0, xArray[i]);
146             matrix.addToEntry(i, 1, yArray[i]);
147         }
148 
149         // compute correlation
150         NaturalRanking ranking = new NaturalRanking(NaNStrategy.REMOVED);
151         SpearmansCorrelation spearman = new SpearmansCorrelation(matrix, ranking);
152 
153         Assert.assertEquals(0.5, spearman.getCorrelationMatrix().getEntry(0, 1), Double.MIN_VALUE);
154     }
