33   * Test for {@link UnivariatePeriodicInterpolator}.

35  public class UnivariatePeriodicInterpolatorTest {
36      private final Random rng = new Random(1224465L);
37  
38      @Test
39      public void testSine() {
40          final int n = 30;
41          final double[] xval = new double[n];
42          final double[] yval = new double[n];
43          final double period = 12.3;
44          final double offset = 45.67;
45  
46          double delta = 0;
47          for (int i = 0; i < n; i++) {
48              delta += rng.nextDouble() * period / n;
49              xval[i] = offset + delta;
50              yval[i] = FastMath.sin(xval[i]);
51          }
52  
53          final UnivariateInterpolator inter = new LinearInterpolator();
54          final UnivariateFunction f = inter.interpolate(xval, yval);
55  
56          final UnivariateInterpolator interP
57              = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
58                                                       period, 1);
59          final UnivariateFunction fP = interP.interpolate(xval, yval);
60  
61          // Comparing with original interpolation algorithm.
62          final double xMin = xval[0];
63          final double xMax = xval[n - 1];
64          for (int i = 0; i < n; i++) {
65              final double x = xMin + (xMax - xMin) * rng.nextDouble();
66              final double y = f.value(x);
67              final double yP = fP.value(x);
68  
69              Assert.assertEquals("x=" + x, y, yP, Math.ulp(1d));
70          }
71  
72          // Test interpolation outside the primary interval.
73          for (int i = 0; i < n; i++) {
74              final double xIn = offset + rng.nextDouble() * period;
75              final double xOut = xIn + rng.nextInt(123456789) * period;
76              final double yIn = fP.value(xIn);
77              final double yOut = fP.value(xOut);
78  
79              Assert.assertEquals(yIn, yOut, 1e-7);
80          }
81      }

83      @Test
84      public void testLessThanOnePeriodCoverage() {
85          final int n = 30;
86          final double[] xval = new double[n];
87          final double[] yval = new double[n];
88          final double period = 12.3;
89          final double offset = 45.67;
90  
91          double delta = period / 2;
92          for (int i = 0; i < n; i++) {
93              delta += period / (2 * n) * rng.nextDouble();
94              xval[i] = offset + delta;
95              yval[i] = FastMath.sin(xval[i]);
96          }
97  
98          final UnivariateInterpolator interP
99              = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
100                                                      period, 1);
101         final UnivariateFunction fP = interP.interpolate(xval, yval);
102 
103         // Test interpolation outside the sample data interval.
104         for (int i = 0; i < n; i++) {
105             final double xIn = offset + rng.nextDouble() * period;
106             final double xOut = xIn + rng.nextInt(123456789) * period;
107             final double yIn = fP.value(xIn);
108             final double yOut = fP.value(xOut);
109 
110             Assert.assertEquals(yIn, yOut, 1e-7);
111         }
112     }

114     @Test
115     public void testMoreThanOnePeriodCoverage() {
116         final int n = 30;
117         final double[] xval = new double[n];
118         final double[] yval = new double[n];
119         final double period = 12.3;
120         final double offset = 45.67;
121 
122         double delta = period / 2;
123         for (int i = 0; i < n; i++) {
124             delta += 10 * period / n * rng.nextDouble();
125             xval[i] = offset + delta;
126             yval[i] = FastMath.sin(xval[i]);
127         }
128 
129         final UnivariateInterpolator interP
130             = new UnivariatePeriodicInterpolator(new LinearInterpolator(),
131                                                      period, 1);
132         final UnivariateFunction fP = interP.interpolate(xval, yval);
133 
134         // Test interpolation outside the sample data interval.
135         for (int i = 0; i < n; i++) {
136             final double xIn = offset + rng.nextDouble() * period;
137             final double xOut = xIn + rng.nextInt(123456789) * period;
138             final double yIn = fP.value(xIn);
139             final double yOut = fP.value(xOut);
140
