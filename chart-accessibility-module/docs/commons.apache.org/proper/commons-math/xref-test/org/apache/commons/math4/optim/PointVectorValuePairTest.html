18  package org.apache.commons.math4.optim;
19  
20  import org.apache.commons.math4.TestUtils;
21  import org.apache.commons.math4.optim.PointVectorValuePair;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  public class PointVectorValuePairTest {
26      @Test
27      public void testSerial() {
28          PointVectorValuePair pv1 = new PointVectorValuePair(new double[] { 1.0, 2.0, 3.0 },
29                                                              new double[] { 4.0, 5.0 });
30          PointVectorValuePair pv2 = (PointVectorValuePair) TestUtils.serializeAndRecover(pv1);
31          Assert.assertEquals(pv1.getKey().length, pv2.getKey().length);
32          for (int i = 0; i < pv1.getKey().length; ++i) {
33              Assert.assertEquals(pv1.getKey()[i], pv2.getKey()[i], 1.0e-15);
34          }
35          Assert.assertEquals(pv1.getValue().length, pv2.getValue().length);
36          for (int i = 0; i < pv1.getValue().length; ++i) {
37              Assert.assertEquals(pv1.getValue()[i], pv2.getValue()[i], 1.0e-15);
38          }
39      }
40  }
