37      private final double[] exampleData = { 20, 17, 30, 42.3, 17, 50,
38              Double.NaN, Double.NEGATIVE_INFINITY, 17 };
39      private final double[] tiesFirst = { 0, 0, 2, 1, 4 };
40      private final double[] tiesLast = { 4, 4, 1, 0 };
41      private final double[] multipleNaNs = { 0, 1, Double.NaN, Double.NaN };
42      private final double[] multipleTies = { 3, 2, 5, 5, 6, 6, 1 };
43      private final double[] allSame = { 0, 0, 0, 0 };
44  
45      @Test
46      public void testDefault() { // Ties averaged, NaNs failed
47          NaturalRanking ranking = new NaturalRanking();
48          double[] ranks;
49  
50          try {
51              ranks = ranking.rank(exampleData);
52              Assert.fail("expected NotANumberException due to NaNStrategy.FAILED");
53          } catch (NotANumberException e) {
54              // expected
55          }
56  
57          ranks = ranking.rank(tiesFirst);
58          double[] correctRanks = new double[] { 1.5, 1.5, 4, 3, 5 };
59          TestUtils.assertEquals(correctRanks, ranks, 0d);
60          ranks = ranking.rank(tiesLast);
61          correctRanks = new double[] { 3.5, 3.5, 2, 1 };
62          TestUtils.assertEquals(correctRanks, ranks, 0d);
63  
64          try {
65              ranks = ranking.rank(multipleNaNs);
66              Assert.fail("expected NotANumberException due to NaNStrategy.FAILED");
67          } catch (NotANumberException e) {
68              // expected
69          }
70  
71          ranks = ranking.rank(multipleTies);
72          correctRanks = new double[] { 3, 2, 4.5, 4.5, 6.5, 6.5, 1 };
73          TestUtils.assertEquals(correctRanks, ranks, 0d);
74          ranks = ranking.rank(allSame);
75          correctRanks = new double[] { 2.5, 2.5, 2.5, 2.5 };
76          TestUtils.assertEquals(correctRanks, ranks, 0d);
77      }

79      @Test
80      public void testNaNsMaximalTiesMinimum() {
81          NaturalRanking ranking = new NaturalRanking(NaNStrategy.MAXIMAL, TiesStrategy.MINIMUM);
82          double[] ranks = ranking.rank(exampleData);
83          double[] correctRanks = { 5, 2, 6, 7, 2, 8, 9, 1, 2 };
84          TestUtils.assertEquals(correctRanks, ranks, 0d);
85          ranks = ranking.rank(tiesFirst);
86          correctRanks = new double[] { 1, 1, 4, 3, 5 };
87          TestUtils.assertEquals(correctRanks, ranks, 0d);
88          ranks = ranking.rank(tiesLast);
89          correctRanks = new double[] { 3, 3, 2, 1 };
90          TestUtils.assertEquals(correctRanks, ranks, 0d);
91          ranks = ranking.rank(multipleNaNs);
92          correctRanks = new double[] { 1, 2, 3, 3 };
93          TestUtils.assertEquals(correctRanks, ranks, 0d);
94          ranks = ranking.rank(multipleTies);
95          correctRanks = new double[] { 3, 2, 4, 4, 6, 6, 1 };
96          TestUtils.assertEquals(correctRanks, ranks, 0d);
97          ranks = ranking.rank(allSame);
98          correctRanks = new double[] { 1, 1, 1, 1 };
99          TestUtils.assertEquals(correctRanks, ranks, 0d);
100     }
