32      private double[][] testSquare = {
33              { 24.0 / 25.0, 43.0 / 25.0 },
34              { 57.0 / 25.0, 24.0 / 25.0 }
35      };
36      private double[][] bigSingular = {
37          { 1.0, 2.0,   3.0,    4.0 },
38          { 2.0, 5.0,   3.0,    4.0 },
39          { 7.0, 3.0, 256.0, 1930.0 },
40          { 3.0, 7.0,   6.0,    8.0 }

45      /** test solve dimension errors */
46      @Test
47      public void testSolveDimensionErrors() {
48          DecompositionSolver solver =
49              new SingularValueDecomposition(MatrixUtils.createRealMatrix(testSquare)).getSolver();
50          RealMatrix b = MatrixUtils.createRealMatrix(new double[3][2]);
51          try {
52              solver.solve(b);
53              Assert.fail("an exception should have been thrown");
54          } catch (MathIllegalArgumentException iae) {
55              // expected behavior
56          }
57          try {
58              solver.solve(b.getColumnVector(0));
59              Assert.fail("an exception should have been thrown");
60          } catch (MathIllegalArgumentException iae) {
61              // expected behavior
62          }
63          try {
64              solver.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
65              Assert.fail("an exception should have been thrown");
66          } catch (MathIllegalArgumentException iae) {
67              // expected behavior
68          }
69      }

71      /** test least square solve */
72      @Test
73      public void testLeastSquareSolve() {
74          RealMatrix m =
75              MatrixUtils.createRealMatrix(new double[][] {
76                                     { 1.0, 0.0 },
77                                     { 0.0, 0.0 }
78                                 });
79          DecompositionSolver solver = new SingularValueDecomposition(m).getSolver();
80          RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
81              { 11, 12 }, { 21, 22 }
82          });
83          RealMatrix xMatrix = solver.solve(b);
84          Assert.assertEquals(11, xMatrix.getEntry(0, 0), 1.0e-15);
85          Assert.assertEquals(12, xMatrix.getEntry(0, 1), 1.0e-15);
86          Assert.assertEquals(0, xMatrix.getEntry(1, 0), 1.0e-15);
87          Assert.assertEquals(0, xMatrix.getEntry(1, 1), 1.0e-15);
88          RealVector xColVec = solver.solve(b.getColumnVector(0));
89          Assert.assertEquals(11, xColVec.getEntry(0), 1.0e-15);
90          Assert.assertEquals(0, xColVec.getEntry(1), 1.0e-15);
91          RealVector xColOtherVec = solver.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
92          Assert.assertEquals(11, xColOtherVec.getEntry(0), 1.0e-15);
93          Assert.assertEquals(0, xColOtherVec.getEntry(1), 1.0e-15);
94      }

96      /** test solve */
97      @Test
98      public void testSolve() {
99          DecompositionSolver solver =
100             new SingularValueDecomposition(MatrixUtils.createRealMatrix(testSquare)).getSolver();
101         RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
102                 { 1, 2, 3 }, { 0, -5, 1 }
103         });
104         RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
105                 { -8.0 / 25.0, -263.0 / 75.0, -29.0 / 75.0 },
106                 { 19.0 / 25.0,   78.0 / 25.0,  49.0 / 25.0 }
107         });
108 
109         // using RealMatrix
110         Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), normTolerance);
111 
112         // using ArrayRealVector
113         for (int i = 0; i < b.getColumnDimension(); ++i) {
114             Assert.assertEquals(0,
115                          solver.solve(b.getColumnVector(i)).subtract(xRef.getColumnVector(i)).getNorm(),
116                          1.0e-13);
117         }
118 
119         // using RealVector with an alternate implementation
120         for (int i = 0; i < b.getColumnDimension(); ++i) {
121             ArrayRealVectorTest.RealVectorTestImpl v =
122                 new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
123             Assert.assertEquals(0,
124                          solver.solve(v).subtract(xRef.getColumnVector(i)).getNorm(),
125                          1.0e-13);
126         }
127 
128     }

139     @Test
140     public void testMath320B() {
141         RealMatrix rm = new Array2DRowRealMatrix(new double[][] {
142             { 1.0, 2.0 }, { 1.0, 2.0 }
143         });
144         SingularValueDecomposition svd =
145             new SingularValueDecomposition(rm);
146         RealMatrix recomposed = svd.getU().multiply(svd.getS()).multiply(svd.getVT());
147         Assert.assertEquals(0.0, recomposed.subtract(rm).getNorm(), 2.0e-15);
148     }

150     @Test
151     public void testSingular() {
152       SingularValueDecomposition svd =
153           new SingularValueDecomposition(MatrixUtils.createRealMatrix(bigSingular));
154       RealMatrix pseudoInverse = svd.getSolver().getInverse();
155       RealMatrix expected = new Array2DRowRealMatrix(new double[][] {
156           {-0.0355022687,0.0512742236,-0.0001045523,0.0157719549},
157           {-0.3214992438,0.3162419255,0.0000348508,-0.0052573183},
158           {0.5437098346,-0.4107754586,-0.0008256918,0.132934376},
159           {-0.0714905202,0.053808742,0.0006279816,-0.0176817782}
160       });
161       Assert.assertEquals(0, expected.subtract(pseudoInverse).getNorm(), 1.0e-9);
162     }
