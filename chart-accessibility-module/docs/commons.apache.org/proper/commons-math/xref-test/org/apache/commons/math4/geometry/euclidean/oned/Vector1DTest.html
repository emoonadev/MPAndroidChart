34  public class Vector1DTest {
35      @Test
36      public void testConstructors() throws DimensionMismatchException {
37          checkVector(new Vector1D(3, new Vector1D(FastMath.PI / 3)),
38                      FastMath.PI);
39          checkVector(new Vector1D(2, Vector1D.ONE, -3, new Vector1D(2)),
40                      -4);
41          checkVector(new Vector1D(2, Vector1D.ONE,
42                                   5, new Vector1D(2),
43                                   -3, new Vector1D(3)),
44                      3);
45          checkVector(new Vector1D(2, Vector1D.ONE,
46                                   5, new Vector1D(2),
47                                   5, new Vector1D(-2),
48                                   -3, new Vector1D(-3)),
49                      11);
50      }
51  
52      @Test
53      public void testSpace() {
54          Space space = new Vector1D(1).getSpace();
55          Assert.assertEquals(1, space.getDimension());
56      }
57  
58      @Test
59      public void testZero() {
60          Assert.assertEquals(0, new Vector1D(1).getZero().getNorm(), 1.0e-15);
61      }
62  
63      @Test
64      public void testEquals() {
65          Vector1D u1 = new Vector1D(1);
66          Vector1D u2 = new Vector1D(1);
67          Assert.assertTrue(u1.equals(u1));
68          Assert.assertTrue(u1.equals(u2));
69          Assert.assertFalse(u1.equals(new Vector1D(1 + 10 * Precision.EPSILON)));
70          Assert.assertTrue(new Vector1D(Double.NaN).equals(new Vector1D(Double.NaN)));
71      }
72  
73      @Test
74      public void testHash() {
75          Assert.assertEquals(new Vector1D(Double.NaN).hashCode(), new Vector1D(Double.NaN).hashCode());
76          Vector1D u = new Vector1D(1);
77          Vector1D v = new Vector1D(1 + 10 * Precision.EPSILON);
78          Assert.assertTrue(u.hashCode() != v.hashCode());
79      }
80  
81      @Test
82      public void testInfinite() {
83          Assert.assertTrue(new Vector1D(Double.NEGATIVE_INFINITY).isInfinite());
84          Assert.assertTrue(new Vector1D(Double.POSITIVE_INFINITY).isInfinite());
85          Assert.assertFalse(new Vector1D(1).isInfinite());
86          Assert.assertFalse(new Vector1D(Double.NaN).isInfinite());
87      }
88  
89      @Test
90      public void testNaN() {
91          Assert.assertTrue(new Vector1D(Double.NaN).isNaN());
92          Assert.assertFalse(new Vector1D(1).isNaN());
93          Assert.assertFalse(new Vector1D(Double.NEGATIVE_INFINITY).isNaN());
94      }

96      @Test
97      public void testToString() {
98          Assert.assertEquals("{3}", new Vector1D(3).toString());
99          NumberFormat format = new DecimalFormat("0.000", new DecimalFormatSymbols(Locale.US));
100         Assert.assertEquals("{3.000}", new Vector1D(3).toString(format));
101     }

133     @Test
134     public void testDistance1() {
135         Vector1D v1 = new Vector1D(1);
136         Vector1D v2 = new Vector1D(-4);
137         Assert.assertEquals(0.0, new Vector1D(-1).distance1(new Vector1D(-1)), 0);
138         Assert.assertEquals(5.0, v1.distance1(v2), 1.0e-12);
139         Assert.assertEquals(v1.subtract(v2).getNorm1(), v1.distance1(v2), 1.0e-12);
140     }
141 
142     @Test
143     public void testDistance() {
144         Vector1D v1 = new Vector1D(1);
145         Vector1D v2 = new Vector1D(-4);
146         Assert.assertEquals(0.0, Vector1D.distance(new Vector1D(-1), new Vector1D(-1)), 0);
147         Assert.assertEquals(5.0, Vector1D.distance(v1, v2), 1.0e-12);
148         Assert.assertEquals(v1.subtract(v2).getNorm(), Vector1D.distance(v1, v2), 1.0e-12);
149     }
150 
151     @Test
152     public void testDistanceSq() {
153         Vector1D v1 = new Vector1D(1);
154         Vector1D v2 = new Vector1D(-4);
155         Assert.assertEquals(0.0, Vector1D.distanceSq(new Vector1D(-1), new Vector1D(-1)), 0);
156         Assert.assertEquals(25.0, Vector1D.distanceSq(v1, v2), 1.0e-12);
157         Assert.assertEquals(Vector1D.distance(v1, v2) * Vector1D.distance(v1, v2),
158                             Vector1D.distanceSq(v1, v2), 1.0e-12);
159   }

161     @Test
162     public void testDistanceInf() {
163         Vector1D v1 = new Vector1D(1);
164         Vector1D v2 = new Vector1D(-4);
165         Assert.assertEquals(0.0, Vector1D.distanceInf(new Vector1D(-1), new Vector1D(-1)), 0);
166         Assert.assertEquals(5.0, Vector1D.distanceInf(v1, v2), 1.0e-12);
167         Assert.assertEquals(v1.subtract(v2).getNormInf(), Vector1D.distanceInf(v1, v2), 1.0e-12);
168     }

170     @Test
171     public void testSubtract() {
172         Vector1D v1 = new Vector1D(1);
173         Vector1D v2 = new Vector1D(-3);
174         v1 = v1.subtract(v2);
175         checkVector(v1, 4);
176 
177         checkVector(v2.subtract(v1), -7);
178         checkVector(v2.subtract(3, v1), -15);
179     }

181     @Test
182     public void testAdd() {
183         Vector1D v1 = new Vector1D(1);
184         Vector1D v2 = new Vector1D(-3);
185         v1 = v1.add(v2);
186         checkVector(v1, -2);
187 
188         checkVector(v2.add(v1), -5);
189         checkVector(v2.add(3, v1), -9);
190     }

192     @Test
193     public void testScalarProduct() {

204         try {

206             Assert.fail("an exception should have been thrown");
207         } catch (MathArithmeticException ae) {
208             // expected behavior
209         }
210     }

215     }

217     private void checkVector(Vector1D v, double x) {
218         Assert.assertEquals(x, v.getX(), 1.0e-12);
