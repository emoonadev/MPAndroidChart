203     @Test
204     public void testParseIgnoredWhitespace() {
205         RealMatrix expected = MatrixUtils.createRealMatrix(new double[][] {{1, 1, 1}, {1, 1, 1}});
206         ParsePosition pos1 = new ParsePosition(0);
207         String source1 = "{{1,1,1},{1,1,1}}";
208         Assert.assertEquals(expected, realMatrixFormat.parse(source1, pos1));
209         Assert.assertEquals(source1.length(), pos1.getIndex());
210         ParsePosition pos2 = new ParsePosition(0);
211         String source2 = " { { 1 , 1 , 1 } , { 1 , 1 , 1 } } ";
212         Assert.assertEquals(expected, realMatrixFormat.parse(source2, pos2));
213         Assert.assertEquals(source2.length() - 1, pos2.getIndex());
214     }

288     @Test
289     public void testParseNan() {
290         String source = "{{(NaN), (NaN), (NaN)}}";
291         RealMatrix actual = realMatrixFormat.parse(source);
292         RealMatrix expected = MatrixUtils.createRealMatrix(new double[][] {{Double.NaN, Double.NaN, Double.NaN}});
293         for (int i = 0; i < expected.getRowDimension(); i++) {
294             for (int j = 0; j < expected.getColumnDimension(); j++) {
295                 Assert.assertTrue(Double.isNaN(actual.getEntry(i, j)));
296             }
297         }
298     }
