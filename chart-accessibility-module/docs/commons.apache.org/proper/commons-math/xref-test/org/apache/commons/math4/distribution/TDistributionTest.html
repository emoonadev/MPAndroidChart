128     @Test
129     public void testMoments() {
130         final double tol = 1e-9;
131         TDistribution dist;
132 
133         dist = new TDistribution(1);
134         Assert.assertTrue(Double.isNaN(dist.getNumericalMean()));
135         Assert.assertTrue(Double.isNaN(dist.getNumericalVariance()));
136 
137         dist = new TDistribution(1.5);
138         Assert.assertEquals(dist.getNumericalMean(), 0, tol);
139         Assert.assertTrue(Double.isInfinite(dist.getNumericalVariance()));
140 
141         dist = new TDistribution(5);
142         Assert.assertEquals(dist.getNumericalMean(), 0, tol);
143         Assert.assertEquals(dist.getNumericalVariance(), 5d / (5d - 2d), tol);
144     }

146     /*
147      * Adding this test to benchmark against tables published by NIST
148      * http://itl.nist.gov/div898/handbook/eda/section3/eda3672.htm
149      * Have chosen tabulated results for degrees of freedom 2,10,30,100
150      * Have chosen problevels from 0.10 to 0.001
151      */
152     @Test
153     public void nistData(){
154         double[] prob = new double[]{ 0.10,0.05,0.025,0.01,0.005,0.001};
155         double[] args2 = new double[]{1.886,2.920,4.303,6.965,9.925,22.327};
156         double[] args10 = new double[]{1.372,1.812,2.228,2.764,3.169,4.143};
157         double[] args30 = new double[]{1.310,1.697,2.042,2.457,2.750,3.385};
158         double[] args100= new double[]{1.290,1.660,1.984,2.364,2.626,3.174};
159         TestUtils.assertEquals(prob, makeNistResults(args2, 2), 1.0e-4);
160         TestUtils.assertEquals(prob, makeNistResults(args10, 10), 1.0e-4);
161         TestUtils.assertEquals(prob, makeNistResults(args30, 30), 1.0e-4);
162         TestUtils.assertEquals(prob, makeNistResults(args100, 100), 1.0e-4);
163         return;
164     }
165     private double[] makeNistResults(double[] args, int df){
166         TDistribution td =  new TDistribution(df);
167         double[] res  = new double[ args.length ];
168         for( int i = 0 ; i < res.length ; i++){
169             res[i] = 1.0 - td.cumulativeProbability(args[i]);
170         }
171         return res;
172     }
