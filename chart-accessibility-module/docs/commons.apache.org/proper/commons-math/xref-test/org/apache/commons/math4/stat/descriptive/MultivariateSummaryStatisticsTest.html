43      @Test
44      public void testSetterInjection() {
45          MultivariateSummaryStatistics u = createMultivariateSummaryStatistics(2, true);
46          u.setMeanImpl(new StorelessUnivariateStatistic[] {
47                          new sumMean(), new sumMean()
48                        });
49          u.addValue(new double[] { 1, 2 });
50          u.addValue(new double[] { 3, 4 });
51          Assert.assertEquals(4, u.getMean()[0], 1E-14);
52          Assert.assertEquals(6, u.getMean()[1], 1E-14);
53          u.clear();
54          u.addValue(new double[] { 1, 2 });
55          u.addValue(new double[] { 3, 4 });
56          Assert.assertEquals(4, u.getMean()[0], 1E-14);
57          Assert.assertEquals(6, u.getMean()[1], 1E-14);
58          u.clear();
59          u.setMeanImpl(new StorelessUnivariateStatistic[] {
60                          new Mean(), new Mean()
61                        }); // OK after clear
62          u.addValue(new double[] { 1, 2 });
63          u.addValue(new double[] { 3, 4 });
64          Assert.assertEquals(2, u.getMean()[0], 1E-14);
65          Assert.assertEquals(3, u.getMean()[1], 1E-14);
66          Assert.assertEquals(2, u.getDimension());
67      }

105     @Test
106     public void testShuffledStatistics() {
107         // the purpose of this test is only to check the get/set methods
108         // we are aware shuffling statistics like this is really not
109         // something sensible to do in production ...
110         MultivariateSummaryStatistics reference = createMultivariateSummaryStatistics(2, true);
111         MultivariateSummaryStatistics shuffled  = createMultivariateSummaryStatistics(2, true);
112 
113         StorelessUnivariateStatistic[] tmp = shuffled.getGeoMeanImpl();
114         shuffled.setGeoMeanImpl(shuffled.getMeanImpl());
115         shuffled.setMeanImpl(shuffled.getMaxImpl());
116         shuffled.setMaxImpl(shuffled.getMinImpl());
117         shuffled.setMinImpl(shuffled.getSumImpl());
118         shuffled.setSumImpl(shuffled.getSumsqImpl());
119         shuffled.setSumsqImpl(shuffled.getSumLogImpl());
120         shuffled.setSumLogImpl(tmp);
121 
122         for (int i = 100; i > 0; --i) {
123             reference.addValue(new double[] {i, i});
124             shuffled.addValue(new double[] {i, i});
125         }
126 
127         TestUtils.assertEquals(reference.getMean(),          shuffled.getGeometricMean(), 1.0e-10);
128         TestUtils.assertEquals(reference.getMax(),           shuffled.getMean(),          1.0e-10);
129         TestUtils.assertEquals(reference.getMin(),           shuffled.getMax(),           1.0e-10);
130         TestUtils.assertEquals(reference.getSum(),           shuffled.getMin(),           1.0e-10);
131         TestUtils.assertEquals(reference.getSumSq(),         shuffled.getSum(),           1.0e-10);
132         TestUtils.assertEquals(reference.getSumLog(),        shuffled.getSumSq(),         1.0e-10);
133         TestUtils.assertEquals(reference.getGeometricMean(), shuffled.getSumLog(),        1.0e-10);
134 
135     }

142         private double sum = 0;
143         private long n = 0;
144         @Override
145         public double evaluate(double[] values, int begin, int length) {
146             return 0;
147         }

150             return 0;
151         }
152         @Override
153         public void clear() {
154           sum = 0;
155           n = 0;
156         }
157         @Override
158         public long getN() {
159             return n;
160         }
161         @Override
162         public double getResult() {
163             return sum;
164         }
165         @Override
166         public void increment(double d) {
167             sum += d;
168             n++;
169         }
170         @Override
171         public void incrementAll(double[] values, int start, int length) {
172         }
173         @Override
174         public void incrementAll(double[] values) {
175         }
176         @Override
177         public StorelessUnivariateStatistic copy() {
178             return new sumMean();
179         }

193     @Test
194     public void testStats() {
195         MultivariateSummaryStatistics u = createMultivariateSummaryStatistics(2, true);
196         Assert.assertEquals(0, u.getN());
197         u.addValue(new double[] { 1, 2 });
198         u.addValue(new double[] { 2, 3 });
199         u.addValue(new double[] { 2, 3 });
200         u.addValue(new double[] { 3, 4 });
201         Assert.assertEquals( 4, u.getN());
202         Assert.assertEquals( 8, u.getSum()[0], 1.0e-10);
203         Assert.assertEquals(12, u.getSum()[1], 1.0e-10);
204         Assert.assertEquals(18, u.getSumSq()[0], 1.0e-10);
205         Assert.assertEquals(38, u.getSumSq()[1], 1.0e-10);
206         Assert.assertEquals( 1, u.getMin()[0], 1.0e-10);
207         Assert.assertEquals( 2, u.getMin()[1], 1.0e-10);
208         Assert.assertEquals( 3, u.getMax()[0], 1.0e-10);
209         Assert.assertEquals( 4, u.getMax()[1], 1.0e-10);
210         Assert.assertEquals(2.4849066497880003102, u.getSumLog()[0], 1.0e-10);
211         Assert.assertEquals( 4.276666119016055311, u.getSumLog()[1], 1.0e-10);
212         Assert.assertEquals( 1.8612097182041991979, u.getGeometricMean()[0], 1.0e-10);
213         Assert.assertEquals( 2.9129506302439405217, u.getGeometricMean()[1], 1.0e-10);
214         Assert.assertEquals( 2, u.getMean()[0], 1.0e-10);
215         Assert.assertEquals( 3, u.getMean()[1], 1.0e-10);
216         Assert.assertEquals(FastMath.sqrt(2.0 / 3.0), u.getStandardDeviation()[0], 1.0e-10);
217         Assert.assertEquals(FastMath.sqrt(2.0 / 3.0), u.getStandardDeviation()[1], 1.0e-10);
218         Assert.assertEquals(2.0 / 3.0, u.getCovariance().getEntry(0, 0), 1.0e-10);
219         Assert.assertEquals(2.0 / 3.0, u.getCovariance().getEntry(0, 1), 1.0e-10);
220         Assert.assertEquals(2.0 / 3.0, u.getCovariance().getEntry(1, 0), 1.0e-10);
221         Assert.assertEquals(2.0 / 3.0, u.getCovariance().getEntry(1, 1), 1.0e-10);
222         u.clear();
223         Assert.assertEquals(0, u.getN());
224     }

282     @Test
283     public void testEqualsAndHashCode() {
284         MultivariateSummaryStatistics u = createMultivariateSummaryStatistics(2, true);
285         MultivariateSummaryStatistics t = null;
286         int emptyHash = u.hashCode();
287         Assert.assertTrue(u.equals(u));
288         Assert.assertFalse(u.equals(t));
289         Assert.assertFalse(u.equals(Double.valueOf(0)));
290         t = createMultivariateSummaryStatistics(2, true);
291         Assert.assertTrue(t.equals(u));
292         Assert.assertTrue(u.equals(t));
293         Assert.assertEquals(emptyHash, t.hashCode());
294 
295         // Add some data to u
296         u.addValue(new double[] { 2d, 1d });
297         u.addValue(new double[] { 1d, 1d });
298         u.addValue(new double[] { 3d, 1d });
299         u.addValue(new double[] { 4d, 1d });
300         u.addValue(new double[] { 5d, 1d });
301         Assert.assertFalse(t.equals(u));
302         Assert.assertFalse(u.equals(t));
303         Assert.assertTrue(u.hashCode() != t.hashCode());
304 
305         //Add data in same order to t
306         t.addValue(new double[] { 2d, 1d });
307         t.addValue(new double[] { 1d, 1d });
308         t.addValue(new double[] { 3d, 1d });
309         t.addValue(new double[] { 4d, 1d });
310         t.addValue(new double[] { 5d, 1d });
311         Assert.assertTrue(t.equals(u));
312         Assert.assertTrue(u.equals(t));
313         Assert.assertEquals(u.hashCode(), t.hashCode());
314 
315         // Clear and make sure summaries are indistinguishable from empty summary
316         u.clear();
317         t.clear();
318         Assert.assertTrue(t.equals(u));
319         Assert.assertTrue(u.equals(t));
320         Assert.assertEquals(emptyHash, t.hashCode());
321         Assert.assertEquals(emptyHash, u.hashCode());
322     }
