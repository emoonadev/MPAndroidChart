17  package org.apache.commons.math4.genetics;
18  
19  
20  import org.apache.commons.math4.exception.MathIllegalArgumentException;
21  import org.apache.commons.math4.genetics.BinaryChromosome;
22  import org.apache.commons.math4.genetics.Chromosome;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class BinaryChromosomeTest {
27  
28      @Test
29      public void testInvalidConstructor() {
30          Integer[][] reprs = new Integer[][] {
31                  new Integer[] {0,1,0,1,2},
32                  new Integer[] {0,1,0,1,-1}
33          };
34  
35          for (Integer[] repr : reprs) {
36              try {
37                  new DummyBinaryChromosome(repr);
38                  Assert.fail("Exception not caught");
39              } catch (MathIllegalArgumentException e) {
40                  // Expected
41              }
42          }
43      }
44  
45      @Test
46      public void testRandomConstructor() {
47          for (int i=0; i<20; i++) {
48              new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(10));
49          }
50      }
51  
52      @Test
53      public void testIsSame() {
54          Chromosome c1 = new DummyBinaryChromosome(new Integer[] {0,1,0,1,0,1});
55          Chromosome c2 = new DummyBinaryChromosome(new Integer[] {0,1,1,0,1});
56          Chromosome c3 = new DummyBinaryChromosome(new Integer[] {0,1,0,1,0,1,1});
57          Chromosome c4 = new DummyBinaryChromosome(new Integer[] {1,1,0,1,0,1});
58          Chromosome c5 = new DummyBinaryChromosome(new Integer[] {0,1,0,1,0,0});
59          Chromosome c6 = new DummyBinaryChromosome(new Integer[] {0,1,0,1,0,1});
60  
61          Assert.assertFalse(c1.isSame(c2));
62          Assert.assertFalse(c1.isSame(c3));
63          Assert.assertFalse(c1.isSame(c4));
64          Assert.assertFalse(c1.isSame(c5));
65          Assert.assertTrue(c1.isSame(c6));
66      }
67  
68  }
