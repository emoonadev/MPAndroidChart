17  package org.apache.commons.math4.ml.distance;
18  
19  import org.junit.Assert;
20  import org.junit.Test;
21  import org.apache.commons.math4.ml.distance.DistanceMeasure;
22  import org.apache.commons.math4.ml.distance.EuclideanDistance;
23  import org.apache.commons.math4.util.FastMath;
24  
25  /**
26   * Tests for {@link EuclideanDistance} class.
27   */
28  public class EuclideanDistanceTest {
29      final DistanceMeasure distance = new EuclideanDistance();
30  
31      @Test
32      public void testZero() {
33          final double[] a = { 0, 1, -2, 3.4, 5, -6.7, 89 };
34          Assert.assertEquals(0, distance.compute(a, a), 0d);
35      }
36  
37      @Test
38      public void test() {
39          final double[] a = { 1, -2, 3, 4 };
40          final double[] b = { -5, -6, 7, 8 };
41          final double expected = FastMath.sqrt(84);
42          Assert.assertEquals(expected, distance.compute(a, b), 0d);
43          Assert.assertEquals(expected, distance.compute(b, a), 0d);
44      }
45  }
