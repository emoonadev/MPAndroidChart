157     @Test
158     public void testValueWithInverseFunction() {
159         final double lo = 2;
160         final double hi = 3;
161         final Logit f = new Logit(lo, hi);
162         final Sigmoid g = new Sigmoid(lo, hi);
163         final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
164                                                                  0x49914cdd9f0b8db5l);
165         final UnivariateDifferentiableFunction id = FunctionUtils.compose((UnivariateDifferentiableFunction) g,
166                                                                 (UnivariateDifferentiableFunction) f);
167 
168         for (int i = 0; i < 10; i++) {
169             final double x = lo + random.nextDouble() * (hi - lo);
170             Assert.assertEquals(x, id.value(new DerivativeStructure(1, 1, 0, x)).getValue(), EPS);
171         }
172 
173         Assert.assertEquals(lo, id.value(new DerivativeStructure(1, 1, 0, lo)).getValue(), EPS);
174         Assert.assertEquals(hi, id.value(new DerivativeStructure(1, 1, 0, hi)).getValue(), EPS);
175     }

177     @Test
178     public void testDerivativesWithInverseFunction() {
179         double[] epsilon = new double[] { 1.0e-20, 4.0e-16, 3.0e-15, 2.0e-11, 3.0e-9, 1.0e-6 };
180         final double lo = 2;
181         final double hi = 3;
182         final Logit f = new Logit(lo, hi);
183         final Sigmoid g = new Sigmoid(lo, hi);
184         final UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A,
185                                                                  0x96885e9c1f81cea6l);
186         final UnivariateDifferentiableFunction id =
187                 FunctionUtils.compose((UnivariateDifferentiableFunction) g, (UnivariateDifferentiableFunction) f);
188         for (int maxOrder = 0; maxOrder < 6; ++maxOrder) {
189             double max = 0;
190             for (int i = 0; i < 10; i++) {
191                 final double x = lo + random.nextDouble() * (hi - lo);
192                 final DerivativeStructure dsX = new DerivativeStructure(1, maxOrder, 0, x);
193                 max = FastMath.max(max, FastMath.abs(dsX.getPartialDerivative(maxOrder) -
194                                                      id.value(dsX).getPartialDerivative(maxOrder)));
195                 Assert.assertEquals("maxOrder = " + maxOrder,
196                                     dsX.getPartialDerivative(maxOrder),
197                                     id.value(dsX).getPartialDerivative(maxOrder),
198                                     epsilon[maxOrder]);
199             }
200 
201             // each function evaluates correctly near boundaries,
202             // but combination leads to NaN as some intermediate point is infinite
203             final DerivativeStructure dsLo = new DerivativeStructure(1, maxOrder, 0, lo);
204             if (maxOrder == 0) {
205                 Assert.assertTrue(Double.isInfinite(f.value(dsLo).getPartialDerivative(maxOrder)));
206                 Assert.assertEquals(lo, id.value(dsLo).getPartialDerivative(maxOrder), epsilon[maxOrder]);
207             } else if (maxOrder == 1) {
208                 Assert.assertTrue(Double.isInfinite(f.value(dsLo).getPartialDerivative(maxOrder)));
209                 Assert.assertTrue(Double.isNaN(id.value(dsLo).getPartialDerivative(maxOrder)));
210             } else {
211                 Assert.assertTrue(Double.isNaN(f.value(dsLo).getPartialDerivative(maxOrder)));
212                 Assert.assertTrue(Double.isNaN(id.value(dsLo).getPartialDerivative(maxOrder)));
213             }
214 
215             final DerivativeStructure dsHi = new DerivativeStructure(1, maxOrder, 0, hi);
216             if (maxOrder == 0) {
217                 Assert.assertTrue(Double.isInfinite(f.value(dsHi).getPartialDerivative(maxOrder)));
218                 Assert.assertEquals(hi, id.value(dsHi).getPartialDerivative(maxOrder), epsilon[maxOrder]);
219             } else if (maxOrder == 1) {
220                 Assert.assertTrue(Double.isInfinite(f.value(dsHi).getPartialDerivative(maxOrder)));
221                 Assert.assertTrue(Double.isNaN(id.value(dsHi).getPartialDerivative(maxOrder)));
222             } else {
223                 Assert.assertTrue(Double.isNaN(f.value(dsHi).getPartialDerivative(maxOrder)));
224                 Assert.assertTrue(Double.isNaN(id.value(dsHi).getPartialDerivative(maxOrder)));
225             }
226 
227         }
