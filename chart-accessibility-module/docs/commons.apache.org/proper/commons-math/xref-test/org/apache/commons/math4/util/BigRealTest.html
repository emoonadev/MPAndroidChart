32      @Test
33      public void testConstructor() {
34          Assert.assertEquals(1.625,
35                              new BigReal(new BigDecimal("1.625")).doubleValue(),
36                              1.0e-15);
37          Assert.assertEquals(-5.0,
38                              new BigReal(new BigInteger("-5")).doubleValue(),
39                              1.0e-15);
40          Assert.assertEquals(-5.0, new BigReal(new BigInteger("-5"),
41                                                MathContext.DECIMAL64)
42              .doubleValue(), 1.0e-15);
43          Assert
44              .assertEquals(0.125,
45                            new BigReal(new BigInteger("125"), 3).doubleValue(),
46                            1.0e-15);
47          Assert.assertEquals(0.125, new BigReal(new BigInteger("125"), 3,
48                                                 MathContext.DECIMAL64)
49              .doubleValue(), 1.0e-15);
50          Assert.assertEquals(1.625, new BigReal(new char[] {
51              '1', '.', '6', '2', '5'
52          }).doubleValue(), 1.0e-15);
53          Assert.assertEquals(1.625, new BigReal(new char[] {
54              'A', 'A', '1', '.', '6', '2', '5', '9'
55          }, 2, 5).doubleValue(), 1.0e-15);
56          Assert.assertEquals(1.625, new BigReal(new char[] {
57              'A', 'A', '1', '.', '6', '2', '5', '9'
58          }, 2, 5, MathContext.DECIMAL64).doubleValue(), 1.0e-15);
59          Assert.assertEquals(1.625, new BigReal(new char[] {
60              '1', '.', '6', '2', '5'
61          }, MathContext.DECIMAL64).doubleValue(), 1.0e-15);
62          Assert.assertEquals(1.625, new BigReal(1.625).doubleValue(), 1.0e-15);
63          Assert.assertEquals(1.625, new BigReal(1.625, MathContext.DECIMAL64)
64              .doubleValue(), 1.0e-15);
65          Assert.assertEquals(-5.0, new BigReal(-5).doubleValue(), 1.0e-15);
66          Assert.assertEquals(-5.0, new BigReal(-5, MathContext.DECIMAL64)
67              .doubleValue(), 1.0e-15);
68          Assert.assertEquals(-5.0, new BigReal(-5l).doubleValue(), 1.0e-15);
69          Assert.assertEquals(-5.0, new BigReal(-5l, MathContext.DECIMAL64)
70              .doubleValue(), 1.0e-15);
71          Assert.assertEquals(1.625, new BigReal("1.625").doubleValue(), 1.0e-15);
72          Assert.assertEquals(1.625, new BigReal("1.625", MathContext.DECIMAL64)
73              .doubleValue(), 1.0e-15);
74      }

76      @Test
77      public void testCompareTo() {
78          BigReal first = new BigReal(1.0 / 2.0);
79          BigReal second = new BigReal(1.0 / 3.0);
80          BigReal third = new BigReal(1.0 / 2.0);
81  
82          Assert.assertEquals(0, first.compareTo(first));
83          Assert.assertEquals(0, first.compareTo(third));
84          Assert.assertEquals(1, first.compareTo(second));
85          Assert.assertEquals(-1, second.compareTo(first));
86  
87      }

126     @Test
127     public void testReciprocal() {
128         BigReal a = new BigReal("1.2345678");
129         double eps = FastMath.pow(10., -a.getScale());
130         BigReal one = new BigReal("1.0000000");
131         BigReal b = a.reciprocal();
132         BigReal r = one.subtract(a.multiply(b));
133         Assert.assertTrue(FastMath.abs(r.doubleValue()) <= eps);
134         r = one.subtract(b.multiply(a));
135         Assert.assertTrue(FastMath.abs(r.doubleValue()) <= eps);
136     }

166     @Test
167     public void testEqualsAndHashCode() {
168         BigReal zero = new BigReal(0.0);
169         BigReal nullReal = null;
170         Assert.assertTrue(zero.equals(zero));
171         Assert.assertFalse(zero.equals(nullReal));
172         Assert.assertFalse(zero.equals(Double.valueOf(0)));
173         BigReal zero2 = new BigReal(0.0);
174         Assert.assertTrue(zero.equals(zero2));
175         Assert.assertEquals(zero.hashCode(), zero2.hashCode());
176         BigReal one = new BigReal(1.0);
177         Assert.assertFalse((one.equals(zero) || zero.equals(one)));
178         Assert.assertTrue(one.equals(BigReal.ONE));
179     }
