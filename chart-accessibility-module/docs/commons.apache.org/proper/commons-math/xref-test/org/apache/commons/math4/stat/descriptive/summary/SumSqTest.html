17  package org.apache.commons.math4.stat.descriptive.summary;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatistic;
20  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
21  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
22  import org.apache.commons.math4.stat.descriptive.summary.SumOfSquares;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test cases for the {@link SumOfSquares} class.
28   */
29  public class SumSqTest extends StorelessUnivariateStatisticAbstractTest{
30  
31      protected SumOfSquares stat;
32  
33      /**
34       * {@inheritDoc}
35       */
36      @Override
37      public UnivariateStatistic getUnivariateStatistic() {
38          return new SumOfSquares();
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public double expectedValue() {
46          return this.sumSq;
47      }
48  
49      @Test
50      public void testSpecialValues() {
51          SumOfSquares sumSq = new SumOfSquares();
52          Assert.assertEquals(0, sumSq.getResult(), 0);
53          sumSq.increment(2d);
54          Assert.assertEquals(4d, sumSq.getResult(), 0);
55          sumSq.increment(Double.POSITIVE_INFINITY);
56          Assert.assertEquals(Double.POSITIVE_INFINITY, sumSq.getResult(), 0);
57          sumSq.increment(Double.NEGATIVE_INFINITY);
58          Assert.assertEquals(Double.POSITIVE_INFINITY, sumSq.getResult(), 0);
59          sumSq.increment(Double.NaN);
60          Assert.assertTrue(Double.isNaN(sumSq.getResult()));
61          sumSq.increment(1);
62          Assert.assertTrue(Double.isNaN(sumSq.getResult()));
63      }
64  
65      @Override
66      protected void checkClearValue(StorelessUnivariateStatistic statistic) {
67          Assert.assertEquals(0, statistic.getResult(), 0);
68      }
69  
70  }
