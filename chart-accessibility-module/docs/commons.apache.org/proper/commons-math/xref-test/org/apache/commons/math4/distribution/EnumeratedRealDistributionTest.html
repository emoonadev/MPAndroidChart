60      /**
61       * Tests if the {@link EnumeratedRealDistribution} constructor throws
62       * exceptions for invalid data.
63       */
64      @Test
65      public void testExceptions() {
66          EnumeratedRealDistribution invalid = null;
67          try {
68              invalid = new EnumeratedRealDistribution(new double[]{1.0, 2.0}, new double[]{0.0});
69              Assert.fail("Expected DimensionMismatchException");
70          } catch (DimensionMismatchException e) {
71          }
72          try{
73          invalid = new EnumeratedRealDistribution(new double[]{1.0, 2.0}, new double[]{0.0, -1.0});
74              Assert.fail("Expected NotPositiveException");
75          } catch (NotPositiveException e) {
76          }
77          try {
78              invalid = new EnumeratedRealDistribution(new double[]{1.0, 2.0}, new double[]{0.0, 0.0});
79              Assert.fail("Expected MathArithmeticException");
80          } catch (MathArithmeticException e) {
81          }
82          try {
83              invalid = new EnumeratedRealDistribution(new double[]{1.0, 2.0}, new double[]{0.0, Double.NaN});
84              Assert.fail("Expected NotANumberException");
85          } catch (NotANumberException e) {
86          }
87          try {
88              invalid = new EnumeratedRealDistribution(new double[]{1.0, 2.0}, new double[]{0.0, Double.POSITIVE_INFINITY});
89              Assert.fail("Expected NotFiniteNumberException");
90          } catch (NotFiniteNumberException e) {
91          }
92          Assert.assertNull("Expected non-initialized DiscreteRealDistribution", invalid);
93      }

174     /**
175      * Tests sampling.
176      */
177     @Test
178     public void testSample() {
179         final int n = 1000000;
180         final RealDistribution.Sampler sampler =
181             testDistribution.createSampler(RandomSource.create(RandomSource.WELL_1024_A, -123456789));
182         final double[] samples = AbstractRealDistribution.sample(n, sampler);
183         Assert.assertEquals(n, samples.length);
184         double sum = 0;
185         double sumOfSquares = 0;
186         for (int i = 0; i < samples.length; i++) {
187             sum += samples[i];
188             sumOfSquares += samples[i] * samples[i];
189         }
190         Assert.assertEquals(testDistribution.getNumericalMean(),
191                 sum / n, 1e-2);
192         Assert.assertEquals(testDistribution.getNumericalVariance(),
193                 sumOfSquares / n - FastMath.pow(sum / n, 2), 1e-2);
194     }

205     @Test
206     public void testIssue1065() {
207         // Test Distribution for inverseCumulativeProbability
208         //
209         //         ^
210         //         |
211         // 1.000   +--------------------------------o===============
212         //         |                               3|
213         //         |                                |
214         //         |                             1o=
215         // 0.750   +-------------------------> o==  .
216         //         |                          3|  . .
217         //         |                   0       |  . .
218         // 0.5625  +---------------> o==o======   . .
219         //         |                 |  .      .  . .
220         //         |                 |  .      .  . .
221         //         |                5|  .      .  . .
222         //         |                 |  .      .  . .
223         //         |             o===   .      .  . .
224         //         |             |   .  .      .  . .
225         //         |            4|   .  .      .  . .
226         //         |             |   .  .      .  . .
227         // 0.000   +=============----+--+------+--+-+--------------->
228         //                      14  18 21     28 31 33
229         //
230         // sum  = 4+5+0+3+1+3 = 16
231 
232         EnumeratedRealDistribution distribution = new EnumeratedRealDistribution(
233                 new double[] { 14.0, 18.0, 21.0, 28.0, 31.0, 33.0 },
234                 new double[] { 4.0 / 16.0, 5.0 / 16.0, 0.0 / 16.0, 3.0 / 16.0, 1.0 / 16.0, 3.0 / 16.0 });
235 
236         assertEquals(14.0, distribution.inverseCumulativeProbability(0.0000), 0.0);
237         assertEquals(14.0, distribution.inverseCumulativeProbability(0.2500), 0.0);
238         assertEquals(33.0, distribution.inverseCumulativeProbability(1.0000), 0.0);
239 
240         assertEquals(18.0, distribution.inverseCumulativeProbability(0.5000), 0.0);
241         assertEquals(18.0, distribution.inverseCumulativeProbability(0.5624), 0.0);
242         assertEquals(28.0, distribution.inverseCumulativeProbability(0.5626), 0.0);
243         assertEquals(31.0, distribution.inverseCumulativeProbability(0.7600), 0.0);
244         assertEquals(18.0, distribution.inverseCumulativeProbability(0.5625), 0.0);
245         assertEquals(28.0, distribution.inverseCumulativeProbability(0.7500), 0.0);
246     }
