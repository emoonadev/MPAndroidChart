88      @Test
89      public void testCrossover() {
90          Integer[] p1 = new Integer[] {1,0,1,0,1,0,1,0,1};
91          Integer[] p2 = new Integer[] {0,1,0,1,0,1,0,1,0};
92  
93          BinaryChromosome p1c = new DummyBinaryChromosome(p1);
94          BinaryChromosome p2c = new DummyBinaryChromosome(p2);
95  
96          final int order = 3;
97          NPointCrossover<Integer> npc = new NPointCrossover<Integer>(order);
98  
99          // the two parent chromosomes are different at each position, so it is easy to detect
100         // the number of crossovers that happened for each child
101         for (int i=0; i<20; i++) {
102             ChromosomePair pair = npc.crossover(p1c,p2c);
103 
104             Integer[] c1 = new Integer[p1.length];
105             Integer[] c2 = new Integer[p2.length];
106 
107             c1 = ((BinaryChromosome) pair.getFirst()).getRepresentation().toArray(c1);
108             c2 = ((BinaryChromosome) pair.getSecond()).getRepresentation().toArray(c2);
109 
110             Assert.assertEquals(order, detectCrossoverPoints(p1c, p2c, (BinaryChromosome) pair.getFirst()));
111             Assert.assertEquals(order, detectCrossoverPoints(p2c, p1c, (BinaryChromosome) pair.getSecond()));
112         }
113     }
114 
115     private int detectCrossoverPoints(BinaryChromosome p1, BinaryChromosome p2, BinaryChromosome c) {
116         int crossovers = 0;
117         final int length = p1.getLength();
118 
119         final List<Integer> p1Rep = p1.getRepresentation();
120         final List<Integer> p2Rep = p2.getRepresentation();
121         final List<Integer> cRep = c.getRepresentation();
122 
123         List<Integer> rep = p1Rep;
124         for (int i = 0; i < length; i++) {
125             if (rep.get(i) != cRep.get(i)) {
126                 crossovers++;
127                 rep = rep == p1Rep ? p2Rep : p1Rep;
128             }
129         }
130 
131         return crossovers;
132     }
