37      /** Formatting. */
38      private static final int DEFAULT_MAX_NAME_WIDTH = 45;
39      /** Formatting. */
40      private static final String ELLIPSIS = "...";
41      /** Formatting. */
42      private static final String TO_STRING_MEMORY_ADDRESS_REGEX = "@\\p{XDigit}{1,8}";
43      /** Formatting. */
44      private static final String JAVA_IDENTIFIER_REGEX =
45          "(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*\\.)*\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*";
46      /** Formatting. */
47      private static final Pattern JAVA_IDENTIFIER_PATTERN =
48          Pattern.compile(JAVA_IDENTIFIER_REGEX);
49      /** Nanoseconds to milliseconds conversion factor ({@value}). */
50      public static final double NANO_TO_MILLI = 1e-6;
51      /** Default number of code repeat per timed block. */
52      private static final int DEFAULT_REPEAT_CHUNK = 1000;
53      /** Default number of code repeats for computing the average run time. */
54      private static final int DEFAULT_REPEAT_STAT = 10000;
55      /** RNG. */
56      private static UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_19937_C);

58      /**
59       * Timing.
60       *
61       * @param repeatChunk Each timing measurement will done done for that
62       * number of repeats of the code.
63       * @param repeatStat Timing will be averaged over that number of runs.
64       * @param runGC Call {@code System.gc()} between each timed block. When
65       * set to {@code true}, the test will run much slower.
66       * @param methods Codes being timed.
67       * @return for each of the given {@code methods}, a
68       * {@link StatisticalSummary} of the average times (in milliseconds)
69       * taken by a single call to the {@code call} method (i.e. the time
70       * taken by each timed block divided by {@code repeatChunk}).
71       */
72      @SafeVarargs
73      public static StatisticalSummary[] time(int repeatChunk,
74                                              int repeatStat,
75                                              boolean runGC,
76                                              Callable<Double> ... methods) {
77          final double[][][] times = timesAndResults(repeatChunk,
78                                                     repeatStat,
79                                                     runGC,
80                                                     methods);
81  
82          final int len = methods.length;
83          final StatisticalSummary[] stats = new StatisticalSummary[len];
84          for (int j = 0; j < len; j++) {
85              final SummaryStatistics s = new SummaryStatistics();
86              for (int k = 0; k < repeatStat; k++) {
87                  s.addValue(times[j][k][0]);
88              }
89              stats[j] = s.getSummary();
90          }
91  
92          return stats;
93      }

95      /**
96       * Timing.
97       *
98       * @param repeatChunk Each timing measurement will done done for that
99       * number of repeats of the code.
100      * @param repeatStat Timing will be averaged over that number of runs.
101      * @param runGC Call {@code System.gc()} between each timed block. When
102      * set to {@code true}, the test will run much slower.
103      * @param methods Codes being timed.
104      * @return for each of the given {@code methods} (first dimension), and
105      * each of the {@code repeatStat} runs (second dimension):
106      * <ul>
107      *  <li>
108      *   the average time (in milliseconds) taken by a single call to the
109      *   {@code call} method (i.e. the time taken by each timed block divided
110      *   by {@code repeatChunk})
111      *  </li>
112      *  <li>
113      *   the result returned by the {@code call} method.
114      *  </li>
115      * </ul>
116      */
117     @SafeVarargs
118     public static double[][][] timesAndResults(int repeatChunk,
119                                                int repeatStat,
120                                                boolean runGC,
121                                                Callable<Double> ... methods) {
122         final int numMethods = methods.length;
123         final double[][][] timesAndResults = new double[numMethods][repeatStat][2];
124 
125         // Indices into the array containing the methods to benchmark.
126         // The purpose is that at each repeat, the "methods" are called in a different order.
127         final int[] methodSequence = MathArrays.natural(numMethods);
128 
129         try {
130             for (int k = 0; k < repeatStat; k++) {
131                 MathArrays.shuffle(methodSequence, rng);
132                 for (int n = 0; n < numMethods; n++) {
133                     final int j = methodSequence[n]; // Index of the timed method.
134 
135                     if (runGC) {
136                         // Try to perform GC outside the timed block.
137                         System.gc();
138                     }
139 
140                     final Callable<Double> r = methods[j];
141                     final double[] result = new double[repeatChunk];
142 
143                     // Timed block.
144                     final long start = System.nanoTime();
145                     for (int i = 0; i < repeatChunk; i++) {
146                         result[i] = r.call().doubleValue();
147                     }
148                     final long stop = System.nanoTime();
149 
150                     // Collect run time.
151                     timesAndResults[j][k][0] = (stop - start) * NANO_TO_MILLI;
152                     // Keep track of a randomly selected result.
153                     timesAndResults[j][k][1] = result[rng.nextInt(repeatChunk)];
154                 }
155             }
156         } catch (Exception e) {
157             // Abort benchmarking if codes throw exceptions.
158             throw new MathIllegalStateException(LocalizedFormats.SIMPLE_MESSAGE, e.getMessage());
159         }
160 
161         final double normFactor = 1d / repeatChunk;
162         for (int j = 0; j < numMethods; j++) {
163             for (int k = 0; k < repeatStat; k++) {
164                 timesAndResults[j][k][0] *= normFactor;
165             }
166         }
167 
168         return timesAndResults;
169     }

171     /**
172      * Timing and report (to standard output) the average time and standard
173      * deviation of a single call.
174      * The timing is performed by calling the
175      * {@link #time(int,int,boolean,Callable[]) time} method.
176      *
177      * @param title Title of the test (for the report).
178      * @param maxNameWidth Maximum width of the first column of the report.
179      * @param repeatChunk Each timing measurement will done done for that
180      * number of repeats of the code.
181      * @param repeatStat Timing will be averaged over that number of runs.
182      * @param runGC Call {@code System.gc()} between each timed block. When
183      * set to {@code true}, the test will run much slower.
184      * @param methods Codes being timed.
185      * @return for each of the given {@code methods}, a statistics of the
186      * average times (in milliseconds) taken by a single call to the
187      * {@code call} method (i.e. the time taken by each timed block divided
188      * by {@code repeatChunk}).
189      */
190     @SuppressWarnings("boxing")
191     public static StatisticalSummary[] timeAndReport(String title,
192                                                      int maxNameWidth,
193                                                      int repeatChunk,
194                                                      int repeatStat,
195                                                      boolean runGC,
196                                                      RunTest ... methods) {
197         // Header format.
198         final String hFormat = "%s (calls per timed block: %d, timed blocks: %d, time unit: ms)";
199 
200         // TODO: user-defined parameter?
201         final boolean removePackageName = false;
202 
203         // Width of the longest name.
204         int nameLength = 0;
205         for (RunTest m : methods) {
206             int len = shorten(m.getName(), removePackageName).length();
207             if (len > nameLength) {
208                 nameLength = len;
209             }
210         }
211         final int actualNameLength = nameLength < maxNameWidth ?
212             nameLength :
213             maxNameWidth;
214         final String nameLengthFormat = "%" + actualNameLength + "s";
215 
216         // Column format.
217         final String cFormat = nameLengthFormat + " %9s %7s %10s %5s %4s %10s";
218         // Result format.
219         final String format = nameLengthFormat + " %.3e %.1e %.4e %.3f %.2f %.4e";
220 
221         System.out.println(String.format(hFormat,
222                                          title,
223                                          repeatChunk,
224                                          repeatStat));
225         System.out.println(String.format(cFormat,
226                                          "name",
227                                          "time/call",
228                                          "std dev",
229                                          "total time",
230                                          "ratio",
231                                          "cv",
232                                          "difference"));
233         final StatisticalSummary[] time = time(repeatChunk,
234                                                repeatStat,
235                                                runGC,
236                                                methods);
237         final double refSum = time[0].getSum() * repeatChunk;
238         for (int i = 0, max = time.length; i < max; i++) {
239             final StatisticalSummary s = time[i];
240             final double sum = s.getSum() * repeatChunk;
241             final double mean = s.getMean();
242             final double sigma = s.getStandardDeviation();
243             System.out.println(String.format(format,
244                                              truncate(shorten(methods[i].getName(),
245                                                               removePackageName),
246                                                       actualNameLength,
247                                                       ELLIPSIS),
248                                              mean,
249                                              sigma,
250                                              sum,
251                                              sum / refSum,
252                                              sigma / mean,
253                                              sum - refSum));
254         }
255 
256         return time;
257     }

259     /**
260      * Timing and report (to standard output).
261      * This method calls {@link #timeAndReport(String,int,int,boolean,RunTest[])
262      * timeAndReport(title, 1000, 10000, false, methods)}.
263      *
264      * @param title Title of the test (for the report).
265      * @param methods Codes being timed.
266      * @return for each of the given {@code methods}, a statistics of the
267      * average times (in milliseconds) taken by a single call to the
268      * {@code call} method (i.e. the time taken by each timed block divided
269      * by {@code repeatChunk}).
270      */
271     public static StatisticalSummary[] timeAndReport(String title,
272                                                      RunTest ... methods) {
273         return timeAndReport(title,
274                              DEFAULT_MAX_NAME_WIDTH,
275                              DEFAULT_REPEAT_CHUNK,
276                              DEFAULT_REPEAT_STAT,
277                              false,
278                              methods);
279     }

281     /**
282      * Utility class for storing a test label.
283      */
284     public static abstract class RunTest implements Callable<Double> {
285         private final String name;
286 
287         /**
288          * @param name Test name.
289          */
290         public RunTest(String name) {
291             this.name = name;
292         }
293 
294         /**
295          * @return the name of this test.
296          */
297         public String getName() {
298             return name;
299         }
300 
301         /** {@inheritDoc} */
302         public abstract Double call() throws Exception;
303     }

305     /**
306      * Truncates a string so that it will not be longer than the
307      * specified length.
308      *
309      * @param str String to truncate.
310      * @param maxLength Maximum length.
311      * @param ellipsis String to use in place of the part being removed
312      * from the original string.
313      * @return the truncated string.
314      * @throws NumberIsTooLargeException if the length of {@code ellipsis}
315      * is larger than {@code maxLength - 2}.
316      */
317     private static String truncate(String str,
318                                    int maxLength,
319                                    String ellipsis) {
320         final int ellSize = ellipsis.length();
321         if (ellSize > maxLength - 2) {
322             throw new NumberIsTooLargeException(ellSize, maxLength - 2, false);
323         }
324 
325         final int strSize = str.length();
326         if (strSize <= maxLength) {
327             // Size is OK.
328             return str;
329         }
330 
331         return str.substring(0, maxLength - ellSize) + ellipsis;
332     }

334     /**
335      * Shortens a string.
336      * It will shorten package names and remove memory addresses
337      * that appear in an instance's name.
338      *
339      * @param str Orginal string.
340      * @param removePackageName Whether package name part of a
341      * fully-quallified name should be removed entirely.
342      * @return the shortened string.
343      */
344     private static String shorten(String str,
345                                   boolean removePackageName) {
346         final Matcher m = JAVA_IDENTIFIER_PATTERN.matcher(str);
347         final StringBuffer sb = new StringBuffer();
348         while (m.find()) {
349             final MatchResult r = m.toMatchResult();
350             m.appendReplacement(sb, shortenPackageName(r.group(),
351                                                        removePackageName));
352         }
353         m.appendTail(sb);
354 
355         return sb.toString().replaceAll(TO_STRING_MEMORY_ADDRESS_REGEX, "");
356     }
357 
358     /**
359      * Shortens package part of the name of a class.
360      *
361      * @param name Class name.
362      * @param remove Whether package name part of a fully-qualified
363      * name should be removed entirely.
364      * @return the shortened name.
365      */
366     private static String shortenPackageName(String name,
367                                              boolean remove) {
368         final String[] comp = name.split("\\.");
369         final int last = comp.length - 1;
370 
371         if (remove) {
372             return comp[last];
373         }
374 
375         final StringBuilder s = new StringBuilder();
376         for (int i = 0; i < last; i++) {
377             s.append(comp[i].substring(0, 1)).append(".");
378         }
379         s.append(comp[last]);
380 
381         return s.toString();
382     }
