193     @Test
194     public void testWithMaxIterations1() {
195         final int maxIter = 20;
196         final double[] init = { 3.5e6, 4.2, 0.1 };
197 
198         GaussianCurveFitter fitter = GaussianCurveFitter.create();
199         double[] parameters = fitter
200             .withMaxIterations(maxIter)
201             .withStartPoint(init)
202             .fit(createDataset(DATASET1).toList());

205         Assert.assertEquals(4.054933085999146, parameters[1], 1e-4);

380     /**
381      * Adds the specified points to specified <code>GaussianCurveFitter</code>
382      * instance.
383      *
384      * @param points Data points where first dimension is a point index and
385      *        second dimension is an array of length two representing the point
386      *        with the first value corresponding to X and the second value
387      *        corresponding to Y.
388      * @return the collection of observed points.
389      */
390     private static WeightedObservedPoints createDataset(double[][] points) {
391         final WeightedObservedPoints obs = new WeightedObservedPoints();
392         for (int i = 0; i < points.length; i++) {
393             obs.add(points[i][0], points[i][1]);
394         }
395         return obs;
396     }
