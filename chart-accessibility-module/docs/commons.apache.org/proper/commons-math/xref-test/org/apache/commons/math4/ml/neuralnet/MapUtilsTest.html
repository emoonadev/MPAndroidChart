35   * Tests for {@link MapUtils} class.

37  public class MapUtilsTest {
38      /*
39       * Test assumes that the network is
40       *
41       *  0-----1-----2
42       */
43      @Test
44      public void testFindClosestNeuron() {
45          final FeatureInitializer init
46              = new OffsetFeatureInitializer(FeatureInitializerFactory.uniform(-0.1, 0.1));
47          final FeatureInitializer[] initArray = { init };
48  
49          final Network net = new NeuronString(3, false, initArray).getNetwork();
50          final DistanceMeasure dist = new EuclideanDistance();
51  
52          final Set<Neuron> allBest = new HashSet<Neuron>();
53          final Set<Neuron> best = new HashSet<Neuron>();
54          double[][] features;
55  
56          // The following tests ensures that
57          // 1. the same neuron is always selected when the input feature is
58          //    in the range of the initializer,
59          // 2. different network's neuron have been selected by inputs features
60          //    that belong to different ranges.
61  
62          best.clear();
63          features = new double[][] {
64              { -1 },
65              { 0.4 },
66          };
67          for (double[] f : features) {
68              best.add(MapUtils.findBest(f, net, dist));
69          }
70          Assert.assertEquals(1, best.size());
71          allBest.addAll(best);
72  
73          best.clear();
74          features = new double[][] {
75              { 0.6 },
76              { 1.4 },
77          };
78          for (double[] f : features) {
79              best.add(MapUtils.findBest(f, net, dist));
80          }
81          Assert.assertEquals(1, best.size());
82          allBest.addAll(best);
83  
84          best.clear();
85          features = new double[][] {
86              { 1.6 },
87              { 3 },
88          };
89          for (double[] f : features) {
90              best.add(MapUtils.findBest(f, net, dist));
91          }
92          Assert.assertEquals(1, best.size());
93          allBest.addAll(best);
94  
95          Assert.assertEquals(3, allBest.size());
96      }
97  
98      @Test
99      public void testSort() {
100         final Set<Neuron> list = new HashSet<Neuron>();
101 
102         for (int i = 0; i < 4; i++) {
103             list.add(new Neuron(i, new double[] { i - 0.5 }));
104         }
105 
106         final Neuron[] sorted = MapUtils.sort(new double[] { 3.4 },
107                                               list,
108                                               new EuclideanDistance());
109 
110         final long[] expected = new long[] { 3, 2, 1, 0 };
111         for (int i = 0; i < list.size(); i++) {
112             Assert.assertEquals(expected[i], sorted[i].getIdentifier());
113         }
114     }
