175     @Test
176     public void testParseIgnoredWhitespace() {
177         ArrayRealVector expected = new ArrayRealVector(new double[] {1, 1, 1});
178         ParsePosition pos1 = new ParsePosition(0);
179         String source1 = "{1;1;1}";
180         Assert.assertEquals(expected, realVectorFormat.parse(source1, pos1));
181         Assert.assertEquals(source1.length(), pos1.getIndex());
182         ParsePosition pos2 = new ParsePosition(0);
183         String source2 = " { 1 ; 1 ; 1 } ";
184         Assert.assertEquals(expected, realVectorFormat.parse(source2, pos2));
185         Assert.assertEquals(source2.length() - 1, pos2.getIndex());
186     }
