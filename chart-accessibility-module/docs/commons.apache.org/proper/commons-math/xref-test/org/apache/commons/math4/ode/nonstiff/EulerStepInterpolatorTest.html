48    @Test
49    public void noReset() throws MaxCountExceededException {
50  
51      double[]   y    =   { 0.0, 1.0, -2.0 };
52      double[][] yDot = { { 1.0, 2.0, -2.0 } };
53      EulerStepInterpolator interpolator = new EulerStepInterpolator();
54      interpolator.reinitialize(new DummyIntegrator(interpolator), y, yDot, true,
55                                new EquationsMapper(0, y.length),
56                                new EquationsMapper[0]);
57      interpolator.storeTime(0);
58      interpolator.shift();
59      interpolator.storeTime(1);
60  
61      double[] result = interpolator.getInterpolatedState();
62      for (int i = 0; i < result.length; ++i) {
63        Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
64      }
65  
66    }

68    @Test
69    public void interpolationAtBounds() throws MaxCountExceededException {
70  
71      double   t0 = 0;
72      double[] y0 = {0.0, 1.0, -2.0};
73  
74      double[] y = y0.clone();
75      double[][] yDot = { new double[y0.length] };
76      EulerStepInterpolator interpolator = new EulerStepInterpolator();
77      interpolator.reinitialize(new DummyIntegrator(interpolator), y, yDot, true,
78                                new EquationsMapper(0, y.length),
79                                new EquationsMapper[0]);
80      interpolator.storeTime(t0);
81  
82      double dt = 1.0;
83      interpolator.shift();
84      y[0] =  1.0;
85      y[1] =  3.0;
86      y[2] = -4.0;
87      yDot[0][0] = (y[0] - y0[0]) / dt;
88      yDot[0][1] = (y[1] - y0[1]) / dt;
89      yDot[0][2] = (y[2] - y0[2]) / dt;
90      interpolator.storeTime(t0 + dt);
91  
92      interpolator.setInterpolatedTime(interpolator.getPreviousTime());
93      double[] result = interpolator.getInterpolatedState();
94      for (int i = 0; i < result.length; ++i) {
95          Assert.assertTrue(FastMath.abs(result[i] - y0[i]) < 1.0e-10);
96      }
97  
98      interpolator.setInterpolatedTime(interpolator.getCurrentTime());
99      result = interpolator.getInterpolatedState();
100     for (int i = 0; i < result.length; ++i) {
101       Assert.assertTrue(FastMath.abs(result[i] - y[i]) < 1.0e-10);
102     }
103 
104   }

106   @Test
107   public void interpolationInside() throws MaxCountExceededException {
108 
109     double[]   y    =   { 0.0, 1.0, -2.0 };
110     double[][] yDot = { { 1.0, 2.0, -2.0 } };
111     EulerStepInterpolator interpolator = new EulerStepInterpolator();
112     interpolator.reinitialize(new DummyIntegrator(interpolator), y, yDot, true,
113                               new EquationsMapper(0, y.length),
114                               new EquationsMapper[0]);
115     interpolator.storeTime(0);
116     interpolator.shift();
117     y[0] =  1.0;
118     y[1] =  3.0;
119     y[2] = -4.0;
120     interpolator.storeTime(1);
121 
122     interpolator.setInterpolatedTime(0.1);
123     double[] result = interpolator.getInterpolatedState();
124     Assert.assertTrue(FastMath.abs(result[0] - 0.1) < 1.0e-10);
125     Assert.assertTrue(FastMath.abs(result[1] - 1.2) < 1.0e-10);
126     Assert.assertTrue(FastMath.abs(result[2] + 2.2) < 1.0e-10);
127 
128     interpolator.setInterpolatedTime(0.5);
129     result = interpolator.getInterpolatedState();
130     Assert.assertTrue(FastMath.abs(result[0] - 0.5) < 1.0e-10);
131     Assert.assertTrue(FastMath.abs(result[1] - 2.0) < 1.0e-10);
132     Assert.assertTrue(FastMath.abs(result[2] + 3.0) < 1.0e-10);
133

146   @Test
147   public void serialization()
148     throws IOException, ClassNotFoundException,
149            DimensionMismatchException, NumberIsTooSmallException,
150            MaxCountExceededException, NoBracketingException {
151 
152     TestProblem1 pb = new TestProblem1();
153     double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
154     EulerIntegrator integ = new EulerIntegrator(step);
155     integ.addStepHandler(new ContinuousOutputModel());
156     integ.integrate(pb,
157                     pb.getInitialTime(), pb.getInitialState(),
158                     pb.getFinalTime(), new double[pb.getDimension()]);
159 
160     ByteArrayOutputStream bos = new ByteArrayOutputStream();
161     ObjectOutputStream    oos = new ObjectOutputStream(bos);
162     for (StepHandler handler : integ.getStepHandlers()) {
163         oos.writeObject(handler);
164     }
165 
166     ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
167     ObjectInputStream     ois = new ObjectInputStream(bis);
168     ContinuousOutputModel cm  = (ContinuousOutputModel) ois.readObject();
169 
170     Random random = new Random(347588535632l);
171     double maxError = 0.0;
172     for (int i = 0; i < 1000; ++i) {
173       double r = random.nextDouble();
174       double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
175       cm.setInterpolatedTime(time);
176       double[] interpolatedY = cm.getInterpolatedState ();
177       double[] theoreticalY  = pb.computeTheoreticalState(time);
178       double dx = interpolatedY[0] - theoreticalY[0];
179       double dy = interpolatedY[1] - theoreticalY[1];
180       double error = dx * dx + dy * dy;
181       if (error > maxError) {
182         maxError = error;
183       }
184     }
185     Assert.assertTrue(maxError < 0.001);
186 
187   }
