17  package org.apache.commons.math4.fitting.leastsquares;
18  
19  import java.util.ArrayList;
20  
21  import org.apache.commons.math4.analysis.MultivariateMatrixFunction;
22  import org.apache.commons.math4.analysis.MultivariateVectorFunction;
23  import org.apache.commons.math4.geometry.euclidean.twod.Vector2D;
24  
25  /**
26   * Class used in the tests.
27   */
28  class CircleVectorial {
29      private ArrayList<Vector2D> points;
30  
31      public CircleVectorial() {
32          points  = new ArrayList<Vector2D>();
33      }
34  
35      public void addPoint(double px, double py) {
36          points.add(new Vector2D(px, py));
37      }
38  
39      public int getN() {
40          return points.size();
41      }
42  
43      public double getRadius(Vector2D center) {
44          double r = 0;
45          for (Vector2D point : points) {
46              r += point.distance(center);
47          }
48          return r / points.size();
49      }
50  
51      public MultivariateVectorFunction getModelFunction() {
52          return new MultivariateVectorFunction() {
53              public double[] value(double[] params) {
54                  Vector2D center = new Vector2D(params[0], params[1]);
55                  double radius = getRadius(center);
56                  double[] residuals = new double[points.size()];
57                  for (int i = 0; i < residuals.length; i++) {
58                      residuals[i] = points.get(i).distance(center) - radius;
59                  }
60  
61                  return residuals;
62              }
63          };
64      }
65  
66      public MultivariateMatrixFunction getModelFunctionJacobian() {
67          return new MultivariateMatrixFunction() {
68              public double[][] value(double[] params) {
69                  final int n = points.size();
70                  final Vector2D center = new Vector2D(params[0], params[1]);
71  
72                  double dRdX = 0;
73                  double dRdY = 0;
74                  for (Vector2D pk : points) {
75                      double dk = pk.distance(center);
76                      dRdX += (center.getX() - pk.getX()) / dk;
77                      dRdY += (center.getY() - pk.getY()) / dk;
78                  }
79                  dRdX /= n;
80                  dRdY /= n;
81  
82                  // Jacobian of the radius residuals.
83                  double[][] jacobian = new double[n][2];
84                  for (int i = 0; i < n; i++) {
85                      final Vector2D pi = points.get(i);
86                      final double di = pi.distance(center);
87                      jacobian[i][0] = (center.getX() - pi.getX()) / di - dRdX;
88                      jacobian[i][1] = (center.getY() - pi.getY()) / di - dRdY;
89                  }
90  
91                  return jacobian;
92              }
93          };
94      }
95  }
