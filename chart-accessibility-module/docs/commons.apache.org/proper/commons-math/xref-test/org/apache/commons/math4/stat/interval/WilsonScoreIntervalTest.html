22  import org.junit.Assert;
23  import org.junit.Test;

25  /**
26   * Test cases for the WilsonScoreInterval class.
27   *
28   */
29  public class WilsonScoreIntervalTest extends BinomialConfidenceIntervalAbstractTest {
30  
31      @Override
32      protected BinomialConfidenceInterval createBinomialConfidenceInterval() {
33          return new WilsonScoreInterval();

36      @Test
37      public void testStandardInterval() {
38          ConfidenceInterval confidenceInterval = createStandardTestInterval();

40          Assert.assertEquals(0.1242664, confidenceInterval.getUpperBound(), 1E-5);
41      }
