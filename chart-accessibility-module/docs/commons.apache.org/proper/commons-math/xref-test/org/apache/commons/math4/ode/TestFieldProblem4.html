18  package org.apache.commons.math4.ode;
19  
20  import java.lang.reflect.Array;
21  
22  import org.apache.commons.math4.Field;
23  import org.apache.commons.math4.RealFieldElement;
24  import org.apache.commons.math4.ode.events.Action;
25  import org.apache.commons.math4.ode.events.FieldEventHandler;
26  import org.apache.commons.math4.util.FastMath;
27  import org.apache.commons.math4.util.MathArrays;
28  
29  /**
30   * This class is used in the junit tests for the ODE integrators.
31  
32   * <p>This specific problem is the following differential equation :
33   * <pre>
34   *    x'' = -x
35   * </pre>
36   * And when x decreases down to 0, the state should be changed as follows :
37   * <pre>
38   *   x' -> -x'
39   * </pre>
40   * The theoretical solution of this problem is x = |sin(t+a)|
41   * </p>
42  
43   * @param <T> the type of the field elements
44   */
45  public class TestFieldProblem4<T extends RealFieldElement<T>>
46      extends TestFieldProblemAbstract<T> {
47  
48      /** Time offset. */
49      private T a;
50  
51      /** Simple constructor.
52       * @param field field to which elements belong
53       */
54      public TestFieldProblem4(Field<T> field) {
55          super(field);
56          a = convert(1.2);
57          T[] y0 = MathArrays.buildArray(field, 2);
58          y0[0] = a.sin();
59          y0[1] = a.cos();;
60          setInitialConditions(convert(0.0), y0);
61          setFinalConditions(convert(15));
62          setErrorScale(convert(1.0, 0.0));
63      }
64  
65      @Override
66      public FieldEventHandler<T>[] getEventsHandlers() {
67          @SuppressWarnings("unchecked")
68          FieldEventHandler<T>[] handlers =
69                          (FieldEventHandler<T>[]) Array.newInstance(FieldEventHandler.class, 2);
70          handlers[0] = new Bounce<T>();
71          handlers[1] = new Stop<T>();
72          return handlers;
73      }
74  
75      /**
76       * Get the theoretical events times.
77       * @return theoretical events times
78       */
79      @Override
80      public T[] getTheoreticalEventsTimes() {
81          T[] array = MathArrays.buildArray(getField(), 5);
82          array[0] = a.negate().add(1 * FastMath.PI);
83          array[1] = a.negate().add(2 * FastMath.PI);
84          array[2] = a.negate().add(3 * FastMath.PI);
85          array[3] = a.negate().add(4 * FastMath.PI);
86          array[4] = convert(120.0);
87          return array;
88      }
89  
90      @Override
91      public T[] doComputeDerivatives(T t, T[] y) {
92          final T[] yDot = MathArrays.buildArray(getField(), getDimension());
93          yDot[0] = y[1];
94          yDot[1] = y[0].negate();
95          return yDot;
96      }
97  
98      @Override
99      public T[] computeTheoreticalState(T t) {
100         T sin = t.add(a).sin();
101         T cos = t.add(a).cos();
102         final T[] y = MathArrays.buildArray(getField(), getDimension());
103         y[0] = sin.abs();
104         y[1] = (sin.getReal() >= 0) ? cos : cos.negate();
105         return y;
106     }
107 
108     private static class Bounce<T extends RealFieldElement<T>> implements FieldEventHandler<T> {
109 
110         private int sign;
111 
112         public Bounce() {
113             sign = +1;
114         }
115 
116         public void init(FieldODEStateAndDerivative<T> state0, T t) {
117         }
118 
119         public T g(FieldODEStateAndDerivative<T> state) {
120             return state.getState()[0].multiply(sign);
121         }
122 
123         public Action eventOccurred(FieldODEStateAndDerivative<T> state, boolean increasing) {
124             // this sign change is needed because the state will be reset soon
125             sign = -sign;
126             return Action.RESET_STATE;
127         }
128 
129         public FieldODEState<T> resetState(FieldODEStateAndDerivative<T> state) {
130             T[] y = state.getState();
131             y[0] = y[0].negate();
132             y[1] = y[1].negate();
133             return new FieldODEState<T>(state.getTime(), y);
134         }
135 
136     }
137 
138     private static class Stop<T extends RealFieldElement<T>> implements FieldEventHandler<T> {
139 
140         public Stop() {
141         }
142 
143         public void init(FieldODEStateAndDerivative<T> state0, T t) {
144         }
145 
146         public T g(FieldODEStateAndDerivative<T> state) {
147             return state.getTime().subtract(12.0);
148         }
149 
150         public Action eventOccurred(FieldODEStateAndDerivative<T> state, boolean increasing) {
151             return Action.STOP;
152         }
153 
154         public FieldODEState<T> resetState(FieldODEStateAndDerivative<T> state) {
155             return state;
156         }
157 
158     }
