18  package org.apache.commons.math4.random;
19  
20  import org.apache.commons.math4.linear.RealMatrix;
21  import org.apache.commons.rng.RandomSource;
22  import org.apache.commons.math4.random.GaussianRandomGenerator;
23  import org.apache.commons.math4.random.UncorrelatedRandomVectorGenerator;
24  import org.apache.commons.math4.stat.descriptive.moment.VectorialCovariance;
25  import org.apache.commons.math4.stat.descriptive.moment.VectorialMean;
26  import org.junit.Test;
27  import org.junit.Assert;
28  
29  public class UncorrelatedRandomVectorGeneratorTest {
30      private double[] mean;
31      private double[] standardDeviation;
32      private UncorrelatedRandomVectorGenerator generator;
33  
34      public UncorrelatedRandomVectorGeneratorTest() {
35          mean              = new double[] {0.0, 1.0, -3.0, 2.3};
36          standardDeviation = new double[] {1.0, 2.0, 10.0, 0.1};
37          generator =
38              new UncorrelatedRandomVectorGenerator(mean, standardDeviation,
39                                                    new GaussianRandomGenerator(RandomSource.create(RandomSource.MT,
40                                                                                                    17399225432l)));
41      }
42  
43      @Test
44      public void testMeanAndCorrelation() {
45  
46          VectorialMean meanStat = new VectorialMean(mean.length);
47          VectorialCovariance covStat = new VectorialCovariance(mean.length, true);
48          for (int i = 0; i < 10000; ++i) {
49              double[] v = generator.nextVector();
50              meanStat.increment(v);
51              covStat.increment(v);
52          }
53  
54          double[] estimatedMean = meanStat.getResult();
55          double scale;
56          RealMatrix estimatedCorrelation = covStat.getResult();
57          for (int i = 0; i < estimatedMean.length; ++i) {
58              Assert.assertEquals(mean[i], estimatedMean[i], 0.07);
59              for (int j = 0; j < i; ++j) {
60                  scale = standardDeviation[i] * standardDeviation[j];
61                  Assert.assertEquals(0, estimatedCorrelation.getEntry(i, j) / scale, 0.03);
62              }
63              scale = standardDeviation[i] * standardDeviation[i];
64              Assert.assertEquals(1, estimatedCorrelation.getEntry(i, i) / scale, 0.025);
65          }
66      }
67  }
