37      @Test
38      public void testNonUnitWeightSum() {
39          final double[] weights = { 1, 2 };
40          final double[][] means = { { -1.5, 2.0 },
41                                     { 4.0, 8.2 } };
42          final double[][][] covariances = { { { 2.0, -1.1 },
43                                               { -1.1, 2.0 } },
44                                             { { 3.5, 1.5 },
45                                               { 1.5, 3.5 } } };
46          final MixtureMultivariateNormalDistribution d
47              = new MixtureMultivariateNormalDistribution(weights, means, covariances);
48  
49          final List<Pair<Double, MultivariateNormalDistribution>> comp = d.getComponents();
50  
51          Assert.assertEquals(1d / 3, comp.get(0).getFirst().doubleValue(), Math.ulp(1d));
52          Assert.assertEquals(2d / 3, comp.get(1).getFirst().doubleValue(), Math.ulp(1d));
53      }

79      /**
80       * Test the accuracy of the density calculation.
81       */
82      @Test
83      public void testDensities() {
84          final double[] weights = { 0.3, 0.7 };
85          final double[][] means = { { -1.5, 2.0 },
86                                     { 4.0, 8.2 } };
87          final double[][][] covariances = { { { 2.0, -1.1 },
88                                               { -1.1, 2.0 } },
89                                             { { 3.5, 1.5 },
90                                               { 1.5, 3.5 } } };
91          final MixtureMultivariateNormalDistribution d
92              = new MixtureMultivariateNormalDistribution(weights, means, covariances);
93  
94          // Test vectors
95          final double[][] testValues = { { -1.5, 2 },
96                                          { 4, 8.2 },
97                                          { 1.5, -2 },
98                                          { 0, 0 } };
99  
100         // Densities that we should get back.
101         // Calculated by assigning weights to multivariate normal distribution
102         // and summing
103         // values from dmvnorm function in R 2.15 CRAN package Mixtools v0.4.
104         // Like: .3*dmvnorm(val,mu1,sigma1)+.7*dmvnorm(val,mu2,sigma2)
105         final double[] correctDensities = { 0.02862037278930575,
106                                             0.03523044847314091,
107                                             0.000416241365629767,
108                                             0.009932042831700297 };
109 
110         for (int i = 0; i < testValues.length; i++) {
111             Assert.assertEquals(correctDensities[i], d.density(testValues[i]), Math.ulp(1d));
112         }
113     }

115     /**
116      * Test the accuracy of sampling from the distribution.
117      */
118     @Ignore@Test
119     public void testSampling() {
120         final double[] weights = { 0.3, 0.7 };
121         final double[][] means = { { -1.5, 2.0 },
122                                    { 4.0, 8.2 } };
123         final double[][][] covariances = { { { 2.0, -1.1 },
124                                              { -1.1, 2.0 } },
125                                            { { 3.5, 1.5 },
126                                              { 1.5, 3.5 } } };
127         final MixtureMultivariateNormalDistribution d =
128             new MixtureMultivariateNormalDistribution(weights, means, covariances);
129         final MultivariateRealDistribution.Sampler sampler =
130             d.createSampler(RandomSource.create(RandomSource.WELL_19937_C, 50));
131 
132         final double[][] correctSamples = getCorrectSamples();
133         final int n = correctSamples.length;
134         final double[][] samples = AbstractMultivariateRealDistribution.sample(n, sampler);
135 
136         for (int i = 0; i < n; i++) {
137             for (int j = 0; j < samples[i].length; j++) {
138                 Assert.assertEquals("sample[" + j + "]",
139                                     correctSamples[i][j], samples[i][j], 1e-16);
140             }
141         }
142     }
