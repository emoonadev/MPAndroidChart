34      @Test
35      public void testSinFunction() {
36          UnivariateFunction f = new Sin();
37          BaseAbstractUnivariateIntegrator integrator
38              = new IterativeLegendreGaussIntegrator(5, 1.0e-14, 1.0e-10, 2, 15);
39          double min, max, expected, result, tolerance;
40  
41          min = 0; max = FastMath.PI; expected = 2;
42          tolerance = FastMath.max(integrator.getAbsoluteAccuracy(),
43                               FastMath.abs(expected * integrator.getRelativeAccuracy()));
44          result = integrator.integrate(10000, f, min, max);
45          Assert.assertEquals(expected, result, tolerance);
46  
47          min = -FastMath.PI/3; max = 0; expected = -0.5;
48          tolerance = FastMath.max(integrator.getAbsoluteAccuracy(),
49                  FastMath.abs(expected * integrator.getRelativeAccuracy()));
50          result = integrator.integrate(10000, f, min, max);
51          Assert.assertEquals(expected, result, tolerance);
52      }

55      public void testQuinticFunction() {
56          UnivariateFunction f = new QuinticFunction();
57          UnivariateIntegrator integrator =
58                  new IterativeLegendreGaussIntegrator(3,
59                                                       BaseAbstractUnivariateIntegrator.DEFAULT_RELATIVE_ACCURACY,
60                                                       BaseAbstractUnivariateIntegrator.DEFAULT_ABSOLUTE_ACCURACY,
61                                                       BaseAbstractUnivariateIntegrator.DEFAULT_MIN_ITERATIONS_COUNT,
62                                                       64);
63          double min, max, expected, result;
64  
65          min = 0; max = 1; expected = -1.0/48;
66          result = integrator.integrate(10000, f, min, max);
67          Assert.assertEquals(expected, result, 1.0e-16);
68  
69          min = 0; max = 0.5; expected = 11.0/768;
70          result = integrator.integrate(10000, f, min, max);
71          Assert.assertEquals(expected, result, 1.0e-16);
72  
73          min = -1; max = 4; expected = 2048/3.0 - 78 + 1.0/48;
74          result = integrator.integrate(10000, f, min, max);
75          Assert.assertEquals(expected, result, 1.0e-16);
76      }

78      @Test
79      public void testExactIntegration() {
80          Random random = new Random(86343623467878363l);
81          for (int n = 2; n < 6; ++n) {
82              IterativeLegendreGaussIntegrator integrator =
83                  new IterativeLegendreGaussIntegrator(n,
84                                                       BaseAbstractUnivariateIntegrator.DEFAULT_RELATIVE_ACCURACY,
85                                                       BaseAbstractUnivariateIntegrator.DEFAULT_ABSOLUTE_ACCURACY,
86                                                       BaseAbstractUnivariateIntegrator.DEFAULT_MIN_ITERATIONS_COUNT,
87                                                       64);
88  
89              // an n points Gauss-Legendre integrator integrates 2n-1 degree polynoms exactly
90              for (int degree = 0; degree <= 2 * n - 1; ++degree) {
91                  for (int i = 0; i < 10; ++i) {
92                      double[] coeff = new double[degree + 1];
93                      for (int k = 0; k < coeff.length; ++k) {
94                          coeff[k] = 2 * random.nextDouble() - 1;
95                      }
96                      PolynomialFunction p = new PolynomialFunction(coeff);
97                      double result    = integrator.integrate(10000, p, -5.0, 15.0);
98                      double reference = exactIntegration(p, -5.0, 15.0);
99                      Assert.assertEquals(n + " " + degree + " " + i, reference, result, 1.0e-12 * (1.0 + FastMath.abs(reference)));
100                 }
101             }
102 
103         }
104     }

124     @Test
125     public void testIssue464() {
126         final double value = 0.2;
127         UnivariateFunction f = new UnivariateFunction() {
128             @Override
129             public double value(double x) {
130                 return (x >= 0 && x <= 5) ? value : 0.0;
131             }
132         };
133         IterativeLegendreGaussIntegrator gauss
134             = new IterativeLegendreGaussIntegrator(5, 3, 100);
135 
136         // due to the discontinuity, integration implies *many* calls
137         double maxX = 0.32462367623786328;
138         Assert.assertEquals(maxX * value, gauss.integrate(Integer.MAX_VALUE, f, -10, maxX), 1.0e-7);
139         Assert.assertTrue(gauss.getEvaluations() > 37000000);
140         Assert.assertTrue(gauss.getIterations() < 30);
141 
142         // setting up limits prevents such large number of calls
143         try {
144             gauss.integrate(1000, f, -10, maxX);
145             Assert.fail("expected TooManyEvaluationsException");
146         } catch (TooManyEvaluationsException tmee) {
147             // expected
148             Assert.assertEquals(1000, tmee.getMax());
149         }
150 
151         // integrating on the two sides should be simpler
152         double sum1 = gauss.integrate(1000, f, -10, 0);
153         int eval1   = gauss.getEvaluations();
154         double sum2 = gauss.integrate(1000, f, 0, maxX);
155         int eval2   = gauss.getEvaluations();
156         Assert.assertEquals(maxX * value, sum1 + sum2, 1.0e-7);
157         Assert.assertTrue(eval1 + eval2 < 200);
158 
159     }

161     private double exactIntegration(PolynomialFunction p, double a, double b) {
162         final double[] coeffs = p.getCoefficients();
163         double yb = coeffs[coeffs.length - 1] / coeffs.length;
164         double ya = yb;
165         for (int i = coeffs.length - 2; i >= 0; --i) {
166             yb = yb * b + coeffs[i] / (i + 1);
167             ya = ya * a + coeffs[i] / (i + 1);
168         }
169         return yb * b - ya * a;
170     }
