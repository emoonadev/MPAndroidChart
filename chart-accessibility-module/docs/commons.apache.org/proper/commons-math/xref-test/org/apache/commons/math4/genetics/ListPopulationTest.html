35      @Test
36      public void testGetFittestChromosome() {
37          Chromosome c1 = new Chromosome() {
38              public double fitness() {
39                  return 0;
40              }
41          };
42          Chromosome c2 = new Chromosome() {
43              public double fitness() {
44                  return 10;
45              }
46          };
47          Chromosome c3 = new Chromosome() {
48              public double fitness() {
49                  return 15;
50              }
51          };
52  
53          ArrayList<Chromosome> chromosomes = new ArrayList<Chromosome> ();
54          chromosomes.add(c1);
55          chromosomes.add(c2);
56          chromosomes.add(c3);
57  
58          ListPopulation population = new ListPopulation(chromosomes, 10) {
59              public Population nextGeneration() {
60                  // not important
61                  return null;
62              }
63          };
64  
65          Assert.assertEquals(c3, population.getFittestChromosome());
66      }

68      @Test
69      public void testChromosomes() {
70          final ArrayList<Chromosome> chromosomes = new ArrayList<Chromosome> ();
71          chromosomes.add(new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(3)));
72          chromosomes.add(new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(3)));
73          chromosomes.add(new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(3)));
74  
75          final ListPopulation population = new ListPopulation(10) {
76              public Population nextGeneration() {
77                  // not important
78                  return null;
79              }
80          };
81  
82          population.addChromosomes(chromosomes);
83  
84          Assert.assertEquals(chromosomes, population.getChromosomes());
85          Assert.assertEquals(chromosomes.toString(), population.toString());
86  
87          population.setPopulationLimit(50);
88          Assert.assertEquals(50, population.getPopulationLimit());
89      }

175         chromosomes.add(new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(3)));
176         chromosomes.add(new DummyBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(3)));
177 
178         final ListPopulation population = new ListPopulation(10) {
179             public Population nextGeneration() {
180                 // not important
181                 return null;
182             }
183         };
184 
185         population.addChromosomes(chromosomes);
186 
187         final Iterator<Chromosome> iter = population.iterator();
188         while (iter.hasNext()) {
189             iter.next();
190             iter.remove();
191         }
