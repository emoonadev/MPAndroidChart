18  package org.apache.commons.math4.ode;
19  
20  import org.apache.commons.math4.Field;
21  import org.apache.commons.math4.RealFieldElement;
22  
23  /**
24   * This class is used in the junit tests for the ODE integrators.
25   * <p>This is the same as problem 1 except integration is done
26   * backward in time</p>
27   * @param <T> the type of the field elements
28   */
29  public class TestFieldProblem5<T extends RealFieldElement<T>>
30      extends TestFieldProblem1<T> {
31  
32      /**
33       * Simple constructor.
34       * @param field field to which elements belong
35       */
36      public TestFieldProblem5(Field<T> field) {
37          super(field);
38          setFinalConditions(getInitialState().getTime().multiply(2).subtract(getFinalTime()));
39      }
40  
41  }
