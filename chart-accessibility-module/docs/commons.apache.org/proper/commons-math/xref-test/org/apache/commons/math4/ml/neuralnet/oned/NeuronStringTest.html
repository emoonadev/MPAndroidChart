43      /*
44       * Test assumes that the network is
45       *
46       *  0-----1-----2-----3
47       */
48      @Test
49      public void testSegmentNetwork() {
50          final FeatureInitializer[] initArray = { init };
51          final Network net = new NeuronString(4, false, initArray).getNetwork();
52  
53          Collection<Neuron> neighbours;
54  
55          // Neuron 0.
56          neighbours = net.getNeighbours(net.getNeuron(0));
57          for (long nId : new long[] { 1 }) {
58              Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
59          }
60          // Ensures that no other neurons is in the neihbourhood set.
61          Assert.assertEquals(1, neighbours.size());
62  
63          // Neuron 1.
64          neighbours = net.getNeighbours(net.getNeuron(1));
65          for (long nId : new long[] { 0, 2 }) {
66              Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
67          }
68          // Ensures that no other neurons is in the neihbourhood set.
69          Assert.assertEquals(2, neighbours.size());
70  
71          // Neuron 2.
72          neighbours = net.getNeighbours(net.getNeuron(2));
73          for (long nId : new long[] { 1, 3 }) {
74              Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
75          }
76          // Ensures that no other neurons is in the neihbourhood set.
77          Assert.assertEquals(2, neighbours.size());
78  
79          // Neuron 3.
80          neighbours = net.getNeighbours(net.getNeuron(3));
81          for (long nId : new long[] { 2 }) {
82              Assert.assertTrue(neighbours.contains(net.getNeuron(nId)));
83          }
84          // Ensures that no other neurons is in the neihbourhood set.
85          Assert.assertEquals(1, neighbours.size());
86      }

138     @Test
139     public void testGetNeighboursWithExclude() {
140         final FeatureInitializer[] initArray = { init };
141         final Network net = new NeuronString(5, true, initArray).getNetwork();
142         final Collection<Neuron> exclude = new ArrayList<Neuron>();
143         exclude.add(net.getNeuron(1));
144         final Collection<Neuron> neighbours = net.getNeighbours(net.getNeuron(0),
145                                                                 exclude);
146         Assert.assertTrue(neighbours.contains(net.getNeuron(4)));
147         Assert.assertEquals(1, neighbours.size());
148     }

150     @Test
151     public void testSerialize()
152         throws IOException,
153                ClassNotFoundException {
154         final FeatureInitializer[] initArray = { init };
155         final NeuronString out = new NeuronString(4, false, initArray);
156 
157         final ByteArrayOutputStream bos = new ByteArrayOutputStream();
158         final ObjectOutputStream oos = new ObjectOutputStream(bos);
159         oos.writeObject(out);
160 
161         final ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
162         final ObjectInputStream ois = new ObjectInputStream(bis);
163         final NeuronString in = (NeuronString) ois.readObject();
164 
165         for (Neuron nOut : out.getNetwork()) {
166             final Neuron nIn = in.getNetwork().getNeuron(nOut.getIdentifier());
167 
168             // Same values.
169             final double[] outF = nOut.getFeatures();
170             final double[] inF = nIn.getFeatures();
171             Assert.assertEquals(outF.length, inF.length);
172             for (int i = 0; i < outF.length; i++) {
173                 Assert.assertEquals(outF[i], inF[i], 0d);
174             }
175 
176             // Same neighbours.
177             final Collection<Neuron> outNeighbours = out.getNetwork().getNeighbours(nOut);
178             final Collection<Neuron> inNeighbours = in.getNetwork().getNeighbours(nIn);
179             Assert.assertEquals(outNeighbours.size(), inNeighbours.size());
180             for (Neuron oN : outNeighbours) {
181                 Assert.assertTrue(inNeighbours.contains(in.getNetwork().getNeuron(oN.getIdentifier())));
182             }
183         }
184     }
