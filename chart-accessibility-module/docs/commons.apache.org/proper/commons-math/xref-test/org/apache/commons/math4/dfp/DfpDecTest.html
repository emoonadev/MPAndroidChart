50      // Generic test function.  Takes params x and y and tests them for
51      // equality.  Then checks the status flags against the flags argument.
52      // If the test fail, it prints the desc string
53      private void test(Dfp x, Dfp y, int flags, String desc) {
54          boolean b = x.equals(y);
55  
56          if (!x.equals(y) && !x.unequal(y))  // NaNs involved
57              b = (x.toString().equals(y.toString()));
58  
59          if (x.equals(new DfpDec(field, 0)))  // distinguish +/- zero
60              b = (b && (x.toString().equals(y.toString())));
61  
62          b = (b && x.getField().getIEEEFlags() == flags);
63  
64          if (!b)
65              Assert.assertTrue("assersion failed "+desc+" x = "+x.toString()+" flags = "+x.getField().getIEEEFlags(), b);
66  
67          x.getField().clearIEEEFlags();
68      }
