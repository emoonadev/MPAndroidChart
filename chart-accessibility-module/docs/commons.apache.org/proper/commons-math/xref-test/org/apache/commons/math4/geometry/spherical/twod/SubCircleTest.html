35      @Test
36      public void testFullCircle() {
37          Circle circle = new Circle(Vector3D.PLUS_K, 1.0e-10);
38          SubCircle set = circle.wholeHyperplane();
39          Assert.assertEquals(MathUtils.TWO_PI, set.getSize(), 1.0e-10);
40          Assert.assertTrue(circle == set.getHyperplane());
41          Assert.assertTrue(circle != set.copySelf().getHyperplane());
42      }

44      @Test
45      public void testSide() {
46  
47          Circle xzPlane = new Circle(Vector3D.PLUS_J, 1.0e-10);
48  
49          SubCircle sc1 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 1.0, 3.0, 5.0, 6.0);
50          Assert.assertEquals(Side.BOTH, sc1.split(xzPlane).getSide());
51  
52          SubCircle sc2 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 1.0, 3.0);
53          Assert.assertEquals(Side.MINUS, sc2.split(xzPlane).getSide());
54  
55          SubCircle sc3 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 5.0, 6.0);
56          Assert.assertEquals(Side.PLUS, sc3.split(xzPlane).getSide());
57  
58          SubCircle sc4 = create(Vector3D.PLUS_J, Vector3D.PLUS_K, Vector3D.PLUS_I, 1.0e-10, 5.0, 6.0);
59          Assert.assertEquals(Side.HYPER, sc4.split(xzPlane).getSide());
60  
61          SubCircle sc5 = create(Vector3D.MINUS_J, Vector3D.PLUS_I, Vector3D.PLUS_K, 1.0e-10, 5.0, 6.0);
62          Assert.assertEquals(Side.HYPER, sc5.split(xzPlane).getSide());
63  
64      }

66      @Test
67      public void testSPlit() {
68  
69          Circle xzPlane = new Circle(Vector3D.PLUS_J, 1.0e-10);
70  
71          SubCircle sc1 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 1.0, 3.0, 5.0, 6.0);
72          SplitSubHyperplane<Sphere2D> split1 = sc1.split(xzPlane);
73          ArcsSet plus1  = (ArcsSet) ((SubCircle) split1.getPlus()).getRemainingRegion();
74          ArcsSet minus1 = (ArcsSet) ((SubCircle) split1.getMinus()).getRemainingRegion();
75          Assert.assertEquals(1, plus1.asList().size());
76          Assert.assertEquals(5.0, plus1.asList().get(0).getInf(), 1.0e-10);
77          Assert.assertEquals(6.0, plus1.asList().get(0).getSup(), 1.0e-10);
78          Assert.assertEquals(1, minus1.asList().size());
79          Assert.assertEquals(1.0, minus1.asList().get(0).getInf(), 1.0e-10);
80          Assert.assertEquals(3.0, minus1.asList().get(0).getSup(), 1.0e-10);
81  
82          SubCircle sc2 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 1.0, 3.0);
83          SplitSubHyperplane<Sphere2D> split2 = sc2.split(xzPlane);
84          Assert.assertNull(split2.getPlus());
85          ArcsSet minus2 = (ArcsSet) ((SubCircle) split2.getMinus()).getRemainingRegion();
86          Assert.assertEquals(1, minus2.asList().size());
87          Assert.assertEquals(1.0, minus2.asList().get(0).getInf(), 1.0e-10);
88          Assert.assertEquals(3.0, minus2.asList().get(0).getSup(), 1.0e-10);
89  
90          SubCircle sc3 = create(Vector3D.PLUS_K, Vector3D.PLUS_I, Vector3D.PLUS_J, 1.0e-10, 5.0, 6.0);
91          SplitSubHyperplane<Sphere2D> split3 = sc3.split(xzPlane);
92          ArcsSet plus3  = (ArcsSet) ((SubCircle) split3.getPlus()).getRemainingRegion();
93          Assert.assertEquals(1, plus3.asList().size());
94          Assert.assertEquals(5.0, plus3.asList().get(0).getInf(), 1.0e-10);
95          Assert.assertEquals(6.0, plus3.asList().get(0).getSup(), 1.0e-10);
96          Assert.assertNull(split3.getMinus());
97  
98          SubCircle sc4 = create(Vector3D.PLUS_J, Vector3D.PLUS_K, Vector3D.PLUS_I, 1.0e-10, 5.0, 6.0);
99          SplitSubHyperplane<Sphere2D> split4 = sc4.split(xzPlane);
100         Assert.assertEquals(Side.HYPER, sc4.split(xzPlane).getSide());
101         Assert.assertNull(split4.getPlus());
102         Assert.assertNull(split4.getMinus());
103 
104         SubCircle sc5 = create(Vector3D.MINUS_J, Vector3D.PLUS_I, Vector3D.PLUS_K, 1.0e-10, 5.0, 6.0);
105         SplitSubHyperplane<Sphere2D> split5 = sc5.split(xzPlane);
106         Assert.assertEquals(Side.HYPER, sc5.split(xzPlane).getSide());
107         Assert.assertNull(split5.getPlus());
108         Assert.assertNull(split5.getMinus());
109 
110     }

112     @Test
113     public void testSideSplitConsistency() {
114 
115         double tolerance = 1.0e-6;
116         Circle hyperplane = new Circle(new Vector3D(9.738804529764676E-5, -0.6772824575010357, -0.7357230887208355),
117                                        tolerance);
118         SubCircle sub = new SubCircle(new Circle(new Vector3D(2.1793884139073498E-4, 0.9790647032675541, -0.20354915700704285),
119                                                  tolerance),
120                                       new ArcsSet(4.7121441684170700, 4.7125386635004760, tolerance));
121         SplitSubHyperplane<Sphere2D> split = sub.split(hyperplane);
122         Assert.assertNotNull(split.getMinus());
123         Assert.assertNull(split.getPlus());
124         Assert.assertEquals(Side.MINUS, sub.split(hyperplane).getSide());
125 
126     }

128     private SubCircle create(Vector3D pole, Vector3D x, Vector3D y,
129                              double tolerance, double ... limits) {
130         RegionFactory<Sphere1D> factory = new RegionFactory<Sphere1D>();
131         Circle circle = new Circle(pole, tolerance);
132         Circle phased =
133                 (Circle) Circle.getTransform(new Rotation(circle.getXAxis(), circle.getYAxis(), x, y)).apply(circle);
134         ArcsSet set = (ArcsSet) factory.getComplement(new ArcsSet(tolerance));
135         for (int i = 0; i < limits.length; i += 2) {
136             set = (ArcsSet) factory.union(set, new ArcsSet(limits[i], limits[i + 1], tolerance));
137         }
138         return new SubCircle(phased, set);
139     }
