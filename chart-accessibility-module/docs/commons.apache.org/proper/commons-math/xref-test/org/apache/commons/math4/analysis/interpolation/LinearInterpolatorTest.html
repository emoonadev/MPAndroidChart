17  package org.apache.commons.math4.analysis.interpolation;

19  import org.apache.commons.math4.TestUtils;
20  import org.apache.commons.math4.analysis.UnivariateFunction;
21  import org.apache.commons.math4.analysis.interpolation.LinearInterpolator;
22  import org.apache.commons.math4.analysis.interpolation.UnivariateInterpolator;
23  import org.apache.commons.math4.analysis.polynomials.PolynomialFunction;
24  import org.apache.commons.math4.analysis.polynomials.PolynomialSplineFunction;
25  import org.apache.commons.math4.exception.DimensionMismatchException;
26  import org.apache.commons.math4.exception.NonMonotonicSequenceException;
27  import org.apache.commons.math4.exception.NumberIsTooSmallException;
28  import org.junit.Assert;
29  import org.junit.Test;
30  
31  /**
32   * Test the LinearInterpolator.
33   */
34  public class LinearInterpolatorTest {
35  
36      /** error tolerance for spline interpolator value at knot points */
37      protected double knotTolerance = 1E-12;
38  
39      /** error tolerance for interpolating polynomial coefficients */
40      protected double coefficientTolerance = 1E-6;
41  
42      /** error tolerance for interpolated values */
43      protected double interpolationTolerance = 1E-12;
44  
45      @Test
46      public void testInterpolateLinearDegenerateTwoSegment()
47          {
48          double x[] = { 0.0, 0.5, 1.0 };
49          double y[] = { 0.0, 0.5, 1.0 };
50          UnivariateInterpolator i = new LinearInterpolator();
51          UnivariateFunction f = i.interpolate(x, y);
52          verifyInterpolation(f, x, y);
53  
54          // Verify coefficients using analytical values
55          PolynomialFunction polynomials[] = ((PolynomialSplineFunction) f).getPolynomials();
56          double target[] = {y[0], 1d};
57          TestUtils.assertEquals(polynomials[0].getCoefficients(), target, coefficientTolerance);
58          target = new double[]{y[1], 1d};
59          TestUtils.assertEquals(polynomials[1].getCoefficients(), target, coefficientTolerance);
60  
61          // Check interpolation
62          Assert.assertEquals(0.0,f.value(0.0), interpolationTolerance);
63          Assert.assertEquals(0.4,f.value(0.4), interpolationTolerance);
64          Assert.assertEquals(1.0,f.value(1.0), interpolationTolerance);
65      }
66  
67      @Test
68      public void testInterpolateLinearDegenerateThreeSegment()
69          {
70          double x[] = { 0.0, 0.5, 1.0, 1.5 };
71          double y[] = { 0.0, 0.5, 1.0, 1.5 };
72          UnivariateInterpolator i = new LinearInterpolator();
73          UnivariateFunction f = i.interpolate(x, y);
74          verifyInterpolation(f, x, y);
75  
76          // Verify coefficients using analytical values
77          PolynomialFunction polynomials[] = ((PolynomialSplineFunction) f).getPolynomials();
78          double target[] = {y[0], 1d};
79          TestUtils.assertEquals(polynomials[0].getCoefficients(), target, coefficientTolerance);
80          target = new double[]{y[1], 1d};
81          TestUtils.assertEquals(polynomials[1].getCoefficients(), target, coefficientTolerance);
82          target = new double[]{y[2], 1d};
83          TestUtils.assertEquals(polynomials[2].getCoefficients(), target, coefficientTolerance);
84  
85          // Check interpolation
86          Assert.assertEquals(0,f.value(0), interpolationTolerance);
87          Assert.assertEquals(1.4,f.value(1.4), interpolationTolerance);
88          Assert.assertEquals(1.5,f.value(1.5), interpolationTolerance);
89      }
90  
91      @Test
92      public void testInterpolateLinear() {
93          double x[] = { 0.0, 0.5, 1.0 };
94          double y[] = { 0.0, 0.5, 0.0 };
95          UnivariateInterpolator i = new LinearInterpolator();
96          UnivariateFunction f = i.interpolate(x, y);
97          verifyInterpolation(f, x, y);
98  
99          // Verify coefficients using analytical values
100         PolynomialFunction polynomials[] = ((PolynomialSplineFunction) f).getPolynomials();
101         double target[] = {y[0], 1d};
102         TestUtils.assertEquals(polynomials[0].getCoefficients(), target, coefficientTolerance);
103         target = new double[]{y[1], -1d};
104         TestUtils.assertEquals(polynomials[1].getCoefficients(), target, coefficientTolerance);
105     }
106 
107     @Test
108     public void testIllegalArguments() {
109         // Data set arrays of different size.
110         UnivariateInterpolator i = new LinearInterpolator();
111         try {
112             double xval[] = { 0.0, 1.0 };
113             double yval[] = { 0.0, 1.0, 2.0 };
114             i.interpolate(xval, yval);
115             Assert.fail("Failed to detect data set array with different sizes.");
116         } catch (DimensionMismatchException iae) {
117             // Expected.
118         }
119         // X values not sorted.
120         try {
121             double xval[] = { 0.0, 1.0, 0.5 };
122             double yval[] = { 0.0, 1.0, 2.0 };
123             i.interpolate(xval, yval);
124             Assert.fail("Failed to detect unsorted arguments.");
125         } catch (NonMonotonicSequenceException iae) {
126             // Expected.
127         }
128         // Not enough data to interpolate.
129         try {
130             double xval[] = { 0.0 };
131             double yval[] = { 0.0 };
132             i.interpolate(xval, yval);
133             Assert.fail("Failed to detect unsorted arguments.");
134         } catch (NumberIsTooSmallException iae) {
135             // Expected.
136         }
137     }
138 
139     /**
140      * verifies that f(x[i]) = y[i] for i = 0..n-1 where n is common length.
141      */
142     protected void verifyInterpolation(UnivariateFunction f, double x[], double y[])
143        {
144         for (int i = 0; i < x.length; i++) {
145             Assert.assertEquals(f.value(x[i]), y[i], knotTolerance);
146         }
147     }
