17  package org.apache.commons.math4.random;
18  
19  import java.util.Random;
20  import org.apache.commons.math4.exception.MathUnsupportedOperationException;
21  import org.apache.commons.rng.UniformRandomProvider;
22  import org.apache.commons.rng.RandomSource;
23  import org.apache.commons.math4.distribution.RealDistribution;
24  import org.apache.commons.math4.distribution.NormalDistribution;
25  import org.apache.commons.math4.TestUtils;
26  import org.junit.Assert;
27  import org.junit.Test;
28  
29  /**
30   * Test cases for {@link JDKRandomAdaptor}.
31   */
32  public class JDKRandomAdaptorTest {
33  
34      @Test
35      public void testUniform() {
36          final RandomSource source = RandomSource.WELL_19937_C;
37          final long seed = RandomSource.createLong(); // Random seed.
38  
39          final UniformRandomProvider reference = RandomSource.create(source, seed);
40          final Random random = new JDKRandomAdaptor(RandomSource.create(source, seed));
41  
42          final int n = 3; // Check several times, reusing the same RNG.
43          for (int i = 0; i < n; i++) {
44              checkUniform(reference, random);
45          }
46      }
47  
48      @Test
49      public void testGaussian() {
50          final RandomSource source = RandomSource.WELL_19937_C;
51          final long seed = RandomSource.createLong(); // Random seed.
52  
53          final UniformRandomProvider reference = RandomSource.create(source, seed);
54          final RealDistribution.Sampler s = new NormalDistribution(0, 1).createSampler(reference);
55  
56          final Random random = new JDKRandomAdaptor(RandomSource.create(source, seed));
57  
58          final int n = 11; // Check several times, reusing the same RNG.
59          for (int i = 0; i < n; i++) {
60              Assert.assertEquals(s.sample(), random.nextGaussian(), 0);
61          }
62      }
63  
64      @Test
65      public void testSeedIsIgnored() {
66          final RandomSource source = RandomSource.WELL_19937_C;
67          final long seed = RandomSource.createLong(); // Random seed.
68  
69          Random random;
70  
71          random = new JDKRandomAdaptor(RandomSource.create(source, seed));
72          final double withoutReseed = random.nextDouble();
73  
74          // Same RNG.
75          random = new JDKRandomAdaptor(RandomSource.create(source, seed));
76          final long differentSeed = seed + 1;
77          random.setSeed(differentSeed); // Is seeding ignored?
78          final double withReseed = random.nextDouble();
79  
80          Assert.assertEquals(withoutReseed, withReseed, 0);
81      }
82  
83      @Test
84      public void testSerializeIsNotSupported() {
85          try {
86              TestUtils.serializeAndRecover(new JDKRandomAdaptor(RandomSource.create(RandomSource.WELL_512_A)));
87          } catch (Exception e) {
88              Throwable cause = e.getCause();
89              Assert.assertTrue(cause instanceof MathUnsupportedOperationException);
90          }
91      }
92  
93      /**
94       * Check uniform random generator.
95       *
96       * @param rand1 Reference generator.
97       * @param rand2 Generator under test.
98       */
99      private void checkUniform(UniformRandomProvider rand1,
100                               Random rand2) {
101         final int len = 11;
102         final byte[] bytes1 = new byte[len];
103         final byte[] bytes2 = new byte[len];
104         rand1.nextBytes(bytes1);
105         rand2.nextBytes(bytes2);
106         Assert.assertArrayEquals(bytes1, bytes2);
107 
108         Assert.assertEquals(rand1.nextBoolean(), rand2.nextBoolean());
109 
110         Assert.assertEquals(rand1.nextInt(), rand2.nextInt());
111         Assert.assertEquals(rand1.nextInt(len), rand2.nextInt(len));
112 
113         Assert.assertEquals(rand1.nextLong(), rand2.nextLong());
114 
115         Assert.assertEquals(rand1.nextDouble(), rand2.nextDouble(), 0);
116         Assert.assertEquals(rand1.nextFloat(), rand2.nextFloat(), 0);
117     }
118 }
