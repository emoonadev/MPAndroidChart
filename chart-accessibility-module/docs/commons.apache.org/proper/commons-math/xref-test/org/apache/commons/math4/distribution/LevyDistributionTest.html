17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.distribution.LevyDistribution;
20  import org.apache.commons.math4.util.Precision;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  public class LevyDistributionTest extends RealDistributionAbstractTest {
25  
26      @Test
27      public void testParameters() {
28          LevyDistribution d = makeDistribution();
29          Assert.assertEquals(1.2, d.getLocation(), Precision.EPSILON);
30          Assert.assertEquals(0.4,   d.getScale(),  Precision.EPSILON);
31      }
32  
33      @Test
34      public void testSupport() {
35          LevyDistribution d = makeDistribution();
36          Assert.assertEquals(d.getLocation(), d.getSupportLowerBound(), Precision.EPSILON);
37          Assert.assertTrue(Double.isInfinite(d.getSupportUpperBound()));
38          Assert.assertTrue(d.isSupportConnected());
39      }
40  
41      @Override
42      public LevyDistribution makeDistribution() {
43          return new LevyDistribution(1.2, 0.4);
44      }
45  
46      @Override
47      public double[] makeCumulativeTestPoints() {
48          return new double[] {
49              1.2001, 1.21, 1.225, 1.25, 1.3, 1.9, 3.4, 5.6
50          };
51      }
