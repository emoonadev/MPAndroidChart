47       * JIRA: MATH-305
48       *
49       * Two points, one cluster, one iteration
50       */
51      @Test
52      public void testPerformClusterAnalysisDegenerate() {
53          KMeansPlusPlusClusterer<DoublePoint> transformer =
54                  new KMeansPlusPlusClusterer<DoublePoint>(1, 1);
55  
56          DoublePoint[] points = new DoublePoint[] {
57                  new DoublePoint(new int[] { 1959, 325100 }),
58                  new DoublePoint(new int[] { 1960, 373200 }), };
59          List<? extends Cluster<DoublePoint>> clusters = transformer.cluster(Arrays.asList(points));
60          Assert.assertEquals(1, clusters.size());
61          Assert.assertEquals(2, (clusters.get(0).getPoints().size()));
62          DoublePoint pt1 = new DoublePoint(new int[] { 1959, 325100 });
63          DoublePoint pt2 = new DoublePoint(new int[] { 1960, 373200 });
64          Assert.assertTrue(clusters.get(0).getPoints().contains(pt1));
65          Assert.assertTrue(clusters.get(0).getPoints().contains(pt2));
66  
67      }

69      @Test
70      public void testCertainSpace() {
71          KMeansPlusPlusClusterer.EmptyClusterStrategy[] strategies = {
72              KMeansPlusPlusClusterer.EmptyClusterStrategy.LARGEST_VARIANCE,
73              KMeansPlusPlusClusterer.EmptyClusterStrategy.LARGEST_POINTS_NUMBER,
74              KMeansPlusPlusClusterer.EmptyClusterStrategy.FARTHEST_POINT
75          };
76          for (KMeansPlusPlusClusterer.EmptyClusterStrategy strategy : strategies) {
77              int numberOfVariables = 27;
78              // initialise testvalues
79              int position1 = 1;
80              int position2 = position1 + numberOfVariables;
81              int position3 = position2 + numberOfVariables;
82              int position4 = position3 + numberOfVariables;
83              // testvalues will be multiplied
84              int multiplier = 1000000;
85  
86              DoublePoint[] breakingPoints = new DoublePoint[numberOfVariables];
87              // define the space which will break the cluster algorithm
88              for (int i = 0; i < numberOfVariables; i++) {
89                  int points[] = { position1, position2, position3, position4 };
90                  // multiply the values
91                  for (int j = 0; j < points.length; j++) {
92                      points[j] *= multiplier;
93                  }
94                  DoublePoint DoublePoint = new DoublePoint(points);
95                  breakingPoints[i] = DoublePoint;
96                  position1 += numberOfVariables;
97                  position2 += numberOfVariables;
98                  position3 += numberOfVariables;
99                  position4 += numberOfVariables;
100             }
101 
102             for (int n = 2; n < 27; ++n) {
103                 KMeansPlusPlusClusterer<DoublePoint> transformer =
104                     new KMeansPlusPlusClusterer<DoublePoint>(n, 100, new EuclideanDistance(), random, strategy);
105 
106                 List<? extends Cluster<DoublePoint>> clusters =
107                         transformer.cluster(Arrays.asList(breakingPoints));
108 
109                 Assert.assertEquals(n, clusters.size());
110                 int sum = 0;
111                 for (Cluster<DoublePoint> cluster : clusters) {
112                     sum += cluster.getPoints().size();
113                 }
114                 Assert.assertEquals(numberOfVariables, sum);
115             }
116         }
117 
118     }

120     /**
121      * A helper class for testSmallDistances(). This class is similar to DoublePoint, but
122      * it defines a different distanceFrom() method that tends to return distances less than 1.
123      */
124     private class CloseDistance extends EuclideanDistance {
125         private static final long serialVersionUID = 1L;
126 
127         @Override
128         public double compute(double[] a, double[] b) {
129             return super.compute(a, b) * 0.001;
130         }
131     }

133     /**
134      * Test points that are very close together. See issue MATH-546.
135      */
136     @Test
137     public void testSmallDistances() {
138         // Create a bunch of CloseDoublePoints. Most are identical, but one is different by a
139         // small distance.
140         int[] repeatedArray = { 0 };
141         int[] uniqueArray = { 1 };
142         DoublePoint repeatedPoint = new DoublePoint(repeatedArray);
143         DoublePoint uniquePoint = new DoublePoint(uniqueArray);
144 
145         Collection<DoublePoint> points = new ArrayList<DoublePoint>();
146         final int NUM_REPEATED_POINTS = 10 * 1000;
147         for (int i = 0; i < NUM_REPEATED_POINTS; ++i) {
148             points.add(repeatedPoint);
149         }
150         points.add(uniquePoint);
151 
152         // Ask a KMeansPlusPlusClusterer to run zero iterations (i.e., to simply choose initial
153         // cluster centers).
154         final int NUM_CLUSTERS = 2;
155         final int NUM_ITERATIONS = 0;
156 
157         KMeansPlusPlusClusterer<DoublePoint> clusterer =
158             new KMeansPlusPlusClusterer<DoublePoint>(NUM_CLUSTERS, NUM_ITERATIONS,
159                     new CloseDistance(), random);
160         List<CentroidCluster<DoublePoint>> clusters = clusterer.cluster(points);
161 
162         // Check that one of the chosen centers is the unique point.
163         boolean uniquePointIsCenter = false;
164         for (CentroidCluster<DoublePoint> cluster : clusters) {
165             if (cluster.getCenter().equals(uniquePoint)) {
166                 uniquePointIsCenter = true;
167             }
168         }
169         Assert.assertTrue(uniquePointIsCenter);
170     }
