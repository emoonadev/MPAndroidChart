165     /**
166      * Check whether the estimate of the (updated) residual corresponds to the
167      * exact residual. This fails to be true for a large number of iterations,
168      * due to the loss of orthogonality of the successive search directions.
169      * Therefore, in the present test, the number of iterations is limited.
170      */
171     @Test
172     public void testUnpreconditionedResidual() {
173         final int n = 10;
174         final int maxIterations = n;
175         final RealLinearOperator a = new HilbertMatrix(n);
176         final ConjugateGradient solver;
177         solver = new ConjugateGradient(maxIterations, 1E-15, true);
178         final RealVector r = new ArrayRealVector(n);
179         final RealVector x = new ArrayRealVector(n);
180         final IterationListener listener = new IterationListener() {
181 
182             public void terminationPerformed(final IterationEvent e) {
183                 // Do nothing
184             }
185 
186             public void iterationStarted(final IterationEvent e) {
187                 // Do nothing
188             }
189 
190             public void iterationPerformed(final IterationEvent e) {
191                 final IterativeLinearSolverEvent evt;
192                 evt = (IterativeLinearSolverEvent) e;
193                 RealVector v = evt.getResidual();
194                 r.setSubVector(0, v);
195                 v = evt.getSolution();
196                 x.setSubVector(0, v);
197             }
198 
199             public void initializationPerformed(final IterationEvent e) {
200                 // Do nothing
201             }
202         };
203         solver.getIterationManager().addIterationListener(listener);
204         final RealVector b = new ArrayRealVector(n);
205         for (int j = 0; j < n; j++) {
206             b.set(0.);
207             b.setEntry(j, 1.);
208 
209             boolean caught = false;
210             try {
211                 solver.solve(a, b);
212             } catch (MaxCountExceededException e) {
213                 caught = true;
214                 final RealVector y = a.operate(x);
215                 for (int i = 0; i < n; i++) {
216                     final double actual = b.getEntry(i) - y.getEntry(i);
217                     final double expected = r.getEntry(i);
218                     final double delta = 1E-6 * FastMath.abs(expected);
219                     final String msg = String
220                         .format("column %d, residual %d", i, j);
221                     Assert.assertEquals(msg, expected, actual, delta);
222                 }
223             }
224             Assert
225                 .assertTrue("MaxCountExceededException should have been caught",
226                             caught);
227         }
228     }

399     @Test
400     public void testPreconditionedSolution2() {
401         final int n = 100;
402         final int maxIterations = 100000;
403         final Array2DRowRealMatrix a = new Array2DRowRealMatrix(n, n);
404         double daux = 1.;
405         for (int i = 0; i < n; i++) {
406             a.setEntry(i, i, daux);
407             daux *= 1.2;
408             for (int j = i + 1; j < n; j++) {
409                 if (i == j) {
410                 } else {
411                     final double value = 1.0;
412                     a.setEntry(i, j, value);
413                     a.setEntry(j, i, value);
414                 }
415             }
416         }
417         final RealLinearOperator m = JacobiPreconditioner.create(a);
418         final PreconditionedIterativeLinearSolver pcg;
419         final IterativeLinearSolver cg;
420         pcg = new ConjugateGradient(maxIterations, 1E-6, true);
421         cg = new ConjugateGradient(maxIterations, 1E-6, true);
422         final RealVector b = new ArrayRealVector(n);
423         final String pattern = "preconditioned gradient (%d iterations) should"
424                                + " have been faster than unpreconditioned (%d iterations)";
425         String msg;
426         for (int j = 0; j < 1; j++) {
427             b.set(0.);
428             b.setEntry(j, 1.);
429             final RealVector px = pcg.solve(a, m, b);
430             final RealVector x = cg.solve(a, b);
431             final int npcg = pcg.getIterationManager().getIterations();
432             final int ncg = cg.getIterationManager().getIterations();
433             msg = String.format(pattern, npcg, ncg);
434             Assert.assertTrue(msg, npcg < ncg);
435             for (int i = 0; i < n; i++) {
436                 msg = String.format("row %d, column %d", i, j);
437                 final double expected = x.getEntry(i);
438                 final double actual = px.getEntry(i);
439                 final double delta = 1E-6 * FastMath.abs(expected);
440                 Assert.assertEquals(msg, expected, actual, delta);
441             }
442         }
443     }

445     @Test
446     public void testEventManagement() {
447         final int n = 5;
448         final int maxIterations = 100;
449         final RealLinearOperator a = new HilbertMatrix(n);
450         final IterativeLinearSolver solver;
451         /*
452          * count[0] = number of calls to initializationPerformed
453          * count[1] = number of calls to iterationStarted
454          * count[2] = number of calls to iterationPerformed
455          * count[3] = number of calls to terminationPerformed
456          */
457         final int[] count = new int[] {0, 0, 0, 0};
458         final IterationListener listener = new IterationListener() {
459             private void doTestVectorsAreUnmodifiable(final IterationEvent e) {
460                 final IterativeLinearSolverEvent evt;
461                 evt = (IterativeLinearSolverEvent) e;
462                 try {
463                     evt.getResidual().set(0.0);
464                     Assert.fail("r is modifiable");
465                 } catch (MathUnsupportedOperationException exc){
466                     // Expected behavior
467                 }
468                 try {
469                     evt.getRightHandSideVector().set(0.0);
470                     Assert.fail("b is modifiable");
471                 } catch (MathUnsupportedOperationException exc){
472                     // Expected behavior
473                 }
474                 try {
475                     evt.getSolution().set(0.0);
476                     Assert.fail("x is modifiable");
477                 } catch (MathUnsupportedOperationException exc){
478                     // Expected behavior
479                 }
480             }
481 
482             public void initializationPerformed(final IterationEvent e) {
483                 ++count[0];
484                 doTestVectorsAreUnmodifiable(e);
485             }
486 
487             public void iterationPerformed(final IterationEvent e) {
488                 ++count[2];
489                 Assert.assertEquals("iteration performed",
490                     count[2], e.getIterations() - 1);
491                 doTestVectorsAreUnmodifiable(e);
492             }
493 
494             public void iterationStarted(final IterationEvent e) {
495                 ++count[1];
496                 Assert.assertEquals("iteration started",
497                     count[1], e.getIterations() - 1);
498                 doTestVectorsAreUnmodifiable(e);
499             }
500 
501             public void terminationPerformed(final IterationEvent e) {
502                 ++count[3];
503                 doTestVectorsAreUnmodifiable(e);
504             }
505         };
506         solver = new ConjugateGradient(maxIterations, 1E-10, true);
507         solver.getIterationManager().addIterationListener(listener);
508         final RealVector b = new ArrayRealVector(n);
509         for (int j = 0; j < n; j++) {
510             Arrays.fill(count, 0);
511             b.set(0.);
512             b.setEntry(j, 1.);
513             solver.solve(a, b);
514             String msg = String.format("column %d (initialization)", j);
515             Assert.assertEquals(msg, 1, count[0]);
516             msg = String.format("column %d (finalization)", j);
517             Assert.assertEquals(msg, 1, count[3]);
518         }
519     }
