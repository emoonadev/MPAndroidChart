76      @Test
77      public void testIncreasingTolerance() throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
78  
79          int previousCalls = Integer.MAX_VALUE;
80          for (int i = -12; i < -2; ++i) {
81              TestProblem1 pb = new TestProblem1();
82              double minStep = 0;
83              double maxStep = pb.getFinalTime() - pb.getInitialTime();
84              double scalAbsoluteTolerance = FastMath.pow(10.0, i);
85              double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
86  
87              FirstOrderIntegrator integ = new AdamsBashforthIntegrator(4, minStep, maxStep,
88                                                                        scalAbsoluteTolerance,
89                                                                        scalRelativeTolerance);
90              TestProblemHandler handler = new TestProblemHandler(pb, integ);
91              integ.addStepHandler(handler);
92              integ.integrate(pb,
93                              pb.getInitialTime(), pb.getInitialState(),
94                              pb.getFinalTime(), new double[pb.getDimension()]);
95  
96              // the 2.6 and 122 factors are only valid for this test
97              // and has been obtained from trial and error
98              // there are no general relationship between local and global errors
99              Assert.assertTrue(handler.getMaximalValueError() > (2.6 * scalAbsoluteTolerance));
100             Assert.assertTrue(handler.getMaximalValueError() < (122 * scalAbsoluteTolerance));
101 
102             int calls = pb.getCalls();
103             Assert.assertEquals(integ.getEvaluations(), calls);
104             Assert.assertTrue(calls <= previousCalls);
105             previousCalls = calls;
106 
107         }

127     @Test
128     public void backward() throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
129 
130         TestProblem5 pb = new TestProblem5();
131         double range = FastMath.abs(pb.getFinalTime() - pb.getInitialTime());
132 
133         AdamsBashforthIntegrator integ = new AdamsBashforthIntegrator(4, 0, range, 1.0e-12, 1.0e-12);
134         integ.setStarterIntegrator(new PerfectStarter(pb, (integ.getNSteps() + 5) / 2));
135         TestProblemHandler handler = new TestProblemHandler(pb, integ);
136         integ.addStepHandler(handler);
137         integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
138                         pb.getFinalTime(), new double[pb.getDimension()]);
139 
140         Assert.assertEquals(0.0, handler.getLastError(), 4.3e-8);
141         Assert.assertEquals(0.0, handler.getMaximalValueError(), 4.3e-8);
142         Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-16);
143         Assert.assertEquals("Adams-Bashforth", integ.getName());
144     }

149         double range = FastMath.abs(pb.getFinalTime() - pb.getInitialTime());

151         for (int nSteps = 2; nSteps < 8; ++nSteps) {
152             AdamsBashforthIntegrator integ =
153                 new AdamsBashforthIntegrator(nSteps, 1.0e-6 * range, 0.1 * range, 1.0e-4, 1.0e-4);
154             integ.setStarterIntegrator(new PerfectStarter(pb, nSteps));
155             TestProblemHandler handler = new TestProblemHandler(pb, integ);
156             integ.addStepHandler(handler);
157             integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
158                             pb.getFinalTime(), new double[pb.getDimension()]);
159             if (nSteps < 5) {
160                 Assert.assertTrue(handler.getMaximalValueError() > 0.005);
161             } else {
162                 Assert.assertTrue(handler.getMaximalValueError() < 5.0e-10);
163             }
164         }

191     private static class PerfectStarter extends AbstractIntegrator {
192 
193         private final PerfectInterpolator interpolator;
194         private final int nbSteps;
195 
196         public PerfectStarter(final TestProblemAbstract problem, final int nbSteps) {
197             this.interpolator = new PerfectInterpolator(problem);
198             this.nbSteps      = nbSteps;
199         }
200 
201         public void integrate(ExpandableStatefulODE equations, double t) {
202             double tStart = equations.getTime() + 0.01 * (t - equations.getTime());
203             getCounter().increment(nbSteps);
204             for (int i = 0; i < nbSteps; ++i) {
205                 double tK = ((nbSteps - 1 - (i + 1)) * equations.getTime() + (i + 1) * tStart) / (nbSteps - 1);
206                 interpolator.setPreviousTime(interpolator.getCurrentTime());
207                 interpolator.setCurrentTime(tK);
208                 interpolator.setInterpolatedTime(tK);
209                 for (StepHandler handler : getStepHandlers()) {
210                     handler.handleStep(interpolator, i == nbSteps - 1);
211                 }
212             }
213         }
214 
215     }

217     private static class PerfectInterpolator implements StepInterpolator {
218         private final TestProblemAbstract problem;
219         private double previousTime;
220         private double currentTime;
221         private double interpolatedTime;
222 
223         public PerfectInterpolator(final TestProblemAbstract problem) {
224             this.problem          = problem;
225             this.previousTime     = problem.getInitialTime();
226             this.currentTime      = problem.getInitialTime();
227             this.interpolatedTime = problem.getInitialTime();
228         }
229 
230         public void readExternal(ObjectInput arg0) {
231         }
232 
233         public void writeExternal(ObjectOutput arg0) {
234         }
235 
236         public double getPreviousTime() {
237             return previousTime;
238         }
239 
240         public void setPreviousTime(double time) {
241             previousTime = time;
242         }
243 
244         public double getCurrentTime() {
245             return currentTime;
246         }
247 
248         public void setCurrentTime(double time) {
249             currentTime = time;
250         }
251 
252         public double getInterpolatedTime() {
253             return interpolatedTime;
254         }
255 
256         public void setInterpolatedTime(double time) {
257             interpolatedTime = time;
258         }
259 
260         public double[] getInterpolatedState() {
261             return problem.computeTheoreticalState(interpolatedTime);
262         }
263 
264         public double[] getInterpolatedDerivatives() {
265             double[] y = problem.computeTheoreticalState(interpolatedTime);
266             double[] yDot = new double[y.length];
267             problem.computeDerivatives(interpolatedTime, y, yDot);
268             return yDot;
269         }
270 
271         public double[] getInterpolatedSecondaryState(int index) {
272             return null;
273         }
274 
275         public double[] getInterpolatedSecondaryDerivatives(int index) {
276             return null;
277         }
278 
279         public boolean isForward() {
280             return problem.getFinalTime() > problem.getInitialTime();
281         }
282 
283         public StepInterpolator copy() {
284             return this;
285         }

287     }
