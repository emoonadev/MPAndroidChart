18  package org.apache.commons.math4.ml.neuralnet.sofm.util;
19  
20  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
21  import org.apache.commons.math4.exception.NumberIsTooLargeException;
22  import org.apache.commons.math4.ml.neuralnet.sofm.util.ExponentialDecayFunction;
23  import org.junit.Test;
24  import org.junit.Assert;
25  
26  /**
27   * Tests for {@link ExponentialDecayFunction} class
28   */
29  public class ExponentialDecayFunctionTest {
30      @Test(expected=NotStrictlyPositiveException.class)
31      public void testPrecondition1() {
32          new ExponentialDecayFunction(0d, 0d, 2);
33      }
34      @Test(expected=NotStrictlyPositiveException.class)
35      public void testPrecondition2() {
36          new ExponentialDecayFunction(1d, 0d, 2);
37      }
38      @Test(expected=NumberIsTooLargeException.class)
39      public void testPrecondition3() {
40          new ExponentialDecayFunction(1d, 1d, 100);
41      }
42      @Test(expected=NotStrictlyPositiveException.class)
43      public void testPrecondition4() {
44          new ExponentialDecayFunction(1d, 0.2, 0);
45      }
46  
47      @Test
48      public void testTrivial() {
49          final int n = 65;
50          final double init = 4;
51          final double valueAtN = 3;
52          final ExponentialDecayFunction f = new ExponentialDecayFunction(init, valueAtN, n);
53  
54          Assert.assertEquals(init, f.value(0), 0d);
55          Assert.assertEquals(valueAtN, f.value(n), 0d);
56          Assert.assertEquals(0, f.value(Long.MAX_VALUE), 0d);
57      }
58  }
