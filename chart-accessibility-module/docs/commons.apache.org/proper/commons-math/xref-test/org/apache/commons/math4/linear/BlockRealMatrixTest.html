214     @Test
215     public void testSeveralBlocks() {
216         RealMatrix m = new BlockRealMatrix(35, 71);
217         for (int i = 0; i < m.getRowDimension(); ++i) {
218             for (int j = 0; j < m.getColumnDimension(); ++j) {
219                 m.setEntry(i, j, i + j / 1024.0);
220             }
221         }
222 
223         RealMatrix mT = m.transpose();
224         Assert.assertEquals(m.getRowDimension(), mT.getColumnDimension());
225         Assert.assertEquals(m.getColumnDimension(), mT.getRowDimension());
226         for (int i = 0; i < mT.getRowDimension(); ++i) {
227             for (int j = 0; j < mT.getColumnDimension(); ++j) {
228                 Assert.assertEquals(m.getEntry(j, i), mT.getEntry(i, j), 0);
229             }
230         }
231 
232         RealMatrix mPm = m.add(m);
233         for (int i = 0; i < mPm.getRowDimension(); ++i) {
234             for (int j = 0; j < mPm.getColumnDimension(); ++j) {
235                 Assert.assertEquals(2 * m.getEntry(i, j), mPm.getEntry(i, j), 0);
236             }
237         }
238 
239         RealMatrix mPmMm = mPm.subtract(m);
240         for (int i = 0; i < mPmMm.getRowDimension(); ++i) {
241             for (int j = 0; j < mPmMm.getColumnDimension(); ++j) {
242                 Assert.assertEquals(m.getEntry(i, j), mPmMm.getEntry(i, j), 0);
243             }
244         }
245 
246         RealMatrix mTm = mT.multiply(m);
247         for (int i = 0; i < mTm.getRowDimension(); ++i) {
248             for (int j = 0; j < mTm.getColumnDimension(); ++j) {
249                 double sum = 0;
250                 for (int k = 0; k < mT.getColumnDimension(); ++k) {
251                     sum += (k + i / 1024.0) * (k + j / 1024.0);
252                 }
253                 Assert.assertEquals(sum, mTm.getEntry(i, j), 0);
254             }
255         }
256 
257         RealMatrix mmT = m.multiply(mT);
258         for (int i = 0; i < mmT.getRowDimension(); ++i) {
259             for (int j = 0; j < mmT.getColumnDimension(); ++j) {
260                 double sum = 0;
261                 for (int k = 0; k < m.getColumnDimension(); ++k) {
262                     sum += (i + k / 1024.0) * (j + k / 1024.0);
263                 }
264                 Assert.assertEquals(sum, mmT.getEntry(i, j), 0);
265             }
266         }
267 
268         RealMatrix sub1 = m.getSubMatrix(2, 9, 5, 20);
269         for (int i = 0; i < sub1.getRowDimension(); ++i) {
270             for (int j = 0; j < sub1.getColumnDimension(); ++j) {

272             }
273         }

277             for (int j = 0; j < sub2.getColumnDimension(); ++j) {

470     /** test examples in user guide */
471     @Test
472     public void testExamples() {
473         // Create a real matrix with two rows and three columns
474         double[][] matrixData = { {1d,2d,3d}, {2d,5d,3d}};
475         RealMatrix m = new BlockRealMatrix(matrixData);
476         // One more with three rows, two columns
477         double[][] matrixData2 = { {1d,2d}, {2d,5d}, {1d, 7d}};
478         RealMatrix n = new BlockRealMatrix(matrixData2);
479         // Now multiply m by n
480         RealMatrix p = m.multiply(n);
481         Assert.assertEquals(2, p.getRowDimension());
482         Assert.assertEquals(2, p.getColumnDimension());
483         // Invert p
484         RealMatrix pInverse = new LUDecomposition(p).getSolver().getInverse();
485         Assert.assertEquals(2, pInverse.getRowDimension());
486         Assert.assertEquals(2, pInverse.getColumnDimension());
487 
488         // Solve example
489         double[][] coefficientsData = {{2, 3, -2}, {-1, 7, 6}, {4, -3, -5}};
490         RealMatrix coefficients = new BlockRealMatrix(coefficientsData);
491         RealVector constants = new ArrayRealVector(new double[]{1, -2, 1}, false);
492         RealVector solution = new LUDecomposition(coefficients).getSolver().solve(constants);
493         final double cst0 = constants.getEntry(0);
494         final double cst1 = constants.getEntry(1);
495         final double cst2 = constants.getEntry(2);
496         final double sol0 = solution.getEntry(0);
497         final double sol1 = solution.getEntry(1);
498         final double sol2 = solution.getEntry(2);
499         Assert.assertEquals(2 * sol0 + 3 * sol1 -2 * sol2, cst0, 1E-12);
500         Assert.assertEquals(-1 * sol0 + 7 * sol1 + 6 * sol2, cst1, 1E-12);
501         Assert.assertEquals(4 * sol0 - 3 * sol1 -5 * sol2, cst2, 1E-12);
502     }

525     private void checkGetSubMatrix(RealMatrix m, double[][] reference,
526                                    int startRow, int endRow, int startColumn, int endColumn) {
527         try {
528             RealMatrix sub = m.getSubMatrix(startRow, endRow, startColumn, endColumn);
529             if (reference != null) {
530                 Assert.assertEquals(new BlockRealMatrix(reference), sub);
531             } else {
532                 Assert.fail("Expecting OutOfRangeException or NumberIsTooSmallException or NoDataException");
533             }
534         } catch (OutOfRangeException e) {
535             if (reference != null) {
