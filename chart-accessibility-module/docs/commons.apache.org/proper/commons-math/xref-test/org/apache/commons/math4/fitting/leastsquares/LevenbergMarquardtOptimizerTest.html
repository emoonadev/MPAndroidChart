129     private void checkEstimate(CircleVectorial circle,
130                                double initialStepBoundFactor, int maxCostEval,
131                                double costRelativeTolerance, double parRelativeTolerance,
132                                double orthoTolerance, boolean shouldFail) {
133         try {
134             final LevenbergMarquardtOptimizer optimizer = new LevenbergMarquardtOptimizer()
135                 .withInitialStepBoundFactor(initialStepBoundFactor)
136                 .withCostRelativeTolerance(costRelativeTolerance)
137                 .withParameterRelativeTolerance(parRelativeTolerance)
138                 .withOrthoTolerance(orthoTolerance)
139                 .withRankingThreshold(Precision.SAFE_MIN);
140 
141             final LeastSquaresProblem problem = builder(circle)
142                     .maxEvaluations(maxCostEval)
143                     .maxIterations(100)
144                     .start(new double[] { 98.680, 47.345 })
145                     .build();
146 
147             optimizer.optimize(problem);
148 
149             Assert.assertTrue(!shouldFail);
150             //TODO check it got the right answer
151 
152         } catch (DimensionMismatchException ee) {
153             Assert.assertTrue(shouldFail);
154         } catch (TooManyEvaluationsException ee) {
155             Assert.assertTrue(shouldFail);
156         }
157     }

161      * Bevington's textbook, "Data reduction and analysis for the physical sciences").

166     @Test
167     public void testBevington() {
168         final double[][] dataPoints = {
169             // column 1 = times
170             { 15, 30, 45, 60, 75, 90, 105, 120, 135, 150,
171               165, 180, 195, 210, 225, 240, 255, 270, 285, 300,
172               315, 330, 345, 360, 375, 390, 405, 420, 435, 450,
173               465, 480, 495, 510, 525, 540, 555, 570, 585, 600,
174               615, 630, 645, 660, 675, 690, 705, 720, 735, 750,
175               765, 780, 795, 810, 825, 840, 855, 870, 885, },
176             // column 2 = measured counts
177             { 775, 479, 380, 302, 185, 157, 137, 119, 110, 89,
178               74, 61, 66, 68, 48, 54, 51, 46, 55, 29,
179               28, 37, 49, 26, 35, 29, 31, 24, 25, 35,
180               24, 30, 26, 28, 21, 18, 20, 27, 17, 17,
181               14, 17, 24, 11, 22, 17, 12, 10, 13, 16,
182               9, 9, 14, 21, 17, 13, 12, 18, 10, },
183         };
184         final double[] start = {10, 900, 80, 27, 225};
185 
186         final BevingtonProblem problem = new BevingtonProblem();
187 
188         final int len = dataPoints[0].length;
189         final double[] weights = new double[len];
190         for (int i = 0; i < len; i++) {
191             problem.addPoint(dataPoints[0][i],
192                              dataPoints[1][i]);
193 
194             weights[i] = 1 / dataPoints[1][i];
195         }
196 
197         final Optimum optimum = optimizer.optimize(
198                 builder(problem)
199                         .target(dataPoints[1])
200                         .weight(new DiagonalMatrix(weights))
201                         .start(start)
202                         .maxIterations(20)
203                         .build()
204         );
205 
206         final RealVector solution = optimum.getPoint();
207         final double[] expectedSolution = { 10.4, 958.3, 131.4, 33.9, 205.0 };
208 
209         final RealMatrix covarMatrix = optimum.getCovariances(1e-14);
210         final double[][] expectedCovarMatrix = {
211             { 3.38, -3.69, 27.98, -2.34, -49.24 },
212             { -3.69, 2492.26, 81.89, -69.21, -8.9 },
213             { 27.98, 81.89, 468.99, -44.22, -615.44 },
214             { -2.34, -69.21, -44.22, 6.39, 53.80 },
215             { -49.24, -8.9, -615.44, 53.8, 929.45 }
216         };
217 
218         final int numParams = expectedSolution.length;
219 
220         // Check that the computed solution is within the reference error range.
221         for (int i = 0; i < numParams; i++) {
222             final double error = FastMath.sqrt(expectedCovarMatrix[i][i]);
223             Assert.assertEquals("Parameter " + i, expectedSolution[i], solution.getEntry(i), error);
224         }
225 
226         // Check that each entry of the computed covariance matrix is within 10%
227         // of the reference matrix entry.
228         for (int i = 0; i < numParams; i++) {
229             for (int j = 0; j < numParams; j++) {
230                 Assert.assertEquals("Covariance matrix [" + i + "][" + j + "]",
231                                     expectedCovarMatrix[i][j],
232                                     covarMatrix.getEntry(i, j),
233                                     FastMath.abs(0.1 * expectedCovarMatrix[i][j]));
234             }
235         }
236 
237         // Check various measures of goodness-of-fit.
238         final double chi2 = optimum.getChiSquare();
239         final double cost = optimum.getCost();
240         final double rms = optimum.getRMS();
241         final double reducedChi2 = optimum.getReducedChiSquare(start.length);
242 
243         // XXX Values computed by the CM code: It would be better to compare
244         // with the results from another library.
245         final double expectedChi2 = 66.07852350839286;
246         final double expectedReducedChi2 = 1.2014277001525975;
247         final double expectedCost = 8.128869755900439;
248         final double expectedRms = 1.0582887010256337;
249 
250         final double tol = 1e14;
251         Assert.assertEquals(expectedChi2, chi2, tol);
252         Assert.assertEquals(expectedReducedChi2, reducedChi2, tol);
253         Assert.assertEquals(expectedCost, cost, tol);
254         Assert.assertEquals(expectedRms, rms, tol);
255     }

257     @Test
258     public void testCircleFitting2() {
259         final double xCenter = 123.456;
260         final double yCenter = 654.321;
261         final double xSigma = 10;
262         final double ySigma = 15;
263         final double radius = 111.111;
264         // The test is extremely sensitive to the seed.
265         final long seed = 59321761412L;
266         final RandomCirclePointGenerator factory
267             = new RandomCirclePointGenerator(xCenter, yCenter, radius,
268                                              xSigma, ySigma,
269                                              seed);
270         final CircleProblem circle = new CircleProblem(xSigma, ySigma);
271 
272         final int numPoints = 10;
273         for (Vector2D p : factory.generate(numPoints)) {
274             circle.addPoint(p.getX(), p.getY());
275         }
276 
277         // First guess for the center's coordinates and radius.
278         final double[] init = { 90, 659, 115 };
279 
280         final Optimum optimum = optimizer.optimize(
281                 builder(circle).maxIterations(50).start(init).build());
282 
283         final double[] paramFound = optimum.getPoint().toArray();
284 
285         // Retrieve errors estimation.
286         final double[] asymptoticStandardErrorFound = optimum.getSigma(1e-14).toArray();
287 
288         // Check that the parameters are found within the assumed error bars.
289         Assert.assertEquals(xCenter, paramFound[0], asymptoticStandardErrorFound[0]);
290         Assert.assertEquals(yCenter, paramFound[1], asymptoticStandardErrorFound[1]);
291         Assert.assertEquals(radius, paramFound[2], asymptoticStandardErrorFound[2]);
292     }

294     @Test
295     public void testParameterValidator() {
296         // Setup.
297         final double xCenter = 123.456;
298         final double yCenter = 654.321;
299         final double xSigma = 10;
300         final double ySigma = 15;
301         final double radius = 111.111;
302         final long seed = 3456789L;
303         final RandomCirclePointGenerator factory
304             = new RandomCirclePointGenerator(xCenter, yCenter, radius,
305                                              xSigma, ySigma,
306                                              seed);
307         final CircleProblem circle = new CircleProblem(xSigma, ySigma);
308 
309         final int numPoints = 10;
310         for (Vector2D p : factory.generate(numPoints)) {
311             circle.addPoint(p.getX(), p.getY());
312         }
313 
314         // First guess for the center's coordinates and radius.
315         final double[] init = { 90, 659, 115 };
316         final Optimum optimum
317             = optimizer.optimize(builder(circle).maxIterations(50).start(init).build());
318         final int numEval = optimum.getEvaluations();
319         Assert.assertTrue(numEval > 1);
320 
321         // Build a new problem with a validator that amounts to cheating.
322         final ParameterValidator cheatValidator
323             = new ParameterValidator() {
324                     public RealVector validate(RealVector params) {
325                         // Cheat: return the optimum found previously.
326                         return optimum.getPoint();
327                     }
328                 };
329 
330         final Optimum cheatOptimum
331             = optimizer.optimize(builder(circle).maxIterations(50).start(init).parameterValidator(cheatValidator).build());
332         final int cheatNumEval = cheatOptimum.getEvaluations();
333         Assert.assertTrue(cheatNumEval < numEval);
334         // System.out.println("n=" + numEval + " nc=" + cheatNumEval);
335     }

358     private static class BevingtonProblem {
359         private List<Double> time;
360         private List<Double> count;
361 
362         public BevingtonProblem() {
363             time = new ArrayList<Double>();
364             count = new ArrayList<Double>();
365         }
366 
367         public void addPoint(double t, double c) {
368             time.add(t);
369             count.add(c);
370         }
371 
372         public MultivariateVectorFunction getModelFunction() {
373             return new MultivariateVectorFunction() {
374                 public double[] value(double[] params) {
375                     double[] values = new double[time.size()];
376                     for (int i = 0; i < values.length; ++i) {
377                         final double t = time.get(i);
378                         values[i] = params[0] +
379                             params[1] * FastMath.exp(-t / params[3]) +
380                             params[2] * FastMath.exp(-t / params[4]);
381                     }
382                     return values;
383                 }
384             };
385         }
386 
387         public MultivariateMatrixFunction getModelFunctionJacobian() {
388             return new MultivariateMatrixFunction() {
389                 public double[][] value(double[] params) {
390                     double[][] jacobian = new double[time.size()][5];
391 
392                     for (int i = 0; i < jacobian.length; ++i) {
393                         final double t = time.get(i);
394                         jacobian[i][0] = 1;
395 
396                         final double p3 =  params[3];
397                         final double p4 =  params[4];
398                         final double tOp3 = t / p3;
399                         final double tOp4 = t / p4;
400                         jacobian[i][1] = FastMath.exp(-tOp3);
401                         jacobian[i][2] = FastMath.exp(-tOp4);
402                         jacobian[i][3] = params[1] * FastMath.exp(-tOp3) * tOp3 / p3;
403                         jacobian[i][4] = params[2] * FastMath.exp(-tOp4) * tOp4 / p4;
404                     }
405                     return jacobian;
406                 }
407             };
408         }
