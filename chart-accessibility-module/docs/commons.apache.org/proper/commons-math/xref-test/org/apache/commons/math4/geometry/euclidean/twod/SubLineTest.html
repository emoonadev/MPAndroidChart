33      @Test
34      public void testEndPoints() {
35          Vector2D p1 = new Vector2D(-1, -7);
36          Vector2D p2 = new Vector2D(7, -1);
37          Segment segment = new Segment(p1, p2, new Line(p1, p2, 1.0e-10));
38          SubLine sub = new SubLine(segment);
39          List<Segment> segments = sub.getSegments();
40          Assert.assertEquals(1, segments.size());
41          Assert.assertEquals(0.0, new Vector2D(-1, -7).distance(segments.get(0).getStart()), 1.0e-10);
42          Assert.assertEquals(0.0, new Vector2D( 7, -1).distance(segments.get(0).getEnd()), 1.0e-10);
43      }

45      @Test
46      public void testNoEndPoints() {
47          SubLine wholeLine = new Line(new Vector2D(-1, 7), new Vector2D(7, 1), 1.0e-10).wholeHyperplane();
48          List<Segment> segments = wholeLine.getSegments();
49          Assert.assertEquals(1, segments.size());
50          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
51                            segments.get(0).getStart().getX() < 0);
52          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
53                            segments.get(0).getStart().getY() > 0);
54          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
55                            segments.get(0).getEnd().getX() > 0);
56          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
57                            segments.get(0).getEnd().getY() < 0);
58      }

60      @Test
61      public void testNoSegments() {
62          SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
63                                      new RegionFactory<Euclidean1D>().getComplement(new IntervalsSet(1.0e-10)));
64          List<Segment> segments = empty.getSegments();
65          Assert.assertEquals(0, segments.size());
66      }

68      @Test
69      public void testSeveralSegments() {
70          SubLine twoSubs = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
71                                      new RegionFactory<Euclidean1D>().union(new IntervalsSet(1, 2, 1.0e-10),
72                                                                             new IntervalsSet(3, 4, 1.0e-10)));
73          List<Segment> segments = twoSubs.getSegments();
74          Assert.assertEquals(2, segments.size());
75      }

77      @Test
78      public void testHalfInfiniteNeg() {
79          SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
80                                      new IntervalsSet(Double.NEGATIVE_INFINITY, 0.0, 1.0e-10));
81          List<Segment> segments = empty.getSegments();
82          Assert.assertEquals(1, segments.size());
83          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
84                            segments.get(0).getStart().getX() < 0);
85          Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
86                            segments.get(0).getStart().getY() < 0);
87          Assert.assertEquals(0.0, new Vector2D(3, -4).distance(segments.get(0).getEnd()), 1.0e-10);
88      }

90      @Test
91      public void testHalfInfinitePos() {
92          SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
93                                      new IntervalsSet(0.0, Double.POSITIVE_INFINITY, 1.0e-10));
94          List<Segment> segments = empty.getSegments();
95          Assert.assertEquals(1, segments.size());
96          Assert.assertEquals(0.0, new Vector2D(3, -4).distance(segments.get(0).getStart()), 1.0e-10);
97          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
98                            segments.get(0).getEnd().getX() > 0);
99          Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
100                           segments.get(0).getEnd().getY() > 0);
101     }

103     @Test
104     public void testIntersectionInsideInside() {
105         SubLine sub1 = new SubLine(new Vector2D(1, 1), new Vector2D(3, 1), 1.0e-10);
106         SubLine sub2 = new SubLine(new Vector2D(2, 0), new Vector2D(2, 2), 1.0e-10);
107         Assert.assertEquals(0.0, new Vector2D(2, 1).distance(sub1.intersection(sub2, true)),  1.0e-12);
108         Assert.assertEquals(0.0, new Vector2D(2, 1).distance(sub1.intersection(sub2, false)), 1.0e-12);
109     }
