74          NormalDistribution distribution = (NormalDistribution) getDistribution();
75          double mu = distribution.getMean();
76          double sigma = distribution.getStandardDeviation();
77          setCumulativeTestPoints( new double[] {mu - 2 *sigma, mu - sigma,

79                  mu + 5 * sigma});
80          // Quantiles computed using R (same as Mathematica)

142     @Test
143     public void testDensity() {
144         double [] x = new double[]{-2, -1, 0, 1, 2};
145         // R 2.5: print(dnorm(c(-2,-1,0,1,2)), digits=10)
146         checkDensity(0, 1, x, new double[]{0.05399096651, 0.24197072452, 0.39894228040, 0.24197072452, 0.05399096651});
147         // R 2.5: print(dnorm(c(-2,-1,0,1,2), mean=1.1), digits=10)
148         checkDensity(1.1, 1, x, new double[]{0.003266819056,0.043983595980,0.217852177033,0.396952547477,0.266085249899});
149     }

151     private void checkDensity(double mean, double sd, double[] x, double[] expected) {
152         NormalDistribution d = new NormalDistribution(mean, sd);
153         for (int i = 0; i < x.length; i++) {
154             Assert.assertEquals(expected[i], d.density(x[i]), 1e-9);
155         }
156     }

158     /**
159      * Check to make sure top-coding of extreme values works correctly.
160      * Verifies fixes for JIRA MATH-167, MATH-414
161      */
162     @Test
163     public void testExtremeValues() {
164         NormalDistribution distribution = new NormalDistribution(0, 1);
165         for (int i = 0; i < 100; i++) { // make sure no convergence exception
166             double lowerTail = distribution.cumulativeProbability(-i);
167             double upperTail = distribution.cumulativeProbability(i);
168             if (i < 9) { // make sure not top-coded
169                 // For i = 10, due to bad tail precision in erf (MATH-364), 1 is returned
170                 // TODO: once MATH-364 is resolved, replace 9 with 30
171                 Assert.assertTrue(lowerTail > 0.0d);
172                 Assert.assertTrue(upperTail < 1.0d);
173             }
174             else { // make sure top coding not reversed
175                 Assert.assertTrue(lowerTail < 0.00001);
176                 Assert.assertTrue(upperTail > 0.99999);
177             }
178         }
179 
180         Assert.assertEquals(distribution.cumulativeProbability(Double.MAX_VALUE), 1, 0);
181         Assert.assertEquals(distribution.cumulativeProbability(-Double.MAX_VALUE), 0, 0);
182         Assert.assertEquals(distribution.cumulativeProbability(Double.POSITIVE_INFINITY), 1, 0);
183         Assert.assertEquals(distribution.cumulativeProbability(Double.NEGATIVE_INFINITY), 0, 0);
184     }

186     @Test
187     public void testMath280() {
188         NormalDistribution normal = new NormalDistribution(0,1);
189         double result = normal.inverseCumulativeProbability(0.9986501019683698);
190         Assert.assertEquals(3.0, result, defaultTolerance);
191         result = normal.inverseCumulativeProbability(0.841344746068543);
192         Assert.assertEquals(1.0, result, defaultTolerance);
193         result = normal.inverseCumulativeProbability(0.9999683287581673);
194         Assert.assertEquals(4.0, result, defaultTolerance);
195         result = normal.inverseCumulativeProbability(0.9772498680518209);
196         Assert.assertEquals(2.0, result, defaultTolerance);
197     }

199     @Test
200     public void testMoments() {
201         final double tol = 1e-9;
202         NormalDistribution dist;
203 
204         dist = new NormalDistribution(0, 1);
205         Assert.assertEquals(dist.getNumericalMean(), 0, tol);
206         Assert.assertEquals(dist.getNumericalVariance(), 1, tol);
207 
208         dist = new NormalDistribution(2.2, 1.4);
209         Assert.assertEquals(dist.getNumericalMean(), 2.2, tol);
210         Assert.assertEquals(dist.getNumericalVariance(), 1.4 * 1.4, tol);
211 
212         dist = new NormalDistribution(-2000.9, 10.4);
213         Assert.assertEquals(dist.getNumericalMean(), -2000.9, tol);
214         Assert.assertEquals(dist.getNumericalVariance(), 10.4 * 10.4, tol);
215     }
