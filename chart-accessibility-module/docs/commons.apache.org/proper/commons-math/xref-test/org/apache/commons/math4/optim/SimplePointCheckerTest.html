17  package org.apache.commons.math4.optim;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.optim.PointValuePair;
21  import org.apache.commons.math4.optim.SimplePointChecker;
22  import org.junit.Test;
23  import org.junit.Assert;
24  
25  public class SimplePointCheckerTest {
26      @Test(expected=NotStrictlyPositiveException.class)
27      public void testIterationCheckPrecondition() {
28          new SimplePointChecker<PointValuePair>(1e-1, 1e-2, 0);
29      }
30  
31      @Test
32      public void testIterationCheck() {
33          final int max = 10;
34          final SimplePointChecker<PointValuePair> checker
35              = new SimplePointChecker<PointValuePair>(1e-1, 1e-2, max);
36          Assert.assertTrue(checker.converged(max, null, null));
37          Assert.assertTrue(checker.converged(max + 1, null, null));
38      }
39  
40      @Test
41      public void testIterationCheckDisabled() {
42          final SimplePointChecker<PointValuePair> checker
43              = new SimplePointChecker<PointValuePair>(1e-8, 1e-8);
44  
45          final PointValuePair a = new PointValuePair(new double[] { 1d }, 1d);
46          final PointValuePair b = new PointValuePair(new double[] { 10d }, 10d);
47  
48          Assert.assertFalse(checker.converged(-1, a, b));
49          Assert.assertFalse(checker.converged(0, a, b));
50          Assert.assertFalse(checker.converged(1000000, a, b));
51  
52          Assert.assertTrue(checker.converged(-1, a, a));
53          Assert.assertTrue(checker.converged(-1, b, b));
54      }
55  }
