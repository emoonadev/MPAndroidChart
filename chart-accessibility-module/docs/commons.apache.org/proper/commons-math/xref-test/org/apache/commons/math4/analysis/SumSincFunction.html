17  package org.apache.commons.math4.analysis;
18  
19  import org.apache.commons.math4.analysis.function.Sinc;
20  
21  /**
22   * Auxiliary class for testing optimizers.
23   *
24   */
25  public class SumSincFunction implements MultivariateFunction {
26      private static final UnivariateFunction sinc = new Sinc();
27  
28      /**
29       * Factor that will multiply each term of the sum.
30       */
31      private final double factor;
32  
33      /**
34       * @param factor Factor that will multiply each term of the sum.
35       */
36      public SumSincFunction(double factor) {
37          this.factor = factor;
38      }
39  
40      /**
41       * @param point Argument.
42       * @return the value of this function at point {@code x}.
43       */
44      @Override
45      public double value(double[] point) {
46          double sum = 0;
47          for (int i = 0, max = point.length; i < max; i++) {
48              final double x = point[i];
49              final double v = sinc.value(x);
50              sum += v;
51          }
52          return factor * sum;
53      }
54  
55  }
