17  package org.apache.commons.math4.random;
18  
19  import org.junit.Assert;
20  
21  import java.io.InputStream;
22  
23  import org.apache.commons.math4.exception.OutOfRangeException;
24  import org.apache.commons.math4.random.SobolSequenceGenerator;
25  import org.junit.Before;
26  import org.junit.Test;
27  
28  public class SobolSequenceGeneratorTest {
29  
30      private double[][] referenceValues = {
31              { 0.0, 0.0, 0.0 },
32              { 0.5, 0.5, 0.5 },
33              { 0.75, 0.25, 0.25 },
34              { 0.25, 0.75, 0.75 },
35              { 0.375, 0.375, 0.625 },
36              { 0.875, 0.875, 0.125 },
37              { 0.625, 0.125, 0.875 },
38              { 0.125, 0.625, 0.375 },
39              { 0.1875, 0.3125, 0.9375 },
40              { 0.6875, 0.8125, 0.4375 }
41      };
42  
43      private SobolSequenceGenerator generator;
44  
45      @Before
46      public void setUp() {
47          generator = new SobolSequenceGenerator(3);
48      }
49  
50      @Test
51      public void test3DReference() {
52          for (int i = 0; i < referenceValues.length; i++) {
53              double[] result = generator.nextVector();
54              Assert.assertArrayEquals(referenceValues[i], result, 1e-6);
55              Assert.assertEquals(i + 1, generator.getNextIndex());
56          }
57      }
58  
59      @Test
60      public void testConstructor() {
61          try {
62              new SobolSequenceGenerator(0);
63              Assert.fail("an exception should have been thrown");
64          } catch (OutOfRangeException e) {
65              // expected
66          }
67  
68          try {
69              new SobolSequenceGenerator(1001);
70              Assert.fail("an exception should have been thrown");
71          } catch (OutOfRangeException e) {
72              // expected
73          }
74      }
75  
76      @Test
77      public void testConstructor2() throws Exception{
78          try {
79              final String RESOURCE_NAME = "/assets/org/apache/commons/math4/random/new-joe-kuo-6.1000";
80              final InputStream is = getClass().getResourceAsStream(RESOURCE_NAME);
81              new SobolSequenceGenerator(1001, is);
82              Assert.fail("an exception should have been thrown");
83          } catch (OutOfRangeException e) {
84              // expected
85          }
86  
87          try {
88              new SobolSequenceGenerator(1001);
89              Assert.fail("an exception should have been thrown");
90          } catch (OutOfRangeException e) {
91              // expected
92          }
93      }
94  
95      @Test
96      public void testSkip() {
97          double[] result = generator.skipTo(5);
98          Assert.assertArrayEquals(referenceValues[5], result, 1e-6);
99          Assert.assertEquals(6, generator.getNextIndex());
100 
101         for (int i = 6; i < referenceValues.length; i++) {
102             result = generator.nextVector();
103             Assert.assertArrayEquals(referenceValues[i], result, 1e-6);
104             Assert.assertEquals(i + 1, generator.getNextIndex());
105         }
106     }
107 
108 }
