17  package org.apache.commons.math4.geometry.spherical.oned;
18  
19  import org.apache.commons.math4.geometry.spherical.oned.LimitAngle;
20  import org.apache.commons.math4.geometry.spherical.oned.S1Point;
21  import org.apache.commons.math4.util.MathUtils;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  public class LimitAngleTest {
26  
27      @Test
28      public void testReversedLimit() {
29          for (int k = -2; k < 3; ++k) {
30              LimitAngle l  = new LimitAngle(new S1Point(1.0 + k * MathUtils.TWO_PI), false, 1.0e-10);
31              Assert.assertEquals(l.getLocation().getAlpha(), l.getReverse().getLocation().getAlpha(), 1.0e-10);
32              Assert.assertEquals(l.getTolerance(), l.getReverse().getTolerance(), 1.0e-10);
33              Assert.assertTrue(l.sameOrientationAs(l));
34              Assert.assertFalse(l.sameOrientationAs(l.getReverse()));
35              Assert.assertEquals(MathUtils.TWO_PI, l.wholeSpace().getSize(), 1.0e-10);
36              Assert.assertEquals(MathUtils.TWO_PI, l.getReverse().wholeSpace().getSize(), 1.0e-10);
37          }
38      }
39  
40  }
