42      @Test
43      public void testIsSame() {
44          DummyRandomKey drk1 = new DummyRandomKey(new Double[] {0.4, 0.1, 0.5, 0.8, 0.2});
45          DummyRandomKey drk2 = new DummyRandomKey(new Double[] {0.4, 0.1, 0.5, 0.8, 0.2});
46          DummyRandomKey drk3 = new DummyRandomKey(new Double[] {0.4, 0.15, 0.5, 0.8, 0.2});
47          DummyRandomKey drk4 = new DummyRandomKey(new Double[] {0.4, 0.25, 0.5, 0.8, 0.2});
48          DummyRandomKey drk5 = new DummyRandomKey(new Double[] {0.4, 0.25, 0.5, 0.8, 0.2, 0.5});
49  
50          Assert.assertTrue(drk1.isSame(drk2));
51          Assert.assertTrue(drk2.isSame(drk3));
52          Assert.assertFalse(drk3.isSame(drk4));
53          Assert.assertFalse(drk4.isSame(drk5));
54      }

56      @Test
57      public void testDecode() {
58          DummyRandomKey drk = new DummyRandomKey(new Double[] {0.4, 0.1, 0.5, 0.8, 0.2});
59          List<String> decoded = drk.decode(Arrays.asList(new String[] {"a", "b", "c", "d", "e"}));
60  
61          Assert.assertEquals("b", decoded.get(0));
62          Assert.assertEquals("e", decoded.get(1));
63          Assert.assertEquals("a", decoded.get(2));
64          Assert.assertEquals("c", decoded.get(3));
65          Assert.assertEquals("d", decoded.get(4));
66      }

94      @Test
95      public void testComparatorPermutation() {
96          List<String> data = Arrays.asList(new String[] {"x", "b", "c", "z", "b"});
97  
98          List<Double> permutation = RandomKey.comparatorPermutation(data, new Comparator<String>() {
99              @Override
100             public int compare(String o1, String o2) {
101                 return o1.compareTo(o2);
102             }
103         });
104         Double[] permArr = new Double[data.size()];
105         permArr = permutation.toArray(permArr);
106         Assert.assertArrayEquals(new Double[] {0.6,0.0,0.4,0.8,0.2}, permArr);
107         List<String> decodedData = new DummyRandomKey(permutation).decode(data);
108         Assert.assertEquals("b", decodedData.get(0));
109         Assert.assertEquals("b", decodedData.get(1));
110         Assert.assertEquals("c", decodedData.get(2));
111         Assert.assertEquals("x", decodedData.get(3));
112         Assert.assertEquals("z", decodedData.get(4));
113 
114         permutation = RandomKey.comparatorPermutation(data, new Comparator<String>() {
115             @Override
116             public int compare(String o1, String o2) {
117                 return o2.compareTo(o1);
118             }
119         });
120         permArr = new Double[data.size()];
121         permArr = permutation.toArray(permArr);
122         Assert.assertArrayEquals(new Double[] {0.2,0.6,0.4,0.0,0.8}, permArr);
123         decodedData = new DummyRandomKey(permutation).decode(data);
124         Assert.assertEquals("z", decodedData.get(0));
125         Assert.assertEquals("x", decodedData.get(1));
126         Assert.assertEquals("c", decodedData.get(2));
127         Assert.assertEquals("b", decodedData.get(3));
128         Assert.assertEquals("b", decodedData.get(4));
129     }

131     @Test
132     public void testInducedPermutation() {
133         List<String> origData = Arrays.asList(new String[] {"a", "b", "c", "d", "d"});
134         List<String> permutedData = Arrays.asList(new String[] {"d", "b", "c", "a", "d"});
135 
136         DummyRandomKey drk = new DummyRandomKey(RandomKey.inducedPermutation(origData, permutedData));
137         List<String> decoded = drk.decode(origData);
138 
139         Assert.assertEquals("d", decoded.get(0));
140         Assert.assertEquals("b", decoded.get(1));
141         Assert.assertEquals("c", decoded.get(2));
142         Assert.assertEquals("a", decoded.get(3));
143         Assert.assertEquals("d", decoded.get(4));
144 
145         try {
146             RandomKey.inducedPermutation(
147                     Arrays.asList(new String[] {"a", "b", "c", "d", "d"}),
148                     Arrays.asList(new String[] {"a", "b", "c", "d"})
149             );
150             Assert.fail("Uncaught exception");
151         } catch (MathIllegalArgumentException e) {
152             // no-op
153         }
154         try {
155             RandomKey.inducedPermutation(
156                     Arrays.asList(new String[] {"a", "b", "c", "d", "d"}),
157                     Arrays.asList(new String[] {"a", "b", "c", "d", "f"})
158             );
159             Assert.fail("Uncaught exception");
160         } catch (MathIllegalArgumentException e) {
161             // no-op
162         }
163     }
