439     @Test

441     public void testAnglesDeprecated()
442             throws CardanEulerSingularityException {
443 
444         DfpField field = new DfpField(15);
445 
446         RotationOrder[] CardanOrders = {
447             RotationOrder.XYZ, RotationOrder.XZY, RotationOrder.YXZ,
448             RotationOrder.YZX, RotationOrder.ZXY, RotationOrder.ZYX
449         };
450 
451         for (int i = 0; i < CardanOrders.length; ++i) {
452             for (double alpha1 = 0.1; alpha1 < 6.2; alpha1 += 2.0) {
453                 for (double alpha2 = -1.55; alpha2 < 1.55; alpha2 += 0.8) {
454                     for (double alpha3 = 0.1; alpha3 < 6.2; alpha3 += 2.0) {
455                         FieldRotation<Dfp> r = new FieldRotation<Dfp>(CardanOrders[i],
456                                                                       field.newDfp(alpha1),
457                                                                       field.newDfp(alpha2),
458                                                                       field.newDfp(alpha3));
459                         Dfp[] angles = r.getAngles(CardanOrders[i]);
460                         checkAngle(angles[0], alpha1);
461                         checkAngle(angles[1], alpha2);
462                         checkAngle(angles[2], alpha3);
463                     }
464                 }
465             }
466         }
467 
468         RotationOrder[] EulerOrders = {
469             RotationOrder.XYX, RotationOrder.XZX, RotationOrder.YXY,
470             RotationOrder.YZY, RotationOrder.ZXZ, RotationOrder.ZYZ
471         };
472 
473         for (int i = 0; i < EulerOrders.length; ++i) {
474             for (double alpha1 = 0.1; alpha1 < 6.2; alpha1 += 2.0) {
475                 for (double alpha2 = 0.05; alpha2 < 3.1; alpha2 += 0.8) {
476                     for (double alpha3 = 0.1; alpha3 < 6.2; alpha3 += 2.0) {
477                         FieldRotation<Dfp> r = new FieldRotation<Dfp>(EulerOrders[i],
478                                                                       field.newDfp(alpha1),
