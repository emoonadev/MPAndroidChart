43      @Test
44      public void testNoError() {
45          final double a = 0.2;
46          final double w = 3.4;
47          final double p = 4.1;
48          final HarmonicOscillator f = new HarmonicOscillator(a, w, p);
49  
50          final WeightedObservedPoints points = new WeightedObservedPoints();
51          for (double x = 0.0; x < 1.3; x += 0.01) {
52              points.add(1, x, f.value(x));
53          }
54  
55          final HarmonicCurveFitter fitter = HarmonicCurveFitter.create();
56          final double[] fitted = fitter.fit(points.toList());
57          Assert.assertEquals(a, fitted[0], 1.0e-13);
58          Assert.assertEquals(w, fitted[1], 1.0e-13);
59          Assert.assertEquals(p, MathUtils.normalizeAngle(fitted[2], p), 1e-13);
60  
61          final HarmonicOscillator ff = new HarmonicOscillator(fitted[0], fitted[1], fitted[2]);
62          for (double x = -1.0; x < 1.0; x += 0.01) {
63              Assert.assertTrue(FastMath.abs(f.value(x) - ff.value(x)) < 1e-13);
64          }
65      }

106         final double a = 0.2;
107         final double w = 3.4;
108         final double p = 4.1;
109         final HarmonicOscillator f = new HarmonicOscillator(a, w, p);

111         final WeightedObservedPoints points = new WeightedObservedPoints();
112         for (double x = 0.0; x < 10.0; x += 0.1) {

114         }

124     @Test
125     public void testUnsorted() {
126         Random randomizer = new Random(64925784252L);
127         final double a = 0.2;
128         final double w = 3.4;
129         final double p = 4.1;
130         final HarmonicOscillator f = new HarmonicOscillator(a, w, p);
131 
132         // Build a regularly spaced array of measurements.
133         final int size = 100;
134         final double[] xTab = new double[size];
135         final double[] yTab = new double[size];
136         for (int i = 0; i < size; i++) {
137             xTab[i] = 0.1 * i;
138             yTab[i] = f.value(xTab[i]) + 0.01 * randomizer.nextGaussian();
139         }
140 
141         // shake it
142         for (int i = 0; i < size; i++) {
143             int i1 = randomizer.nextInt(size);
144             int i2 = randomizer.nextInt(size);
145             double xTmp = xTab[i1];
146             double yTmp = yTab[i1];
147             xTab[i1] = xTab[i2];
148             yTab[i1] = yTab[i2];
149             xTab[i2] = xTmp;
150             yTab[i2] = yTmp;
151         }
152 
153         // Pass it to the fitter.
154         final WeightedObservedPoints points = new WeightedObservedPoints();
155         for (int i = 0; i < size; ++i) {
156             points.add(1, xTab[i], yTab[i]);
157         }
158 
159         final HarmonicCurveFitter fitter = HarmonicCurveFitter.create();
160         final double[] fitted = fitter.fit(points.toList());
161         Assert.assertEquals(a, fitted[0], 7.6e-4);
162         Assert.assertEquals(w, fitted[1], 3.5e-3);
163         Assert.assertEquals(p, MathUtils.normalizeAngle(fitted[2], p), 1.5e-2);
164     }

166     @Test(expected=MathIllegalStateException.class)
167     public void testMath844() {
168         final double[] y = { 0, 1, 2, 3, 2, 1,
169                              0, -1, -2, -3, -2, -1,
170                              0, 1, 2, 3, 2, 1,
171                              0, -1, -2, -3, -2, -1,
172                              0, 1, 2, 3, 2, 1, 0 };
173         final List<WeightedObservedPoint> points = new ArrayList<WeightedObservedPoint>();
174         for (int i = 0; i < y.length; i++) {
175             points.add(new WeightedObservedPoint(1, i, y[i]));
176         }
177 
178         // The guesser fails because the function is far from an harmonic
179         // function: It is a triangular periodic function with amplitude 3
180         // and period 12, and all sample points are taken at integer abscissae
181         // so function values all belong to the integer subset {-3, -2, -1, 0,
182         // 1, 2, 3}.
183         new HarmonicCurveFitter.ParameterGuesser(points);
184     }
