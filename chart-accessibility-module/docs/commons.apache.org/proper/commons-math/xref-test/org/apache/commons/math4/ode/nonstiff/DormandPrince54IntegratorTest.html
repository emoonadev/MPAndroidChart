158   @Test
159   public void testIncreasingTolerance()
160       throws DimensionMismatchException, NumberIsTooSmallException,
161              MaxCountExceededException, NoBracketingException {
162 
163     int previousCalls = Integer.MAX_VALUE;
164     for (int i = -12; i < -2; ++i) {
165       TestProblem1 pb = new TestProblem1();
166       double minStep = 0;
167       double maxStep = pb.getFinalTime() - pb.getInitialTime();
168       double scalAbsoluteTolerance = FastMath.pow(10.0, i);
169       double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
170 
171       EmbeddedRungeKuttaIntegrator integ =
172           new DormandPrince54Integrator(minStep, maxStep,
173                                         scalAbsoluteTolerance, scalRelativeTolerance);
174       TestProblemHandler handler = new TestProblemHandler(pb, integ);
175       integ.setSafety(0.8);
176       integ.setMaxGrowth(5.0);
177       integ.setMinReduction(0.3);
178       integ.addStepHandler(handler);
179       integ.integrate(pb,
180                       pb.getInitialTime(), pb.getInitialState(),
181                       pb.getFinalTime(), new double[pb.getDimension()]);
182       Assert.assertEquals(0.8, integ.getSafety(), 1.0e-12);
183       Assert.assertEquals(5.0, integ.getMaxGrowth(), 1.0e-12);
184       Assert.assertEquals(0.3, integ.getMinReduction(), 1.0e-12);
185 
186       // the 0.7 factor is only valid for this test
187       // and has been obtained from trial and error
188       // there is no general relation between local and global errors
189       Assert.assertTrue(handler.getMaximalValueError() < (0.7 * scalAbsoluteTolerance));
190       Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
191 
192       int calls = pb.getCalls();
193       Assert.assertEquals(integ.getEvaluations(), calls);
194       Assert.assertTrue(calls <= previousCalls);
195       previousCalls = calls;
196 
197     }

201   @Test
202   public void testEvents()
203       throws DimensionMismatchException, NumberIsTooSmallException,
204              MaxCountExceededException, NoBracketingException {
205 
206     TestProblem4 pb = new TestProblem4();
207     double minStep = 0;
208     double maxStep = pb.getFinalTime() - pb.getInitialTime();
209     double scalAbsoluteTolerance = 1.0e-8;
210     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
211 
212     FirstOrderIntegrator integ = new DormandPrince54Integrator(minStep, maxStep,
213                                                                scalAbsoluteTolerance,
214                                                                scalRelativeTolerance);
215     TestProblemHandler handler = new TestProblemHandler(pb, integ);
216     integ.addStepHandler(handler);
217     EventHandler[] functions = pb.getEventsHandlers();
218     double convergence = 1.0e-8 * maxStep;
219     for (int l = 0; l < functions.length; ++l) {
220       integ.addEventHandler(functions[l],
221                                  Double.POSITIVE_INFINITY, convergence, 1000);
222     }
223     Assert.assertEquals(functions.length, integ.getEventHandlers().size());
224     integ.integrate(pb,
225                     pb.getInitialTime(), pb.getInitialState(),
226                     pb.getFinalTime(), new double[pb.getDimension()]);
227 
228     Assert.assertTrue(handler.getMaximalValueError() < 5.0e-6);
229     Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
230     Assert.assertEquals(12.0, handler.getLastTime(), convergence);
231     integ.clearEventHandlers();
232     Assert.assertEquals(0, integ.getEventHandlers().size());
233 
234   }

280   private static class KeplerHandler implements StepHandler {
281     public KeplerHandler(TestProblem3 pb) {
282       this.pb = pb;
283     }
284     public void init(double t0, double[] y0, double t) {
285       nbSteps = 0;
286       maxError = 0;
287     }
288     public void handleStep(StepInterpolator interpolator, boolean isLast)
289         throws MaxCountExceededException {
290 
291       ++nbSteps;
292       for (int a = 1; a < 10; ++a) {
293 
294         double prev   = interpolator.getPreviousTime();
295         double curr   = interpolator.getCurrentTime();
296         double interp = ((10 - a) * prev + a * curr) / 10;
297         interpolator.setInterpolatedTime(interp);
298 
299         double[] interpolatedY = interpolator.getInterpolatedState ();
300         double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
301         double dx = interpolatedY[0] - theoreticalY[0];
302         double dy = interpolatedY[1] - theoreticalY[1];
303         double error = dx * dx + dy * dy;
304         if (error > maxError) {
305           maxError = error;
306         }
307       }
308       if (isLast) {
309         Assert.assertTrue(maxError < 7.0e-10);
310         Assert.assertTrue(nbSteps < 400);
311       }
312     }
313     private int nbSteps;
314     private double maxError;
315     private TestProblem3 pb;
316   }

318   private static class VariableHandler implements StepHandler {
319     public VariableHandler() {
320       firstTime = true;
321       minStep = 0;
322       maxStep = 0;
323     }
324     public void init(double t0, double[] y0, double t) {
325       firstTime = true;
326       minStep = 0;
327       maxStep = 0;
328     }
329     public void handleStep(StepInterpolator interpolator,
330                            boolean isLast) {
331 
332       double step = FastMath.abs(interpolator.getCurrentTime()
333                              - interpolator.getPreviousTime());
334       if (firstTime) {
335         minStep   = FastMath.abs(step);
336         maxStep   = minStep;
337         firstTime = false;
338       } else {
339         if (step < minStep) {
340           minStep = step;
341         }
342         if (step > maxStep) {
343           maxStep = step;
344         }
345       }
346 
347       if (isLast) {
348         Assert.assertTrue(minStep < (1.0 / 450.0));
349         Assert.assertTrue(maxStep > (1.0 / 4.2));
350       }
351     }
352     private boolean firstTime;
353     private double  minStep;
354     private double  maxStep;
355   }
