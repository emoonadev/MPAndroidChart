17  package org.apache.commons.math4.genetics;
18  
19  import java.util.List;
20  
21  import org.apache.commons.math4.genetics.AbstractListChromosome;
22  import org.apache.commons.math4.genetics.RandomKey;
23  
24  /**
25   * Implementation of RandomKey for testing purposes
26   */
27  public class DummyRandomKey extends RandomKey<String> {
28  
29      public DummyRandomKey(List<Double> representation) {
30          super(representation);
31      }
32  
33      public DummyRandomKey(Double[] representation) {
34          super(representation);
35      }
36  
37      @Override
38      public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> chromosomeRepresentation) {
39          return new DummyRandomKey(chromosomeRepresentation);
40      }
41  
42      public double fitness() {
43          // unimportant
44          return 0;
45      }
46  
47  }
