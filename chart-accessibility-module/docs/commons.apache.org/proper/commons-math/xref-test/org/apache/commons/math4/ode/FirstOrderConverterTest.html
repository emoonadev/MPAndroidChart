20  import org.apache.commons.math4.exception.DimensionMismatchException;
21  import org.apache.commons.math4.exception.MaxCountExceededException;
22  import org.apache.commons.math4.exception.NoBracketingException;
23  import org.apache.commons.math4.exception.NumberIsTooSmallException;
24  import org.apache.commons.math4.ode.FirstOrderConverter;
25  import org.apache.commons.math4.ode.SecondOrderDifferentialEquations;
26  import org.apache.commons.math4.ode.nonstiff.ClassicalRungeKuttaIntegrator;
27  import org.apache.commons.math4.util.FastMath;
28  import org.junit.Assert;
29  import org.junit.Test;

31  
32  public class FirstOrderConverterTest {
33  
34    @Test
35    public void testDoubleDimension() {
36      for (int i = 1; i < 10; ++i) {
37        SecondOrderDifferentialEquations eqn2 = new Equations(i, 0.2);
38        FirstOrderConverter eqn1 = new FirstOrderConverter(eqn2);
39        Assert.assertTrue(eqn1.getDimension() == (2 * eqn2.getDimension()));
40      }
41    }
42  
43    @Test
44    public void testDecreasingSteps()
45        throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
46  
47      double previousError = Double.NaN;
48      for (int i = 0; i < 10; ++i) {
49  
50        double step  = FastMath.pow(2.0, -(i + 1));
51        double error = integrateWithSpecifiedStep(4.0, 0.0, 1.0, step)
52                     - FastMath.sin(4.0);
53        if (i > 0) {
54          Assert.assertTrue(FastMath.abs(error) < FastMath.abs(previousError));
55        }
56        previousError = error;
57  
58      }
59    }
60  
61    @Test
62    public void testSmallStep()
63        throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
64      double error = integrateWithSpecifiedStep(4.0, 0.0, 1.0, 1.0e-4)
65                     - FastMath.sin(4.0);
66      Assert.assertTrue(FastMath.abs(error) < 1.0e-10);
67    }
68  
69    @Test
70    public void testBigStep()
71        throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
72      double error = integrateWithSpecifiedStep(4.0, 0.0, 1.0, 0.5)
73                     - FastMath.sin(4.0);
74      Assert.assertTrue(FastMath.abs(error) > 0.1);
75    }
76  
77    private static class Equations
78      implements SecondOrderDifferentialEquations {
79  
80       private int n;
81  
82        private double omega2;
83  
84        public Equations(int n, double omega) {
85          this.n = n;
86          omega2 = omega * omega;
87        }
88  
89        public int getDimension() {
90          return n;
91        }
92  
93        public void computeSecondDerivatives(double t, double[] y, double[] yDot,
94                                             double[] yDDot) {
95          for (int i = 0; i < n; ++i) {
96            yDDot[i] = -omega2 * y[i];
97          }
98      }
99  
100   }
101 
102   private double integrateWithSpecifiedStep(double omega,
103                                             double t0, double t,
104                                             double step) throws DimensionMismatchException, NumberIsTooSmallException, MaxCountExceededException, NoBracketingException {
105     double[] y0 = new double[2];
106     y0[0] = FastMath.sin(omega * t0);
107     y0[1] = omega * FastMath.cos(omega * t0);
108     ClassicalRungeKuttaIntegrator i = new ClassicalRungeKuttaIntegrator(step);
109     double[] y = new double[2];
110     i.integrate(new FirstOrderConverter(new Equations(1, omega)), t0, y0, t, y);
111     return y[0];
112   }
