30      @Test
31      public void testInsufficientData() {
32          double[] nothing = null;
33          SemiVariance sv = new SemiVariance();
34          try {
35              sv.evaluate(nothing);
36              Assert.fail("null is not a valid data array.");
37          } catch (NullArgumentException nae) {
38          }
39  
40          try {
41              sv.setVarianceDirection(SemiVariance.UPSIDE_VARIANCE);
42              sv.evaluate(nothing);
43              Assert.fail("null is not a valid data array.");
44          } catch (NullArgumentException nae) {
45          }
46          nothing = new double[] {};
47          Assert.assertTrue(Double.isNaN(sv.evaluate(nothing)));
48      }

66      @Test
67      public void testSample() {
68          final double[] values = { -2.0d, 2.0d, 4.0d, -2.0d, 22.0d, 11.0d, 3.0d, 14.0d, 5.0d };
69          final int length = values.length;
70          final double mean = StatUtils.mean(values); // 6.333...
71          final SemiVariance sv = new SemiVariance();  // Default bias correction is true
72          final double downsideSemiVariance = sv.evaluate(values); // Downside is the default
73          Assert.assertEquals(TestUtils.sumSquareDev(new double[] {-2d, 2d, 4d, -2d, 3d, 5d}, mean) / (length - 1),
74                  downsideSemiVariance, 1E-14);
75  
76          sv.setVarianceDirection(SemiVariance.UPSIDE_VARIANCE);
77          final double upsideSemiVariance = sv.evaluate(values);
78          Assert.assertEquals(TestUtils.sumSquareDev(new double[] {22d, 11d, 14d}, mean) / (length - 1),
79                  upsideSemiVariance, 1E-14);
80  
81          // Verify that upper + lower semivariance against the mean sum to variance
82          Assert.assertEquals(StatUtils.variance(values), downsideSemiVariance + upsideSemiVariance, 10e-12);
83      }

98      @Test
99      public void testNonMeanCutoffs() {
100         double[] values = { -2.0d, 2.0d, 4.0d, -2.0d, 22.0d, 11.0d, 3.0d, 14.0d, 5.0d };
101         SemiVariance sv = new SemiVariance(false); // Turn off bias correction - use df = length
102 
103         double singletest = sv.evaluate(values, 1.0d, SemiVariance.DOWNSIDE_VARIANCE, false, 0, values.length);
104         Assert.assertEquals(TestUtils.sumSquareDev(new double[] { -2d, -2d }, 1.0d) / values.length,
105                 singletest, 0.01d);
106 
107         singletest = sv.evaluate(values, 3.0d, SemiVariance.UPSIDE_VARIANCE, false, 0, values.length);
108         Assert.assertEquals(TestUtils.sumSquareDev(new double[] { 4d, 22d, 11d, 14d, 5d }, 3.0d) / values.length, singletest,
109                 0.01d);
110     }

112     /**
113      * Check that the lower + upper semivariance against the mean sum to the
114      * variance.
115      */
116     @Test
117     public void testVarianceDecompMeanCutoff() {
118         double[] values = { -2.0d, 2.0d, 4.0d, -2.0d, 22.0d, 11.0d, 3.0d, 14.0d, 5.0d };
119         double variance = StatUtils.variance(values);
120         SemiVariance sv = new SemiVariance(true); // Bias corrected
121         sv.setVarianceDirection(SemiVariance.DOWNSIDE_VARIANCE);
122         final double lower = sv.evaluate(values);
123         sv.setVarianceDirection(SemiVariance.UPSIDE_VARIANCE);
124         final double upper = sv.evaluate(values);
125         Assert.assertEquals(variance, lower + upper, 10e-12);
126     }
127 
128     /**
129      * Check that upper and lower semivariances against a cutoff sum to the sum
130      * of squared deviations of the full set of values against the cutoff
131      * divided by df = length - 1 (assuming bias-corrected).
132      */
133     @Test
134     public void testVarianceDecompNonMeanCutoff() {
135         double[] values = { -2.0d, 2.0d, 4.0d, -2.0d, 22.0d, 11.0d, 3.0d, 14.0d, 5.0d };
136         double target = 0;
137         double totalSumOfSquares = TestUtils.sumSquareDev(values, target);
138         SemiVariance sv = new SemiVariance(true); // Bias corrected
139         sv.setVarianceDirection(SemiVariance.DOWNSIDE_VARIANCE);
140         double lower = sv.evaluate(values, target);
141         sv.setVarianceDirection(SemiVariance.UPSIDE_VARIANCE);
142         double upper = sv.evaluate(values, target);
143         Assert.assertEquals(totalSumOfSquares / (values.length - 1), lower + upper, 10e-12);
144     }

146     @Test
147     public void testNoVariance() {
148         final double[] values = {100d, 100d, 100d, 100d};
149         SemiVariance sv = new SemiVariance();
150         Assert.assertEquals(0, sv.evaluate(values), 10E-12);
151         Assert.assertEquals(0, sv.evaluate(values, 100d), 10E-12);
152         Assert.assertEquals(0, sv.evaluate(values, 100d, SemiVariance.UPSIDE_VARIANCE, false, 0, values.length), 10E-12);
153     }
