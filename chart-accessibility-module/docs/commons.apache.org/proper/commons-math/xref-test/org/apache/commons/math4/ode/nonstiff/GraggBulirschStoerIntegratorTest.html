115   @Test
116   public void testIncreasingTolerance()
117       throws DimensionMismatchException, NumberIsTooSmallException,
118              MaxCountExceededException, NoBracketingException {
119 
120     int previousCalls = Integer.MAX_VALUE;
121     for (int i = -12; i < -4; ++i) {
122       TestProblem1 pb     = new TestProblem1();
123       double minStep      = 0;
124       double maxStep      = pb.getFinalTime() - pb.getInitialTime();
125       double absTolerance = FastMath.pow(10.0, i);
126       double relTolerance = absTolerance;
127 
128       FirstOrderIntegrator integ =
129         new GraggBulirschStoerIntegrator(minStep, maxStep,
130                                          absTolerance, relTolerance);
131       TestProblemHandler handler = new TestProblemHandler(pb, integ);
132       integ.addStepHandler(handler);
133       integ.integrate(pb,
134                       pb.getInitialTime(), pb.getInitialState(),
135                       pb.getFinalTime(), new double[pb.getDimension()]);
136 
137       // the coefficients are only valid for this test
138       // and have been obtained from trial and error
139       // there is no general relation between local and global errors
140       double ratio =  handler.getMaximalValueError() / absTolerance;
141       Assert.assertTrue(ratio < 2.4);
142       Assert.assertTrue(ratio > 0.02);
143       Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
144 
145       int calls = pb.getCalls();
146       Assert.assertEquals(integ.getEvaluations(), calls);
147       Assert.assertTrue(calls <= previousCalls);
148       previousCalls = calls;
149 
150     }

154   @Test
155   public void testIntegratorControls()
156       throws DimensionMismatchException, NumberIsTooSmallException,
157              MaxCountExceededException, NoBracketingException {
158 
159     TestProblem3 pb = new TestProblem3(0.999);
160     GraggBulirschStoerIntegrator integ =
161         new GraggBulirschStoerIntegrator(0, pb.getFinalTime() - pb.getInitialTime(),
162                 1.0e-8, 1.0e-10);
163 
164     double errorWithDefaultSettings = getMaxError(integ, pb);
165 
166     // stability control
167     integ.setStabilityCheck(true, 2, 1, 0.99);
168     Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
169     integ.setStabilityCheck(true, -1, -1, -1);
170 
171     integ.setControlFactors(0.5, 0.99, 0.1, 2.5);
172     Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
173     integ.setControlFactors(-1, -1, -1, -1);
174 
175     integ.setOrderControl(10, 0.7, 0.95);
176     Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
177     integ.setOrderControl(-1, -1, -1);
178 
179     integ.setInterpolationControl(true, 3);
180     Assert.assertTrue(errorWithDefaultSettings < getMaxError(integ, pb));
181     integ.setInterpolationControl(true, -1);
182 
183   }

185   private double getMaxError(FirstOrderIntegrator integrator, TestProblemAbstract pb)
186       throws DimensionMismatchException, NumberIsTooSmallException,
187              MaxCountExceededException, NoBracketingException {
188       TestProblemHandler handler = new TestProblemHandler(pb, integrator);
189       integrator.addStepHandler(handler);
190       integrator.integrate(pb,
191                            pb.getInitialTime(), pb.getInitialState(),
192                            pb.getFinalTime(), new double[pb.getDimension()]);
193       return handler.getMaximalValueError();
194   }

196   @Test
197   public void testEvents()
198       throws DimensionMismatchException, NumberIsTooSmallException,
199              MaxCountExceededException, NoBracketingException {
200 
201     TestProblem4 pb = new TestProblem4();
202     double minStep = 0;
203     double maxStep = pb.getFinalTime() - pb.getInitialTime();
204     double scalAbsoluteTolerance = 1.0e-10;
205     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
206 
207     FirstOrderIntegrator integ = new GraggBulirschStoerIntegrator(minStep, maxStep,
208                                                                   scalAbsoluteTolerance,
209                                                                   scalRelativeTolerance);
210     TestProblemHandler handler = new TestProblemHandler(pb, integ);
211     integ.addStepHandler(handler);
212     EventHandler[] functions = pb.getEventsHandlers();
213     double convergence = 1.0e-8 * maxStep;
214     for (int l = 0; l < functions.length; ++l) {
215       integ.addEventHandler(functions[l], Double.POSITIVE_INFINITY, convergence, 1000);
216     }
217     Assert.assertEquals(functions.length, integ.getEventHandlers().size());
218     integ.integrate(pb,
219                     pb.getInitialTime(), pb.getInitialState(),
220                     pb.getFinalTime(), new double[pb.getDimension()]);
221 
222     Assert.assertTrue(handler.getMaximalValueError() < 4.0e-7);
223     Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
224     Assert.assertEquals(12.0, handler.getLastTime(), convergence);
225     integ.clearEventHandlers();
226     Assert.assertEquals(0, integ.getEventHandlers().size());
227 
228   }

351   private static class KeplerStepHandler implements StepHandler {
352     public KeplerStepHandler(TestProblem3 pb) {
353       this.pb = pb;
354     }
355     public void init(double t0, double[] y0, double t) {
356       nbSteps = 0;
357       maxError = 0;
358     }
359     public void handleStep(StepInterpolator interpolator, boolean isLast)
360         throws MaxCountExceededException {
361 
362       ++nbSteps;
363       for (int a = 1; a < 100; ++a) {
364 
365         double prev   = interpolator.getPreviousTime();
366         double curr   = interpolator.getCurrentTime();
367         double interp = ((100 - a) * prev + a * curr) / 100;
368         interpolator.setInterpolatedTime(interp);
369 
370         double[] interpolatedY = interpolator.getInterpolatedState ();
371         double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
372         double dx = interpolatedY[0] - theoreticalY[0];
373         double dy = interpolatedY[1] - theoreticalY[1];
374         double error = dx * dx + dy * dy;
375         if (error > maxError) {
376           maxError = error;
377         }
378       }
379       if (isLast) {
380         Assert.assertTrue(maxError < 2.7e-6);
381         Assert.assertTrue(nbSteps < 80);
382       }
383     }
384     private int nbSteps;
385     private double maxError;
386     private TestProblem3 pb;
387   }

389   public static class VariableStepHandler implements StepHandler {
390     public VariableStepHandler() {
391         firstTime = true;
392         minStep = 0;
393         maxStep = 0;
394     }
395     public void init(double t0, double[] y0, double t) {
396       firstTime = true;
397       minStep = 0;
398       maxStep = 0;
399     }
400     public void handleStep(StepInterpolator interpolator,
401                            boolean isLast) {
402 
403       double step = FastMath.abs(interpolator.getCurrentTime()
404                              - interpolator.getPreviousTime());
405       if (firstTime) {
406         minStep   = FastMath.abs(step);
407         maxStep   = minStep;
408         firstTime = false;
409       } else {
410         if (step < minStep) {
411           minStep = step;
412         }
413         if (step > maxStep) {
414           maxStep = step;
415         }
416       }
417 
418       if (isLast) {
419         Assert.assertTrue(minStep < 8.2e-3);
420         Assert.assertTrue(maxStep > 1.5);
421       }
422     }
423     private boolean firstTime;
424     private double  minStep;
425     private double  maxStep;
426   }
