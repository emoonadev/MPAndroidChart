17  package org.apache.commons.math4.distribution;

19  import org.apache.commons.math4.analysis.UnivariateFunction;
20  import org.apache.commons.math4.analysis.integration.RombergIntegrator;
21  import org.apache.commons.math4.analysis.integration.UnivariateIntegrator;
22  import org.apache.commons.math4.distribution.AbstractRealDistribution;
23  import org.apache.commons.math4.exception.OutOfRangeException;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  /** Various tests related to MATH-699. */
28  public class AbstractRealDistributionTest {
29  
30      @Test
31      public void testContinuous() {
32          final double x0 = 0.0;
33          final double x1 = 1.0;
34          final double x2 = 2.0;
35          final double x3 = 3.0;
36          final double p12 = 0.5;
37          final AbstractRealDistribution distribution;
38          distribution = new AbstractRealDistribution() {
39              private static final long serialVersionUID = 1L;
40  
41              @Override
42              public double cumulativeProbability(final double x) {
43                  if ((x < x0) || (x > x3)) {
44                      throw new OutOfRangeException(x, x0, x3);
45                  }
46                  if (x <= x1) {
47                      return p12 * (x - x0) / (x1 - x0);
48                  } else if (x <= x2) {
49                      return p12;
50                  } else if (x <= x3) {
51                      return p12 + (1.0 - p12) * (x - x2) / (x3 - x2);
52                  }
53                  return 0.0;
54              }
55  
56              @Override
57              public double density(final double x) {
58                  if ((x < x0) || (x > x3)) {
59                      throw new OutOfRangeException(x, x0, x3);
60                  }
61                  if (x <= x1) {
62                      return p12 / (x1 - x0);
63                  } else if (x <= x2) {
64                      return 0.0;
65                  } else if (x <= x3) {
66                      return (1.0 - p12) / (x3 - x2);
67                  }
68                  return 0.0;
69              }
70  
71              @Override
72              public double getNumericalMean() {
73                  return ((x0 + x1) * p12 + (x2 + x3) * (1.0 - p12)) / 2.0;
74              }
75  
76              @Override
77              public double getNumericalVariance() {
78                  final double meanX = getNumericalMean();
79                  final double meanX2;
80                  meanX2 = ((x0 * x0 + x0 * x1 + x1 * x1) * p12 + (x2 * x2 + x2
81                          * x3 + x3 * x3)
82                          * (1.0 - p12)) / 3.0;
83                  return meanX2 - meanX * meanX;
84              }
85  
86              @Override
87              public double getSupportLowerBound() {
88                  return x0;
89              }
90  
91              @Override
92              public double getSupportUpperBound() {
93                  return x3;
94              }
95  
96              @Override
97              public boolean isSupportConnected() {
98                  return false;
99              }
100 
101             @Override
102             public double probability(final double x) {
103                 throw new UnsupportedOperationException();
104             }
105         };
106         final double expected = x1;
107         final double actual = distribution.inverseCumulativeProbability(p12);
108         Assert.assertEquals("", expected, actual,
109                 distribution.getSolverAbsoluteAccuracy());
110     }
111 
112     @Test
113     public void testDiscontinuous() {
114         final double x0 = 0.0;
115         final double x1 = 0.25;
116         final double x2 = 0.5;
117         final double x3 = 0.75;
118         final double x4 = 1.0;
119         final double p12 = 1.0 / 3.0;
120         final double p23 = 2.0 / 3.0;
121         final AbstractRealDistribution distribution;
122         distribution = new AbstractRealDistribution() {
123             private static final long serialVersionUID = 1L;
124 
125             @Override
126             public double cumulativeProbability(final double x) {
127                 if ((x < x0) || (x > x4)) {
128                     throw new OutOfRangeException(x, x0, x4);
129                 }
130                 if (x <= x1) {
131                     return p12 * (x - x0) / (x1 - x0);
132                 } else if (x <= x2) {
133                     return p12;
134                 } else if (x <= x3) {
135                     return p23;
136                 } else {
137                     return (1.0 - p23) * (x - x3) / (x4 - x3) + p23;
138                 }
139             }
140 
141             @Override
142             public double density(final double x) {
143                 if ((x < x0) || (x > x4)) {
144                     throw new OutOfRangeException(x, x0, x4);
145                 }
146                 if (x <= x1) {
147                     return p12 / (x1 - x0);
148                 } else if (x <= x2) {
149                     return 0.0;
150                 } else if (x <= x3) {
151                     return 0.0;
152                 } else {
153                     return (1.0 - p23) / (x4 - x3);
154                 }
155             }
156 
157             @Override
158             public double getNumericalMean() {
159                 final UnivariateFunction f = new UnivariateFunction() {
160 
161                     public double value(final double x) {
162                         return x * density(x);
163                     }
164                 };
165                 final UnivariateIntegrator integrator = new RombergIntegrator();
166                 return integrator.integrate(Integer.MAX_VALUE, f, x0, x4);
167             }
168 
169             @Override
170             public double getNumericalVariance() {
171                 final double meanX = getNumericalMean();
172                 final UnivariateFunction f = new UnivariateFunction() {
173 
174                     public double value(final double x) {
175                         return x * x * density(x);
176                     }
177                 };
178                 final UnivariateIntegrator integrator = new RombergIntegrator();
179                 final double meanX2 = integrator.integrate(Integer.MAX_VALUE,
180                         f, x0, x4);
181                 return meanX2 - meanX * meanX;
182             }
183 
184             @Override
185             public double getSupportLowerBound() {
186                 return x0;
187             }
188 
189             @Override
190             public double getSupportUpperBound() {
191                 return x4;
192             }
193 
194             @Override
195             public boolean isSupportConnected() {
196                 return false;
197             }
198 
199             @Override
200             public double probability(final double x) {
201                 throw new UnsupportedOperationException();
202             }
203         };
204         final double expected = x2;
205         final double actual = distribution.inverseCumulativeProbability(p23);
206         Assert.assertEquals("", expected, actual,
207                 distribution.getSolverAbsoluteAccuracy());
208 
209     }
