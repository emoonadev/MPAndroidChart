74      @Test
75      public void testEquals() {
76          Vector3D u1 = new Vector3D(1, 2, 3);
77          Vector3D u2 = new Vector3D(1, 2, 3);
78          Assert.assertTrue(u1.equals(u1));
79          Assert.assertTrue(u1.equals(u2));
80          Assert.assertFalse(u1.equals(new Rotation(1, 0, 0, 0, false)));
81          Assert.assertFalse(u1.equals(new Vector3D(1, 2, 3 + 10 * Precision.EPSILON)));
82          Assert.assertFalse(u1.equals(new Vector3D(1, 2 + 10 * Precision.EPSILON, 3)));
83          Assert.assertFalse(u1.equals(new Vector3D(1 + 10 * Precision.EPSILON, 2, 3)));
84          Assert.assertTrue(new Vector3D(0, Double.NaN, 0).equals(new Vector3D(0, 0, Double.NaN)));

333     @Test
334     public void testAccurateDotProduct() {
335         // the following two vectors are nearly but not exactly orthogonal
336         // naive dot product (i.e. computing u1.x * u2.x + u1.y * u2.y + u1.z * u2.z
337         // leads to a result of 0.0, instead of the correct -1.855129...
338         Vector3D u1 = new Vector3D(-1321008684645961.0 /  268435456.0,
339                                    -5774608829631843.0 /  268435456.0,
340                                    -7645843051051357.0 / 8589934592.0);
341         Vector3D u2 = new Vector3D(-5712344449280879.0 /    2097152.0,
342                                    -4550117129121957.0 /    2097152.0,
343                                     8846951984510141.0 /     131072.0);
344         double sNaive = u1.getX() * u2.getX() + u1.getY() * u2.getY() + u1.getZ() * u2.getZ();
345         double sAccurate = u1.dotProduct(u2);
346         Assert.assertEquals(0.0, sNaive, 1.0e-30);
347         Assert.assertEquals(-2088690039198397.0 / 1125899906842624.0, sAccurate, 1.0e-15);
348     }

350     @Test
351     public void testDotProduct() {
352         // we compare accurate versus naive dot product implementations
353         // on regular vectors (i.e. not extreme cases like in the previous test)
354         UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A, 553267312521321237l);
355         for (int i = 0; i < 10000; ++i) {
356             double ux = 10000 * random.nextDouble();
357             double uy = 10000 * random.nextDouble();
358             double uz = 10000 * random.nextDouble();
359             double vx = 10000 * random.nextDouble();
360             double vy = 10000 * random.nextDouble();
361             double vz = 10000 * random.nextDouble();
362             double sNaive = ux * vx + uy * vy + uz * vz;
363             double sAccurate = new Vector3D(ux, uy, uz).dotProduct(new Vector3D(vx, vy, vz));
364             Assert.assertEquals(sNaive, sAccurate, 2.5e-16 * sAccurate);
365         }
366     }

368     @Test
369     public void testAccurateCrossProduct() {
370         // the vectors u1 and u2 are nearly but not exactly anti-parallel
371         // (7.31e-16 degrees from 180 degrees) naive cross product (i.e.
372         // computing u1.x * u2.x + u1.y * u2.y + u1.z * u2.z
373         // leads to a result of   [0.0009765, -0.0001220, -0.0039062],
374         // instead of the correct [0.0006913, -0.0001254, -0.0007909]
375         final Vector3D u1 = new Vector3D(-1321008684645961.0 /   268435456.0,
376                                          -5774608829631843.0 /   268435456.0,
377                                          -7645843051051357.0 /  8589934592.0);
378         final Vector3D u2 = new Vector3D( 1796571811118507.0 /  2147483648.0,
379                                           7853468008299307.0 /  2147483648.0,
380                                           2599586637357461.0 / 17179869184.0);
381         final Vector3D u3 = new Vector3D(12753243807587107.0 / 18446744073709551616.0,
382                                          -2313766922703915.0 / 18446744073709551616.0,
383                                           -227970081415313.0 /   288230376151711744.0);
384         Vector3D cNaive = new Vector3D(u1.getY() * u2.getZ() - u1.getZ() * u2.getY(),
385                                        u1.getZ() * u2.getX() - u1.getX() * u2.getZ(),
386                                        u1.getX() * u2.getY() - u1.getY() * u2.getX());
387         Vector3D cAccurate = u1.crossProduct(u2);
388         Assert.assertTrue(u3.distance(cNaive) > 2.9 * u3.getNorm());
389         Assert.assertEquals(0.0, u3.distance(cAccurate), 1.0e-30 * cAccurate.getNorm());
390     }

393     public void testCrossProduct() {
394         // we compare accurate versus naive cross product implementations
395         // on regular vectors (i.e. not extreme cases like in the previous test)
396         UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A, 885362227452043215l);
397         for (int i = 0; i < 10000; ++i) {
398             double ux = 10000 * random.nextDouble();
399             double uy = 10000 * random.nextDouble();
400             double uz = 10000 * random.nextDouble();
401             double vx = 10000 * random.nextDouble();
402             double vy = 10000 * random.nextDouble();
403             double vz = 10000 * random.nextDouble();
404             Vector3D cNaive = new Vector3D(uy * vz - uz * vy, uz * vx - ux * vz, ux * vy - uy * vx);
405             Vector3D cAccurate = new Vector3D(ux, uy, uz).crossProduct(new Vector3D(vx, vy, vz));
406             Assert.assertEquals(0.0, cAccurate.distance(cNaive), 6.0e-15 * cAccurate.getNorm());
407         }
