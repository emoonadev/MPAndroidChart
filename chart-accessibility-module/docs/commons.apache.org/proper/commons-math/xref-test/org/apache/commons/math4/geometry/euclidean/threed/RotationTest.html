212   @Test
213   public void testVectorTwoPairs() throws MathArithmeticException {
214 
215     Vector3D u1 = new Vector3D(3, 0, 0);
216     Vector3D u2 = new Vector3D(0, 5, 0);
217     Vector3D v1 = new Vector3D(0, 0, 2);
218     Vector3D v2 = new Vector3D(-2, 0, 2);
219     Rotation r = new Rotation(u1, u2, v1, v2);
220     checkVector(r.applyTo(Vector3D.PLUS_I), Vector3D.PLUS_K);
221     checkVector(r.applyTo(Vector3D.PLUS_J), Vector3D.MINUS_I);
222 
223     r = new Rotation(u1, u2, u1.negate(), u2.negate());
224     Vector3D axis = r.getAxis(RotationConvention.VECTOR_OPERATOR);
225     if (Vector3D.dotProduct(axis, Vector3D.PLUS_K) > 0) {
226       checkVector(axis, Vector3D.PLUS_K);
227     } else {
228       checkVector(axis, Vector3D.MINUS_K);
229     }
230     checkAngle(r.getAngle(), FastMath.PI);
231 
232     double sqrt = FastMath.sqrt(2) / 2;
233     r = new Rotation(Vector3D.PLUS_I,  Vector3D.PLUS_J,
234                      new Vector3D(0.5, 0.5,  sqrt),
235                      new Vector3D(0.5, 0.5, -sqrt));
236     checkRotation(r, sqrt, 0.5, 0.5, 0);
237 
238     r = new Rotation(u1, u2, u1, Vector3D.crossProduct(u1, u2));
239     checkRotation(r, sqrt, -sqrt, 0, 0);
240 
241     checkRotation(new Rotation(u1, u2, u1, u2), 1, 0, 0, 0);
242 
243     try {
244         new Rotation(u1, u2, Vector3D.ZERO, v2);
245         Assert.fail("an exception should have been thrown");
246     } catch (MathArithmeticException e) {
247       // expected behavior
248     }

331     double d01 = m2[0][1] - m3[0][1];
332     double d02 = m2[0][2] - m3[0][2];
333     double d10 = m2[1][0] - m3[1][0];
334     double d11 = m2[1][1] - m3[1][1];
335     double d12 = m2[1][2] - m3[1][2];
