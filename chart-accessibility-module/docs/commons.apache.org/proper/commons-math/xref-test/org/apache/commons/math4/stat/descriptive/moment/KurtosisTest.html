17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.moment.Kurtosis;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link UnivariateStatistic} class.
27   */
28  public class KurtosisTest extends StorelessUnivariateStatisticAbstractTest{
29  
30      protected Kurtosis stat;
31  
32      /**
33       * {@inheritDoc}
34       */
35      @Override
36      public UnivariateStatistic getUnivariateStatistic() {
37          return new Kurtosis();
38      }
39  
40      /**
41       * {@inheritDoc}
42       */
43      @Override
44      public double expectedValue() {
45          return this.kurt;
46      }
47  
48      /**
49       * Make sure Double.NaN is returned iff n < 4
50       */
51      @Test
52      public void testNaN() {
53          Kurtosis kurt = new Kurtosis();
54          Assert.assertTrue(Double.isNaN(kurt.getResult()));
55          kurt.increment(1d);
56          Assert.assertTrue(Double.isNaN(kurt.getResult()));
57          kurt.increment(1d);
58          Assert.assertTrue(Double.isNaN(kurt.getResult()));
59          kurt.increment(1d);
60          Assert.assertTrue(Double.isNaN(kurt.getResult()));
61          kurt.increment(1d);
62          Assert.assertFalse(Double.isNaN(kurt.getResult()));
63      }
64  
65  }
