17  package org.apache.commons.math4.genetics;
18  
19  
20  import org.apache.commons.math4.genetics.Chromosome;
21  import org.apache.commons.math4.genetics.MutationPolicy;
22  import org.apache.commons.math4.genetics.RandomKey;
23  import org.apache.commons.math4.genetics.RandomKeyMutation;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  public class RandomKeyMutationTest {
28  
29      @Test
30      public void testMutate() {
31          MutationPolicy mutation = new RandomKeyMutation();
32          int l=10;
33          for (int i=0; i<20; i++) {
34              DummyRandomKey origRk = new DummyRandomKey(RandomKey.randomPermutation(l));
35              Chromosome mutated = mutation.mutate(origRk);
36              DummyRandomKey mutatedRk = (DummyRandomKey) mutated;
37  
38              int changes = 0;
39              for (int j=0; j<origRk.getLength(); j++) {
40                  if (origRk.getRepresentation().get(j) != mutatedRk.getRepresentation().get(j)) {
41                      changes++;
42                  }
43              }
44              Assert.assertEquals(1,changes);
45          }
46      }
47  
48  }
