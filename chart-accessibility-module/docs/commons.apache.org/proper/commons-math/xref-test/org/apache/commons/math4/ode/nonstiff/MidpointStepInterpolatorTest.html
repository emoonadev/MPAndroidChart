41  public class MidpointStepInterpolatorTest {
42  
43    @Test
44    public void testDerivativesConsistency()
45        throws DimensionMismatchException, NumberIsTooSmallException,
46               MaxCountExceededException, NoBracketingException {
47      TestProblem3 pb = new TestProblem3();
48      double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
49      MidpointIntegrator integ = new MidpointIntegrator(step);
50      StepInterpolatorTestUtils.checkDerivativesConsistency(integ, pb, 0.01, 6.6e-12);
51    }
52  
53    @Test
54    public void serialization()
55      throws IOException, ClassNotFoundException,
56             DimensionMismatchException, NumberIsTooSmallException,
57             MaxCountExceededException, NoBracketingException {
58  
59      TestProblem1 pb = new TestProblem1();
60      double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
61      MidpointIntegrator integ = new MidpointIntegrator(step);
62      integ.addStepHandler(new ContinuousOutputModel());
63      integ.integrate(pb,
64                      pb.getInitialTime(), pb.getInitialState(),
65                      pb.getFinalTime(), new double[pb.getDimension()]);
66  
67      ByteArrayOutputStream bos = new ByteArrayOutputStream();
68      ObjectOutputStream    oos = new ObjectOutputStream(bos);
69      for (StepHandler handler : integ.getStepHandlers()) {
70          oos.writeObject(handler);
71      }
72  
73      Assert.assertTrue(bos.size () > 135000);
74      Assert.assertTrue(bos.size () < 145000);
75  
76      ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
77      ObjectInputStream     ois = new ObjectInputStream(bis);
78      ContinuousOutputModel cm  = (ContinuousOutputModel) ois.readObject();
79  
80      Random random = new Random(347588535632l);
81      double maxError = 0.0;
82      for (int i = 0; i < 1000; ++i) {
83        double r = random.nextDouble();
84        double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
85        cm.setInterpolatedTime(time);
86        double[] interpolatedY = cm.getInterpolatedState ();
87        double[] theoreticalY  = pb.computeTheoreticalState(time);
88        double dx = interpolatedY[0] - theoreticalY[0];
89        double dy = interpolatedY[1] - theoreticalY[1];
90        double error = dx * dx + dy * dy;
91        if (error > maxError) {
92          maxError = error;
93        }
94      }
95  
96      Assert.assertTrue(maxError < 1.0e-6);
97  
98    }
