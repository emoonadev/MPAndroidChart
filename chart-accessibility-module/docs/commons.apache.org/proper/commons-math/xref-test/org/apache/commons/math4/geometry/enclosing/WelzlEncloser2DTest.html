87          UniformRandomProvider random = RandomSource.create(RandomSource.WELL_1024_A, 0xa2a63cad12c01fb2l);
88          for (int k = 0; k < 100; ++k) {
89              int nbPoints = random.nextInt(10000);
90              List<Vector2D> points = new ArrayList<Vector2D>();
91              for (int i = 0; i < nbPoints; ++i) {
92                  double x = random.nextDouble();
93                  double y = random.nextDouble();

95              }
96              checkDisk(points);

100     private List<Vector2D> buildList(final double ... coordinates) {
101         List<Vector2D> list = new ArrayList<Vector2D>(coordinates.length / 2);
102         for (int i = 0; i < coordinates.length; i += 2) {
103             list.add(new Vector2D(coordinates[i], coordinates[i + 1]));
104         }
105         return list;
106     }

108     private void checkDisk(List<Vector2D> points, List<Vector2D> refSupport) {
109 
110         EnclosingBall<Euclidean2D, Vector2D> disk = checkDisk(points);
111 
112         // compare computed disk with expected disk
113         DiskGenerator generator = new DiskGenerator();
114         EnclosingBall<Euclidean2D, Vector2D> expected = generator.ballOnSupport(refSupport);
115         Assert.assertEquals(refSupport.size(), disk.getSupportSize());
116         Assert.assertEquals(expected.getRadius(),        disk.getRadius(),        1.0e-10);
117         Assert.assertEquals(expected.getCenter().getX(), disk.getCenter().getX(), 1.0e-10);
118         Assert.assertEquals(expected.getCenter().getY(), disk.getCenter().getY(), 1.0e-10);
119 
120         for (Vector2D s : disk.getSupport()) {
121             boolean found = false;
122             for (Vector2D rs : refSupport) {
123                 if (s == rs) {
124                     found = true;
125                 }
126             }
127             Assert.assertTrue(found);
128         }
129 
130         // check removing any point of the support disk fails to enclose the point
131         for (int i = 0; i < disk.getSupportSize(); ++i) {
132             List<Vector2D> reducedSupport = new ArrayList<Vector2D>();
133             int count = 0;
134             for (Vector2D s : disk.getSupport()) {
135                 if (count++ != i) {
136                     reducedSupport.add(s);
137                 }
138             }
139             EnclosingBall<Euclidean2D, Vector2D> reducedDisk = generator.ballOnSupport(reducedSupport);
140             boolean foundOutside = false;
141             for (int j = 0; j < points.size() && !foundOutside; ++j) {
142                 if (!reducedDisk.contains(points.get(j), 1.0e-10)) {
143                     foundOutside = true;
144                 }
145             }
146             Assert.assertTrue(foundOutside);
147         }
148 
149     }
150 
151     private EnclosingBall<Euclidean2D, Vector2D> checkDisk(List<Vector2D> points) {
152 
153         WelzlEncloser<Euclidean2D, Vector2D> encloser =
154                 new WelzlEncloser<Euclidean2D, Vector2D>(1.0e-10, new DiskGenerator());
155         EnclosingBall<Euclidean2D, Vector2D> disk = encloser.enclose(points);
156 
157         // all points are enclosed
158         for (Vector2D v : points) {
159             Assert.assertTrue(disk.contains(v, 1.0e-10));
160         }
161 
162         for (Vector2D v : points) {
163             boolean inSupport = false;
164             for (Vector2D s : disk.getSupport()) {
165                 if (v == s) {
166                     inSupport = true;
167                 }
168             }
169             if (inSupport) {
170                 // points on the support should be outside of reduced ball
171                 Assert.assertFalse(disk.contains(v, -0.001));
172             }
173         }
174 
175         return disk;
176 
177     }
