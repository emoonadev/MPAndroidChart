46   * </code></pre>

49      /** Number of runs. */
50      private static final int MONTE_CARLO_RUNS = Integer.parseInt(System.getProperty("mcRuns",
51                                                                                      "100"));

53      /**
54       * Using a Monte-Carlo procedure, this test checks the error estimations
55       * as provided by the square-root of the diagonal elements of the
56       * covariance matrix.
57       * <br/>
58       * The test generates sets of observations, each sampled from
59       * a Gaussian distribution.
60       * <br/>
61       * The optimization problem solved is defined in class
62       * {@link StraightLineProblem}.
63       * <br/>
64       * The output (on stdout) will be a table summarizing the distribution
65       * of parameters generated by the Monte-Carlo process and by the direct
66       * estimation provided by the diagonal elements of the covariance matrix.
67       */
68      @Test
69      public void testParametersErrorMonteCarloObservations() {
70          // Error on the observations.
71          final double yError = 15;
72  
73          // True values of the parameters.
74          final double slope = 123.456;
75          final double offset = -98.765;
76  
77          // Samples generator.
78          final RandomStraightLinePointGenerator lineGenerator
79              = new RandomStraightLinePointGenerator(slope, offset,
80                                                     yError,
81                                                     -1e3, 1e4,
82                                                     138577L);
83  
84          // Number of observations.
85          final int numObs = 100; // XXX Should be a command-line option.
86          // number of parameters.
87          final int numParams = 2;
88  
89          // Parameters found for each of Monte-Carlo run.
90          final SummaryStatistics[] paramsFoundByDirectSolution = new SummaryStatistics[numParams];
91          // Sigma estimations (square-root of the diagonal elements of the
92          // covariance matrix), for each Monte-Carlo run.
93          final SummaryStatistics[] sigmaEstimate = new SummaryStatistics[numParams];
94  
95          // Initialize statistics accumulators.
96          for (int i = 0; i < numParams; i++) {
97              paramsFoundByDirectSolution[i] = new SummaryStatistics();
98              sigmaEstimate[i] = new SummaryStatistics();
99          }
100 
101         final RealVector init = new ArrayRealVector(new double[]{ slope, offset }, false);
102 
103         // Monte-Carlo (generates many sets of observations).
104         final int mcRepeat = MONTE_CARLO_RUNS;
105         int mcCount = 0;
106         while (mcCount < mcRepeat) {
107             // Observations.
108             final Point2D.Double[] obs = lineGenerator.generate(numObs);
109 
110             final StraightLineProblem problem = new StraightLineProblem(yError);
111             for (int i = 0; i < numObs; i++) {
112                 final Point2D.Double p = obs[i];
113                 problem.addPoint(p.x, p.y);
114             }
115 
116             // Direct solution (using simple regression).
117             final double[] regress = problem.solve();
118 
119             // Estimation of the standard deviation (diagonal elements of the
120             // covariance matrix).
121             final LeastSquaresProblem lsp = builder(problem).build();
122 
123             final RealVector sigma = lsp.evaluate(init).getSigma(1e-14);
124 
125             // Accumulate statistics.
126             for (int i = 0; i < numParams; i++) {
127                 paramsFoundByDirectSolution[i].addValue(regress[i]);
128                 sigmaEstimate[i].addValue(sigma.getEntry(i));
129             }
130 
131             // Next Monte-Carlo.
132             ++mcCount;
133         }
134 
135         // Print statistics.
136         final String line = "--------------------------------------------------------------";
137         System.out.println("                 True value       Mean        Std deviation");
138         for (int i = 0; i < numParams; i++) {
139             System.out.println(line);
140             System.out.println("Parameter #" + i);
141 
142             StatisticalSummary s = paramsFoundByDirectSolution[i].getSummary();
143             System.out.printf("              %+.6e   %+.6e   %+.6e\n",
144                               init.getEntry(i),
145                               s.getMean(),
146                               s.getStandardDeviation());
147 
148             s = sigmaEstimate[i].getSummary();
149             System.out.printf("sigma: %+.6e (%+.6e)\n",
150                               s.getMean(),
151                               s.getStandardDeviation());
152         }
153         System.out.println(line);
154 
155         // Check the error estimation.
156         for (int i = 0; i < numParams; i++) {
157             Assert.assertEquals(paramsFoundByDirectSolution[i].getSummary().getStandardDeviation(),
158                                 sigmaEstimate[i].getSummary().getMean(),
159                                 8e-2);
160         }
161     }

163     /**
164      * In this test, the set of observations is fixed.
165      * Using a Monte-Carlo procedure, it generates sets of parameters,
166      * and determine the parameter change that will result in the
167      * normalized chi-square becoming larger by one than the value from
168      * the best fit solution.
169      * <br/>
170      * The optimization problem solved is defined in class
171      * {@link StraightLineProblem}.
172      * <br/>
173      * The output (on stdout) will be a list of lines containing:
174      * <ul>
175      *  <li>slope of the straight line,</li>
176      *  <li>intercept of the straight line,</li>
177      *  <li>chi-square of the solution defined by the above two values.</li>
178      * </ul>
179      * The output is separated into two blocks (with a blank line between
180      * them); the first block will contain all parameter sets for which
181      * {@code chi2 < chi2_b + 1}
182      * and the second block, all sets for which
183      * {@code chi2 >= chi2_b + 1}
184      * where {@code chi2_b} is the lowest chi-square (corresponding to the
185      * best solution).
186      */
187     @Test
188     public void testParametersErrorMonteCarloParameters() {
189         // Error on the observations.
190         final double yError = 15;
191 
192         // True values of the parameters.
193         final double slope = 123.456;
194         final double offset = -98.765;
195 
196         // Samples generator.
197         final RandomStraightLinePointGenerator lineGenerator
198             = new RandomStraightLinePointGenerator(slope, offset,
199                                                    yError,
200                                                    -1e3, 1e4,
201                                                    13839013L);
202 
203         // Number of observations.
204         final int numObs = 10;
205         // number of parameters.
206 
207         // Create a single set of observations.
208         final Point2D.Double[] obs = lineGenerator.generate(numObs);
209 
210         final StraightLineProblem problem = new StraightLineProblem(yError);
211         for (int i = 0; i < numObs; i++) {
212             final Point2D.Double p = obs[i];
213             problem.addPoint(p.x, p.y);
214         }
215 
216         // Direct solution (using simple regression).
217         final RealVector regress = new ArrayRealVector(problem.solve(), false);
218 
219         // Dummy optimizer (to compute the chi-square).
220         final LeastSquaresProblem lsp = builder(problem).build();
221 
222         // Get chi-square of the best parameters set for the given set of
223         // observations.
224         final double bestChi2N = getChi2N(lsp, regress);
225         final RealVector sigma = lsp.evaluate(regress).getSigma(1e-14);
226 
227         // Monte-Carlo (generates a grid of parameters).
228         final int mcRepeat = MONTE_CARLO_RUNS;
229         final int gridSize = (int) FastMath.sqrt(mcRepeat);
230 
231         // Parameters found for each of Monte-Carlo run.
232         // Index 0 = slope
233         // Index 1 = offset
234         // Index 2 = normalized chi2
235         final List<double[]> paramsAndChi2 = new ArrayList<double[]>(gridSize * gridSize);
236 
237         final double slopeRange = 10 * sigma.getEntry(0);
238         final double offsetRange = 10 * sigma.getEntry(1);
239         final double minSlope = slope - 0.5 * slopeRange;
240         final double minOffset = offset - 0.5 * offsetRange;
241         final double deltaSlope =  slopeRange/ gridSize;
242         final double deltaOffset = offsetRange / gridSize;
243         for (int i = 0; i < gridSize; i++) {
244             final double s = minSlope + i * deltaSlope;
245             for (int j = 0; j < gridSize; j++) {
246                 final double o = minOffset + j * deltaOffset;
247                 final double chi2N = getChi2N(lsp,
248                         new ArrayRealVector(new double[] {s, o}, false));
249 
250                 paramsAndChi2.add(new double[] {s, o, chi2N});
251             }
252         }
253 
254         // Output (for use with "gnuplot").
255 
256         // Some info.
257 
258         // For plotting separately sets of parameters that have a large chi2.
259         final double chi2NPlusOne = bestChi2N + 1;
260         int numLarger = 0;
261 
262         final String lineFmt = "%+.10e %+.10e   %.8e\n";
263 
264         // Point with smallest chi-square.
265         System.out.printf(lineFmt, regress.getEntry(0), regress.getEntry(1), bestChi2N);
266         System.out.println(); // Empty line.
267 
268         // Points within the confidence interval.
269         for (double[] d : paramsAndChi2) {
270             if (d[2] <= chi2NPlusOne) {
271                 System.out.printf(lineFmt, d[0], d[1], d[2]);
272             }
273         }
274         System.out.println(); // Empty line.
275 
276         // Points outside the confidence interval.
277         for (double[] d : paramsAndChi2) {
278             if (d[2] > chi2NPlusOne) {
279                 ++numLarger;
280                 System.out.printf(lineFmt, d[0], d[1], d[2]);
281             }
282         }
283         System.out.println(); // Empty line.
284 
285         System.out.println("# sigma=" + sigma.toString());
286         System.out.println("# " + numLarger + " sets filtered out");
287     }

289     LeastSquaresBuilder builder(StraightLineProblem problem){
290         return new LeastSquaresBuilder()
291                 .model(problem.getModelFunction(), problem.getModelFunctionJacobian())
292                 .target(problem.target())
293                 .weight(new DiagonalMatrix(problem.weight()))
294                 //unused start point to avoid NPE
295                 .start(new double[2]);
296     }
297     /**
298      * @return the normalized chi-square.
299      */
300     private double getChi2N(LeastSquaresProblem lsp,
301                             RealVector params) {
302         final double cost = lsp.evaluate(params).getCost();
303         return cost * cost / (lsp.getObservationSize() - params.getDimension());
304     }
