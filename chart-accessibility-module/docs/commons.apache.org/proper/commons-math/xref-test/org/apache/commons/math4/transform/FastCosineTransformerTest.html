132     @Override
133     double[] transform(final double[] x, final TransformType type) {
134         final int n = x.length;
135         final double[] y = new double[n];
136         final double[] cos = new double[2 * (n - 1)];
137         for (int i = 0; i < cos.length; i++) {
138             cos[i] = FastMath.cos(FastMath.PI * i / (n - 1.0));
139         }
140         int sgn = 1;
141         for (int j = 0; j < n; j++) {
142             double yj = 0.5 * (x[0] + sgn * x[n - 1]);
143             for (int i = 1; i < n - 1; i++) {
144                 yj += x[i] * cos[(i * j) % cos.length];
145             }
146             y[j] = yj;
147             sgn *= -1;
148         }
149         final double s;
150         if (type == TransformType.FORWARD) {
151             if (normalization == DctNormalization.STANDARD_DCT_I) {
152                 s = 1.0;
153             } else if (normalization == DctNormalization.ORTHOGONAL_DCT_I) {
154                 s = FastMath.sqrt(2.0 / (n - 1.0));
155             } else {
156                 throw new MathIllegalStateException();
157             }
158         } else if (type == TransformType.INVERSE) {
159             if (normalization == DctNormalization.STANDARD_DCT_I) {
160                 s = 2.0 / (n - 1.0);
161             } else if (normalization == DctNormalization.ORTHOGONAL_DCT_I) {
162                 s = FastMath.sqrt(2.0 / (n - 1.0));
163             } else {
164                 throw new MathIllegalStateException();
165             }
166         } else {
167             /*
168              * Should never occur. This clause is a safeguard in case other
169              * types are used to TransformType (which should not be done).
170              */
171             throw new MathIllegalStateException();
172         }
173         TransformUtils.scaleArray(y, s);
174         return y;
175     }

181     /** Test of transformer for the ad hoc data. */
182     @Test
183     public void testAdHocData() {
184         FastCosineTransformer transformer;
185         transformer = new FastCosineTransformer(DctNormalization.STANDARD_DCT_I);
186         double result[], tolerance = 1E-12;
187 
188         double x[] = {
189             0.0, 1.0, 4.0, 9.0, 16.0, 25.0, 36.0, 49.0, 64.0
190         };
191         double y[] =
192             {
193                 172.0, -105.096569476353, 27.3137084989848, -12.9593152353742,
194                 8.0, -5.78585076868676, 4.68629150101524, -4.15826451958632,
195                 4.0
196             };
197 
198         result = transformer.transform(x, TransformType.FORWARD);
199         for (int i = 0; i < result.length; i++) {
200             Assert.assertEquals(y[i], result[i], tolerance);
201         }
202 
203         result = transformer.transform(y, TransformType.INVERSE);
204         for (int i = 0; i < result.length; i++) {
205             Assert.assertEquals(x[i], result[i], tolerance);
206         }
207 
208         TransformUtils.scaleArray(x, FastMath.sqrt(0.5 * (x.length - 1)));
209 
210         transformer = new FastCosineTransformer(DctNormalization.ORTHOGONAL_DCT_I);
211         result = transformer.transform(y, TransformType.FORWARD);
212         for (int i = 0; i < result.length; i++) {
213             Assert.assertEquals(x[i], result[i], tolerance);
214         }
215 
216         result = transformer.transform(x, TransformType.INVERSE);
217         for (int i = 0; i < result.length; i++) {
218             Assert.assertEquals(y[i], result[i], tolerance);
219         }
220     }

222     /** Test of parameters for the transformer. */
223     @Test
224     public void testParameters()
225         throws Exception {
226         UnivariateFunction f = new Sin();
227         FastCosineTransformer transformer;
228         transformer = new FastCosineTransformer(DctNormalization.STANDARD_DCT_I);
229 
230         try {
231             // bad interval
232             transformer.transform(f, 1, -1, 65, TransformType.FORWARD);
233             Assert.fail("Expecting MathIllegalArgumentException - bad interval");
234         } catch (MathIllegalArgumentException ex) {
235             // expected
236         }
237         try {
238             // bad samples number
239             transformer.transform(f, -1, 1, 1, TransformType.FORWARD);
240             Assert.fail("Expecting MathIllegalArgumentException - bad samples number");
241         } catch (MathIllegalArgumentException ex) {
242             // expected
243         }
244         try {
245             // bad samples number
246             transformer.transform(f, -1, 1, 64, TransformType.FORWARD);
247             Assert.fail("Expecting MathIllegalArgumentException - bad samples number");
248         } catch (MathIllegalArgumentException ex) {
249             // expected
250         }
251     }

253     /** Test of transformer for the sine function. */
254     @Test
255     public void testSinFunction() {
256         UnivariateFunction f = new Sin();
257         FastCosineTransformer transformer;
258         transformer = new FastCosineTransformer(DctNormalization.STANDARD_DCT_I);
259         double min, max, result[], tolerance = 1E-12;
260         int N = 9;
261 
262         double expected[] =
263             {
264                 0.0, 3.26197262739567, 0.0, -2.17958042710327, 0.0,
265                 -0.648846697642915, 0.0, -0.433545502649478, 0.0
266             };
267         min = 0.0;
268         max = 2.0 * FastMath.PI * N / (N - 1);
269         result = transformer.transform(f, min, max, N, TransformType.FORWARD);
270         for (int i = 0; i < N; i++) {
271             Assert.assertEquals(expected[i], result[i], tolerance);
272         }
273 
274         min = -FastMath.PI;
275         max = FastMath.PI * (N + 1) / (N - 1);
276         result = transformer.transform(f, min, max, N, TransformType.FORWARD);
277         for (int i = 0; i < N; i++) {
278             Assert.assertEquals(-expected[i], result[i], tolerance);
279         }
280     }
