18  package org.apache.commons.math4.ml.clustering.evaluation;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.junit.Assert.assertFalse;
22  import static org.junit.Assert.assertTrue;
23  
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  import org.apache.commons.math4.ml.clustering.Cluster;
28  import org.apache.commons.math4.ml.clustering.DoublePoint;
29  import org.apache.commons.math4.ml.clustering.evaluation.ClusterEvaluator;
30  import org.apache.commons.math4.ml.clustering.evaluation.SumOfClusterVariances;
31  import org.apache.commons.math4.ml.distance.EuclideanDistance;
32  import org.junit.Before;
33  import org.junit.Test;
34  
35  public class SumOfClusterVariancesTest {
36  
37      private ClusterEvaluator<DoublePoint> evaluator;
38  
39      @Before
40      public void setUp() {
41          evaluator = new SumOfClusterVariances<DoublePoint>(new EuclideanDistance());
42      }
43  
44      @Test
45      public void testScore() {
46          final DoublePoint[] points1 = new DoublePoint[] {
47                  new DoublePoint(new double[] { 1 }),
48                  new DoublePoint(new double[] { 2 }),
49                  new DoublePoint(new double[] { 3 })
50          };
51  
52          final DoublePoint[] points2 = new DoublePoint[] {
53                  new DoublePoint(new double[] { 1 }),
54                  new DoublePoint(new double[] { 5 }),
55                  new DoublePoint(new double[] { 10 })
56          };
57  
58          final List<Cluster<DoublePoint>> clusters = new ArrayList<Cluster<DoublePoint>>();
59  
60          final Cluster<DoublePoint> cluster1 = new Cluster<DoublePoint>();
61          for (DoublePoint p : points1) {
62              cluster1.addPoint(p);
63          }
64          clusters.add(cluster1);
65  
66          assertEquals(1.0/3.0, evaluator.score(clusters), 1e-6);
67  
68          final Cluster<DoublePoint> cluster2 = new Cluster<DoublePoint>();
69          for (DoublePoint p : points2) {
70              cluster2.addPoint(p);
71          }
72          clusters.add(cluster2);
73  
74          assertEquals(6.148148148, evaluator.score(clusters), 1e-6);
75      }
76  
77      @Test
78      public void testOrdering() {
79          assertTrue(evaluator.isBetterScore(10, 20));
80          assertFalse(evaluator.isBetterScore(20, 1));
81      }
82  }
