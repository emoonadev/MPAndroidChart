112     @Test
113     public void testPositiveOctantByIntersection() {
114         double tol = 0.01;
115         double sinTol = FastMath.sin(tol);
116         RegionFactory<Sphere2D> factory = new RegionFactory<Sphere2D>();
117         SphericalPolygonsSet plusX = new SphericalPolygonsSet(Vector3D.PLUS_I, tol);
118         SphericalPolygonsSet plusY = new SphericalPolygonsSet(Vector3D.PLUS_J, tol);
119         SphericalPolygonsSet plusZ = new SphericalPolygonsSet(Vector3D.PLUS_K, tol);
120         SphericalPolygonsSet octant =
121                 (SphericalPolygonsSet) factory.intersection(factory.intersection(plusX, plusY), plusZ);
122         UnitSphereRandomVectorGenerator random =
123                 new UnitSphereRandomVectorGenerator(3, RandomSource.create(RandomSource.WELL_1024_A,
124                                                                            0x9c9802fde3cbcf25l));
125         for (int i = 0; i < 1000; ++i) {
126             Vector3D v = new Vector3D(random.nextVector());
127             if ((v.getX() > sinTol) && (v.getY() > sinTol) && (v.getZ() > sinTol)) {
128                 Assert.assertEquals(Location.INSIDE, octant.checkPoint(new S2Point(v)));
129             } else if ((v.getX() < -sinTol) || (v.getY() < -sinTol) || (v.getZ() < -sinTol)) {
130                 Assert.assertEquals(Location.OUTSIDE, octant.checkPoint(new S2Point(v)));
131             } else {
132                 Assert.assertEquals(Location.BOUNDARY, octant.checkPoint(new S2Point(v)));
133             }
134         }
135 
136         List<Vertex> loops = octant.getBoundaryLoops();
137         Assert.assertEquals(1, loops.size());
138         boolean xPFound = false;
139         boolean yPFound = false;
140         boolean zPFound = false;
141         boolean xVFound = false;
142         boolean yVFound = false;
143         boolean zVFound = false;
144         Vertex first = loops.get(0);
145         int count = 0;
146         for (Vertex v = first; count == 0 || v != first; v = v.getOutgoing().getEnd()) {
147             ++count;
148             Edge e = v.getIncoming();
149             Assert.assertTrue(v == e.getStart().getOutgoing().getEnd());
150             xPFound = xPFound || e.getCircle().getPole().distance(Vector3D.PLUS_I) < 1.0e-10;
151             yPFound = yPFound || e.getCircle().getPole().distance(Vector3D.PLUS_J) < 1.0e-10;
152             zPFound = zPFound || e.getCircle().getPole().distance(Vector3D.PLUS_K) < 1.0e-10;
153             Assert.assertEquals(0.5 * FastMath.PI, e.getLength(), 1.0e-10);
154             xVFound = xVFound || v.getLocation().getVector().distance(Vector3D.PLUS_I) < 1.0e-10;
155             yVFound = yVFound || v.getLocation().getVector().distance(Vector3D.PLUS_J) < 1.0e-10;
156             zVFound = zVFound || v.getLocation().getVector().distance(Vector3D.PLUS_K) < 1.0e-10;
157         }
158         Assert.assertTrue(xPFound);
159         Assert.assertTrue(yPFound);
160         Assert.assertTrue(zPFound);
161         Assert.assertTrue(xVFound);
162         Assert.assertTrue(yVFound);
163         Assert.assertTrue(zVFound);
164         Assert.assertEquals(3, count);
165 
166         Assert.assertEquals(0.0,
167                             ((S2Point) octant.getBarycenter()).distance(new S2Point(new Vector3D(1, 1, 1))),
168                             1.0e-10);
169         Assert.assertEquals(0.5 * FastMath.PI, octant.getSize(), 1.0e-10);
170 
171         EnclosingBall<Sphere2D, S2Point> cap = octant.getEnclosingCap();
172         Assert.assertEquals(0.0, octant.getBarycenter().distance(cap.getCenter()), 1.0e-10);
173         Assert.assertEquals(FastMath.acos(1.0 / FastMath.sqrt(3)), cap.getRadius(), 1.0e-10);
174 
175         EnclosingBall<Sphere2D, S2Point> reversedCap =
176                 ((SphericalPolygonsSet) factory.getComplement(octant)).getEnclosingCap();
177         Assert.assertEquals(0, reversedCap.getCenter().distance(new S2Point(new Vector3D(-1, -1, -1))), 1.0e-10);
178         Assert.assertEquals(FastMath.PI - FastMath.asin(1.0 / FastMath.sqrt(3)), reversedCap.getRadius(), 1.0e-10);
179

202     @Test
203     public void testNonConvex() {
204         double tol = 0.01;
205         double sinTol = FastMath.sin(tol);
206         RegionFactory<Sphere2D> factory = new RegionFactory<Sphere2D>();
207         SphericalPolygonsSet plusX = new SphericalPolygonsSet(Vector3D.PLUS_I, tol);
208         SphericalPolygonsSet plusY = new SphericalPolygonsSet(Vector3D.PLUS_J, tol);
209         SphericalPolygonsSet plusZ = new SphericalPolygonsSet(Vector3D.PLUS_K, tol);
210         SphericalPolygonsSet threeOctants =
211                 (SphericalPolygonsSet) factory.difference(plusZ, factory.intersection(plusX, plusY));
212 
213         UnitSphereRandomVectorGenerator random =
214                 new UnitSphereRandomVectorGenerator(3, RandomSource.create(RandomSource.WELL_1024_A,
215                                                                            0x9c9802fde3cbcf25l));
216         for (int i = 0; i < 1000; ++i) {
217             Vector3D v = new Vector3D(random.nextVector());
218             if (((v.getX() < -sinTol) || (v.getY() < -sinTol)) && (v.getZ() > sinTol)) {
219                 Assert.assertEquals(Location.INSIDE, threeOctants.checkPoint(new S2Point(v)));
220             } else if (((v.getX() > sinTol) && (v.getY() > sinTol)) || (v.getZ() < -sinTol)) {
221                 Assert.assertEquals(Location.OUTSIDE, threeOctants.checkPoint(new S2Point(v)));
222             } else {
223                 Assert.assertEquals(Location.BOUNDARY, threeOctants.checkPoint(new S2Point(v)));
224             }
225         }
226 
227         List<Vertex> loops = threeOctants.getBoundaryLoops();
228         Assert.assertEquals(1, loops.size());
229         boolean xPFound = false;
230         boolean yPFound = false;
231         boolean zPFound = false;
232         boolean xVFound = false;
233         boolean yVFound = false;
234         boolean zVFound = false;
235         Vertex first = loops.get(0);
236         int count = 0;
237         double sumPoleX = 0;
238         double sumPoleY = 0;
239         double sumPoleZ = 0;
240         for (Vertex v = first; count == 0 || v != first; v = v.getOutgoing().getEnd()) {
241             ++count;
242             Edge e = v.getIncoming();
243             Assert.assertTrue(v == e.getStart().getOutgoing().getEnd());
244             if (e.getCircle().getPole().distance(Vector3D.MINUS_I) < 1.0e-10) {
245                 xPFound = true;
246                 sumPoleX += e.getLength();
247             } else if (e.getCircle().getPole().distance(Vector3D.MINUS_J) < 1.0e-10) {
248                 yPFound = true;
249                 sumPoleY += e.getLength();
250             } else {
251                 Assert.assertEquals(0.0, e.getCircle().getPole().distance(Vector3D.PLUS_K), 1.0e-10);
252                 zPFound = true;
253                 sumPoleZ += e.getLength();
254             }
255             xVFound = xVFound || v.getLocation().getVector().distance(Vector3D.PLUS_I) < 1.0e-10;
256             yVFound = yVFound || v.getLocation().getVector().distance(Vector3D.PLUS_J) < 1.0e-10;
257             zVFound = zVFound || v.getLocation().getVector().distance(Vector3D.PLUS_K) < 1.0e-10;
258         }
259         Assert.assertTrue(xPFound);
260         Assert.assertTrue(yPFound);
261         Assert.assertTrue(zPFound);
262         Assert.assertTrue(xVFound);
263         Assert.assertTrue(yVFound);
264         Assert.assertTrue(zVFound);
265         Assert.assertEquals(0.5 * FastMath.PI, sumPoleX, 1.0e-10);
266         Assert.assertEquals(0.5 * FastMath.PI, sumPoleY, 1.0e-10);
267         Assert.assertEquals(1.5 * FastMath.PI, sumPoleZ, 1.0e-10);
268 
269         Assert.assertEquals(1.5 * FastMath.PI, threeOctants.getSize(), 1.0e-10);
270 
271     }

373     @Test
374     public void testPartWithHole() {
375         double tol = 0.01;
376         double alpha = 0.7;
377         S2Point center = new S2Point(new Vector3D(1, 1, 1));
378         SphericalPolygonsSet hexa = new SphericalPolygonsSet(center.getVector(), Vector3D.PLUS_K, alpha, 6, tol);
379         SphericalPolygonsSet hole  = new SphericalPolygonsSet(tol,
380                                                               new S2Point(FastMath.PI / 6, FastMath.PI / 3),
381                                                               new S2Point(FastMath.PI / 3, FastMath.PI / 3),
382                                                               new S2Point(FastMath.PI / 4, FastMath.PI / 6));
383         SphericalPolygonsSet hexaWithHole =
384                 (SphericalPolygonsSet) new RegionFactory<Sphere2D>().difference(hexa, hole);
385 
386         for (double phi = center.getPhi() - alpha + 0.1; phi < center.getPhi() + alpha - 0.1; phi += 0.07) {
387             Location l = hexaWithHole.checkPoint(new S2Point(FastMath.PI / 4, phi));
388             if (phi < FastMath.PI / 6 || phi > FastMath.PI / 3) {
389                 Assert.assertEquals(Location.INSIDE,  l);
390             } else {
391                 Assert.assertEquals(Location.OUTSIDE, l);
392             }
393         }
394 
395         // there should be two separate boundary loops
396         Assert.assertEquals(2, hexaWithHole.getBoundaryLoops().size());
397 
398         Assert.assertEquals(hexa.getBoundarySize() + hole.getBoundarySize(), hexaWithHole.getBoundarySize(), 1.0e-10);
399         Assert.assertEquals(hexa.getSize() - hole.getSize(), hexaWithHole.getSize(), 1.0e-10);
400 
401     }

533     private SubCircle create(Vector3D pole, Vector3D x, Vector3D y,
534                              double tolerance, double ... limits) {
535         RegionFactory<Sphere1D> factory = new RegionFactory<Sphere1D>();
536         Circle circle = new Circle(pole, tolerance);
537         Circle phased =
538                 (Circle) Circle.getTransform(new Rotation(circle.getXAxis(), circle.getYAxis(), x, y)).apply(circle);
539         ArcsSet set = (ArcsSet) factory.getComplement(new ArcsSet(tolerance));
540         for (int i = 0; i < limits.length; i += 2) {
541             set = (ArcsSet) factory.union(set, new ArcsSet(limits[i], limits[i + 1], tolerance));
542         }
543         return new SubCircle(phased, set);
544     }

546     private SphericalPolygonsSet buildSimpleZone(double[][] points) {
547         final S2Point[] vertices = new S2Point[points.length];
548         for (int i = 0; i < points.length; ++i) {
549             vertices[i] = s2Point(points[i][0], points[i][1]);
550         }
551         return new SphericalPolygonsSet(1.0e-10, vertices);
552     }
553 
554     private S2Point s2Point(double latitude, double longitude) {
555         return new S2Point(FastMath.toRadians(longitude), FastMath.toRadians(90.0 - latitude));
556     }
