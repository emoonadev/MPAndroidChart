17  package org.apache.commons.math4.analysis;
18  
19  
20  /**
21   * Wrapper class for counting functions calls.
22   *
23   */
24  public class MonitoredFunction implements UnivariateFunction {
25  
26      public MonitoredFunction(UnivariateFunction f) {
27          callsCount = 0;
28          this.f = f;
29      }
30  
31      public void setCallsCount(int callsCount) {
32          this.callsCount = callsCount;
33      }
34  
35      public int getCallsCount() {
36          return callsCount;
37      }
38  
39      @Override
40      public double value(double x) {
41          ++callsCount;
42          return f.value(x);
43      }
44  
45      private int callsCount;
46      private final UnivariateFunction f;
47  
48  }
