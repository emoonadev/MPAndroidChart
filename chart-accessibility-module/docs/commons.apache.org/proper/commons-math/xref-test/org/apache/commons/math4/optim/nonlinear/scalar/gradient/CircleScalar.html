18  package org.apache.commons.math4.optim.nonlinear.scalar.gradient;
19  
20  import java.util.ArrayList;
21  
22  import org.apache.commons.math4.analysis.MultivariateFunction;
23  import org.apache.commons.math4.analysis.MultivariateVectorFunction;
24  import org.apache.commons.math4.geometry.euclidean.twod.Vector2D;
25  import org.apache.commons.math4.optim.nonlinear.scalar.ObjectiveFunction;
26  import org.apache.commons.math4.optim.nonlinear.scalar.ObjectiveFunctionGradient;
27  
28  /**
29   * Class used in the tests.
30   */
31  public class CircleScalar {
32      private ArrayList<Vector2D> points;
33  
34      public CircleScalar() {
35          points  = new ArrayList<Vector2D>();
36      }
37  
38      public void addPoint(double px, double py) {
39          points.add(new Vector2D(px, py));
40      }
41  
42      public double getRadius(Vector2D center) {
43          double r = 0;
44          for (Vector2D point : points) {
45              r += point.distance(center);
46          }
47          return r / points.size();
48      }
49  
50      public ObjectiveFunction getObjectiveFunction() {
51          return new ObjectiveFunction(new MultivariateFunction() {
52                  public double value(double[] params)  {
53                      Vector2D center = new Vector2D(params[0], params[1]);
54                      double radius = getRadius(center);
55                      double sum = 0;
56                      for (Vector2D point : points) {
57                          double di = point.distance(center) - radius;
58                          sum += di * di;
59                      }
60                      return sum;
61                  }
62              });
63      }
64  
65      public ObjectiveFunctionGradient getObjectiveFunctionGradient() {
66          return new ObjectiveFunctionGradient(new MultivariateVectorFunction() {
67                  public double[] value(double[] params) {
68                      Vector2D center = new Vector2D(params[0], params[1]);
69                      double radius = getRadius(center);
70                      // gradient of the sum of squared residuals
71                      double dJdX = 0;
72                      double dJdY = 0;
73                      for (Vector2D pk : points) {
74                          double dk = pk.distance(center);
75                          dJdX += (center.getX() - pk.getX()) * (dk - radius) / dk;
76                          dJdY += (center.getY() - pk.getY()) * (dk - radius) / dk;
77                      }
78                      dJdX *= 2;
79                      dJdY *= 2;
80  
81                      return new double[] { dJdX, dJdY };
82                  }
83              });
84      }
85  }
