18  package org.apache.commons.math4;
19  
20  import java.util.Random;
21  
22  import org.apache.commons.math4.exception.MathIllegalStateException;
23  import org.junit.Test;
24  import org.junit.runner.RunWith;
25  
26  /**
27   * Test for the "Retry" functionality (retrying Junit test methods).
28   */
29  @RunWith(RetryRunner.class)
30  public class RetryRunnerTest {
31      final Random rng = new Random();
32  
33      /**
34       * Shows that an always failing test will fail even if it is retried.
35       */
36      @Test(expected=MathIllegalStateException.class)
37      @Retry
38      public void testRetryFailAlways() {
39          throw new MathIllegalStateException();
40      }
41  
42      /**
43       * Shows that a test that sometimes fail might succeed if it is retried.
44       * In this case the high number of retries makes it quite unlikely that
45       * the exception will be thrown by all of the calls.
46       */
47      @Test
48      @Retry(100)
49      public void testRetryFailSometimes() {
50          if (rng.nextBoolean()) {
51              throw new MathIllegalStateException();
52          }
53      }
54  }
