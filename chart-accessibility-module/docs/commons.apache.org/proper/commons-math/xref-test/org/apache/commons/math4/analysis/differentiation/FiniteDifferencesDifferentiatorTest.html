63                  new FiniteDifferencesDifferentiator(5, 0.01);

65                  differentiator.differentiate(new UnivariateFunction() {
66                      @Override
67                      public double value(double x) {

98      @Test
99      public void testGaussian() {
100         FiniteDifferencesDifferentiator differentiator =
101                 new FiniteDifferencesDifferentiator(9, 0.02);
102         UnivariateDifferentiableFunction gaussian = new Gaussian(1.0, 2.0);
103         UnivariateDifferentiableFunction f =
104                 differentiator.differentiate(gaussian);
105         double[] expectedError = new double[] {
106             6.939e-18, 1.284e-15, 2.477e-13, 1.168e-11, 2.840e-9, 7.971e-8
107         };
108        double[] maxError = new double[expectedError.length];
109         for (double x = -10; x < 10; x += 0.1) {
110             DerivativeStructure dsX  = new DerivativeStructure(1, maxError.length - 1, 0, x);
111             DerivativeStructure yRef = gaussian.value(dsX);
112             DerivativeStructure y    = f.value(dsX);
113             Assert.assertEquals(f.value(dsX.getValue()), f.value(dsX).getValue(), 1.0e-15);
114             for (int order = 0; order <= yRef.getOrder(); ++order) {
115                 maxError[order] = FastMath.max(maxError[order],
116                                         FastMath.abs(yRef.getPartialDerivative(order) -
117                                                      y.getPartialDerivative(order)));
118             }
119         }
120         for (int i = 0; i < maxError.length; ++i) {
121             Assert.assertEquals(expectedError[i], maxError[i], 0.01 * expectedError[i]);
122         }
123     }

125     @Test
126     public void testStepSizeUnstability() {
127         UnivariateDifferentiableFunction quintic = new QuinticFunction();
128         UnivariateDifferentiableFunction goodStep =
129                 new FiniteDifferencesDifferentiator(7, 0.25).differentiate(quintic);
130         UnivariateDifferentiableFunction badStep =
131                 new FiniteDifferencesDifferentiator(7, 1.0e-6).differentiate(quintic);
132         double[] maxErrorGood = new double[7];
133         double[] maxErrorBad  = new double[7];
134         for (double x = -10; x < 10; x += 0.1) {
135             DerivativeStructure dsX  = new DerivativeStructure(1, 6, 0, x);
136             DerivativeStructure yRef  = quintic.value(dsX);
137             DerivativeStructure yGood = goodStep.value(dsX);
138             DerivativeStructure yBad  = badStep.value(dsX);
139             for (int order = 0; order <= 6; ++order) {
140                 maxErrorGood[order] = FastMath.max(maxErrorGood[order],
141                                                    FastMath.abs(yRef.getPartialDerivative(order) -
142                                                                 yGood.getPartialDerivative(order)));
143                 maxErrorBad[order]  = FastMath.max(maxErrorBad[order],
144                                                    FastMath.abs(yRef.getPartialDerivative(order) -
145                                                                 yBad.getPartialDerivative(order)));
146             }
147         }
148 
149         // the 0.25 step size is good for finite differences in the quintic on this abscissa range for 7 points
150         // the errors are fair
151         final double[] expectedGood = new double[] {
152             7.276e-12, 7.276e-11, 9.968e-10, 3.092e-9, 5.432e-8, 8.196e-8, 1.818e-6
153         };
154 
155         // the 1.0e-6 step size is far too small for finite differences in the quintic on this abscissa range for 7 points
156         // the errors are huge!
157         final double[] expectedBad = new double[] {
158             2.910e-11, 2.087e-5, 147.7, 3.820e7, 6.354e14, 6.548e19, 1.543e27
159         };
160 
161         for (int i = 0; i < maxErrorGood.length; ++i) {
162             Assert.assertEquals(expectedGood[i], maxErrorGood[i], 0.01 * expectedGood[i]);
163             Assert.assertEquals(expectedBad[i],  maxErrorBad[i],  0.01 * expectedBad[i]);
164         }
165 
166     }

215     @Test
216     public void testBounds() {
217 
218         final double slope = 2.5;
219         UnivariateFunction f = new UnivariateFunction() {
220             @Override
221             public double value(double x) {
222                 if (x < 0) {
223                     throw new NumberIsTooSmallException(x, 0, true);
224                 } else if (x > 1) {
225                     throw new NumberIsTooLargeException(x, 1, true);
226                 } else {
227                     return slope * x;
228                 }
229             }
230         };
231 
232         UnivariateDifferentiableFunction missingBounds =
233                 new FiniteDifferencesDifferentiator(3, 0.1).differentiate(f);
234         UnivariateDifferentiableFunction properlyBounded =
235                 new FiniteDifferencesDifferentiator(3, 0.1, 0.0, 1.0).differentiate(f);
236         DerivativeStructure tLow  = new DerivativeStructure(1, 1, 0, 0.05);
237         DerivativeStructure tHigh = new DerivativeStructure(1, 1, 0, 0.95);
238 
239         try {
240             // here, we did not set the bounds, so the differences are evaluated out of domain
241             // using f(-0.05), f(0.05), f(0.15)
242             missingBounds.value(tLow);
243             Assert.fail("an exception should have been thrown");
244         } catch (NumberIsTooSmallException nse) {
245             Assert.assertEquals(-0.05, nse.getArgument().doubleValue(), 1.0e-10);
246         } catch (Exception e) {
247             Assert.fail("wrong exception caught: " + e.getClass().getName());
248         }
249 
250         try {
251             // here, we did not set the bounds, so the differences are evaluated out of domain
252             // using f(0.85), f(0.95), f(1.05)
253             missingBounds.value(tHigh);
254             Assert.fail("an exception should have been thrown");
255         } catch (NumberIsTooLargeException nle) {
256             Assert.assertEquals(1.05, nle.getArgument().doubleValue(), 1.0e-10);
257         } catch (Exception e) {
258             Assert.fail("wrong exception caught: " + e.getClass().getName());
259         }
260 
261         // here, we did set the bounds, so evaluations are done within domain
262         // using f(0.0), f(0.1), f(0.2)
263         Assert.assertEquals(slope, properlyBounded.value(tLow).getPartialDerivative(1), 1.0e-10);
264 
265         // here, we did set the bounds, so evaluations are done within domain
266         // using f(0.8), f(0.9), f(1.0)
267         Assert.assertEquals(slope, properlyBounded.value(tHigh).getPartialDerivative(1), 1.0e-10);
268 
269     }

271     @Test
272     public void testBoundedSqrt() {
273 
274         UnivariateFunctionDifferentiator differentiator =
275                 new FiniteDifferencesDifferentiator(9, 1.0 / 32, 0.0, Double.POSITIVE_INFINITY);
276         UnivariateDifferentiableFunction sqrt = differentiator.differentiate(new UnivariateFunction() {
277             @Override
278             public double value(double x) {
279                 return FastMath.sqrt(x);
280             }
281         });
282 
283         // we are able to compute derivative near 0, but the accuracy is much poorer there
284         DerivativeStructure t001 = new DerivativeStructure(1, 1, 0, 0.01);
285         Assert.assertEquals(0.5 / FastMath.sqrt(t001.getValue()), sqrt.value(t001).getPartialDerivative(1), 1.6);
286         DerivativeStructure t01 = new DerivativeStructure(1, 1, 0, 0.1);
287         Assert.assertEquals(0.5 / FastMath.sqrt(t01.getValue()), sqrt.value(t01).getPartialDerivative(1), 7.0e-3);
288         DerivativeStructure t03 = new DerivativeStructure(1, 1, 0, 0.3);
289         Assert.assertEquals(0.5 / FastMath.sqrt(t03.getValue()), sqrt.value(t03).getPartialDerivative(1), 2.1e-7);
290 
291     }

293     @Test
294     public void testVectorFunction() {
295 
296         FiniteDifferencesDifferentiator differentiator =
297                 new FiniteDifferencesDifferentiator(7, 0.01);
298         UnivariateDifferentiableVectorFunction f =
299                 differentiator.differentiate(new UnivariateVectorFunction() {
300 
301             @Override
302             public double[] value(double x) {
303                 return new double[] { FastMath.cos(x), FastMath.sin(x) };
304             }
305 
306         });
307 
308         for (double x = -10; x < 10; x += 0.1) {
309             DerivativeStructure dsX = new DerivativeStructure(1, 2, 0, x);
310             DerivativeStructure[] y = f.value(dsX);
311             double cos = FastMath.cos(x);
312             double sin = FastMath.sin(x);
313             double[] f1 = f.value(dsX.getValue());
314             DerivativeStructure[] f2 = f.value(dsX);
315             Assert.assertEquals(f1.length, f2.length);
316             for (int i = 0; i < f1.length; ++i) {
317                 Assert.assertEquals(f1[i], f2[i].getValue(), 1.0e-15);
318             }
319             Assert.assertEquals( cos, y[0].getValue(), 7.0e-16);
320             Assert.assertEquals( sin, y[1].getValue(), 7.0e-16);
321             Assert.assertEquals(-sin, y[0].getPartialDerivative(1), 6.0e-14);
322             Assert.assertEquals( cos, y[1].getPartialDerivative(1), 6.0e-14);
323             Assert.assertEquals(-cos, y[0].getPartialDerivative(2), 2.0e-11);
324             Assert.assertEquals(-sin, y[1].getPartialDerivative(2), 2.0e-11);
325         }

329     @Test
330     public void testMatrixFunction() {
331 
332         FiniteDifferencesDifferentiator differentiator =
333                 new FiniteDifferencesDifferentiator(7, 0.01);
334         UnivariateDifferentiableMatrixFunction f =
335                 differentiator.differentiate(new UnivariateMatrixFunction() {
336 
337             @Override
338             public double[][] value(double x) {
339                 return new double[][] {
340                     { FastMath.cos(x),  FastMath.sin(x)  },
341                     { FastMath.cosh(x), FastMath.sinh(x) }
342                 };
343             }
344 
345         });
346 
347         for (double x = -1; x < 1; x += 0.02) {
348             DerivativeStructure dsX = new DerivativeStructure(1, 2, 0, x);
349             DerivativeStructure[][] y = f.value(dsX);
350             double cos = FastMath.cos(x);
351             double sin = FastMath.sin(x);
352             double cosh = FastMath.cosh(x);
353             double sinh = FastMath.sinh(x);
354             double[][] f1 = f.value(dsX.getValue());
355             DerivativeStructure[][] f2 = f.value(dsX);
356             Assert.assertEquals(f1.length, f2.length);
357             for (int i = 0; i < f1.length; ++i) {
358                 Assert.assertEquals(f1[i].length, f2[i].length);
359                 for (int j = 0; j < f1[i].length; ++j) {
360                     Assert.assertEquals(f1[i][j], f2[i][j].getValue(), 1.0e-15);
361                 }
362             }
363             Assert.assertEquals(cos,   y[0][0].getValue(), 7.0e-18);
364             Assert.assertEquals(sin,   y[0][1].getValue(), 6.0e-17);
365             Assert.assertEquals(cosh,  y[1][0].getValue(), 3.0e-16);
366             Assert.assertEquals(sinh,  y[1][1].getValue(), 3.0e-16);
367             Assert.assertEquals(-sin,  y[0][0].getPartialDerivative(1), 2.0e-14);
368             Assert.assertEquals( cos,  y[0][1].getPartialDerivative(1), 2.0e-14);
369             Assert.assertEquals( sinh, y[1][0].getPartialDerivative(1), 3.0e-14);
370             Assert.assertEquals( cosh, y[1][1].getPartialDerivative(1), 3.0e-14);
371             Assert.assertEquals(-cos,  y[0][0].getPartialDerivative(2), 3.0e-12);
372             Assert.assertEquals(-sin,  y[0][1].getPartialDerivative(2), 3.0e-12);
373             Assert.assertEquals( cosh, y[1][0].getPartialDerivative(2), 6.0e-12);
374             Assert.assertEquals( sinh, y[1][1].getPartialDerivative(2), 6.0e-12);
375         }

379     @Test
380     public void testSeveralFreeParameters() {
381         FiniteDifferencesDifferentiator differentiator =
382                 new FiniteDifferencesDifferentiator(5, 0.001);
383         UnivariateDifferentiableFunction sine = new Sin();
384         UnivariateDifferentiableFunction f =
385                 differentiator.differentiate(sine);
386         double[] expectedError = new double[] {
387             6.696e-16, 1.371e-12, 2.007e-8, 1.754e-5
388         };
389         double[] maxError = new double[expectedError.length];
390        for (double x = -2; x < 2; x += 0.1) {
391            for (double y = -2; y < 2; y += 0.1) {
392                DerivativeStructure dsX  = new DerivativeStructure(2, maxError.length - 1, 0, x);
393                DerivativeStructure dsY  = new DerivativeStructure(2, maxError.length - 1, 1, y);
394                DerivativeStructure dsT  = dsX.multiply(3).subtract(dsY.multiply(2));
395                DerivativeStructure sRef = sine.value(dsT);
396                DerivativeStructure s    = f.value(dsT);
397                for (int xOrder = 0; xOrder <= sRef.getOrder(); ++xOrder) {
398                    for (int yOrder = 0; yOrder <= sRef.getOrder(); ++yOrder) {
399                        if (xOrder + yOrder <= sRef.getOrder()) {
400                            maxError[xOrder +yOrder] = FastMath.max(maxError[xOrder + yOrder],
401                                                                     FastMath.abs(sRef.getPartialDerivative(xOrder, yOrder) -
402                                                                                  s.getPartialDerivative(xOrder, yOrder)));
403                        }
404                    }
405                }
406            }
407        }
408        for (int i = 0; i < maxError.length; ++i) {
409            Assert.assertEquals(expectedError[i], maxError[i], 0.01 * expectedError[i]);
410        }
