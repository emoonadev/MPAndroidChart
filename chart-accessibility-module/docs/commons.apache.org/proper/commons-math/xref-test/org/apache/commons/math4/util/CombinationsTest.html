147     /**
148      * Verifies that the iterator generates a lexicographically
149      * increasing sequence of b(n,k) arrays, each having length k
150      * and each array itself increasing.
151      *
152      * @param c Combinations.
153      */
154     private void checkLexicographicIterator(Combinations c) {
155         final Comparator<int[]> comp = c.comparator();
156         final int n = c.getN();
157         final int k = c.getK();
158 
159         int[] lastIterate = null;
160 
161         long numIterates = 0;
162         for (int[] iterate : c) {
163             Assert.assertEquals(k, iterate.length);
164 
165             // Check that the sequence of iterates is ordered.
166             if (lastIterate != null) {
167                 Assert.assertTrue(comp.compare(iterate, lastIterate) == 1);
168             }
169 
170             // Check that each iterate is ordered.
171             for (int i = 1; i < iterate.length; i++) {
172                 Assert.assertTrue(iterate[i] > iterate[i - 1]);
173             }
174 
175             lastIterate = iterate;
176             ++numIterates;
177         }
178 
179         // Check the number of iterates.
180         Assert.assertEquals(CombinatoricsUtils.binomialCoefficient(n, k),
181                             numIterates);
182     }
