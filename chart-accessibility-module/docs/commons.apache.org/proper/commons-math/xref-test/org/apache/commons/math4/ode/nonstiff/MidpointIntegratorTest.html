56    @Test
57    public void testDecreasingSteps()
58        throws DimensionMismatchException, NumberIsTooSmallException,
59               MaxCountExceededException, NoBracketingException {
60  
61        for (TestProblemAbstract pb : new TestProblemAbstract[] {
62            new TestProblem1(), new TestProblem2(), new TestProblem3(),
63            new TestProblem4(), new TestProblem5(), new TestProblem6()
64        }) {
65  
66        double previousValueError = Double.NaN;
67        double previousTimeError = Double.NaN;
68        for (int i = 4; i < 10; ++i) {
69  
70          double step = (pb.getFinalTime() - pb.getInitialTime()) * FastMath.pow(2.0, -i);
71          FirstOrderIntegrator integ = new MidpointIntegrator(step);
72          TestProblemHandler handler = new TestProblemHandler(pb, integ);
73          integ.addStepHandler(handler);
74          EventHandler[] functions = pb.getEventsHandlers();
75          for (int l = 0; l < functions.length; ++l) {
76            integ.addEventHandler(functions[l],
77                                       Double.POSITIVE_INFINITY, 1.0e-6 * step, 1000);
78          }
79          double stopTime = integ.integrate(pb,
80                                            pb.getInitialTime(), pb.getInitialState(),
81                                            pb.getFinalTime(), new double[pb.getDimension()]);
82          if (functions.length == 0) {
83              Assert.assertEquals(pb.getFinalTime(), stopTime, 1.0e-10);
84          }
85  
86          double valueError = handler.getMaximalValueError();
87          if (i > 4) {
88            Assert.assertTrue(valueError < FastMath.abs(previousValueError));
89          }
90          previousValueError = valueError;
91  
92          double timeError = handler.getMaximalTimeError();
93          if (i > 4) {
94            Assert.assertTrue(timeError <= FastMath.abs(previousTimeError));
95          }
96          previousTimeError = timeError;
97  
98        }
99  
100     }

104   @Test
105   public void testSmallStep()
106       throws DimensionMismatchException, NumberIsTooSmallException,
107              MaxCountExceededException, NoBracketingException {
108 
109     TestProblem1 pb  = new TestProblem1();
110     double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
111 
112     FirstOrderIntegrator integ = new MidpointIntegrator(step);
113     TestProblemHandler handler = new TestProblemHandler(pb, integ);
114     integ.addStepHandler(handler);
115     integ.integrate(pb,
116                     pb.getInitialTime(), pb.getInitialState(),
117                     pb.getFinalTime(), new double[pb.getDimension()]);
118 
119     Assert.assertTrue(handler.getLastError() < 2.0e-7);
120     Assert.assertTrue(handler.getMaximalValueError() < 1.0e-6);
121     Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
122     Assert.assertEquals("midpoint", integ.getName());
123 
124   }

167   @Test
168   public void testStepSize()
169       throws DimensionMismatchException, NumberIsTooSmallException,
170              MaxCountExceededException, NoBracketingException {
171       final double step = 1.23456;
172       FirstOrderIntegrator integ = new MidpointIntegrator(step);
173       integ.addStepHandler(new StepHandler() {
174           public void handleStep(StepInterpolator interpolator, boolean isLast) {
175               if (! isLast) {
176                   Assert.assertEquals(step,
177                                interpolator.getCurrentTime() - interpolator.getPreviousTime(),
178                                1.0e-12);
179               }
180           }
181           public void init(double t0, double[] y0, double t) {
182           }
183       });
184       integ.integrate(new FirstOrderDifferentialEquations() {
185           public void computeDerivatives(double t, double[] y, double[] dot) {
186               dot[0] = 1.0;
187           }
188           public int getDimension() {
189               return 1;
190           }
191       }, 0.0, new double[] { 0.0 }, 5.0, new double[1]);
192   }
