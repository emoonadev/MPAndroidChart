17  package org.apache.commons.math4.optim.univariate;
18  
19  import org.apache.commons.math4.analysis.UnivariateFunction;
20  import org.apache.commons.math4.optim.nonlinear.scalar.GoalType;
21  import org.apache.commons.math4.optim.univariate.BracketFinder;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test for {@link BracketFinder}.
27   */
28  public class BracketFinderTest {
29  
30      @Test
31      public void testCubicMin() {
32          final BracketFinder bFind = new BracketFinder();
33          final UnivariateFunction func = new UnivariateFunction() {
34                  public double value(double x) {
35                      if (x < -2) {
36                          return value(-2);
37                      }
38                      else  {
39                          return (x - 1) * (x + 2) * (x + 3);
40                      }
41                  }
42              };
43  
44          bFind.search(func, GoalType.MINIMIZE, -2 , -1);
45          final double tol = 1e-15;
46          // Comparing with results computed in Python.
47          Assert.assertEquals(-2, bFind.getLo(), tol);
48          Assert.assertEquals(-1, bFind.getMid(), tol);
49          Assert.assertEquals(0.61803399999999997, bFind.getHi(), tol);
50      }
51  
52      @Test
53      public void testCubicMax() {
54          final BracketFinder bFind = new BracketFinder();
55          final UnivariateFunction func = new UnivariateFunction() {
56                  public double value(double x) {
57                      if (x < -2) {
58                          return value(-2);
59                      }
60                      else  {
61                          return -(x - 1) * (x + 2) * (x + 3);
62                      }
63                  }
64              };
65  
66          bFind.search(func, GoalType.MAXIMIZE, -2 , -1);
67          final double tol = 1e-15;
68          Assert.assertEquals(-2, bFind.getLo(), tol);
69          Assert.assertEquals(-1, bFind.getMid(), tol);
70          Assert.assertEquals(0.61803399999999997, bFind.getHi(), tol);
71      }
72  
73      @Test
74      public void testMinimumIsOnIntervalBoundary() {
75          final UnivariateFunction func = new UnivariateFunction() {
76                  public double value(double x) {
77                      return x * x;
78                  }
79              };
80  
81          final BracketFinder bFind = new BracketFinder();
82  
83          bFind.search(func, GoalType.MINIMIZE, 0, 1);
84          Assert.assertTrue(bFind.getLo() <= 0);
85          Assert.assertTrue(0 <= bFind.getHi());
86  
87          bFind.search(func, GoalType.MINIMIZE, -1, 0);
88          Assert.assertTrue(bFind.getLo() <= 0);
89          Assert.assertTrue(0 <= bFind.getHi());
90      }
91  
92      @Test
93      public void testIntervalBoundsOrdering() {
94          final UnivariateFunction func = new UnivariateFunction() {
95                  public double value(double x) {
96                      return x * x;
97                  }
98              };
99  
100         final BracketFinder bFind = new BracketFinder();
101 
102         bFind.search(func, GoalType.MINIMIZE, -1, 1);
103         Assert.assertTrue(bFind.getLo() <= 0);
104         Assert.assertTrue(0 <= bFind.getHi());
105 
106         bFind.search(func, GoalType.MINIMIZE, 1, -1);
107         Assert.assertTrue(bFind.getLo() <= 0);
108         Assert.assertTrue(0 <= bFind.getHi());
109 
110         bFind.search(func, GoalType.MINIMIZE, 1, 2);
111         Assert.assertTrue(bFind.getLo() <= 0);
112         Assert.assertTrue(0 <= bFind.getHi());
113 
114         bFind.search(func, GoalType.MINIMIZE, 2, 1);
115         Assert.assertTrue(bFind.getLo() <= 0);
116         Assert.assertTrue(0 <= bFind.getHi());
117     }
