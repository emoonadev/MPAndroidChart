18  package org.apache.commons.math4.linear;
19  
20  import org.apache.commons.math4.exception.MathIllegalArgumentException;
21  import org.apache.commons.math4.linear.CholeskyDecomposition;
22  import org.apache.commons.math4.linear.DecompositionSolver;
23  import org.apache.commons.math4.linear.MatrixUtils;
24  import org.apache.commons.math4.linear.RealMatrix;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  
29  public class CholeskySolverTest {
30  
31      private double[][] testData = new double[][] {
32              {  1,  2,   4,   7,  11 },
33              {  2, 13,  23,  38,  58 },
34              {  4, 23,  77, 122, 182 },
35              {  7, 38, 122, 294, 430 },
36              { 11, 58, 182, 430, 855 }
37      };
38  
39      /** test solve dimension errors */
40      @Test
41      public void testSolveDimensionErrors() {
42          DecompositionSolver solver =
43              new CholeskyDecomposition(MatrixUtils.createRealMatrix(testData)).getSolver();
44          RealMatrix b = MatrixUtils.createRealMatrix(new double[2][2]);
45          try {
46              solver.solve(b);
47              Assert.fail("an exception should have been thrown");
48          } catch (MathIllegalArgumentException iae) {
49              // expected behavior
50          }
51          try {
52              solver.solve(b.getColumnVector(0));
53              Assert.fail("an exception should have been thrown");
54          } catch (MathIllegalArgumentException iae) {
55              // expected behavior
56          }
57          try {
58              solver.solve(new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(0)));
59              Assert.fail("an exception should have been thrown");
60          } catch (MathIllegalArgumentException iae) {
61              // expected behavior
62          }
63      }
64  
65      /** test solve */
66      @Test
67      public void testSolve() {
68          DecompositionSolver solver =
69              new CholeskyDecomposition(MatrixUtils.createRealMatrix(testData)).getSolver();
70          RealMatrix b = MatrixUtils.createRealMatrix(new double[][] {
71                  {   78,  -13,    1 },
72                  {  414,  -62,   -1 },
73                  { 1312, -202,  -37 },
74                  { 2989, -542,  145 },
75                  { 5510, -1465, 201 }
76          });
77          RealMatrix xRef = MatrixUtils.createRealMatrix(new double[][] {
78                  { 1,  0,  1 },
79                  { 0,  1,  1 },
80                  { 2,  1, -4 },
81                  { 2,  2,  2 },
82                  { 5, -3,  0 }
83          });
84  
85          // using RealMatrix
86          Assert.assertEquals(0, solver.solve(b).subtract(xRef).getNorm(), 1.0e-13);
87  
88          // using ArrayRealVector
89          for (int i = 0; i < b.getColumnDimension(); ++i) {
90              Assert.assertEquals(0,
91                           solver.solve(b.getColumnVector(i)).subtract(xRef.getColumnVector(i)).getNorm(),
92                           1.0e-13);
93          }
94  
95          // using RealVector with an alternate implementation
96          for (int i = 0; i < b.getColumnDimension(); ++i) {
97              ArrayRealVectorTest.RealVectorTestImpl v =
98                  new ArrayRealVectorTest.RealVectorTestImpl(b.getColumn(i));
99              Assert.assertEquals(0,
100                          solver.solve(v).subtract(xRef.getColumnVector(i)).getNorm(),
101                          1.0e-13);
102         }
103 
104     }
105 
106     /** test determinant */
107     @Test
108     public void testDeterminant() {
109         Assert.assertEquals(7290000.0, getDeterminant(MatrixUtils.createRealMatrix(testData)), 1.0e-15);
110     }
111 
112     private double getDeterminant(RealMatrix m) {
113         return new CholeskyDecomposition(m).getDeterminant();
114     }

116 }
