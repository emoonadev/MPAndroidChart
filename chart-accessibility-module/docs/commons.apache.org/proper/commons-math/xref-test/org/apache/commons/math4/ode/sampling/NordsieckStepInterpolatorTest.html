40  public class NordsieckStepInterpolatorTest {
41  
42      @Test
43      public void derivativesConsistency()
44          throws NumberIsTooSmallException, DimensionMismatchException,
45                 MaxCountExceededException, NoBracketingException {
46          TestProblem3 pb = new TestProblem3();
47          AdamsBashforthIntegrator integ = new AdamsBashforthIntegrator(4, 0.0, 1.0, 1.0e-10, 1.0e-10);
48          StepInterpolatorTestUtils.checkDerivativesConsistency(integ, pb, 0.05, 2.8e-9);
49      }
50  
51      @Test
52      public void serialization()
53      throws IOException, ClassNotFoundException,
54             NumberIsTooSmallException, DimensionMismatchException,
55             MaxCountExceededException, NoBracketingException {
56  
57          TestProblem1 pb = new TestProblem1();
58          AdamsBashforthIntegrator integ = new AdamsBashforthIntegrator(4, 0.0, 1.0, 1.0e-10, 1.0e-10);
59          integ.addStepHandler(new ContinuousOutputModel());
60          integ.integrate(pb,
61                          pb.getInitialTime(), pb.getInitialState(),
62                          pb.getFinalTime(), new double[pb.getDimension()]);
63  
64          ByteArrayOutputStream bos = new ByteArrayOutputStream();
65          ObjectOutputStream    oos = new ObjectOutputStream(bos);
66          for (StepHandler handler : integ.getStepHandlers()) {
67              oos.writeObject(handler);
68          }
69  
70          Assert.assertTrue(bos.size() > 47000);
71          Assert.assertTrue(bos.size() < 48000);
72  
73          ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
74          ObjectInputStream     ois = new ObjectInputStream(bis);
75          ContinuousOutputModel cm  = (ContinuousOutputModel) ois.readObject();
76  
77          Random random = new Random(347588535632l);
78          double maxError = 0.0;
79          for (int i = 0; i < 1000; ++i) {
80              double r = random.nextDouble();
81              double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
82              cm.setInterpolatedTime(time);
83              double[] interpolatedY = cm.getInterpolatedState();
84              double[] theoreticalY  = pb.computeTheoreticalState(time);
85              double dx = interpolatedY[0] - theoreticalY[0];
86              double dy = interpolatedY[1] - theoreticalY[1];
87              double error = dx * dx + dy * dy;
88              if (error > maxError) {
89                  maxError = error;
90              }
91          }
92  
93          Assert.assertTrue(maxError < 1.0e-6);
94  
95      }
