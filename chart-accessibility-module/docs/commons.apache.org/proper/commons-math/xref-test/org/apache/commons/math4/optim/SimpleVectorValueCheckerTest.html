17  package org.apache.commons.math4.optim;
18  
19  import org.apache.commons.math4.exception.NotStrictlyPositiveException;
20  import org.apache.commons.math4.optim.PointVectorValuePair;
21  import org.apache.commons.math4.optim.SimpleVectorValueChecker;
22  import org.junit.Test;
23  import org.junit.Assert;
24  
25  public class SimpleVectorValueCheckerTest {
26      @Test(expected=NotStrictlyPositiveException.class)
27      public void testIterationCheckPrecondition() {
28          new SimpleVectorValueChecker(1e-1, 1e-2, 0);
29      }
30  
31      @Test
32      public void testIterationCheck() {
33          final int max = 10;
34          final SimpleVectorValueChecker checker = new SimpleVectorValueChecker(1e-1, 1e-2, max);
35          Assert.assertTrue(checker.converged(max, null, null));
36          Assert.assertTrue(checker.converged(max + 1, null, null));
37      }
38  
39      @Test
40      public void testIterationCheckDisabled() {
41          final SimpleVectorValueChecker checker = new SimpleVectorValueChecker(1e-8, 1e-8);
42  
43          final PointVectorValuePair a = new PointVectorValuePair(new double[] { 1d },
44                                                                  new double[] { 1d });
45          final PointVectorValuePair b = new PointVectorValuePair(new double[] { 10d },
46                                                                  new double[] { 10d });
47  
48          Assert.assertFalse(checker.converged(-1, a, b));
49          Assert.assertFalse(checker.converged(0, a, b));
50          Assert.assertFalse(checker.converged(1000000, a, b));
51  
52          Assert.assertTrue(checker.converged(-1, a, a));
53          Assert.assertTrue(checker.converged(-1, b, b));
54      }
55  }
