17  package org.apache.commons.math4.geometry.euclidean.oned;
18  
19  import org.apache.commons.math4.geometry.euclidean.oned.Interval;
20  import org.apache.commons.math4.geometry.partitioning.Region;
21  import org.apache.commons.math4.util.FastMath;
22  import org.apache.commons.math4.util.Precision;
23  import org.apache.commons.math4.exception.NumberIsTooSmallException;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  public class IntervalTest {
28  
29      @Test
30      public void testInterval() {
31          Interval interval = new Interval(2.3, 5.7);
32          Assert.assertEquals(3.4, interval.getSize(), 1.0e-10);
33          Assert.assertEquals(4.0, interval.getBarycenter(), 1.0e-10);
34          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(2.3, 1.0e-10));
35          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(5.7, 1.0e-10));
36          Assert.assertEquals(Region.Location.OUTSIDE,  interval.checkPoint(1.2, 1.0e-10));
37          Assert.assertEquals(Region.Location.OUTSIDE,  interval.checkPoint(8.7, 1.0e-10));
38          Assert.assertEquals(Region.Location.INSIDE,   interval.checkPoint(3.0, 1.0e-10));
39          Assert.assertEquals(2.3, interval.getInf(), 1.0e-10);
40          Assert.assertEquals(5.7, interval.getSup(), 1.0e-10);
41      }
42  
43      @Test
44      public void testTolerance() {
45          Interval interval = new Interval(2.3, 5.7);
46          Assert.assertEquals(Region.Location.OUTSIDE,  interval.checkPoint(1.2, 1.0));
47          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(1.2, 1.2));
48          Assert.assertEquals(Region.Location.OUTSIDE,  interval.checkPoint(8.7, 2.9));
49          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(8.7, 3.1));
50          Assert.assertEquals(Region.Location.INSIDE,   interval.checkPoint(3.0, 0.6));
51          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(3.0, 0.8));
52      }
53  
54      @Test
55      public void testInfinite() {
56          Interval interval = new Interval(9.0, Double.POSITIVE_INFINITY);
57          Assert.assertEquals(Region.Location.BOUNDARY, interval.checkPoint(9.0, 1.0e-10));
58          Assert.assertEquals(Region.Location.OUTSIDE,  interval.checkPoint(8.4, 1.0e-10));
59          for (double e = 1.0; e <= 6.0; e += 1.0) {
60              Assert.assertEquals(Region.Location.INSIDE,
61                                  interval.checkPoint(FastMath.pow(10.0, e), 1.0e-10));
62          }
63          Assert.assertTrue(Double.isInfinite(interval.getSize()));
64          Assert.assertEquals(9.0, interval.getInf(), 1.0e-10);
65          Assert.assertTrue(Double.isInfinite(interval.getSup()));
66  
67      }
68  
69      @Test
70      public void testSinglePoint() {
71          Interval interval = new Interval(1.0, 1.0);
72          Assert.assertEquals(0.0, interval.getSize(), Precision.SAFE_MIN);
73          Assert.assertEquals(1.0, interval.getBarycenter(), Precision.EPSILON);
74      }
75  
76      // MATH-1256
77      @Test(expected=NumberIsTooSmallException.class)
78      public void testStrictOrdering() {
79          new Interval(0, -1);
80      }
81  }
