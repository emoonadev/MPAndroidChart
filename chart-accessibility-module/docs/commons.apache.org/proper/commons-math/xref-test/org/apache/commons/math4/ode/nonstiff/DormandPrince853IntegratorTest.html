43    @Test
44    public void testMissedEndEvent()
45        throws DimensionMismatchException, NumberIsTooSmallException,
46               MaxCountExceededException, NoBracketingException {
47        final double   t0     = 1878250320.0000029;
48        final double   tEvent = 1878250379.9999986;
49        final double[] k  = { 1.0e-4, 1.0e-5, 1.0e-6 };
50        FirstOrderDifferentialEquations ode = new FirstOrderDifferentialEquations() {
51  
52            public int getDimension() {
53                return k.length;
54            }
55  
56            public void computeDerivatives(double t, double[] y, double[] yDot) {
57                for (int i = 0; i < y.length; ++i) {
58                    yDot[i] = k[i] * y[i];
59                }
60            }
61        };
62  
63        DormandPrince853Integrator integrator = new DormandPrince853Integrator(0.0, 100.0,
64                                                                               1.0e-10, 1.0e-10);
65  
66        double[] y0   = new double[k.length];
67        for (int i = 0; i < y0.length; ++i) {
68            y0[i] = i + 1;
69        }
70        double[] y    = new double[k.length];
71  
72        integrator.setInitialStepSize(60.0);
73        double finalT = integrator.integrate(ode, t0, y0, tEvent, y);
74        Assert.assertEquals(tEvent, finalT, 5.0e-6);
75        for (int i = 0; i < y.length; ++i) {
76            Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-9);
77        }
78  
79        integrator.setInitialStepSize(60.0);
80        integrator.addEventHandler(new EventHandler() {
81  
82            public void init(double t0, double[] y0, double t) {
83            }
84  
85            public void resetState(double t, double[] y) {
86            }
87  
88            public double g(double t, double[] y) {
89                return t - tEvent;
90            }
91  
92            public Action eventOccurred(double t, double[] y, boolean increasing) {
93                Assert.assertEquals(tEvent, t, 5.0e-6);
94                return Action.CONTINUE;
95            }
96        }, Double.POSITIVE_INFINITY, 1.0e-20, 100);
97        finalT = integrator.integrate(ode, t0, y0, tEvent + 120, y);
98        Assert.assertEquals(tEvent + 120, finalT, 5.0e-6);
99        for (int i = 0; i < y.length; ++i) {
100           Assert.assertEquals(y0[i] * FastMath.exp(k[i] * (finalT - t0)), y[i], 1.0e-9);
101       }
102 
103   }

154   @Test
155   public void testIncreasingTolerance()
156       throws DimensionMismatchException, NumberIsTooSmallException,
157              MaxCountExceededException, NoBracketingException {
158 
159     int previousCalls = Integer.MAX_VALUE;
160     AdaptiveStepsizeIntegrator integ =
161         new DormandPrince853Integrator(0, Double.POSITIVE_INFINITY,
162                                        Double.NaN, Double.NaN);
163     for (int i = -12; i < -2; ++i) {
164       TestProblem1 pb = new TestProblem1();
165       double minStep = 0;
166       double maxStep = pb.getFinalTime() - pb.getInitialTime();
167       double scalAbsoluteTolerance = FastMath.pow(10.0, i);
168       double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
169       integ.setStepSizeControl(minStep, maxStep, scalAbsoluteTolerance, scalRelativeTolerance);
170 
171       TestProblemHandler handler = new TestProblemHandler(pb, integ);
172       integ.addStepHandler(handler);
173       integ.integrate(pb,
174                       pb.getInitialTime(), pb.getInitialState(),
175                       pb.getFinalTime(), new double[pb.getDimension()]);
176 
177       // the 1.3 factor is only valid for this test
178       // and has been obtained from trial and error
179       // there is no general relation between local and global errors
180       Assert.assertTrue(handler.getMaximalValueError() < (1.3 * scalAbsoluteTolerance));
181       Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
182 
183       int calls = pb.getCalls();
184       Assert.assertEquals(integ.getEvaluations(), calls);
185       Assert.assertTrue(calls <= previousCalls);
186       previousCalls = calls;
187 
188     }

245   @Test
246   public void testEvents()
247       throws DimensionMismatchException, NumberIsTooSmallException,
248              MaxCountExceededException, NoBracketingException {
249 
250     TestProblem4 pb = new TestProblem4();
251     double minStep = 0;
252     double maxStep = pb.getFinalTime() - pb.getInitialTime();
253     double scalAbsoluteTolerance = 1.0e-9;
254     double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
255 
256     FirstOrderIntegrator integ = new DormandPrince853Integrator(minStep, maxStep,
257                                                                 scalAbsoluteTolerance,
258                                                                 scalRelativeTolerance);
259     TestProblemHandler handler = new TestProblemHandler(pb, integ);
260     integ.addStepHandler(handler);
261     EventHandler[] functions = pb.getEventsHandlers();
262     double convergence = 1.0e-8 * maxStep;
263     for (int l = 0; l < functions.length; ++l) {
264       integ.addEventHandler(functions[l], Double.POSITIVE_INFINITY, convergence, 1000);
265     }
266     Assert.assertEquals(functions.length, integ.getEventHandlers().size());
267     integ.integrate(pb,
268                     pb.getInitialTime(), pb.getInitialState(),
269                     pb.getFinalTime(), new double[pb.getDimension()]);
270 
271     Assert.assertEquals(0, handler.getMaximalValueError(), 2.1e-7);
272     Assert.assertEquals(0, handler.getMaximalTimeError(), convergence);
273     Assert.assertEquals(12.0, handler.getLastTime(), convergence);
274     integ.clearEventHandlers();
275     Assert.assertEquals(0, integ.getEventHandlers().size());
276 
277   }

338   @Test
339   public void testEventsScheduling() {
340 
341       FirstOrderDifferentialEquations sincos = new FirstOrderDifferentialEquations() {
342 
343           public int getDimension() {
344               return 2;
345           }
346 
347           public void computeDerivatives(double t, double[] y, double[] yDot) {
348               yDot[0] =  y[1];
349               yDot[1] = -y[0];
350           }
351 
352       };
353 
354       SchedulingChecker sinChecker = new SchedulingChecker(0); // events at 0, PI, 2PI ...
355       SchedulingChecker cosChecker = new SchedulingChecker(1); // events at PI/2, 3PI/2, 5PI/2 ...
356 
357       FirstOrderIntegrator integ =
358               new DormandPrince853Integrator(0.001, 1.0, 1.0e-12, 0.0);
359       integ.addEventHandler(sinChecker, 0.01, 1.0e-7, 100);
360       integ.addStepHandler(sinChecker);
361       integ.addEventHandler(cosChecker, 0.01, 1.0e-7, 100);
362       integ.addStepHandler(cosChecker);
363       double   t0 = 0.5;
364       double[] y0 = new double[] { FastMath.sin(t0), FastMath.cos(t0) };
365       double   t  = 10.0;
366       double[] y  = new double[2];
367       integ.integrate(sincos, t0, y0, t, y);
368 
369   }

371   private static class SchedulingChecker implements StepHandler, EventHandler {
372 
373       int index;
374       double tMin;
375 
376       public SchedulingChecker(int index) {
377           this.index = index;
378       }
379 
380       public void init(double t0, double[] y0, double t) {
381           tMin = t0;
382       }
383 
384       public void handleStep(StepInterpolator interpolator, boolean isLast) {
385           tMin = interpolator.getCurrentTime();
386       }
387 
388       public double g(double t, double[]  y) {
389           // once a step has been handled by handleStep,
390           // events checking should only refer to dates after the step
391           Assert.assertTrue(t >= tMin);
392           return y[index];
393       }
394 
395       public Action eventOccurred(double t, double[] y, boolean increasing) {
396           return Action.RESET_STATE;
397       }
398 
399       public void resetState(double t, double[] y) {
400           // in fact, we don't need to reset anything for the test
401       }

405   private static class KeplerHandler implements StepHandler {
406     public KeplerHandler(TestProblem3 pb) {
407       this.pb = pb;
408     }
409     public void init(double t0, double[] y0, double t) {
410       nbSteps = 0;
411       maxError = 0;
412     }
413     public void handleStep(StepInterpolator interpolator, boolean isLast)
414         throws MaxCountExceededException {
415 
416       ++nbSteps;
417       for (int a = 1; a < 10; ++a) {
418 
419         double prev   = interpolator.getPreviousTime();
420         double curr   = interpolator.getCurrentTime();
421         double interp = ((10 - a) * prev + a * curr) / 10;
422         interpolator.setInterpolatedTime(interp);
423 
424         double[] interpolatedY = interpolator.getInterpolatedState ();
425         double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getInterpolatedTime());
426         double dx = interpolatedY[0] - theoreticalY[0];
427         double dy = interpolatedY[1] - theoreticalY[1];
428         double error = dx * dx + dy * dy;
429         if (error > maxError) {
430           maxError = error;
431         }
432       }
433       if (isLast) {
434         Assert.assertTrue(maxError < 2.4e-10);
435         Assert.assertTrue(nbSteps < 150);
436       }
437     }
438     private int nbSteps;
439     private double maxError;
440     private TestProblem3 pb;
441   }

443   private static class VariableHandler implements StepHandler {
444     public VariableHandler() {
445         firstTime = true;
446         minStep = 0;
447         maxStep = 0;
448     }
449     public void init(double t0, double[] y0, double t) {
450       firstTime = true;
451       minStep = 0;
452       maxStep = 0;
453     }
454     public void handleStep(StepInterpolator interpolator,
455                            boolean isLast) {
456 
457       double step = FastMath.abs(interpolator.getCurrentTime()
458                              - interpolator.getPreviousTime());
459       if (firstTime) {
460         minStep   = FastMath.abs(step);
461         maxStep   = minStep;
462         firstTime = false;
463       } else {
464         if (step < minStep) {
465           minStep = step;
466         }
467         if (step > maxStep) {
468           maxStep = step;
469         }
470       }
471 
472       if (isLast) {
473         Assert.assertTrue(minStep < (1.0 / 100.0));
474         Assert.assertTrue(maxStep > (1.0 / 2.0));
475       }
476     }
477     private boolean firstTime = true;
478     private double  minStep = 0;
479     private double  maxStep = 0;
480   }
