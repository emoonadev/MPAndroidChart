17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.exception.NumberIsTooLargeException;
20  import org.apache.commons.math4.linear.OpenMapRealMatrix;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  public final class OpenMapRealMatrixTest {
25  
26      @Test(expected=NumberIsTooLargeException.class)
27      public void testMath679() {
28          new OpenMapRealMatrix(3, Integer.MAX_VALUE);
29      }
30  
31      @Test
32      public void testMath870() {
33          // Caveat: This implementation assumes that, for any {@code x},
34          // the equality {@code x * 0d == 0d} holds. But it is is not true for
35          // {@code NaN}. Moreover, zero entries will lose their sign.
36          // Some operations (that involve {@code NaN} and/or infinities) may
37          // thus give incorrect results.
38          OpenMapRealMatrix a = new OpenMapRealMatrix(3, 3);
39          OpenMapRealMatrix x = new OpenMapRealMatrix(3, 1);
40          x.setEntry(0, 0, Double.NaN);
41          x.setEntry(2, 0, Double.NEGATIVE_INFINITY);
42          OpenMapRealMatrix b = a.multiply(x);
43          for (int i = 0; i < b.getRowDimension(); ++i) {
44              for (int j = 0; j < b.getColumnDimension(); ++j) {
45                  // NaNs and infinities have disappeared, this is a limitation of our implementation
46                  Assert.assertEquals(0.0, b.getEntry(i, j), 1.0e-20);
47              }
48          }
49      }
50  }
