56    @Test
57    public void testDecreasingSteps()
58        throws DimensionMismatchException, NumberIsTooSmallException,
59               MaxCountExceededException, NoBracketingException {
60  
61        for (TestProblemAbstract pb : new TestProblemAbstract[] {
62            new TestProblem1(), new TestProblem2(), new TestProblem3(),
63            new TestProblem4(), new TestProblem5(), new TestProblem6()
64        }) {
65  
66        double previousValueError = Double.NaN;
67        double previousTimeError = Double.NaN;
68        for (int i = 4; i < 10; ++i) {
69  
70          double step = (pb.getFinalTime() - pb.getInitialTime()) * FastMath.pow(2.0, -i);
71  
72          FirstOrderIntegrator integ = new ThreeEighthesIntegrator(step);
73          TestProblemHandler handler = new TestProblemHandler(pb, integ);
74          integ.addStepHandler(handler);
75          EventHandler[] functions = pb.getEventsHandlers();
76          for (int l = 0; l < functions.length; ++l) {
77            integ.addEventHandler(functions[l],
78                                       Double.POSITIVE_INFINITY, 1.0e-6 * step, 1000);
79          }
80          double stopTime = integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
81                                            pb.getFinalTime(), new double[pb.getDimension()]);
82          if (functions.length == 0) {
83              Assert.assertEquals(pb.getFinalTime(), stopTime, 1.0e-10);
84          }
85  
86          double error = handler.getMaximalValueError();
87          if (i > 4) {
88            Assert.assertTrue(error < 1.01 * FastMath.abs(previousValueError));
89          }
90          previousValueError = error;
91  
92          double timeError = handler.getMaximalTimeError();
93          if (i > 4) {
94            Assert.assertTrue(timeError <= FastMath.abs(previousTimeError));
95          }
96          previousTimeError = timeError;
97  
98        }
99  
100     }

104  @Test
105  public void testSmallStep()
106      throws DimensionMismatchException, NumberIsTooSmallException,
107             MaxCountExceededException, NoBracketingException {
108 
109     TestProblem1 pb = new TestProblem1();
110     double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
111 
112     FirstOrderIntegrator integ = new ThreeEighthesIntegrator(step);
113     TestProblemHandler handler = new TestProblemHandler(pb, integ);
114     integ.addStepHandler(handler);
115     integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
116                     pb.getFinalTime(), new double[pb.getDimension()]);
117 
118     Assert.assertTrue(handler.getLastError() < 2.0e-13);
119     Assert.assertTrue(handler.getMaximalValueError() < 4.0e-12);
120     Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
121     Assert.assertEquals("3/8", integ.getName());
122 
123   }

165   @Test
166   public void testKepler()
167       throws DimensionMismatchException, NumberIsTooSmallException,
168              MaxCountExceededException, NoBracketingException {
169 
170     final TestProblem3 pb  = new TestProblem3(0.9);
171     double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.0003;
172 
173     FirstOrderIntegrator integ = new ThreeEighthesIntegrator(step);
174     integ.addStepHandler(new KeplerHandler(pb));
175     integ.integrate(pb,
176                     pb.getInitialTime(), pb.getInitialState(),
177                     pb.getFinalTime(), new double[pb.getDimension()]);
178   }
179 
180   private static class KeplerHandler implements StepHandler {
181 
182     public KeplerHandler(TestProblem3 pb) {
183       this.pb = pb;
184       maxError = 0;
185     }
186 
187     public void init(double t0, double[] y0, double t) {
188       maxError = 0;
189     }
190 
191     public void handleStep(StepInterpolator interpolator, boolean isLast)
192         throws MaxCountExceededException {
193 
194       double[] interpolatedY = interpolator.getInterpolatedState();
195       double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getCurrentTime());
196       double dx = interpolatedY[0] - theoreticalY[0];
197       double dy = interpolatedY[1] - theoreticalY[1];
198       double error = dx * dx + dy * dy;
199       if (error > maxError) {
200         maxError = error;
201       }
202       if (isLast) {
203         // even with more than 1000 evaluations per period,
204         // RK4 is not able to integrate such an eccentric
205         // orbit with a good accuracy
206         Assert.assertTrue(maxError > 0.005);
207       }
208     }
209 
210     private TestProblem3 pb;
211     private double maxError = 0;
212 
213   }

215   @Test
216   public void testStepSize()
217       throws DimensionMismatchException, NumberIsTooSmallException,
218              MaxCountExceededException, NoBracketingException {
219       final double step = 1.23456;
220       FirstOrderIntegrator integ = new ThreeEighthesIntegrator(step);
221       integ.addStepHandler(new StepHandler() {
222           public void handleStep(StepInterpolator interpolator, boolean isLast) {
223               if (! isLast) {
224                   Assert.assertEquals(step,
225                                interpolator.getCurrentTime() - interpolator.getPreviousTime(),
226                                1.0e-12);
227               }
228           }
229           public void init(double t0, double[] y0, double t) {
230           }
231       });
232       integ.integrate(new FirstOrderDifferentialEquations() {
233           public void computeDerivatives(double t, double[] y, double[] dot) {
234               dot[0] = 1.0;
235           }
236           public int getDimension() {
237               return 1;
238           }
239       }, 0.0, new double[] { 0.0 }, 5.0, new double[1]);
240   }
