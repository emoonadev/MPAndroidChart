17  package org.apache.commons.math4.linear;
18  
19  import org.apache.commons.math4.linear.MatrixDimensionMismatchException;
20  import org.junit.Assert;
21  import org.junit.Test;
22  
23  /**
24   * Test for {@link MatrixDimensionMismatchException}.
25   *
26   */
27  public class MatrixDimensionMismatchExceptionTest {
28      @Test
29      public void testAccessors() {
30          final MatrixDimensionMismatchException e
31              = new MatrixDimensionMismatchException(1, 2, 3, 4);
32          Assert.assertEquals(1, e.getWrongRowDimension());
33          Assert.assertEquals(2, e.getWrongColumnDimension());
34          Assert.assertEquals(3, e.getExpectedRowDimension());
35          Assert.assertEquals(4, e.getExpectedColumnDimension());
36      }
37  }
