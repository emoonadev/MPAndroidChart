17  package org.apache.commons.math4.stat.interval;
18  
19  import org.apache.commons.math4.stat.interval.AgrestiCoullInterval;
20  import org.apache.commons.math4.stat.interval.ClopperPearsonInterval;
21  import org.apache.commons.math4.stat.interval.ConfidenceInterval;
22  import org.apache.commons.math4.stat.interval.IntervalUtils;
23  import org.apache.commons.math4.stat.interval.NormalApproximationInterval;
24  import org.apache.commons.math4.stat.interval.WilsonScoreInterval;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  /**
29   * Test cases for the IntervalUtils class.
30   *
31   */
32  public class IntervalUtilsTest {
33  
34      private final int successes = 50;
35      private final int trials = 500;
36      private final double confidenceLevel = 0.9;
37  
38      // values to test must be exactly the same
39      private final double eps = 0.0;
40  
41      @Test
42      public void testAgrestiCoull() {
43          checkConfidenceIntervals(new AgrestiCoullInterval().createInterval(trials, successes, confidenceLevel),
44                                   IntervalUtils.getAgrestiCoullInterval(trials, successes, confidenceLevel));
45      }
46  
47      @Test
48      public void testClopperPearson() {
49          checkConfidenceIntervals(new ClopperPearsonInterval().createInterval(trials, successes, confidenceLevel),
50                                   IntervalUtils.getClopperPearsonInterval(trials, successes, confidenceLevel));
51      }
52  
53      @Test
54      public void testNormalApproximation() {
55          checkConfidenceIntervals(new NormalApproximationInterval().createInterval(trials, successes, confidenceLevel),
56                                   IntervalUtils.getNormalApproximationInterval(trials, successes, confidenceLevel));
57      }
58  
59      @Test
60      public void testWilsonScore() {
61          checkConfidenceIntervals(new WilsonScoreInterval().createInterval(trials, successes, confidenceLevel),
62                                   IntervalUtils.getWilsonScoreInterval(trials, successes, confidenceLevel));
63      }
64  
65      private void checkConfidenceIntervals(ConfidenceInterval expected, ConfidenceInterval actual) {
66          Assert.assertEquals(expected.getLowerBound(), actual.getLowerBound(), eps);
67          Assert.assertEquals(expected.getUpperBound(), actual.getUpperBound(), eps);
68          Assert.assertEquals(expected.getConfidenceLevel(), actual.getConfidenceLevel(), eps);
69      }
70  }
