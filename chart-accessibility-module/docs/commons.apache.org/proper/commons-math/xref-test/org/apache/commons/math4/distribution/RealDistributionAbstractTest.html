42  /**
43   * Abstract base class for {@link RealDistribution} tests.
44   * <p>
45   * To create a concrete test class for a continuous distribution
46   * implementation, first implement makeDistribution() to return a distribution
47   * instance to use in tests. Then implement each of the test data generation
48   * methods below.  In each case, the test points and test values arrays
49   * returned represent parallel arrays of inputs and expected values for the
50   * distribution returned by makeDistribution().  Default implementations
51   * are provided for the makeInverseXxx methods that just invert the mapping
52   * defined by the arrays returned by the makeCumulativeXxx methods.
53   * <p>
54   * makeCumulativeTestPoints() -- arguments used to test cumulative probabilities
55   * makeCumulativeTestValues() -- expected cumulative probabilities
56   * makeDensityTestValues() -- expected density values at cumulativeTestPoints
57   * makeInverseCumulativeTestPoints() -- arguments used to test inverse cdf
58   * makeInverseCumulativeTestValues() -- expected inverse cdf values
59   * <p>
60   * To implement additional test cases with different distribution instances and
61   * test data, use the setXxx methods for the instance data in test cases and
62   * call the verifyXxx methods to verify results.
63   * <p>
64   * Error tolerance can be overridden by implementing getTolerance().
65   * <p>
66   * Test data should be validated against reference tables or other packages
67   * where possible, and the source of the reference data and/or validation
68   * should be documented in the test cases.  A framework for validating
69   * distribution data against R is included in the /src/test/R source tree.
70   * <p>
71   * See {@link NormalDistributionTest} and {@link ChiSquaredDistributionTest}
72   * for examples.
73   *
74   */
75  public abstract class RealDistributionAbstractTest {

116     /** Creates the default logarithmic density test expected values.
117      * The default implementation simply computes the logarithm
118      * of each value returned by {@link #makeDensityTestValues()}.*/
119     public double[] makeLogDensityTestValues() {
120         final double[] densityTestValues = makeDensityTestValues();
121         final double[] logDensityTestValues = new double[densityTestValues.length];
122         for (int i = 0; i < densityTestValues.length; i++) {
123             logDensityTestValues[i] = FastMath.log(densityTestValues[i]);
124         }
125         return logDensityTestValues;
126     }

172     /**
173      * Verifies that cumulative probability density calculations match expected values
174      * using current test instance data
175      */
176     protected void verifyCumulativeProbabilities() {
177         // verify cumulativeProbability(double)
178         for (int i = 0; i < cumulativeTestPoints.length; i++) {
179             TestUtils.assertEquals("Incorrect cumulative probability value returned for "
180                 + cumulativeTestPoints[i], cumulativeTestValues[i],
181                 distribution.cumulativeProbability(cumulativeTestPoints[i]),
182                 getTolerance());
183         }
184         // verify probability(double, double)
185         for (int i = 0; i < cumulativeTestPoints.length; i++) {
186             for (int j = 0; j < cumulativeTestPoints.length; j++) {
187                 if (cumulativeTestPoints[i] <= cumulativeTestPoints[j]) {
188                     TestUtils.assertEquals(cumulativeTestValues[j] - cumulativeTestValues[i],
189                         distribution.probability(cumulativeTestPoints[i], cumulativeTestPoints[j]),
190                         getTolerance());
191                 } else {
192                     try {
193                         distribution.probability(cumulativeTestPoints[i], cumulativeTestPoints[j]);
194                     } catch (NumberIsTooLargeException e) {
195                         continue;
196                     }
197                     Assert.fail("distribution.probability(double, double) should have thrown an exception that second argument is too large");
198                 }
199             }
200         }
201     }
202

278     /**
279      * Verifies that probability computations are consistent
280      */
281     @Test
282     public void testConsistency() {
283         for (int i=1; i < cumulativeTestPoints.length; i++) {
284 
285             // check that cdf(x, x) = 0
286             TestUtils.assertEquals(0d,
287                distribution.probability
288                  (cumulativeTestPoints[i], cumulativeTestPoints[i]), tolerance);
289 
290             // check that P(a < X <= b) = P(X <= b) - P(X <= a)
291             double upper = FastMath.max(cumulativeTestPoints[i], cumulativeTestPoints[i -1]);
292             double lower = FastMath.min(cumulativeTestPoints[i], cumulativeTestPoints[i -1]);
293             double diff = distribution.cumulativeProbability(upper) -
294                 distribution.cumulativeProbability(lower);
295             double direct = distribution.probability(lower, upper);
296             TestUtils.assertEquals("Inconsistent probability for ("
297                     + lower + "," + upper + ")", diff, direct, tolerance);
298         }
299     }

327      * Test sampling
328      */
329     @Test
330     public void testSampler() {
331         final int sampleSize = 1000;
332         final RealDistribution.Sampler sampler =
333             distribution.createSampler(RandomSource.create(RandomSource.WELL_19937_C, 123456789L));
334         final double[] sample = AbstractRealDistribution.sample(sampleSize, sampler);
335         final double[] quartiles = TestUtils.getDistributionQuartiles(distribution);
336         final double[] expected = {250, 250, 250, 250};
337         final long[] counts = new long[4];
338 
339         for (int i = 0; i < sampleSize; i++) {
340             TestUtils.updateCounts(sample[i], counts, quartiles);
341         }
342         TestUtils.assertChiSquareAccept(expected, counts, 0.001);
343     }

345     /**
346      * Verify that density integrals match the distribution.
347      * The (filtered, sorted) cumulativeTestPoints array is used to source
348      * integration limits. The integral of the density (estimated using a
349      * Legendre-Gauss integrator) is compared with the cdf over the same
350      * interval. Test points outside of the domain of the density function
351      * are discarded.
352      */
353     @Test
354     public void testDensityIntegrals() {
355         final double tol = 1.0e-9;
356         final BaseAbstractUnivariateIntegrator integrator =
357             new IterativeLegendreGaussIntegrator(5, 1.0e-12, 1.0e-10);
358         final UnivariateFunction d = new UnivariateFunction() {
359             public double value(double x) {
360                 return distribution.density(x);
361             }
362         };
363         final ArrayList<Double> integrationTestPoints = new ArrayList<Double>();
364         for (int i = 0; i < cumulativeTestPoints.length; i++) {
365             if (Double.isNaN(cumulativeTestValues[i]) ||
366                     cumulativeTestValues[i] < 1.0e-5 ||
367                     cumulativeTestValues[i] > 1 - 1.0e-5) {
368                 continue; // exclude integrals outside domain.
369             }
370             integrationTestPoints.add(cumulativeTestPoints[i]);
371         }
372         Collections.sort(integrationTestPoints);
373         for (int i = 1; i < integrationTestPoints.size(); i++) {
374             Assert.assertEquals(
375                     distribution.probability(
376                             integrationTestPoints.get(0), integrationTestPoints.get(i)),
377                             integrator.integrate(
378                                     1000000, // Triangle integrals are very slow to converge
379                                     d, integrationTestPoints.get(0),
380                                     integrationTestPoints.get(i)), tol);
381         }
382     }

384     @Test
385     public void testDistributionClone()
386         throws IOException,
387                ClassNotFoundException {
388         // Construct a distribution and initialize its internal random
389         // generator, using a fixed seed for deterministic results.
390         final long seed = 123;
391         RandomSource source = RandomSource.WELL_512_A;
392         RealDistribution.Sampler origSampler = distribution.createSampler(RandomSource.create(source, seed));
393 
394         // Clone the distribution.
395         final RealDistribution cloned = deepClone();
396         RealDistribution.Sampler clonedSampler = cloned.createSampler(RandomSource.create(source, seed));
397 
398         // Make sure they still produce the same samples.
399         Assert.assertEquals(origSampler.sample(),
400                             clonedSampler.sample(),
401                             0d);
402     }

497     /**
498      * Serialization and deserialization loop of the {@link #distribution}.
499      */
500     private RealDistribution deepClone()
501         throws IOException,
502                ClassNotFoundException {
503         // Serialize to byte array.
504         final ByteArrayOutputStream bOut = new ByteArrayOutputStream();
505         final ObjectOutputStream oOut = new ObjectOutputStream(bOut);
506         oOut.writeObject(distribution);
507         final byte[] data = bOut.toByteArray();
508 
509         // Deserialize from byte array.
510         final ByteArrayInputStream bIn = new ByteArrayInputStream(data);
511         final ObjectInputStream oIn = new ObjectInputStream(bIn);
512         final Object clone = oIn.readObject();
513         oIn.close();
514 
515         return (RealDistribution) clone;
516     }
