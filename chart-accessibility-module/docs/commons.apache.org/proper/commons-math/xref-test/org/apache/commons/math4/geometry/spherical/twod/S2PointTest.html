17  package org.apache.commons.math4.geometry.spherical.twod;
18  
19  import org.apache.commons.math4.exception.OutOfRangeException;
20  import org.apache.commons.math4.geometry.spherical.oned.Sphere1D;
21  import org.apache.commons.math4.geometry.spherical.twod.S2Point;
22  import org.apache.commons.math4.geometry.spherical.twod.Sphere2D;
23  import org.apache.commons.math4.util.FastMath;
24  import org.apache.commons.math4.util.MathUtils;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  public class S2PointTest {
29  
30  
31      @Test
32      public void testS2Point() {
33          for (int k = -2; k < 3; ++k) {
34              S2Point p = new S2Point(1.0 + k * MathUtils.TWO_PI, 1.4);
35              Assert.assertEquals(1.0 + k * MathUtils.TWO_PI, p.getTheta(), 1.0e-10);
36              Assert.assertEquals(1.4, p.getPhi(), 1.0e-10);
37              Assert.assertEquals(FastMath.cos(1.0) * FastMath.sin(1.4), p.getVector().getX(), 1.0e-10);
38              Assert.assertEquals(FastMath.sin(1.0) * FastMath.sin(1.4), p.getVector().getY(), 1.0e-10);
39              Assert.assertEquals(FastMath.cos(1.4), p.getVector().getZ(), 1.0e-10);
40              Assert.assertFalse(p.isNaN());
41          }
42      }
43  
44      @Test(expected=OutOfRangeException.class)
45      public void testNegativePolarAngle() {
46          new S2Point(1.0, -1.0);
47      }
48  
49      @Test(expected=OutOfRangeException.class)
50      public void testTooLargePolarAngle() {
51          new S2Point(1.0, 3.5);
52      }
53  
54      @Test
55      public void testNaN() {
56          Assert.assertTrue(S2Point.NaN.isNaN());
57          Assert.assertTrue(S2Point.NaN.equals(new S2Point(Double.NaN, 1.0)));
58          Assert.assertFalse(new S2Point(1.0, 1.3).equals(S2Point.NaN));
59      }
60  
61      @Test
62      public void testEquals() {
63          S2Point a = new S2Point(1.0, 1.0);
64          S2Point b = new S2Point(1.0, 1.0);
65          Assert.assertEquals(a.hashCode(), b.hashCode());
66          Assert.assertFalse(a == b);
67          Assert.assertTrue(a.equals(b));
68          Assert.assertTrue(a.equals(a));
69          Assert.assertFalse(a.equals('a'));
70      }
71  
72      @Test
73      public void testDistance() {
74          S2Point a = new S2Point(1.0, 0.5 * FastMath.PI);
75          S2Point b = new S2Point(a.getTheta() + 0.5 * FastMath.PI, a.getPhi());
76          Assert.assertEquals(0.5 * FastMath.PI, a.distance(b), 1.0e-10);
77          Assert.assertEquals(FastMath.PI, a.distance(a.negate()), 1.0e-10);
78          Assert.assertEquals(0.5 * FastMath.PI, S2Point.MINUS_I.distance(S2Point.MINUS_K), 1.0e-10);
79          Assert.assertEquals(0.0, new S2Point(1.0, 0).distance(new S2Point(2.0, 0)), 1.0e-10);
80      }
81  
82      @Test
83      public void testSpace() {
84          S2Point a = new S2Point(1.0, 1.0);
85          Assert.assertTrue(a.getSpace() instanceof Sphere2D);
86          Assert.assertEquals(2, a.getSpace().getDimension());
87          Assert.assertTrue(a.getSpace().getSubSpace() instanceof Sphere1D);
88      }
