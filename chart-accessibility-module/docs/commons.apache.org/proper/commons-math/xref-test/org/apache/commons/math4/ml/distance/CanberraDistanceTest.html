17  package org.apache.commons.math4.ml.distance;
18  
19  import org.apache.commons.math4.ml.distance.CanberraDistance;
20  import org.apache.commons.math4.ml.distance.DistanceMeasure;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Tests for {@link CanberraDistance} class.
26   */
27  public class CanberraDistanceTest {
28      final DistanceMeasure distance = new CanberraDistance();
29  
30      @Test
31      public void testZero() {
32          final double[] a = { 0, 1, -2, 3.4, 5, -6.7, 89 };
33          Assert.assertEquals(0, distance.compute(a, a), 0d);
34      }
35  
36      @Test
37      public void testZero2() {
38          final double[] a = { 0, 0 };
39          Assert.assertEquals(0, distance.compute(a, a), 0d);
40      }
41  
42      @Test
43      public void test() {
44          final double[] a = { 1, 2, 3, 4, 9 };
45          final double[] b = { -5, -6, 7, 4, 3 };
46          final double expected = 2.9;
47          Assert.assertEquals(expected, distance.compute(a, b), 0d);
48          Assert.assertEquals(expected, distance.compute(b, a), 0d);
49      }
50  }
