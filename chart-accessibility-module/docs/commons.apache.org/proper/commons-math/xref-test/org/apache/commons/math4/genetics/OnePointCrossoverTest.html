17  package org.apache.commons.math4.genetics;
18  
19  import org.apache.commons.math4.genetics.BinaryChromosome;
20  import org.apache.commons.math4.genetics.ChromosomePair;
21  import org.apache.commons.math4.genetics.OnePointCrossover;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  public class OnePointCrossoverTest {
26  
27      @Test
28      public void testCrossover() {
29          @SuppressWarnings("boxing")
30          Integer[] p1 = new Integer[] {1,0,1,0,0,1,0,1,1};
31          @SuppressWarnings("boxing")
32          Integer[] p2 = new Integer[] {0,1,1,0,1,0,1,1,1};
33  
34          BinaryChromosome p1c = new DummyBinaryChromosome(p1);
35          BinaryChromosome p2c = new DummyBinaryChromosome(p2);
36  
37          OnePointCrossover<Integer> opc = new OnePointCrossover<Integer>();
38  
39          // how to test a stochastic method?
40          for (int i=0; i<20; i++) {
41              ChromosomePair pair = opc.crossover(p1c,p2c);
42  
43              Integer[] c1 = new Integer[p1.length];
44              Integer[] c2 = new Integer[p2.length];
45  
46              c1 = ((BinaryChromosome) pair.getFirst()).getRepresentation().toArray(c1);
47              c2 = ((BinaryChromosome) pair.getSecond()).getRepresentation().toArray(c2);
48  
49              // first and last values will be the same
50              Assert.assertEquals(p1[0],  c1[0]);
51              Assert.assertEquals(p2[0], c2[0]);
52              Assert.assertEquals(p1[p1.length-1], c1[c1.length-1]);
53              Assert.assertEquals(p2[p2.length-1], c2[c2.length-1]);
54              // moreover, in the above setting, the 2nd, 3rd and 7th values will be the same
55              Assert.assertEquals(p1[2], c1[2]);
56              Assert.assertEquals(p2[2], c2[2]);
57              Assert.assertEquals(p1[3], c1[3]);
58              Assert.assertEquals(p2[3], c2[3]);
59              Assert.assertEquals(p1[7], c1[7]);
60              Assert.assertEquals(p2[7], c2[7]);
61          }
62      }
63  
64  }
