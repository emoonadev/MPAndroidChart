36      /** error tolerance for spline interpolator value at knot points */
37      protected double knotTolerance = 1E-14;

39      /** error tolerance for interpolating polynomial coefficients */
40      protected double coefficientTolerance = 1E-14;

42      /** error tolerance for interpolated values -- high value is from sin test */
43      protected double interpolationTolerance = 1E-14;

45      @Test
46      public void testInterpolateLinearDegenerateTwoSegment()
47          {
48          double tolerance = 1e-15;
49          double x[] = { 0.0, 0.5, 1.0 };
50          double y[] = { 0.0, 0.5, 1.0 };
51          UnivariateInterpolator i = new SplineInterpolator();
52          UnivariateFunction f = i.interpolate(x, y);
53          verifyInterpolation(f, x, y);
54          verifyConsistency((PolynomialSplineFunction) f, x);
55  
56          // Verify coefficients using analytical values
57          PolynomialFunction polynomials[] = ((PolynomialSplineFunction) f).getPolynomials();
58          double target[] = {y[0], 1d};
59          TestUtils.assertEquals(polynomials[0].getCoefficients(), target, coefficientTolerance);
60          target = new double[]{y[1], 1d};
61          TestUtils.assertEquals(polynomials[1].getCoefficients(), target, coefficientTolerance);
62  
63          // Check interpolation
64          Assert.assertEquals(0.0,f.value(0.0), tolerance);
65          Assert.assertEquals(0.4,f.value(0.4), tolerance);
66          Assert.assertEquals(1.0,f.value(1.0), tolerance);
67      }

111     @Test
112     public void testInterpolateSin() {
113         double sineCoefficientTolerance = 1e-6;
114         double sineInterpolationTolerance = 0.0043;
115         double x[] =
116             {
117                 0.0,
118                 FastMath.PI / 6d,
119                 FastMath.PI / 2d,
120                 5d * FastMath.PI / 6d,
121                 FastMath.PI,
122                 7d * FastMath.PI / 6d,
123                 3d * FastMath.PI / 2d,
124                 11d * FastMath.PI / 6d,
125                 2.d * FastMath.PI };
126         double y[] = { 0d, 0.5d, 1d, 0.5d, 0d, -0.5d, -1d, -0.5d, 0d };
127         UnivariateInterpolator i = new SplineInterpolator();
128         UnivariateFunction f = i.interpolate(x, y);
129         verifyInterpolation(f, x, y);
130         verifyConsistency((PolynomialSplineFunction) f, x);
131 
132         /* Check coefficients against values computed using R (version 1.8.1, Red Hat Linux 9)
133          *
134          * To replicate in R:
135          *     x[1] <- 0
136          *     x[2] <- pi / 6, etc, same for y[] (could use y <- scan() for y values)
137          *     g <- splinefun(x, y, "natural")
138          *     splinecoef <- eval(expression(z), envir = environment(g))
139          *     print(splinecoef)
140          */
141         PolynomialFunction polynomials[] = ((PolynomialSplineFunction) f).getPolynomials();
142         double target[] = {y[0], 1.002676d, 0d, -0.17415829d};
143         TestUtils.assertEquals(polynomials[0].getCoefficients(), target, sineCoefficientTolerance);
144         target = new double[]{y[1], 8.594367e-01, -2.735672e-01, -0.08707914};
145         TestUtils.assertEquals(polynomials[1].getCoefficients(), target, sineCoefficientTolerance);
146         target = new double[]{y[2], 1.471804e-17,-5.471344e-01, 0.08707914};
147         TestUtils.assertEquals(polynomials[2].getCoefficients(), target, sineCoefficientTolerance);
148         target = new double[]{y[3], -8.594367e-01, -2.735672e-01, 0.17415829};
149         TestUtils.assertEquals(polynomials[3].getCoefficients(), target, sineCoefficientTolerance);
150         target = new double[]{y[4], -1.002676, 6.548562e-17, 0.17415829};
151         TestUtils.assertEquals(polynomials[4].getCoefficients(), target, sineCoefficientTolerance);
152         target = new double[]{y[5], -8.594367e-01, 2.735672e-01, 0.08707914};
153         TestUtils.assertEquals(polynomials[5].getCoefficients(), target, sineCoefficientTolerance);
154         target = new double[]{y[6], 3.466465e-16, 5.471344e-01, -0.08707914};
155         TestUtils.assertEquals(polynomials[6].getCoefficients(), target, sineCoefficientTolerance);
156         target = new double[]{y[7], 8.594367e-01, 2.735672e-01, -0.17415829};
157         TestUtils.assertEquals(polynomials[7].getCoefficients(), target, sineCoefficientTolerance);
158 
159         //Check interpolation
160         Assert.assertEquals(FastMath.sqrt(2d) / 2d,f.value(FastMath.PI/4d),sineInterpolationTolerance);
161         Assert.assertEquals(FastMath.sqrt(2d) / 2d,f.value(3d*FastMath.PI/4d),sineInterpolationTolerance);
162     }

164     @Test
165     public void testIllegalArguments() {
166         // Data set arrays of different size.
167         UnivariateInterpolator i = new SplineInterpolator();
168         try {
169             double xval[] = { 0.0, 1.0 };
170             double yval[] = { 0.0, 1.0, 2.0 };
171             i.interpolate(xval, yval);
172             Assert.fail("Failed to detect data set array with different sizes.");
173         } catch (DimensionMismatchException iae) {
174             // Expected.
175         }
176         // X values not sorted.
177         try {
178             double xval[] = { 0.0, 1.0, 0.5 };
179             double yval[] = { 0.0, 1.0, 2.0 };
180             i.interpolate(xval, yval);
181             Assert.fail("Failed to detect unsorted arguments.");
182         } catch (NonMonotonicSequenceException iae) {
183             // Expected.
184         }
185         // Not enough data to interpolate.
186         try {
187             double xval[] = { 0.0, 1.0 };
188             double yval[] = { 0.0, 1.0 };
189             i.interpolate(xval, yval);
190             Assert.fail("Failed to detect unsorted arguments.");
191         } catch (NumberIsTooSmallException iae) {
192             // Expected.
193         }
194     }

196     /**
197      * verifies that f(x[i]) = y[i] for i = 0..n-1 where n is common length.
198      */
199     protected void verifyInterpolation(UnivariateFunction f, double x[], double y[])
200        {
201         for (int i = 0; i < x.length; i++) {
202             Assert.assertEquals(f.value(x[i]), y[i], knotTolerance);
203         }
204     }
205 
206     /**
207      * Verifies that interpolating polynomials satisfy consistency requirement:
208      *    adjacent polynomials must agree through two derivatives at knot points
209      */
210     protected void verifyConsistency(PolynomialSplineFunction f, double x[])
211         {
212         PolynomialFunction polynomials[] = f.getPolynomials();
213         for (int i = 1; i < x.length - 2; i++) {
214             // evaluate polynomials and derivatives at x[i + 1]
215             Assert.assertEquals(polynomials[i].value(x[i +1] - x[i]), polynomials[i + 1].value(0), 0.1);
216             Assert.assertEquals(polynomials[i].polynomialDerivative().value(x[i +1] - x[i]),
217                                 polynomials[i + 1].polynomialDerivative().value(0), 0.5);
218             Assert.assertEquals(polynomials[i].polynomialDerivative().polynomialDerivative().value(x[i +1] - x[i]),
219                                 polynomials[i + 1].polynomialDerivative().polynomialDerivative().value(0), 0.5);
220         }
221     }
