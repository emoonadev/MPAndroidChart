17  package org.apache.commons.math4.fraction;
18  
19  
20  import org.apache.commons.math4.TestUtils;
21  import org.apache.commons.math4.fraction.BigFraction;
22  import org.apache.commons.math4.fraction.BigFractionField;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class BigFractionFieldTest {
27  
28      @Test
29      public void testZero() {
30          Assert.assertEquals(BigFraction.ZERO, BigFractionField.getInstance().getZero());
31      }
32  
33      @Test
34      public void testOne() {
35          Assert.assertEquals(BigFraction.ONE, BigFractionField.getInstance().getOne());
36      }
37  
38      @Test
39      public void testSerial() {
40          // deserializing the singleton should give the singleton itself back
41          BigFractionField field = BigFractionField.getInstance();
42          Assert.assertTrue(field == TestUtils.serializeAndRecover(field));
43      }
44  
45  }
