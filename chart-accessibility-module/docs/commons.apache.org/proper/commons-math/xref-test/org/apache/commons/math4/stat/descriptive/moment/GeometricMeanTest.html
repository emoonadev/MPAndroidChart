17  package org.apache.commons.math4.stat.descriptive.moment;
18  
19  import org.apache.commons.math4.stat.descriptive.StorelessUnivariateStatisticAbstractTest;
20  import org.apache.commons.math4.stat.descriptive.UnivariateStatistic;
21  import org.apache.commons.math4.stat.descriptive.moment.GeometricMean;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Test cases for the {@link UnivariateStatistic} class.
27   */
28  public class GeometricMeanTest extends StorelessUnivariateStatisticAbstractTest{
29  
30      protected GeometricMean stat;
31  
32      /**
33       * {@inheritDoc}
34       */
35      @Override
36      public UnivariateStatistic getUnivariateStatistic() {
37          return new GeometricMean();
38      }
39  
40      /**
41       * {@inheritDoc}
42       */
43      @Override
44      public double expectedValue() {
45          return this.geoMean;
46      }
47  
48      @Test
49      public void testSpecialValues() {
50          GeometricMean mean = new GeometricMean();
51          // empty
52          Assert.assertTrue(Double.isNaN(mean.getResult()));
53  
54          // finite data
55          mean.increment(1d);
56          Assert.assertFalse(Double.isNaN(mean.getResult()));
57  
58          // add 0 -- makes log sum blow to minus infinity, should make 0
59          mean.increment(0d);
60          Assert.assertEquals(0d, mean.getResult(), 0);
61  
62          // add positive infinity - note the minus infinity above
63          mean.increment(Double.POSITIVE_INFINITY);
64          Assert.assertTrue(Double.isNaN(mean.getResult()));
65  
66          // clear
67          mean.clear();
68          Assert.assertTrue(Double.isNaN(mean.getResult()));
69  
70          // positive infinity by itself
71          mean.increment(Double.POSITIVE_INFINITY);
72          Assert.assertEquals(Double.POSITIVE_INFINITY, mean.getResult(), 0);
73  
74          // negative value -- should make NaN
75          mean.increment(-2d);
76          Assert.assertTrue(Double.isNaN(mean.getResult()));
77      }
78  
79  }
