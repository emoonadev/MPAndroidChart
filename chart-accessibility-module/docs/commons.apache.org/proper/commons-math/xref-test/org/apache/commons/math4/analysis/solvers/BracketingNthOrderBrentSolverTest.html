72      @Test
73      public void testConvergenceOnFunctionAccuracy() {
74          BracketingNthOrderBrentSolver solver =
75                  new BracketingNthOrderBrentSolver(1.0e-12, 1.0e-10, 0.001, 3);
76          QuinticFunction f = new QuinticFunction();
77          double result = solver.solve(20, f, 0.2, 0.9, 0.4, AllowedSolution.BELOW_SIDE);
78          Assert.assertEquals(0, f.value(result), solver.getFunctionValueAccuracy());
79          Assert.assertTrue(f.value(result) <= 0);
80          Assert.assertTrue(result - 0.5 > solver.getAbsoluteAccuracy());
81          result = solver.solve(20, f, -0.9, -0.2,  -0.4, AllowedSolution.ABOVE_SIDE);
82          Assert.assertEquals(0, f.value(result), solver.getFunctionValueAccuracy());
83          Assert.assertTrue(f.value(result) >= 0);
84          Assert.assertTrue(result + 0.5 < -solver.getAbsoluteAccuracy());
85      }

96          double result = solver.solve(100, sharpTurn, -0.9999999, 30, 15, AllowedSolution.RIGHT_SIDE);
97          Assert.assertEquals(0, sharpTurn.value(result), solver.getFunctionValueAccuracy());
98          Assert.assertTrue(sharpTurn.value(result) >= 0);
99          Assert.assertEquals(-0.5, result, 1.0e-10);

102     @Test
103     public void testFasterThanNewton() {
104         // the following test functions come from Beny Neta's paper:
105         // "Several New Methods for solving Equations"
106         // intern J. Computer Math Vol 23 pp 265-282
107         // available here: http://www.math.nps.navy.mil/~bneta/SeveralNewMethods.PDF
108         // the reference roots have been computed by the Dfp solver to more than
109         // 80 digits and checked with emacs (only the first 20 digits are reproduced here)
110         compare(new TestFunction(0.0, -2, 2) {
111             @Override
112             public DerivativeStructure value(DerivativeStructure x) {
113                 return x.sin().subtract(x.multiply(0.5));
114             }
115         });
116         compare(new TestFunction(6.3087771299726890947, -5, 10) {
117             @Override
118             public DerivativeStructure value(DerivativeStructure x) {
119                 return x.pow(5).add(x).subtract(10000);
120             }
121         });
122         compare(new TestFunction(9.6335955628326951924, 0.001, 10) {
123             @Override
124             public DerivativeStructure value(DerivativeStructure x) {
125                 return x.sqrt().subtract(x.reciprocal()).subtract(3);
126             }
127         });
128         compare(new TestFunction(2.8424389537844470678, -5, 5) {
129             @Override
130             public DerivativeStructure value(DerivativeStructure x) {
131                 return x.exp().add(x).subtract(20);
132             }
133         });
134         compare(new TestFunction(8.3094326942315717953, 0.001, 10) {
135             @Override
136             public DerivativeStructure value(DerivativeStructure x) {
137                 return x.log().add(x.sqrt()).subtract(5);
138             }
139         });
140         compare(new TestFunction(1.4655712318767680266, -0.5, 1.5) {
141             @Override
142             public DerivativeStructure value(DerivativeStructure x) {
143                 return x.subtract(1).multiply(x).multiply(x).subtract(1);
144             }
145         });

149     private void compare(TestFunction f) {
150         compare(f, f.getRoot(), f.getMin(), f.getMax());
151     }
152 
153     private void compare(final UnivariateDifferentiableFunction f,
154                          double root, double min, double max) {
155         NewtonRaphsonSolver newton = new NewtonRaphsonSolver(1.0e-12);
156         BracketingNthOrderBrentSolver bracketing =
157                 new BracketingNthOrderBrentSolver(1.0e-12, 1.0e-12, 1.0e-18, 5);
158         double resultN;
159         try {
160             resultN = newton.solve(100, f, min, max);
161         } catch (TooManyEvaluationsException tmee) {
162             resultN = Double.NaN;
163         }
164         double resultB;
165         try {
166             resultB = bracketing.solve(100, f, min, max);
167         } catch (TooManyEvaluationsException tmee) {
168             resultB = Double.NaN;
169         }
170         Assert.assertEquals(root, resultN, newton.getAbsoluteAccuracy());
171         Assert.assertEquals(root, resultB, bracketing.getAbsoluteAccuracy());
172 
173         // bracketing solver evaluates only function value, we set the weight to 1
174         final int weightedBracketingEvaluations = bracketing.getEvaluations();
175 
176         // Newton-Raphson solver evaluates both function value and derivative, we set the weight to 2
177         final int weightedNewtonEvaluations = 2 * newton.getEvaluations();
178 
179         Assert.assertTrue(weightedBracketingEvaluations < weightedNewtonEvaluations);
180 
181     }

183     private static abstract class TestFunction implements UnivariateDifferentiableFunction {
184 
185         private final double root;
186         private final double min;
187         private final double max;
188 
189         protected TestFunction(final double root, final double min, final double max) {
190             this.root = root;
191             this.min  = min;
192             this.max  = max;
193         }
194 
195         public double getRoot() {
196             return root;
197         }
198 
199         public double getMin() {
200             return min;
201         }
202 
203         public double getMax() {
204             return max;
205         }
206 
207         public double value(final double x) {
208             return value(new DerivativeStructure(0, 0, x)).getValue();
209         }
210 
211         public abstract DerivativeStructure value(final DerivativeStructure t);
212 
213     }
