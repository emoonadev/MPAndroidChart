67      @Test
68      public void testAddObsGetNClear() {
69          MillerUpdatingRegression instance = new MillerUpdatingRegression(3, true);
70          double[][] xAll = new double[airdata[0].length][];
71          double[] y = new double[airdata[0].length];
72          for (int i = 0; i < airdata[0].length; i++) {
73              xAll[i] = new double[3];
74              xAll[i][0] = FastMath.log(airdata[3][i]);
75              xAll[i][1] = FastMath.log(airdata[4][i]);
76              xAll[i][2] = airdata[5][i];
77              y[i] = FastMath.log(airdata[2][i]);
78          }
79          instance.addObservations(xAll, y);
80          if (instance.getN() != xAll.length) {
81              Assert.fail("Number of observations not correct in bulk addition");
82          }
83          instance.clear();
84          for (int i = 0; i < xAll.length; i++) {
85              instance.addObservation(xAll[i], y[i]);
86          }
87          if (instance.getN() != xAll.length) {
88              Assert.fail("Number of observations not correct in drip addition");
89          }
90          return;
91      }

972         int off2 = 6;
973         for (int i = 0; i < 4; i++) {
974             for (int j = 0; j < i; j++) {
975                 if (FastMath.abs(pc[idx] - cp[off] / (diag[i] * diag[j])) > 1.0e-8) {
976                     Assert.fail("Failed cross products... i = " + i + " j = " + j);
977                 }
978                 ++idx;
979                 ++off;
980             }
981             ++off;
982             if (FastMath.abs(pc[i+off2] - yxcorr[ i] / (FastMath.sqrt(sumysq) * diag[i])) > 1.0e-8) {
983                 Assert.fail("Assert.failed cross product i = " + i + " y");
984             }
985         }
986         double[] pc2 = instance.getPartialCorrelations(1);

991             for (int j = 1; j < i; j++) {
992                 if (FastMath.abs(pc2[idx] - corr.getEntry(j, i)) > 1.0e-8) {
993                     Assert.fail("Failed cross products... i = " + i + " j = " + j);
994                 }
995                 ++idx;
