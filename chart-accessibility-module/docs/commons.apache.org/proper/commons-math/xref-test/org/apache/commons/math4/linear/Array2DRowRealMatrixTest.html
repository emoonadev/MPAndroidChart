426     /** test examples in user guide */
427     @Test
428     public void testExamples() {
429         // Create a real matrix with two rows and three columns
430         double[][] matrixData = { {1d,2d,3d}, {2d,5d,3d}};
431         RealMatrix m = new Array2DRowRealMatrix(matrixData);
432         // One more with three rows, two columns
433         double[][] matrixData2 = { {1d,2d}, {2d,5d}, {1d, 7d}};
434         RealMatrix n = new Array2DRowRealMatrix(matrixData2);
435         // Now multiply m by n
436         RealMatrix p = m.multiply(n);
437         Assert.assertEquals(2, p.getRowDimension());
438         Assert.assertEquals(2, p.getColumnDimension());
439         // Invert p
440         RealMatrix pInverse = new LUDecomposition(p).getSolver().getInverse();
441         Assert.assertEquals(2, pInverse.getRowDimension());
442         Assert.assertEquals(2, pInverse.getColumnDimension());
443 
444         // Solve example
445         double[][] coefficientsData = {{2, 3, -2}, {-1, 7, 6}, {4, -3, -5}};
446         RealMatrix coefficients = new Array2DRowRealMatrix(coefficientsData);
447         RealVector constants = new ArrayRealVector(new double[]{1, -2, 1}, false);
448         RealVector solution = new LUDecomposition(coefficients).getSolver().solve(constants);
449         final double cst0 = constants.getEntry(0);
450         final double cst1 = constants.getEntry(1);
451         final double cst2 = constants.getEntry(2);
452         final double sol0 = solution.getEntry(0);
453         final double sol1 = solution.getEntry(1);
454         final double sol2 = solution.getEntry(2);
455         Assert.assertEquals(2 * sol0 + 3 * sol1 -2 * sol2, cst0, 1E-12);
456         Assert.assertEquals(-1 * sol0 + 7 * sol1 + 6 * sol2, cst1, 1E-12);
457         Assert.assertEquals(4 * sol0 - 3 * sol1 -5 * sol2, cst2, 1E-12);
458     }

1114     /** extracts the l  and u matrices from compact lu representation */
1115     protected void splitLU(RealMatrix lu, double[][] lowerData, double[][] upperData) {
1116         if (!lu.isSquare()) {
1117             throw new NonSquareMatrixException(lu.getRowDimension(), lu.getColumnDimension());
1118         }
1119         if (lowerData.length != lowerData[0].length) {
1120             throw new DimensionMismatchException(lowerData.length, lowerData[0].length);
1121         }
1122         if (upperData.length != upperData[0].length) {
1123             throw new DimensionMismatchException(upperData.length, upperData[0].length);
1124         }
1125         if (lowerData.length != upperData.length) {
1126             throw new DimensionMismatchException(lowerData.length, upperData.length);
1127         }
1128         if (lowerData.length != lu.getRowDimension()) {
1129             throw new DimensionMismatchException(lowerData.length, lu.getRowDimension());
1130         }
1131 
1132         int n = lu.getRowDimension();
1133         for (int i = 0; i < n; i++) {
1134             for (int j = 0; j < n; j++) {
1135                 if (j < i) {
1136                     lowerData[i][j] = lu.getEntry(i, j);
1137                     upperData[i][j] = 0d;
1138                 } else if (i == j) {
1139                     lowerData[i][j] = 1d;
1140                     upperData[i][j] = lu.getEntry(i, j);
1141                 } else {
1142                     lowerData[i][j] = 0d;
1143                     upperData[i][j] = lu.getEntry(i, j);
1144                 }
1145             }
1146         }
1147     }
