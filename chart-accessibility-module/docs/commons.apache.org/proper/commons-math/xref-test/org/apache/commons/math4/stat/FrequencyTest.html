53      /** test freq counts */
54      @Test
55      public void testCounts() {
56          Assert.assertEquals("total count",0,f.getSumFreq());
57          f.addValue(ONE_LONG);
58          f.addValue(TWO_LONG);
59          f.addValue(1);
60          f.addValue(ONE);
61          Assert.assertEquals("one frequency count",3,f.getCount(1));
62          Assert.assertEquals("two frequency count",1,f.getCount(2));
63          Assert.assertEquals("three frequency count",0,f.getCount(3));
64          Assert.assertEquals("total count",4,f.getSumFreq());
65          Assert.assertEquals("zero cumulative frequency", 0, f.getCumFreq(0));
66          Assert.assertEquals("one cumulative frequency", 3,  f.getCumFreq(1));
67          Assert.assertEquals("two cumulative frequency", 4,  f.getCumFreq(2));
68          Assert.assertEquals("Integer argument cum freq",4, f.getCumFreq(Integer.valueOf(2)));
69          Assert.assertEquals("five cumulative frequency", 4,  f.getCumFreq(5));
70          Assert.assertEquals("foo cumulative frequency", 0,  f.getCumFreq("foo"));
71  
72          f.clear();
73          Assert.assertEquals("total count",0,f.getSumFreq());
74  
75          // userguide examples -------------------------------------------------------------------
76          f.addValue("one");
77          f.addValue("One");
78          f.addValue("oNe");
79          f.addValue("Z");
80          Assert.assertEquals("one cumulative frequency", 1 ,  f.getCount("one"));
81          Assert.assertEquals("Z cumulative pct", 0.5,  f.getCumPct("Z"), TOLERANCE);
82          Assert.assertEquals("z cumulative pct", 1.0,  f.getCumPct("z"), TOLERANCE);
83          Assert.assertEquals("Ot cumulative pct", 0.25,  f.getCumPct("Ot"), TOLERANCE);
84          f.clear();
85  
86          f = null;
87          Frequency f = new Frequency();
88          f.addValue(1);
89          f.addValue(Integer.valueOf(1));
90          f.addValue(Long.valueOf(1));
91          f.addValue(2);
92          f.addValue(Integer.valueOf(-1));
93          Assert.assertEquals("1 count", 3, f.getCount(1));
94          Assert.assertEquals("1 count", 3, f.getCount(Integer.valueOf(1)));
95          Assert.assertEquals("0 cum pct", 0.2, f.getCumPct(0), TOLERANCE);
96          Assert.assertEquals("1 pct", 0.6, f.getPct(Integer.valueOf(1)), TOLERANCE);
97          Assert.assertEquals("-2 cum pct", 0, f.getCumPct(-2), TOLERANCE);
98          Assert.assertEquals("10 cum pct", 1, f.getCumPct(10), TOLERANCE);
99  
100         f = null;
101         f = new Frequency(String.CASE_INSENSITIVE_ORDER);
102         f.addValue("one");
103         f.addValue("One");
104         f.addValue("oNe");
105         f.addValue("Z");
106         Assert.assertEquals("one count", 3 ,  f.getCount("one"));
107         Assert.assertEquals("Z cumulative pct -- case insensitive", 1 ,  f.getCumPct("Z"), TOLERANCE);
108         Assert.assertEquals("z cumulative pct -- case insensitive", 1 ,  f.getCumPct("z"), TOLERANCE);
109 
110         f = null;
111         f = new Frequency();
112         Assert.assertEquals(0L, f.getCount('a'));
113         Assert.assertEquals(0L, f.getCumFreq('b'));
114         TestUtils.assertEquals(Double.NaN, f.getPct('a'), 0.0);
115         TestUtils.assertEquals(Double.NaN, f.getCumPct('b'), 0.0);
116         f.addValue('a');
117         f.addValue('b');
118         f.addValue('c');
119         f.addValue('d');
120         Assert.assertEquals(1L, f.getCount('a'));
121         Assert.assertEquals(2L, f.getCumFreq('b'));
122         Assert.assertEquals(0.25, f.getPct('a'), 0.0);
123         Assert.assertEquals(0.5, f.getCumPct('b'), 0.0);
124         Assert.assertEquals(1.0, f.getCumPct('e'), 0.0);
125     }

148         Assert.assertEquals("zero cum pct",0.0,f.getCumPct(0),TOLERANCE);
149         Assert.assertEquals("foo cum pct",0,f.getCumPct("foo"),TOLERANCE);

327     public void testMerge() {
328         Assert.assertEquals(0, f.getUniqueCount());
329         f.addValue(ONE_LONG);
330         f.addValue(TWO_LONG);
331         f.addValue(ONE);
332         f.addValue(TWO);
333 
334         Assert.assertEquals(2, f.getUniqueCount());
335         Assert.assertEquals(2, f.getCount(ONE));
336         Assert.assertEquals(2, f.getCount(TWO));
337 
338         Frequency g = new Frequency();
339         g.addValue(ONE_LONG);
340         g.addValue(THREE_LONG);
341         g.addValue(THREE);
342 
343         Assert.assertEquals(2, g.getUniqueCount());
344         Assert.assertEquals(1, g.getCount(ONE));
345         Assert.assertEquals(2, g.getCount(THREE));
346 
347         f.merge(g);
348 
349         Assert.assertEquals(3, f.getUniqueCount());

352         Assert.assertEquals(2, f.getCount(THREE));

378         Assert.assertEquals(1, f.getCount(THREE));
