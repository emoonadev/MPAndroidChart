34  public class MultivariateNormalDistributionTest {
35      /**
36       * Test the ability of the distribution to report its mean value parameter.
37       */
38      @Test
39      public void testGetMean() {
40          final double[] mu = { -1.5, 2 };
41          final double[][] sigma = { { 2, -1.1 },
42                                     { -1.1, 2 } };
43          final MultivariateNormalDistribution d = new MultivariateNormalDistribution(mu, sigma);
44  
45          final double[] m = d.getMeans();
46          for (int i = 0; i < m.length; i++) {
47              Assert.assertEquals(mu[i], m[i], 0);
48          }
49      }

51      /**
52       * Test the ability of the distribution to report its covariance matrix parameter.
53       */
54      @Test
55      public void testGetCovarianceMatrix() {
56          final double[] mu = { -1.5, 2 };
57          final double[][] sigma = { { 2, -1.1 },
58                                     { -1.1, 2 } };
59          final MultivariateNormalDistribution d = new MultivariateNormalDistribution(mu, sigma);
60  
61          final RealMatrix s = d.getCovariances();
62          final int dim = d.getDimension();
63          for (int i = 0; i < dim; i++) {
64              for (int j = 0; j < dim; j++) {
65                  Assert.assertEquals(sigma[i][j], s.getEntry(i, j), 0);
66              }
67          }
68      }

70      /**
71       * Test the accuracy of sampling from the distribution.
72       */
73      @Test
74      public void testSampling() {
75          final double[] mu = { -1.5, 2 };
76          final double[][] sigma = { { 2, -1.1 },
77                                     { -1.1, 2 } };
78          final MultivariateNormalDistribution d = new MultivariateNormalDistribution(mu, sigma);
79          final MultivariateRealDistribution.Sampler sampler =
80              d.createSampler(RandomSource.create(RandomSource.WELL_19937_C, 50));
81  
82          final int n = 500000;
83          final double[][] samples = AbstractMultivariateRealDistribution.sample(n, sampler);
84  
85          final int dim = d.getDimension();
86          final double[] sampleMeans = new double[dim];
87  
88          for (int i = 0; i < samples.length; i++) {
89              for (int j = 0; j < dim; j++) {
90                  sampleMeans[j] += samples[i][j];
91              }
92          }
93  
94          final double sampledValueTolerance = 1e-2;
95          for (int j = 0; j < dim; j++) {
96              sampleMeans[j] /= samples.length;
97              Assert.assertEquals(mu[j], sampleMeans[j], sampledValueTolerance);
98          }
99  
100         final double[][] sampleSigma = new Covariance(samples).getCovarianceMatrix().getData();
101         for (int i = 0; i < dim; i++) {
102             for (int j = 0; j < dim; j++) {
103                 Assert.assertEquals(sigma[i][j], sampleSigma[i][j], sampledValueTolerance);
104             }
105         }
106     }

108     /**
109      * Test the accuracy of the distribution when calculating densities.
110      */
111     @Test
112     public void testDensities() {
113         final double[] mu = { -1.5, 2 };
114         final double[][] sigma = { { 2, -1.1 },
115                                    { -1.1, 2 } };
116         final MultivariateNormalDistribution d = new MultivariateNormalDistribution(mu, sigma);
117 
118         final double[][] testValues = { { -1.5, 2 },
119                                         { 4, 4 },
120                                         { 1.5, -2 },
121                                         { 0, 0 } };
122         final double[] densities = new double[testValues.length];
123         for (int i = 0; i < densities.length; i++) {
124             densities[i] = d.density(testValues[i]);
125         }
126 
127         // From dmvnorm function in R 2.15 CRAN package Mixtools v0.4.5
128         final double[] correctDensities = { 0.09528357207691344,
129                                             5.80932710124009e-09,
130                                             0.001387448895173267,
131                                             0.03309922090210541 };
132 
133         for (int i = 0; i < testValues.length; i++) {
134             Assert.assertEquals(correctDensities[i], densities[i], 1e-16);
135         }
136     }

138     /**
139      * Test the accuracy of the distribution when calculating densities.
140      */
141     @Test
142     public void testUnivariateDistribution() {
143         final double[] mu = { -1.5 };
144         final double[][] sigma = { { 1 } };
145 
146         final MultivariateNormalDistribution multi = new MultivariateNormalDistribution(mu, sigma);
147 
148         final NormalDistribution uni = new NormalDistribution(mu[0], sigma[0][0]);
149         final Random rng = new Random();
150         final int numCases = 100;
151         final double tol = Math.ulp(1d);
152         for (int i = 0; i < numCases; i++) {
153             final double v = rng.nextDouble() * 10 - 5;
154             Assert.assertEquals(uni.density(v), multi.density(new double[] { v }), tol);
155         }
156     }
