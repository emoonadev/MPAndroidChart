29      private double[][] testSquare = {
30              { 24.0 / 25.0, 43.0 / 25.0 },
31              { 57.0 / 25.0, 24.0 / 25.0 }
32      };

34      private double[][] testNonSquare = {
35          {  -540.0 / 625.0,  963.0 / 625.0, -216.0 / 625.0 },
36          { -1730.0 / 625.0, -744.0 / 625.0, 1008.0 / 625.0 },
37          {  -720.0 / 625.0, 1284.0 / 625.0, -288.0 / 625.0 },
38          {  -360.0 / 625.0,  192.0 / 625.0, 1756.0 / 625.0 },

41      @Test
42      public void testDimensions() {
43          checkdimensions(MatrixUtils.createRealMatrix(testSquare));
44          checkdimensions(MatrixUtils.createRealMatrix(testNonSquare));
45          checkdimensions(MatrixUtils.createRealMatrix(testNonSquare).transpose());
46      }
47  
48      private void checkdimensions(RealMatrix matrix) {
49          final int m = matrix.getRowDimension();
50          final int n = matrix.getColumnDimension();
51          BiDiagonalTransformer transformer = new BiDiagonalTransformer(matrix);
52          Assert.assertEquals(m, transformer.getU().getRowDimension());
53          Assert.assertEquals(m, transformer.getU().getColumnDimension());
54          Assert.assertEquals(m, transformer.getB().getRowDimension());
55          Assert.assertEquals(n, transformer.getB().getColumnDimension());
56          Assert.assertEquals(n, transformer.getV().getRowDimension());
57          Assert.assertEquals(n, transformer.getV().getColumnDimension());
58  
59      }

61      @Test
62      public void testAEqualUSVt() {
63          checkAEqualUSVt(MatrixUtils.createRealMatrix(testSquare));
64          checkAEqualUSVt(MatrixUtils.createRealMatrix(testNonSquare));
65          checkAEqualUSVt(MatrixUtils.createRealMatrix(testNonSquare).transpose());
66      }
67  
68      private void checkAEqualUSVt(RealMatrix matrix) {
69          BiDiagonalTransformer transformer = new BiDiagonalTransformer(matrix);
70          RealMatrix u = transformer.getU();
71          RealMatrix b = transformer.getB();
72          RealMatrix v = transformer.getV();
73          double norm = u.multiply(b).multiply(v.transpose()).subtract(matrix).getNorm();
74          Assert.assertEquals(0, norm, 1.0e-14);
75      }

78      public void testUOrthogonal() {
79          checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getU());
80          checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getU());

84      @Test
85      public void testVOrthogonal() {
86          checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getV());
87          checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getV());
88          checkOrthogonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).getV());
89      }
90  
91      private void checkOrthogonal(RealMatrix m) {
92          RealMatrix mTm = m.transpose().multiply(m);
93          RealMatrix id  = MatrixUtils.createRealIdentityMatrix(mTm.getRowDimension());
94          Assert.assertEquals(0, mTm.subtract(id).getNorm(), 1.0e-14);
95      }

97      @Test
98      public void testBBiDiagonal() {
99          checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare)).getB());
100         checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare)).getB());
101         checkBiDiagonal(new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testNonSquare).transpose()).getB());
102     }
103 
104     private void checkBiDiagonal(RealMatrix m) {
105         final int rows = m.getRowDimension();
106         final int cols = m.getColumnDimension();
107         for (int i = 0; i < rows; ++i) {
108             for (int j = 0; j < cols; ++j) {
109                 if (rows < cols) {
110                     if ((i < j) || (i > j + 1)) {
111                         Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
112                     }
113                 } else {
114                     if ((i < j - 1) || (i > j)) {
115                         Assert.assertEquals(0, m.getEntry(i, j), 1.0e-16);
116                     }
117                 }
118             }
119         }
120     }

122     @Test
123     public void testSingularMatrix() {
124        BiDiagonalTransformer transformer =
125             new BiDiagonalTransformer(MatrixUtils.createRealMatrix(new double[][] {
126                 { 1.0, 2.0, 3.0 },
127                 { 2.0, 3.0, 4.0 },
128                 { 3.0, 5.0, 7.0 }
129             }));
130        final double s3  = FastMath.sqrt(3.0);
131        final double s14 = FastMath.sqrt(14.0);
132        final double s1553 = FastMath.sqrt(1553.0);
133        RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
134            {  -1.0 / s14,  5.0 / (s3 * s14),  1.0 / s3 },
135            {  -2.0 / s14, -4.0 / (s3 * s14),  1.0 / s3 },
136            {  -3.0 / s14,  1.0 / (s3 * s14), -1.0 / s3 }
137        });
138        RealMatrix bRef = MatrixUtils.createRealMatrix(new double[][] {
139            { -s14, s1553 / s14,   0.0 },
140            {  0.0, -87 * s3 / (s14 * s1553), -s3 * s14 / s1553 },
141            {  0.0, 0.0, 0.0 }
142        });
143        RealMatrix vRef = MatrixUtils.createRealMatrix(new double[][] {
144            { 1.0,   0.0,         0.0        },
145            { 0.0,  -23 / s1553,  32 / s1553 },
146            { 0.0,  -32 / s1553, -23 / s1553 }
147        });
148 
149        // check values against known references
150        RealMatrix u = transformer.getU();
151        Assert.assertEquals(0, u.subtract(uRef).getNorm(), 1.0e-14);
152        RealMatrix b = transformer.getB();
153        Assert.assertEquals(0, b.subtract(bRef).getNorm(), 1.0e-14);
154        RealMatrix v = transformer.getV();
155        Assert.assertEquals(0, v.subtract(vRef).getNorm(), 1.0e-14);
156 
157        // check the same cached instance is returned the second time
158        Assert.assertTrue(u == transformer.getU());
159        Assert.assertTrue(b == transformer.getB());
160        Assert.assertTrue(v == transformer.getV());
161 
162     }

164     @Test
165     public void testMatricesValues() {
166        BiDiagonalTransformer transformer =
167             new BiDiagonalTransformer(MatrixUtils.createRealMatrix(testSquare));
168        final double s17 = FastMath.sqrt(17.0);
169         RealMatrix uRef = MatrixUtils.createRealMatrix(new double[][] {
170                 {  -8 / (5 * s17), 19 / (5 * s17) },
171                 { -19 / (5 * s17), -8 / (5 * s17) }
172         });
173         RealMatrix bRef = MatrixUtils.createRealMatrix(new double[][] {
174                 { -3 * s17 / 5, 32 * s17 / 85 },
175                 {      0.0,     -5 * s17 / 17 }
176         });
177         RealMatrix vRef = MatrixUtils.createRealMatrix(new double[][] {
178                 { 1.0,  0.0 },
179                 { 0.0, -1.0 }
180         });
181 
182         // check values against known references
183         RealMatrix u = transformer.getU();
184         Assert.assertEquals(0, u.subtract(uRef).getNorm(), 1.0e-14);
185         RealMatrix b = transformer.getB();
186         Assert.assertEquals(0, b.subtract(bRef).getNorm(), 1.0e-14);
187         RealMatrix v = transformer.getV();
188         Assert.assertEquals(0, v.subtract(vRef).getNorm(), 1.0e-14);
189 
190         // check the same cached instance is returned the second time
191         Assert.assertTrue(u == transformer.getU());
192         Assert.assertTrue(b == transformer.getB());
193         Assert.assertTrue(v == transformer.getV());
194 
195     }
