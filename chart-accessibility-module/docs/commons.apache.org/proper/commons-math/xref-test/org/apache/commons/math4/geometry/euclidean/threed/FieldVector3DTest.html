131     @Test
132     public void testEquals() {
133         FieldVector3D<DerivativeStructure> u1 = createVector(1, 2, 3, 3);
134         FieldVector3D<DerivativeStructure> v  = createVector(1, 2, 3 + 10 * Precision.EPSILON, 3);
135         Assert.assertTrue(u1.equals(u1));
136         Assert.assertTrue(u1.equals(new FieldVector3D<DerivativeStructure>(new DerivativeStructure(3, 1, 0, 1.0),
137                                                    new DerivativeStructure(3, 1, 1, 2.0),
138                                                    new DerivativeStructure(3, 1, 2, 3.0))));
139         Assert.assertFalse(u1.equals(new FieldVector3D<DerivativeStructure>(new DerivativeStructure(3, 1, 1.0),
140                                                     new DerivativeStructure(3, 1, 1, 2.0),
141                                                     new DerivativeStructure(3, 1, 2, 3.0))));
142         Assert.assertFalse(u1.equals(new FieldVector3D<DerivativeStructure>(new DerivativeStructure(3, 1, 0, 1.0),
143                                                     new DerivativeStructure(3, 1, 2.0),
144                                                     new DerivativeStructure(3, 1, 2, 3.0))));
145         Assert.assertFalse(u1.equals(new FieldVector3D<DerivativeStructure>(new DerivativeStructure(3, 1, 0, 1.0),
146                                                     new DerivativeStructure(3, 1, 1, 2.0),
147                                                     new DerivativeStructure(3, 1, 3.0))));
148         Assert.assertFalse(u1.equals(v));
149         Assert.assertFalse(u1.equals(u1.toVector3D()));
150         Assert.assertTrue(createVector(0, Double.NaN, 0, 3).equals(createVector(0, 0, Double.NaN, 3)));
151     }
