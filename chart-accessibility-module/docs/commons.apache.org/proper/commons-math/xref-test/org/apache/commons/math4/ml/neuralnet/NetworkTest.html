62      @Test
63      public void testDeleteLink() {
64          final FeatureInitializer[] initArray = { init };
65          final Network net = new NeuronSquareMesh2D(2, false,
66                                                     2, false,
67                                                     SquareNeighbourhood.VON_NEUMANN,
68                                                     initArray).getNetwork();
69          Collection<Neuron> neighbours;
70  
71          // Delete 0 --> 1.
72          net.deleteLink(net.getNeuron(0),
73                         net.getNeuron(1));
74  
75          // Link from 0 to 1 was deleted.
76          neighbours = net.getNeighbours(net.getNeuron(0));
77          Assert.assertFalse(neighbours.contains(net.getNeuron(1)));
78          // Link from 1 to 0 still exists.
79          neighbours = net.getNeighbours(net.getNeuron(1));
80          Assert.assertTrue(neighbours.contains(net.getNeuron(0)));
81      }

95                                                     2, false,
96                                                     SquareNeighbourhood.VON_NEUMANN,

98  
99          Assert.assertEquals(2, net.getNeighbours(net.getNeuron(0)).size());
100         Assert.assertEquals(2, net.getNeighbours(net.getNeuron(3)).size());
101 
102         // Delete neuron 1.
103         net.deleteNeuron(net.getNeuron(1));
104 
105         try {
106             net.getNeuron(1);
107         } catch (NoSuchElementException expected) {}
108 
109         Assert.assertEquals(1, net.getNeighbours(net.getNeuron(0)).size());
110         Assert.assertEquals(1, net.getNeighbours(net.getNeuron(3)).size());

113     @Test
114     public void testIterationOrder() {
115         final FeatureInitializer[] initArray = { init };
116         final Network net = new NeuronSquareMesh2D(4, false,
117                                                    3, true,
118                                                    SquareNeighbourhood.VON_NEUMANN,
119                                                    initArray).getNetwork();
120 
121         // Check that the comparator provides a specific order.
122         boolean isUnspecifiedOrder = false;
123         long previousId = Long.MIN_VALUE;
124         for (Neuron n : net.getNeurons(new Network.NeuronIdentifierComparator())) {
125             final long currentId = n.getIdentifier();
126             if (currentId < previousId) {
127                 isUnspecifiedOrder = true;
128                 break;
129             }
130             previousId = currentId;
131         }
132         Assert.assertFalse(isUnspecifiedOrder);
133     }

143     @Test
144     public void testCopy() {
145         final FeatureInitializer[] initArray = { init };
146         final Network net = new NeuronSquareMesh2D(2, false,
147                                                    2, false,
148                                                    SquareNeighbourhood.VON_NEUMANN,
149                                                    initArray).getNetwork();
150 
151         final Network copy = net.copy();
152 
153         final Neuron netNeuron0 = net.getNeuron(0);
154         final Neuron copyNeuron0 = copy.getNeuron(0);
155         final Neuron netNeuron1 = net.getNeuron(1);
156         final Neuron copyNeuron1 = copy.getNeuron(1);
157         Collection<Neuron> netNeighbours;
158         Collection<Neuron> copyNeighbours;
159 
160         // Check that both networks have the same connections.
161         netNeighbours = net.getNeighbours(netNeuron0);
162         copyNeighbours = copy.getNeighbours(copyNeuron0);
163         Assert.assertTrue(netNeighbours.contains(netNeuron1));
164         Assert.assertTrue(copyNeighbours.contains(copyNeuron1));
165 
166         // Delete neuron 1 from original.
167         net.deleteNeuron(netNeuron1);
168 
169         // Check that the networks now differ.
170         netNeighbours = net.getNeighbours(netNeuron0);
171         copyNeighbours = copy.getNeighbours(copyNeuron0);
172         Assert.assertFalse(netNeighbours.contains(netNeuron1));
173         Assert.assertTrue(copyNeighbours.contains(copyNeuron1));
174     }

176     @Test
177     public void testSerialize()
178         throws IOException,
179                ClassNotFoundException {
180         final FeatureInitializer[] initArray = { init };
181         final Network out = new NeuronSquareMesh2D(4, false,
182                                                    3, true,
183                                                    SquareNeighbourhood.VON_NEUMANN,
184                                                    initArray).getNetwork();
185 
186         final ByteArrayOutputStream bos = new ByteArrayOutputStream();
187         final ObjectOutputStream oos = new ObjectOutputStream(bos);
188         oos.writeObject(out);
189 
190         final ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
191         final ObjectInputStream ois = new ObjectInputStream(bis);
192         final Network in = (Network) ois.readObject();
193 
194         for (Neuron nOut : out) {
195             final Neuron nIn = in.getNeuron(nOut.getIdentifier());
196 
197             // Same values.
198             final double[] outF = nOut.getFeatures();
199             final double[] inF = nIn.getFeatures();
200             Assert.assertEquals(outF.length, inF.length);
201             for (int i = 0; i < outF.length; i++) {
202                 Assert.assertEquals(outF[i], inF[i], 0d);
203             }
204 
205             // Same neighbours.
206             final Collection<Neuron> outNeighbours = out.getNeighbours(nOut);
207             final Collection<Neuron> inNeighbours = in.getNeighbours(nIn);
208             Assert.assertEquals(outNeighbours.size(), inNeighbours.size());
209             for (Neuron oN : outNeighbours) {
210                 Assert.assertTrue(inNeighbours.contains(in.getNeuron(oN.getIdentifier())));
211             }
212         }
213     }
