29      private void testRegularizedGamma(double expected, double a, double x) {
30          double actualP = Gamma.regularizedGammaP(a, x);
31          double actualQ = Gamma.regularizedGammaQ(a, x);
32          TestUtils.assertEquals(expected, actualP, 10e-15);
33          TestUtils.assertEquals(actualP, 1.0 - actualQ, 10e-15);
34      }

36      private void testLogGamma(double expected, double x) {
37          double actual = Gamma.logGamma(x);
38          TestUtils.assertEquals(expected, actual, 10e-15);
39      }

944     @Test
945     public void testGamma() {
946 
947         for (int i = 0; i < GAMMA_REF.length; i++) {
948             final double[] ref = GAMMA_REF[i];
949             final double x = ref[0];
950             final double expected = ref[1];
951             final double actual = Gamma.gamma(x);
952             final double absX = FastMath.abs(x);
953             final int ulps;
954             if (absX <= 8.0) {
955                 ulps = 3;
956             } else if (absX <= 20.0) {
957                 ulps = 5;
958             } else if (absX <= 30.0) {
959                 ulps = 50;
960             } else if (absX <= 50.0) {
961                 ulps = 180;
962             } else {
963                 ulps = 500;
964             }
965             final double tol = ulps * FastMath.ulp(expected);
966             Assert.assertEquals(Double.toString(x), expected, actual, tol);
967         }
968     }

981         // see: https://en.wikipedia.org/wiki/Gamma_function
982 
983         double previousGamma = Gamma.gamma(-18.5);
984         for (double x = -19.5; x > -25; x -= 1.0) {
985             double gamma = Gamma.gamma(x);
986             Assert.assertEquals(  (int) FastMath.signum(previousGamma),
987                                 - (int) FastMath.signum(gamma));

989             previousGamma = gamma;
