17  package org.apache.commons.math4.fitting;
18  
19  import java.util.Random;
20  
21  import org.apache.commons.math4.TestUtils;
22  import org.apache.commons.math4.analysis.ParametricUnivariateFunction;
23  import org.apache.commons.math4.analysis.polynomials.PolynomialFunction;
24  import org.apache.commons.math4.distribution.RealDistribution;
25  import org.apache.commons.math4.distribution.UniformRealDistribution;
26  import org.apache.commons.math4.fitting.SimpleCurveFitter;
27  import org.apache.commons.math4.fitting.WeightedObservedPoints;
28  import org.apache.commons.rng.RandomSource;
29  import org.junit.Test;
30  
31  /**
32   * Test for class {@link SimpleCurveFitter}.
33   */
34  public class SimpleCurveFitterTest {
35      @Test
36      public void testPolynomialFit() {
37          final Random randomizer = new Random(53882150042L);
38          final RealDistribution.Sampler rng
39              = new UniformRealDistribution(-100, 100).createSampler(RandomSource.create(RandomSource.WELL_512_A,
40                                                                                         64925784253L));
41  
42          final double[] coeff = { 12.9, -3.4, 2.1 }; // 12.9 - 3.4 x + 2.1 x^2
43          final PolynomialFunction f = new PolynomialFunction(coeff);
44  
45          // Collect data from a known polynomial.
46          final WeightedObservedPoints obs = new WeightedObservedPoints();
47          for (int i = 0; i < 100; i++) {
48              final double x = rng.sample();
49              obs.add(x, f.value(x) + 0.1 * randomizer.nextGaussian());
50          }
51  
52          final ParametricUnivariateFunction function = new PolynomialFunction.Parametric();
53          // Start fit from initial guesses that are far from the optimal values.
54          final SimpleCurveFitter fitter
55              = SimpleCurveFitter.create(function,
56                                         new double[] { -1e20, 3e15, -5e25 });
57          final double[] best = fitter.fit(obs.toList());
58  
59          TestUtils.assertEquals("best != coeff", coeff, best, 2e-2);
60      }
61  }
