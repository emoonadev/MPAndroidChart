17  package org.apache.commons.math4.geometry.euclidean.twod;
18  
19  import org.apache.commons.math4.geometry.euclidean.twod.Line;
20  import org.apache.commons.math4.geometry.euclidean.twod.Segment;
21  import org.apache.commons.math4.geometry.euclidean.twod.Vector2D;
22  import org.apache.commons.math4.util.FastMath;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class SegmentTest {
27  
28      @Test
29      public void testDistance() {
30          Vector2D start = new Vector2D(2, 2);
31          Vector2D end = new Vector2D(-2, -2);
32          Segment segment = new Segment(start, end, new Line(start, end, 1.0e-10));
33  
34          // distance to center of segment
35          Assert.assertEquals(FastMath.sqrt(2), segment.distance(new Vector2D(1, -1)), 1.0e-10);
36  
37          // distance a point on segment
38          Assert.assertEquals(FastMath.sin(Math.PI / 4.0), segment.distance(new Vector2D(0, -1)), 1.0e-10);
39  
40          // distance to end point
41          Assert.assertEquals(FastMath.sqrt(8), segment.distance(new Vector2D(0, 4)), 1.0e-10);
42  
43          // distance to start point
44          Assert.assertEquals(FastMath.sqrt(8), segment.distance(new Vector2D(0, -4)), 1.0e-10);
45      }
46  }
