17  package org.apache.commons.math4.stat.descriptive;
18  
19  
20  import java.util.Locale;
21  
22  import org.apache.commons.math4.TestUtils;
23  import org.apache.commons.math4.stat.descriptive.StatisticalSummaryValues;
24  import org.junit.Assert;
25  import org.junit.Test;
26  /**
27   * Test cases for the {@link StatisticalSummaryValues} class.
28   *
29   */
30  
31  public final class StatisticalSummaryValuesTest {
32  
33      @Test
34      public void testSerialization() {
35          StatisticalSummaryValues u = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
36          TestUtils.checkSerializedEquality(u);
37          StatisticalSummaryValues t = (StatisticalSummaryValues) TestUtils.serializeAndRecover(u);
38          verifyEquality(u, t);
39      }
40  
41      @Test
42      public void testEqualsAndHashCode() {
43          StatisticalSummaryValues u  = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
44          StatisticalSummaryValues t = null;
45          Assert.assertTrue("reflexive", u.equals(u));
46          Assert.assertFalse("non-null compared to null", u.equals(t));
47          Assert.assertFalse("wrong type", u.equals(Double.valueOf(0)));
48          t = new StatisticalSummaryValues(1, 2, 3, 4, 5, 6);
49          Assert.assertTrue("instances with same data should be equal", t.equals(u));
50          Assert.assertEquals("hash code", u.hashCode(), t.hashCode());
51  
52          u = new StatisticalSummaryValues(Double.NaN, 2, 3, 4, 5, 6);
53          t = new StatisticalSummaryValues(1, Double.NaN, 3, 4, 5, 6);
54          Assert.assertFalse("instances based on different data should be different",
55                  (u.equals(t) ||t.equals(u)));
56      }
57  
58      private void verifyEquality(StatisticalSummaryValues s, StatisticalSummaryValues u) {
59          Assert.assertEquals("N",s.getN(),u.getN());
60          TestUtils.assertEquals("sum",s.getSum(),u.getSum(), 0);
61          TestUtils.assertEquals("var",s.getVariance(),u.getVariance(), 0);
62          TestUtils.assertEquals("std",s.getStandardDeviation(),u.getStandardDeviation(), 0);
63          TestUtils.assertEquals("mean",s.getMean(),u.getMean(), 0);
64          TestUtils.assertEquals("min",s.getMin(),u.getMin(), 0);
65          TestUtils.assertEquals("max",s.getMax(),u.getMax(), 0);
66      }
67  
68      @Test
69      public void testToString() {
70          StatisticalSummaryValues u  = new StatisticalSummaryValues(4.5, 16, 10, 5, 4, 45);
71          Locale d = Locale.getDefault();
72          Locale.setDefault(Locale.US);
73          Assert.assertEquals("StatisticalSummaryValues:\n" +
74                       "n: 10\n" +
75                       "min: 4.0\n" +
76                       "max: 5.0\n" +
77                       "mean: 4.5\n" +
78                       "std dev: 4.0\n" +
79                       "variance: 16.0\n" +
80                       "sum: 45.0\n",  u.toString());
81          Locale.setDefault(d);
82      }
83  }
