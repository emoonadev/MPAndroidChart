17  package org.apache.commons.math4.genetics;
18  
19  
20  import java.util.Iterator;
21  
22  import org.apache.commons.math4.genetics.Chromosome;
23  import org.apache.commons.math4.genetics.FixedGenerationCount;
24  import org.apache.commons.math4.genetics.Population;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  public class FixedGenerationCountTest {
29  
30      @Test
31      public void testIsSatisfied() {
32          FixedGenerationCount fgc = new FixedGenerationCount(20);
33  
34          int cnt = 0;
35          Population pop = new Population() {
36              public void addChromosome(Chromosome chromosome) {
37                  // unimportant
38              }
39              public Chromosome getFittestChromosome() {
40                  // unimportant
41                  return null;
42              }
43              public int getPopulationLimit() {
44                  // unimportant
45                  return 0;
46              }
47              public int getPopulationSize() {
48                  // unimportant
49                  return 0;
50              }
51              public Population nextGeneration() {
52                  // unimportant
53                  return null;
54              }
55              public Iterator<Chromosome> iterator() {
56                  // unimportant
57                  return null;
58              }
59          };
60  
61          while (!fgc.isSatisfied(pop)) {
62              cnt++;
63          }
64          Assert.assertEquals(20, cnt);
65      }
66  
67  }
