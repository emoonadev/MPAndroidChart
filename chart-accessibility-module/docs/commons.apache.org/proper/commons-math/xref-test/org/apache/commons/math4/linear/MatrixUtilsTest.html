208     /**
209      * Verifies that the matrix is an identity matrix
210      */
211     protected void checkIdentityMatrix(RealMatrix m) {
212         for (int i = 0; i < m.getRowDimension(); i++) {
213             for (int j =0; j < m.getColumnDimension(); j++) {
214                 if (i == j) {
215                     Assert.assertEquals(m.getEntry(i, j), 1d, 0);
216                 } else {
217                     Assert.assertEquals(m.getEntry(i, j), 0d, 0);
218                 }
219             }
220         }
221     }

290     public static final Fraction[][] asFraction(double[][] data) {
291         Fraction d[][] = new Fraction[data.length][];
292         try {
293             for (int i = 0; i < data.length; ++i) {
294                 double[] dataI = data[i];
295                 Fraction[] dI  = new Fraction[dataI.length];
296                 for (int j = 0; j < dataI.length; ++j) {
297                     dI[j] = new Fraction(dataI[j]);
298                 }
299                 d[i] = dI;
300             }
301         } catch (FractionConversionException fce) {
302             Assert.fail(fce.getMessage());
303         }
304         return d;
305     }
306 
307     public static final Fraction[] asFraction(double[] data) {
308         Fraction d[] = new Fraction[data.length];
309         try {
310             for (int i = 0; i < data.length; ++i) {
311                 d[i] = new Fraction(data[i]);
312             }
313         } catch (FractionConversionException fce) {
314             Assert.fail(fce.getMessage());
315         }
316         return d;
317     }

348     @Test
349     public void testBlockInverse() {
350         final double[][] data = {
351             { -1, 0, 123, 4 },
352             { -56, 78.9, -0.1, -23.4 },
353             { 5.67, 8, -9, 1011 },
354             { 12, 345, -67.8, 9 },
355         };
356 
357         final RealMatrix m = new Array2DRowRealMatrix(data);
358         final int len = data.length;
359         final double tol = 1e-14;
360 
361         for (int splitIndex = 0; splitIndex < 3; splitIndex++) {
362             final RealMatrix mInv = MatrixUtils.blockInverse(m, splitIndex);
363             final RealMatrix id = m.multiply(mInv);
364 
365             // Check that we recovered the identity matrix.
366             for (int i = 0; i < len; i++) {
367                 for (int j = 0; j < len; j++) {
368                     final double entry = id.getEntry(i, j);
369                     if (i == j) {
370                         Assert.assertEquals("[" + i + "][" + j + "]",
371                                             1, entry, tol);
372                     } else {
373                         Assert.assertEquals("[" + i + "][" + j + "]",
374                                             0, entry, tol);
375                     }
376                 }
377             }
378         }
379     }

414         final double eps = 1e-4;
415 
416         final double[][] dataSym1 = {
417             { 1,   1, 1.00009 },
418             { 1,   1, 1       },
419             { 1.0, 1, 1       },

422         final double[][] dataSym2 = {
423             { 1,   1, 0.99990 },
424             { 1,   1, 1       },
425             { 1.0, 1, 1       },
426         };
427         Assert.assertTrue(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataSym2), eps));
428 
429         final double[][] dataNonSym1 = {
430             { 1,   1, 1.00011 },
431             { 1,   1, 1       },
432             { 1.0, 1, 1       },
433         };
434         Assert.assertFalse(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataNonSym1), eps));

436             { 1,   1, 0.99989 },
437             { 1,   1, 1       },
438             { 1.0, 1, 1       },
439         };
440         Assert.assertFalse(MatrixUtils.isSymmetric(MatrixUtils.createRealMatrix(dataNonSym2), eps));
