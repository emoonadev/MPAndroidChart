62      private Map<Integer, Double> generate() {
63          Map<Integer, Double> map = new HashMap<Integer, Double>();
64          Random r = new Random();
65          for (int i = 0; i < 2000; ++i)
66              map.put(r.nextInt(), r.nextDouble());
67          return map;
68      }

70      private OpenIntToDoubleHashMap createFromJavaMap() {
71          OpenIntToDoubleHashMap map = new OpenIntToDoubleHashMap();
72          for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
73              map.put(mapEntry.getKey(), mapEntry.getValue());
74          }
75          return map;
76      }

100     private void assertPutAndGet(OpenIntToDoubleHashMap map, int mapSize,
101             Set<Integer> keysInMap) {
102         Assert.assertEquals(mapSize, map.size());
103         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
104             map.put(mapEntry.getKey(), mapEntry.getValue());
105             if (!keysInMap.contains(mapEntry.getKey()))
106                 ++mapSize;
107             Assert.assertEquals(mapSize, map.size());
108             Assert.assertTrue(Precision.equals(mapEntry.getValue(), map.get(mapEntry.getKey()), 1));
109         }
110     }

205     /**
206      * Returns a map with at least 100 elements where each element is absent from javaMap.
207      */
208     private Map<Integer, Double> generateAbsent() {
209         Map<Integer, Double> generated = new HashMap<Integer, Double>();
210         do {
211             generated.putAll(generate());
212             for (Integer key : javaMap.keySet())
213                 generated.remove(key);
214         } while (generated.size() < 100);
215         return generated;
216     }

228     @Test
229     public void testContainsKey() {
230         OpenIntToDoubleHashMap map = createFromJavaMap();
231         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
232             Assert.assertTrue(map.containsKey(mapEntry.getKey()));
233         }
234         for (Map.Entry<Integer, Double> mapEntry : generateAbsent().entrySet()) {
235             Assert.assertFalse(map.containsKey(mapEntry.getKey()));
236         }
237         for (Map.Entry<Integer, Double> mapEntry : javaMap.entrySet()) {
238             int key = mapEntry.getKey();
239             Assert.assertTrue(map.containsKey(key));
240             map.remove(key);
241             Assert.assertFalse(map.containsKey(key));
242         }
243     }

245     @Test
246     public void testIterator() {
247         OpenIntToDoubleHashMap map = createFromJavaMap();
248         OpenIntToDoubleHashMap.Iterator iterator = map.iterator();
249         for (int i = 0; i < map.size(); ++i) {
250             Assert.assertTrue(iterator.hasNext());
251             iterator.advance();
252             int key = iterator.key();
253             Assert.assertTrue(map.containsKey(key));
254             Assert.assertEquals(javaMap.get(key), map.get(key), 0);
255             Assert.assertEquals(javaMap.get(key), iterator.value(), 0);
256             Assert.assertTrue(javaMap.containsKey(key));
257         }
258         Assert.assertFalse(iterator.hasNext());
259         try {
260             iterator.advance();
261             Assert.fail("an exception should have been thrown");
262         } catch (NoSuchElementException nsee) {
263             // expected
264         }
265     }

280     /**
281      * Regression test for a bug in findInsertionIndex where the hashing in the second probing
282      * loop was inconsistent with the first causing duplicate keys after the right sequence
283      * of puts and removes.
284      */
285     @Test
286     public void testPutKeysWithCollisions() {
287         OpenIntToDoubleHashMap map = new OpenIntToDoubleHashMap();
288         int key1 = -1996012590;
289         double value1 = 1.0;
290         map.put(key1, value1);
291         int key2 = 835099822;
292         map.put(key2, value1);
293         int key3 = 1008859686;
294         map.put(key3, value1);
295         Assert.assertTrue(Precision.equals(value1, map.get(key3), 1));
296         Assert.assertEquals(3, map.size());
297 
298         map.remove(key2);
299         double value2 = 2.0;
300         map.put(key3, value2);
301         Assert.assertTrue(Precision.equals(value2, map.get(key3), 1));
302         Assert.assertEquals(2, map.size());
303     }
