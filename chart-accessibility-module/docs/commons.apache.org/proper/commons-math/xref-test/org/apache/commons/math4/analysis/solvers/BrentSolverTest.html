38  /**
39   * Test case for {@link BrentSolver Brent} solver.
40   * Because Brent-Dekker is guaranteed to converge in less than the default
41   * maximum iteration count due to bisection fallback, it is quite hard to
42   * debug. I include measured iteration counts plus one in order to detect
43   * regressions. On average Brent-Dekker should use 4..5 iterations for the
44   * default absolute accuracy of 10E-8 for sinus and the quintic function around
45   * zero, and 5..10 iterations for the other zeros.
46   *
47   */
48  public final class BrentSolverTest {
49      @Test
50      public void testSinZero() {
51          // The sinus function is behaved well around the root at pi. The second
52          // order derivative is zero, which means linar approximating methods will
53          // still converge quadratically.
54          UnivariateFunction f = new Sin();
55          double result;
56          UnivariateSolver solver = new BrentSolver();
57          // Somewhat benign interval. The function is monotone.
58          result = solver.solve(100, f, 3, 4);
59          // System.out.println(
60          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
61          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
62          Assert.assertTrue(solver.getEvaluations() <= 7);
63          // Larger and somewhat less benign interval. The function is grows first.
64          result = solver.solve(100, f, 1, 4);
65          // System.out.println(
66          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
67          Assert.assertEquals(result, FastMath.PI, solver.getAbsoluteAccuracy());
68          Assert.assertTrue(solver.getEvaluations() <= 8);
69      }

71      @Test
72      public void testQuinticZero() {
73          // The quintic function has zeros at 0, +-0.5 and +-1.
74          // Around the root of 0 the function is well behaved, with a second derivative
75          // of zero a 0.
76          // The other roots are less well to find, in particular the root at 1, because
77          // the function grows fast for x>1.
78          // The function has extrema (first derivative is zero) at 0.27195613 and 0.82221643,
79          // intervals containing these values are harder for the solvers.
80          UnivariateFunction f = new QuinticFunction();
81          double result;
82          // Brent-Dekker solver.
83          UnivariateSolver solver = new BrentSolver();
84          // Symmetric bracket around 0. Test whether solvers can handle hitting
85          // the root in the first iteration.
86          result = solver.solve(100, f, -0.2, 0.2);
87          //System.out.println(
88          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
89          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
90          Assert.assertTrue(solver.getEvaluations() <= 3);
91          // 1 iterations on i586 JDK 1.4.1.
92          // Asymmetric bracket around 0, just for fun. Contains extremum.
93          result = solver.solve(100, f, -0.1, 0.3);
94          //System.out.println(
95          //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
96          Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
97          // 5 iterations on i586 JDK 1.4.1.
98          Assert.assertTrue(solver.getEvaluations() <= 7);
99          // Large bracket around 0. Contains two extrema.
100         result = solver.solve(100, f, -0.3, 0.45);
101         //System.out.println(
102         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
103         Assert.assertEquals(result, 0, solver.getAbsoluteAccuracy());
104         // 6 iterations on i586 JDK 1.4.1.
105         Assert.assertTrue(solver.getEvaluations() <= 8);
106         // Benign bracket around 0.5, function is monotonous.
107         result = solver.solve(100, f, 0.3, 0.7);
108         //System.out.println(
109         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
110         Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
111         // 6 iterations on i586 JDK 1.4.1.
112         Assert.assertTrue(solver.getEvaluations() <= 9);
113         // Less benign bracket around 0.5, contains one extremum.
114         result = solver.solve(100, f, 0.2, 0.6);
115         // System.out.println(
116         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
117         Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
118         Assert.assertTrue(solver.getEvaluations() <= 10);
119         // Large, less benign bracket around 0.5, contains both extrema.
120         result = solver.solve(100, f, 0.05, 0.95);
121         //System.out.println(
122         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
123         Assert.assertEquals(result, 0.5, solver.getAbsoluteAccuracy());
124         Assert.assertTrue(solver.getEvaluations() <= 11);
125         // Relatively benign bracket around 1, function is monotonous. Fast growth for x>1
126         // is still a problem.
127         result = solver.solve(100, f, 0.85, 1.25);
128         //System.out.println(
129         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
130         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
131         Assert.assertTrue(solver.getEvaluations() <= 11);
132         // Less benign bracket around 1 with extremum.
133         result = solver.solve(100, f, 0.8, 1.2);
134         //System.out.println(
135         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
136         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
137         Assert.assertTrue(solver.getEvaluations() <= 11);
138         // Large bracket around 1. Monotonous.
139         result = solver.solve(100, f, 0.85, 1.75);
140         //System.out.println(
141         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
142         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
143         Assert.assertTrue(solver.getEvaluations() <= 13);
144         // Large bracket around 1. Interval contains extremum.
145         result = solver.solve(100, f, 0.55, 1.45);
146         //System.out.println(
147         //    "Root: " + result + " Evaluations: " + solver.getEvaluations());
148         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
149         Assert.assertTrue(solver.getEvaluations() <= 10);
150         // Very large bracket around 1 for testing fast growth behaviour.
151         result = solver.solve(100, f, 0.85, 5);
152         //System.out.println(
153        //     "Root: " + result + " Evaluations: " + solver.getEvaluations());
154         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
155         Assert.assertTrue(solver.getEvaluations() <= 15);
156 
157         try {
158             result = solver.solve(5, f, 0.85, 5);
159             Assert.fail("Expected TooManyEvaluationsException");
160         } catch (TooManyEvaluationsException e) {
161             // Expected.
162         }
163     }

165     @Test
166     public void testRootEndpoints() {
167         UnivariateFunction f = new Sin();
168         BrentSolver solver = new BrentSolver();
169 
170         // endpoint is root
171         double result = solver.solve(100, f, FastMath.PI, 4);
172         Assert.assertEquals(FastMath.PI, result, solver.getAbsoluteAccuracy());
173 
174         result = solver.solve(100, f, 3, FastMath.PI);
175         Assert.assertEquals(FastMath.PI, result, solver.getAbsoluteAccuracy());
176 
177         result = solver.solve(100, f, FastMath.PI, 4, 3.5);
178         Assert.assertEquals(FastMath.PI, result, solver.getAbsoluteAccuracy());
179 
180         result = solver.solve(100, f, 3, FastMath.PI, 3.07);
181         Assert.assertEquals(FastMath.PI, result, solver.getAbsoluteAccuracy());
182     }

202             Assert.fail("Expecting NoBracketingException - non-bracketing");
203         } catch (NoBracketingException ex) {
204             // expected
205         }

208     @Test
209     public void testInitialGuess() {
210         MonitoredFunction f = new MonitoredFunction(new QuinticFunction());
211         BrentSolver solver = new BrentSolver();
212         double result;
213 
214         // no guess
215         result = solver.solve(100, f, 0.6, 7.0);
216         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
217         int referenceCallsCount = f.getCallsCount();
218         Assert.assertTrue(referenceCallsCount >= 13);
219 
220         // invalid guess (it *is* a root, but outside of the range)
221         try {
222           result = solver.solve(100, f, 0.6, 7.0, 0.0);
223           Assert.fail("a NumberIsTooLargeException was expected");
224         } catch (NumberIsTooLargeException iae) {
225             // expected behaviour
226         }
227 
228         // bad guess
229         f.setCallsCount(0);
230         result = solver.solve(100, f, 0.6, 7.0, 0.61);
231         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
232         Assert.assertTrue(f.getCallsCount() > referenceCallsCount);
233 
234         // good guess
235         f.setCallsCount(0);
236         result = solver.solve(100, f, 0.6, 7.0, 0.999999);
237         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
238         Assert.assertTrue(f.getCallsCount() < referenceCallsCount);
239 
240         // perfect guess
241         f.setCallsCount(0);
242         result = solver.solve(100, f, 0.6, 7.0, 1.0);
243         Assert.assertEquals(result, 1.0, solver.getAbsoluteAccuracy());
244         Assert.assertEquals(1, solver.getEvaluations());
245         Assert.assertEquals(1, f.getCallsCount());
246     }

248     @Test
249     public void testMath832() {
250         final UnivariateFunction f = new UnivariateFunction() {
251                 private final UnivariateDifferentiableFunction sqrt = new Sqrt();
252                 private final UnivariateDifferentiableFunction inv = new Inverse();
253                 private final UnivariateDifferentiableFunction func
254                     = FunctionUtils.add(FunctionUtils.multiply(new Constant(1e2), sqrt),
255                                         FunctionUtils.multiply(new Constant(1e6), inv),
256                                         FunctionUtils.multiply(new Constant(1e4),
257                                                                FunctionUtils.compose(inv, sqrt)));
258 
259                 public double value(double x) {
260                     return func.value(new DerivativeStructure(1, 1, 0, x)).getPartialDerivative(1);
261                 }
262 
263             };
264 
265         BrentSolver solver = new BrentSolver();
266         final double result = solver.solve(99, f, 1, 1e30, 1 + 1e-10);
267         Assert.assertEquals(804.93558250, result, 1e-8);
268     }
