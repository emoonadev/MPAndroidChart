17  package org.apache.commons.math4.genetics;
18

20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.apache.commons.math4.genetics.Chromosome;
24  import org.apache.commons.math4.genetics.ListPopulation;
25  import org.apache.commons.math4.genetics.Population;
26  import org.junit.Assert;
27  import org.junit.Test;
28  
29  public class ChromosomeTest {
30  
31      @Test
32      public void testCompareTo() {
33          Chromosome c1 = new Chromosome() {
34              public double fitness() {
35                  return 0;
36              }
37          };
38          Chromosome c2 = new Chromosome() {
39              public double fitness() {
40                  return 10;
41              }
42          };
43          Chromosome c3 = new Chromosome() {
44              public double fitness() {
45                  return 10;
46              }
47          };
48  
49          Assert.assertTrue(c1.compareTo(c2) < 0);
50          Assert.assertTrue(c2.compareTo(c1) > 0);
51          Assert.assertEquals(0,c3.compareTo(c2));
52          Assert.assertEquals(0,c2.compareTo(c3));
53      }
54  
55      private abstract static class DummyChromosome extends Chromosome {
56          private final int repr;
57  
58          public DummyChromosome(final int repr) {
59              this.repr = repr;
60          }
61          @Override
62          protected boolean isSame(Chromosome another) {
63              return ((DummyChromosome) another).repr == repr;
64          }
65      }
66  
67      @Test
68      public void testFindSameChromosome() {
69          Chromosome c1 = new DummyChromosome(1) {
70              public double fitness() {
71                  return 1;
72              }
73          };
74          Chromosome c2 = new DummyChromosome(2) {
75              public double fitness() {
76                  return 2;
77              }
78          };
79          Chromosome c3 = new DummyChromosome(3) {
80              public double fitness() {
81                  return 3;
82              }
83          };
84          Chromosome c4 = new DummyChromosome(1) {
85              public double fitness() {
86                  return 5;
87              }
88          };
89          Chromosome c5 = new DummyChromosome(15) {
90              public double fitness() {
91                  return 15;
92              }
93          };
94  
95          List<Chromosome> popChr = new ArrayList<Chromosome>();
96          popChr.add(c1);
97          popChr.add(c2);
98          popChr.add(c3);
99          Population pop = new ListPopulation(popChr,3) {
100             public Population nextGeneration() {
101                 // not important
102                 return null;
103             }
104         };
105 
106         Assert.assertNull(c5.findSameChromosome(pop));
107         Assert.assertEquals(c1, c4.findSameChromosome(pop));
108 
109         c4.searchForFitnessUpdate(pop);
110         Assert.assertEquals(1, c4.getFitness(),0);
111     }
112 
113 }
