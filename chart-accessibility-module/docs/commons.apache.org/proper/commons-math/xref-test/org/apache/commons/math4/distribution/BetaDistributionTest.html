154     private void checkCumulative(double alpha, double beta, double[] x, double[] cumes) {
155         BetaDistribution d = new BetaDistribution(alpha, beta);
156         for (int i = 0; i < x.length; i++) {
157             Assert.assertEquals(cumes[i], d.cumulativeProbability(x[i]), 1e-8);
158         }
159 
160         for (int i = 1; i < x.length - 1; i++) {
161             Assert.assertEquals(x[i], d.inverseCumulativeProbability(cumes[i]), 1e-5);
162         }
163     }

318     @Test
319     public void testMomentsSampling() {
320         final UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_1024_A,
321                                                               123456789L);
322         final int numSamples = 1000;
323         for (final double alpha : alphaBetas) {
324             for (final double beta : alphaBetas) {
325                 final BetaDistribution betaDistribution = new BetaDistribution(alpha, beta);
326                 final double[] observed = AbstractRealDistribution.sample(numSamples,
327                         betaDistribution.createSampler(rng));
328                 Arrays.sort(observed);
329 
330                 final String distribution = String.format("Beta(%.2f, %.2f)", alpha, beta);
331                 Assert.assertEquals(String.format("E[%s]", distribution),
332                                     betaDistribution.getNumericalMean(),
333                                     StatUtils.mean(observed), epsilon);
334                 Assert.assertEquals(String.format("Var[%s]", distribution),
335                                     betaDistribution.getNumericalVariance(),
336                                     StatUtils.variance(observed), epsilon);
337             }
338         }
339     }

341     @Test
342     public void testGoodnessOfFit() {
343         final UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_19937_A,
344                                                               123456789L);
345         final int numSamples = 1000;
346         final double level = 0.01;
347         for (final double alpha : alphaBetas) {
348             for (final double beta : alphaBetas) {
349                 final BetaDistribution betaDistribution = new BetaDistribution(alpha, beta);
350 
351                 final RealDistribution.Sampler sampler = betaDistribution.createSampler(rng);
352                 final double[] observed = AbstractRealDistribution.sample(numSamples, sampler);
353 
354                 Assert.assertFalse("G goodness-of-fit test rejected null at alpha = " + level,
355                                    gTest(betaDistribution, observed) < level);
356                 Assert.assertFalse("KS goodness-of-fit test rejected null at alpha = " + level,
357                                    new KolmogorovSmirnovTest(RandomSource.JDK, 3448845623L).kolmogorovSmirnovTest(betaDistribution, observed) < level);
358             }
359         }
360     }

362     private double gTest(final RealDistribution expectedDistribution, final double[] values) {
363         final int numBins = values.length / 30;
364         final double[] breaks = new double[numBins];
365         for (int b = 0; b < breaks.length; b++) {
366             breaks[b] = expectedDistribution.inverseCumulativeProbability((double) b / numBins);
367         }
368 
369         final long[] observed = new long[numBins];
370         for (final double value : values) {
371             int b = 0;
372             do {
373                 b++;
374             } while (b < numBins && value >= breaks[b]);
375 
376             observed[b - 1]++;
377         }
378 
379         final double[] expected = new double[numBins];
380         Arrays.fill(expected, (double) values.length / numBins);
381 
382         return InferenceTestUtils.gTest(expected, observed);
383     }
