17  package org.apache.commons.math4.exception.util;
18  
19  import java.util.List;
20  import java.util.ArrayList;
21  
22  import org.apache.commons.math4.exception.util.ArgUtils;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  /**
27   * Test for {@link ArgUtils}.
28   *
29   */
30  public class ArgUtilsTest {
31      @Test
32      public void testFlatten() {
33          final List<Object> orig = new ArrayList<Object>();
34  
35          final Object[] struct = new Object[] {
36              new Object[] {
37                  new Object[] {
38                      create(orig),
39                      create(orig),
40                  },
41                  create(orig),
42                  new Object[] {
43                      create(orig),
44                  }
45              },
46              create(orig),
47              new Object[] {
48                  create(orig),
49                  new Object[] {
50                      create(orig),
51                      create(orig),
52                  }
53              },
54              create(orig),
55          };
56  
57          Object[] flat = ArgUtils.flatten(struct);
58          Assert.assertEquals(flat.length, orig.size());
59  
60          for (int i = 0, max = orig.size(); i < max; i++) {
61              Assert.assertEquals(orig.get(i), flat[i]);
62          }
63      }
64  
65      /**
66       * Create and store an {@code Object}.
67       *
68       * @param list List to store to.
69       * @return the stored object.
70       */
71      private Object create(List<Object> list) {
72          final Object o = new Object();
73          list.add(o);
74          return o;
75      }
76  }
