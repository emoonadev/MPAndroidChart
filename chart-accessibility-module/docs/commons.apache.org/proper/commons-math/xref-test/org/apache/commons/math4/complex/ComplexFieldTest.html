17  package org.apache.commons.math4.complex;
18  
19  import org.apache.commons.math4.TestUtils;
20  import org.apache.commons.math4.complex.Complex;
21  import org.apache.commons.math4.complex.ComplexField;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  public class ComplexFieldTest {
26  
27      @Test
28      public void testZero() {
29          Assert.assertEquals(Complex.ZERO, ComplexField.getInstance().getZero());
30      }
31  
32      @Test
33      public void testOne() {
34          Assert.assertEquals(Complex.ONE, ComplexField.getInstance().getOne());
35      }
36  
37      @Test
38      public void testSerial() {
39          // deserializing the singleton should give the singleton itself back
40          ComplexField field = ComplexField.getInstance();
41          Assert.assertTrue(field == TestUtils.serializeAndRecover(field));
42      }
43  
44  }
