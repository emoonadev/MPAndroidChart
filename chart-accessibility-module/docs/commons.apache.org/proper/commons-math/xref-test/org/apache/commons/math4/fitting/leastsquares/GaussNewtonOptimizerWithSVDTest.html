57      @Test
58      public void testMaxEvaluations() throws Exception {
59          try{
60          CircleVectorial circle = new CircleVectorial();
61          circle.addPoint( 30.0,  68.0);
62          circle.addPoint( 50.0,  -6.0);
63          circle.addPoint(110.0, -20.0);
64          circle.addPoint( 35.0,  15.0);
65          circle.addPoint( 45.0,  97.0);
66  
67          LeastSquaresProblem lsp = builder(circle)
68                  .checkerPair(new SimpleVectorValueChecker(1e-30, 1e-30))
69                  .maxIterations(Integer.MAX_VALUE)
70                  .start(new double[]{98.680, 47.345})
71                  .build();
72  
73          optimizer.optimize(lsp);
74  
75              fail(optimizer);
76          }catch (TooManyEvaluationsException e){
77              //expected
78          }
79      }

127     @Test
128     @Override
129     public void testNonInvertible() throws Exception {
130         /*  SVD can compute a solution to singular problems.
131          *  In this case the target vector, b, is not in the
132          *  span of the jacobian matrix, A. The closes point
133          *  to b on the plane spanned by A is computed.
134          */
135         LinearProblem problem = new LinearProblem(new double[][]{
136                 {1, 2, -3},
137                 {2, 1, 3},
138                 {-3, 0, -9}
139         }, new double[]{1, 1, 1});
140 
141         Optimum optimum = optimizer.optimize(problem.getBuilder().build());
142 
143         Plane span = new Plane(Vector3D.ZERO, new Vector3D(1, 2, -3), new Vector3D(2, 1, 0), TOl);
144         double expected = FastMath.abs(span.getOffset(new Vector3D(1, 1, 1)));
145         double actual = optimum.getResiduals().getNorm();
146 
147         //verify
148         Assert.assertEquals(expected, actual, TOl);
149     }
