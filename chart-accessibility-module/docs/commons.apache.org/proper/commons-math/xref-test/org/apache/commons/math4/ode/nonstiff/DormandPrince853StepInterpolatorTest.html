59    @Test
60    public void serialization()
61      throws IOException, ClassNotFoundException,
62             DimensionMismatchException, NumberIsTooSmallException,
63             MaxCountExceededException, NoBracketingException {
64  
65      TestProblem3 pb = new TestProblem3(0.9);
66      double minStep = 0;
67      double maxStep = pb.getFinalTime() - pb.getInitialTime();
68      double scalAbsoluteTolerance = 1.0e-8;
69      double scalRelativeTolerance = scalAbsoluteTolerance;
70      DormandPrince853Integrator integ = new DormandPrince853Integrator(minStep, maxStep,
71                                                                        scalAbsoluteTolerance,
72                                                                        scalRelativeTolerance);
73      integ.addStepHandler(new ContinuousOutputModel());
74      integ.integrate(pb,
75                      pb.getInitialTime(), pb.getInitialState(),
76                      pb.getFinalTime(), new double[pb.getDimension()]);
77  
78      ByteArrayOutputStream bos = new ByteArrayOutputStream();
79      ObjectOutputStream    oos = new ObjectOutputStream(bos);
80      for (StepHandler handler : integ.getStepHandlers()) {
81          oos.writeObject(handler);
82      }
83  
84      Assert.assertTrue(bos.size () > 90000);
85      Assert.assertTrue(bos.size () < 100000);
86  
87      ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
88      ObjectInputStream     ois = new ObjectInputStream(bis);
89      ContinuousOutputModel cm  = (ContinuousOutputModel) ois.readObject();
90  
91      Random random = new Random(347588535632l);
92      double maxError = 0.0;
93      for (int i = 0; i < 1000; ++i) {
94        double r = random.nextDouble();
95        double time = r * pb.getInitialTime() + (1.0 - r) * pb.getFinalTime();
96        cm.setInterpolatedTime(time);
97        double[] interpolatedY = cm.getInterpolatedState ();
98        double[] theoreticalY  = pb.computeTheoreticalState(time);
99        double dx = interpolatedY[0] - theoreticalY[0];
100       double dy = interpolatedY[1] - theoreticalY[1];
101       double error = dx * dx + dy * dy;
102       if (error > maxError) {
103         maxError = error;
104       }
105     }
106 
107     Assert.assertTrue(maxError < 2.4e-10);
108 
109   }

111   @Test
112   public void checklone()
113       throws DimensionMismatchException, NumberIsTooSmallException,
114              MaxCountExceededException, NoBracketingException {
115     TestProblem3 pb = new TestProblem3(0.9);
116     double minStep = 0;
117     double maxStep = pb.getFinalTime() - pb.getInitialTime();
118     double scalAbsoluteTolerance = 1.0e-8;
119     double scalRelativeTolerance = scalAbsoluteTolerance;
120     DormandPrince853Integrator integ = new DormandPrince853Integrator(minStep, maxStep,
121                                                                       scalAbsoluteTolerance,
122                                                                       scalRelativeTolerance);
123     integ.addStepHandler(new StepHandler() {
124         public void handleStep(StepInterpolator interpolator, boolean isLast)
125             throws MaxCountExceededException {
126             StepInterpolator cloned = interpolator.copy();
127             double tA = cloned.getPreviousTime();
128             double tB = cloned.getCurrentTime();
129             double halfStep = FastMath.abs(tB - tA) / 2;
130             Assert.assertEquals(interpolator.getPreviousTime(), tA, 1.0e-12);
131             Assert.assertEquals(interpolator.getCurrentTime(), tB, 1.0e-12);
132             for (int i = 0; i < 10; ++i) {
133                 double t = (i * tB + (9 - i) * tA) / 9;
134                 interpolator.setInterpolatedTime(t);
135                 Assert.assertTrue(FastMath.abs(cloned.getInterpolatedTime() - t) > (halfStep / 10));
136                 cloned.setInterpolatedTime(t);
137                 Assert.assertEquals(t, cloned.getInterpolatedTime(), 1.0e-12);
138                 double[] referenceState = interpolator.getInterpolatedState();
139                 double[] cloneState     = cloned.getInterpolatedState();
140                 for (int j = 0; j < referenceState.length; ++j) {
141                     Assert.assertEquals(referenceState[j], cloneState[j], 1.0e-12);
142                 }
143             }
144         }
145         public void init(double t0, double[] y0, double t) {
146         }
147     });
148     integ.integrate(pb,
149             pb.getInitialTime(), pb.getInitialState(),
150             pb.getFinalTime(), new double[pb.getDimension()]);
151 
152   }
