31  /**
32   * Test case for Romberg integrator.
33   * <p>
34   * Romberg algorithm is very fast for good behavior integrand. Test runs
35   * show that for a default relative accuracy of 1E-6, it generally takes
36   * takes less than 5 iterations for the integral to converge.
37   *
38   */
39  public final class RombergIntegratorTest {
40  
41      /**
42       * Test of integrator for the sine function.
43       */
44      @Test
45      public void testSinFunction() {
46          UnivariateFunction f = new Sin();
47          UnivariateIntegrator integrator = new RombergIntegrator();
48          double min, max, expected, result, tolerance;
49  
50          min = 0; max = FastMath.PI; expected = 2;
51          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
52          result = integrator.integrate(100, f, min, max);
53          Assert.assertTrue(integrator.getEvaluations() < 50);
54          Assert.assertTrue(integrator.getIterations()  < 10);
55          Assert.assertEquals(expected, result, tolerance);
56  
57          min = -FastMath.PI/3; max = 0; expected = -0.5;
58          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
59          result = integrator.integrate(100, f, min, max);
60          Assert.assertTrue(integrator.getEvaluations() < 50);
61          Assert.assertTrue(integrator.getIterations()  < 10);
62          Assert.assertEquals(expected, result, tolerance);
63      }
64  
65      /**
66       * Test of integrator for the quintic function.
67       */
68      @Test
69      public void testQuinticFunction() {
70          UnivariateFunction f = new QuinticFunction();
71          UnivariateIntegrator integrator = new RombergIntegrator();
72          double min, max, expected, result, tolerance;
73  
74          min = 0; max = 1; expected = -1.0/48;
75          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
76          result = integrator.integrate(100, f, min, max);
77          Assert.assertTrue(integrator.getEvaluations() < 10);
78          Assert.assertTrue(integrator.getIterations()  < 5);
79          Assert.assertEquals(expected, result, tolerance);
80  
81          min = 0; max = 0.5; expected = 11.0/768;
82          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
83          result = integrator.integrate(100, f, min, max);
84          Assert.assertTrue(integrator.getEvaluations() < 10);
85          Assert.assertTrue(integrator.getIterations()  < 5);
86          Assert.assertEquals(expected, result, tolerance);
87  
88          min = -1; max = 4; expected = 2048/3.0 - 78 + 1.0/48;
89          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
90          result = integrator.integrate(100, f, min, max);
91          Assert.assertTrue(integrator.getEvaluations() < 10);
92          Assert.assertTrue(integrator.getIterations()  < 5);
93          Assert.assertEquals(expected, result, tolerance);
94      }

96      /**
97       * Test of parameters for the integrator.
98       */
99      @Test
100     public void testParameters() {
101         UnivariateFunction f = new Sin();
102 
103         try {
104             // bad interval
105             new RombergIntegrator().integrate(1000, f, 1, -1);
106             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
107         } catch (NumberIsTooLargeException ex) {
108             // expected
109         }
110         try {
111             // bad iteration limits
112             new RombergIntegrator(5, 4);
113             Assert.fail("Expecting NumberIsTooSmallException - bad iteration limits");
114         } catch (NumberIsTooSmallException ex) {
115             // expected
116         }
117         try {
118             // bad iteration limits
119             new RombergIntegrator(10, 50);
120             Assert.fail("Expecting NumberIsTooLargeException - bad iteration limits");
121         } catch (NumberIsTooLargeException ex) {
122             // expected
123         }
124     }
