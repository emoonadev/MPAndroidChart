17  package org.apache.commons.math4.distribution;
18  
19  import org.apache.commons.math4.distribution.NakagamiDistribution;
20  import org.apache.commons.math4.util.Precision;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test cases for NakagamiDistribution.
26   */
27  public class NakagamiDistributionTest extends RealDistributionAbstractTest {
28  
29      @Test
30      public void testParameters() {
31          NakagamiDistribution d = makeDistribution();
32          Assert.assertEquals(0.5, d.getShape(), Precision.EPSILON);
33          Assert.assertEquals(1, d.getScale(), Precision.EPSILON);
34      }
35  
36      @Test
37      public void testSupport() {
38          NakagamiDistribution d = makeDistribution();
39          Assert.assertEquals(d.getSupportLowerBound(), 0, Precision.EPSILON);
40          Assert.assertTrue(Double.isInfinite(d.getSupportUpperBound()));
41          Assert.assertTrue(d.isSupportConnected());
42      }
43  
44      @Override
45      public NakagamiDistribution makeDistribution() {
46          return new NakagamiDistribution(0.5, 1);
47      }
48  
49      @Override
50      public double[] makeCumulativeTestPoints() {
51          return new double[] {
52              0, 0.2, 0.4, 0.6, 0.8, 1, 1.2, 1.4, 1.6, 1.8, 2
53          };
54      }
55  
56      @Override
57      public double[] makeDensityTestValues() {
58          return new double[] {
59              0.0000000, 0.7820854, 0.7365403, 0.6664492, 0.5793831, 0.4839414,
60              0.3883721, 0.2994549, 0.2218417, 0.1579003, 0.1079819
61          };
62      }
63  
64      @Override
65      public double[] makeCumulativeTestValues() {
66          return new double[] {
67              0.0000000, 0.1585194, 0.3108435, 0.4514938, 0.5762892, 0.6826895,
68              0.7698607, 0.8384867, 0.8904014, 0.9281394, 0.9544997
69          };
70      }
71  
72  }
