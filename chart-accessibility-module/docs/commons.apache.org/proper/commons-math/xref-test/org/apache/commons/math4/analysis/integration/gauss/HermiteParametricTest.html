17  package org.apache.commons.math4.analysis.integration.gauss;
18  
19  import java.util.ArrayList;
20  import java.util.Collection;
21  
22  import org.junit.runner.RunWith;
23  import org.junit.runners.Parameterized;
24  import org.junit.runners.Parameterized.Parameters;
25  import org.apache.commons.math4.analysis.integration.gauss.GaussIntegratorFactory;
26  import org.apache.commons.math4.analysis.integration.gauss.HermiteRuleFactory;
27  import org.apache.commons.math4.util.FastMath;
28  
29  /**
30   * Test of the {@link HermiteRuleFactory}.
31   * This parameterized test extends the standard test for Gaussian quadrature
32   * rule, where each monomial is tested in turn.
33   * Parametrization allows to test automatically 0, 1, ... , {@link #MAX_NUM_POINTS}
34   * quadrature rules.
35   *
36   */
37  @RunWith(value=Parameterized.class)
38  public class HermiteParametricTest extends GaussianQuadratureAbstractTest {
39      private static final double SQRT_PI = FastMath.sqrt(Math.PI);
40      private static final GaussIntegratorFactory factory = new GaussIntegratorFactory();
41  
42      /**
43       * The highest order quadrature rule to be tested.
44       */
45      public static final int MAX_NUM_POINTS = 30;
46  
47      /**
48       * Creates a new instance of this test, with the specified number of nodes
49       * for the Gauss-Hermite quadrature rule.
50       *
51       * @param numberOfPoints Order of integration rule.
52       * @param maxDegree Maximum degree of monomials to be tested.
53       * @param eps Value of &epsilon;.
54       * @param numUlps Value of the maximum relative error (in ulps).
55       */
56      public HermiteParametricTest(int numberOfPoints,
57                                   int maxDegree,
58                                   double eps,
59                                   double numUlps) {
60          super(factory.hermite(numberOfPoints),
61                maxDegree, eps, numUlps);
62      }
63  
64      /**
65       * Returns the collection of parameters to be passed to the constructor of
66       * this class.
67       * Gauss-Hermite quadrature rules of order 1, ..., {@link #MAX_NUM_POINTS}
68       * will be constructed.
69       *
70       * @return the collection of parameters for this parameterized test.
71       */
72      @SuppressWarnings("boxing") // OK here
73      @Parameters
74      public static Collection<Object[]> getParameters() {
75          final ArrayList<Object[]> parameters = new ArrayList<Object[]>();
76          for (int k = 1; k <= MAX_NUM_POINTS; k++) {
77              parameters.add(new Object[] { k, 2 * k - 1, Math.ulp(1d), 195 });
78          }
79          return parameters;
80      }
81  
82      @Override
83      public double getExpectedValue(final int n) {
84          if (n % 2 == 1) {
85              return 0;
86          }
87  
88          final int iMax = n / 2;
89          double p = 1;
90          double q = 1;
91          for (int i = 0; i < iMax; i++) {
92              p *= 2 * i + 1;
93              q *= 2;
94          }
95  
96          return p / q * SQRT_PI;
97      }
98  }
