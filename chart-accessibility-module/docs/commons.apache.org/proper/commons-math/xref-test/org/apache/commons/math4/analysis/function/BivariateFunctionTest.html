18  package org.apache.commons.math4.analysis.function;
19  
20  import org.apache.commons.math4.util.FastMath;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test for all uncovered classes in org.apache.commons.math4.analysis.function that implement BivariateFunction explicitly.
26   */
27  public class BivariateFunctionTest {
28  
29      private static final double EPS = Math.ulp(1d);
30  
31      @Test
32      public void testAtan2() {
33          Atan2 atan2 = new Atan2();
34          Assert.assertEquals(FastMath.PI/4,atan2.value(1,1), EPS);
35      }
36  
37      @Test
38      public void testSubtract() {
39          Subtract subtract = new Subtract();
40          Assert.assertEquals(5, subtract.value(10,5), EPS);
41          Assert.assertEquals(-5, subtract.value(5,10), EPS);
42      }
43  
44  }
