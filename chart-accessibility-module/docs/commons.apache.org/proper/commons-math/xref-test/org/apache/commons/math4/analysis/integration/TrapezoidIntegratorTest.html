31  /**
32   * Test case for trapezoid integrator.
33   * <p>
34   * Test runs show that for a default relative accuracy of 1E-6, it
35   * generally takes 10 to 15 iterations for the integral to converge.
36   *
37   */
38  public final class TrapezoidIntegratorTest {
39  
40      /**
41       * Test of integrator for the sine function.
42       */
43      @Test
44      public void testSinFunction() {
45          UnivariateFunction f = new Sin();
46          UnivariateIntegrator integrator = new TrapezoidIntegrator();
47          double min, max, expected, result, tolerance;
48  
49          min = 0; max = FastMath.PI; expected = 2;
50          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
51          result = integrator.integrate(10000, f, min, max);
52          Assert.assertTrue(integrator.getEvaluations() < 2500);
53          Assert.assertTrue(integrator.getIterations()  < 15);
54          Assert.assertEquals(expected, result, tolerance);
55  
56          min = -FastMath.PI/3; max = 0; expected = -0.5;
57          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
58          result = integrator.integrate(10000, f, min, max);
59          Assert.assertTrue(integrator.getEvaluations() < 2500);
60          Assert.assertTrue(integrator.getIterations()  < 15);
61          Assert.assertEquals(expected, result, tolerance);
62      }
63  
64      /**
65       * Test of integrator for the quintic function.
66       */
67      @Test
68      public void testQuinticFunction() {
69          UnivariateFunction f = new QuinticFunction();
70          UnivariateIntegrator integrator = new TrapezoidIntegrator();
71          double min, max, expected, result, tolerance;
72  
73          min = 0; max = 1; expected = -1.0/48;
74          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
75          result = integrator.integrate(10000, f, min, max);
76          Assert.assertTrue(integrator.getEvaluations() < 5000);
77          Assert.assertTrue(integrator.getIterations()  < 15);
78          Assert.assertEquals(expected, result, tolerance);
79  
80          min = 0; max = 0.5; expected = 11.0/768;
81          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
82          result = integrator.integrate(10000, f, min, max);
83          Assert.assertTrue(integrator.getEvaluations() < 2500);
84          Assert.assertTrue(integrator.getIterations()  < 15);
85          Assert.assertEquals(expected, result, tolerance);
86  
87          min = -1; max = 4; expected = 2048/3.0 - 78 + 1.0/48;
88          tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
89          result = integrator.integrate(10000, f, min, max);
90          Assert.assertTrue(integrator.getEvaluations() < 5000);
91          Assert.assertTrue(integrator.getIterations()  < 15);
92          Assert.assertEquals(expected, result, tolerance);
93  
94      }

96      /**
97       * Test of parameters for the integrator.
98       */
99      @Test
100     public void testParameters() {
101         UnivariateFunction f = new Sin();
102 
103         try {
104             // bad interval
105             new TrapezoidIntegrator().integrate(1000, f, 1, -1);
106             Assert.fail("Expecting NumberIsTooLargeException - bad interval");
107         } catch (NumberIsTooLargeException ex) {
108             // expected
109         }
110         try {
111             // bad iteration limits
112             new TrapezoidIntegrator(5, 4);
113             Assert.fail("Expecting NumberIsTooSmallException - bad iteration limits");
114         } catch (NumberIsTooSmallException ex) {
115             // expected
116         }
117         try {
118             // bad iteration limits
119             new TrapezoidIntegrator(10,99);
120             Assert.fail("Expecting NumberIsTooLargeException - bad iteration limits");
121         } catch (NumberIsTooLargeException ex) {
122             // expected
123         }
124     }
