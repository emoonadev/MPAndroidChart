26  /**
27   * Test case for Newton form of polynomial function.

34      /**
35       * Test of polynomial for the linear function.
36       */
37      @Test
38      public void testLinearFunction() {
39          PolynomialFunctionNewtonForm p;
40          double coefficients[], z, expected, result, tolerance = 1E-12;
41  
42          // p(x) = 1.5x - 4 = 2 + 1.5(x-4)
43          double a[] = { 2.0, 1.5 };
44          double c[] = { 4.0 };
45          p = new PolynomialFunctionNewtonForm(a, c);
46  
47          z = 2.0; expected = -1.0; result = p.value(z);
48          Assert.assertEquals(expected, result, tolerance);
49  
50          z = 4.5; expected = 2.75; result = p.value(z);
51          Assert.assertEquals(expected, result, tolerance);
52  
53          z = 6.0; expected = 5.0; result = p.value(z);
54          Assert.assertEquals(expected, result, tolerance);
55  
56          Assert.assertEquals(1, p.degree());
57  
58          coefficients = p.getCoefficients();
59          Assert.assertEquals(2, coefficients.length);
60          Assert.assertEquals(-4.0, coefficients[0], tolerance);
61          Assert.assertEquals(1.5, coefficients[1], tolerance);
62      }

64      /**
65       * Test of polynomial for the quadratic function.
66       */
67      @Test
68      public void testQuadraticFunction() {
69          PolynomialFunctionNewtonForm p;
70          double coefficients[], z, expected, result, tolerance = 1E-12;
71  
72          // p(x) = 2x^2 + 5x - 3 = 4 + 3(x-1) + 2(x-1)(x+2)
73          double a[] = { 4.0, 3.0, 2.0 };
74          double c[] = { 1.0, -2.0 };
75          p = new PolynomialFunctionNewtonForm(a, c);
76  
77          z = 1.0; expected = 4.0; result = p.value(z);
78          Assert.assertEquals(expected, result, tolerance);
79  
80          z = 2.5; expected = 22.0; result = p.value(z);
81          Assert.assertEquals(expected, result, tolerance);
82  
83          z = -2.0; expected = -5.0; result = p.value(z);
84          Assert.assertEquals(expected, result, tolerance);
85  
86          Assert.assertEquals(2, p.degree());
87  
88          coefficients = p.getCoefficients();
89          Assert.assertEquals(3, coefficients.length);
90          Assert.assertEquals(-3.0, coefficients[0], tolerance);
91          Assert.assertEquals(5.0, coefficients[1], tolerance);
92          Assert.assertEquals(2.0, coefficients[2], tolerance);
93      }

95      /**
96       * Test of polynomial for the quintic function.
97       */
98      @Test
99      public void testQuinticFunction() {
100         PolynomialFunctionNewtonForm p;
101         double coefficients[], z, expected, result, tolerance = 1E-12;
102 
103         // p(x) = x^5 - x^4 - 7x^3 + x^2 + 6x
104         //      = 6x - 6x^2 -6x^2(x-1) + x^2(x-1)(x+1) + x^2(x-1)(x+1)(x-2)
105         double a[] = { 0.0, 6.0, -6.0, -6.0, 1.0, 1.0 };
106         double c[] = { 0.0, 0.0, 1.0, -1.0, 2.0 };
107         p = new PolynomialFunctionNewtonForm(a, c);
108 
109         z = 0.0; expected = 0.0; result = p.value(z);
110         Assert.assertEquals(expected, result, tolerance);
111 
112         z = -2.0; expected = 0.0; result = p.value(z);
113         Assert.assertEquals(expected, result, tolerance);
114 
115         z = 4.0; expected = 360.0; result = p.value(z);
116         Assert.assertEquals(expected, result, tolerance);
117 
118         Assert.assertEquals(5, p.degree());
119 
120         coefficients = p.getCoefficients();
121         Assert.assertEquals(6, coefficients.length);
122         Assert.assertEquals(0.0, coefficients[0], tolerance);
123         Assert.assertEquals(6.0, coefficients[1], tolerance);
124         Assert.assertEquals(1.0, coefficients[2], tolerance);
125         Assert.assertEquals(-7.0, coefficients[3], tolerance);
126         Assert.assertEquals(-1.0, coefficients[4], tolerance);
127         Assert.assertEquals(1.0, coefficients[5], tolerance);
128     }

130     /**
131      * Test for derivatives.
132      */
133     @Test
134     public void testDerivative() {
135 
136         // x^3 = 0 * [1] + 1 * [x] + 3 * [x(x-1)] + 1 * [x(x-1)(x-2)]
137         PolynomialFunctionNewtonForm p =
138                 new PolynomialFunctionNewtonForm(new double[] { 0, 1, 3, 1 },
139                                                  new double[] { 0, 1, 2 });
140 
141         double eps = 2.0e-14;
142         for (double t = 0.0; t < 10.0; t += 0.1) {
143             DerivativeStructure x = new DerivativeStructure(1, 4, 0, t);
144             DerivativeStructure y = p.value(x);
145             Assert.assertEquals(t * t * t,   y.getValue(),              eps * t * t * t);
146             Assert.assertEquals(3.0 * t * t, y.getPartialDerivative(1), eps * 3.0 * t * t);
147             Assert.assertEquals(6.0 * t,     y.getPartialDerivative(2), eps * 6.0 * t);
148             Assert.assertEquals(6.0,         y.getPartialDerivative(3), eps * 6.0);
149             Assert.assertEquals(0.0,         y.getPartialDerivative(4), eps);
150         }
151 
152     }

168         }

173             new PolynomialFunctionNewtonForm(a, c);
174             Assert.fail("Expecting MathIllegalArgumentException - mismatch input arrays");
175         } catch (MathIllegalArgumentException ex) {
176             // expected
177         }
178     }
