57      @Test
58      public void testSinMin() {
59          UnivariateFunction f = new Sin();
60          UnivariateOptimizer underlying = new BrentOptimizer(1e-10, 1e-14);
61          UniformRandomProvider g = RandomSource.create(RandomSource.TWO_CMRES, 44428400075l);
62          MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 10, g);
63          optimizer.optimize(new MaxEval(300),
64                             new UnivariateObjectiveFunction(f),
65                             GoalType.MINIMIZE,
66                             new SearchInterval(-100.0, 100.0));
67          UnivariatePointValuePair[] optima = optimizer.getOptima();
68          for (int i = 1; i < optima.length; ++i) {
69              double d = (optima[i].getPoint() - optima[i-1].getPoint()) / (2 * FastMath.PI);
70              Assert.assertTrue(FastMath.abs(d - FastMath.rint(d)) < 1.0e-8);
71              Assert.assertEquals(-1.0, f.value(optima[i].getPoint()), 1.0e-10);
72              Assert.assertEquals(f.value(optima[i].getPoint()), optima[i].getValue(), 1.0e-10);
73          }
74          Assert.assertTrue(optimizer.getEvaluations() > 200);
75          Assert.assertTrue(optimizer.getEvaluations() < 300);
76      }

78      @Test
79      public void testQuinticMin() {
80          UnivariateFunction f = new QuinticFunction();
81          UnivariateOptimizer underlying = new BrentOptimizer(1e-9, 1e-14);
82          UniformRandomProvider g = RandomSource.create(RandomSource.TWO_CMRES, 4312000053L);
83          MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 5, g);
84  
85          UnivariatePointValuePair optimum
86              = optimizer.optimize(new MaxEval(300),
87                                   new UnivariateObjectiveFunction(f),
88                                   GoalType.MINIMIZE,
89                                   new SearchInterval(-0.3, -0.2));
90          Assert.assertEquals(-0.271956128, optimum.getPoint(), 1e-9);
91          Assert.assertEquals(-0.0443342695, optimum.getValue(), 1e-9);
92  
93          UnivariatePointValuePair[] optima = optimizer.getOptima();
94          for (int i = 0; i < optima.length; ++i) {
95              Assert.assertEquals(f.value(optima[i].getPoint()), optima[i].getValue(), 1e-9);
96          }
97          Assert.assertTrue(optimizer.getEvaluations() >= 50);
98          Assert.assertTrue(optimizer.getEvaluations() <= 100);
99      }

101     @Test
102     public void testBadFunction() {
103         UnivariateFunction f = new UnivariateFunction() {
104                 public double value(double x) {
105                     if (x < 0) {
106                         throw new LocalException();
107                     }
108                     return 0;
109                 }
110             };
111         UnivariateOptimizer underlying = new BrentOptimizer(1e-9, 1e-14);
112         UniformRandomProvider g = RandomSource.create(RandomSource.TWO_CMRES, 4312000053L);
113         MultiStartUnivariateOptimizer optimizer = new MultiStartUnivariateOptimizer(underlying, 5, g);
114 
115         try {
116             optimizer.optimize(new MaxEval(300),
117                                new UnivariateObjectiveFunction(f),
118                                GoalType.MINIMIZE,
119                                new SearchInterval(-0.3, -0.2));
120             Assert.fail();
121         } catch (LocalException e) {
122             // Expected.
123         }
124 
125         // Ensure that the exception was thrown because no optimum was found.
126         Assert.assertTrue(optimizer.getOptima()[0] == null);
127     }
