326     public void testShift() {
327         // f1(x) = 1 + x + 2 x^2
328         PolynomialFunction f1x = new PolynomialFunction(new double[] { 1, 1, 2 });
329 
330         PolynomialFunction f1x1
331             = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), 1));
332         checkPolynomial(f1x1, "4 + 5 x + 2 x^2");
333 
334         PolynomialFunction f1xM1
335             = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), -1));
336         checkPolynomial(f1xM1, "2 - 3 x + 2 x^2");
337 
338         PolynomialFunction f1x3
339             = new PolynomialFunction(PolynomialsUtils.shift(f1x.getCoefficients(), 3));
340         checkPolynomial(f1x3, "22 + 13 x + 2 x^2");
341 
342         // f2(x) = 2 + 3 x^2 + 8 x^3 + 121 x^5
343         PolynomialFunction f2x = new PolynomialFunction(new double[]{2, 0, 3, 8, 0, 121});
344 
345         PolynomialFunction f2x1
346             = new PolynomialFunction(PolynomialsUtils.shift(f2x.getCoefficients(), 1));
347         checkPolynomial(f2x1, "134 + 635 x + 1237 x^2 + 1218 x^3 + 605 x^4 + 121 x^5");

351         checkPolynomial(f2x3, "29648 + 49239 x + 32745 x^2 + 10898 x^3 + 1815 x^4 + 121 x^5");
352     }

356         PolynomialFunction q = new PolynomialFunction(new double[] { denominator});
357         Assert.assertEquals(reference, p.multiply(q).toString());

364     private void checkNullPolynomial(PolynomialFunction p) {
365         for (double coefficient : p.getCoefficients()) {
366             Assert.assertEquals(0, coefficient, 1e-13);
367         }
368     }

370     private void checkOrthogonality(final PolynomialFunction p1,
371                                     final PolynomialFunction p2,
372                                     final UnivariateFunction weight,
373                                     final double a, final double b,
374                                     final double nonZeroThreshold,
375                                     final double zeroThreshold) {
376         UnivariateFunction f = new UnivariateFunction() {
377             @Override
378             public double value(double x) {
379                 return weight.value(x) * p1.value(x) * p2.value(x);
380             }
381         };
382         double dotProduct =
383                 new IterativeLegendreGaussIntegrator(5, 1.0e-9, 1.0e-8, 2, 15).integrate(1000000, f, a, b);
384         if (p1.degree() == p2.degree()) {
385             // integral should be non-zero
386             Assert.assertTrue("I(" + p1.degree() + ", " + p2.degree() + ") = "+ dotProduct,
387                               FastMath.abs(dotProduct) > nonZeroThreshold);
388         } else {
389             // integral should be zero
390             Assert.assertEquals("I(" + p1.degree() + ", " + p2.degree() + ") = "+ dotProduct,
391                                 0.0, FastMath.abs(dotProduct), zeroThreshold);
392         }
