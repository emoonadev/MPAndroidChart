20  import java.util.List;

22  import org.apache.commons.math4.exception.MathIllegalArgumentException;
23  import org.apache.commons.math4.stat.descriptive.SummaryStatistics;
24  import org.apache.commons.math4.stat.inference.OneWayAnova;
25  import org.junit.Assert;

29  /**
30   * Test cases for the OneWayAnovaImpl class.
31   *
32   */
33  
34  public class OneWayAnovaTest {
35  
36      protected OneWayAnova testStatistic = new OneWayAnova();
37  
38      private double[] emptyArray = {};
39  
40      private double[] classA =
41              {93.0, 103.0, 95.0, 101.0, 91.0, 105.0, 96.0, 94.0, 101.0 };
42      private double[] classB =
43              {99.0, 92.0, 102.0, 100.0, 102.0, 89.0 };
44      private double[] classC =
45              {110.0, 115.0, 111.0, 117.0, 128.0, 117.0 };
46  
47      @Test
48      public void testAnovaFValue() {
49          // Target comparison values computed using R version 2.6.0 (Linux version)
50          List<double[]> threeClasses = new ArrayList<double[]>();
51          threeClasses.add(classA);
52          threeClasses.add(classB);
53          threeClasses.add(classC);
54  
55          Assert.assertEquals("ANOVA F-value",  24.67361709460624,
56                   testStatistic.anovaFValue(threeClasses), 1E-12);
57  
58          List<double[]> twoClasses = new ArrayList<double[]>();
59          twoClasses.add(classA);
60          twoClasses.add(classB);
61  
62          Assert.assertEquals("ANOVA F-value",  0.0150579150579,
63                   testStatistic.anovaFValue(twoClasses), 1E-12);
64  
65          List<double[]> emptyContents = new ArrayList<double[]>();
66          emptyContents.add(emptyArray);
67          emptyContents.add(classC);
68          try {
69              testStatistic.anovaFValue(emptyContents);
70              Assert.fail("empty array for key classX, MathIllegalArgumentException expected");
71          } catch (MathIllegalArgumentException ex) {
72              // expected
73          }
74  
75          List<double[]> tooFew = new ArrayList<double[]>();
76          tooFew.add(classA);
77          try {
78              testStatistic.anovaFValue(tooFew);
79              Assert.fail("less than two classes, MathIllegalArgumentException expected");
80          } catch (MathIllegalArgumentException ex) {
81              // expected
82          }
83      }

86      @Test
87      public void testAnovaPValue() {
88          // Target comparison values computed using R version 2.6.0 (Linux version)
89          List<double[]> threeClasses = new ArrayList<double[]>();
90          threeClasses.add(classA);
91          threeClasses.add(classB);
92          threeClasses.add(classC);
93  
94          Assert.assertEquals("ANOVA P-value", 6.959446E-06,
95                   testStatistic.anovaPValue(threeClasses), 1E-12);
96  
97          List<double[]> twoClasses = new ArrayList<double[]>();
98          twoClasses.add(classA);
99          twoClasses.add(classB);
100 
101         Assert.assertEquals("ANOVA P-value",  0.904212960464,
102                  testStatistic.anovaPValue(twoClasses), 1E-12);
103 
104     }
105 
106     @Test
107     public void testAnovaPValueSummaryStatistics() {
108         // Target comparison values computed using R version 2.6.0 (Linux version)
109         List<SummaryStatistics> threeClasses = new ArrayList<SummaryStatistics>();
110         SummaryStatistics statsA = new SummaryStatistics();
111         for (double a : classA) {
112             statsA.addValue(a);
113         }
114         threeClasses.add(statsA);
115         SummaryStatistics statsB = new SummaryStatistics();
116         for (double b : classB) {
117             statsB.addValue(b);
118         }
119         threeClasses.add(statsB);
120         SummaryStatistics statsC = new SummaryStatistics();
121         for (double c : classC) {
122             statsC.addValue(c);
123         }
124         threeClasses.add(statsC);
125 
126         Assert.assertEquals("ANOVA P-value", 6.959446E-06,
127                  testStatistic.anovaPValue(threeClasses, true), 1E-12);
128 
129         List<SummaryStatistics> twoClasses = new ArrayList<SummaryStatistics>();
130         twoClasses.add(statsA);
131         twoClasses.add(statsB);
132 
133         Assert.assertEquals("ANOVA P-value",  0.904212960464,
134                  testStatistic.anovaPValue(twoClasses, false), 1E-12);
135 
136     }

138     @Test
139     public void testAnovaTest() {
140         // Target comparison values computed using R version 2.3.1 (Linux version)
141         List<double[]> threeClasses = new ArrayList<double[]>();
142         threeClasses.add(classA);
143         threeClasses.add(classB);
144         threeClasses.add(classC);
145 
146         Assert.assertTrue("ANOVA Test P<0.01", testStatistic.anovaTest(threeClasses, 0.01));
147 
148         List<double[]> twoClasses = new ArrayList<double[]>();
149         twoClasses.add(classA);
150         twoClasses.add(classB);
151 
152         Assert.assertFalse("ANOVA Test P>0.01", testStatistic.anovaTest(twoClasses, 0.01));
153     }
