78      @Test
79      public void testPerfectFit() {
80          double[] betaHat = regression.estimateRegressionParameters();
81          TestUtils.assertEquals(betaHat,
82                                 new double[]{ 11.0, 1.0 / 2.0, 2.0 / 3.0, 3.0 / 4.0, 4.0 / 5.0, 5.0 / 6.0 },
83                                 1e-14);
84          double[] residuals = regression.estimateResiduals();
85          TestUtils.assertEquals(residuals, new double[]{0d,0d,0d,0d,0d,0d},
86                                 1e-14);
87          RealMatrix errors =
88              new Array2DRowRealMatrix(regression.estimateRegressionParametersVariance(), false);
89          final double[] s = { 1.0, -1.0 /  2.0, -1.0 /  3.0, -1.0 /  4.0, -1.0 /  5.0, -1.0 /  6.0 };
90          RealMatrix referenceVariance = new Array2DRowRealMatrix(s.length, s.length);
91          referenceVariance.walkInOptimizedOrder(new DefaultRealMatrixChangingVisitor() {
92              @Override
93              public double visit(int row, int column, double value) {
94                  if (row == 0) {
95                      return s[column];
96                  }
97                  double x = s[row] * s[column];
98                  return (row == column) ? 2 * x : x;
99              }
100         });
101        Assert.assertEquals(0.0,
102                      errors.subtract(referenceVariance).getNorm(),
103                      5.0e-16 * referenceVariance.getNorm());
104        Assert.assertEquals(1, ((OLSMultipleLinearRegression) regression).calculateRSquared(), 1E-12);
105     }

374     /**
375      * Test hat matrix computation
376      *
377      */
378     @Test
379     public void testHat() {
380 
381         /*
382          * This example is from "The Hat Matrix in Regression and ANOVA",
383          * David C. Hoaglin and Roy E. Welsch,
384          * The American Statistician, Vol. 32, No. 1 (Feb., 1978), pp. 17-22.
385          *
386          */
387         double[] design = new double[] {
388                 11.14, .499, 11.1,
389                 12.74, .558, 8.9,
390                 13.13, .604, 8.8,
391                 11.51, .441, 8.9,
392                 12.38, .550, 8.8,
393                 12.60, .528, 9.9,
394                 11.13, .418, 10.7,
395                 11.7, .480, 10.5,
396                 11.02, .406, 10.5,
397                 11.41, .467, 10.7
398         };
399 
400         int nobs = 10;
401         int nvars = 2;
402 
403         // Estimate the model
404         OLSMultipleLinearRegression model = new OLSMultipleLinearRegression();
405         model.newSampleData(design, nobs, nvars);
406 
407         RealMatrix hat = model.calculateHat();
408 
409         // Reference data is upper half of symmetric hat matrix
410         double[] referenceData = new double[] {
411                 .418, -.002,  .079, -.274, -.046,  .181,  .128,  .222,  .050,  .242,
412                        .242,  .292,  .136,  .243,  .128, -.041,  .033, -.035,  .004,
413                               .417, -.019,  .273,  .187, -.126,  .044, -.153,  .004,
414                                      .604,  .197, -.038,  .168, -.022,  .275, -.028,
415                                             .252,  .111, -.030,  .019, -.010, -.010,
416                                                    .148,  .042,  .117,  .012,  .111,
417                                                           .262,  .145,  .277,  .174,
418                                                                  .154,  .120,  .168,
419                                                                         .315,  .148,
420                                                                                .187
421         };
422 
423         // Check against reference data and verify symmetry
424         int k = 0;
425         for (int i = 0; i < 10; i++) {
426             for (int j = i; j < 10; j++) {
427                 Assert.assertEquals(referenceData[k], hat.getEntry(i, j), 10e-3);
428                 Assert.assertEquals(hat.getEntry(i, j), hat.getEntry(j, i), 10e-12);
429                 k++;
430             }
431         }
432 
433         /*
434          * Verify that residuals computed using the hat matrix are close to
435          * what we get from direct computation, i.e. r = (I - H) y
436          */
437         double[] residuals = model.estimateResiduals();
438         RealMatrix I = MatrixUtils.createRealIdentityMatrix(10);
439         double[] hatResiduals = I.subtract(hat).operate(model.getY()).toArray();
440         TestUtils.assertEquals(residuals, hatResiduals, 10e-12);
441     }

473     /**
474      * Verifies that setting X and Y separately has the same effect as newSample(X,Y).
475      */
476     @Test
477     public void testNewSample2() {
478         double[] y = new double[] {1, 2, 3, 4};
479         double[][] x = new double[][] {
480           {19, 22, 33},
481           {20, 30, 40},
482           {25, 35, 45},
483           {27, 37, 47}
484         };
485         OLSMultipleLinearRegression regression = new OLSMultipleLinearRegression();
486         regression.newSampleData(y, x);
487         RealMatrix combinedX = regression.getX().copy();
488         RealVector combinedY = regression.getY().copy();
489         regression.newXSampleData(x);
490         regression.newYSampleData(y);
491         Assert.assertEquals(combinedX, regression.getX());
492         Assert.assertEquals(combinedY, regression.getY());
493 
494         // No intercept
