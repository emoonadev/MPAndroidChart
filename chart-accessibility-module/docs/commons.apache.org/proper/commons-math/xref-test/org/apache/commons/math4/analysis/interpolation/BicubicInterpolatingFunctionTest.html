116     @Test
117     public void testIsValidPoint() {
118         final double xMin = -12;
119         final double xMax = 34;
120         final double yMin = 5;
121         final double yMax = 67;
122         final double[] xval = new double[] { xMin, xMax };
123         final double[] yval = new double[] { yMin, yMax };
124         final double[][] f = new double[][] { { 1, 2 },
125                                               { 3, 4 } };
126         final double[][] dFdX = f;
127         final double[][] dFdY = f;
128         final double[][] dFdXdY = f;
129 
130         final BicubicInterpolatingFunction bcf
131             = new BicubicInterpolatingFunction(xval, yval, f,
132                                                      dFdX, dFdY, dFdXdY);
133 
134         double x, y;
135 
136         x = xMin;
137         y = yMin;
138         Assert.assertTrue(bcf.isValidPoint(x, y));
139         // Ensure that no exception is thrown.
140         bcf.value(x, y);
141 
142         x = xMax;
143         y = yMax;
144         Assert.assertTrue(bcf.isValidPoint(x, y));
145         // Ensure that no exception is thrown.
146         bcf.value(x, y);
147 
148         final double xRange = xMax - xMin;
149         final double yRange = yMax - yMin;
150         x = xMin + xRange / 3.4;
151         y = yMin + yRange / 1.2;
152         Assert.assertTrue(bcf.isValidPoint(x, y));
153         // Ensure that no exception is thrown.
154         bcf.value(x, y);
155 
156         final double small = 1e-8;
157         x = xMin - small;
158         y = yMax;
159         Assert.assertFalse(bcf.isValidPoint(x, y));
160         // Ensure that an exception would have been thrown.
161         try {
162             bcf.value(x, y);
163             Assert.fail("OutOfRangeException expected");
164         } catch (OutOfRangeException expected) {}
165 
166         x = xMin;
167         y = yMax + small;
168         Assert.assertFalse(bcf.isValidPoint(x, y));
169         // Ensure that an exception would have been thrown.
170         try {
171             bcf.value(x, y);
172             Assert.fail("OutOfRangeException expected");
173         } catch (OutOfRangeException expected) {}
174     }

237      * z = 2 x<sup>2</sup> - 3 y<sup>2</sup> + 4 x y - 5
238      */
239     @Test
240     public void testParaboloid() {
241         final int numberOfElements = 10;
242         final double minimumX = -10;
243         final double maximumX = 10;
244         final double minimumY = -10;
245         final double maximumY = 10;
246         final int numberOfSamples = 1000;
247 
248         final double interpolationTolerance = 2e-14;
249         final double maxTolerance = 1e-12;
250 
251         // Function values
252         BivariateFunction f = new BivariateFunction() {
253                 @Override
254                 public double value(double x, double y) {
255                     return 2 * x * x - 3 * y * y + 4 * x * y - 5;
256                 }
257             };
258         BivariateFunction dfdx = new BivariateFunction() {
259                 @Override
260                 public double value(double x, double y) {
261                     return 4 * (x + y);
262                 }
263             };
264         BivariateFunction dfdy = new BivariateFunction() {
265                 @Override
266                 public double value(double x, double y) {
267                     return 4 * x - 6 * y;
268                 }
269             };
270         BivariateFunction d2fdxdy = new BivariateFunction() {
271                 @Override
272                 public double value(double x, double y) {
273                     return 4;
274                 }
275             };
276 
277         testInterpolation(minimumX,
278                           maximumX,
279                           minimumY,
280                           maximumY,
281                           numberOfElements,
282                           numberOfSamples,
283                           f,
284                           dfdx,
285                           dfdy,
286                           d2fdxdy,
287                           interpolationTolerance,
288                           maxTolerance,
289                           false);
290     }

292     /**
293      * @param minimumX Lower bound of interpolation range along the x-coordinate.
294      * @param maximumX Higher bound of interpolation range along the x-coordinate.
295      * @param minimumY Lower bound of interpolation range along the y-coordinate.
296      * @param maximumY Higher bound of interpolation range along the y-coordinate.
297      * @param numberOfElements Number of data points (along each dimension).
298      * @param numberOfSamples Number of test points.
299      * @param f Function to test.
300      * @param dfdx Partial derivative w.r.t. x of the function to test.
301      * @param dfdy Partial derivative w.r.t. y of the function to test.
302      * @param d2fdxdy Second partial cross-derivative of the function to test.
303      * @param meanTolerance Allowed average error (mean error on all interpolated values).
304      * @param maxTolerance Allowed error on each interpolated value.
305      */
306     private void testInterpolation(double minimumX,
307                                    double maximumX,
308                                    double minimumY,
309                                    double maximumY,
310                                    int numberOfElements,
311                                    int numberOfSamples,
312                                    BivariateFunction f,
313                                    BivariateFunction dfdx,
314                                    BivariateFunction dfdy,
315                                    BivariateFunction d2fdxdy,
316                                    double meanTolerance,
317                                    double maxTolerance,
318                                    boolean print) {
319         double expected;
320         double actual;
321         double currentX;
322         double currentY;
323         final double deltaX = (maximumX - minimumX) / numberOfElements;
324         final double deltaY = (maximumY - minimumY) / numberOfElements;
325         final double[] xValues = new double[numberOfElements];
326         final double[] yValues = new double[numberOfElements];
327         final double[][] zValues = new double[numberOfElements][numberOfElements];
328         final double[][] dzdx = new double[numberOfElements][numberOfElements];
329         final double[][] dzdy = new double[numberOfElements][numberOfElements];
330         final double[][] d2zdxdy = new double[numberOfElements][numberOfElements];
331 
332         for (int i = 0; i < numberOfElements; i++) {
333             xValues[i] = minimumX + deltaX * i;
334             final double x = xValues[i];
335             for (int j = 0; j < numberOfElements; j++) {
336                 yValues[j] = minimumY + deltaY * j;
337                 final double y = yValues[j];
338                 zValues[i][j] = f.value(x, y);
339                 dzdx[i][j] = dfdx.value(x, y);
340                 dzdy[i][j] = dfdy.value(x, y);
341                 d2zdxdy[i][j] = d2fdxdy.value(x, y);
342             }
343         }
344 
345         final BivariateFunction interpolation
346             = new BicubicInterpolatingFunction(xValues,
347                                                yValues,
348                                                zValues,
349                                                dzdx,
350                                                dzdy,
351                                                d2zdxdy);
352 
353         for (int i = 0; i < numberOfElements; i++) {
354             currentX = xValues[i];
355             for (int j = 0; j < numberOfElements; j++) {
356                 currentY = yValues[j];
357                 expected = f.value(currentX, currentY);
358                 actual = interpolation.value(currentX, currentY);
359                 Assert.assertTrue("On data point: " + expected + " != " + actual,
360                                   Precision.equals(expected, actual));
361             }
362         }
363 
364         final UniformRandomProvider rng = RandomSource.create(RandomSource.WELL_19937_C, 1234567L);
365         final RealDistribution.Sampler distX = new UniformRealDistribution(xValues[0], xValues[xValues.length - 1]).createSampler(rng);
366         final RealDistribution.Sampler distY = new UniformRealDistribution(yValues[0], yValues[yValues.length - 1]).createSampler(rng);
367 
368         double sumError = 0;
369         for (int i = 0; i < numberOfSamples; i++) {
370             currentX = distX.sample();
371             currentY = distY.sample();
372             expected = f.value(currentX, currentY);
373 
374             if (print) {
375                 System.out.println(currentX + " " + currentY + " -> ");
376             }
377 
378             actual = interpolation.value(currentX, currentY);
379             sumError += FastMath.abs(actual - expected);
380 
381             if (print) {
382                 System.out.println(actual + " (diff=" + (expected - actual) + ")");
383             }
384 
385             Assert.assertEquals(expected, actual, maxTolerance);
386         }
387 
388         final double meanError = sumError / numberOfSamples;
389         Assert.assertEquals(0, meanError, meanTolerance);
390     }
