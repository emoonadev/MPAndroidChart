59              throws DimensionMismatchException, NumberIsTooSmallException,

62            TestProblem1 pb = new TestProblem1();
63            double minStep = 0.1 * (pb.getFinalTime() - pb.getInitialTime());
64            double maxStep = pb.getFinalTime() - pb.getInitialTime();
65            double[] vecAbsoluteTolerance = { 1.0e-15, 1.0e-16 };
66            double[] vecRelativeTolerance = { 1.0e-15, 1.0e-16 };
67  
68            FirstOrderIntegrator integ = new AdamsMoultonIntegrator(4, minStep, maxStep,

79      @Test
80      public void testIncreasingTolerance()
81              throws DimensionMismatchException, NumberIsTooSmallException,
82              MaxCountExceededException, NoBracketingException {
83  
84          int previousCalls = Integer.MAX_VALUE;
85          for (int i = -12; i < -2; ++i) {
86              TestProblem1 pb = new TestProblem1();
87              double minStep = 0;
88              double maxStep = pb.getFinalTime() - pb.getInitialTime();
89              double scalAbsoluteTolerance = FastMath.pow(10.0, i);
90              double scalRelativeTolerance = 0.01 * scalAbsoluteTolerance;
91  
92              FirstOrderIntegrator integ = new AdamsMoultonIntegrator(4, minStep, maxStep,
93                                                                      scalAbsoluteTolerance,
94                                                                      scalRelativeTolerance);
95              TestProblemHandler handler = new TestProblemHandler(pb, integ);
96              integ.addStepHandler(handler);
97              integ.integrate(pb,
98                              pb.getInitialTime(), pb.getInitialState(),
99                              pb.getFinalTime(), new double[pb.getDimension()]);
100 
101             // the 0.45 and 8.69 factors are only valid for this test
102             // and has been obtained from trial and error
103             // there is no general relation between local and global errors
104             Assert.assertTrue(handler.getMaximalValueError() > (0.45 * scalAbsoluteTolerance));
105             Assert.assertTrue(handler.getMaximalValueError() < (8.69 * scalAbsoluteTolerance));
106             Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-16);
107 
108             int calls = pb.getCalls();
109             Assert.assertEquals(integ.getEvaluations(), calls);
110             Assert.assertTrue(calls <= previousCalls);
111             previousCalls = calls;
112 
113         }

155     @Test
156     public void polynomial()
157             throws DimensionMismatchException, NumberIsTooSmallException,
158             MaxCountExceededException, NoBracketingException {
159         TestProblem6 pb = new TestProblem6();
160         double range = FastMath.abs(pb.getFinalTime() - pb.getInitialTime());
161 
162         for (int nSteps = 2; nSteps < 8; ++nSteps) {
163             AdamsMoultonIntegrator integ =
164                 new AdamsMoultonIntegrator(nSteps, 1.0e-6 * range, 0.1 * range, 1.0e-5, 1.0e-5);
165             integ.setStarterIntegrator(new PerfectStarter(pb, nSteps));
166             TestProblemHandler handler = new TestProblemHandler(pb, integ);
167             integ.addStepHandler(handler);
168             integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
169                             pb.getFinalTime(), new double[pb.getDimension()]);
170             if (nSteps < 5) {
171                 Assert.assertTrue(handler.getMaximalValueError() > 2.2e-05);
172             } else {
173                 Assert.assertTrue(handler.getMaximalValueError() < 1.1e-11);
174             }
175         }

179     private static class PerfectStarter extends AbstractIntegrator {
180 
181         private final PerfectInterpolator interpolator;
182         private final int nbSteps;
183 
184         public PerfectStarter(final TestProblemAbstract problem, final int nbSteps) {
185             this.interpolator = new PerfectInterpolator(problem);
186             this.nbSteps      = nbSteps;
187         }
188 
189         public void integrate(ExpandableStatefulODE equations, double t) {
190             double tStart = equations.getTime() + 0.01 * (t - equations.getTime());
191             getCounter().increment(nbSteps);
192             for (int i = 0; i < nbSteps; ++i) {
193                 double tK = ((nbSteps - 1 - (i + 1)) * equations.getTime() + (i + 1) * tStart) / (nbSteps - 1);
194                 interpolator.setPreviousTime(interpolator.getCurrentTime());
195                 interpolator.setCurrentTime(tK);
196                 interpolator.setInterpolatedTime(tK);
197                 for (StepHandler handler : getStepHandlers()) {
198                     handler.handleStep(interpolator, i == nbSteps - 1);
199                 }
200             }
201         }
202 
203     }

205     private static class PerfectInterpolator implements StepInterpolator {
206         private final TestProblemAbstract problem;
207         private double previousTime;
208         private double currentTime;
209         private double interpolatedTime;
210 
211         public PerfectInterpolator(final TestProblemAbstract problem) {
212             this.problem          = problem;
213             this.previousTime     = problem.getInitialTime();
214             this.currentTime      = problem.getInitialTime();
215             this.interpolatedTime = problem.getInitialTime();
216         }
217 
218         public void readExternal(ObjectInput arg0) {
219         }
220 
221         public void writeExternal(ObjectOutput arg0) {
222         }
223 
224         public double getPreviousTime() {
225             return previousTime;
226         }
227 
228         public void setPreviousTime(double time) {
229             previousTime = time;
230         }
231 
232         public double getCurrentTime() {
233             return currentTime;
234         }
235 
236         public void setCurrentTime(double time) {
237             currentTime = time;
238         }
239 
240         public double getInterpolatedTime() {
241             return interpolatedTime;
242         }
243 
244         public void setInterpolatedTime(double time) {
245             interpolatedTime = time;
246         }
247 
248         public double[] getInterpolatedState() {
249             return problem.computeTheoreticalState(interpolatedTime);
250         }
251 
252         public double[] getInterpolatedDerivatives() {
253             double[] y = problem.computeTheoreticalState(interpolatedTime);
254             double[] yDot = new double[y.length];
255             problem.computeDerivatives(interpolatedTime, y, yDot);
256             return yDot;
257         }
258 
259         public double[] getInterpolatedSecondaryState(int index) {
260             return null;
261         }
262 
263         public double[] getInterpolatedSecondaryDerivatives(int index) {
264             return null;
265         }
266 
267         public boolean isForward() {
268             return problem.getFinalTime() > problem.getInitialTime();
269         }
270 
271         public StepInterpolator copy() {
272             return this;
273         }
274 
275     }
