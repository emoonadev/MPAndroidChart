17  package org.apache.commons.math4.stat.inference;

19  import org.apache.commons.math4.exception.NoDataException;
20  import org.apache.commons.math4.exception.NullArgumentException;
21  import org.apache.commons.math4.stat.inference.MannWhitneyUTest;
22  import org.junit.Assert;
23  import org.junit.Test;

26  /**
27   * Test cases for the MannWhitneyUTestImpl class.
28   *
29   */
30  
31  public class MannWhitneyUTestTest {
32  
33      protected MannWhitneyUTest testStatistic = new MannWhitneyUTest();
34  
35      @Test
36      public void testMannWhitneyUSimple() {
37          /* Target values computed using R version 2.11.1
38           * x <- c(19, 22, 16, 29, 24)
39           * y <- c(20, 11, 17, 12)
40           * wilcox.test(x, y, alternative = "two.sided", mu = 0, paired = FALSE, exact = FALSE, correct = FALSE)
41           * W = 17, p-value = 0.08641
42           */
43          final double x[] = {19, 22, 16, 29, 24};
44          final double y[] = {20, 11, 17, 12};
45  
46          Assert.assertEquals(17, testStatistic.mannWhitneyU(x, y), 1e-10);
47          Assert.assertEquals(0.08641, testStatistic.mannWhitneyUTest(x, y), 1e-5);
48      }
49  
50  
51      @Test
52      public void testMannWhitneyUInputValidation() {
53          /* Samples must be present, i.e. length > 0
54           */
55          try {
56              testStatistic.mannWhitneyUTest(new double[] { }, new double[] { 1.0 });
57              Assert.fail("x does not contain samples (exact), NoDataException expected");
58          } catch (NoDataException ex) {
59              // expected
60          }
61  
62          try {
63              testStatistic.mannWhitneyUTest(new double[] { 1.0 }, new double[] { });
64              Assert.fail("y does not contain samples (exact), NoDataException expected");
65          } catch (NoDataException ex) {
66              // expected
67          }
68  
69          /*
70           * x and y is null
71           */
72          try {
73              testStatistic.mannWhitneyUTest(null, null);
74              Assert.fail("x and y is null (exact), NullArgumentException expected");
75          } catch (NullArgumentException ex) {
76              // expected
77          }
78  
79          try {
80              testStatistic.mannWhitneyUTest(null, null);
81              Assert.fail("x and y is null (asymptotic), NullArgumentException expected");
82          } catch (NullArgumentException ex) {
83              // expected
84          }
85  
86          /*
87           * x or y is null
88           */
89          try {
90              testStatistic.mannWhitneyUTest(null, new double[] { 1.0 });
91              Assert.fail("x is null (exact), NullArgumentException expected");
92          } catch (NullArgumentException ex) {
93              // expected
94          }
95  
96          try {
97              testStatistic.mannWhitneyUTest(new double[] { 1.0 }, null);
98              Assert.fail("y is null (exact), NullArgumentException expected");
99          } catch (NullArgumentException ex) {
100             // expected
101         }
102     }
103 
104     @Test
105     public void testBigDataSet() {
106         double[] d1 = new double[1500];
107         double[] d2 = new double[1500];
108         for (int i = 0; i < 1500; i++) {
109             d1[i] = 2 * i;
110             d2[i] = 2 * i + 1;
111         }
112         double result = testStatistic.mannWhitneyUTest(d1, d2);
113         Assert.assertTrue(result > 0.1);
114     }
115 
116     @Test
117     public void testBigDataSetOverflow() {
118         // MATH-1145
119         double[] d1 = new double[110000];
120         double[] d2 = new double[110000];
121         for (int i = 0; i < 110000; i++) {
122             d1[i] = i;
123             d2[i] = i;
124         }
125         double result = testStatistic.mannWhitneyUTest(d1, d2);
126         Assert.assertTrue(result == 1.0);
127     }
128 }
