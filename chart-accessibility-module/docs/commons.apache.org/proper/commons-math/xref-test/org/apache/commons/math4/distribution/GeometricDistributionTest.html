159     @Test
160     public void testMoments() {
161         final double tol = 1e-9;
162         GeometricDistribution dist;
163 
164         dist = new GeometricDistribution(0.5);
165         Assert.assertEquals(dist.getNumericalMean(), (1.0d - 0.5d) / 0.5d, tol);
166         Assert.assertEquals(dist.getNumericalVariance(), (1.0d - 0.5d) / (0.5d * 0.5d), tol);
167 
168         dist = new GeometricDistribution(0.3);
169         Assert.assertEquals(dist.getNumericalMean(), (1.0d - 0.3d) / 0.3d, tol);
170         Assert.assertEquals(dist.getNumericalVariance(), (1.0d - 0.3d) / (0.3d * 0.3d), tol);
171     }
