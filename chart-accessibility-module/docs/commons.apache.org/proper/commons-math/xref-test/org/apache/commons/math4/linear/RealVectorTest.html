192     @Test
193     public void testSparseIterator() {
194         /*
195          * For non-default values, use x + 1, x + 2, etc... to make sure that
196          * these values are really different from x.
197          */
198         final double x = getPreferredEntryValue();
199         final double[] data = {
200             x, x + 1d, x, x, x + 2d, x + 3d, x + 4d, x, x, x, x + 5d, x + 6d, x
201         };
202 
203         RealVector v = create(data);
204         Entry e;
205         int i = 0;
206         final double[] nonDefault = {
207             x + 1d, x + 2d, x + 3d, x + 4d, x + 5d, x + 6d
208         };
209         for (Iterator<Entry> it = v.sparseIterator(); it.hasNext(); i++) {
210             e = it.next();
211             Assert.assertEquals(nonDefault[i], e.getValue(), 0);
212         }
213         double [] onlyOne = {x, x + 1d, x};
214         v = create(onlyOne);
215         for(Iterator<Entry> it = v.sparseIterator(); it.hasNext(); ) {
216             e = it.next();
217             Assert.assertEquals(onlyOne[1], e.getValue(), 0);
218         }
219     }
