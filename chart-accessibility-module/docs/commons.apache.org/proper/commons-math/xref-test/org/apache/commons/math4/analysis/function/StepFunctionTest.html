21  import org.apache.commons.math4.analysis.function.StepFunction;

23  import org.apache.commons.math4.exception.NoDataException;
24  import org.apache.commons.math4.exception.NonMonotonicSequenceException;
25  import org.apache.commons.math4.exception.NullArgumentException;

29  /**
30   * Test for class {@link StepFunction}.
31   */
32  public class StepFunctionTest {
33      private final double EPS = Math.ulp(1d);
34  
35      @Test(expected=NullArgumentException.class)
36      public void testPreconditions1() {
37          new StepFunction(null, new double[] {0, -1, -2});
38      }
39  
40      @Test(expected=NullArgumentException.class)
41      public void testPreconditions2() {
42          new StepFunction(new double[] {0, 1}, null);
43      }
44  
45      @Test(expected=NoDataException.class)
46      public void testPreconditions3() {
47          new StepFunction(new double[] {0}, new double[] {});
48      }
49  
50      @Test(expected=NoDataException.class)
51      public void testPreconditions4() {
52          new StepFunction(new double[] {}, new double[] {0});
53      }
54  
55      @Test(expected=DimensionMismatchException.class)
56      public void testPreconditions5() {
57          new StepFunction(new double[] {0, 1}, new double[] {0, -1, -2});
58      }
59  
60      @Test(expected=NonMonotonicSequenceException.class)
61      public void testPreconditions6() {
62          new StepFunction(new double[] {1, 0, 1}, new double[] {0, -1, -2});
63      }
64  
65      @Test
66      public void testSomeValues() {
67          final double[] x = { -2, -0.5, 0, 1.9, 7.4, 21.3 };
68          final double[] y = { 4, -1, -5.5, 0.4, 5.8, 51.2 };
69  
70          final UnivariateFunction f = new StepFunction(x, y);
71  
72          Assert.assertEquals(4, f.value(Double.NEGATIVE_INFINITY), EPS);
73          Assert.assertEquals(4, f.value(-10), EPS);
74          Assert.assertEquals(-1, f.value(-0.4), EPS);
75          Assert.assertEquals(-5.5, f.value(0), EPS);
76          Assert.assertEquals(0.4, f.value(2), EPS);
77          Assert.assertEquals(5.8, f.value(10), EPS);
78          Assert.assertEquals(51.2, f.value(30), EPS);
79          Assert.assertEquals(51.2, f.value(Double.POSITIVE_INFINITY), EPS);
80      }
81  
82      @Test
83      public void testEndpointBehavior() {
84          final double[] x = {0, 1, 2, 3};
85          final double[] xp = {-8, 1, 2, 3};
86          final double[] y = {1, 2, 3, 4};
87          final UnivariateFunction f = new StepFunction(x, y);
88          final UnivariateFunction fp = new StepFunction(xp, y);
89          Assert.assertEquals(f.value(-8), fp.value(-8), EPS);
90          Assert.assertEquals(f.value(-10), fp.value(-10), EPS);
91          Assert.assertEquals(f.value(0), fp.value(0), EPS);
92          Assert.assertEquals(f.value(0.5), fp.value(0.5), EPS);
93          for (int i = 0; i < x.length; i++) {
94             Assert.assertEquals(y[i], f.value(x[i]), EPS);
95             if (i > 0) {
96                 Assert.assertEquals(y[i - 1], f.value(x[i] - 0.5), EPS);
97             } else {
98                 Assert.assertEquals(y[0], f.value(x[i] - 0.5), EPS);
99             }
100         }
101     }
102 
103     @Test
104     public void testHeaviside() {
105         final UnivariateFunction h = new StepFunction(new double[] {-1, 0},
106                                                           new double[] {0, 1});
107 
108         Assert.assertEquals(0, h.value(Double.NEGATIVE_INFINITY), 0);
109         Assert.assertEquals(0, h.value(-Double.MAX_VALUE), 0);
110         Assert.assertEquals(0, h.value(-2), 0);
111         Assert.assertEquals(0, h.value(-Double.MIN_VALUE), 0);
112         Assert.assertEquals(1, h.value(0), 0);
113         Assert.assertEquals(1, h.value(2), 0);
114         Assert.assertEquals(1, h.value(Double.POSITIVE_INFINITY), 0);
115     }
116 }
