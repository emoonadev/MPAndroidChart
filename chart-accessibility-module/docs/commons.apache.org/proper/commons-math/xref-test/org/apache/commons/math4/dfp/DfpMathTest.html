59      // Generic test function.  Takes params x and y and tests them for
60      // equality.  Then checks the status flags against the flags argument.
61      // If the test fail, it prints the desc string
62      private void test(Dfp x, Dfp y, int flags, String desc)
63      {
64          boolean b = x.equals(y);
65  
66          if (!x.equals(y) && !x.unequal(y))  // NaNs involved
67              b = (x.toString().equals(y.toString()));
68  
69          if (x.equals(factory.newDfp("0")))  // distinguish +/- zero
70              b = (b && (x.toString().equals(y.toString())));
71  
72          b = (b && x.getField().getIEEEFlags() == flags);
73  
74          if (!b)
75              Assert.assertTrue("assersion failed "+desc+" x = "+x.toString()+" flags = "+x.getField().getIEEEFlags(), b);
76  
77          x.getField().clearIEEEFlags();
78      }
