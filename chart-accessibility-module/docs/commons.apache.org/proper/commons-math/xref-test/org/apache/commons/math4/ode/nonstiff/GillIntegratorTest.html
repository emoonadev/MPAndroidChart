56    @Test
57    public void testDecreasingSteps()
58        throws DimensionMismatchException, NumberIsTooSmallException,
59               MaxCountExceededException, NoBracketingException {
60  
61        for (TestProblemAbstract pb : new TestProblemAbstract[] {
62            new TestProblem1(), new TestProblem2(), new TestProblem3(),
63            new TestProblem4(), new TestProblem5(), new TestProblem6()
64        }) {
65  
66        double previousValueError = Double.NaN;
67        double previousTimeError = Double.NaN;
68        for (int i = 5; i < 10; ++i) {
69  
70          double step = (pb.getFinalTime() - pb.getInitialTime()) * FastMath.pow(2.0, -i);
71  
72          FirstOrderIntegrator integ = new GillIntegrator(step);
73          TestProblemHandler handler = new TestProblemHandler(pb, integ);
74          integ.addStepHandler(handler);
75          EventHandler[] functions = pb.getEventsHandlers();
76          for (int l = 0; l < functions.length; ++l) {
77            integ.addEventHandler(functions[l],
78                                       Double.POSITIVE_INFINITY, 1.0e-6 * step, 1000);
79          }
80          double stopTime = integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
81                                            pb.getFinalTime(), new double[pb.getDimension()]);
82          if (functions.length == 0) {
83              Assert.assertEquals(pb.getFinalTime(), stopTime, 1.0e-10);
84          }
85  
86          double valueError = handler.getMaximalValueError();
87          if (i > 5) {
88            Assert.assertTrue(valueError < 1.01 * FastMath.abs(previousValueError));
89          }
90          previousValueError = valueError;
91  
92          double timeError = handler.getMaximalTimeError();
93          if (i > 5) {
94            Assert.assertTrue(timeError <= FastMath.abs(previousTimeError));
95          }
96          previousTimeError = timeError;
97  
98        }
99  
100     }

104   @Test
105   public void testSmallStep()
106       throws DimensionMismatchException, NumberIsTooSmallException,
107              MaxCountExceededException, NoBracketingException {
108 
109     TestProblem1 pb = new TestProblem1();
110     double step = (pb.getFinalTime() - pb.getInitialTime()) * 0.001;
111 
112     FirstOrderIntegrator integ = new GillIntegrator(step);
113     TestProblemHandler handler = new TestProblemHandler(pb, integ);
114     integ.addStepHandler(handler);
115     integ.integrate(pb, pb.getInitialTime(), pb.getInitialState(),
116                     pb.getFinalTime(), new double[pb.getDimension()]);
117 
118     Assert.assertTrue(handler.getLastError() < 2.0e-13);
119     Assert.assertTrue(handler.getMaximalValueError() < 4.0e-12);
120     Assert.assertEquals(0, handler.getMaximalTimeError(), 1.0e-12);
121     Assert.assertEquals("Gill", integ.getName());
122 
123   }

180   @Test
181   public void testUnstableDerivative()
182       throws DimensionMismatchException, NumberIsTooSmallException,
183              MaxCountExceededException, NoBracketingException {
184     final StepProblem stepProblem = new StepProblem(0.0, 1.0, 2.0);
185     FirstOrderIntegrator integ = new GillIntegrator(0.3);
186     integ.addEventHandler(stepProblem, 1.0, 1.0e-12, 1000);
187     double[] y = { Double.NaN };
188     integ.integrate(stepProblem, 0.0, new double[] { 0.0 }, 10.0, y);
189     Assert.assertEquals(8.0, y[0], 1.0e-12);
190   }
191 
192   private static class KeplerStepHandler implements StepHandler {
193     public KeplerStepHandler(TestProblem3 pb) {
194       this.pb = pb;
195     }
196     public void init(double t0, double[] y0, double t) {
197       maxError = 0;
198     }
199     public void handleStep(StepInterpolator interpolator, boolean isLast)
200         throws MaxCountExceededException {
201 
202       double[] interpolatedY = interpolator.getInterpolatedState();
203       double[] theoreticalY  = pb.computeTheoreticalState(interpolator.getCurrentTime());
204       double dx = interpolatedY[0] - theoreticalY[0];
205       double dy = interpolatedY[1] - theoreticalY[1];
206       double error = dx * dx + dy * dy;
207       if (error > maxError) {
208         maxError = error;
209       }
210       if (isLast) {
211         // even with more than 1000 evaluations per period,
212         // Gill is not able to integrate such an eccentric
213         // orbit with a good accuracy
214         Assert.assertTrue(maxError > 0.001);
215       }
216     }
217     private double maxError;
218     private TestProblem3 pb;
219   }

221   @Test
222   public void testStepSize()
223       throws DimensionMismatchException, NumberIsTooSmallException,
224              MaxCountExceededException, NoBracketingException {
225       final double step = 1.23456;
226       FirstOrderIntegrator integ = new GillIntegrator(step);
227       integ.addStepHandler(new StepHandler() {
228           public void handleStep(StepInterpolator interpolator, boolean isLast) {
229               if (! isLast) {
230                   Assert.assertEquals(step,
231                                interpolator.getCurrentTime() - interpolator.getPreviousTime(),
232                                1.0e-12);
233               }
234           }
235           public void init(double t0, double[] y0, double t) {
236           }
237       });
238       integ.integrate(new FirstOrderDifferentialEquations() {
239           public void computeDerivatives(double t, double[] y, double[] dot) {
240               dot[0] = 1.0;
241           }
242           public int getDimension() {
243               return 1;
244           }
245       }, 0.0, new double[] { 0.0 }, 5.0, new double[1]);
246   }
