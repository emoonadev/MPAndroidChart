19  import org.junit.Assert;

21  import java.util.Arrays;
22  
23  import org.apache.commons.math4.analysis.solvers.PegasusSolver;
24  import org.apache.commons.math4.exception.DimensionMismatchException;
25  import org.apache.commons.math4.exception.MaxCountExceededException;
26  import org.apache.commons.math4.exception.NoBracketingException;
27  import org.apache.commons.math4.exception.NumberIsTooSmallException;
28  import org.apache.commons.math4.ode.FirstOrderDifferentialEquations;
29  import org.apache.commons.math4.ode.FirstOrderIntegrator;
30  import org.apache.commons.math4.ode.events.EventHandler;
31  import org.apache.commons.math4.ode.nonstiff.DormandPrince853Integrator;
32  import org.apache.commons.math4.ode.nonstiff.GraggBulirschStoerIntegrator;
33  import org.junit.Test;
34  
35  public class ReappearingEventTest {
36      @Test
37      public void testDormandPrince()
38          throws DimensionMismatchException, NumberIsTooSmallException,
39                 MaxCountExceededException, NoBracketingException {
40          double tEnd = test(1);
41          Assert.assertEquals(10.0, tEnd, 1e-7);
42      }
43  
44      @Test
45      public void testGragg()
46          throws DimensionMismatchException, NumberIsTooSmallException,
47                 MaxCountExceededException, NoBracketingException {
48          double tEnd = test(2);
49          Assert.assertEquals(10.0, tEnd, 1e-7);
50      }
51  
52      public double test(int integratorType)
53          throws DimensionMismatchException, NumberIsTooSmallException,
54                 MaxCountExceededException, NoBracketingException {
55          double e = 1e-15;
56          FirstOrderIntegrator integrator;
57          integrator = (integratorType == 1)
58                       ? new DormandPrince853Integrator(e, 100.0, 1e-7, 1e-7)
59                       : new GraggBulirschStoerIntegrator(e, 100.0, 1e-7, 1e-7);
60          PegasusSolver rootSolver = new PegasusSolver(e, e);
61          integrator.addEventHandler(new Event(), 0.1, e, 1000, rootSolver);
62          double t0 = 6.0;
63          double tEnd = 10.0;
64          double[] y = {2.0, 2.0, 2.0, 4.0, 2.0, 7.0, 15.0};
65          return integrator.integrate(new Ode(), t0, y, tEnd, y);
66      }
67  
68      private static class Ode implements FirstOrderDifferentialEquations {
69          public int getDimension() {
70              return 7;
71          }
72  
73          public void computeDerivatives(double t, double[] y, double[] yDot) {
74              Arrays.fill(yDot, 1.0);
75          }
76      }
77  
78      /** State events for this unit test. */
79      protected static class Event implements EventHandler {
80  
81          public void init(double t0, double[] y0, double t) {
82          }
83  
84          public double g(double t, double[] y) {
85              return y[6] - 15.0;
86          }
87  
88          public Action eventOccurred(double t, double[] y, boolean increasing) {
89              return Action.STOP;
90          }
91  
92          public void resetState(double t, double[] y) {
93              // Never called.
94          }
95      }
96  }
