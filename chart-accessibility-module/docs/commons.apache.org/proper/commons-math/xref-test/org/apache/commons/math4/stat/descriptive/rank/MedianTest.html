39  /**
40   * Test cases for the {@link UnivariateStatistic} class.
41   */
42  public class MedianTest extends UnivariateStatisticAbstractTest{
43  
44      protected Median stat;
45  
46      /**
47       * {@link  org.apache.commons.math4.stat.descriptive.rank.Percentile.EstimationType type}
48       *  to be used while calling
49       * {@link #getUnivariateStatistic()}
50       */
51      protected EstimationType estimationType = LEGACY;
52  
53      /**
54       * {@inheritDoc}
55       */
56      @Override
57      public UnivariateStatistic getUnivariateStatistic() {
58          return new Median();
59      }
60  
61      private Median getTestMedian(EstimationType type) {
62          NaNStrategy strategy = (type == LEGACY) ? NaNStrategy.FIXED : NaNStrategy.REMOVED;
63          return new Median().withEstimationType(type).withNaNStrategy(strategy);
64      }
65  
66      /**
67       * {@inheritDoc}
68       */
69      @Override
70      public double expectedValue() {
71          return this.median;
72      }
73  
74      @Before
75      public void before() {
76          estimationType=LEGACY;
77      }
78  
79      @Test
80      public void testAllTechniquesSingleton() {
81          double[] singletonArray = new double[] { 1d };
82          for (EstimationType e : EstimationType.values()) {
83              UnivariateStatistic percentile = getTestMedian(e);
84              Assert.assertEquals(1d, percentile.evaluate(singletonArray), 0);
85              Assert.assertEquals(1d, percentile.evaluate(singletonArray, 0, 1), 0);
86              Assert.assertEquals(1d, new Median().evaluate(singletonArray, 0, 1, 5), 0);
87              Assert.assertEquals(1d, new Median().evaluate(singletonArray, 0, 1, 100), 0);
88              Assert.assertTrue(Double.isNaN(percentile.evaluate(singletonArray, 0, 0)));
89          }
90      }
91  
92      @Test
93      public void testAllTechniquesMedian() {
94          double[] d = new double[] { 1, 3, 2, 4 };
95          testAssertMappedValues(d, new Object[][] { { LEGACY, 2.5d },
96              { R_1, 2d }, { R_2, 2.5d }, { R_3, 2d }, { R_4, 2d }, { R_5, 2.5 },
97              { R_6, 2.5 },{ R_7, 2.5 },{ R_8, 2.5 }, { R_9 , 2.5 } },  1.0e-05);
98      }
99  
100     /**
101      * Simple test assertion utility method
102      *
103      * @param d input data
104      * @param map of expected result against a {@link EstimationType}
105      * @param tolerance the tolerance of difference allowed
106      */
107     protected void testAssertMappedValues(double[] d, Object[][] map, Double tolerance) {
108         for (Object[] o : map) {
109             EstimationType e = (EstimationType) o[0];
110             double expected = (Double) o[1];
111             double result = getTestMedian(e).evaluate(d);
112             Assert.assertEquals("expected[" + e + "] = " + expected +
113                     " but was = " + result, expected, result, tolerance);
114         }
115     }
116 }
