17  package org.apache.commons.math4.genetics;
18  
19  
20  import java.util.LinkedList;
21  import java.util.List;
22  
23  import org.apache.commons.math4.genetics.BinaryChromosome;
24  import org.apache.commons.math4.genetics.BinaryMutation;
25  import org.apache.commons.math4.genetics.Chromosome;
26  import org.apache.commons.math4.genetics.ElitisticListPopulation;
27  import org.apache.commons.math4.genetics.FixedGenerationCount;
28  import org.apache.commons.math4.genetics.GeneticAlgorithm;
29  import org.apache.commons.math4.genetics.OnePointCrossover;
30  import org.apache.commons.math4.genetics.Population;
31  import org.apache.commons.math4.genetics.StoppingCondition;
32  import org.apache.commons.math4.genetics.TournamentSelection;
33  import org.junit.Assert;
34  import org.junit.Test;
35  
36  
37  public class FitnessCachingTest {
38  
39      // parameters for the GA
40      private static final int DIMENSION = 50;
41      private static final double CROSSOVER_RATE = 1;
42      private static final double MUTATION_RATE = 0.1;
43      private static final int TOURNAMENT_ARITY = 5;
44  
45      private static final int POPULATION_SIZE = 10;
46      private static final int NUM_GENERATIONS = 50;
47      private static final double ELITISM_RATE = 0.2;
48  
49      // how many times was the fitness computed
50      private static int fitnessCalls = 0;
51  
52  
53      @Test
54      public void testFitnessCaching() {
55          // initialize a new genetic algorithm
56          GeneticAlgorithm ga = new GeneticAlgorithm(
57                  new OnePointCrossover<Integer>(),
58                  CROSSOVER_RATE, // all selected chromosomes will be recombined (=crosssover)
59                  new BinaryMutation(),
60                  MUTATION_RATE, // no mutation
61                  new TournamentSelection(TOURNAMENT_ARITY)
62          );
63  
64          // initial population
65          Population initial = randomPopulation();
66          // stopping conditions
67          StoppingCondition stopCond = new FixedGenerationCount(NUM_GENERATIONS);
68  
69          // run the algorithm
70          ga.evolve(initial, stopCond);
71  
72          int neededCalls =
73              POPULATION_SIZE /*initial population*/ +
74              (NUM_GENERATIONS - 1) /*for each population*/ * (int)(POPULATION_SIZE * (1.0 - ELITISM_RATE)) /*some chromosomes are copied*/
75              ;
76          Assert.assertTrue(fitnessCalls <= neededCalls); // some chromosomes after crossover may be the same os old ones
77      }
78  
79  
80      /**
81       * Initializes a random population.
82       */
83      private static ElitisticListPopulation randomPopulation() {
84          List<Chromosome> popList = new LinkedList<Chromosome>();
85  
86          for (int i=0; i<POPULATION_SIZE; i++) {
87              BinaryChromosome randChrom = new DummyCountingBinaryChromosome(BinaryChromosome.randomBinaryRepresentation(DIMENSION));
88              popList.add(randChrom);
89          }
90          return new ElitisticListPopulation(popList, popList.size(), ELITISM_RATE);
91      }
92  
93      private static class DummyCountingBinaryChromosome extends DummyBinaryChromosome {
94  
95          public DummyCountingBinaryChromosome(List<Integer> representation) {
96              super(representation);
97          }
98  
99          @Override
100         public double fitness() {
101             fitnessCalls++;
102             return 0;
103         }
104     }
105 }
