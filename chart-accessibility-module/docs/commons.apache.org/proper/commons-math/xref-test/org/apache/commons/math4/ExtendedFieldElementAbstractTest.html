500     private void checkRelative(double expected, T obtained) {
501         Assert.assertEquals(expected, obtained.getReal(), 1.0e-15 * (1 + FastMath.abs(expected)));
502     }

504     @Test
505     public void testEquals() {
506         T t1a = build(1.0);
507         T t1b = build(1.0);
508         T t2  = build(2.0);
509         Assert.assertTrue(t1a.equals(t1a));
510         Assert.assertTrue(t1a.equals(t1b));
511         Assert.assertFalse(t1a.equals(t2));
512         Assert.assertFalse(t1a.equals(new Object()));
513     }

517         T t1a = build(1.0);
518         T t1b = build(1.0);

520         Assert.assertEquals(t1a.hashCode(), t1b.hashCode());
521         Assert.assertTrue(t1a.hashCode() != t2.hashCode());

524     private double[] generateDouble (final UniformRandomProvider r, int n) {
525         double[] a = new double[n];
526         for (int i = 0; i < n; ++i) {
527             a[i] = r.nextDouble();
528         }
529         return a;
530     }
531 
532     private T[] toFieldArray (double[] a) {
533         T[] f = MathArrays.buildArray(build(0).getField(), a.length);
534         for (int i = 0; i < a.length; ++i) {
535             f[i] = build(a[i]);
536         }
537         return f;
538     }
