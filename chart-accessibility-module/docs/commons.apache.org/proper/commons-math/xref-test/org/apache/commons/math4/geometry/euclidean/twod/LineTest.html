33      @Test
34      public void testContains() {
35          Line l = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
36          Assert.assertTrue(l.contains(new Vector2D(0, 1)));
37          Assert.assertTrue(l.contains(new Vector2D(1, 2)));
38          Assert.assertTrue(l.contains(new Vector2D(7, 8)));
39          Assert.assertTrue(! l.contains(new Vector2D(8, 7)));
40      }

42      @Test
43      public void testAbscissa() {
44          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
45          Assert.assertEquals(0.0,
46                              (l.toSubSpace(new Vector2D(-3,  4))).getX(),
47                              1.0e-10);
48          Assert.assertEquals(0.0,
49                              (l.toSubSpace(new Vector2D( 3, -4))).getX(),
50                              1.0e-10);
51          Assert.assertEquals(-5.0,
52                              (l.toSubSpace(new Vector2D( 7, -1))).getX(),
53                              1.0e-10);
54          Assert.assertEquals( 5.0,
55                               (l.toSubSpace(new Vector2D(-1, -7))).getX(),
56                               1.0e-10);
57      }

59      @Test
60      public void testOffset() {
61          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
62          Assert.assertEquals(-5.0, l.getOffset(new Vector2D(5, -3)), 1.0e-10);
63          Assert.assertEquals(+5.0, l.getOffset(new Vector2D(-5, 2)), 1.0e-10);
64      }

66      @Test
67      public void testDistance() {
68          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
69          Assert.assertEquals(+5.0, l.distance(new Vector2D(5, -3)), 1.0e-10);
70          Assert.assertEquals(+5.0, l.distance(new Vector2D(-5, 2)), 1.0e-10);
71      }

73      @Test
74      public void testPointAt() {
75          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
76          for (double a = -2.0; a < 2.0; a += 0.2) {
77              Point<Euclidean1D> pA = new Vector1D(a);
78              Point<Euclidean2D> point = l.toSpace(pA);
79              Assert.assertEquals(a, (l.toSubSpace(point)).getX(), 1.0e-10);
80              Assert.assertEquals(0.0, l.getOffset(point),   1.0e-10);
81              for (double o = -2.0; o < 2.0; o += 0.2) {
82                  point = l.getPointAt((Vector1D) pA, o);
83                  Assert.assertEquals(a, (l.toSubSpace(point)).getX(), 1.0e-10);
84                  Assert.assertEquals(o, l.getOffset(point),   1.0e-10);
85              }
86          }
87      }

89      @Test
90      public void testOriginOffset() {
91          Line l1 = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
92          Assert.assertEquals(FastMath.sqrt(0.5), l1.getOriginOffset(), 1.0e-10);
93          Line l2 = new Line(new Vector2D(1, 2), new Vector2D(0, 1), 1.0e-10);
94          Assert.assertEquals(-FastMath.sqrt(0.5), l2.getOriginOffset(), 1.0e-10);
95      }

97      @Test
98      public void testParallel() {
99          Line l1 = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
100         Line l2 = new Line(new Vector2D(2, 2), new Vector2D(3, 3), 1.0e-10);
101         Assert.assertTrue(l1.isParallelTo(l2));
102         Line l3 = new Line(new Vector2D(1, 0), new Vector2D(0.5, -0.5), 1.0e-10);
103         Assert.assertTrue(l1.isParallelTo(l3));
104         Line l4 = new Line(new Vector2D(1, 0), new Vector2D(0.5, -0.51), 1.0e-10);
105         Assert.assertTrue(! l1.isParallelTo(l4));
106     }

108     @Test
109     public void testTransform() throws MathIllegalArgumentException {
110 
111         Line l1 = new Line(new Vector2D(1.0 ,1.0), new Vector2D(4.0 ,1.0), 1.0e-10);
112         Transform<Euclidean2D, Euclidean1D> t1 =
113             Line.getTransform(0.0, 0.5, -1.0, 0.0, 1.0, 1.5);
114         Assert.assertEquals(0.5 * FastMath.PI,
115                             ((Line) t1.apply(l1)).getAngle(),
116                             1.0e-10);
117 
118         Line l2 = new Line(new Vector2D(0.0, 0.0), new Vector2D(1.0, 1.0), 1.0e-10);
119         Transform<Euclidean2D, Euclidean1D> t2 =
120             Line.getTransform(0.0, 0.5, -1.0, 0.0, 1.0, 1.5);
121         Assert.assertEquals(FastMath.atan2(1.0, -2.0),
122                             ((Line) t2.apply(l2)).getAngle(),
123                             1.0e-10);
124 
125     }

127     @Test
128     public void testIntersection() {
129         Line    l1 = new Line(new Vector2D( 0, 1), new Vector2D(1, 2), 1.0e-10);
130         Line    l2 = new Line(new Vector2D(-1, 2), new Vector2D(2, 1), 1.0e-10);
131         Vector2D p  = l1.intersection(l2);
132         Assert.assertEquals(0.5, p.getX(), 1.0e-10);
133         Assert.assertEquals(1.5, p.getY(), 1.0e-10);
134     }
