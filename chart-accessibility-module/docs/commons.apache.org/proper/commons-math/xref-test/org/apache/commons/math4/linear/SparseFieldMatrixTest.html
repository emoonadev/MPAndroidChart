381     /** test examples in user guide */
382     @Test
383     public void testExamples() {
384         // Create a real matrix with two rows and three columns
385         Fraction[][] matrixData = { { new Fraction(1), new Fraction(2), new Fraction(3) }, { new Fraction(2), new Fraction(5), new Fraction(3) } };
386         FieldMatrix<Fraction> m = createSparseMatrix(matrixData);
387         // One more with three rows, two columns
388         Fraction[][] matrixData2 = { { new Fraction(1), new Fraction(2) }, { new Fraction(2), new Fraction(5) }, { new Fraction(1), new Fraction(7) } };
389         FieldMatrix<Fraction> n = createSparseMatrix(matrixData2);
390         // Now multiply m by n
391         FieldMatrix<Fraction> p = m.multiply(n);
392         Assert.assertEquals(2, p.getRowDimension());
393         Assert.assertEquals(2, p.getColumnDimension());
394         // Invert p
395         FieldMatrix<Fraction> pInverse = new FieldLUDecomposition<Fraction>(p).getSolver().getInverse();
396         Assert.assertEquals(2, pInverse.getRowDimension());
397         Assert.assertEquals(2, pInverse.getColumnDimension());
398 
399         // Solve example
400         Fraction[][] coefficientsData = { { new Fraction(2), new Fraction(3), new Fraction(-2) }, { new Fraction(-1), new Fraction(7), new Fraction(6) },
401                 { new Fraction(4), new Fraction(-3), new Fraction(-5) } };
402         FieldMatrix<Fraction> coefficients = createSparseMatrix(coefficientsData);
403         Fraction[] constants = { new Fraction(1), new Fraction(-2), new Fraction(1) };
404         Fraction[] solution;
405         solution = new FieldLUDecomposition<Fraction>(coefficients)
406             .getSolver()
407             .solve(new ArrayFieldVector<Fraction>(constants, false)).toArray();
408         Assert.assertEquals((new Fraction(2).multiply((solution[0])).add(new Fraction(3).multiply(solution[1])).subtract(new Fraction(2).multiply(solution[2]))).doubleValue(),
409                 constants[0].doubleValue(), 1E-12);
410         Assert.assertEquals(((new Fraction(-1).multiply(solution[0])).add(new Fraction(7).multiply(solution[1])).add(new Fraction(6).multiply(solution[2]))).doubleValue(),
411                 constants[1].doubleValue(), 1E-12);
412         Assert.assertEquals(((new Fraction(4).multiply(solution[0])).subtract(new Fraction(3).multiply( solution[1])).subtract(new Fraction(5).multiply(solution[2]))).doubleValue(),
413                 constants[2].doubleValue(), 1E-12);
414

682     protected void assertClose(String msg, FieldMatrix<Fraction> m, FieldMatrix<Fraction> n,
683             double tolerance) {
684         for(int i=0; i < m.getRowDimension(); i++){
685             for(int j=0; j < m.getColumnDimension(); j++){
686                 Assert.assertEquals(msg, m.getEntry(i,j).doubleValue(), n.getEntry(i,j).doubleValue(), tolerance);
687             }
688 
689         }
690     }

692     /** verifies that two vectors are close (sup norm) */
693     protected void assertClose(String msg, Fraction[] m, Fraction[] n,
694             double tolerance) {
695         if (m.length != n.length) {
696             Assert.fail("vectors not same length");
697         }
698         for (int i = 0; i < m.length; i++) {
699             Assert.assertEquals(msg + " " + i + " elements differ", m[i].doubleValue(), n[i].doubleValue(),
700                     tolerance);
701         }
702     }

704     private SparseFieldMatrix<Fraction> createSparseMatrix(Fraction[][] data) {
705         SparseFieldMatrix<Fraction> matrix = new SparseFieldMatrix<Fraction>(field, data.length, data[0].length);
706         for (int row = 0; row < data.length; row++) {
707             for (int col = 0; col < data[row].length; col++) {
708                 matrix.setEntry(row, col, data[row][col]);
709             }
710         }
711         return matrix;
712     }
