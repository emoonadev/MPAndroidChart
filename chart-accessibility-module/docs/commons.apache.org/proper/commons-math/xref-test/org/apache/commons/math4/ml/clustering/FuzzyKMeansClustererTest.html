44      @Test
45      public void testCluster() {
46          final List<DoublePoint> points = new ArrayList<DoublePoint>();
47  
48          // create 10 data points: [1], ... [10]
49          for (int i = 1; i <= 10; i++) {
50              final DoublePoint p = new DoublePoint(new double[] { i } );
51              points.add(p);
52          }
53  
54          final FuzzyKMeansClusterer<DoublePoint> transformer =
55                  new FuzzyKMeansClusterer<DoublePoint>(3, 2.0);
56          final List<CentroidCluster<DoublePoint>> clusters = transformer.cluster(points);
57  
58          // we expect 3 clusters:
59          //   [1], [2], [3]
60          //   [4], [5], [6], [7]
61          //   [8], [9], [10]
62          final List<DoublePoint> clusterOne = Arrays.asList(points.get(0), points.get(1), points.get(2));
63          final List<DoublePoint> clusterTwo = Arrays.asList(points.get(3), points.get(4), points.get(5), points.get(6));
64          final List<DoublePoint> clusterThree = Arrays.asList(points.get(7), points.get(8), points.get(9));
65  
66          boolean cluster1Found = false;
67          boolean cluster2Found = false;
68          boolean cluster3Found = false;
69          Assert.assertEquals(3, clusters.size());
70          for (final Cluster<DoublePoint> cluster : clusters) {
71              if (cluster.getPoints().containsAll(clusterOne)) {
72                  cluster1Found = true;
73              }
74              if (cluster.getPoints().containsAll(clusterTwo)) {
75                  cluster2Found = true;
76              }
77              if (cluster.getPoints().containsAll(clusterThree)) {
78                  cluster3Found = true;
79              }
80          }
81          Assert.assertTrue(cluster1Found);
82          Assert.assertTrue(cluster2Found);
83          Assert.assertTrue(cluster3Found);
84      }

97      @Test
98      public void testGetters() {
99          final DistanceMeasure measure = new CanberraDistance();
100         final UniformRandomProvider random = RandomSource.create(RandomSource.MT_64);
101         final FuzzyKMeansClusterer<DoublePoint> clusterer =
102                 new FuzzyKMeansClusterer<DoublePoint>(3, 2.0, 100, measure, 1e-6, random);
103 
104         Assert.assertEquals(3, clusterer.getK());
105         Assert.assertEquals(2.0, clusterer.getFuzziness(), 1e-6);
106         Assert.assertEquals(100, clusterer.getMaxIterations());
107         Assert.assertEquals(1e-6, clusterer.getEpsilon(), 1e-12);
108         Assert.assertThat(clusterer.getDistanceMeasure(), CoreMatchers.is(measure));
109         Assert.assertThat(clusterer.getRandomGenerator(), CoreMatchers.is(random));
110     }
